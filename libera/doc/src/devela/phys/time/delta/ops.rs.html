16impl TimeDelta {
17    /// Add two time deltas together. If overflow occurs, then `None` is returned.
18    pub const fn checked_add(self, rhs: TimeDelta) -> Option<TimeDelta> {
19        let Some(mut secs) = self.secs.checked_add(rhs.secs) else {
20            return None;
21        };
22        // OK because `-999_999_999 <= nanos <= 999_999_999`, and so adding
23        // them together will never overflow an i32.
24        let mut nanos = self.nanos + rhs.nanos;
25        // The below is effectively TimeDelta::new, but with checked
26        // arithmetic. My suspicion is that there is probably a better way
27        // to do this. The main complexity here is that 1) `|nanos|` might
28        // now exceed 1 second and 2) the signs of `secs` and `nanos` might
29        // not be the same. The other difference from TimeDelta::new is
30        // that we know that `-1_999_999_998 <= nanos <= 1_999_999_998` since
31        // `|TimeDelta::nanos|` is guaranteed to be less than 1 second. So
32        // we can skip the div and modulus operations.
33
34        // When |nanos| exceeds 1 second, we balance the excess up to seconds.
35        if nanos != 0 {
36            if nanos >= NANOS_PER_SEC {
37                nanos -= NANOS_PER_SEC;
38                secs = match secs.checked_add(1) {
39                    None => return None,
40                    Some(secs) => secs,
41                };
42            } else if nanos <= -NANOS_PER_SEC {
43                nanos += NANOS_PER_SEC;
44                secs = match secs.checked_sub(1) {
45                    None => return None,
46                    Some(secs) => secs,
47                };
48            }
49            if secs != 0 && nanos != 0 && secs.signum() != (nanos.signum() as i64) {
50                if secs < 0 {
51                    debug_assert!(nanos > 0);
52                    // OK because secs<0.
53                    secs += 1;
54                    // OK because nanos>0.
55                    nanos -= NANOS_PER_SEC;
56                } else {
57                    debug_assert!(secs > 0);
58                    debug_assert!(nanos < 0);
59                    // OK because secs>0.
60                    secs -= 1;
61                    // OK because nanos<0.
62                    nanos += NANOS_PER_SEC;
63                }
64            }
65        }
66        Some(TimeDelta::new_unchecked(secs, nanos))
67    }
68
69    /// Add two time deltas together. If overflow occurs, then arithmetic saturates.
70    pub const fn saturating_add(self, rhs: TimeDelta) -> TimeDelta {
71        let Some(sum) = self.checked_add(rhs) else {
72            return if rhs.is_negative() { TimeDelta::MIN } else { TimeDelta::MAX };
73        };
74        sum
75    }

77    /// Subtract one time delta from another. If overflow occurs, then `None` is returned.
78    pub const fn checked_sub(self, rhs: TimeDelta) -> Option<TimeDelta> {
79        let Some(rhs) = rhs.checked_neg() else {
80            return None;
81        };
82        self.checked_add(rhs)
83    }

85    /// Add two time deltas together. If overflow occurs, then arithmetic saturates.
86    pub const fn saturating_sub(self, rhs: TimeDelta) -> TimeDelta {
87        let Some(diff) = self.checked_sub(rhs) else {
88            return if rhs.is_positive() { TimeDelta::MIN } else { TimeDelta::MAX };
89        };
90        diff
91    }

94/// # Operations involving integers.
95impl TimeDelta {
96    /// Multiply this time delta by an integer.
97    /// If the multiplication overflows, then `None` is returned.
98    pub const fn checked_mul(self, rhs: i32) -> Option<TimeDelta> {
99        let rhs = rhs as i64;
100        // Multiplying any two i32 values never overflows an i64.
101        let nanos = (self.nanos as i64) * rhs;
102        // OK since NANOS_PER_SEC!={-1,0}.
103        let addsecs = nanos / (NANOS_PER_SEC as i64);
104        // OK since NANOS_PER_SEC!={-1,0}.
105        let nanos = (nanos % (NANOS_PER_SEC as i64)) as i32;
106        let Some(secs) = self.secs.checked_mul(rhs) else {
107            return None;
108        };
109        let Some(secs) = secs.checked_add(addsecs) else {
110            return None;
111        };
112        Some(TimeDelta::new_unchecked(secs, nanos))
113    }
114
115    /// Multiply this time delta by an integer.
116    ///
117    /// If the multiplication overflows, then the result saturates to either
118    /// the minimum or maximum duration depending on the sign of the product.
119    pub const fn saturating_mul(self, rhs: i32) -> TimeDelta {
120        let Some(product) = self.checked_mul(rhs) else {
121            let sign = (self.signum() as i64) * (rhs as i64).signum();
122            return if sign.is_negative() { TimeDelta::MIN } else { TimeDelta::MAX };
123        };
124        product
125    }
126
127    /// Divide this duration by an integer. If the division overflows, then `None` is returned.
128    pub const fn checked_div(self, rhs: i32) -> Option<TimeDelta> {
129        if rhs == 0 || (self.secs == i64::MIN && rhs == -1) {
130            return None;
131        }
132        // OK since rhs!={-1,0}.
133        let secs = self.secs / (rhs as i64);
134        // OK since rhs!={-1,0}.
135        let addsecs = self.secs % (rhs as i64);
136        // OK since rhs!=0 and self.nanos>i32::MIN.
137        let mut nanos = self.nanos / rhs;
138        // OK since rhs!=0 and self.nanos>i32::MIN.
139        let addnanos = self.nanos % rhs;
140        let leftover_nanos = (addsecs * (NANOS_PER_SEC as i64)) + (addnanos as i64);
141        nanos += (leftover_nanos / (rhs as i64)) as i32;
142        debug_assert!(nanos < NANOS_PER_SEC);
143        Some(TimeDelta::new_unchecked(secs, nanos))
144    }
145}

178    /// Returns a time delta corresponding to the number of seconds.
179    ///
180    /// The number given may have a fractional nanosecond component.
181    ///
182    /// If the given float overflows the minimum or maximum time delta values,
183    /// then an error is returned.
184    ///
185    /// # Features
186    /// - Uses `std` or `_float_f64` when available, leveraging
187    ///   `trunc`, `fract`, and `round` for precise, bias-free conversion.
188    /// - In strict `no_std` mode, manually rounds using integer arithmetic,
189    ///   ensuring correctness while lacking exact fractional nanosecond precision.
190    #[rustfmt::skip]
191    pub fn try_from_secs_f64(secs: f64) -> Result<TimeDelta, &'static str> {
192        if !secs.is_finite() {
193            return Err("could not convert non-finite seconds {secs} to time delta"); }
194        if secs < (i64::MIN as f64) {
195            return Err("floating point seconds {secs} overflows TimeDelta::MIN"); }
196        if secs > (i64::MAX as f64) {
197            return Err("floating point seconds {secs} overflows TimeDelta::MAX"); }
198        let (isecs, nanos);
199        #[cfg(any(feature = "std", feature = "_float_f64"))] {
200            isecs = secs.trunc() as i64;
201            nanos = (secs.fract() * NANOS_PER_SEC as f64).round() as i32;
202        }
203        #[cfg(not(any(feature = "std", feature = "_float_f64")))] {
204            let secs_rounded = if secs >= 0.0 { secs + 0.5 }  // Round normally
205            else { secs - 0.5 };  // Round away from zero for negatives
206            isecs = secs_rounded as i64;
207            nanos = ((secs_rounded - isecs as f64) * NANOS_PER_SEC as f64) as i32;
208        }
209        Ok(TimeDelta::new_unchecked(isecs, nanos))
210    }

265    /// Returns a time delta corresponding to the number of milliseconds.
266    ///
267    /// The number given may have a fractional nanosecond component.
268    ///
269    /// If the given float overflows the minimum or maximum time delta
270    /// values, then an error is returned.
271    ///
272    /// # Features
273    /// - Uses `std` or `_float_f64` when available, leveraging `round`, `div_euclid`,
274    ///   and `rem_euclid` for precise, bias-free conversion.
275    /// - In strict `no_std` mode, manually rounds using integer arithmetic, ensuring correctness
276    ///   while lacking exact Euclidean division for negatives.
277    #[rustfmt::skip]
278    pub fn try_from_millis_f64(millis: f64) -> Result<TimeDelta, &'static str> {
279        if !millis.is_finite() {
280            return Err("could not convert non-finite milliseconds {millis} to time delta"); }
281        if millis < (i64::MIN as f64) {
282            return Err("floating point milliseconds {millis} overflows TimeDelta::MIN"); }
283        if millis > (i64::MAX as f64) {
284            return Err("floating point milliseconds {millis} overflows TimeDelta::MAX"); }
285        let (millis_rounded, secs, nanos);
286        #[cfg(any(feature = "std", feature = "_float_f64"))]
287        {
288            millis_rounded = millis.round();
289            secs = millis_rounded.div_euclid(1_000.0) as i64;
290            nanos = (millis_rounded.rem_euclid(1_000.0) * 1_000_000.0) as i32;
291        }
292        #[cfg(not(any(feature = "std", feature = "_float_f64")))]
293        {
294            millis_rounded = if millis >= 0.0 { millis + 0.5 }  // Round normally
295            else { millis - 0.5 };  // Round away from zero for negatives
296            let millis_i64 = millis_rounded as i64;
297            secs = millis_i64 / MILLIS_PER_SEC;
298            nanos = ((millis_i64 % MILLIS_PER_SEC) * NANOS_PER_MILLI as i64) as i32;
299        }
300        Ok(TimeDelta::new_unchecked(secs, nanos))
301    }
302    /// Compile-time friendly version of `try_from_millis_f64`.
303    ///
304    /// # Features
305    /// - Uses `_float_f64` if enabled, leveraging [`Float`]'s `const_round`, `div_euclid`,
306    ///   and `rem_euclid` for precise, bias-free conversion.
307    /// - Without `_float_f64`, rounds manually using integer arithmetic, preventing
308    ///   systematic underestimation but lacking exact Euclidean division for negatives.
309    #[rustfmt::skip]
310    pub const fn const_try_from_millis_f64(millis: f64) -> Result<TimeDelta, &'static str> {
311        if !millis.is_finite() {
312            return Err("could not convert non-finite milliseconds {millis} to time delta"); }
313        if millis < (i64::MIN as f64) {
314            return Err("floating point milliseconds {millis} overflows TimeDelta::MIN"); }
315        if millis > (i64::MAX as f64) {
316            return Err("floating point milliseconds {millis} overflows TimeDelta::MAX"); }
317        let (millis_rounded, secs, nanos);
318        #[cfg(feature = "_float_f64")]
319        {
320            millis_rounded = Float(millis).const_round().0 as i64;
321            secs = Float(millis_rounded as f64).div_euclid(MILLIS_PER_SEC as f64).0 as i64;
322            nanos = Float(millis_rounded as f64).rem_euclid(MILLIS_PER_SEC as f64).0 as i64
323            * NANOS_PER_MILLI as i64;
324        }
325        #[cfg(not(feature = "_float_f64"))]
326        {
327            // millis_rounded = millis as i64; // slight systematic underestimation
328            millis_rounded = if millis >= 0.0 { (millis + 0.5) as i64 }  // Round normally
329            else { (millis - 0.5) as i64 };  // Round away from zero for negatives
330            secs = millis_rounded / MILLIS_PER_SEC;
331            nanos = (millis_rounded % MILLIS_PER_SEC) * NANOS_PER_MILLI as i64;
332        }
333        Ok(TimeDelta::new_unchecked(secs, nanos as i32))
334    }

376    /// Divides this time delta by another time delta.
377    pub const fn div_delta_f64(self, rhs: TimeDelta) -> f64 {
378        let lhs_nanos = (self.secs as f64) * (NANOS_PER_SEC as f64) + (self.nanos as f64);
379        let rhs_nanos = (rhs.secs as f64) * (NANOS_PER_SEC as f64) + (rhs.nanos as f64);
380        lhs_nanos / rhs_nanos
381    }
