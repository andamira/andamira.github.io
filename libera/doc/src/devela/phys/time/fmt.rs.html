19#[doc = crate::TAG_TIME!()]
20/// Timecode splitting and formatting.
21///
22/// # Examples
23/// ```
24/// # use devela::Timecode;
25/// #[cfg(feature = "_str_u8")]
26/// #[cfg(any(feature = "std", feature = "_float_f64"))]
27/// assert_eq!(Timecode::secs_f64(3661.5), "01:01:01.500");
28///
29/// #[cfg(feature = "_str_u8")]
30/// assert_eq!(Timecode::nanos_u64(1_002_003_004), "001s 002ms 003µs 004ns");
31/// ```
32pub struct Timecode;
33
34impl Timecode {
35    /* time splitting */
36
37    /// Decomposes a number of `seconds` in `{ h, m, s, ms }`.
38    ///
39    /// The maximum decomposition for [`u64::MAX`] is
40    /// `{ h: 5_124_095_576_030_431, .. }` (more than 584_942_417 millenia).
41    // -> 64 bits
42    #[must_use]
43    #[cfg(any(feature = "std", feature = "_float_f64"))]
44    #[cfg_attr(nightly_doc, doc(cfg(any(feature = "std", feature = "_float_f64"))))]
45    pub fn split_secs_f64(seconds: f64) -> TimeSplitHourNano<u32, u8, u8, u16, NoTime, NoTime> {
46        let ms = (seconds.fract() * 1000.) as u16;
47        let mut ts = seconds.trunc() as u64;
48        let h = (ts / 3600) as u32;
49        ts %= 3600;
50        let m = (ts / 60) as u8;
51        let s = (ts % 60) as u8;
52        TimeSplit::new_hour_nano(h, m, s, ms, (), ())
53    }
54
55    /// Splits a number of `nanoseconds` in `{ s, ms, µs, ns }`.
56    ///
57    /// The maximum decomposition for [`u64::MAX`] is
58    /// `{ s: 1_266_874_889, .. }` (more than 40 years).
59    // -> 80 bits
60    #[must_use]
61    pub const fn split_nanos_u64(
62        nanos: u64,
63    ) -> TimeSplitHourNano<NoTime, NoTime, u32, u16, u16, u16> {
64        let (us_tmp, ns) = (nanos / 1000, (nanos % 1000) as u16);
65        let (ms_tmp, us) = (us_tmp / 1000, (us_tmp % 1000) as u16);
66        let (s, ms) = ((ms_tmp / 1000) as u32, (ms_tmp % 1000) as u16);
67        TimeSplit::new_hour_nano((), (), s, ms, us, ns)
68    }
69
70    /// Splits a number of `nanoseconds` in `{ s, ms, µs, ns }`.
71    ///
72    /// The maximum decomposition for [`u32::MAX`] is
73    /// `{ ns: 295, µs: 967, ms: 294, s: 4 }` (more than 4 seconds).
74    // -> 56 bits
75    #[must_use]
76    pub const fn split_nanos_u32(
77        nanos: u32,
78    ) -> TimeSplitHourNano<NoTime, NoTime, u8, u16, u16, u16> {
79        let (us_tmp, ns) = (nanos / 1000, (nanos % 1000) as u16);
80        let (ms_tmp, us) = (us_tmp / 1000, (us_tmp % 1000) as u16);
81        let (s, ms) = ((ms_tmp / 1000) as u8, (ms_tmp % 1000) as u16);
82        TimeSplit::new_hour_nano((), (), s, ms, us, ns)
83    }
84
85    /// Returns the time code as `HH:MM:SS:MIL` or `MM:SS:MIL`.
86    ///
87    /// The hours are clamped to 99 (more than 4 days).
88    ///
89    /// # Features
90    /// Makes use of the `unsafe_str` feature if enabled.
91    //
92    // -> 96 bits
93    #[cfg(any(feature = "std", feature = "_float_f64"))]
94    #[cfg_attr(nightly_doc, doc(cfg(any(feature = "std", feature = "_float_f64"))))]
95    #[cfg(feature = "_str_u8")]
96    #[cfg_attr(nightly_doc, doc(cfg(feature = "_str_u8")))]
97    pub fn secs_f64(seconds: f64) -> StringU8<12> {
98        let TimeSplitHourNano { h, m, s, ms, .. } = Self::split_secs_f64(seconds);
99        let m = Ascii(m as u32).digits_str(2);
100        let s = Ascii(s as u32).digits_str(2);
101        let ms = Ascii(ms as u32).digits_str(3);
102
103        let mut buf = [0; 12];
104        let mut buf_len = 12;
105
106        if h > 0 {
107            let h = Ascii(h.min(99)).digits_str(2);
108            let _str = format_buf![&mut buf, "{h}:{m}:{s}.{ms}"];
109        } else {
110            buf_len = 9;
111            let _str = format_buf![&mut buf, "{m}:{s}.{ms}"];
112        }
113
114        #[cfg(any(feature = "safe_time", not(feature = "unsafe_str")))]
115        return StringU8::<12>::from_bytes_nleft(buf, buf_len).unwrap();
116
117        #[cfg(all(not(feature = "safe_time"), feature = "unsafe_str"))]
118        // SAFETY: the buffer contains only ASCII characters.
119        unsafe {
120            StringU8::<12>::from_bytes_nleft_unchecked(buf, buf_len)
121        }
122    }
123
124    /// Returns the time code, up to seconds, as `001s 012ms 012µs 012ns`.
125    ///
126    /// The seconds are clamped to 999 (more than 16 minutes).
127    #[cfg(feature = "alloc")]
128    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
129    // -> 192 bits in "64", 92 bits in "32"
130    pub fn nanos_u64_alloc(ns: u64) -> String {
131        let (us, ns_rem) = (ns / 1_000, ns % 1_000);
132        let (ms, us_rem) = (us / 1_000, us % 1_000);
133        let (s, ms_rem) = (ms / 1_000, ms % 1_000);
134        let s = s.min(999);
135
136        if s > 0 {
137            format!["{s}s {ms_rem:03}ms {us_rem:03}µs {ns_rem:03}ns"]
138        } else if ms > 0 {
139            format!["{ms_rem}ms {us_rem:03}µs {ns_rem:03}ns"]
140        } else if us > 0 {
141            format!["{us_rem}µs {ns_rem:03}ns"]
142        } else {
143            format!["{ns_rem:03}ns"]
144        }
145    }
146
147    /// Returns the time code, up to seconds, as `001s 012ms 012µs 012345ns`.
148    ///
149    /// The seconds are clamped to 999 (more than 16 minutes).
150    // -> 208 bits
151    #[cfg(feature = "_str_u8")]
152    #[cfg_attr(nightly_doc, doc(cfg(feature = "_str_u8")))]
153    pub fn nanos_u64(nanos: u64) -> StringU8<23> {
154        let TimeSplitHourNano { s, ms, us, ns, .. } = Self::split_nanos_u64(nanos);
155        let s_str = Ascii(s.min(999)).digits_str(3);
156        let ms_str = Ascii(ms as u32).digits_str(3);
157        let us_str = Ascii(us as u32).digits_str(3);
158        let ns_str = Ascii(ns as u32).digits_str(3);
159
160        let mut buf = [0; 23];
161        let mut buf_len = 23; // = 18 + 3digits + 1name(s) + 1space
162
163        if s > 0 {
164            let _ = format_buf![&mut buf, "{s_str}s {ms_str}ms {us_str}µs {ns_str}ns"];
165        } else if ms > 0 {
166            buf_len = 18; // = 18 + 3digits + 2name(ms) + 1space
167            let _ = format_buf![&mut buf, "{ms_str}ms {us_str}µs {ns_str}ns"];
168        } else if us > 0 {
169            buf_len = 12; // = 5 + 3digits + 3name(µs) + 1space
170            let _ = format_buf![&mut buf, "{us_str}µs {ns_str}ns"];
171        } else {
172            buf_len = 5; // = 0 + 3digits + 2name(ns)
173            let _ = format_buf![&mut buf, "{ns_str}ns"];
174        }
175
176        #[cfg(any(feature = "safe_time", not(feature = "unsafe_str")))]
177        return StringU8::<23>::from_bytes_nleft(buf, buf_len).unwrap();
178
179        #[cfg(all(not(feature = "safe_time"), feature = "unsafe_str"))]
180        // SAFETY: the buffer contains only ASCII characters.
181        unsafe {
182            StringU8::<23>::from_bytes_nleft_unchecked(buf, buf_len)
183        }
184    }
185}
