#[derive(Clone, Copy, Default, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub struct Square<T> {
    _marker: PhantomData<T>,

        Self { _marker: PhantomData }
    }
}

///
#[derive(Clone, Copy, Default, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub struct SquareRoot<T> {
    _marker: PhantomData<T>,
}
impl<T> SquareRoot<T> {
    pub fn new() -> Self {
        Self { _marker: PhantomData }
    }
}

#[cfg(feature = "_float_f64")]
impl NumFn for Square<f64> {
    type Arg = f64;
    type Out = f64;

    fn apply(&self, x: f64) -> NumResult<f64> {
        Ok(x * x)
    }

    fn inverse(&self) -> NumResult<SquareRoot<f64>> {
        Ok(SquareRoot::<f64>::new())
    }
}

#[cfg(feature = "_float_f64")]
impl NumFn for SquareRoot<f64> {
    type Arg = f64;
    type Out = f64;

    fn apply(&self, x: f64) -> NumResult<f64> {
        Ok(x.sqrt())
    }

    fn inverse(&self) -> NumResult<Square<f64>> {
        Ok(Square::<f64>::new())
    }
}
