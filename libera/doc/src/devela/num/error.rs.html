8#[doc = crate::TAG_NUM!()]
9#[doc = crate::TAG_RESULT!()]
10/// A numeric-related result.
11pub type NumResult<T> = crate::Result<T, NumError>;
12
13#[doc = crate::TAG_NUM!()]
14#[doc = crate::TAG_ERROR_COMPOSITE!()]
15/// A numeric-related error.
16#[non_exhaustive]
17#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
18pub enum NumError {
19    /// The requested numerical functionality is not implemented.
20    ///
21    /// This is the default implementation of every numeric trait method.
22    NotImplemented,
23
24    /// The requested numerical functionality is not supported.
25    NotSupported,
26
27    /// Unspecified error.
28    ///
29    /// When no clearer error can be given.
30    // RETHINK
31    Unspecified,
32
33    /// An invalid value was received for the given type or operation.
34    Invalid,
35
36    /// An inverse doesn't exist.
37    NoInverse,
38
39    /// The provided values are not compatible in size.
40    MismatchedSizes,
41
42    /// The given bounds are not compatible.
43    IncompatibleBounds,
44
45    /// A non-negative value is required.
46    NonNegativeRequired,
47
48    /// A positive value is required.
49    PositiveRequired,
50
51    /// A non-zero value is required.
52    NonZeroRequired,
53
54    /// An arithmetic overflow error, with an optional associated sign.
55    Overflow(Option<Sign>),
56}
57
58#[allow(dead_code)]
59impl NumError {
60    pub(crate) const fn ni<T>() -> NumResult<T> {
61        Err(NumError::NotImplemented)
62    }
63    pub(crate) const fn ns<T>() -> NumResult<T> {
64        Err(NumError::NotSupported)
65    }
66}

68mod core_impls {
69    use crate::{Display, FmtResult, Formatter, NumError, Sign};
70
71    impl crate::Error for NumError {}
72    impl crate::ExtError for NumError {
73        type Kind = ();
74        fn error_eq(&self, other: &Self) -> bool {
75            self == other
76        }
77        fn error_kind(&self) -> Self::Kind {}
78    }
79
80    impl Display for NumError {
81        fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
82            use NumError as E;
83            match self {
84                E::NotImplemented => write!(f, "Not implemented."),
85                E::NotSupported => write!(f, "Not supported."),
86                E::Unspecified => write!(f, "Unspecified."),
87                E::Invalid => write!(f, "Invalid value."),
88                E::NoInverse => write!(f, "An inverse doesn't exist."),
89                E::IncompatibleBounds => {
90                    write!(f, "The given bounds are incompatible.")
91                }
92                E::MismatchedSizes => {
93                    write!(f, "The provided values are not compatible in size.")
94                }
95                E::NonNegativeRequired => write!(f, "A non-negative value is required."),
96                E::PositiveRequired => write!(f, "A positive value is required.."),
97                E::NonZeroRequired => write!(f, "A non-zero value is required."),
98                E::Overflow(sign) => {
99                    if let Some(sign) = sign {
100                        match sign {
101                            Sign::Positive => write!(f, "Positive overflow."),
102                            Sign::Negative => write!(f, "Negative overflow."),
103                            Sign::None => write!(f, "Unsigned overflow."), // not meaningful
104                        }
105                    } else {
106                        write!(f, "Overflow.")
107                    }
108                }
109            }
110        }
111    }
