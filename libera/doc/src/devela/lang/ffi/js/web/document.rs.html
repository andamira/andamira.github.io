6#[cfg(feature = "alloc")]
7use crate::String;
8use crate::js_doc;
9#[allow(unused_imports, reason = "not(windows)")]
10use crate::{Js, js_bool, js_int32, js_reexport, js_uint32};
11
12/// Handle to the brower's global [Document] associated APIs.
13///
14/// [Document](https://developer.mozilla.org/en-US/docs/Web/API/Document)
15#[repr(C)]
16#[derive(Copy, Clone)]
17pub struct WebDocument;
18
19#[rustfmt::skip]
20#[cfg(all(feature = "unsafe_ffi", not(windows)))]
21#[cfg_attr(nightly_doc, doc(cfg(all(feature = "unsafe_ffi", target_arch = "wasm32"))))]
22impl WebDocument {
23    /* flags */
24
25    #[doc = js_doc!("Document", "compatMode")]
26    /// Returns `true` if the browser is in no-quirks mode, ("CSS1Compat") or `false` otherwise.
27    ///
28    /// - <https://developer.mozilla.org/en-US/docs/Web/HTML/Guides/Quirks_mode_and_standards_mode>
29    pub fn is_compat_mode() -> js_bool { document_is_compat_mode() }
30
31    #[doc = js_doc!("Document", "hidden")]
32    /// Whether the current document is hidden.
33    pub fn is_hidden() -> js_bool { document_is_hidden() }
34
35    /* */
36
37    #[doc = js_doc!("Document", "contentType")]
38    /// Returns the document's content type.
39    #[cfg(feature = "alloc")]
40    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
41    pub fn content_type() -> String {
42        Js::read_string(|ptr, len| unsafe { document_content_type(ptr, len) })
43    }
44
45    #[doc = js_doc!("Document", "contentType")]
46    /// Returns the document's content type using the provided buffer.
47    pub fn content_type_buf(buffer: &mut [u8]) -> &str {
48        Js::read_str(buffer, |ptr, len| unsafe { document_content_type(ptr, len) })
49    }
50}
51js_reexport! {
52    [module: "api_document"]
53    safe fn document_is_compat_mode() -> js_bool;
54    safe fn document_is_hidden() -> js_bool;
55    unsafe fn document_content_type(ptr: *mut u8, max_len: js_uint32) -> js_int32;
56}
