/// An enumeration of byte multiples.
#[derive(Clone, Copy, Debug, Default)]
#[non_exhaustive]
pub enum ByteUnit {
    /// Byte.
    #[default]
    B,

    /// Kibibyte = 2^10 bytes.
    KiB,
    /// Mebibyte = 2^20 bytes.
    MiB,
    /// Gibibyte = 2^30 bytes.
    GiB,
    /// Tebibyte = 2^40 bytes.
    TiB,
    /// Pebibyte = 2^50 bytes.
    PiB,
    /// Exbibyte = 2^60 bytes.
    EiB,

    /// Megabyte = 10^6 bytes.
    MB,
    /// Kilobyte = 10^3 bytes.
    KB,
    /// Gigabyte = 10^9 bytes.
    GB,
    /// Terabyte = 10^12 bytes.
    TB,
    /// Petabyte = 10^15 bytes.
    PB,
    /// Exabyte = 10^18 bytes.
    EB,
}

/// Aliases
impl ByteUnit {
    /// Alias of `B`.
    pub const BYTE: Self = Self::B;
    //
    /// Alias of `KB`.
    #[allow(non_upper_case_globals)]
    pub const kB: Self = Self::KB;
    /// Alias of `KB`.
    pub const KILOBYTE: Self = Self::KB;
    /// Alias of `MB`.
    pub const MEGABYTE: Self = Self::MB;
    /// Alias of `GB`.
    pub const GIGABYTE: Self = Self::GB;
    /// Alias of `TB`.
    pub const TERABYTE: Self = Self::TB;
    /// Alias of `PB`.
    pub const PETABYTE: Self = Self::PB;
    /// Alias of `EB`.
    pub const EXABYTE: Self = Self::EB;
    //
    /// Alias of `KiB`.
    pub const KIBIBYTE: Self = Self::KiB;
    /// Alias of `MiB`.
    pub const MEBIBYTE: Self = Self::MiB;
    /// Alias of `GiB`.
    pub const GIBIBYTE: Self = Self::GiB;
    /// Alias of `TiB`.
    pub const TEBIBYTE: Self = Self::TiB;
    /// Alias of `PiB`.
    pub const PEBIBYTE: Self = Self::PiB;
    /// Alias of `EiB`.
    pub const EXBIBYTE: Self = Self::EiB;
}

impl ByteUnit {
    /// Returns the symbol of the unit.
    pub const fn symbol(&self) -> &str {
        match self {
            ByteUnit::B => "B",
            //
            ByteUnit::KB => "kB",
            ByteUnit::MB => "MB",
            ByteUnit::GB => "GB",
            ByteUnit::TB => "TB",
            ByteUnit::PB => "PB",
            ByteUnit::EB => "EB",
            //
            ByteUnit::KiB => "KiB",
            ByteUnit::MiB => "MiB",
            ByteUnit::GiB => "GiB",
            ByteUnit::TiB => "TiB",
            ByteUnit::PiB => "PiB",
            ByteUnit::EiB => "EiB",
        }
    }

    /// Returns the name of the unit.
    pub const fn name(&self) -> &str {
        match self {
            ByteUnit::B => "byte",
            //
            ByteUnit::KB => "kilobyte",
            ByteUnit::MB => "megabyte",
            ByteUnit::GB => "gigabyte",
            ByteUnit::TB => "terabyte",
            ByteUnit::PB => "petabyte",
            ByteUnit::EB => "exabyte",
            //
            ByteUnit::KiB => "kibibyte",
            ByteUnit::MiB => "mebibyte",
            ByteUnit::GiB => "gibibyte",
            ByteUnit::TiB => "tebibyte",
            ByteUnit::PiB => "pebibyte",
            ByteUnit::EiB => "exbibyte",
        }
    }

    /// Returns the base value corresponding to the chosen unit prefix.
    #[inline]
    #[must_use]
    pub const fn base(&self) -> i32 {
        use ByteUnit::{EiB, GiB, KiB, MiB, PiB, TiB, B, EB, GB, KB, MB, PB, TB};
        match self {
            KiB | MiB | GiB | TiB | PiB | EiB => 2,
            KB | MB | GB | TB | PB | EB => 10,
            B => 0,
        }
    }

    /// Returns the exponent corresponding to the chosen unit prefix.
    #[inline]
    #[must_use]
    pub const fn exp(&self) -> i32 {
        match self {
            ByteUnit::KiB => UnitBi::Kibi.exp(),
            ByteUnit::MiB => UnitBi::Mebi.exp(),
            ByteUnit::GiB => UnitBi::Gibi.exp(),
            ByteUnit::TiB => UnitBi::Tebi.exp(),
            ByteUnit::PiB => UnitBi::Pebi.exp(),
            ByteUnit::EiB => UnitBi::Exbi.exp(),
            //
            ByteUnit::KB => UnitSi::Kilo.exp(),
            ByteUnit::MB => UnitSi::Mega.exp(),
            ByteUnit::GB => UnitSi::Giga.exp(),
            ByteUnit::TB => UnitSi::Tera.exp(),
            ByteUnit::PB => UnitSi::Peta.exp(),
            ByteUnit::EB => UnitSi::Exa.exp(),
            //
            ByteUnit::B => 0,
        }
    }

    /// Returns the multiplication factor for the unit prefix as an f64.
    pub const fn factor(&self) -> f64 {
        match self {
            ByteUnit::KiB => UnitBi::Kibi.factor(),
            ByteUnit::MiB => UnitBi::Mebi.factor(),
            ByteUnit::GiB => UnitBi::Gibi.factor(),
            ByteUnit::TiB => UnitBi::Tebi.factor(),
            ByteUnit::PiB => UnitBi::Pebi.factor(),
            ByteUnit::EiB => UnitBi::Exbi.factor(),
            //
            ByteUnit::KB => UnitSi::Kilo.factor(),
            ByteUnit::MB => UnitSi::Mega.factor(),
            ByteUnit::GB => UnitSi::Giga.factor(),
            ByteUnit::TB => UnitSi::Tera.factor(),
            ByteUnit::PB => UnitSi::Peta.factor(),
            ByteUnit::EB => UnitSi::Exa.factor(),
            //
            ByteUnit::B => 1.,
        }
    }

    /// Returns the multiplication factor for the unit prefix as an i64.
    pub const fn factor_i64(&self) -> i64 {
        match self {
            ByteUnit::KiB => UnitBi::Kibi.factor_i64(),
            ByteUnit::MiB => UnitBi::Mebi.factor_i64(),
            ByteUnit::GiB => UnitBi::Gibi.factor_i64(),
            ByteUnit::TiB => UnitBi::Tebi.factor_i64(),
            ByteUnit::PiB => UnitBi::Pebi.factor_i64(),
            ByteUnit::EiB => UnitBi::Exbi.factor_i64(),
            //
            ByteUnit::KB => UnitSi::Kilo.factor_i64(),
            ByteUnit::MB => UnitSi::Mega.factor_i64(),
            ByteUnit::GB => UnitSi::Giga.factor_i64(),
            ByteUnit::TB => UnitSi::Tera.factor_i64(),
            ByteUnit::PB => UnitSi::Peta.factor_i64(),
            ByteUnit::EB => UnitSi::Exa.factor_i64(),
            //
            ByteUnit::B => 1,
        }
    }

    /// Returns the multiplication factor for the unit prefix as an i64.
    pub const fn factor_i128(&self) -> i128 {
        match self {
            ByteUnit::KiB => UnitBi::Kibi.factor_i128(),
            ByteUnit::MiB => UnitBi::Mebi.factor_i128(),
            ByteUnit::GiB => UnitBi::Gibi.factor_i128(),
            ByteUnit::TiB => UnitBi::Tebi.factor_i128(),
            ByteUnit::PiB => UnitBi::Pebi.factor_i128(),
            ByteUnit::EiB => UnitBi::Exbi.factor_i128(),
            //
            ByteUnit::KB => UnitSi::Kilo.factor_i128(),
            ByteUnit::MB => UnitSi::Mega.factor_i128(),
            ByteUnit::GB => UnitSi::Giga.factor_i128(),
            ByteUnit::TB => UnitSi::Tera.factor_i128(),
            ByteUnit::PB => UnitSi::Peta.factor_i128(),
            ByteUnit::EB => UnitSi::Exa.factor_i128(),
            //
            ByteUnit::B => 1,
        }
    }

    /// Reduces the given `value` to the most appropriate binary prefix as an f64,
    /// returning a tuple of the reduced size and the prefix.
    #[must_use]
    #[cfg(any(feature = "std", feature = "_float_f64"))]
    #[cfg_attr(feature = "nightly_doc", doc(cfg(any(feature = "std", feature = "_float_f64"))))]
    pub fn reduce(bytes: f64) -> (f64, Self) {
        let (value, unit) = UnitBi::reduce(bytes);
        (value, unit.try_into().expect("UnitBi to ByteUnit"))
    }

    /// Reduces the given `value` to the most appropriate SI prefix as an f64,
    /// returning a tuple of the reduced size and the prefix.
    #[must_use]
    #[cfg(any(feature = "std", feature = "_float_f64"))]
    #[cfg_attr(feature = "nightly_doc", doc(cfg(any(feature = "std", feature = "_float_f64"))))]
    pub fn reduce_si(bytes: f64) -> (f64, Self) {
        let (value, unit) = UnitBi::reduce(bytes);
        (value, unit.try_into().expect("UnitSi to ByteUnit"))
    }

    /// Reduces the `bytes` value to the most appropriate binary prefix as an `i64`,
    /// returning a tuple of the reduced magnitude, the prefix, and the remainder.
    ///
    /// The input `bytes` value is assumed to be a non-negative number of bytes.
    // FIX: overflow
    #[must_use]
    pub const fn reduce_i64(bytes: i64) -> (i64, Self, i64) {
        let (value, unit, remainder) = UnitBi::reduce_i64(bytes);
        (value, unwrap![ok_expect Self::try_from_unit_bi(unit), "UnitBi to ByteUnit"], remainder)
    }

    /// Reduces the `bytes` value to the most appropriate binary prefix as an `i64`,
    /// returning a tuple of the reduced magnitude, the prefix, and the remainder.
    ///
    /// The input `bytes` value is assumed to be a non-negative number of bytes.
    #[must_use]
    // FIX: panic for reduced unit too high
    pub const fn reduce_i128(bytes: i128) -> (i128, Self, i128) {
        let (value, unit, remainder) = UnitBi::reduce_i128(bytes);
        (value, unwrap![ok_expect Self::try_from_unit_bi(unit), "UnitBi to ByteUnit"], remainder)
    }

impl ByteUnit {
    /// Returns an iterator of binary units in ascending order of magnitude.
    pub fn asc_iter() -> impl Iterator<Item = Self> {
        const UNITS: [ByteUnit; 7] = [
            ByteUnit::B,
            ByteUnit::KiB,
            ByteUnit::MiB,
            ByteUnit::GiB,
            ByteUnit::TiB,
            ByteUnit::PiB,
            ByteUnit::EiB,
        ];
        UNITS.iter().copied()
    }

    /// Returns an iterator of binary units in descending order of magnitude.
    pub fn desc_iter() -> impl Iterator<Item = Self> {
        const UNITS: [ByteUnit; 7] = [
            ByteUnit::EiB,
            ByteUnit::PiB,
            ByteUnit::TiB,
            ByteUnit::GiB,
            ByteUnit::MiB,
            ByteUnit::KiB,
            ByteUnit::B,
        ];
        UNITS.iter().copied()
    }

    /// Returns an iterator in ascending order of magnitude.
    pub fn asc_iter_si() -> impl Iterator<Item = Self> {
        const UNITS: [ByteUnit; 7] = [
            ByteUnit::B,
            ByteUnit::KB,
            ByteUnit::MB,
            ByteUnit::GB,
            ByteUnit::TB,
            ByteUnit::PB,
            ByteUnit::EB,
        ];
        UNITS.iter().copied()
    }

    /// Returns an iterator in descending order of magnitude.
    pub fn desc_iter_si() -> impl Iterator<Item = Self> {
        const UNITS: [ByteUnit; 7] = [
            ByteUnit::EB,
            ByteUnit::PB,
            ByteUnit::TB,
            ByteUnit::GB,
            ByteUnit::MB,
            ByteUnit::KB,
            ByteUnit::B,
        ];
        UNITS.iter().copied()
    }
}

/* conversions from/into UnitBi/UnitSi */

impl ByteUnit {
    /// const `TryInto<UnitBi> for ByteUnit`.
    pub const fn try_into_unit_bi(from: ByteUnit) -> Result<UnitBi, NumError> {
        match from {
            ByteUnit::KiB => Ok(UnitBi::Kibi),
            ByteUnit::MiB => Ok(UnitBi::Mebi),
            ByteUnit::GiB => Ok(UnitBi::Gibi),
            ByteUnit::TiB => Ok(UnitBi::Tebi),
            ByteUnit::PiB => Ok(UnitBi::Pebi),
            ByteUnit::EiB => Ok(UnitBi::Exbi),
            ByteUnit::B => Ok(UnitBi::None),
            _ => Err(NumError::NotSupported),
        }
    }
    /// const `TryInto<ByteUnit> for UnitBi`.
    pub const fn try_from_unit_bi(from: UnitBi) -> Result<ByteUnit, NumError> {
        match from {
            UnitBi::Exbi => Ok(ByteUnit::EiB),
            UnitBi::Pebi => Ok(ByteUnit::PiB),
            UnitBi::Tebi => Ok(ByteUnit::TiB),
            UnitBi::Gibi => Ok(ByteUnit::GiB),
            UnitBi::Mebi => Ok(ByteUnit::MiB),
            UnitBi::Kibi => Ok(ByteUnit::KiB),
            UnitBi::None => Ok(ByteUnit::B),
            _ => Err(NumError::NotSupported),
        }
    }
    /// const `TryInto<UnitBi> for ByteUnit`.
    pub const fn try_into_unit_si(from: ByteUnit) -> Result<UnitSi, NumError> {
        match from {
            ByteUnit::KB => Ok(UnitSi::Kilo),
            ByteUnit::MB => Ok(UnitSi::Mega),
            ByteUnit::GB => Ok(UnitSi::Giga),
            ByteUnit::TB => Ok(UnitSi::Tera),
            ByteUnit::PB => Ok(UnitSi::Peta),
            ByteUnit::EB => Ok(UnitSi::Exa),
            ByteUnit::B => Ok(UnitSi::None),
            _ => Err(NumError::NotSupported),
        }
    }
    /// const `TryInto<ByteUnit> for UnitBi`.
    pub const fn try_from_unit_si(from: UnitSi) -> Result<ByteUnit, NumError> {
        match from {
            UnitSi::Exa => Ok(ByteUnit::EB),
            UnitSi::Peta => Ok(ByteUnit::PB),
            UnitSi::Tera => Ok(ByteUnit::TB),
            UnitSi::Giga => Ok(ByteUnit::GB),
            UnitSi::Mega => Ok(ByteUnit::MB),
            UnitSi::Kilo => Ok(ByteUnit::KB),
            UnitSi::None => Ok(ByteUnit::B),
            _ => Err(NumError::NotSupported),
        }
    }
}

impl TryFrom<ByteUnit> for UnitBi {
    type Error = NumError;
    #[inline]
    fn try_from(from: ByteUnit) -> Result<UnitBi, Self::Error> {
        ByteUnit::try_into_unit_bi(from)
    }
}
impl TryFrom<UnitBi> for ByteUnit {
    type Error = NumError;
    #[inline]
    fn try_from(from: UnitBi) -> Result<ByteUnit, Self::Error> {
        ByteUnit::try_from_unit_bi(from)
    }
}
impl TryFrom<ByteUnit> for UnitSi {
    type Error = NumError;
    fn try_from(from: ByteUnit) -> Result<UnitSi, Self::Error> {
        ByteUnit::try_into_unit_si(from)
    }
}
impl TryFrom<UnitSi> for ByteUnit {
    type Error = NumError;
    fn try_from(from: UnitSi) -> Result<ByteUnit, Self::Error> {
        ByteUnit::try_from_unit_si(from)
    }
}

impl<T> ByteSized for T {}

/// Type size information in bytes.
///
/// This trait is automatically implemented for every `Sized` type.
// (this allows to have associated constants depending on Self)
pub trait ByteSized: Sized {
    /// The alignment of this type in bytes.
    const BYTE_ALIGN: usize = align_of::<Self>();

    /// The size of this type in bytes.
    const BYTE_SIZE: usize = size_of::<Self>();

    /// The size of a pointer in bytes, for the current platform.
    const PTR_BYTES: usize = size_of::<usize>();

    /// The size of a pointer in bits, for the current platform.
    const PTR_BITS: usize = usize::BITS as usize;

    /// True if the system's architecture is little-endian.
    const LITTLE_ENDIAN: bool = cfg!(target_endian = "little");

    /// True if the system's architecture is big-endian.
    const BIG_ENDIAN: bool = cfg!(target_endian = "big");

    /// Returns the alignment of this type in bytes.
    #[inline]
    fn byte_align(&self) -> usize {
        align_of_val(self)
    }

    /// Returns the size of this type in bytes.
    ///
    /// Ignores any allocated resources in the heap.
    #[inline]
    fn byte_size(&self) -> usize {
        size_of_val(self)
    }

    /// Returns the size ratio between [`PTR_BYTES`][Self#associatedconstant.PTR_BYTES]
    /// and [`BYTE_SIZE`][Self#associatedconstant.BYTE_SIZE].
    ///
    /// For example: the ratio will be (1, 1) if both sizes are the same,
    /// (2, 1) if a pointer is double the byte size, and (1, 2) if a pointer is
    /// half the byte size.
    ///
    /// # Examples
    /// ```
    /// use devela::mem::ByteSized;
    ///
    /// assert_eq![().ptr_size_ratio(), [1, 0]];
    /// assert_eq![1_usize.ptr_size_ratio(), [1, 1]];
    /// assert_eq!["slice".ptr_size_ratio(), [1, 2]];
    /// assert_eq![String::from("hello").ptr_size_ratio(), [1, 3]];
    ///
    /// #[cfg(target_pointer_width = "64")]
    /// {
    ///     assert_eq![0_u8.ptr_size_ratio(), [8, 1]];
    ///     assert_eq![0_u16.ptr_size_ratio(), [4, 1]];
    ///     assert_eq![0_u32.ptr_size_ratio(), [2, 1]];
    ///     assert_eq![0_u64.ptr_size_ratio(), [1, 1]];
    ///     assert_eq![0_u128.ptr_size_ratio(), [1, 2]];
    ///     assert_eq!['c'.ptr_size_ratio(), [2, 1]];
    ///     assert_eq!["slice".ptr_size_ratio(), [1, 2]];
    /// }
    /// ```
    ///
    /// For the `const` version see [`ptr_size_ratio`].
    #[inline]
    fn ptr_size_ratio(&self) -> [usize; 2] {
        ptr_size_ratio(Self::BYTE_SIZE)
    }
}

/* fn definitions */

/// Returns the ratio of a `usize` in respect to `other_size`.
///
/// For example: the ratio will be `(1, 1)` if both sizes are equal, `(2, 1)`
/// if the pointer size is double the other size, and `(1, 2)` if is is half
/// the other byte size.
///
/// # Examples
/// ```
/// use devela::mem::ptr_size_ratio;
///
/// assert_eq![ptr_size_ratio(0), [1, 0]];
/// assert_eq![ptr_size_ratio(size_of::<usize>()), [1, 1]];
/// assert_eq![ptr_size_ratio(size_of::<&str>()), [1, 2]];
/// assert_eq![ptr_size_ratio(size_of::<String>()), [1, 3]];
///
/// #[cfg(target_pointer_width = "64")]
/// assert_eq![ptr_size_ratio(size_of::<char>()), [2,1]];
/// ```
///
/// Note that when `other_size == 0` it returns `(1, 0)` which is an invalid ratio.
///
#[inline]
#[allow(dead_code)]
pub const fn ptr_size_ratio(other_size: usize) -> [usize; 2] {
    #[inline]
    const fn gcd(m: usize, n: usize) -> usize {
        iif![n == 0; m; gcd(n, m % n)]
    }
    let g = gcd(size_of::<usize>(), other_size);
    [size_of::<usize>() / g, other_size / g]
}

/// Returns the rounded up size in bytes from a size in bits.
///
/// This is equivalent to `(bit_size + 7) / 8` but handles potential overflow.
#[must_use]
#[inline]
pub const fn bytes_from_bits(bit_size: usize) -> usize {
    if let Some(t) = bit_size.checked_add(8 - 1) {
        t / 8
    } else {
        bytes_from_bits_cold()
    }
}
#[cold] #[rustfmt::skip]
const fn bytes_from_bits_cold() -> usize { usize::MAX / 8 }
