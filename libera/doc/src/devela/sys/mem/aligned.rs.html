6/// Marker trait to prevent downstream implementations of the [`MemAligned`] trait.
7trait Sealed {}
8impl<Candidate, Requirement> Sealed for (Candidate, Requirement) {}

10/// Marker trait to verify memory alignment between two types.
11///
12/// This trait ensures that a value of the `Candidate` type can be safely placed
13/// into a storage medium designed for the `Requirement` type, adhering to
14/// alignment requirements.
15///
16/// For runtime alignment adjustments, see [`Mem::align_down`] and [`Mem::align_up`].
17//
18// WAIT:DONE:1.79 [ptr.is_aligned](https://github.com/rust-lang/rust/pull/121948)
19// WAIT: [pointer_is_aligned_to](https://github.com/rust-lang/rust/issues/96284)
20// WAIT: [const_pointer_is_aligned](https://github.com/rust-lang/rust/issues/104203)
21#[cfg_attr(nightly_doc, doc(notable_trait))]
22#[expect(private_bounds, reason = "Sealed")]
23pub trait MemAligned: Sealed {
24    /// Checks if the `Candidate`'s alignment is compatible with the `Requirement`'s alignment.
25    ///
26    /// Returns `true` if `Candidate` type's alignment is less than or equal to the
27    /// `Requirement` type's alignment.
28    ///
29    /// Alignment represents the byte boundary to which a type must be aligned
30    /// in memory. It is crucial to ensure that types are placed in memory
31    /// locations that respect their alignment requirements.
32    #[must_use]
33    fn is_compatible() -> bool;
34
35    /// Asserts that the memory alignment of the `Candidate` type meets or
36    /// exceeds the `Requirement` type's alignment.
37    ///
38    /// # Panics
39    /// Panics if the `Candidate` type's alignment is greater than the
40    /// `Requirement` type's alignment, indicating an incompatibility that could
41    /// lead to undefined behavior or performance penalties.
42    fn assert_compatibility();
43}
44
45// #[cfg(not(feature = "nightly_const"))]
46impl<Candidate, Requirement> MemAligned for (Candidate, Requirement) {
47    fn is_compatible() -> bool {
48        align_of::<Candidate>() <= align_of::<Requirement>()
49    }
50
51    fn assert_compatibility() {
52        assert!(
53            Self::is_compatible(),
54            "`Candidate`'s alignment ({}) exceeds `Requirement`'s alignment ({}).",
55            align_of::<Candidate>(),
56            align_of::<Requirement>(),
57        );
58    }
59}
