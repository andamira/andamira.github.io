42/// # Read-related methods.

45impl Linux {
46    /// Checks if there is input available to read from stdin.
47    pub fn has_input() -> bool { Self::available_bytes().unwrap_or(0) > 0 }
48
49    /// Returns the number of bytes available to be read from *stdin*.
50    pub fn available_bytes() -> Result<usize> {
51        let mut n = 0;
52        let result = unsafe {
53            Linux::sys_ioctl(FILENO::STDIN, IOCTL::FIONREAD, &mut n as *mut i32 as *mut u8)
54        };
55        if result < 0 { Err(LinuxError::Sys(result)) } else { Ok(n as usize) }
56    }
57
58    /// Gets a single byte from *stdin*.
59    pub fn get_byte() -> Result<u8> {
60        let mut c = 0;
61        loop {
62            let n = unsafe { Linux::sys_read(FILENO::STDIN, &mut c as *mut u8, 1) };
63            if n < 0 { return Err(LinuxError::Sys(n)); }
64            if n == 1 { break; }
65        }
66        Ok(c)
67    }
68
69    /// Attempts to get a single byte from stdin without blocking.
70    ///
71    /// Checks if there is input available using `has_input`,
72    /// and if so, reads a byte using `get_byte`.
73    ///
74    /// # Returns
75    /// - `Some(u8)` if a byte is successfully read.
76    /// - `None` if no data is available.
77    pub fn try_get_byte() -> Option<u8> {
78        if Linux::has_input() { Linux::get_byte().ok() } else { None }
79    }
80
81    /// Pauses execution until receiving from *stdin* any `char` in the `list`.
82    pub fn pause_until_char(list: &[char]) {
83        loop { if list.contains(&Linux::get_dirty_char()) { break; } }
84    }
85
86    /// Gets a single `char` from *stdin*.
87    ///
88    /// # Returns
89    /// - `Ok(char)` if a valid UTF-8 character is read.
90    /// - `Err(LinuxError::InvalidUtf8)` if the bytes are not valid UTF-8.
91    /// - `Err(LinuxError::NoInput)` if the UTF-8 sequence is valid but empty.
92    pub fn get_char() -> Result<char> {
93        let bytes = Linux::get_utf8_bytes()?;
94        let s = unsafe { from_utf8_unchecked(&bytes) };
95        s.chars().next().ok_or(LinuxError::NoInput)
96    }
97
98    /// Gets either a single `char` from *stdin*, or the replacement character.
99    ///
100    /// If the bytes received doesn't form a valid unicode scalar then the
101    /// [replacement character (�)][char::REPLACEMENT_CHARACTER] will be returned.
102    pub fn get_dirty_char() -> char {
103        match Linux::get_utf8_bytes() {
104            Ok(bytes) => {
105                let s = unsafe { from_utf8_unchecked(&bytes) };
106                s.chars().next().unwrap()
107            }
108            Err(_) => char::REPLACEMENT_CHARACTER,
109        }
110    }
111
112    /// Gets a UTF-8 encoded byte sequence from *stdin* representing a `char`.
113    pub fn get_utf8_bytes() -> Result<[u8; 4]> {
114        let mut bytes = [0u8; 4];
115        let len;
116        // read the first byte to determine the length of the character
117        bytes[0] = Linux::get_byte()?;
118        if bytes[0] & 0x80 == 0 { return Ok([bytes[0], 0, 0, 0]); } // ASCII char
119        // IMPROVE: use text module functionality
120        else if bytes[0] & 0xE0 == 0xC0 { len = 2; }
121        else if bytes[0] & 0xF0 == 0xE0 { len = 3; }
122        else if bytes[0] & 0xF8 == 0xF0 { len = 4; }
123        else { return Err(LinuxError::InvalidUtf8) }
124        // read the remaining bytes of the character
125        for i in 1..len {
126            bytes[i as usize] = Linux::get_byte()?;
127            if bytes[i as usize] & 0xC0 != 0x80 { return Err(LinuxError::InvalidUtf8); }
128        }
129        Ok(bytes)
130    }
131
132    /// Reads all available bytes from stdin.
133    ///
134    /// # Returns
135    /// - `Ok(Vec<u8>)` containing the bytes read.
136    /// - `Err(isize)` if the read fails, returning the error code.
137    #[cfg(feature = "alloc")]
138    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
139    pub fn read_available_bytes() -> Result<Vec<u8>> {
140        let count = Linux::available_bytes()?;
141        if count == 0 { return Ok(Vec::new()); }
142        let mut buffer = crate::vec_![0u8; count];
143        let n = unsafe { Linux::sys_read(FILENO::STDIN, buffer.as_mut_ptr(), count) };
144        if n < 0 { Err(LinuxError::Sys(n)) } else { buffer.truncate(n as usize); Ok(buffer) }
145    }
146
147    /// Prompts the user for a string from *stdin*, backed by a `buffer`.
148    ///
149    /// # Examples
150    /// ```no_run
151    /// # use devela::Linux;
152    /// let mut name_buffer = [0_u8; 32];
153    /// let name: &str = Linux::prompt::<32>("Enter your name: ", &mut name_buffer).unwrap();
154    /// ```
155    pub fn prompt<'i, const CAP: usize>(text: &str, buffer: &'i mut [u8; CAP]) -> Result<&'i str> {
156        Linux::print(text)?; Linux::get_line(buffer)
157    }
158
159    /// Gets a string from *stdin* backed by a `buffer`, until a newline.
160    ///
161    /// # Examples
162    /// ```no_run
163    /// # use devela::Linux;
164    /// let mut buf = [0_u8; 32];
165    /// let name: &str = Linux::get_line::<32>(&mut buf).unwrap();
166    /// ```
167    pub fn get_line<const CAP: usize>(buffer: &mut [u8; CAP]) -> Result<&str> {
168        Linux::get_str(buffer, '\n')
169    }
170
171    /// Gets a string from *stdin* backed by a `buffer`, until the `stop` char is received.
172    ///
173    /// # Examples
174    /// ```no_run
175    /// # use devela::Linux;
176    /// let mut buf = [0_u8; 32];
177    /// let name: &str = Linux::get_str::<32>(&mut buf, '\n').unwrap();
178    /// ```
179    pub fn get_str<const CAP: usize>(buffer: &mut [u8; CAP], stop: char) -> Result<&str> {
180        let mut index = 0;
181        loop {
182            if let Ok(c) = Linux::get_char() {
183                let mut c_buf = [0; 4];
184                let c_str = c.encode_utf8(&mut c_buf);
185                if c == stop {
186                    break;
187                } else if index + c_str.len() <= CAP {
188                    // Linux::print(c_str)?;
189                    for &b in c_str.as_bytes() {
190                        buffer[index] = b;
191                        index += 1;
192                    }
193                }
194            }
195        }
196        Ok(unsafe { from_utf8_unchecked(&buffer[..index]) })
197    }
198}

206    const STACK_BUFFER_LEN: usize = 512; // PIPE_BUF is typically 4096 but 512 is safe
207
208    /// Writes all bytes to a file descriptor, handling partial writes.
209    fn write_all(fd: c_int, mut buf: &[u8]) -> Result<()> {
210        while !buf.is_empty() {
211            let n = unsafe { Linux::sys_write(fd, buf.as_ptr(), buf.len()) };
212            if n < 0 { return Err(LinuxError::Sys(n)); }
213            buf = &buf[n as usize..];
214        }
215        Ok(())
216    }
217    /// Writes all bytes to stdout. Returns error on syscall failure.
218    fn write_all_unchecked(fd: c_int, mut buf: &[u8]) {
219        while !buf.is_empty() {
220            let n = unsafe { Linux::sys_write(fd, buf.as_ptr(), buf.len()) };
221            if n <= 0 { panic!("write failed with return value {n}"); }
222            buf = &buf[n as usize..];
223        }
224    }
225
226    /* public methods */
227
228    /// Writes string + newline to stdout. Optimized for small strings.
229    pub fn print(s: &str) -> Result<()> {
230        Self::write_all(FILENO::STDOUT, s.as_bytes())
231    }
232    /// Like `print`, but panics on failure instead of returning errors.
233    pub fn print_unchecked(s: &str) { Self::write_all_unchecked(FILENO::STDOUT, s.as_bytes()); }
234
235    /// Writes string + newline to stdout. Optimized for small strings.
236    pub fn println(s: &str) -> Result<()> {
237        if s.len() <= Self::STACK_BUFFER_LEN { // small string optimization:
238            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
239            let bytes = s.as_bytes();
240            buf[..bytes.len()].copy_from_slice(bytes);
241            buf[bytes.len()] = b'\n';
242            Self::write_all(FILENO::STDOUT, &buf[..bytes.len() + 1])
243        } else { // fallback for large strings:
244            Self::write_all(FILENO::STDOUT, s.as_bytes())?;
245            Self::write_all(FILENO::STDOUT, b"\n")
246        }
247    }
248    /// Like `println`, but panics on failure instead of returning errors.
249    pub fn println_unchecked(s: &str) {
250        if s.len() <= Self::STACK_BUFFER_LEN {
251            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
252            let bytes = s.as_bytes();
253            buf[..bytes.len()].copy_from_slice(bytes);
254            buf[bytes.len()] = b'\n';
255            Self::write_all_unchecked(FILENO::STDOUT, &buf[..bytes.len() + 1]);
256        } else {
257            Self::write_all_unchecked(FILENO::STDOUT, s.as_bytes());
258            Self::write_all_unchecked(FILENO::STDOUT, b"\n");
259        }
260    }
261    /// Ultra-fast stdout write. Panics if not all bytes are written at once.
262    /// Only use for small strings (<512 bytes).
263    pub fn print_unchecked_fast(s: &str) {
264        let bytes = s.as_bytes();
265        let n = unsafe { Linux::sys_write(FILENO::STDOUT, bytes.as_ptr(), bytes.len()) };
266        if n != bytes.len() as isize {
267            panic!("write failed with return value {n}");
268        }
269    }
270    /// Ultra-fast stdout write with newline. Panics if write isn't atomic.
271    /// Only use for small strings (<511 bytes).
272    pub fn println_unchecked_fast(s: &str) {
273        let bytes = s.as_bytes();
274        let mut buf = MaybeUninit::<[u8; Self::STACK_BUFFER_LEN]>::uninit();
275        // SAFETY: we're about to initialize all accessed bytes
276        unsafe {
277            let buf_ptr = buf.as_mut_ptr() as *mut u8;
278            core::ptr::copy_nonoverlapping(bytes.as_ptr(), buf_ptr, bytes.len());
279            *buf_ptr.add(bytes.len()) = b'\n'; // Add newline
280            let n = Linux::sys_write(FILENO::STDOUT, buf_ptr, bytes.len() + 1); // single syscall
281            if n != (bytes.len() + 1) as isize { panic!("write failed with return value {n}"); }
282        }
283    }
284    /// Writes bytes to stdout. Returns error on syscall failure.
285    pub fn print_bytes(b: &[u8]) -> Result<()> { Self::write_all(FILENO::STDOUT, b) }
286    /// Like `print_bytes`, but panics on failure instead of returning errors.
287    pub fn print_bytes_unchecked(b: &[u8]) { Self::write_all_unchecked(FILENO::STDOUT, b); }
288
289    /// Writes all bytes to stderr. Returns error on syscall failure.
290    pub fn eprint(s: &str) -> Result<()> { Self::write_all(FILENO::STDERR, s.as_bytes()) }
291    /// Writes string + newline to stderr. Optimized for small strings.
292    pub fn eprintln(s: &str) -> Result<()> {
293        if s.len() <= Self::STACK_BUFFER_LEN { // small string optimization:
294            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
295            let bytes = s.as_bytes();
296            buf[..bytes.len()].copy_from_slice(bytes);
297            buf[bytes.len()] = b'\n';
298            Self::write_all(FILENO::STDERR, &buf[..bytes.len() + 1])
299        } else { // fallback for large strings:
300            Self::write_all(FILENO::STDERR, s.as_bytes())?;
301            Self::write_all(FILENO::STDERR, b"\n")
302        }
303    }
304    /// Writes bytes to stderr. Returns error on syscall failure.
305    pub fn eprint_bytes(b: &[u8]) -> Result<()> {
306        Self::write_all(FILENO::STDERR, b)
307    }

373    /// Suspends execution of calling thread for the given `duration`.
374    pub fn sleep(duration: Duration) -> Result<()> {
375        let mut req = LinuxTimespec::with(duration);
376        let mut rem = LinuxTimespec::default();
377        loop {
378            let n = unsafe { Linux::sys_nanosleep(req.as_ptr(), rem.as_mut_ptr()) };
379            if n == 0 {
380                break Ok(()); // Success
381            } else if n == -ERRNO::EINTR {
382                req = rem; // Retry with remaining time
383                continue;
384            } else { // Actual error
385                return Err(LinuxError::Sys(n));
386            }
387        }
388    }
389    /// Suspends execution of calling thread for the given `milliseconds`.
390    pub fn sleep_ms(milliseconds: u64) -> Result<()> {
391        Linux::sleep(Duration::from_millis(milliseconds))
392    }

394    /// Returns the current process number.
395    pub fn getpid() -> i32 {
396        unsafe { Linux::sys_getpid() }

405    const LINUX_SIG_MAX: usize = 30;
406    /// A static array to match signals with handlers.
407    static LINUX_SIG_HANDLERS: [AtomicPtr<fn(i32)>; LINUX_SIG_MAX] = {

409        [INIT; LINUX_SIG_MAX]

413/// # Signaling-related methods.

417    /// Registers multiple signals using a handler function.
418    ///
419    /// # Notes
420    /// - Unknown flags in the `flags` slice are ignored, and a warning is printed.
421    ///
422    /// # Arguments
423    /// - `handler`: A function that will be called when one of the specified signals is received.
424    ///   The function takes a single argument for the signal number.
425    /// - `signals`: A slice of [`LINUX_SIGNAL`][super::LINUX_SIGNAL] values specifying the signals to handle.
426    /// - `flags`: An optional slice of [`LINUX_SIGACTION`][super::LINUX_SIGACTION] flags.
427    ///   If `None`, only the `SA_RESTORER` flag is used.
428    ///
429    /// # Notes
430    /// - The `SA_RESTORER` flag is always included.
431    /// - The `SA_SIGINFO` flag is ignored. Use [`sig_handler_info`][Self::sig_handler_info] instead.
432    /// - Unknown flags in the `flags` slice are ignored, and a warning is printed.
433    ///
434    /// # Examples
435    /// ```no_run
436    /// # #[cfg(feature = "std")] {
437    /// # use devela::{Linux, LINUX_SIGNAL, sleep4};
438    /// fn handler(sig: i32) {
439    ///    println!("\nsignal `{sig}` received! continuing. . .");
440    /// }
441    /// fn main() {
442    ///     println!("press Ctrl+C, or resize the terminal to catch the signals");
443    ///     Linux::sig_handler(handler, &[LINUX_SIGNAL::SIGINT, LINUX_SIGNAL::SIGWINCH], None);
444    ///     sleep4!(5);
445    ///     println!("bye");
446    /// }
447    /// # }
448    /// ```
449    pub fn sig_handler(handler: fn(i32), signals: &[i32], flags: Option<&[usize]>) {
450        extern "C" fn c_handler(sig: i32) {
451            if sig >= 0 && (sig as usize) < LINUX_SIG_MAX {
452                let handler = LINUX_SIG_HANDLERS[sig as usize].load(AtomicOrdering::SeqCst);
453                if !handler.is_null() {
454                    unsafe {
455                        let handler: fn(i32) = transmute(handler);
456                        handler(sig);
457                    }
458                }
459            }
460        }
461        // Use the restorer function defined in assembly.
462        unsafe extern "C" { safe fn __devela_linux_restore_rt(); }
463
464        // Start with the SA_RESTORER flag, as it is always required.
465        let mut combined_flags = SIGACTION::SA_RESTORER;
466        // Add additional flags if provided.
467        if let Some(provided_flags) = flags {
468            for &flag in provided_flags {
469                match flag { // Validate the flag to ensure it is one of the known values.
470                    SIGACTION::SA_NOCLDSTOP
471                    | SIGACTION::SA_NOCLDWAIT
472                    | SIGACTION::SA_NODEFER
473                    | SIGACTION::SA_ONSTACK
474                    | SIGACTION::SA_RESETHAND
475                    | SIGACTION::SA_RESTART => {
476                        combined_flags |= flag;
477                    }
478                    _ => {
479                        #[cfg(feature = "std")]
480                        eprintln!("Warning: Ignoring unknown flag: {flag:#x}");
481                    }
482                }
483            }
484        }
485        let mask = LinuxSigset::empty();
486        for &sig in signals {
487            if (1..=LINUX_SIG_MAX).contains(&(sig as usize)) { // make sure the signal is a valid number
488                // Store the handler in the appropriate slot
489                LINUX_SIG_HANDLERS[sig as usize].store(handler as *mut _, AtomicOrdering::SeqCst);
490                // Register the handler with the OS
491                let c_handler = c_handler as extern "C" fn(i32);
492                let sigaction = LinuxSigaction::new(c_handler,
493                    combined_flags, mask, Some(__devela_linux_restore_rt));
494                unsafe {
495                    let _ = Linux::sys_rt_sigaction(sig, &sigaction,
496                        Ptr::null_mut(), LinuxSigset::size());
497                }
498            }
499        }
500    }
501
502    /// Registers multiple signals using a handler function accepting additional signal information.
503    ///
504    /// # Arguments
505    /// - `handler`: A function that will be called when one of the specified signals is received.
506    ///   The function takes three arguments:
507    ///   - `i32`: The signal number.
508    ///   - `LinuxSiginfo`: Additional information about the signal.
509    ///   - `*mut c_void`: A pointer to the signal context (ucontext_t).
510    /// - `signals`: A slice of [`LINUX_SIGNAL`][super::LINUX_SIGNAL] values specifying the signals to handle.
511    /// - `flags`: An optional slice of [`LINUX_SIGACTION`][super::LINUX_SIGACTION] flags.
512    ///   If `None`, only the `SA_RESTORER` and `SA_SIGINFO` flags are used.
513    ///
514    /// # Notes
515    /// - The `SA_RESTORER` and `SA_SIGINFO` flags are always included.
516    /// - Unknown flags in the `flags` slice are ignored, and a warning is printed.
517    ///
518    /// # Examples
519    /// ```no_run
520    /// # use devela::{c_void, sleep4, Linux, LinuxSiginfo, LINUX_SIGNAL, LINUX_SIGACTION};
521    /// fn handler(sig: i32, info: LinuxSiginfo, _context: *mut c_void) {
522    ///     println!("Signal {} received from PID {}!", sig, info.pid());
523    /// }
524    /// fn main() {
525    ///     Linux::sig_handler_info(handler, &[LINUX_SIGNAL::SIGINT],
526    ///         Some(&[LINUX_SIGACTION::SA_RESETHAND, LINUX_SIGACTION::SA_RESTART]),
527    ///     );
528    ///     println!("press Ctrl+C, to trigger SIGINT");
529    ///     loop {
530    ///         println!("Waiting for signal...");
531    ///         sleep4!(1);
532    ///     }
533    /// }
534    /// ```
535    pub fn sig_handler_info(
536        handler: fn(i32, LinuxSiginfo, *mut c_void),
537        signals: &[i32],
538        flags: Option<&[usize]>,
539    ) {
540        // We store the given `handler` function in a static to be able to call it
541        // from the new extern function which can't capture its environment.
542        static HANDLER: AtomicPtr<fn(i32, LinuxSiginfo, *mut c_void)>
543            = AtomicPtr::new(Ptr::null_mut());
544        HANDLER.store(handler as *mut _, AtomicOrdering::SeqCst);
545
546        extern "C" fn c_handler_siginfo(sig: i32, info: LinuxSiginfo, context: *mut c_void) {
547            let handler = HANDLER.load(AtomicOrdering::SeqCst);
548            if !handler.is_null() {
549                #[expect(clippy::crosspointer_transmute)]
550                let handler = unsafe {
551                    transmute::<*mut fn(i32, LinuxSiginfo, *mut c_void),
552                        fn(i32, LinuxSiginfo, *mut c_void)>(handler)
553                };
554                handler(sig, info, context);
555            }
556        }
557        // Use the restorer function defined in assembly.
558        unsafe extern "C" { safe fn __devela_linux_restore_rt(); }
559
560        // Start with the SA_RESTORER and SA_SIGINFO flags, as they are always required.
561        let mut combined_flags = SIGACTION::SA_RESTORER | SIGACTION::SA_SIGINFO;
562        // Add additional flags if provided.
563        if let Some(provided_flags) = flags {
564            for &flag in provided_flags {
565                match flag {
566                    SIGACTION::SA_NOCLDSTOP
567                    | SIGACTION::SA_NOCLDWAIT
568                    | SIGACTION::SA_NODEFER
569                    | SIGACTION::SA_ONSTACK
570                    | SIGACTION::SA_RESETHAND
571                    | SIGACTION::SA_RESTART => {
572                        combined_flags |= flag;
573                    }
574                    _ => {
575                        #[cfg(feature = "std")]
576                        eprintln!("Warning: Ignoring unknown flag: {flag:#x}");
577                    }
578                }
579            }
580        }
581        let mask = LinuxSigset::empty();
582        let sigaction = LinuxSigaction::new_siginfo(
583            c_handler_siginfo,
584            combined_flags,
585            mask,
586            Some(__devela_linux_restore_rt),
587        );
588        for s in signals {
589            if (1..=31).contains(s) { // make sure the signal is a valid number
590                unsafe {
591                    let _ = Linux::sys_rt_sigaction(*s, &sigaction,
592                        Ptr::null_mut(), LinuxSigset::size());
593                }
594            }
595        }
596    }
597}

599/// Generates a `random_*` function for each given integer primitive
600#[cfg(all(feature = "unsafe_syscall", not(miri)))]
601macro_rules! impl_random_fns {
602    // $prim: the unsigned integer primitive
603    // $len:  the length of the primitive in bytes
604    ($($prim:ident : $len:literal),+) => { $crate::paste! { $(
605        #[doc = "Generates a random `" $prim "` value that may not be criptographically secure."]
606        pub fn [<random_ $prim>]() -> Result<$prim> {
607            #[cold] fn getrandom_failed_cold(n: isize) -> Result<$prim> { Err(LinuxError::Sys(n)) }
608
609            let (mut r, mut attempts) = ([0; $len], 0);
610            loop {
611                let n = unsafe { Linux::sys_getrandom(r.as_mut_ptr(), $len, Linux::RAND_FLAGS) };
612                if n == $len { break; } // ← hot path
613                else if n == -ERRNO::EAGAIN { // ←↓ cold paths
614                    is![!Linux::getrandom_try_again_cold(&mut attempts); break];
615                } else { return getrandom_failed_cold(n); } // n < 0
616            }
617            Ok($prim::from_ne_bytes(r))
618        }
619    )+ }};
620}

622/// # Randomness-related methods.
623///
624/// They make use of the `GRND_NONBLOCK` and `GRND_INSECURE` flags. So when the randomness
625/// source is not ready, instead of blocking it may return less secure data in linux >= 5.6
626/// or retry it a certain number of times, or even return 0 in some cases.
627#[rustfmt::skip]
628#[cfg(all(feature = "unsafe_syscall", not(miri)))]
629impl Linux {
630    impl_random_fns![u8:1, u16:2, u32:4, u64:8, u128:16];
631
632    /// Fills the given `buffer` with random bytes that may not be cryptographically secure.
633    ///
634    /// # Panics
635    /// Panics in debug if `buffer.len() > `[`isize::MAX`]
636    pub fn random_bytes(buffer: &mut [u8]) -> Result<()> {
637        #[cold] fn getrandom_failed_cold(n: isize) -> Result<()> { Err(LinuxError::Sys(n)) }
638
639        debug_assert![buffer.len() <= isize::MAX as usize];
640        let (mut attempts, mut offset) = (0, 0);
641        while offset < buffer.len() {
642            let n = unsafe { Linux::sys_getrandom(buffer[offset..].as_mut_ptr(),
643                buffer.len() - offset, Linux::RAND_FLAGS) };
644            if n == -ERRNO::EAGAIN { is![!Linux::getrandom_try_again_cold(&mut attempts); break]; }
645            else if n < 0 { return getrandom_failed_cold(n); } // ←↑ cold paths
646            else { offset += n as usize; } // ← hot path
647        }
648        Ok(())
649    }
650
651    // from `sys/random.h`
652    const GRND_NONBLOCK: c_uint = 0x0001;
653    // const GRND_RANDOM: isize = 0x0002;
654    const GRND_INSECURE: c_uint = 0x0004;
655    const RAND_FLAGS: c_uint = Linux::GRND_NONBLOCK | Linux::GRND_INSECURE;
656    const RAND_MAX_ATTEMPTS: usize = 15;
657    /// the cold path for trying again
658    #[cold] #[must_use]
659    fn getrandom_try_again_cold(attempts: &mut usize) -> bool {
660        // if *attempts >= Linux::RAND_MAX_ATTEMPTS { Linux::getrandom_failed_cold(); }
661        *attempts += 1;
662        *attempts <= Linux::RAND_MAX_ATTEMPTS
663    }
664}
