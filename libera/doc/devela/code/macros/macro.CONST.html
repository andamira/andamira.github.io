Macro CONST

Source 
macro_rules! CONST {
    ( // function
    $(
        $(#[$attrs:meta])*
        $vis:vis $NAME_FN:ident =
        $(#[$fn_attrs:meta])*
        $fn_vis:vis $($mod1:ident)? fn $fn:ident($($param:ident: $param_ty:ty),* $(,)?)
        $(-> $ret_ty:ty)? $body:block
    );* $(;)?) => { ... };
    ( // expression
    $(
        $(#[$attrs:meta])*
        $vis:vis $NAME_EXPR:ident = $expr:expr
     );* $(;)?) => { ... };
}
Expand description

A helper for constructing macro constants.

§Examples

CONST!{ /* Supports empty declarations */ }

// Supports any expresion
CONST!{EXPR = 2 * 15 / 3}
assert_eq![EXPR![], 10u8];
assert_eq![EXPR![], 10i8];

CONST!{
    /// Supports docs, attributes, and visibility.
    #[macro_export] // attribute needed if public
    pub ARRAY = [1, 2, 3]
}
assert_eq![ARRAY![], [1u16, 2, 3]];
assert_eq![ARRAY![], [1i32, 2, 3]];

CONST!{
    /// Supports functions (only 1 fn qualifier [async, const, extern, unsafe]
    FN_1 =
    /// Returns `true` if a != 0.
    #[inline] #[must_use]
    pub const fn fn_1(a: i32) -> bool { a != 0 };

    /// You can repeat functions.
    FN_2 = #[inline] #[must_use] pub const fn fn_2(a: i32) { };

    // FN_3 = pub fn fn_3() {}; // NOTE FIXME: ambiguity... maybe separate CONST_FN macro?
}
pub struct FNS;
impl FNS {
    FN_1!{}
    FN_2!{}
    // FN_3!{}
}

// Supports multiple definitions, ended by ;
CONST! {
    DOC_1 = "Doc 1." ;
    pub(crate) DOC_2 = "Doc 2.";

    // NOTE: You can't mix expressions and functions in the same invocation.
    // FN_4 = fn fn_4() {}; // compile fails
}
assert_eq![DOC_1![], "Doc 1."];
assert_eq![DOC_2![], "Doc 2."];
//impl FNS { FN_2!{} }

// A good use-case is for repeated documentation
/// Function 1, version a.
#[doc = DOC_1!()]
pub fn version_1a() {}
/// Function 1, version b.
#[doc = DOC_1!()]
pub fn version_1b() {}
