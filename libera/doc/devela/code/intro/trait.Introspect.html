Minimal introspection interface.

enum MyKind { A, B, C }

struct MyTypeA;
struct MyTypeB;

impl Introspect for MyTypeA {
    type Kind = MyKind;
    fn intro_kind(&self) -> Self::Kind { MyKind::A }
}

impl Introspect for MyTypeB {
    type Kind = MyKind;
    fn intro_kind(&self) -> Self::Kind { MyKind::B }
}
