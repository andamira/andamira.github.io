10/// Helper for Web API doc links.
11#[rustfmt::skip]
12macro_rules! _js_doc {
13    ( // Link to an API Object
14        $API:literal) => { concat!["([", $API,
15        "](https://developer.mozilla.org/en-US/docs/Web/API/", $API, "))"] };
16    ( // Link to a method of an API Object
17        $API:literal, $method:literal) => { concat![ "([", $method,
18        "](https://developer.mozilla.org/en-US/docs/Web/API/", $API, "/", $method, "))"] };
19
20    /* special cases */
21
22    (canvas $method:literal) => { concat!["([", $method,
23        "](https://developer.mozilla.org/en-US/docs/Web/API/CanvasRenderingContext2D/",
24        $method, "))"] };
25    (console $method:literal) => { concat!["([", $method,
26        "](https://developer.mozilla.org/en-US/docs/Web/API/console/",
27        $method, "_static))"] };
28}
29pub(crate) use _js_doc;

31/// Helps re-exporting javascript functions.
32///
33/// # Example
34/// ```ignore
35/// # use devela::_js_extern;
36/// _js_extern! {
37///     [ module: "env" ]
38///     pub safe fn same_fn_name(x: f64, y: f64, w: f64, h: f64);
39///     pub(crate) safe fn "js_fn_name" rust_fn_name(x: f64, y: f64, w: f64, h: f64);
40///     unsafe fn "js_fn" rs_fn(ptr: *const u8, len: usize, x: f64, y: f64);
41/// }
42/// // The previous code generates:
43///
44/// // #[link(wasm_import_module = "env")]
45/// // unsafe extern "C" {
46/// //     pub safe fn same_fn_name(x: f64, y: f64, w: f64, h: f64);
47/// //
48/// //     #[link_name = "js_fn_name"]
49/// //     pub(cate) safe fn rust_fn_name(x: f64, y: f64, w: f64, h: f64);
50/// //
51/// //     #[link_name = "js_fn"]
52/// //     unsafe fn rs_fn(ptr: *const u8, len: usize, x: f64, y: f64);
53/// // }
54/// ```
55///
56/// Use *safe* fn when:
57/// - The function does not perform pointer dereferencing or other memory-unsafe operations.
58/// - It always behaves safely (e.g., a function that just draws to the Canvas API).
59///
60/// Use *unsafe* fn if:
61/// - The function can mutate raw memory (e.g., passing buffers, pointers).
62/// - It performs DOM manipulations that might trigger undefined behavior.
63/// - It can throw exceptions that Rust cannot catch.
64#[rustfmt::skip]
65#[cfg(all(feature = "unsafe_ffi", not(windows)))]
66macro_rules! _js_extern {
67    (
68        // # Args
69        // [            header section
70        // $module:     optional js module name (defaults to "env")
71        // ]
72        // $fn_attrs:   optional attributes and doc comments
73        // $vis:        visibility of the extern function (defaults to private)
74        // safe|unsafe  optional safety specifier (defaults to unsafe)
75        // $js_fn:      optional link_name (different javascript function name)
76        // $fn:         imported rust function name (default same js name)
77
78        [ $(module: $module:literal)? $(,)? ]
79        $(
80            $(#[$fn_attrs:meta])*
81            $vis:vis $(safe$($_s:block)?)? $(unsafe$($_u:block)?)?
82            fn
83            $($js_fn:literal)?
84            $fn:ident
85            ($($param:ident: $param_ty:ty),* $(,)?) $(-> $fn_return:ty)?;
86        )*
87    ) => {
88        $( #[link(wasm_import_module = $module)] )?
89        #[cfg(not(feature = "safe_lang"))]
90        unsafe extern "C" { $(
91            $(#[$fn_attrs])*
92            $( #[link_name = $js_fn] )?
93            $vis $(safe$($_s)?)? $(unsafe$($_u)?)?
94            fn $fn($($param: $param_ty),*) $(-> $fn_return)?;
95        )* }
96    };
97}
98/// Dummy safe fallback of the real `_js_extern!` macro.
99#[rustfmt::skip]
100#[cfg(any(not(feature = "unsafe_ffi"), windows))]
101macro_rules! _js_extern { ($($tt:tt)*) => {}; }

103#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_ffi")))]
104pub(crate) use _js_extern;

106/// Helper macro to generate JS methods that return strings.
107///
108/// Generates two versions of each method:
109/// 1. A buffer version (default name) that writes into the given buffer and retuns `&str`.
110/// 2. An allocating version (with `_alloc` suffix) that returns a `String`.
111///
112/// # Example:
113/// The following code
114/// ```ignore
115/// _js_method_buf_string! {
116///     #[doc = _js_doc!("Window", "name")]
117///     /// Returns the window name.
118///     name, window_name
119/// }
120/// ```
121/// would produce:
122/// ```ignore
123/// #[doc = _js_doc!("Window", "name")]
124/// /// Returns the window name.
125/// pub fn name(buffer: &mut [u8]) -> &str {
126///     Js::read_str(buffer, |ptr, len| unsafe { window_name(ptr, len) }) }
127///
128/// #[doc = _js_doc!("Window", "name")]
129/// /// Returns the window name.
130/// #[cfg(feature = "alloc")]
131/// #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
132/// pub fn name_alloc() -> String {
133///     Js::read_string(|ptr, len| unsafe { window_name(ptr, len) })
134/// }
135/// ```
136#[rustfmt::skip]
137macro_rules! _js_method_str_alloc {
138    (
139    $(#[$fn_attrs:meta])*
140    $method_name:ident,
141    $call_fn:ident $(,)?
142    ) => { $crate::paste! {
143        /* default string slice version */
144
145        $(#[$fn_attrs])*
146        pub fn $method_name(buffer: &mut [u8]) -> &str {
147            $crate::Js::read_str(buffer, |ptr, len| unsafe { $call_fn(ptr, len) })
148        }
149
150        /* allocating String version */
151
152        $(#[$fn_attrs])*
153        #[cfg(feature = "alloc")]
154        #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
155        pub fn [<$method_name _alloc>]() -> String {
156            $crate::Js::read_string(|ptr, len| unsafe { $call_fn(ptr, len) })
157        }
158    }};
159}
160pub(crate) use _js_method_str_alloc;
