16/// A handle to a JavaScript Web Worker.
17///
18/// - <https://developer.mozilla.org/en-US/docs/Web/API/Worker>.
19#[repr(transparent)]
20#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
21pub struct WebWorker {
22    pub(in crate::lang::ffi::js) id: js_uint32,
23}
24
25#[rustfmt::skip]
26impl WebWorker {
27    /// Returns a new invalid worker.
28    pub const fn invalid() -> Self { WebWorker { id: 0 } }
29    /// Returns the worker's ID.
30    pub const fn id(self) -> js_uint32 { self.id }
31}
32
33#[rustfmt::skip]
34#[cfg(not(feature = "safe_lang"))]
35#[cfg(all(feature = "unsafe_ffi", not(windows)))]
36#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_ffi")))]
37#[cfg_attr(nightly_doc, doc(cfg(target_arch = "wasm32")))]
38impl WebWorker {
39    /// Spawns a new JavaScript Web Worker from a script.
40    pub fn spawn(script: &str) -> Result<Self, WebWorkerError> { Web::worker_spawn(script) }
41    /// Stops this Web Worker.
42    pub fn stop(self) { Web::worker_stop(self); }
43    /// Checks if this worker is still active by querying JavaScript.
44    pub fn is_active(self) -> bool { Web::worker_is_active(self) }
45    /// Sends a message to this Web Worker.
46    pub fn send_message(self, message: &str) { Web::worker_send_message(self, message); }
47    /// Executes JavaScript inside this worker, returning a `WebWorkerJob`.
48    pub fn eval(self, js_code: &str) -> Result<WebWorkerJob, WebWorkerError> {
49        if !self.is_active() { return Err(WebWorkerError::WorkerNotFound); }
50        let job = Web::worker_eval(self, js_code);
51        if job.id == 0 { return Err(WebWorkerError::WorkerNotFound); }
52        Ok(job)
53    }
54}

56/// Errors that can occur when working with JavaScript Web Workers.
57#[derive(Clone, Copy, Debug, PartialEq, Eq)]
58pub enum WebWorkerError {
59    /// The worker script provided was invalid.
60    InvalidScript,
61    /// The worker was not found.
62    WorkerNotFound,
63    /// The job was not found.
64    JobNotFound,
65}

67/// Represents a job running inside a [`WebWorker`].
68#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
69pub struct WebWorkerJob {
70    pub(in crate::lang::ffi::js) worker: WebWorker,
71    pub(in crate::lang::ffi::js) id: js_uint32,
72}
73
74#[rustfmt::skip]
75impl WebWorkerJob {
76    /// Returns the associated worker.
77    pub const fn worker(self) -> WebWorker { self.worker }
78    /// Returns the job's ID.
79    pub const fn id(self) -> js_uint32 { self.id }
80}

92        Web::worker_poll(self, buffer)

94    /// Polls the result of this job.

96    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]

98        Web::worker_poll_alloc(self)
99    }
100    /// Cancels this job.
101    pub fn cancel(self) { Web::worker_eval_cancel(self); }
