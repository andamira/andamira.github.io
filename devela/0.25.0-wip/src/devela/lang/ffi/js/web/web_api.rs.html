138    /// Callback dispatcher for WebAssembly events without args.
139    ///
140    /// - This function is used by JavaScript to invoke a Rust function pointer.
141    /// - It allows Rust event listeners to execute when triggered by JS.
142    /// - The `callback_ptr` is a function pointer cast from JS,
143    ///   which is then transmuted into a callable Rust function.
144    ///
145    /// # Safety
146    /// `callback_ptr` must be a valid function pointer.
147    ///
148    /// # Example
149    /// ```ignore
150    /// #[unsafe(no_mangle)]
151    /// pub extern "C" fn my_callback() { Web::console_log("Button clicked!"); }
152    /// Web::event_add_listener("#my_button", WebEventKind::Click, my_callback);
153    /// ```
154    #[unsafe(no_mangle)]
155    #[cfg_attr(not(feature = "__force_test_no_mangle"), cfg(not(test)))]
156    pub unsafe extern "C" fn wasm_callback(callback_ptr: usize) {
157        let callback = callback_ptr as *const ();
158        let callback: extern "C" fn() = unsafe { transmute(callback) };
159        callback();
160    }
161    /// WebAssembly mouse event callback dispatcher.
162    ///
163    /// - Called from JavaScript when a mouse event is fired.
164    /// - Passes the `WebEventMouse` struct to the Rust callback.
165    ///
166    /// # Safety
167    /// - `callback_ptr` must be a valid function pointer.
168    #[unsafe(no_mangle)]
169    #[cfg_attr(not(feature = "__force_test_no_mangle"), cfg(not(test)))]
170    pub unsafe extern "C" fn wasm_callback_mouse(callback_ptr: usize, button: js_int32,
171        buttons: js_int32, x: js_number, y: js_number, etype: js_int32, time_stamp: js_number) {
172        let callback = callback_ptr as *const ();
173        let callback: extern "C" fn(WebEventMouse) = unsafe { transmute(callback) };
174        let etype = WebEventKind::from_repr(etype as u8);
175        let time_stamp = JsInstant::from_millis_f64(time_stamp);
176        callback(WebEventMouse::new(x, y, button as u8, buttons as u8, etype, time_stamp));
177    }
178    /// WebAssembly mouse event callback dispatcher.
179    ///
180    /// - Called from JavaScript when a pointer event is fired.
181    /// - Passes the `WebEventPointer` struct to the Rust callback.
182    ///
183    /// # Safety
184    /// - `callback_ptr` must be a valid function pointer.
185    // WAIT: In firefox + linux + X11 needs the env variable `MOZ_USE_XINPUT2=1`
186    // https://bugzilla.mozilla.org/show_bug.cgi?id=1606832 (closed)
187    // https://bugzilla.mozilla.org/show_bug.cgi?id=1207700
188    // https://bugzilla.mozilla.org/show_bug.cgi?id=1822714
189    #[unsafe(no_mangle)]
190    #[cfg_attr(not(feature = "__force_test_no_mangle"), cfg(not(test)))]
191    pub unsafe extern "C" fn wasm_callback_pointer(callback_ptr: usize, id: js_int32,
192        x: js_number, y: js_number, pressure: js_number, tilt_x: js_int32, tilt_y: js_int32,
193        twist: js_int32, etype: js_int32, time_stamp: js_number) {
194        let callback = callback_ptr as *const ();
195        let callback: extern "C" fn(WebEventPointer) = unsafe { transmute(callback) };
196        let etype = WebEventKind::from_repr(etype as u8);
197        let time_stamp = JsInstant::from_millis_f64(time_stamp);
198        callback(WebEventPointer::new(x, y, pressure, id, tilt_x as i8, tilt_y as i8, twist as u16,
199            etype, time_stamp));
200    }

456/// Web API workers
457#[rustfmt::skip]
458#[cfg(not(feature = "safe_lang"))]
459#[cfg(all(feature = "unsafe_ffi", not(windows)))]
460#[cfg_attr(nightly_doc, doc(cfg(all(feature = "unsafe_ffi", target_arch = "wasm32"))))]
461impl Web {
462    /// Spawns a Web Worker and returns its ID.
463    pub fn worker_spawn(script: &str) -> Result<WebWorker, WebWorkerError> {
464        let id = unsafe { worker_spawn(script.as_ptr(), script.len()) };
465        if id == 0 { Err(WebWorkerError::InvalidScript) } else { Ok(WebWorker { id }) }
466    }
467    /// Checks if this worker is still active by querying JavaScript.
468    pub fn worker_is_active(worker: WebWorker) -> js_bool { worker_is_active(worker.id()) }
469    /// Stops a specific Web Worker by ID.
470    pub fn worker_stop(worker: WebWorker) { worker_stop(worker.id()); }
471    /// Stops all Web Workers.
472    pub fn worker_stop_all() { worker_stop_all(); }
473    /// Returns the number of active workers.
474    pub fn worker_list_len() -> usize { worker_list_len() as usize }
475    /// Returns the list of active worker IDs.
476    #[cfg(feature = "alloc")]
477    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
478    pub fn worker_list() -> Vec<WebWorker> {
479        let len = worker_list_len() as usize;
480        let mut workers = vec![WebWorker::default(); len];
481        let count = Web::worker_list_buf(&mut workers);
482        workers.truncate(count);
483        workers
484    }
485    /// Writes active worker handles into a buffer and returns the number written.
486    pub fn worker_list_buf(buffer: &mut [WebWorker]) -> usize {
487        let len = worker_list_len() as usize;
488        let count = len.min(buffer.len());
489        unsafe { worker_list(buffer.as_mut_ptr() as *mut js_uint32, count as js_uint32); }
490        count
491    }
492    /// Sends a message to a specific Web Worker.
493    pub fn worker_send_message(worker: WebWorker, msg: &str) {
494        unsafe { worker_send_message(worker.id(), msg.as_ptr(), msg.len()); }
495    }
496    /// Requests execution of JavaScript inside a worker.
497    pub fn worker_eval(worker: WebWorker, js_code: &str) -> WebWorkerJob {
498        let id = unsafe { worker_eval(worker.id(), js_code.as_ptr(), js_code.len()) };
499        WebWorkerJob { worker, id }
500    }
501    /// Polls for the result of a JavaScript execution in a worker, and writes it into `buffer`.
502    ///
503    /// If ready, returns the number of bytes written to the buffer.
504    pub fn worker_poll(job: WebWorkerJob, buffer: &mut [u8])
505        -> TaskPoll<Result<usize, WebWorkerError>> {
506        if !job.worker().is_active() { return TaskPoll::Ready(Err(WebWorkerError::WorkerNotFound)); }
507        let written = unsafe { worker_poll(job.id(), buffer.as_mut_ptr(), buffer.len()) };
508        match written {
509            0 => TaskPoll::Pending,
510            js_uint32::MAX => TaskPoll::Ready(Err(WebWorkerError::JobNotFound)),
511            _ => TaskPoll::Ready(Ok(written as usize)),
512        }
513    }
514    /// Polls for the result of a JavaScript execution in a worker.
515    #[cfg(feature = "alloc")]
516    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
517    pub fn worker_poll_alloc(job: WebWorkerJob) -> TaskPoll<Result<String, WebWorkerError>> {
518        if !job.worker().is_active() { return TaskPoll::Ready(Err(WebWorkerError::WorkerNotFound)); }
519        let mut first_check = true;
520        let result = Js::read_string_capped(128, false, |ptr, cap| {
521            let res = unsafe { worker_poll(job.id(), ptr, cap as usize) as js_int32 };
522            if first_check {
523                first_check = false; // Intercept status codes before bothering with decoding
524                if res == 0 { return 0; } else if res == -1 { return -1; } // pending or not found
525            }
526            res
527        });
528        match result.as_str() {
529            "" => TaskPoll::Pending, // Covers 0 and -1 (mapped above)
530            _ => TaskPoll::Ready(Ok(result)),
531        }
532    }
533    /// Cancels an ongoing JavaScript evaluation.
534    pub fn worker_eval_cancel(job: WebWorkerJob) { worker_cancel_eval(job.id()); }
