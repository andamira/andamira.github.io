16/// Handle to the brower's global [Document] associated APIs.
17///
18/// [Document]: https://developer.mozilla.org/en-US/docs/Web/API/Document
19#[repr(C)]
20#[derive(Copy, Clone, Debug)]
21pub struct WebDocument;
22
23#[rustfmt::skip]
24#[cfg(not(feature = "safe_lang"))]
25#[cfg(all(feature = "unsafe_ffi", not(windows)))]
26#[cfg_attr(nightly_doc, doc(cfg(all(feature = "unsafe_ffi", target_arch = "wasm32"))))]
27impl WebDocument {
28    /// Creates an element (delegates to `WebElement::new`).
29    pub fn create_element(local_name: &str) -> Result<WebElement, JsError> {
30        let handle = unsafe { document_create_element(local_name.as_ptr(), local_name.len()) };
31        if handle == 0 { Err(JsError::Exception) }
32        else { Ok(WebElement { handle: unsafe { JsObject::from_handle(handle) } }) }
33    }
34
35    /* flags */
36
37    #[doc = _js_doc!("Document", "compatMode")]
38    /// Returns `true` if the browser is in no-quirks mode, ("CSS1Compat") or `false` otherwise.
39    ///
40    /// - <https://developer.mozilla.org/en-US/docs/Web/HTML/Guides/Quirks_mode_and_standards_mode>
41    pub fn is_compat_mode() -> js_bool { document_is_compat_mode() }
42
43    #[doc = _js_doc!("Document", "hidden")]
44    /// Whether the current document is hidden.
45    pub fn is_hidden() -> js_bool { document_is_hidden() }
46

50        #[doc = _js_doc!("Document", "contentType")]
51        /// Returns the document's content type.
52        content_type, document_content_type
53    }
