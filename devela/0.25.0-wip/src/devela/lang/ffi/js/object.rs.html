10/// Handle to a JavaScript [Object].
11///
12/// [Object]: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Object
13#[repr(transparent)]
14#[derive(Debug, PartialEq, Eq, Hash)]
15pub struct JsObject {
16    pub(in crate::lang::ffi::js) handle: js_uint32,
17}
18#[cfg(not(feature = "safe_lang"))]
19#[cfg(feature = "unsafe_ffi")]
20#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_ffi")))]
21impl Drop for JsObject {
22    fn drop(&mut self) {
23        if self.handle != 0 {
24            object_release(self.handle);
25        }
26    }
27}
28#[cfg(not(feature = "safe_lang"))]
29#[cfg(feature = "unsafe_ffi")]
30#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_ffi")))]
31impl Clone for JsObject {
32    fn clone(&self) -> Self {
33        if self.handle != 0 {
34            object_add_ref(self.handle);
35        }
36        JsObject { handle: self.handle }
37    }
38}

90    /// Gets a property of the JS object.
91    pub fn get_property<'buf>(&self, property: &str, buffer: &'buf mut [u8]) -> Result<JsValue<'buf>, JsError> {
92        if self.handle == 0 { return Err(JsError::InvalidHandle); }
93        let result = unsafe {
94            object_get_property(
95                self.handle,
96                property.as_ptr(),
97                property.len(),
98                buffer.as_mut_ptr(),
99                buffer.len()
100            )
101        };
102        match result {
103            n if n > 0 => JsValue::from_buffer(buffer, n as usize),
104            0 => Ok(JsValue::Null),
105            -1 => Err(JsError::InvalidHandle),
106            -2 => Err(JsError::PropertyNotFound),
107            -3 => Err(JsError::InvalidTypeTag),
108            n if n < 0 => Err(JsError::BufferTooSmall((-n) as usize)),
109            _ => unreachable!(),
110        }
111    }
