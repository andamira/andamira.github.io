17/// A JavaScript value for FFI communication.
18#[non_exhaustive]
19#[derive(Debug)]
20pub enum JsValue<'a> {
21    /// Explicit `null`.
22    Null,
23    /// Explicit `undefined`.
24    Undefined,
25
26    /// Boolean (`true`/`false`).
27    Boolean(js_bool),
28
29    /// `number` (always 64-bit float)
30    Number(js_number),
31
32    /// String slice
33    Str(&'a str),
34
35    /// Heap-allocated string.
36    #[cfg(feature = "alloc")]
37    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
38    String(String),
39
40    /// Handle to JS object (registry index).
41    Object(js_uint32),
42
43    /// A raw binary data buffer backed by a byte slice.
44    ArrayBuffer(&'a [u8]),
45
46    /// A raw binary data buffer backed by a byte vector.
47    #[cfg(feature = "alloc")]
48    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
49    VecBuffer(Vec<u8>),
50
51    /// An array backed by a slice.
52    Array(&'a [JsValue<'a>]),
53
54    /// An array backed by a vector.
55    #[cfg(feature = "alloc")]
56    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
57    Vec(Vec<JsValue<'a>>),
58}

62    pub const fn unwrap_str(&self) -> &str {
63        // match self { Self::Str(s) => s, _ => panic!["not JsValue::Str"] }
64        // if let Self::Str(s) = self { s } else { panic!["not JsValue::Str"] }
65        is![let Self::Str(s) = self; s; panic!["not JsValue::Str"]]
66    }
67    ///
68    #[cfg(feature = "alloc")]
69    pub fn unwrap_string(self) -> String {
70        is![let Self::String(s) = self; s; panic!["not JsValue::String"]]
71    }

73    /// Returns the number tag associated with the current variant.
74    #[inline(always)]
75    pub const fn type_tag(&self) -> u8 {
76        match self {
77            Self::Null => 0,
78            Self::Undefined => 1,
79            Self::Boolean(_) => 2,
80            Self::Number(_) => 3,
81
82            Self::Str(_) => 4,
83            #[cfg(feature = "alloc")]
84            Self::String(_) => 4,
85
86            Self::Object(_) => 5,
87
88            Self::ArrayBuffer(_) => 6,
89            #[cfg(feature = "alloc")]
90            Self::VecBuffer(_) => 6,
91
92            Self::Array(_) => 7,
93            #[cfg(feature = "alloc")]
94            Self::Vec(_) => 7,
95        }
96    }

98    /// Returns the size of the payload data.
99    // WIP MAYBE RECONSIDER
100    #[inline(always)]
101    pub fn encoded_size(&self) -> usize {
102        match self {
103            Self::Null | Self::Undefined => 1,
104            Self::Boolean(_) => 1 + 1,
105            Self::Number(_) => 1 + 8, // f64
106
107            Self::Str(s) => 1 + 4 + s.len(),
108            #[cfg(feature = "alloc")]
109            Self::String(s) => 1 + 4 + s.len(),
110
111            Self::Object(_) => 1 + 4, // u32 handle
112            //
113            Self::ArrayBuffer(buf) => 1 + 4 + buf.len(),
114            #[cfg(feature = "alloc")]
115            Self::VecBuffer(buf) => 1 + 4 + buf.len(),
116
117            Self::Array(elems) => 1 + 4 + elems.iter().map(|e| e.encoded_size()).sum::<usize>(),
118            #[cfg(feature = "alloc")]
119            Self::Vec(elems) => 1 + 4 + elems.iter().map(|e| e.encoded_size()).sum::<usize>(),
120        }
121    }

123    /// Encodes the value into a byte buffer for WASM FFI.
124    ///
125    /// Returns the total bytes written.
126    ///
127    /// # Layout
128    /// ```text
129    /// [1B: type tag][payload...]
130    /// ```
131    /// # Panics
132    /// If buffer is too small (debug builds only).
133    pub fn encode_into(&self, buf: &mut [u8]) -> usize {
134        let mut offset = 0;
135        buf[offset] = self.type_tag();
136        offset += 1;
137
138        match self {
139            Self::Boolean(b) => {
140                buf[offset] = *b as u8;
141                offset += 1;
142            }
143            Self::Number(n) => {
144                buf[offset..offset + 8].copy_from_slice(&n.to_le_bytes());
145                offset += 8;
146            }
147            Self::Str(s) => {
148                offset += self.encode_str(s, &mut buf[offset..]);
149            }
150            #[cfg(feature = "alloc")]
151            Self::String(s) => {
152                offset += self.encode_str(s.as_str(), &mut buf[offset..]);
153            }
154            Self::Object(handle) => {
155                buf[offset..offset + 4].copy_from_slice(&handle.to_le_bytes());
156                offset += 4;
157            }
158            Self::ArrayBuffer(slice) => {
159                offset += self.encode_buffer(slice, &mut buf[offset..]);
160            }
161            #[cfg(feature = "alloc")]
162            Self::VecBuffer(vec) => {
163                offset += self.encode_buffer(vec, &mut buf[offset..]);
164            }
165            Self::Array(elements) => {
166                let len_pos = offset;
167                offset += 4; // Reserve space for length
168                let mut elem_count = 0;
169                for elem in elements.iter() {
170                    let bytes_written = elem.encode_into(&mut buf[offset..]);
171                    offset += bytes_written;
172                    elem_count += 1;
173                }
174                // Backfill array length
175                buf[len_pos..len_pos + 4].copy_from_slice(&(elem_count as u32).to_le_bytes());
176            }
177            #[cfg(feature = "alloc")]
178            Self::Vec(elements) => {
179                let len_pos = offset;
180                offset += 4;
181
182                let mut elem_count = 0;
183                for elem in elements.iter() {
184                    let bytes_written = elem.encode_into(&mut buf[offset..]);
185                    offset += bytes_written;
186                    elem_count += 1;
187                }
188
189                buf[len_pos..len_pos + 4].copy_from_slice(&(elem_count as u32).to_le_bytes());
190            }
191            Self::Null | Self::Undefined => (),
192        }
193        offset
194    }
195
196    // Layout: [4B length][N bytes data]
197    #[inline(always)]
198    fn encode_buffer(&self, data: &[u8], buf: &mut [u8]) -> usize {
199        let len = data.len();
200        buf[..4].copy_from_slice(&(len as u32).to_le_bytes());
201        buf[4..4 + len].copy_from_slice(data);
202        4 + len // total bytes written
203    }
204
205    #[inline(always)]
206    fn encode_str(&self, s: &str, buf: &mut [u8]) -> usize {
207        let (bytes, len) = (s.as_bytes(), s.len());
208        buf[..4].copy_from_slice(&(len as u32).to_le_bytes()); // encode length prefix
209        buf[4..4 + len].copy_from_slice(bytes); // encode string byte content
210        4 + len // total bytes written
211    }

237    /// WIP
238    pub fn from_buffer(buffer: &'buf [u8], len: usize) -> Result<Self, JsError> {
239        if len == 0 || buffer.is_empty() {
240            return Err(JsError::EmptyBuffer);
241        }
242
243        match buffer[0] {
244            0 => Ok(JsValue::Null),
245            1 => Ok(JsValue::Undefined),
246            2 => Ok(JsValue::Boolean(buffer.get(1).is_some_and(|&b| b != 0))),
247            3 => Ok(JsValue::Number(f64::from_le_bytes(
248                buffer.get(1..9).ok_or(JsError::InvalidBuffer)?.try_into().unwrap(),
249            ))),
250            4 => {
251                // String
252                let str_bytes = &buffer[1..len];
253                let str_len = str_bytes.iter().position(|&b| b == 0).unwrap_or(str_bytes.len());
254                Ok(JsValue::Str(
255                    Str::from_utf8(&str_bytes[..str_len]).map_err(|_| JsError::StringDecoding)?,
256                ))
257            }
258            // Object
259            5 => Ok(JsValue::Object(u32::from_le_bytes(
260                buffer.get(1..5).ok_or(JsError::InvalidBuffer)?.try_into().unwrap(),
261            ))),
262            _ => Err(JsError::InvalidTypeTag),
263        }
264    }
