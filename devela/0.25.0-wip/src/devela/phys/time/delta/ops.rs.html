14/// # Operations.
15impl TimeDelta {
16    /// Add two time deltas together. If overflow occurs, then `None` is returned.
17    pub const fn checked_add(self, rhs: TimeDelta) -> Option<TimeDelta> {
18        let Some(mut secs) = self.secs.checked_add(rhs.secs) else {
19            return None;
20        };
21        // OK because `-999_999_999 <= nanos <= 999_999_999`, and so adding
22        // them together will never overflow an i32.
23        let mut nanos = self.nanos + rhs.nanos;
24        // The below is effectively TimeDelta::new, but with checked
25        // arithmetic. My suspicion is that there is probably a better way
26        // to do this. The main complexity here is that 1) `|nanos|` might
27        // now exceed 1 second and 2) the signs of `secs` and `nanos` might
28        // not be the same. The other difference from TimeDelta::new is
29        // that we know that `-1_999_999_998 <= nanos <= 1_999_999_998` since
30        // `|TimeDelta::nanos|` is guaranteed to be less than 1 second. So
31        // we can skip the div and modulus operations.
32
33        // When |nanos| exceeds 1 second, we balance the excess up to seconds.
34        if nanos != 0 {
35            if nanos >= NANOS_PER_SEC {
36                nanos -= NANOS_PER_SEC;
37                secs = match secs.checked_add(1) {
38                    None => return None,
39                    Some(secs) => secs,
40                };
41            } else if nanos <= -NANOS_PER_SEC {
42                nanos += NANOS_PER_SEC;
43                secs = match secs.checked_sub(1) {
44                    None => return None,
45                    Some(secs) => secs,
46                };
47            }
48            if secs != 0 && nanos != 0 && secs.signum() != (nanos.signum() as i64) {
49                if secs < 0 {
50                    debug_assert!(nanos > 0);
51                    // OK because secs<0.
52                    secs += 1;
53                    // OK because nanos>0.
54                    nanos -= NANOS_PER_SEC;
55                } else {
56                    debug_assert!(secs > 0);
57                    debug_assert!(nanos < 0);
58                    // OK because secs>0.
59                    secs -= 1;
60                    // OK because nanos<0.
61                    nanos += NANOS_PER_SEC;
62                }
63            }
64        }
65        Some(TimeDelta::new_unchecked(secs, nanos))
66    }
67
68    /// Add two time deltas together. If overflow occurs, then arithmetic saturates.
69    pub const fn saturating_add(self, rhs: TimeDelta) -> TimeDelta {
70        let Some(sum) = self.checked_add(rhs) else {
71            return if rhs.is_negative() { TimeDelta::MIN } else { TimeDelta::MAX };
72        };
73        sum
74    }
75
76    /// Subtract one time delta from another. If overflow occurs, then `None` is returned.
77    pub const fn checked_sub(self, rhs: TimeDelta) -> Option<TimeDelta> {
78        let Some(rhs) = rhs.checked_neg() else {
79            return None;
80        };
81        self.checked_add(rhs)
82    }
83
84    /// Add two time deltas together. If overflow occurs, then arithmetic saturates.
85    pub const fn saturating_sub(self, rhs: TimeDelta) -> TimeDelta {
86        let Some(diff) = self.checked_sub(rhs) else {
87            return if rhs.is_positive() { TimeDelta::MIN } else { TimeDelta::MAX };
88        };
89        diff
90    }
91}

93/// # Operations involving integers.
94impl TimeDelta {
95    /// Multiply this time delta by an integer.
96    /// If the multiplication overflows, then `None` is returned.
97    pub const fn checked_mul(self, rhs: i32) -> Option<TimeDelta> {
98        let rhs = rhs as i64;
99        // Multiplying any two i32 values never overflows an i64.
100        let nanos = (self.nanos as i64) * rhs;
101        // OK since NANOS_PER_SEC!={-1,0}.
102        let addsecs = nanos / (NANOS_PER_SEC as i64);
103        // OK since NANOS_PER_SEC!={-1,0}.
104        let nanos = (nanos % (NANOS_PER_SEC as i64)) as i32;
105        let Some(secs) = self.secs.checked_mul(rhs) else {
106            return None;
107        };
108        let Some(secs) = secs.checked_add(addsecs) else {
109            return None;
110        };
111        Some(TimeDelta::new_unchecked(secs, nanos))
112    }
113
114    /// Multiply this time delta by an integer.
115    ///
116    /// If the multiplication overflows, then the result saturates to either
117    /// the minimum or maximum duration depending on the sign of the product.
118    pub const fn saturating_mul(self, rhs: i32) -> TimeDelta {
119        let Some(product) = self.checked_mul(rhs) else {
120            let sign = (self.signum() as i64) * (rhs as i64).signum();
121            return if sign.is_negative() { TimeDelta::MIN } else { TimeDelta::MAX };
122        };
123        product
124    }
125
126    /// Divide this duration by an integer. If the division overflows, then `None` is returned.
127    pub const fn checked_div(self, rhs: i32) -> Option<TimeDelta> {
128        if rhs == 0 || (self.secs == i64::MIN && rhs == -1) {
129            return None;
130        }
131        // OK since rhs!={-1,0}.
132        let secs = self.secs / (rhs as i64);
133        // OK since rhs!={-1,0}.
134        let addsecs = self.secs % (rhs as i64);
135        // OK since rhs!=0 and self.nanos>i32::MIN.
136        let mut nanos = self.nanos / rhs;
137        // OK since rhs!=0 and self.nanos>i32::MIN.
138        let addnanos = self.nanos % rhs;
139        let leftover_nanos = (addsecs * (NANOS_PER_SEC as i64)) + (addnanos as i64);
140        nanos += (leftover_nanos / (rhs as i64)) as i32;
141        debug_assert!(nanos < NANOS_PER_SEC);
142        Some(TimeDelta::new_unchecked(secs, nanos))
143    }
144}

301    /// Compile-time friendly version of `try_from_millis_f64`.
302    //
303    // TODO: RETHINK
304    // # Features
305    // - Uses `_float_f64` if enabled, leveraging [`Float`]'s `const_round`, `div_euclid`,
306    //   and `rem_euclid` for precise, bias-free conversion.
307    // - Without `_float_f64`, rounds manually using integer arithmetic, preventing
308    //   systematic underestimation but lacking exact Euclidean division for negatives.
309    #[rustfmt::skip]
310    pub const fn const_try_from_millis_f64(millis: f64) -> Result<TimeDelta, &'static str> {
311        if !millis.is_finite() {
312            return Err("could not convert non-finite milliseconds {millis} to time delta"); }
313        if millis < (i64::MIN as f64) {
314            return Err("floating point milliseconds {millis} overflows TimeDelta::MIN"); }

316            return Err("floating point milliseconds {millis} overflows TimeDelta::MAX"); }
317        let (millis_rounded, secs, nanos);
318        millis_rounded = Float(millis).const_round().0 as i64;
319        secs = Float(millis_rounded as f64).div_euclid(MILLIS_PER_SEC as f64).0 as i64;
320        nanos = Float(millis_rounded as f64).rem_euclid(MILLIS_PER_SEC as f64).0 as i64
321            * NANOS_PER_MILLI as i64;
322        // #[cfg(not(feature = "_float_f64"))]
323        // {
324        //     // millis_rounded = millis as i64; // slight systematic underestimation
325        //     millis_rounded = if millis >= 0.0 { (millis + 0.5) as i64 }  // Round normally
326        //     else { (millis - 0.5) as i64 };  // Round away from zero for negatives
327        //     secs = millis_rounded / MILLIS_PER_SEC;
328        //     nanos = (millis_rounded % MILLIS_PER_SEC) * NANOS_PER_MILLI as i64;
329        // }
330        Ok(TimeDelta::new_unchecked(secs, nanos as i32))

365    /// Divides this time delta by another time delta.
366    pub const fn div_delta_f64(self, rhs: TimeDelta) -> f64 {
367        let lhs_nanos = (self.secs as f64) * (NANOS_PER_SEC as f64) + (self.nanos as f64);
368        let rhs_nanos = (rhs.secs as f64) * (NANOS_PER_SEC as f64) + (rhs.nanos as f64);
369        lhs_nanos / rhs_nanos
370    }
