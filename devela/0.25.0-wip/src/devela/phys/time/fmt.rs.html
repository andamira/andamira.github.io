21/// Timecode splitting and formatting.
22///
23/// # Examples
24/// ```
25/// # use devela::Timecode;
26/// assert_eq!(Timecode::secs_f64(3661.5), "01:01:01.500");
27/// assert_eq!(Timecode::nanos_u64(1_002_003_004), "001s 002ms 003Âµs 004ns");
28/// ```
29#[derive(Debug)]
30pub struct Timecode;

35    /// Decomposes a number of `seconds` in `{ h, m, s, ms }`.
36    ///
37    /// The maximum decomposition for [`u64::MAX`] is
38    /// `{ h: 5_124_095_576_030_431, .. }` (more than 584_942_417 millenia).
39    // -> 64 bits
40    #[must_use]
41    pub fn split_secs_f64(seconds: f64) -> TimeSplitHourNano<u32, u8, u8, u16, NoTime, NoTime> {
42        let ms = (seconds.fract() * 1000.) as u16;
43        let mut ts = seconds.trunc() as u64;
44        let h = (ts / 3600) as u32;
45        ts %= 3600;
46        let m = (ts / 60) as u8;
47        let s = (ts % 60) as u8;
48        TimeSplit::new_hour_nano(h, m, s, ms, (), ())
49    }
50
51    /// Splits a number of `nanoseconds` in `{ s, ms, Âµs, ns }`.
52    ///
53    /// The maximum decomposition for [`u64::MAX`] is
54    /// `{ s: 1_266_874_889, .. }` (more than 40 years).
55    // -> 80 bits
56    #[must_use]
57    pub const fn split_nanos_u64(
58        nanos: u64,
59    ) -> TimeSplitHourNano<NoTime, NoTime, u32, u16, u16, u16> {
60        let (us_tmp, ns) = (nanos / 1000, (nanos % 1000) as u16);
61        let (ms_tmp, us) = (us_tmp / 1000, (us_tmp % 1000) as u16);
62        let (s, ms) = ((ms_tmp / 1000) as u32, (ms_tmp % 1000) as u16);
63        TimeSplit::new_hour_nano((), (), s, ms, us, ns)
64    }
65
66    /// Splits a number of `nanoseconds` in `{ s, ms, Âµs, ns }`.
67    ///
68    /// The maximum decomposition for [`u32::MAX`] is
69    /// `{ ns: 295, Âµs: 967, ms: 294, s: 4 }` (more than 4 seconds).
70    // -> 56 bits
71    #[must_use]
72    pub const fn split_nanos_u32(
73        nanos: u32,
74    ) -> TimeSplitHourNano<NoTime, NoTime, u8, u16, u16, u16> {
75        let (us_tmp, ns) = (nanos / 1000, (nanos % 1000) as u16);
76        let (ms_tmp, us) = (us_tmp / 1000, (us_tmp % 1000) as u16);
77        let (s, ms) = ((ms_tmp / 1000) as u8, (ms_tmp % 1000) as u16);
78        TimeSplit::new_hour_nano((), (), s, ms, us, ns)
79    }

81    /// Returns the time code as `HH:MM:SS:MIL` or `MM:SS:MIL`.
82    ///
83    /// The hours are clamped to 99 (more than 4 days).
84    ///
85    /// # Features
86    /// Makes use of the `unsafe_str` feature if enabled.
87    //
88    // -> 96 bits
89    pub fn secs_f64(seconds: f64) -> StringU8<12> {
90        let TimeSplitHourNano { h, m, s, ms, .. } = Self::split_secs_f64(seconds);
91        let m = Ascii(m as u32).digits_str(2);
92        let s = Ascii(s as u32).digits_str(2);
93        let ms = Ascii(ms as u32).digits_str(3);
94
95        let mut buf = [0; 12];
96        let mut buf_len = 12;
97
98        if h > 0 {
99            let h = Ascii(h.min(99)).digits_str(2);
100            let _str = format_buf![&mut buf, "{h}:{m}:{s}.{ms}"];
101        } else {
102            buf_len = 9;
103            let _str = format_buf![&mut buf, "{m}:{s}.{ms}"];
104        }
105
106        #[cfg(any(feature = "safe_time", not(feature = "unsafe_str")))]
107        return StringU8::<12>::from_bytes_nleft(buf, buf_len).unwrap();
108
109        #[cfg(all(not(feature = "safe_time"), feature = "unsafe_str"))]
110        // SAFETY: the buffer contains only ASCII characters.
111        unsafe {
112            StringU8::<12>::from_bytes_nleft_unchecked(buf, buf_len)
113        }
114    }

116    /// Returns the time code, up to seconds, as `001s 012ms 012Âµs 012ns`.
117    ///
118    /// The seconds are clamped to 999 (more than 16 minutes).
119    #[cfg(feature = "alloc")]

122    pub fn nanos_u64_alloc(ns: u64) -> String {
123        let (us, ns_rem) = (ns / 1_000, ns % 1_000);
124        let (ms, us_rem) = (us / 1_000, us % 1_000);
125        let (s, ms_rem) = (ms / 1_000, ms % 1_000);
126        let s = s.min(999);
127
128        if s > 0 {
129            format!["{s}s {ms_rem:03}ms {us_rem:03}Âµs {ns_rem:03}ns"]
130        } else if ms > 0 {
131            format!["{ms_rem}ms {us_rem:03}Âµs {ns_rem:03}ns"]
132        } else if us > 0 {
133            format!["{us_rem}Âµs {ns_rem:03}ns"]
134        } else {
135            format!["{ns_rem:03}ns"]
136        }
137    }

139    /// Returns the time code, up to seconds, as `001s 012ms 012Âµs 012345ns`.
140    ///
141    /// The seconds are clamped to 999 (more than 16 minutes).
142    // -> 208 bits
143    pub fn nanos_u64(nanos: u64) -> StringU8<23> {
144        let TimeSplitHourNano { s, ms, us, ns, .. } = Self::split_nanos_u64(nanos);
145        let s_str = Ascii(s.min(999)).digits_str(3);
146        let ms_str = Ascii(ms as u32).digits_str(3);
147        let us_str = Ascii(us as u32).digits_str(3);
148        let ns_str = Ascii(ns as u32).digits_str(3);
149
150        let mut buf = [0; 23];
151        let mut buf_len = 23; // = 18 + 3digits + 1name(s) + 1space
152
153        if s > 0 {
154            let _ = format_buf![&mut buf, "{s_str}s {ms_str}ms {us_str}Âµs {ns_str}ns"];
155        } else if ms > 0 {
156            buf_len = 18; // = 18 + 3digits + 2name(ms) + 1space
157            let _ = format_buf![&mut buf, "{ms_str}ms {us_str}Âµs {ns_str}ns"];
158        } else if us > 0 {
159            buf_len = 12; // = 5 + 3digits + 3name(Âµs) + 1space
160            let _ = format_buf![&mut buf, "{us_str}Âµs {ns_str}ns"];
161        } else {
162            buf_len = 5; // = 0 + 3digits + 2name(ns)
163            let _ = format_buf![&mut buf, "{ns_str}ns"];
164        }
165
166        #[cfg(any(feature = "safe_time", not(feature = "unsafe_str")))]
167        return StringU8::<23>::from_bytes_nleft(buf, buf_len).unwrap();
168
169        #[cfg(all(not(feature = "safe_time"), feature = "unsafe_str"))]
170        // SAFETY: the buffer contains only ASCII characters.
171        unsafe {
172            StringU8::<23>::from_bytes_nleft_unchecked(buf, buf_len)
173        }
174    }
