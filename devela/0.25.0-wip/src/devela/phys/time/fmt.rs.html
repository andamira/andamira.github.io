23/// Timecode splitting and formatting.
24///
25/// # Examples
26/// ```
27/// # use devela::Timecode;
28/// #[cfg(feature = "_str_u8")]
29/// #[cfg(any(feature = "std", feature = "_float_f64"))]
30/// assert_eq!(Timecode::secs_f64(3661.5), "01:01:01.500");
31///
32/// #[cfg(feature = "_str_u8")]
33/// assert_eq!(Timecode::nanos_u64(1_002_003_004), "001s 002ms 003µs 004ns");
34/// ```
35#[derive(Debug)]
36pub struct Timecode;
37
38impl Timecode {
39    /* time splitting */
40
41    /// Decomposes a number of `seconds` in `{ h, m, s, ms }`.
42    ///
43    /// The maximum decomposition for [`u64::MAX`] is
44    /// `{ h: 5_124_095_576_030_431, .. }` (more than 584_942_417 millenia).
45    // -> 64 bits
46    #[must_use]
47    #[cfg(any(feature = "std", feature = "_float_f64"))]
48    #[cfg_attr(nightly_doc, doc(cfg(any(feature = "std", feature = "_float_f64"))))]
49    pub fn split_secs_f64(seconds: f64) -> TimeSplitHourNano<u32, u8, u8, u16, NoTime, NoTime> {
50        let ms = (seconds.fract() * 1000.) as u16;
51        let mut ts = seconds.trunc() as u64;
52        let h = (ts / 3600) as u32;
53        ts %= 3600;
54        let m = (ts / 60) as u8;
55        let s = (ts % 60) as u8;
56        TimeSplit::new_hour_nano(h, m, s, ms, (), ())
57    }
58
59    /// Splits a number of `nanoseconds` in `{ s, ms, µs, ns }`.
60    ///
61    /// The maximum decomposition for [`u64::MAX`] is
62    /// `{ s: 1_266_874_889, .. }` (more than 40 years).
63    // -> 80 bits
64    #[must_use]
65    pub const fn split_nanos_u64(
66        nanos: u64,
67    ) -> TimeSplitHourNano<NoTime, NoTime, u32, u16, u16, u16> {
68        let (us_tmp, ns) = (nanos / 1000, (nanos % 1000) as u16);
69        let (ms_tmp, us) = (us_tmp / 1000, (us_tmp % 1000) as u16);
70        let (s, ms) = ((ms_tmp / 1000) as u32, (ms_tmp % 1000) as u16);
71        TimeSplit::new_hour_nano((), (), s, ms, us, ns)
72    }
73
74    /// Splits a number of `nanoseconds` in `{ s, ms, µs, ns }`.
75    ///
76    /// The maximum decomposition for [`u32::MAX`] is
77    /// `{ ns: 295, µs: 967, ms: 294, s: 4 }` (more than 4 seconds).
78    // -> 56 bits
79    #[must_use]
80    pub const fn split_nanos_u32(
81        nanos: u32,
82    ) -> TimeSplitHourNano<NoTime, NoTime, u8, u16, u16, u16> {
83        let (us_tmp, ns) = (nanos / 1000, (nanos % 1000) as u16);
84        let (ms_tmp, us) = (us_tmp / 1000, (us_tmp % 1000) as u16);
85        let (s, ms) = ((ms_tmp / 1000) as u8, (ms_tmp % 1000) as u16);
86        TimeSplit::new_hour_nano((), (), s, ms, us, ns)
87    }
88
89    /// Returns the time code as `HH:MM:SS:MIL` or `MM:SS:MIL`.
90    ///
91    /// The hours are clamped to 99 (more than 4 days).
92    ///
93    /// # Features
94    /// Makes use of the `unsafe_str` feature if enabled.
95    //
96    // -> 96 bits
97    #[cfg(any(feature = "std", feature = "_float_f64"))]
98    #[cfg_attr(nightly_doc, doc(cfg(any(feature = "std", feature = "_float_f64"))))]
99    #[cfg(feature = "_str_u8")]
100    #[cfg_attr(nightly_doc, doc(cfg(feature = "_str_u8")))]
101    pub fn secs_f64(seconds: f64) -> StringU8<12> {
102        let TimeSplitHourNano { h, m, s, ms, .. } = Self::split_secs_f64(seconds);
103        let m = Ascii(m as u32).digits_str(2);
104        let s = Ascii(s as u32).digits_str(2);
105        let ms = Ascii(ms as u32).digits_str(3);
106
107        let mut buf = [0; 12];
108        let mut buf_len = 12;
109
110        if h > 0 {
111            let h = Ascii(h.min(99)).digits_str(2);
112            let _str = format_buf![&mut buf, "{h}:{m}:{s}.{ms}"];
113        } else {
114            buf_len = 9;
115            let _str = format_buf![&mut buf, "{m}:{s}.{ms}"];
116        }
117
118        #[cfg(any(feature = "safe_time", not(feature = "unsafe_str")))]
119        return StringU8::<12>::from_bytes_nleft(buf, buf_len).unwrap();
120
121        #[cfg(all(not(feature = "safe_time"), feature = "unsafe_str"))]
122        // SAFETY: the buffer contains only ASCII characters.
123        unsafe {
124            StringU8::<12>::from_bytes_nleft_unchecked(buf, buf_len)
125        }
126    }
127
128    /// Returns the time code, up to seconds, as `001s 012ms 012µs 012ns`.
129    ///
130    /// The seconds are clamped to 999 (more than 16 minutes).
131    #[cfg(feature = "alloc")]
132    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
133    // -> 192 bits in "64", 92 bits in "32"
134    pub fn nanos_u64_alloc(ns: u64) -> String {
135        let (us, ns_rem) = (ns / 1_000, ns % 1_000);
136        let (ms, us_rem) = (us / 1_000, us % 1_000);
137        let (s, ms_rem) = (ms / 1_000, ms % 1_000);
138        let s = s.min(999);
139
140        if s > 0 {
141            format!["{s}s {ms_rem:03}ms {us_rem:03}µs {ns_rem:03}ns"]
142        } else if ms > 0 {
143            format!["{ms_rem}ms {us_rem:03}µs {ns_rem:03}ns"]
144        } else if us > 0 {
145            format!["{us_rem}µs {ns_rem:03}ns"]
146        } else {
147            format!["{ns_rem:03}ns"]
148        }
149    }
150
151    /// Returns the time code, up to seconds, as `001s 012ms 012µs 012345ns`.
152    ///
153    /// The seconds are clamped to 999 (more than 16 minutes).
154    // -> 208 bits
155    #[cfg(feature = "_str_u8")]
156    #[cfg_attr(nightly_doc, doc(cfg(feature = "_str_u8")))]
157    pub fn nanos_u64(nanos: u64) -> StringU8<23> {
158        let TimeSplitHourNano { s, ms, us, ns, .. } = Self::split_nanos_u64(nanos);
159        let s_str = Ascii(s.min(999)).digits_str(3);
160        let ms_str = Ascii(ms as u32).digits_str(3);
161        let us_str = Ascii(us as u32).digits_str(3);
162        let ns_str = Ascii(ns as u32).digits_str(3);
163
164        let mut buf = [0; 23];
165        let mut buf_len = 23; // = 18 + 3digits + 1name(s) + 1space
166
167        if s > 0 {
168            let _ = format_buf![&mut buf, "{s_str}s {ms_str}ms {us_str}µs {ns_str}ns"];
169        } else if ms > 0 {
170            buf_len = 18; // = 18 + 3digits + 2name(ms) + 1space
171            let _ = format_buf![&mut buf, "{ms_str}ms {us_str}µs {ns_str}ns"];
172        } else if us > 0 {
173            buf_len = 12; // = 5 + 3digits + 3name(µs) + 1space
174            let _ = format_buf![&mut buf, "{us_str}µs {ns_str}ns"];
175        } else {
176            buf_len = 5; // = 0 + 3digits + 2name(ns)
177            let _ = format_buf![&mut buf, "{ns_str}ns"];
178        }
179
180        #[cfg(any(feature = "safe_time", not(feature = "unsafe_str")))]
181        return StringU8::<23>::from_bytes_nleft(buf, buf_len).unwrap();
182
183        #[cfg(all(not(feature = "safe_time"), feature = "unsafe_str"))]
184        // SAFETY: the buffer contains only ASCII characters.
185        unsafe {
186            StringU8::<23>::from_bytes_nleft_unchecked(buf, buf_len)
187        }
188    }
189}
