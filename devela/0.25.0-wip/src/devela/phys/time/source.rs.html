18/// A source of timestamps with a known granularity and monotonicity.
19///
20/// Provides a consistent API for querying timestamps at various precisions,
21/// abstracting over different time sources.
22pub trait TimeSource<const MONOTONIC: bool> {
23    /// Returns the granularity of this time source.
24    fn granularity() -> Oneof<2, TimeGranularity, Ratio<u32, u32>>;
25
26    /// Returns the current timestamp in milliseconds.
27    fn now_millis() -> u64;
28
29    /* provided methods */
30
31    /// Returns the epoch offset in milliseconds.
32    ///
33    /// - For absolute sources (e.g. `SystemTime`), returns the absolute epoch (e.g. `UNIX_EPOCH`).
34    /// - For monotonic sources with a known meaningful reference (e.g. `JsInstant`),
35    ///   returns a meaningful offset.
36    /// - For monotonic sources without a meaningful absolute reference (e.g. `SystemInstant`),
37    ///   returns `0` (default).
38    fn epoch_millis() -> u64 { 0 }
39
40    /// Returns the current timestamp in microseconds.
41    ///
42    /// Default: Uses `now_millis()`.
43    fn now_micros() -> u64 { Self::now_millis() * 1_000 }
44
45    /// Returns the epoch offset in microseconds.
46    ///
47    /// Default: Uses `epoch_millis()`.
48    fn epoch_micros() -> u64 { Self::epoch_millis() * 1_000 }
49
50    /// Returns the current timestamp in nanoseconds.
51    ///
52    /// Default: Uses `now_millis()`.
53    fn now_nanos() -> u64 { Self::now_millis() * 1_000_000 }
54
55    /// Returns the epoch offset in nanoseconds.
56    ///
57    /// Default: Uses `epoch_millis()`.
58    fn epoch_nanos() -> u64 { Self::epoch_millis() * 1_000_000 }
59
60    /// Returns the current timestamp as an `f64` value in milliseconds.
61    ///
62    /// Default: Converts `now_millis()` to `f64`.
63    fn now_millis_f64() -> f64 { Self::now_millis() as f64 }
64
65    /// Returns the current timestamp as an `f64` value in milliseconds.
66    ///
67    /// Default: Converts `now_millis()` to `f64`.
68    fn epoch_millis_f64() -> f64 { Self::epoch_millis() as f64 }
69}

141    /// A test-friendly time source that allows manual control.
142    ///
143    /// `TimeSourceFake` provides a controlled, adjustable timestamp source for tests.
144    /// This enables predictable behavior when testing time-dependent systems.
145    ///
146    /// # Features:
147    /// - **Manually set the time** with `set_time()`.
148    /// - **Manually advance time** with `advance_time()`.
149    /// - **Implements `TimeSource`**, so it works seamlessly in tests.
150    ///
151    /// # Example:
152    /// ```
153    /// # use devela::TimeSourceFake;
154    /// let ts = TimeSourceFake::new(1_700_000_000_000);
155    /// assert_eq!(ts.now_millis(), 1_700_000_000_000);
156    /// ts.advance_time(1000);
157    /// assert_eq!(ts.now_millis(), 1_700_000_001_000);
158    /// ```
159    pub(crate) struct TimeSourceFake {
160        /// Atomic for safe multi-threaded testing
161        now: AtomicU64,
162    }
163    impl TimeSourceFake {
164        /// Creates a new `TimeSourceFake` with the given starting fake time (in milliseconds).
165        pub const fn new(start_time: u64) -> Self {
166            Self { now: AtomicU64::new(start_time) }
167        }
168        /// Manually sets the fake time to a specific value (in milliseconds).
169        pub fn get_time(&self, new_time: u64) {
170            self.now.load(AtomicOrdering::SeqCst);
171        }
172        /// Manually sets the fake time to a specific value (in milliseconds).
173        pub fn set_time(&self, new_time: u64) {
174            self.now.store(new_time, AtomicOrdering::SeqCst);
175        }
176        /// Advances the fake time by a given amount (in milliseconds).
177        pub fn advance_time(&self, millis: u64) {
178            self.now.fetch_add(millis, AtomicOrdering::SeqCst);
179        }
180    }
181    impl TimeSource<true> for TimeSourceFake {
182        fn granularity() -> Oneof<2, TimeGranularity, Ratio<u32, u32>> {
183            Oneof::_0(TimeGranularity::Millis)
184        }
185        fn now_millis() -> u64 {
186            TIME_SOURCE_FAKE.now.load(AtomicOrdering::SeqCst)
187        }
188        fn epoch_millis() -> u64 {
189            1_700_000_000_000 // Default testing epoch
190        }
191    }
