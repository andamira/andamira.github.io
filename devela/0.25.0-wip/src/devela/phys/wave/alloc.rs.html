18/// Trait for lossy compression of wavelet coefficients.
19///
20/// Compresses coefficients based on a given tolerance, zeroing out those
21/// deemed insignificant to the desired error threshold.
22pub trait WaveletCompressionVec {

25}

27/// Trait defining essential wavelet transform operations.
28pub trait WaveletTransformVec {

30    #[must_use]

33    /// Computes the inverse wavelet transform on the given coefficients.

38/// A single unit of the wavelet decomposition at a specific level and position.
39#[derive(Debug)]
40pub struct WaveletUnitVec {

42    pub component_type: WaveletUnitRole,
43    /// Resolution level of the component, indicating its level of detail.
44    pub level: usize,
45    /// Position index of the component within the resolution level.
46    pub position: usize,
47    /// Coefficient values defining the component's shape.
48    pub values: Vec<f64>,
49}

65impl WaveletCompressionVec for WaveletHaar {
66    fn compress(&self, coeffs: &[f64], tolerance: f64) -> Vec<f64> {
67        coeffs.iter().map(|&c| if c.abs() < tolerance { 0.0 } else { c }).collect()
68    }
69}
70
71impl WaveletTransformVec for WaveletHaar {
72    fn forward(&self, input: &[f64]) -> Vec<f64> {
73        let mut output = input.to_vec();
74        let mut length = output.len();
75
76        // Iteratively apply averaging and differencing until we reach the coarsest level.
77        while length > 1 {
78            let mut temp = vec![0.0; length];
79            for i in 0..length / 2 {
80                // Calculate average and detail coefficients for each pair.
81                let average = (output[2 * i] + output[2 * i + 1]) / 2.0;
82                let difference = (output[2 * i] - output[2 * i + 1]) / 2.0;
83                temp[i] = average;
84                temp[length / 2 + i] = difference; // Store details in the second half
85            }
86            // Copy temporary results back to output for the next level.
87            output[..length].clone_from_slice(&temp);
88            length /= 2; // Reduce length to work on the next coarser level
89        }
90        output
91    }
92
93    fn inverse(&self, coeffs: &[f64]) -> Vec<f64> {
94        let mut output = coeffs.to_vec();
95        let mut length = 2;
96
97        // Iteratively reconstruct from coarsest to finest level
98        while length <= output.len() {
99            let mut temp = vec![0.0; length];
100            for i in 0..length / 2 {
101                // Reconstruct original values from average and detail coefficients.
102                let average = output[i];
103                let difference = output[length / 2 + i];
104                temp[2 * i] = average + difference; // First reconstructed value
105                temp[2 * i + 1] = average - difference; // Second reconstructed value
106            }
107            // Copy reconstructed values back to output for the next finer level.
108            output[..length].clone_from_slice(&temp);
109            length *= 2; // Expand length to work on the next finer level
110        }
111        output
112    }
113}
