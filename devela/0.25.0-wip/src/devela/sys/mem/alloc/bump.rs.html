6use crate::{

11#[doc = crate::TAG_ALLOCATOR!()]
12/// A simple, single-threaded bump allocator.
13///
14/// - Allocates memory linearly from a fixed memory region.
15/// - Does not support deallocation.
16/// - Suitable for embedded and WASM.
17///
18/// # Safety
19/// This allocator is thread-safe only when allocations are uncontended.
20/// For full thread safety, use a lock or atomic coordination externally.
21///
22/// # Example
23/// ```ignore
24/// # use devela::BumpAlloc;
25/// #[global_allocator]
26/// static ALLOCATOR: BumpAlloc = BumpAlloc::new(0x10000, 0x20000);
27/// ```
28#[derive(Debug)]
29pub struct BumpAlloc {
30    current: AtomicUsize,
31    end: usize,
32}
33
34impl BumpAlloc {
35    /// Creates a new `BumpAlloc` from `start..end` addresses.
36    ///
37    /// # Panics
38    /// Panics if `start >= end`.
39    pub const fn new(start: usize, end: usize) -> Self {
40        debug_assert![start < end];
41        BumpAlloc { current: AtomicUsize::new(start), end }
42    }
43}
44
45unsafe impl GlobalAlloc for BumpAlloc {
46    unsafe fn alloc(&self, layout: MemLayout) -> *mut u8 {
47        let (align, size) = (layout.align(), layout.size());
48        // Load the current allocation pointer atomically (no ordering dependency)
49        let mut current = self.current.load(Relaxed);
50
51        loop {
52            // Align the current address up to the requested alignment
53            let aligned = (current + align - 1) & !(align - 1);
54            // Compute the next allocation pointer (may overflow)
55            let next = aligned.checked_add(size).unwrap_or(self.end + 1);
56            // If allocation would exceed the end of the memory region, fail
57            is![next > self.end; return Ptr::null_mut()];
58            // Attempt to atomically reserve the space by advancing the pointer
59            match self.current.compare_exchange_weak(current, next, SeqCst, Relaxed) {
60                Ok(_) => return aligned as *mut u8, // Success: return aligned pointer
61                Err(updated) => current = updated,  // Fail: retry with updated `current`
62            }
63        }
64    }
65    unsafe fn dealloc(&self, _ptr: *mut u8, _layout: MemLayout) {} // IMPROVE
66}
