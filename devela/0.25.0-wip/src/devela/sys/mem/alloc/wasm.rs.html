7#[cfg(target_arch = "wasm32")]

10#[doc = crate::TAG_ALLOCATOR!()]
11/// A WebAssembly global memory allocator that uses a bump allocation strategy.
12///
13/// This allocator manages memory starting from `__heap_base` (the WASM heap start)
14/// and grows memory as needed using `memory.grow`. It's designed for single-threaded
15/// WASM environments where synchronization isn't required.
16///
17/// # Safety
18/// - This allocator is not thread-safe. The unsafe `Sync` implementation is only
19///   valid in single-threaded WASM environments.
20///
21/// # Example
22/// ```ignore
23/// # use devela::WasmAlloc;
24/// #[global_allocator]
25/// static ALLOCATOR: WasmAlloc = WasmAlloc::INIT;
26/// ```
27#[doc = crate::_doc!(vendor: "mini-alloc")]
28#[derive(Debug)]
29pub struct WasmAlloc;
30impl WasmAlloc {
31    /// Initial value for `WasmAlloc`.
32    pub const INIT: Self = WasmAlloc;
33}
34/// This is safe in single-threaded WASM environments bug invalid in multi-threaded contexts.
35unsafe impl Sync for WasmAlloc {}
36
37#[rustfmt::skip]
38unsafe impl GlobalAlloc for WasmAlloc {
39    /// Allocates memory with the given layout.
40    #[allow(unused_variables, reason = "cross-platform")]
41    unsafe fn alloc(&self, l: MemLayout) -> *mut u8 {
42        #[cfg(target_arch = "wasm32")]
43        return unsafe { alloc_impl(l) }.unwrap_or(Ptr::null_mut());
44
45        #[cfg(not(target_arch = "wasm32"))]
46        Ptr::null_mut()
47    }
48    /// Allocates zeroed memory with the given layout.
49    /// This is the same as `alloc` since WASM memory is always zero-initialized.
50    #[inline(always)]
51    unsafe fn alloc_zeroed(&self, l: MemLayout) -> *mut u8 { unsafe { self.alloc(l) } }
52    /// No-op. Memory is only reclaimed when the entire WASM instance is dropped.
53    #[inline(always)]
54    unsafe fn dealloc(&self, _ptr: *mut u8, _l: MemLayout) {}
55}
56
57/// Global allocator state tracking the current bump offset and memory boundary.
58///
59/// The state stores negative values because:
60/// 1. Aligning negative offsets requires fewer instructions than positive offsets.
61/// 2. Allows using saturating arithmetic to detect out-of-memory conditions.
62///
63/// The tuple contains:
64/// - `.0`: Negative of the current bump pointer offset from `__heap_base`
65/// - `.1`: Negative of the current memory boundary (total available memory)
66#[cfg(target_arch = "wasm32")]
67static mut WASM_ALLOC_STATE: Option<(NonZeroUsize, usize)> = None;
68
69/// Internal allocation implementation using bump allocation strategy.
70///
71/// # Safety
72/// - This function accesses and modifies global mutable state.
73/// - The caller must ensure no other allocations are happening concurrently.
74///
75/// Returns `None` if allocation fails (cannot grow memory or layout is invalid).
76#[cfg(target_arch = "wasm32")]
77#[allow(clippy::deref_addrof, reason = "safe reference to static mut")]
78unsafe fn alloc_impl(layout: MemLayout) -> Option<*mut u8> {
79    // SAFETY: Single-threaded WASM, the static is only referenced here, the function not reentrant.
80    // https://doc.rust-lang.org/nightly/edition-guide/rust-2024/static-mut-references.html#safe-references
81    let state_ref = unsafe { &mut *&raw mut WASM_ALLOC_STATE };
82    let (neg_offset, neg_bound) = state_ref.get_or_insert_with(|| {
83        let bound = Wasm::PAGE_BYTES * Wasm::memory_pages() - 1;
84        (
85            unsafe { NonZeroUsize::new_unchecked(Wasm::heap_base().wrapping_neg()) },
86            bound.wrapping_neg(),
87        )
88    });
89    // Align the current offset for the requested alignment
90    let neg_aligned = Mem::align_down(neg_offset.get(), layout.align());
91    // Calculate new offset after allocating the requested size
92    let next_neg_offset = neg_aligned.checked_sub(layout.size())?;
93    // Check if we need to grow memory
94    let bytes_needed = neg_bound.saturating_sub(next_neg_offset + 1);
95    if bytes_needed != 0 {
96        let pages_needed = 1 + (bytes_needed - 1) / Wasm::PAGE_BYTES;
97        Wasm::memory_grow_checked(pages_needed)?;
98        *neg_bound -= Wasm::PAGE_BYTES * pages_needed;
99    }
100    // Update state and return the allocated pointer
101    *neg_offset = unsafe { NonZeroUsize::new_unchecked(next_neg_offset) };
102    Some(neg_aligned.wrapping_neg() as *mut u8)
103}
