6#[cfg(feature = "alloc")]
7use devela::{Borrow, String, Vec};
8
9/// Defines the relationship between a borrowed type and its owned counterpart.
10///
11/// This enables abstracting over allocation strategies in [`MaybeOwned`][crate::MaybeOwned].
12/// Implement this for types where:
13/// - The borrowed form is `&T`
14/// - The owned form implements `Borrow<T>` (can be viewed as `&T`)
15///
16/// # Allocation Requirement
17/// This trait's associated type **always requires `alloc`** because:
18/// 1. The `Owned` type must implement `Borrow<Self>`
19/// 2. Non-alloc owned types would violate the borrowing relationship
20///
21/// However, the trait itself is `core`-compatible
22/// # Examples
23/// ```ignore
24/// # use devela::{Ownership, MaybeOwned};
25/// #[cfg(feature = "alloc")]
26/// impl Ownership for str {
27///     type Owned = String;  // String implements Borrow<str>
28/// }
29/// ```
30///
31/// # Alloc Consideration
32/// The trait itself is `core`-compatible, but most implementations will require
33/// the `alloc` feature for their `Owned` types.
34pub trait Ownership {
35    /// The owned counterpart.
36    #[cfg(feature = "alloc")]
37    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
38    type Backing: Borrow<Self>;
39}
