3//! Defines the [`Backing`] enum.

6#[cfg(feature = "alloc")]
7use devela::RangeFull;
8
9/// Controls how memory is provided for a value.
10///
11/// Complements [`MaybeOwned`][super::MaybeOwned] by specifying whether to:
12/// - Use an existing mutable buffer (`Buf`)
13/// - Allocate new memory (`Alloc`)
14///
15/// Enables APIs to flexibly accept either caller-provided buffers or fall back to
16/// allocations (when `alloc` is enabled)
17///
18/// # Example
19/// ```ignore
20/// # use devela::{Backing, MaybeOwned};
21/// fn process(mode: Backing<'_>) -> MaybeOwned<'_, str> {
22///     match mode {
23///         Backing::Buf(buffer) => MaybeOwned::borrowed(/*...*/),
24///         Backing::Alloc => MaybeOwned::owned(/*...*/),
25///     }
26/// }
27/// ```
28/// ## Features
29/// The `Alloc` variant disappears when `alloc` is disabled.
30#[derive(Debug, PartialEq)]
31pub enum Backing<'a> {
32    /// Reuse an existing exclusive buffer.
33    Buf(&'a mut [u8]),
34
35    /// Allocate new memory via the global allocator.
36    #[cfg(feature = "alloc")]
37    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
38    Alloc,
39}
40
41impl Backing<'_> {
42    /// Returns `true` if this mode requires allocation.
43    #[inline(always)]
44    pub const fn is_buf(&self) -> bool {
45        #[cfg(not(feature = "alloc"))]
46        return false;
47        #[cfg(feature = "alloc")]
48        matches!(self, Self::Alloc)
49    }
50
51    /// Returns `true` if this mode uses existing memory.
52    #[inline(always)]
53    pub const fn is_alloc(&self) -> bool {
54        matches!(self, Self::Buf(_))
55    }
56}
57
58// The following conversions can be used when accepting Into<Backing>.
59// The problem is the buffer would get moved if not re-referenced.
60
61impl<'a> From<&'a mut [u8]> for Backing<'a> {
62    /// Convert an exclusive slice into borrowing mode.
63    fn from(buf: &'a mut [u8]) -> Self {
64        Self::Buf(buf)
65    }
66}
67impl<'a, const LEN: usize> From<&'a mut [u8; LEN]> for Backing<'a> {
68    /// Convert an exclusive reference to an array into borrowing mode.
69    fn from(buf: &'a mut [u8; LEN]) -> Self {
70        Self::Buf(buf)
71    }
72}
73#[cfg(feature = "alloc")]
74#[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
75impl<'a> From<RangeFull> for Backing<'a> {
76    /// Convert `..` into allocation mode.
77    ///
78    /// The range syntax visually suggests "fill everything":
79    /// ```ignore
80    /// #[cfg(feature = "alloc")]
81    /// let s = load_string(..); // Indicates allocation
82    /// ```
83    fn from(_: RangeFull) -> Self {
84        Self::Alloc
85    }
86}
