15// Stores the current pointers for concrete types.
16define_static_map![typeid KeyCurrentMap];
17thread_local! {
18    static CURRENT_PTR_MAP: RefCell<KeyCurrentMap<u64, PtrNonNull<u8>, 64>>
19        = RefCell::new(KeyCurrentMap::new_cloned(PtrNonNull::<u8>::dangling()));
20}

22/// A guard that temporarily sets a global current pointer for `T`, restoring the old one on drop.
23///
24/// When dropped, it restores the previous pointer or sets a placeholder if none existed."
25///
26/// This is useful for tracking the current instance of a type within a thread.
27#[cfg_attr(doc, doc = ::devela::_doc!(vendor: "current"))]
28#[derive(Debug)]
29pub struct CurrentGuard<'a, T: Any> {
30    /// The active instance of `T` for the duration of this guard.
31    _current: &'a mut T,
32    /// The previous pointer, restored when this guard is dropped.
33    prev_ptr: Option<PtrNonNull<T>>,
34}
35impl<T: Any> CurrentGuard<'_, T> {
36    /// Creates a new *current guard* for the given value.
37    ///
38    /// When this guard is dropped, the previous pointer (if any) will be restored.
39    ///
40    /// # Safety
41    /// - This function modifies global state within a thread-local map.
42    /// - It ensures that only one mutable reference to a given `T` exists at a time.
43    /// - Improper use may lead to stale pointers if lifetimes are not respected.
44    ///
45    /// # Example
46    /// ```
47    /// # use devela::{CurrentGuard};
48    /// # struct MyType { data: u64} impl MyType { fn new() -> Self { Self { data:0 } }}
49    /// let mut my_value = MyType::new();
50    /// let guard = CurrentGuard::new(&mut my_value);
51    /// ```
52    pub fn new(current: &mut T) -> CurrentGuard<'_, T> {
53        let ptr = PtrNonNull::from(&mut *current).cast::<u8>();
54        let prev_ptr = CURRENT_PTR_MAP.with(|current| {
55            let mut map = current.borrow_mut();
56            if let Some(entry) = map.get_mut_type::<T>() {
57                Some(Mem::replace(entry, ptr).cast::<T>())
58            } else {
59                map.insert_type::<T>(ptr.cast()).ok();
60                None
61            }
62        });
63        CurrentGuard { prev_ptr, _current: current }
64    }
65}
66impl<T: Any> Drop for CurrentGuard<'_, T> {
67    fn drop(&mut self) {
68        CURRENT_PTR_MAP.with(|current| {
69            let mut map = current.borrow_mut();
70            match self.prev_ptr {
71                None => map.insert_type::<T>(PtrNonNull::<u8>::dangling().cast()).ok(),
72                Some(prev_ptr) => map.insert_type::<T>(prev_ptr.cast()).ok(),
73            }
74        });
75    }
76}

78/// A marker object representing the current instance of a type `T`.
79///
80/// This struct does not hold any actual value but instead allows access to
81/// a globally tracked instance of `T`, typically managed through `CurrentGuard`.
82///
83/// The primary purpose of `Current<T>` is to:
84/// - Provide safe, structured access to a global instance of `T`.
85/// - Prevent direct global mutable access in safe code.
86///
87/// Not a smart pointer; instead, it acts as a reference handle for thread-local state.
88#[cfg_attr(doc, doc = ::devela::_doc!(vendor: "current"))]
89#[derive(Debug)]
90pub struct Current<T>(PhantomData<T>);
91
92impl<T: Any> Current<T> {
93    /// Creates a new `Current<T>` marker object.
94    ///
95    /// # Safety
96    /// - This function does not initialize an actual value.
97    /// - Dereferencing without an active `CurrentGuard` leads to undefined behavior.
98    /// - Ensure that `CurrentGuard` is properly managed before using this.
99    ///
100    /// # Example
101    /// ```ignore
102    /// let current = unsafe { Current::<MyType>::new() };
103    /// ```
104    pub unsafe fn new() -> Current<T> {
105        Current(PhantomData)
106    }
107
108    /// Retrieves an exclusive reference to the current instance of `T`, if set.
109    ///
110    /// # Safety
111    /// - May return a dangling reference if the associated `CurrentGuard<T>` has been dropped.
112    /// - The caller must ensure the reference is only used while the `CurrentGuard<T>` is alive.
113    /// - If the same `T` is set multiple times, the reference may be stale.
114    ///
115    /// # Example
116    /// ```ignore
117    /// let mut my_value = MyType::new();
118    /// let guard = CurrentGuard::new(&mut my_value);
119    /// let current = unsafe { Current::<MyType>::new().current() };
120    /// assert!(current.is_some());
121    /// ```
122    #[rustfmt::skip]
123    pub unsafe fn current(&mut self) -> Option<&mut T> {
124        let ptr: PtrNonNull<u8> = CURRENT_PTR_MAP.with(|current| current.borrow().get_type::<T>())?;
125        // SAFETY: The pointer is non-null but may not be valid; ensure `CurrentGuard<T>` is active.
126        Some(unsafe { &mut *ptr.cast::<T>().as_ptr() })
127    }
128
129    /// Retrieves an exclusive reference to the current instance of `T`, or panics.
130    ///
131    /// # Safety
132    /// Same as [`current`][Self::current]: the caller must ensure the reference remains valid.
133    ///
134    /// # Panics
135    /// Panics if no instance of `T` is currently set.
136    ///
137    /// # Example
138    /// ```ignore
139    /// let mut my_value = MyType::new();
140    /// let guard = CurrentGuard::new(&mut my_value);
141    /// let current = unsafe { Current::<MyType>::new().current_unwrap() };
142    /// ```
143    pub unsafe fn current_unwrap(&mut self) -> &mut T {
144        // SAFETY: Panics if no `CurrentGuard<T>` exists, preventing invalid access.
145        match unsafe { self.current() } {
146            None => panic!("No current `{}` is set", any_type_name::<T>()),
147            Some(x) => x,
148        }
149    }
150}
151impl<T: Any> Deref for Current<T> {
152    type Target = T;
153
154    #[inline(always)] #[rustfmt::skip]
155    fn deref<'a>(&'a self) -> &'a T {
156        // SAFETY:
157        // - `Current<T>` is only an access point, not an actual value.
158        // - Transmuting `&Current<T>` to `&mut Current<T>` is safe since it's never directly used.
159        // - Caller must ensure `CurrentGuard<T>` is active; otherwise, `current_unwrap` panics.
160        #[allow(mutable_transmutes, reason = "`Current` only acts as a reference handle")]
161        unsafe { transmute::<&Current<T>, &'a mut Current<T>>(self).current_unwrap() }
162    }
163}
164impl<T: Any> DerefMut for Current<T> {
165    #[inline(always)]
166    fn deref_mut(&mut self) -> &mut T {
167        // SAFETY:
168        // - Requires an active `CurrentGuard<T>`, ensuring a valid instance.
169        // - If no guard exists, `current_unwrap` panics instead of returning an invalid reference.
170        unsafe { self.current_unwrap() }
171    }
172}

175fn main() {
176    struct State {

178    }
179    impl State {
180        // prints the current text, and changes it to "world!"
181        fn print() {
182            let mut ctx = unsafe { Current::<State>::new() };
183            println!("{}", ctx.text);
184            ctx.text = "world!".to_string();
185        }
186        // changes the text to "good bye" and calls print() two times.
187        fn bar() {
188            let mut bar = State { text: "good bye".to_string() };
189            let guard = CurrentGuard::new(&mut bar);
190            State::print(); // good bye
191            State::print(); // world!
192        }
193    }
194    let mut ctx = State { text: "hello".to_string() };
195    let guard = CurrentGuard::new(&mut ctx);
196    State::print(); // hello
197    State::print(); // world!
198    State::bar(); // good bye world!
