43/// Type size information in bits.
44///
45/// Indicates a size of exactly `LEN` bits for the relevant data part of this type.
46///
47/// E.g. a `bool` has a BitSized of 1 bit.
48pub trait BitSized<const LEN: usize>: ByteSized {
49    /// The bit size of this type (only the relevant data part, without padding).
50    ///
51    /// # Panics
52    /// Panics if `MIN_BYTE_SIZE > `[`BYTE_SIZE`][ByteSized::BYTE_SIZE],
53    const BIT_SIZE: usize = {
54        let min_byte_size = Mem::bytes_from_bits(LEN);
55        if min_byte_size > Self::BYTE_SIZE {
56            panic!["BitSized::MIN_BYTE_SIZE > ByteSized::BYTE_SIZE"];
57        }
58        LEN
59    };
60
61    /// The rounded up byte size for this type.
62    ///
63    /// This is the minimum number of full bytes needed to represent this type.
64    /// Basically `(LEN + 7) / 8`.
65    ///
66    /// # Panics
67    /// Panics if `MIN_BYTE_SIZE > `[`BYTE_SIZE`][ByteSized::BYTE_SIZE],
68    const MIN_BYTE_SIZE: usize = {
69        let min_byte_size = Mem::bytes_from_bits(LEN);
70        if min_byte_size > Self::BYTE_SIZE {
71            panic!["BitSized::MIN_BYTE_SIZE > ByteSized::BYTE_SIZE"];
72        }
73        min_byte_size
74    };
75
76    /// Returns the bit size of this type (only the relevant data part, without padding).
77    ///
78    /// # Panics
79    /// Panics if `MIN_BYTE_SIZE > `[`BYTE_SIZE`][ByteSized::BYTE_SIZE],
80    #[must_use]
81    fn bit_size(&self) -> usize {
82        Self::BIT_SIZE
83    }
84
85    /// Returns the rounded up byte size for this type.
86    ///
87    /// This is the minimum number of full bytes needed to represent this type.
88    /// Basically `(LEN + 7) / 8`.
89    ///
90    /// # Panics
91    /// Panics if `MIN_BYTE_SIZE > `[`BYTE_SIZE`][ByteSized::BYTE_SIZE],
92    #[must_use]
93    fn min_byte_size(&self) -> usize {
94        Self::MIN_BYTE_SIZE
95    }
96}

98/// Helper macro to implement [`BitSized`].
99macro_rules! bit_sized {
100    /* primitives */
101
102    (= $bits:expr; for $($t:ty),+) => { $( impl $crate::BitSized<$bits> for $t {} )+ };
103
104    /* primitives generic on $T */
105
106    (<$T:ident> = $bits:expr; for $($t:ty),+) => {
107        $( impl<$T> $crate::BitSized<$bits> for $t {} )+
108    };
109    (<const $T:ident: $Tt:ty> = $bits:expr; for $($t:ty),+) => {
110        $( impl<const $T: $Tt> $crate::BitSized<$bits> for $t {} )+
111    };
112
113    /* primitives generic on $K, $V */
114
115    (<$K:ident, $V:ident> = $bits:expr; for $($t:ty),+) => {
116        $( impl<$K, $V> $crate::BitSized<$bits> for $t {} )+
117    };
118    (<const $K:ident: $Kt:ty, const $V:ident: $Vt:ty> = $bits:expr; for $($t:ty),+) => {
119        $( impl<const $K: $Kt, const $V: $Vt> $crate::BitSized<$bits> for $t {} )+
120    };
121
122    /* pointer primitives */
123
124    // implements BitSized<$PTR_BITS> for pointer-sized related types.
125    (pointer = $PTR_BITS:literal) => {
126        bit_sized![= $PTR_BITS; for isize, usize];
127
128        bit_sized![= $PTR_BITS; for NonZeroIsize, NonZeroUsize];
129
130        #[cfg(all(feature = "work", any(feature = "dep_portable_atomic", target_has_atomic = "ptr")))]
131        bit_sized![= $PTR_BITS; for AtomicIsize, AtomicUsize];
132        #[cfg(all(feature = "work", any(feature = "dep_portable_atomic", target_has_atomic = "ptr")))]
133        bit_sized![<T> = $PTR_BITS; for AtomicPtr<T>];
134
135        #[cfg(feature = "std")]
136        bit_sized![<T> = {$PTR_BITS * 1}; for Rc<T>, Arc<T>];
137
138        bit_sized![= {$PTR_BITS * 2}; for &str];
139        bit_sized![<T> = {$PTR_BITS * 2}; for &[T], &mut [T]];
140
141        #[cfg(feature = "alloc")]
142        bit_sized![= {$PTR_BITS * 3}; for String];
143
144        #[cfg(feature = "alloc")]
145        bit_sized![= {$PTR_BITS * 3}; for GraphemeString];
146
147        #[cfg(feature = "alloc")]
148        bit_sized![<T> = {$PTR_BITS * 3};
149            for Vec<T>, LinkedList<T>, VecDeque<T>, BTreeSet<T>, BinaryHeap<T>];
150        #[cfg(feature = "std")]
151        bit_sized![<T> = {$PTR_BITS * 3}; for HashSet<T>, Mutex<T>];
152
153        // K, V
154        #[cfg(feature = "alloc")]
155        bit_sized![<K, V> = {$PTR_BITS * 3}; for BTreeMap<K, V>];
156        #[cfg(feature = "std")]
157        bit_sized![<K, V> = {$PTR_BITS * 3}; for HashMap<K, V>];
158    };
159
160    /* arrays */
161
162    (array = $bits:literal * len for T: $tsize:literal * len: $($len:literal),+) => {
163        $(
164        impl<T: $crate::BitSized<$tsize>> $crate::BitSized<{$bits*$len}> for [T; $len] {}
165        )+
166    };
167}
