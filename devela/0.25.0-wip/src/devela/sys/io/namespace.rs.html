15#[doc = crate::TAG_NAMESPACE!()]
16/// I/O-related operations.
17#[derive(Debug)]
18pub struct Io;
19
20/// # Methods available in `no_std`.
21#[rustfmt::skip]
22impl Io {
23    #[cfg(not(feature = "std"))]
24    /// Copies the entire contents of a reader into a writer.
25    ///
26    /// See <https://doc.rust-lang.org/std/io/fn.copy.html>.
27    ///
28    /// # Features
29    /// Makes use of the `unsafe_array` feature to employ [`MaybeUninit`].
30    pub fn copy<R, W, const LEN: usize>(reader: &mut R, writer: &mut W) -> IoResult<u64>
31    where R: IoRead + ?Sized, W: IoWrite + ?Sized { copy::<R, W, LEN>(reader, writer) }
32    #[cfg(feature = "std")]
33    /// Copies the entire contents of a reader into a writer.
34    ///
35    /// See `std::io::`[`copy`].
36    pub fn copy<R, W>(reader: &mut R, writer: &mut W) -> IoResult<u64>
37    where R: IoRead + ?Sized, W: IoWrite + ?Sized { copy(reader, writer) }
38
39    /// Creates a value that is always at EOF for reads, and ignores all data written.
40    ///
41    /// See `std::io::`[`empty`].
42    ///
43    /// [`empty`]: ::std::io::empty
44    #[must_use]
45    pub const fn empty() -> IoEmpty { empty() }
46
47    /// Creates an instance of a reader that infinitely repeats one byte.
48    ///
49    /// See `std::io::`[`repeat`].
50    ///
51    /// [`repeat`]: ::std::io::repeat
52    #[must_use]
53    pub const fn repeat(byte: u8) -> IoRepeat { repeat(byte) }
54}
55
56/// # Methods only available in `std`.
57#[rustfmt::skip]
58#[cfg(feature = "std")]
59#[cfg_attr(nightly_doc, doc(cfg(feature = "std")))]
60impl Io {
61    /// Reads all bytes from a reader into a new String.
62    ///
63    /// See `std::io::`[`read_to_string`].
64    pub fn read_to_string<R: IoRead>(reader: R) -> IoResult<String> { read_to_string(reader) }
65
66    /// Creates an instance of a writer which will successfully consume all data.
67    ///
68    /// See `std::io::`[`sink`].
69    pub const fn sink() -> IoSink { sink() }
70
71    /// Constructs a new handle to the standard error of the current process.
72    ///
73    /// See `std::io::`[`stderr`].
74    pub fn stderr() -> Stderr { stderr() }
75
76    /// Constructs a new handle to the standard input of the current process.
77    ///
78    /// See `std::io::`[`stdin`].
79    pub fn stdin() -> Stdin { stdin() }
80
81    /// Constructs a new handle to the standard output of the current process.
82    ///
83    /// See `std::io::`[`stdout`].
84    pub fn stdout() -> Stdout { stdout() }
85}
