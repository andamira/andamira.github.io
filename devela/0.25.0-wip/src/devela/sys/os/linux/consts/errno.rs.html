11/// Standard exit codes for Linux processes
12//
13// - /usr/include/sysexits.h
14#[derive(Debug)]
15pub struct LINUX_EXIT;
16
17/// Standard success/failure codes (0-2)
18impl LINUX_EXIT {
19    /// Successful termination.
20    pub const SUCCESS: c_int = 0;
21    //
22    /// Alias of [`SUCCESS`][Self::SUCCESS].
23    pub const OK: c_int = Self::SUCCESS;
24
25    /// General error.
26    pub const FAILURE: c_int = 1;
27    //
28    /// Alias of [`FAILURE`][Self::FAILURE].
29    pub const ERR: c_int = Self::FAILURE;
30
31    /// Misuse of shell builtins.
32    pub const MISUSE: c_int = 2;
33}
34/// `sysexits.h` codes (64-78)
35impl LINUX_EXIT {
36    //  Error numbers begin at EX__BASE to reduce the possibility of clashing
37    //  with other exit statuses that random programs may already return.
38    // pub const BASE: c_int = 64;
39
40    /// The command was used incorrectly
41    ///
42    /// E.g., with the wrong number of arguments, a bad flag, a bad syntax in a parameter…
43    pub const USAGE: c_int = 64;
44
45    /// The input data was incorrect in some way.
46    ///
47    /// This should only be used for user's data & not system files.
48    pub const DATAERR: c_int = 65;
49
50    /// An input file (not a system file) did not exist or was not readable.
51    ///
52    /// This could also include errors like "No message" to a mailer (if it cared to catch it).
53    pub const NOINPUT: c_int = 66;
54
55    /// The user specified did not exist.
56    ///
57    /// This might be used for mail addresses or remote logins.
58    pub const NOUSER: c_int = 67;
59
60    /// The host specified did not exist.
61    ///
62    /// This is used in mail addresses or network requests.
63    pub const NOHOST: c_int = 68;
64
65    /// A service is unavailable.
66    ///
67    /// This can occur if a support program or file does not exist. This can also be used as a
68    /// catchall message when something you wanted to do doesn't work, but you don't know why.
69    pub const UNAVAILABLE: c_int = 69; // service unavailable
70
71    /// An internal software error has been detected.
72    ///
73    /// This should be limited to non-operating system related errors as possible.
74    pub const SOFTWARE: c_int = 70; // internal software error
75
76    /// An operating system error has been detected.
77    ///
78    /// This is intended to be used for such things as "cannot fork", "cannot create pipe"…
79    /// It includes things like getuid returning a user that does not exist in the passwd file.
80    pub const OSERR: c_int = 71; // system error
81
82    /// Some system file (e.g., /etc/passwd, /etc/utmp, etc.)
83    /// does not exist, cannot be opened, or has some sort of error (e.g., syntax error).
84    pub const OSFILE: c_int = 72; // critical OS file missing
85
86    /// A (user specified) output file cannot be created.
87    pub const CANTCREAT: c_int = 73;
88
89    /// An error occurred while doing I/O on some file.
90    pub const IOERR: c_int = 74; // input/output error
91
92    /// Temporary failure, indicating something that is not really an error.
93    ///
94    /// In sendmail, this means that a mailer (e.g.) could not create a connection,
95    /// and the request should be reattempted later.
96    pub const TEMPFAIL: c_int = 75; // temp failure; user is invited to retry
97
98    /// The remote system returned something that was "not possible" during a protocol exchange.
99    pub const PROTOCOL: c_int = 76; // remote error in protocol
100
101    /// You did not have sufficient permission to perform the operation.
102    ///
103    /// This is not intended for file system problems, which should use `NOINPUT` or `CANTCREAT`,
104    /// but rather for higher level permissions.
105    pub const NOPERM: c_int = 77; // permission denied
106
107    /// Configuration error.
108    pub const CONFIG: c_int = 78; // configuration error
109
110    /// Maximum valid exit code.
111    pub const MAX: c_int = 255; // maximum listed value (everything should be modulo 256 anyway)
112}
113/// Special shell codes (126-128+)
114impl LINUX_EXIT {
115    /// Command invoked cannot execute.
116    pub const CANNOT_EXEC: c_int = 126;
117
118    /// Command not found.
119    pub const NOT_FOUND: c_int = 127;
120
121    /// Invalid argument to exit.
122    pub const INVALID_ARG: c_int = 128;
123
124    /// Termination by signal N.
125    pub const fn signal(n: c_int) -> c_int {
126        128 + n
127    }
128}
129/// Custom error codes.
130impl LINUX_EXIT {
131    /// Used when an invalid exit code would have been generated.
132    ///
133    /// This helps distinguish between explicit failures and coding errors.
134    pub const INTERNAL_ERROR: c_int = Self::MAX - 1;
135}

138/// [`Linux`][crate::Linux] `sys/errno.h` constants.
139//
140// - /usr/include/asm-generic/errno.h
141#[derive(Debug)]
142pub struct LINUX_ERRNO;
143#[allow(missing_docs)]
144impl LINUX_ERRNO {
145    /// “Operation not permitted.”
146    ///
147    /// Only the owner of the file (or other resource) or processes with special
148    /// privileges can perform the operation.
149    pub const EPERM: isize = 1;
150
151    /// “No such file or directory.”
152    ///
153    /// This is a “file doesn’t exist” error for ordinary files that are referenced
154    /// in contexts where they are expected to already exist.
155    pub const ENOENT: isize = 2;
156
157    /// “No such process.”
158    ///
159    /// No process matches the specified process ID.
160    pub const ESRCH: isize = 3;
161
162    /// “Interrupted system call.”
163    ///
164    /// An asynchronous signal occurred and prevented completion of the call.
165    /// When this happens, you should try the call again.
166    ///
167    /// You can choose to have functions resume after a signal that is handled,
168    /// rather than failing with `EINTR`.
169    pub const EINTR: isize = 4;
170
171    /// “Input/output error.”
172    ///
173    /// Usually used for physical read or write errors.
174    pub const EIO: isize = 5;
175
176    /// “No such device or address.”
177    ///
178    /// The system tried to use the device represented by a file you specified, and
179    /// it couldn’t find the device. This can mean that the device file was
180    /// installed incorrectly, or that the physical device is missing or not
181    /// correctly attached to the computer.
182    pub const ENXIO: isize = 6;
183
184    /// “Argument list too long.”
185    ///
186    /// Used when the arguments passed to a new program being executed with one of
187    /// the exec functions (see Executing a File) occupy too much memory space. This
188    /// condition never arises on GNU/Hurd systems.
189    pub const E2BIG: isize = 7;
190
191    /// “Exec format error.”
192    ///
193    /// Invalid executable file format. This condition is detected by the exec functions.
194    pub const ENOEXEC: isize = 8;
195
196    /// “Bad file descriptor.”
197    ///
198    /// For example, I/O on a descriptor that has been closed or reading from a
199    /// descriptor open only for writing (or vice versa).
200    pub const EBADF: isize = 9;
201
202    /// “No child processes.”
203    ///
204    /// This error happens on operations that are supposed to manipulate child
205    /// processes, when there aren’t any processes to manipulate.
206    pub const ECHILD: isize = 10;
207
208    /// “Resource temporarily unavailable.”
209    ///
210    /// The call might work if you try again later. The constant
211    /// [`EWOULDBLOCK`][Self::EWOULDBLOCK] is another name for `EAGAIN`.
212    pub const EAGAIN: isize = 11;
213
214    /// “Cannot allocate memory.”
215    ///
216    /// The system cannot allocate more virtual memory because its capacity is full.
217    pub const ENOMEM: isize = 12;
218
219    /// “Permission denied.”
220    ///
221    /// The file permissions do not allow the attempted operation.
222    pub const EACCES: isize = 13;
223
224    /// “Bad address.”
225    ///
226    /// An invalid pointer was detected. On GNU/Hurd systems, this error never
227    /// happens; you get a signal instead.
228    pub const EFAULT: isize = 14;
229
230    /// “Block device required.”
231    ///
232    /// A file that isn’t a block special file was given in a situation that
233    /// requires one. For example, trying to mount an ordinary file as a file system
234    /// in Unix gives this error.
235    pub const ENOTBLK: isize = 15;
236
237    /// “Device or resource busy.”
238    ///
239    /// A system resource that can’t be shared is already in use. For example, if
240    /// you try to delete a file that is the root of a currently mounted filesystem,
241    /// you get this error.
242    pub const EBUSY: isize = 16;
243
244    /// “File exists.”
245    ///
246    /// An existing file was specified in a context where it only makes sense to
247    /// specify a new file.
248    pub const EEXIST: isize = 17;
249
250    /// “Invalid cross-device link.”
251    ///
252    /// An attempt to make an improper link across file systems was detected. This
253    /// happens not only when you use link (see Hard Links) but also when you rename
254    /// a file with rename (see Renaming Files).
255    pub const EXDEV: isize = 18;
256
257    /// “No such device.”
258    ///
259    /// The wrong type of device was given to a function that expects a particular
260    /// sort of device.
261    pub const ENODEV: isize = 19;
262
263    /// “Not a directory.”
264    ///
265    /// A file that isn’t a directory was specified when a directory is required.
266    pub const ENOTDIR: isize = 20;
267
268    /// “Is a directory.”
269    ///
270    /// You cannot open a directory for writing, or create or remove hard links to it.
271    pub const EISDIR: isize = 21;
272
273    /// “Invalid argument.”
274    ///
275    /// This is used to indicate various kinds of problems with passing the wrong
276    /// argument to a library function.
277    pub const EINVAL: isize = 22;
278
279    /// “Too many open files in system.”
280    ///
281    /// There are too many distinct file openings in the entire system. Note that
282    /// any number of linked channels count as just one file opening; see Linked
283    /// Channels. This error never occurs on GNU/Hurd systems.
284    pub const ENFILE: isize = 23;
285
286    /// “Too many open files.”
287    ///
288    /// The current process has too many files open and can’t open any more.
289    /// Duplicate descriptors do count toward this limit.
290    pub const EMFILE: isize = 24;
291
292    /// “Inappropriate ioctl for device.”
293    ///
294    /// Inappropriate I/O control operation, such as trying to set terminal modes on
295    /// an ordinary file.
296    pub const ENOTTY: isize = 25;
297
298    /// “Text file busy.”
299    ///
300    /// An attempt to execute a file that is currently open for writing, or write to
301    /// a file that is currently being executed. Often using a debugger to run a
302    /// program is considered having it open for writing and will cause this error.
303    /// (The name stands for “text file busy”.) This is not an error on GNU/Hurd
304    /// systems; the text is copied as necessary.
305    pub const ETXTBSY: isize = 26;
306
307    /// “File too large.”
308    ///
309    /// The size of a file would be larger than allowed by the system.
310    pub const EFBIG: isize = 27;
311
312    /// “No space left on device.”
313    ///
314    /// Write operation on a file failed because the disk is full.
315    pub const ENOSPC: isize = 28;
316
317    /// “Illegal seek.”
318    ///
319    /// Invalid seek operation (such as on a pipe).
320    pub const ESPIPE: isize = 29;
321
322    /// “Read-only file system.”
323    ///
324    /// An attempt was made to modify something on a read-only file system.
325    pub const EROFS: isize = 30;
326
327    /// “Too many links.”
328    ///
329    /// The link count of a single file would become too large. rename can cause
330    /// this error if the file being renamed already has as many links as it can take.
331    pub const EMLINK: isize = 31;
332
333    /// “Broken pipe.”
334    ///
335    /// There is no process reading from the other end of a pipe. Every library
336    /// function that returns this error code also generates a `SIGPIPE` signal; this
337    /// signal terminates the program if not handled or blocked. Thus, your program
338    /// will never actually see `EPIPE` unless it has handled or blocked `SIGPIPE`.
339    pub const EPIPE: isize = 32;
340
341    /// “Numerical argument out of domain.”
342    ///
343    /// Used by mathematical functions when an argument value does not fall into the
344    /// domain over which the function is defined.
345    pub const EDOM: isize = 33;
346
347    /// “Numerical result out of range.”
348    ///
349    /// Used by mathematical functions when the result value is not representable
350    /// because of overflow or underflow.
351    pub const ERANGE: isize = 34;
352
353    /// “Operation would block.”
354    ///
355    /// In the GNU C Library, this is another name for [`EAGAIN`][Self::EAGAIN].
356    /// The values are always the same, on every operating system.
357    ///
358    /// C libraries in many older Unix systems have EWOULDBLOCK as a separate error code.
359    pub const EWOULDBLOCK: isize = Self::EAGAIN;
