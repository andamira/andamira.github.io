3//! Linux file-related constants.
4//!
5//! Defines:

17/// [`Linux`][crate::Linux] Standard file descriptor numbers.
18#[allow(non_camel_case_types)]
19#[derive(Debug)]
20pub struct LINUX_FILENO;
21impl LINUX_FILENO {
22    /// Standard input.
23    pub const STDIN: c_int = 0;
24    /// Standard output.
25    pub const STDOUT: c_int = 1;
26    /// Standard error.
27    pub const STDERR: c_int = 2;
28}

30/// [`Linux`][crate::Linux] file creation and status flags.
31///
32/// Used with `sys_open`, `sys_fcntl`, etc.
33//
34// - /usr/include/asm-generic/fcntl.h
35// - /usr/include/linux/fcntl.h
36#[derive(Debug)]
37pub struct LINUX_O_FLAGS;
38impl LINUX_O_FLAGS {
39    /* basic access */
40
41    /// Open for reading only.
42    pub const RDONLY: c_int = 0o0;
43    /// Open for writing only.
44    pub const WRONLY: c_int = 0o1;
45    /// Open for reading and writing.
46    pub const RDWR: c_int = 0o2;
47    /// Mask for access mode.
48    pub const ACCMODE: c_int = 0o3;
49
50    /* File creation and status flags */
51
52    /// Create file if it doesn't exist.
53    pub const CREAT: c_int = 0o100;
54    /// Fail if file exists (when used with CREAT).
55    pub const EXCL: c_int = 0o200;
56    /// Don't make this fd the controlling terminal (for devices).
57    pub const NOCTTY: c_int = 0o400;
58    /// Truncate file upon open.
59    pub const TRUNC: c_int = 0o1_000;
60    /// Append to file on each write.
61    pub const APPEND: c_int = 0o2_000;
62    /// Non-blocking mode.
63    pub const NONBLOCK: c_int = 0o4_000;
64    /// Synchronous I/O.
65    pub const SYNC: c_int = 0o4_010_000;
66    /// Synchronized I/O data integrity (wait for data writes only).
67    pub const DSYNC: c_int = 0o10_000;
68    /// Direct I/O (no buffering).
69    pub const DIRECT: c_int = 0o40_000;
70    /// Large file support.
71    pub const LARGEFILE: c_int = 0o100_000;
72    /// Don't follow symbolic links.
73    pub const NOFOLLOW: c_int = 0o200_000;
74    /// Create a directory if one doesn't exist.
75    pub const DIRECTORY: c_int = 0o400_000;
76    /// Don't update file access time on reads.
77    pub const NOATIME: c_int = 0o1_000_000;
78    /// Close on exec.
79    pub const CLOEXEC: c_int = 0o2_000_000;
80    /// Obtain fd for path resolution only (no file access).
81    pub const PATH: c_int = 0o10_000_000; // For openat2()
82
83    /* Special flags for tmpfile */
84
85    /// Create an unnamed temporary file (automatically deleted on close).
86    ///
87    /// Must be used with `DIRECTORY` to specify where to create it.
88    pub const TMPFILE: c_int = 0o20_000_000 | Self::DIRECTORY;
89
90    /* Async I/O (not available on all architectures) */
91
92    /// Enable signal-driven I/O (SIGIO on file activity).
93    pub const ASYNC: c_int = 0o20_000;
94
95    /* Additional flags from newer kernels */
96
97    /// Resolve path without escaping starting dir (openat2()).
98    pub const RESOLVE_BENEATH: c_int = 0o100_000_000;
99    /// Don't cross mount points during resolution.
100    pub const RESOLVE_NO_XDEV: c_int = 0o200_000_000;
101    /// Block magic-link traversal (procfs-style links).
102    pub const RESOLVE_NO_MAGICLINKS: c_int = 0o400_000_000;
103    /// Block all symlink traversal.
104    pub const RESOLVE_NO_SYMLINKS: c_int = 0o1_000_000_000;
105    /// Treat starting dir as root during resolution.
106    pub const RESOLVE_IN_ROOT: c_int = 0o2_000_000_000;
107}

109/// [`Linux`][crate::Linux] File mode and permission bits.
110//
111// Architecture independent
112// - /usr/include/linux/stat.h
113// - /usr/include/bits/stat.h (glibc)
114//
115#[derive(Debug)]
116pub struct LINUX_S_IFMT;
117impl LINUX_S_IFMT {
118    /// File type mask (S_IFMT in POSIX).
119    pub const TYPE_MASK: c_uint = 0o170_000;
120
121    /* File types */
122
123    /// Regular file (S_IFREG in POSIX).
124    pub const FILE: c_uint = 0o100_000;
125    /// Directory (S_IFDIR in POSIX).
126    pub const DIRECTORY: c_uint = 0o040_000;
127    /// Character device (S_IFCHR in POSIX).
128    pub const CHAR_DEVICE: c_uint = 0o020_000;
129    /// Block device (S_IFBLK in POSIX).
130    pub const BLOCK_DEVICE: c_uint = 0o060_000;
131    /// FIFO/pipe (S_IFIFO in POSIX).
132    pub const FIFO: c_uint = 0o010_000;
133    /// Symbolic link (S_IFLNK in POSIX).
134    pub const SYMLINK: c_uint = 0o120_000;
135    /// Socket (S_IFSOCK in POSIX).
136    pub const SOCKET: c_uint = 0o140_000;
137
138    /* Special mode bits */
139
140    /// Set user ID on execution (`S_ISUID` in POSIX).
141    pub const SET_UID: c_uint = 0o4_000;
142    /// Set group ID on execution (`S_ISGID` in POSIX).
143    pub const SET_GID: c_uint = 0o2_000;
144    /// Sticky bit - restricted deletion (`S_ISVTX` in POSIX).
145    pub const STICKY_BIT: c_uint = 0o1_000;
146
147    /* Permission bits */
148
149    /// User has read permission (`S_IRUSR` in POSIX).
150    pub const USER_READ: c_uint = 0o400;
151    /// User has write permission (`S_IWUSR` in POSIX).
152    pub const USER_WRITE: c_uint = 0o200;
153    /// User has execute permission (`S_IXUSR` in POSIX).
154    pub const USER_EXEC: c_uint = 0o100;
155
156    /// Group has read permission (`S_IRGRP` in POSIX).
157    pub const GROUP_READ: c_uint = 0o040;
158    /// Group has write permission (`S_IWGRP` in POSIX).
159    pub const GROUP_WRITE: c_uint = 0o020;
160    /// Group has execute permission (`S_IXGRP` in POSIX).
161    pub const GROUP_EXEC: c_uint = 0o010;
162
163    /// Others have read permission (`S_IROTH` in POSIX).
164    pub const OTHER_READ: c_uint = 0o004;
165    /// Others have write permission (`S_IWOTH` in POSIX).
166    pub const OTHER_WRITE: c_uint = 0o002;
167    /// Others have execute permission (`S_IXOTH` in POSIX).
168    pub const OTHER_EXEC: c_uint = 0o001;
169}

171/// [`Linux`][crate::Linux] Seek commands (for `lseek`).
172//
173// Architecture independent
174// - /usr/include/unistd.h
175// - /usr/include/bits/fcntl-linux.h
176#[derive(Debug)]
177pub struct LINUX_SEEK;
178impl LINUX_SEEK {
179    /// Seek from start of file.
180    pub const SET: c_int = 0;
181    /// Seek from current position.
182    pub const CUR: c_int = 1;
183    /// Seek from end of file.
184    pub const END: c_int = 2;
185
186    /// Seek to next data (for sparse files).
187    pub const DATA: c_int = 3;
188    /// Seek to next hole (for sparse files).
189    pub const HOLE: c_int = 4;
190
191    /// Max supported seek type.
192    pub const MAX: c_int = Self::HOLE;
193}

195/// [`Linux`][crate::Linux] File descriptor commands (for `fcntl`).
196//
197// - /usr/include/asm-generic/fcntl.h
198// - /usr/include/bits/fcntl-linux.h
199//
200// Possible arch-specific variations (F_GETLK, F_SETLK, F_SETLKW may have different numeric values)
201// - /usr/include/asm/fcntl.h
202#[derive(Debug)]
203pub struct LINUX_F_CMD;
204impl LINUX_F_CMD {
205    /// Duplicate file descriptor.
206    pub const DUPFD: c_int = 0;
207    /// Get file descriptor flags.
208    pub const GETFD: c_int = 1;
209    /// Set file descriptor flags.
210    pub const SETFD: c_int = 2;
211
212    /// Get file status flags.
213    pub const GETFL: c_int = 3;
214    /// Set file status flags.
215    pub const SETFL: c_int = 4;
216
217    /// Get record locking info.
218    pub const GETLK: c_int = 5;
219    /// Set record locking info.
220    pub const SETLK: c_int = 6;
221    /// Set record locking wait.
222    pub const SETLKW: c_int = 7;
223
224    /// Duplicate FD with CLOSE_ON_EXEC.
225    pub const DUPFD_CLOEXEC: c_int = 1024;
226
227    /// Get pipe buffer size.
228    pub const GETPIPE_SZ: c_int = 1032;
229    /// Set pipe buffer size.
230    pub const SETPIPE_SZ: c_int = 1031;
231
232    /// Add seals to file.
233    pub const ADD_SEALS: c_int = 1033;
234    /// Get seals from file.
235    pub const GET_SEALS: c_int = 1034;
236
237    /// Get owner's process ID.
238    pub const GETOWN: c_int = 9;
239    /// Set owner's process ID.
240    pub const SETOWN: c_int = 8;
241
242    /// Get SIGIO/SIGURG signals.
243    pub const GETSIG: c_int = 11;
244    /// Set SIGIO/SIGURG signals.
245    pub const SETSIG: c_int = 10;
246}
