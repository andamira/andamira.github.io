90/// # Software flow control
91/// ---
92impl LINUX_IOCTL {
93    /// Equivalent to tcflow(fd, arg).
94    ///
95    /// Argument: int arg
96    ///
97    /// See tcflow(3) for the argument values [`TCOOFF`][Self::TCOOFF],
98    /// [`TCOON`][Self::TCOON], [`TCIOFF`][Self::TCIOFF], [`TCION`][Self::TCION].
99    pub const TCXONC: c_ulong = 0x540A;
100
101    /// Suspends output.
102    pub const TCOOFF: c_int = 0;
103    /// Resumes suspended output.
104    pub const TCOON: c_int = 1;
105    /// Sends a STOP character to the terminal, to stop the terminal from sending
106    /// any further input.
107    pub const TCIOFF: c_int = 2;
108    /// Sends a START character to the terminal, to tell the terminal that it can
109    /// resume sending input.
110    pub const TCION: c_int = 3;
111}

260/// # Redirecting console output
261/// ---
262impl LINUX_IOCTL {
263    /// Redirect output that would have gone to `/dev/console` or `/dev/tty0` to
264    /// the given terminal.
265    ///
266    /// Argument: void
267    ///
268    /// If that was a pseudoterminal master, send it to the slave. Before Linux
269    /// 2.6.10, anybody can do this as long as the output was not redirected
270    /// yet; since Linux 2.6.10, only a process with the `CAP_SYS_ADMIN`
271    /// capability may do this. If output was redirected already, then EBUSY is
272    /// returned, but redirection can be stopped by using this ioctl with fd
273    /// pointing at /dev/console or /dev/tty0.
274    pub const TIOCCONS: c_ulong = 0x541D;
275}

277/// # Pseudoterminal ioctls
278/// ---
279impl LINUX_IOCTL {
280    /// Enable (when *argp is nonzero) or disable packet mode.
281    ///
282    /// Argument: const int *argp
283    ///
284    /// Can be applied to the master side of a pseudoterminal only (and will
285    /// return ENOTTY otherwise).
286    ///
287    /// In packet mode, each subsequent read(2) will return a packet that either
288    /// contains a single nonzero control byte, or has a single byte containing
289    /// zero ('\0') followed by data written on the slave side of the
290    /// pseudoterminal.
291    ///
292    /// If the first byte is not TIOCPKT_DATA (0), it is an OR of one or more of
293    /// the following bits:
294    ///
295    /// - `TIOCPKT_FLUSHREAD`    The read queue for the terminal is flushed.
296    /// - `TIOCPKT_FLUSHWRITE`   The write queue for the terminal is flushed.
297    /// - `TIOCPKT_STOP`         Output to the terminal is stopped.
298    /// - `TIOCPKT_START`        Output to the terminal is restarted.
299    /// - `TIOCPKT_DOSTOP`       The start and stop characters are ^S/^Q.
300    /// - `TIOCPKT_NOSTOP`       The start and stop characters are not ^S/^Q.
301    pub const TIOCPKT: c_ulong = 0x5420;
302
303    /// Set (if *argp is nonzero) or remove (if *argp is zero) the lock on the
304    /// pseudoterminal slave device.
305    ///
306    /// Argument: int *argp
307    pub const TIOCSPTLCK: c_ulong = 0x4004_5431;
308
309    /// Return the current packet mode setting in the integer pointed to by
310    /// argp.
311    ///
312    /// Argument: const int *argp
313    pub const TIOCGPKT: c_ulong = 0x8004_5438;
314
315    /// Place the current lock state of the pseudoterminal slave device in the
316    /// location pointed to by argp.
317    ///
318    /// Argument: int *argp
319    pub const TIOCGPTLCK: c_ulong = 0x8004_5439;
320
321    /// Open and return a new file descriptor that refers to the peer
322    /// pseudoterminal slave device.
323    ///
324    /// Argument: int flags
325    ///
326    /// Given a file descriptor in fd that refers to a pseudoterminal master,
327    /// open (with the given open(2)-style flags) and return a new file
328    /// descriptor that refers to the peer pseudoterminal slave device.
329    ///
330    /// This operation can be performed regardless of whether the pathname of
331    /// the slave device is accessible through the calling process's mount
332    /// namespace.
333    pub const TIOCGPTPEER: c_ulong = 0x5441;
334}

336/// # Controlling terminal
337/// ---
338impl LINUX_IOCTL {
339    /// Make the given terminal the controlling terminal of the calling process.
340    ///
341    /// Argument: int arg
342    ///
343    /// The calling process must be a session leader and not have a controlling
344    /// terminal already. For this case, arg should be specified as zero.
345    ///
346    /// If this terminal is already the controlling terminal of a different
347    /// session group, then the ioctl fails with `EPERM`, unless the caller has
348    /// the `CAP_SYS_ADMIN` capability and arg equals 1, in which case the
349    /// terminal is stolen, and all processes that had it as controlling
350    /// terminal lose it.
351    pub const TIOCSCTTY: c_ulong = 0x540E;
352
353    /// Give up this controlling terminal.
354    ///
355    /// Argument: void
356    ///
357    /// If the process was session leader, then send `SIGHUP` and `SIGCONT` to
358    /// the foreground process group and all processes in the current session
359    /// lose their controlling terminal.
360    pub const TIOCNOTTY: c_ulong = 0x5422;
361}

377/// # Sending a break
378/// ---
379impl LINUX_IOCTL {
380    /// Equivalent to tcsendbreak(fd, arg).
381    ///
382    /// Argument: int arg
383    ///
384    /// If the terminal is using asynchronous serial data transmission, and arg is
385    /// zero, then send a break (a stream of zero bits) for between 0.25 and 0.5
386    /// seconds.  If the terminal is not using asynchronous serial data
387    /// transmission, then either a break is sent, or the function returns without
388    /// doing anything.  When arg is nonzero, nobody knows what will happen.
389    pub const TCSBRK: c_ulong = 0x5409;
390
391    /// So-called "POSIX version" of [`TCSBRK`][Self::TCSBRK].
392    ///
393    /// It treats nonzero arg as a time interval measured in deciseconds, and does
394    /// nothing when the driver does not support breaks.
395    ///
396    /// Argument: int arg
397    pub const TCSBRKP: c_ulong = 0x5425;
398
399    /// Turn break on, that is, start sending zero bits.
400    ///
401    /// Argument: void
402    pub const TIOCSBRK: c_ulong = 0x5427;
403
404    /// Turn break off, that is, stop sending zero bits.
405    ///
406    /// Argument: void
407    pub const TIOCCBRK: c_ulong = 0x5428;
408}
