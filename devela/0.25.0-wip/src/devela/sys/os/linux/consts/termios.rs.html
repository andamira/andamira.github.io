31impl LINUX_TERMIOS_IFLAG {
32    /// Ignore BREAK condition on input.
33    pub const IGNBRK: c_uint = 0_000_001;
34
35    /// Signal interrupt on BREAK.
36    /// If [`IGNBRK`](Self::IGNBRK) is set, a BREAK is ignored.
37    ///
38    /// If it is not set but `BRKINT` is set, then a BREAK causes the input and
39    /// output queues to be flushed, and if the terminal is the controlling
40    /// terminal of a foreground process group, it will cause a `SIGINT` to be
41    /// sent to this foreground process group.
42    ///
43    /// When neither `IGNBRK` nor `BRKINT` are set, a BREAK reads as a null byte
44    /// ('\0'), except when [`PARMRK`](Self::PARMRK) is set, in which case it
45    /// reads as the sequence `\377` `\0` `\0`.
46    pub const BRKINT: c_uint = 0_000_002;
47
48    /// Ignore framing errors and parity errors.
49    pub const IGNPAR: c_uint = 0_000_004;
50
51    /// Mark parity and framing errors.
52    ///
53    /// If this bit is set, input bytes with parity or framing
54    /// errors are marked when passed to the program.
55    ///
56    /// This bit is meaningful only when [`INPCK`](Self::INPCK) is set and
57    /// [`IGNPAR`](Self::IGNPAR) is not set.
58    ///
59    /// The way erroneous bytes are marked is with two preceding
60    /// bytes, `\377` and `\0`.
61    ///
62    /// Thus, the program actually reads three bytes for one erroneous byte
63    /// received from the terminal.
64    ///
65    /// If a valid byte has the value `\377`, and [`ISTRIP`](Self::ISTRIP) is
66    /// not set, the program might confuse it with the prefix that marks a
67    /// parity error.
68    ///
69    /// Therefore, a valid byte `\377` is passed to the program as two bytes,
70    /// `\377` `\377`, in this case.
71    ///
72    /// If neither `IGNPAR` nor `PARMRK` is set, read a character with a parity
73    /// error or framing error as `\0`.
74    pub const PARMRK: c_uint = 0_000_010;
75
76    /// Enable input parity checking.
77    pub const INPCK: c_uint = 0_000_020;
78
79    /// Strip off eighth bit.
80    pub const ISTRIP: c_uint = 0_000_040;
81
82    /// Translate `NL` to `CR` on input.
83    pub const INLCR: c_uint = 0_000_100;
84
85    /// Ignore `CR` (carriage return) on input.
86    pub const IGNCR: c_uint = 0_000_200;
87
88    /// Translate CR to NL (carriage return to newline) on input
89    /// (unless [`IGNCR`](Self::IGNCR) is set).
90    pub const ICRNL: c_uint = 0_000_400;
91
92    /// Map uppercase characters to lowercase on input (not in POSIX).
93    pub const IUCLC: c_uint = 0_001_000;
94
95    /// Enable XON/XOFF flow control on output.
96    pub const IXON: c_uint = 0_002_000;
97
98    /// (XSI) Typing any character will restart stopped output.
99    ///
100    /// (The default is to allow just the START character to
101    /// restart output.)
102    pub const IXANY: c_uint = 0_004_000;
103
104    /// Enable XON/XOFF flow control on input.
105    pub const IXOFF: c_uint = 0_010_000;
106
107    /// Ring bell when input queue is full (not in POSIX).
108    ///
109    /// Linux does not implement this bit, and acts as if it is always set.
110    pub const IMAXBEL: c_uint = 0_020_000;
111
112    /// Input is UTF8. (not in POSIX)
113    ///
114    /// This allows character-erase to be correctly performed in cooked mode.
115    ///
116    /// (since Linux 2.6.4)
117    pub const IUTF8: c_uint = 0_040_000;
118}

120impl LINUX_TERMIOS_OFLAG {
121    /// Enable implementation-defined output processing.
122    pub const OPOST: c_uint = 0_000_001;
123
124    /// Map lowercase characters to uppercase on output. (not in POSIX)
125    pub const OLCUC: c_uint = 0_000_002;
126
127    /// (XSI) Map NL to CR-NL on output.
128    pub const ONLCR: c_uint = 0_000_004;
129
130    /// Map CR to NL on output.
131    pub const OCRNL: c_uint = 0_000_010;
132
133    /// Don't output CR at column 0.
134    pub const ONOCR: c_uint = 0_000_020;
135
136    /// The `NL` character is assumed to do the carriage-return
137    /// function; the kernel's idea of the current column is set
138    /// to 0 after both `NL` and `CR`.
139    pub const ONLRET: c_uint = 0_000_040;
140
141    /// Send fill characters for a delay, rather than using a timed delay.
142    pub const OFILL: c_uint = 0_000_100;
143
144    /// Fill character is ASCII `DEL` (0177).
145    ///
146    /// If unset, fill character is ASCII `NUL` ('\0'). (Not implemented on Linux.)
147    pub const OFDEL: c_uint = 0_000_200;
148
149    /// Newline delay mask.
150    /// Values are [`NL0`](Self::NL0) and [`NL1`](Self::NL1).
151    ///
152    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE` or `_XOPEN_SOURCE`]
153    pub const NLDLY: c_uint = 0_000_400;
154    /// `NLDLY`: Newline type 0_.
155    pub const NL0: c_uint = 0_000_000;
156    /// `NLDLY`: Newline type 1.
157    pub const NL1: c_uint = 0_000_400;
158
159    /// Carriage return delay mask. Values are [`CR0`](Self::CR0),
160    /// [`CR1`](Self::CR1), [`CR2`](Self::CR2), or [`CR3`](Self::CR3).
161    ///
162    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE` or `_XOPEN_SOURCE`]
163    pub const CRDLY: c_uint = 0_003_000;
164    /// `CRDLY`: Carriage-return delay type 0.
165    pub const CR0: c_uint = 0_000_000;
166    /// `CRDLY`: Carriage-return delay type 1.
167    pub const CR1: c_uint = 0_001_000;
168    /// `CRDLY`: Carriage-return delay type 2.
169    pub const CR2: c_uint = 0_002_000;
170    /// `CRDLY`: Carriage-return delay type 3.
171    pub const CR3: c_uint = 0_003_000;
172
173    /// Horizontal tab delay mask.
174    ///
175    /// Values are [`TAB0`](Self::TAB0), [`TAB1`](Self::TAB1),
176    /// [`TAB2`](Self::TAB2), [`TAB3`](Self::TAB3)
177    /// (or `XTABS`, but see the BUGS section).
178    ///
179    /// A value of `TAB3`, that is, `XTABS`, expands tabs to spaces
180    /// (with tab stops every eight columns).
181    ///
182    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE` or `_XOPEN_SOURCE`]
183    ///
184    /// ## BUGS
185    /// On the Alpha architecture before Linux 4.16 (and glibc before
186    /// glibc 2.28), the `XTABS` value was different from `TAB3` and it was
187    /// ignored by the `N_TTY` line discipline code of the terminal driver
188    /// as a result (because as it wasn't part of the `TABDLY` mask).
189    pub const TABDLY: c_uint = 0_014000;
190    /// `TABDLY`: Horizontal-tab delay type 0.
191    pub const TAB0: c_uint = 0_000_000;
192    /// `TABDLY`: Horizontal-tab delay type 1.
193    pub const TAB1: c_uint = 0_004_000;
194    /// `TABDLY`: Horizontal-tab delay type 2.
195    pub const TAB2: c_uint = 0_010_000;
196    /// `TABDLY`: Horizontal-tab delay type 3.
197    pub const TAB3: c_uint = 0_014_000;
198
199    /// Backspace delay mask.
200    /// Values are [`BS0`](Self::BS0) or [`BS1`](Self::BS1).
201    ///
202    /// (Has never been implemented.)
203    ///
204    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE` or `_XOPEN_SOURCE`]
205    pub const BSDLY: c_uint = 0_020_000;
206    /// Backspace-delay type 0.
207    pub const BS0: c_uint = 0_000_000;
208    /// Backspace-delay type 1.
209    pub const BS1: c_uint = 0_020_000;
210
211    /// Vertical tab delay mask.
212    /// Values are [`VT0`](Self::VT0) or [`VT1`](Self::VT1).
213    pub const VTDLY: c_uint = 0_040_000;
214    /// `VTDLY`: Vertical-tab delay type 0.
215    pub const VT0: c_uint = 0_000_000;
216    /// `VTDLY`: Vertical-tab delay type 1.
217    pub const VT1: c_uint = 0_040_000;
218
219    /// Form feed delay mask.
220    /// Values are [`FF0`](Self::FF0) or [`FF1`](Self::FF1).
221    ///
222    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE` or `_XOPEN_SOURCE`]
223    pub const FFDLY: c_uint = 0_100_000;
224    /// `FFDLY`: Form-feed delay type 0.
225    pub const FF0: c_uint = 0_000_000;
226    /// `FFDLY`: Form-feed delay type 1.
227    pub const FF1: c_uint = 0_100_000;
228
229    /// See: [`TAB3`](Self::TAB3)
230    pub const XTABS: c_uint = 0_014_000;
231}

233impl LINUX_TERMIOS_CFLAG {
234    /// Character size mask.
235    /// Values are [`CS5`](Self::CS5), [`CS6`](Self::CS5), [`CS7`](Self::CS5),
236    /// or [`CS8`](Self::CS5).
237    pub const CSIZE: c_uint = 0_000_060;
238    /// `CSIZE`: Character size mask type 5.
239    pub const CS5: c_uint = 0_000_000;
240    /// `CSIZE`: Character size mask type 6.
241    pub const CS6: c_uint = 0_000_020;
242    /// `CSIZE`: Character size mask type 7.
243    pub const CS7: c_uint = 0_000_040;
244    /// `CSIZE`: Character size mask type 8.
245    pub const CS8: c_uint = 0_000_060;
246
247    /// Set two stop bits, rather than one.
248    pub const CSTOPB: c_uint = 0_000_100;
249
250    /// Enable receiver.
251    pub const CREAD: c_uint = 0_000_200;
252
253    /// Enable parity generation on output and parity checking for input.
254    pub const PARENB: c_uint = 0_000_400;
255
256    /// If set, then parity for input and output is odd;
257    /// otherwise even parity is used.
258    pub const PARODD: c_uint = 0_001_000;
259
260    /// Lower modem control lines after last process closes the
261    /// device (hang up).
262    pub const HUPCL: c_uint = 0_002_000;
263
264    /// Ignore modem control lines.
265    pub const CLOCAL: c_uint = 0_004_000;
266
267    // (not in POSIX) Baud speed mask (4+1 bits).
268    // [requires _BSD_SOURCE or _SVID_SOURCE]
269    // CBAUD
270
271    // (not in POSIX) Extra baud speed mask (1 bit), included in
272    // CBAUD.  [requires _BSD_SOURCE or _SVID_SOURCE]
273    //
274    // (POSIX says that the baud speed is stored in the termios
275    // structure without specifying where precisely, and provides
276    // cfgetispeed() and cfsetispeed() for getting at it.  Some
277    // systems use bits selected by CBAUD in c_cflag, other
278    // systems use separate fields, for example, sg_ispeed and
279    // sg_ospeed.)
280    // CBAUDEX
281
282    // (not in POSIX) Block output from a noncurrent shell layer.
283    // For use by shl (shell layers).  (Not implemented on
284    // Linux.)
285    // LOBLK
286
287    // (not in POSIX) Mask for input speeds.  The values for the
288    // CIBAUD bits are the same as the values for the CBAUD bits,
289    // shifted left IBSHIFT bits.  [requires _BSD_SOURCE or
290    // _SVID_SOURCE] (Not implemented in glibc, supported on
291    // Linux via TCGET* and TCSET* ioctls; see ioctl_tty(2))
292    // CIBAUD
293
294    // (not in POSIX) Use "stick" (mark/space) parity (supported
295    // on certain serial devices): if PARODD is set, the parity
296    // bit is always 1; if PARODD is not set, then the parity bit
297    // is always 0.  [requires _BSD_SOURCE or _SVID_SOURCE]
298    // CMSPAR
299
300    // (not in POSIX) Enable RTS/CTS (hardware) flow control.
301    // [requires _BSD_SOURCE or _SVID_SOURCE]
302    // CRTSCTS
303}

305impl LINUX_TERMIOS_LFLAG {
306    /// Enable signals.
307    ///
308    /// When any of the characters `INTR`, `QUIT`, `SUSP`, or `DSUSP` are
309    /// received, generate the corresponding signal.
310    pub const ISIG: c_uint = 0_000_001;
311
312    /// Enable canonical mode (erase and kill processing).
313    pub const ICANON: c_uint = 0_000_002;
314
315    /// If ICANON is also set, terminal is uppercase only.
316    /// (not in POSIX; not supported under Linux)
317    ///
318    /// Input is converted to lowercase, except for characters preceded by `\`.
319    ///
320    /// On output, uppercase characters are preceded by `\` and lowercase characters
321    /// are converted to uppercase.
322    ///
323    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE` or `_XOPEN_SOURCE`]
324    pub const XCASE: c_uint = 0_000_004;
325
326    /// Echo input characters.
327    pub const ECHO: c_uint = 0_000_010;
328
329    /// If [`ICANON`](Self::ICANON) is also set, the `ERASE` character erases
330    /// the preceding input character, and `WERASE` erases the preceding word.
331    pub const ECHOE: c_uint = 0_000_020;
332
333    /// If [`ICANON`](Self::ICANON) is also set, the `KILL` character
334    /// erases the current line.
335    pub const ECHOK: c_uint = 0_000_040;
336
337    /// If [`ICANON`](Self::ICANON) is also set, echo the `NL` character
338    /// even if [`ECHO`](Self::ECHO) is not set.
339    pub const ECHONL: c_uint = 0_000_100;
340
341    /// Disable flushing the input and output queues when
342    /// generating signals for the `INT`, `QUIT`, and `SUSP` characters.
343    pub const NOFLSH: c_uint = 0_000_200;
344
345    /// Send the `SIGTTOU` signal to the process group of a background process
346    /// which tries to write to its controlling terminal.
347    pub const TOSTOP: c_uint = 0_000_400;
348
349    /// If [`ECHO`](Self::ECHO) is also set, terminal special characters
350    /// other than `TAB`, `NL`, `START`, and `STOP` are echoed as `^X`,
351    /// where `X` is the character with ASCII code `0x40` greater than
352    /// the special character.
353    ///
354    /// (not in POSIX)
355    ///
356    /// For example, character `0x08` (BS) is echoed as `^H`.
357    ///
358    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE`]
359    pub const ECHOCTL: c_uint = 0_001_000;
360
361    /// If [`ICANON`](Self::ICANON) and [`ECHO`](Self::ECHO) are also set,
362    /// characters are printed as they are being erased.
363    ///
364    /// (not in POSIX)
365    ///
366    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE`]
367    pub const ECHOPRT: c_uint = 0_002_000;
368
369    /// If [`ICANON`](Self::ICANON) is also set, `KILL` is echoed by erasing
370    /// each character on the line, as specified by [`ECHOE`](Self::ECHOE)
371    /// and [`ECHOPRT`](Self::ECHOPRT).
372    ///
373    /// (not in POSIX)
374    ///
375    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE`]
376    pub const ECHOKE: c_uint = 0_004_000;
377
378    // /// Echo only when a process is reading.
379    // ///
380    // /// (not in POSIX) (Not implemented on Linux)
381    // pub const DEFECHO: c_uint = ?;
382
383    /// Output is being flushed.
384    ///
385    /// This flag is toggled by typing the `DISCARD` character.
386    ///
387    /// (not in POSIX; not supported under Linux)
388    ///
389    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE`]
390    pub const FLUSHO: c_uint = 0_010_000;
391
392    /// All characters in the input queue are reprinted
393    /// when the next character is read.
394    ///
395    /// (not in POSIX; not supported under Linux)
396    ///
397    /// (bash(1) handles typeahead this way.)
398    ///
399    /// [requires `_BSD_SOURCE` or `_SVID_SOURCE`]
400    pub const PENDING: c_uint = 0_040_000;
401
402    /// Enable implementation-defined input processing.
403    ///
404    /// This flag, as well as [`ICANON`](Self::ICANON) must be enabled for
405    /// the special characters `EOL2`, `LNEXT`, `REPRINT`, `WERASE` to be
406    /// interpreted, and for the [`IUCLC`](LINUX_TERMIOS_IFLAG::IUCLC) flag
407    /// to be effective.
408    pub const IEXTEN: c_uint = 0_100_000;
409
410    /// enable "LINEMODE"; useful with high latency links
411    pub const EXTPROC: c_uint = 0_200_000;
