38/// # Read-related methods.

42    /// Checks if there is input available to read from stdin.
43    pub fn has_input() -> bool { Self::available_bytes().unwrap_or(0) > 0 }
44
45    /// Returns the number of bytes available to be read from *stdin*.
46    pub fn available_bytes() -> Result<usize> {
47        let mut n = 0;
48        let result = unsafe {
49            Linux::sys_ioctl(FILENO::STDIN, IOCTL::FIONREAD, &mut n as *mut i32 as *mut u8)
50        };
51        if result < 0 { Err(LinuxError::Sys(result)) } else { Ok(n as usize) }
52    }
53
54    /// Gets a single byte from *stdin*.
55    pub fn get_byte() -> Result<u8> {
56        let mut c = 0;
57        loop {
58            let n = unsafe { Linux::sys_read(FILENO::STDIN, &mut c as *mut u8, 1) };
59            if n < 0 { return Err(LinuxError::Sys(n)); }
60            if n == 1 { break; }
61        }
62        Ok(c)
63    }
64
65    /// Attempts to get a single byte from stdin without blocking.
66    ///
67    /// Checks if there is input available using `has_input`,
68    /// and if so, reads a byte using `get_byte`.
69    ///
70    /// # Returns
71    /// - `Some(u8)` if a byte is successfully read.
72    /// - `None` if no data is available.
73    pub fn try_get_byte() -> Option<u8> {
74        if Linux::has_input() { Linux::get_byte().ok() } else { None }
75    }
76
77    /// Pauses execution until receiving from *stdin* any `char` in the `list`.
78    pub fn pause_until_char(list: &[char]) {
79        loop { if list.contains(&Linux::get_dirty_char()) { break; } }
80    }
81
82    /// Gets a single `char` from *stdin*.
83    ///
84    /// # Returns
85    /// - `Ok(char)` if a valid UTF-8 character is read.
86    /// - `Err(LinuxError::InvalidUtf8)` if the bytes are not valid UTF-8.
87    /// - `Err(LinuxError::NoInput)` if the UTF-8 sequence is valid but empty.
88    pub fn get_char() -> Result<char> {
89        let bytes = Linux::get_utf8_bytes()?;
90        let s = unsafe { from_utf8_unchecked(&bytes) };
91        s.chars().next().ok_or(LinuxError::NoInput)
92    }
93
94    /// Gets either a single `char` from *stdin*, or the replacement character.
95    ///
96    /// If the bytes received doesn't form a valid unicode scalar then the
97    /// [replacement character (�)][char::REPLACEMENT_CHARACTER] will be returned.
98    pub fn get_dirty_char() -> char {
99        match Linux::get_utf8_bytes() {
100            Ok(bytes) => {
101                let s = unsafe { from_utf8_unchecked(&bytes) };
102                s.chars().next().unwrap()
103            }
104            Err(_) => char::REPLACEMENT_CHARACTER,
105        }
106    }
107
108    /// Gets a UTF-8 encoded byte sequence from *stdin* representing a `char`.
109    pub fn get_utf8_bytes() -> Result<[u8; 4]> {
110        let mut bytes = [0u8; 4];
111        let len;
112        // read the first byte to determine the length of the character
113        bytes[0] = Linux::get_byte()?;
114        if bytes[0] & 0x80 == 0 { return Ok([bytes[0], 0, 0, 0]); } // ASCII char
115        // IMPROVE: use text module functionality
116        else if bytes[0] & 0xE0 == 0xC0 { len = 2; }
117        else if bytes[0] & 0xF0 == 0xE0 { len = 3; }
118        else if bytes[0] & 0xF8 == 0xF0 { len = 4; }
119        else { return Err(LinuxError::InvalidUtf8) }
120        // read the remaining bytes of the character
121        for i in 1..len {
122            bytes[i as usize] = Linux::get_byte()?;
123            if bytes[i as usize] & 0xC0 != 0x80 { return Err(LinuxError::InvalidUtf8); }
124        }
125        Ok(bytes)
126    }

128    /// Reads all available bytes from stdin.
129    ///
130    /// # Returns
131    /// - `Ok(Vec<u8>)` containing the bytes read.
132    /// - `Err(isize)` if the read fails, returning the error code.
133    #[cfg(feature = "alloc")]
134    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
135    pub fn read_available_bytes() -> Result<Vec<u8>> {
136        let count = Linux::available_bytes()?;
137        if count == 0 { return Ok(Vec::new()); }
138        let mut buffer = crate::vec_![0u8; count];
139        let n = unsafe { Linux::sys_read(FILENO::STDIN, buffer.as_mut_ptr(), count) };
140        if n < 0 { Err(LinuxError::Sys(n)) } else { buffer.truncate(n as usize); Ok(buffer) }
141    }

143    /// Prompts the user for a string from *stdin*, backed by a `buffer`.
144    ///
145    /// # Examples
146    /// ```no_run
147    /// # use devela::Linux;
148    /// let mut name_buffer = [0_u8; 32];
149    /// let name: &str = Linux::prompt::<32>("Enter your name: ", &mut name_buffer).unwrap();
150    /// ```
151    pub fn prompt<'i, const CAP: usize>(text: &str, buffer: &'i mut [u8; CAP]) -> Result<&'i str> {
152        Linux::print(text)?; Linux::get_line(buffer)
153    }
154
155    /// Gets a string from *stdin* backed by a `buffer`, until a newline.
156    ///
157    /// # Examples
158    /// ```no_run
159    /// # use devela::Linux;
160    /// let mut buf = [0_u8; 32];
161    /// let name: &str = Linux::get_line::<32>(&mut buf).unwrap();
162    /// ```
163    pub fn get_line<const CAP: usize>(buffer: &mut [u8; CAP]) -> Result<&str> {
164        Linux::get_str(buffer, '\n')
165    }
166
167    /// Gets a string from *stdin* backed by a `buffer`, until the `stop` char is received.
168    ///
169    /// # Examples
170    /// ```no_run
171    /// # use devela::Linux;
172    /// let mut buf = [0_u8; 32];
173    /// let name: &str = Linux::get_str::<32>(&mut buf, '\n').unwrap();
174    /// ```
175    pub fn get_str<const CAP: usize>(buffer: &mut [u8; CAP], stop: char) -> Result<&str> {
176        let mut index = 0;
177        loop {
178            if let Ok(c) = Linux::get_char() {
179                let mut c_buf = [0; 4];
180                let c_str = c.encode_utf8(&mut c_buf);
181                if c == stop {
182                    break;
183                } else if index + c_str.len() <= CAP {
184                    // Linux::print(c_str)?;
185                    for &b in c_str.as_bytes() {
186                        buffer[index] = b;
187                        index += 1;
188                    }
189                }
190            }
191        }
192        Ok(unsafe { from_utf8_unchecked(&buffer[..index]) })
193    }

202    const STACK_BUFFER_LEN: usize = 512; // PIPE_BUF is typically 4096 but 512 is safe
203
204    /// Writes all bytes to a file descriptor, handling partial writes.
205    fn write_all(fd: c_int, mut buf: &[u8]) -> Result<()> {
206        while !buf.is_empty() {
207            let n = unsafe { Linux::sys_write(fd, buf.as_ptr(), buf.len()) };
208            if n < 0 { return Err(LinuxError::Sys(n)); }
209            buf = &buf[n as usize..];
210        }
211        Ok(())
212    }
213    /// Writes all bytes to stdout. Returns error on syscall failure.
214    fn write_all_unchecked(fd: c_int, mut buf: &[u8]) {
215        while !buf.is_empty() {
216            let n = unsafe { Linux::sys_write(fd, buf.as_ptr(), buf.len()) };
217            if n <= 0 { panic!("write failed with return value {n}"); }
218            buf = &buf[n as usize..];
219        }
220    }
221
222    /* public methods */
223
224    /// Writes string + newline to stdout. Optimized for small strings.
225    pub fn print(s: &str) -> Result<()> {
226        Self::write_all(FILENO::STDOUT, s.as_bytes())
227    }
228    /// Like `print`, but panics on failure instead of returning errors.
229    pub fn print_unchecked(s: &str) { Self::write_all_unchecked(FILENO::STDOUT, s.as_bytes()); }
230
231    /// Writes string + newline to stdout. Optimized for small strings.
232    pub fn println(s: &str) -> Result<()> {
233        if s.len() <= Self::STACK_BUFFER_LEN { // small string optimization:
234            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
235            let bytes = s.as_bytes();
236            buf[..bytes.len()].copy_from_slice(bytes);
237            buf[bytes.len()] = b'\n';
238            Self::write_all(FILENO::STDOUT, &buf[..bytes.len() + 1])
239        } else { // fallback for large strings:
240            Self::write_all(FILENO::STDOUT, s.as_bytes())?;
241            Self::write_all(FILENO::STDOUT, b"\n")
242        }
243    }
244    /// Like `println`, but panics on failure instead of returning errors.
245    pub fn println_unchecked(s: &str) {
246        if s.len() <= Self::STACK_BUFFER_LEN {
247            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
248            let bytes = s.as_bytes();
249            buf[..bytes.len()].copy_from_slice(bytes);
250            buf[bytes.len()] = b'\n';
251            Self::write_all_unchecked(FILENO::STDOUT, &buf[..bytes.len() + 1]);
252        } else {
253            Self::write_all_unchecked(FILENO::STDOUT, s.as_bytes());
254            Self::write_all_unchecked(FILENO::STDOUT, b"\n");
255        }
256    }
257    /// Ultra-fast stdout write. Panics if not all bytes are written at once.
258    /// Only use for small strings (<512 bytes).
259    pub fn print_unchecked_fast(s: &str) {
260        let bytes = s.as_bytes();
261        let n = unsafe { Linux::sys_write(FILENO::STDOUT, bytes.as_ptr(), bytes.len()) };
262        if n != bytes.len() as isize {
263            panic!("write failed with return value {n}");
264        }
265    }
266    /// Ultra-fast stdout write with newline. Panics if write isn't atomic.
267    /// Only use for small strings (<511 bytes).
268    pub fn println_unchecked_fast(s: &str) {
269        let bytes = s.as_bytes();
270        let mut buf = MaybeUninit::<[u8; Self::STACK_BUFFER_LEN]>::uninit();
271        // SAFETY: we're about to initialize all accessed bytes
272        unsafe {
273            let buf_ptr = buf.as_mut_ptr() as *mut u8;
274            core::ptr::copy_nonoverlapping(bytes.as_ptr(), buf_ptr, bytes.len());
275            *buf_ptr.add(bytes.len()) = b'\n'; // Add newline
276            let n = Linux::sys_write(FILENO::STDOUT, buf_ptr, bytes.len() + 1); // single syscall
277            if n != (bytes.len() + 1) as isize { panic!("write failed with return value {n}"); }
278        }
279    }
280    /// Writes bytes to stdout. Returns error on syscall failure.
281    pub fn print_bytes(b: &[u8]) -> Result<()> { Self::write_all(FILENO::STDOUT, b) }
282    /// Like `print_bytes`, but panics on failure instead of returning errors.
283    pub fn print_bytes_unchecked(b: &[u8]) { Self::write_all_unchecked(FILENO::STDOUT, b); }
284
285    /// Writes all bytes to stderr. Returns error on syscall failure.
286    pub fn eprint(s: &str) -> Result<()> { Self::write_all(FILENO::STDERR, s.as_bytes()) }
287    /// Writes string + newline to stderr. Optimized for small strings.
288    pub fn eprintln(s: &str) -> Result<()> {
289        if s.len() <= Self::STACK_BUFFER_LEN { // small string optimization:
290            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
291            let bytes = s.as_bytes();
292            buf[..bytes.len()].copy_from_slice(bytes);
293            buf[bytes.len()] = b'\n';
294            Self::write_all(FILENO::STDERR, &buf[..bytes.len() + 1])
295        } else { // fallback for large strings:
296            Self::write_all(FILENO::STDERR, s.as_bytes())?;
297            Self::write_all(FILENO::STDERR, b"\n")
298        }
299    }
300    /// Writes bytes to stderr. Returns error on syscall failure.
301    pub fn eprint_bytes(b: &[u8]) -> Result<()> {
302        Self::write_all(FILENO::STDERR, b)
303    }

369    /// Suspends execution of calling thread for the given `duration`.
370    pub fn sleep(duration: Duration) -> Result<()> {
371        let mut req = LinuxTimespec::with(duration);
372        let mut rem = LinuxTimespec::default();
373        loop {
374            let n = unsafe { Linux::sys_nanosleep(req.as_ptr(), rem.as_mut_ptr()) };
375            if n == 0 {
376                break Ok(()); // Success
377            } else if n == -ERRNO::EINTR {
378                req = rem; // Retry with remaining time
379                continue;
380            } else { // Actual error
381                return Err(LinuxError::Sys(n));
382            }
383        }
384    }
385    /// Suspends execution of calling thread for the given `milliseconds`.
386    pub fn sleep_ms(milliseconds: u64) -> Result<()> {
387        Linux::sleep(Duration::from_millis(milliseconds))
388    }

390    /// Returns the current process number.
391    pub fn getpid() -> i32 {
392        unsafe { Linux::sys_getpid() }
393    }

401    const LINUX_SIG_MAX: usize = 30;
402    /// A static array to match signals with handlers.
403    static LINUX_SIG_HANDLERS: [AtomicPtr<fn(i32)>; LINUX_SIG_MAX] = {
404        const INIT: AtomicPtr<fn(i32)> = AtomicPtr::new(Ptr::null_mut());
405        [INIT; LINUX_SIG_MAX]

409/// # Signaling-related methods.

413    /// Registers multiple signals using a handler function.
414    ///
415    /// # Notes
416    /// - Unknown flags in the `flags` slice are ignored, and a warning is printed.
417    ///
418    /// # Arguments
419    /// - `handler`: A function that will be called when one of the specified signals is received.
420    ///   The function takes a single argument for the signal number.
421    /// - `signals`: A slice of [`LINUX_SIGNAL`][super::LINUX_SIGNAL] values specifying the signals to handle.
422    /// - `flags`: An optional slice of [`LINUX_SIGACTION`][super::LINUX_SIGACTION] flags.
423    ///   If `None`, only the `SA_RESTORER` flag is used.
424    ///
425    /// # Notes
426    /// - The `SA_RESTORER` flag is always included.
427    /// - The `SA_SIGINFO` flag is ignored. Use [`sig_handler_info`][Self::sig_handler_info] instead.
428    /// - Unknown flags in the `flags` slice are ignored, and a warning is printed.
429    ///
430    /// # Examples
431    /// ```no_run
432    /// # #[cfg(feature = "std")] {
433    /// # use devela::{Linux, LINUX_SIGNAL, sleep4};
434    /// fn handler(sig: i32) {
435    ///    println!("\nsignal `{sig}` received! continuing. . .");
436    /// }
437    /// fn main() {
438    ///     println!("press Ctrl+C, or resize the terminal to catch the signals");
439    ///     Linux::sig_handler(handler, &[LINUX_SIGNAL::SIGINT, LINUX_SIGNAL::SIGWINCH], None);
440    ///     sleep4!(5);
441    ///     println!("bye");
442    /// }
443    /// # }
444    /// ```
445    pub fn sig_handler(handler: fn(i32), signals: &[i32], flags: Option<&[usize]>) {
446        extern "C" fn c_handler(sig: i32) {
447            if sig >= 0 && (sig as usize) < LINUX_SIG_MAX {
448                let handler = LINUX_SIG_HANDLERS[sig as usize].load(AtomicOrdering::SeqCst);
449                if !handler.is_null() {
450                    unsafe {
451                        let handler: fn(i32) = transmute(handler);
452                        handler(sig);
453                    }
454                }
455            }
456        }
457        // Use the restorer function defined in assembly.
458        unsafe extern "C" { safe fn __devela_linux_restore_rt(); }
459
460        // Start with the SA_RESTORER flag, as it is always required.
461        let mut combined_flags = SIGACTION::SA_RESTORER;
462        // Add additional flags if provided.
463        if let Some(provided_flags) = flags {
464            for &flag in provided_flags {
465                match flag { // Validate the flag to ensure it is one of the known values.
466                    SIGACTION::SA_NOCLDSTOP
467                    | SIGACTION::SA_NOCLDWAIT
468                    | SIGACTION::SA_NODEFER
469                    | SIGACTION::SA_ONSTACK
470                    | SIGACTION::SA_RESETHAND
471                    | SIGACTION::SA_RESTART => {
472                        combined_flags |= flag;
473                    }
474                    _ => {
475                        #[cfg(feature = "std")]
476                        eprintln!("Warning: Ignoring unknown flag: {flag:#x}");
477                    }
478                }
479            }
480        }
481        let mask = LinuxSigset::empty();
482        for &sig in signals {
483            if (1..=LINUX_SIG_MAX).contains(&(sig as usize)) { // make sure the signal is a valid number
484                // Store the handler in the appropriate slot
485                LINUX_SIG_HANDLERS[sig as usize].store(handler as *mut _, AtomicOrdering::SeqCst);
486                // Register the handler with the OS
487                let c_handler = c_handler as extern "C" fn(i32);
488                let sigaction = LinuxSigaction::new(c_handler,
489                    combined_flags, mask, Some(__devela_linux_restore_rt));
490                unsafe {
491                    let _ = Linux::sys_rt_sigaction(sig, &sigaction,
492                        Ptr::null_mut(), LinuxSigset::size());
493                }
494            }
495        }
496    }
497
498    /// Registers multiple signals using a handler function accepting additional signal information.
499    ///
500    /// # Arguments
501    /// - `handler`: A function that will be called when one of the specified signals is received.
502    ///   The function takes three arguments:
503    ///   - `i32`: The signal number.
504    ///   - `LinuxSiginfo`: Additional information about the signal.
505    ///   - `*mut c_void`: A pointer to the signal context (ucontext_t).
506    /// - `signals`: A slice of [`LINUX_SIGNAL`][super::LINUX_SIGNAL] values specifying the signals to handle.
507    /// - `flags`: An optional slice of [`LINUX_SIGACTION`][super::LINUX_SIGACTION] flags.
508    ///   If `None`, only the `SA_RESTORER` and `SA_SIGINFO` flags are used.
509    ///
510    /// # Notes
511    /// - The `SA_RESTORER` and `SA_SIGINFO` flags are always included.
512    /// - Unknown flags in the `flags` slice are ignored, and a warning is printed.
513    ///
514    /// # Examples
515    /// ```no_run
516    /// # use devela::{c_void, sleep4, Linux, LinuxSiginfo, LINUX_SIGNAL, LINUX_SIGACTION};
517    /// fn handler(sig: i32, info: LinuxSiginfo, _context: *mut c_void) {
518    ///     println!("Signal {} received from PID {}!", sig, info.pid());
519    /// }
520    /// fn main() {
521    ///     Linux::sig_handler_info(handler, &[LINUX_SIGNAL::SIGINT],
522    ///         Some(&[LINUX_SIGACTION::SA_RESETHAND, LINUX_SIGACTION::SA_RESTART]),
523    ///     );
524    ///     println!("press Ctrl+C, to trigger SIGINT");
525    ///     loop {
526    ///         println!("Waiting for signal...");
527    ///         sleep4!(1);
528    ///     }
529    /// }
530    /// ```
531    pub fn sig_handler_info(
532        handler: fn(i32, LinuxSiginfo, *mut c_void),
533        signals: &[i32],
534        flags: Option<&[usize]>,
535    ) {
536        // We store the given `handler` function in a static to be able to call it
537        // from the new extern function which can't capture its environment.
538        static HANDLER: AtomicPtr<fn(i32, LinuxSiginfo, *mut c_void)>
539            = AtomicPtr::new(Ptr::null_mut());
540        HANDLER.store(handler as *mut _, AtomicOrdering::SeqCst);
541
542        extern "C" fn c_handler_siginfo(sig: i32, info: LinuxSiginfo, context: *mut c_void) {
543            let handler = HANDLER.load(AtomicOrdering::SeqCst);
544            if !handler.is_null() {
545                #[expect(clippy::crosspointer_transmute)]
546                let handler = unsafe {
547                    transmute::<*mut fn(i32, LinuxSiginfo, *mut c_void),
548                        fn(i32, LinuxSiginfo, *mut c_void)>(handler)
549                };
550                handler(sig, info, context);
551            }
552        }
553        // Use the restorer function defined in assembly.
554        unsafe extern "C" { safe fn __devela_linux_restore_rt(); }
555
556        // Start with the SA_RESTORER and SA_SIGINFO flags, as they are always required.
557        let mut combined_flags = SIGACTION::SA_RESTORER | SIGACTION::SA_SIGINFO;
558        // Add additional flags if provided.
559        if let Some(provided_flags) = flags {
560            for &flag in provided_flags {
561                match flag {
562                    SIGACTION::SA_NOCLDSTOP
563                    | SIGACTION::SA_NOCLDWAIT
564                    | SIGACTION::SA_NODEFER
565                    | SIGACTION::SA_ONSTACK
566                    | SIGACTION::SA_RESETHAND
567                    | SIGACTION::SA_RESTART => {
568                        combined_flags |= flag;
569                    }
570                    _ => {
571                        #[cfg(feature = "std")]
572                        eprintln!("Warning: Ignoring unknown flag: {flag:#x}");
573                    }
574                }
575            }
576        }
577        let mask = LinuxSigset::empty();
578        let sigaction = LinuxSigaction::new_siginfo(
579            c_handler_siginfo,
580            combined_flags,
581            mask,
582            Some(__devela_linux_restore_rt),
583        );
584        for s in signals {
585            if (1..=31).contains(s) { // make sure the signal is a valid number
586                unsafe {
587                    let _ = Linux::sys_rt_sigaction(*s, &sigaction,
588                        Ptr::null_mut(), LinuxSigset::size());
589                }
590            }
591        }
592    }
593}

595/// Generates a `random_*` function for each given integer primitive
596#[cfg(all(feature = "unsafe_syscall", not(miri)))]
597macro_rules! impl_random_fns {
598    // $prim: the unsigned integer primitive
599    // $len:  the length of the primitive in bytes
600    ($($prim:ident : $len:literal),+) => { $crate::paste! { $(
601        #[doc = "Generates a random `" $prim "` value that may not be criptographically secure."]
602        pub fn [<random_ $prim>]() -> Result<$prim> {
603            #[cold] fn getrandom_failed_cold(n: isize) -> Result<$prim> { Err(LinuxError::Sys(n)) }
604
605            let (mut r, mut attempts) = ([0; $len], 0);
606            loop {
607                let n = unsafe { Linux::sys_getrandom(r.as_mut_ptr(), $len, Linux::RAND_FLAGS) };
608                if n == $len { break; } // ← hot path
609                else if n == -ERRNO::EAGAIN { // ←↓ cold paths
610                    is![!Linux::getrandom_try_again_cold(&mut attempts); break];
611                } else { return getrandom_failed_cold(n); } // n < 0
612            }
613            Ok($prim::from_ne_bytes(r))
614        }
615    )+ }};
616}

618/// # Randomness-related methods.
619///
620/// They make use of the `GRND_NONBLOCK` and `GRND_INSECURE` flags. So when the randomness
621/// source is not ready, instead of blocking it may return less secure data in linux >= 5.6
622/// or retry it a certain number of times, or even return 0 in some cases.
623#[rustfmt::skip]
624#[cfg(all(feature = "unsafe_syscall", not(miri)))]
625impl Linux {
626    impl_random_fns![u8:1, u16:2, u32:4, u64:8, u128:16];
627
628    /// Fills the given `buffer` with random bytes that may not be cryptographically secure.
629    ///
630    /// # Panics
631    /// Panics in debug if `buffer.len() > `[`isize::MAX`]
632    pub fn random_bytes(buffer: &mut [u8]) -> Result<()> {
633        #[cold] fn getrandom_failed_cold(n: isize) -> Result<()> { Err(LinuxError::Sys(n)) }
634
635        debug_assert![buffer.len() <= isize::MAX as usize];
636        let (mut attempts, mut offset) = (0, 0);
637        while offset < buffer.len() {
638            let n = unsafe { Linux::sys_getrandom(buffer[offset..].as_mut_ptr(),
639                buffer.len() - offset, Linux::RAND_FLAGS) };
640            if n == -ERRNO::EAGAIN { is![!Linux::getrandom_try_again_cold(&mut attempts); break]; }
641            else if n < 0 { return getrandom_failed_cold(n); } // ←↑ cold paths
642            else { offset += n as usize; } // ← hot path
643        }
644        Ok(())
645    }
646
647    // from `sys/random.h`
648    const GRND_NONBLOCK: c_uint = 0x0001;
649    // const GRND_RANDOM: isize = 0x0002;
650    const GRND_INSECURE: c_uint = 0x0004;
651    const RAND_FLAGS: c_uint = Linux::GRND_NONBLOCK | Linux::GRND_INSECURE;
652    const RAND_MAX_ATTEMPTS: usize = 15;
653    /// the cold path for trying again
654    #[cold] #[must_use]
655    fn getrandom_try_again_cold(attempts: &mut usize) -> bool {
656        // if *attempts >= Linux::RAND_MAX_ATTEMPTS { Linux::getrandom_failed_cold(); }
657        *attempts += 1;
658        *attempts <= Linux::RAND_MAX_ATTEMPTS
659    }
660}
