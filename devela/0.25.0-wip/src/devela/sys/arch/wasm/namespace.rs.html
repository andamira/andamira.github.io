17#[cfg(target_family = "wasm")]
18use ::core::arch::wasm32::{memory_grow, memory_size};
19
20#[doc = crate::TAG_NAMESPACE!()]
21/// A Wasm namespace.
22#[derive(Debug)]
23pub struct Wasm;
24
25impl Wasm {
26    /// Number of bytes per memory page.
27    pub const PAGE_BYTES: usize = 65_536;
28
29    /// Maximum possible memory pages.
30    pub const MAX_PAGES: usize = 65_536;
31
32    /// Maximum addressable memory in bytes.
33    pub const MAX_MEMORY: usize = Self::MAX_PAGES.saturating_mul(Self::PAGE_BYTES);
34
35    /// Returns `true` if the crate was compiled for a WebAssembly target.
36    ///
37    /// This is a **compile-time check**, not a runtime check.
38    ///
39    /// Equivalent to `cfg!(target_family = "wasm")`.
40    pub const fn is_wasm_target() -> bool {
41        cfg!(target_family = "wasm")
42    }
43
44    /// Checks if WebAssembly atomics are enabled.
45    ///
46    /// Returns `true` if it's been compiled with `RUSTFLAGS='-C target-feature=+atomics`.
47    ///
48    /// See <https://doc.rust-lang.org/core/arch/wasm32/index.html#atomics>.
49    pub const fn has_atomics() -> bool {
50        cfg!(target_feature = "atomics")
51    }
52
53    /// Checks if bulk memory operations are available.
54    ///
55    /// Returns `true` if it's been compiled with `RUSTFLAGS='-C target-feature=+bulk-memory`.
56    pub fn has_bulk_memory() -> bool {
57        cfg!(target_feature = "bulk-memory")
58    }
59
60    /// Checks if mutable-globals are available.
61    ///
62    /// Returns `true` if it's been compiled with `RUSTFLAGS='-C target-feature=+mutable-globals`.
63    pub fn has_mutable_globals() -> bool {
64        cfg!(target_feature = "mutable-globals")
65    }
66
67    /// Checks if SIMD is available.
68    ///
69    /// Returns `true` if it's been compiled with `RUSTFLAGS='-C target-feature=+simd128`.
70    ///
71    /// See <https://doc.rust-lang.org/core/arch/wasm32/index.html#simd>.
72    pub fn has_simd() -> bool {
73        cfg!(target_feature = "simd128")
74    }
75
76    /// Returns the starting address of the WASM heap (`__heap_base`).
77    ///
78    /// # Safety
79    /// - On non-WASM targets, this always returns 0.
80    /// - On WASM targets, the value comes from linker-defined symbol.
81    #[cfg(feature = "unsafe_layout")]
82    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_layout")))]
83    pub fn heap_base() -> usize {
84        #[cfg(target_family = "wasm")]
85        return &raw const __heap_base as usize;
86        #[cfg(not(target_family = "wasm"))]
87        0
88    }
89
90    /// Returns the current memory size in units of pages.
91    ///
92    /// On non-WASM platforms, always returns `0`.
93    ///
94    /// See `core::arch::wasm32::`[`memory_size`].
95    #[allow(rustdoc::broken_intra_doc_links, reason = "cross-platform")]
96    pub fn memory_pages() -> usize {
97        #[cfg(target_family = "wasm")]
98        return memory_size(0);
99        #[cfg(not(target_family = "wasm"))]
100        0
101    }
102
103    /// Returns the current memory size in bytes.
104    ///
105    /// On non-WASM platforms it always returns `0`.
106    pub fn memory_bytes() -> usize {
107        Self::memory_pages().saturating_mul(Self::PAGE_BYTES)
108    }
109
110    /// Attempts to grow the default linear memory by the specified `delta` of pages.
111    ///
112    /// If memory is successfully grown then the previous size of memory, in pages,
113    /// is returned. If memory cannot be grown then `usize::MAX` is returned.
114    ///
115    /// On non-WASM platforms it always returns `usize::MAX`.
116    ///
117    /// See `core::arch::wasm32::`[`memory_grow`].
118    #[inline(always)]
119    #[allow(unused_variables, rustdoc::broken_intra_doc_links, reason = "cross-platform")]
120    pub fn memory_grow(delta: usize) -> usize {
121        #[cfg(target_family = "wasm")]
122        return memory_grow(0, delta);
123        #[cfg(not(target_family = "wasm"))]
124        usize::MAX
125    }
126
127    /// Attempts to grow the default linear memory by the specified `delta` of pages.
128    ///
129    /// - Returns the previous size of memory on success, or `None` otherwise.
130    /// - On non-WASM platforms it always returns `None`.
131    #[allow(unused_variables, rustdoc::broken_intra_doc_links, reason = "cross-platform")]
132    pub fn memory_grow_checked(delta: usize) -> Option<usize> {
133        #[cfg(target_family = "wasm")]
134        {
135            let result = memory_grow(0, delta);
136            if result == usize::MAX { None } else { Some(result) }
137        }
138        #[cfg(not(target_family = "wasm"))]
139        None
140    }
141
142    /// Returns `true` if memory can still grow.
143    ///
144    /// On non-WASM platforms it always returns `false`.
145    pub fn memory_can_grow() -> bool {
146        #[cfg(target_family = "wasm")]
147        return Self::memory_pages() < Self::MAX_PAGES;
148        #[cfg(not(target_family = "wasm"))]
149        false
150    }
151
152    /// Returns the remaining available memory.
153    #[inline(always)]
154    pub fn remaining_memory() -> usize {
155        Self::MAX_MEMORY.saturating_sub(Self::memory_bytes())
156    }
157}
158
159#[cfg(not(feature = "safe_sys"))]
160#[cfg(feature = "unsafe_layout")]
161unsafe extern "C" {
162    /// Reference to the start of the WASM heap (&__heap_base).
163    ///
164    /// This symbol is provided by the LLVM linker and marks where dynamic memory begins.
165    static __heap_base: u8;
166}
