8/// Marker trait to prevent downstream implementations of the [`ExtVec`] trait.
9trait Sealed {}
10impl<T> Sealed for Vec<T> {}
11
12#[doc = crate::TAG_DATA_STRUCTURE!()]
13/// Extension trait providing additional methods for [`Vec`].
14///
15/// This trait is sealed and cannot be implemented for any other type.
16#[cfg_attr(nightly_doc, doc(notable_trait))]
17#[expect(private_bounds, reason = "Sealed")]
18pub trait ExtVec<T>: Sealed {
19    /* convert */
20
21    /// Converts `Vec<T>` to `Vec<U>` when `U` implements `From<T>`.
22    /// # Examples
23    /// ```
24    /// # use devela::ExtVec;
25    /// assert_eq![vec![1_u16, 2, 3], vec![1_u8, 2, 3].vec_into_vec::<u16>()];
26    /// assert_eq![vec![1_u16, 2, 3], vec![1_u8, 2, 3].vec_into_vec::<u16>()];
27    /// ```
28    #[must_use]
29    fn vec_into_vec<U>(self) -> Vec<U>
30    where
31        U: From<T>;
32
33    /// Tries to convert `Vec<T>` to `Vec<U>` when `U` implements `TryFrom<T>`.
34    /// # Examples
35    /// ```
36    /// # use devela::ExtVec;
37    /// assert_eq![Ok(vec![1_i32, 2, 3]), vec![1_i64, 2, 3].vec_try_into_vec()];
38    /// assert_eq![Ok(vec![1_i32, 2, 3]), vec![1_i64, 2, 3].vec_try_into_vec::<_, i32>()];
39    /// ```
40    fn vec_try_into_vec<E, U>(self) -> Result<Vec<U>, E>
41    where
42        U: TryFrom<T, Error = E>;
43}
44
45impl<T> ExtVec<T> for Vec<T> {
46    /* convert */
47
48    #[rustfmt::skip]
49    fn vec_into_vec<U>(self) -> Vec<U> where U: From<T> {
50        self.into_iter().map(|t| U::from(t)).collect::<Vec<_>>().into_iter().collect()
51    }
52    #[rustfmt::skip]
53    fn vec_try_into_vec<E, U>(self) -> Result<Vec<U>, E> where U: TryFrom<T, Error = E> {
54        self.into_iter()
55            // 1. Vec<Result<_>>:
56            .map(U::try_from)
57            .collect::<Vec<_>>()
58            // 2. Result<Vec<_>>:
59            .into_iter()
60            .collect::<Result<Vec<_>, _>>()
61    }
62}
