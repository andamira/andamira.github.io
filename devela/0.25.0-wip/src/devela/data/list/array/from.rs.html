3//! Defines [`ArrayFrom`].
4//
5
6/// Compile-time conversion from slice-like types to arrays.
7///
8/// # Features
9/// Uses `Ptr::copy_nonoverlapping` internally when unsafe operations are allowed.
10///
11/// # Panics
12/// When `N` does not match the actual length of the input.
13///
14/// # Example
15/// ```
16/// # use devela::ArrayFrom;
17/// let arr = ArrayFrom("hello").to_array::<5>();
18/// assert_eq!(arr, *b"hello");
19/// ```
20#[derive(Debug)]
21pub struct ArrayFrom<T>(pub T);
22
23#[rustfmt::skip]
24#[allow(clippy::len_without_is_empty)]
25mod impls {
26    use crate::{Slice, ArrayFrom};
27
28    // byte
29    impl ArrayFrom<u8> {
30        /// Returns 1 (length of single byte).
31        pub const fn len(self) -> usize { 1 }
32        /// Converts to `[u8; N]` (must have N == 1).
33        pub const fn to_array<const N: usize>(self) -> [u8; N] { Slice::to_array(&[self.0]) }
34    }
35    // byte array
36    impl<const L: usize> ArrayFrom<[u8; L]> {
37        /// Returns array length.
38        pub const fn len(&self) -> usize { L }
39        /// Converts to `[u8; N]` (must have N == len()).
40        pub const fn to_array<const N: usize>(&self) -> [u8; N] { Slice::to_array(&self.0) }
41    }
42    // byte array ref
43    impl<const L: usize> ArrayFrom<&[u8; L]> {
44        /// Returns L (array length).
45        pub const fn len(&self) -> usize { L }
46        /// Converts to `[u8; N]` (must have N == L).
47        pub const fn to_array<const N: usize>(&self) -> [u8; N] { Slice::to_array(self.0) }
48    }
49    // byte slice
50    impl ArrayFrom<&[u8]> {
51        /// Returns slice length.
52        pub const fn len(&self) -> usize { self.0.len() }
53        /// Converts to `[u8; N]` (must have N == len()).
54        pub const fn to_array<const N: usize>(&self) -> [u8; N] { Slice::to_array(self.0) }
55    }
56    // slice of byte slices
57    impl ArrayFrom<&[&[u8]]> {
58        /// Returns total length of all slices.
59        pub const fn len(&self) -> usize {
60            let parts = self.0; let mut sum = 0; let mut i = 0;
61            while i < parts.len() { sum += parts[i].len(); i += 1; }
62            sum
63        }
64        /// Concatenates into `[u8; N]` (must have N == len()).
65        pub const fn to_array<const N: usize>(&self) -> [u8; N] {
66            let (mut buf, mut pos, parts, mut i) = ([0; N], 0, self.0, 0);
67            while i < parts.len() {
68                let part = parts[i];
69                Slice::copy_array_at(&mut buf, part, pos);
70                pos += part.len();
71                i += 1;
72            }
73            assert!(pos == N);
74            buf
75        }
76    }
77    // string slice
78    impl ArrayFrom<&str> {
79        /// Returns byte length.
80        pub const fn len(&self) -> usize { self.0.len() }
81        /// Converts to `[u8; N]` (must have N == len()).
82        pub const fn to_array<const N: usize>(&self) -> [u8; N] {
83            Slice::to_array(self.0.as_bytes())
84        }
85    }
86    // slice of string slices
87    impl ArrayFrom<&[&str]> {
88        /// Returns total length of all string slices.
89        pub const fn len(&self) -> usize {
90            let (mut len, mut rem) = (0, self.0);
91            while let [first, rest @ ..] = rem {
92                len += first.len();
93                rem = rest;
94            }
95            len
96        }
97        /// Concatenates into `[u8; N]` (must have N == len()).
98        pub const fn to_array<const N: usize>(&self) -> [u8; N] {
99            let (mut buf, mut pos, mut i) = ([0; N], 0, 0);
100            while i < self.0.len() {
101                let bytes = self.0[i].as_bytes();
102                crate::Slice::<u8>::copy_array_at(&mut buf, bytes, pos);
103                pos += bytes.len();
104                i += 1;
105            }
106            assert!(pos == N);
107            buf
108        }
109    }
110}
