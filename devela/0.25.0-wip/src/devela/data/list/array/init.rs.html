16/// Initializes a [`[$T; $LEN]`][array] array in multiple ways.
17///
18/// # Arguments
19/// - `[$T; $LEN]`: the array's elements' type and length.
20/// - `$init`: a function with an `usize` argument that returns `$T`.
21/// - `$const_init`: a const fn with an `usize` argument that returns `$T: Copy`.
22/// - `$copiable`: an expression that returns an element of type `$T: Copy`.
23/// - `$clonable`: an expression that returns an element of type `$T: Clone`.
24/// - `$fsafe`: a feature that forbids the use of `unsafe` when enabled.
25/// - `$funsafe`: a feature that enables the use of `unsafe` when enabled.
26/// - `$intoiter`: an item that implements [`IntoIterator`].
27///
28/// # Examples
29/// ```
30/// # use devela::array_init;
31/// assert_eq![[2,4,6], array_init![safe_init [i32; 3], |n| (n as i32 + 1) * 2]];
32/// #[cfg(feature = "unsafe_array")]
33/// assert_eq![[3,6,9], array_init![unsafe_init [i32; 3], |n| (n as i32 + 1) * 3]];
34///
35/// assert_eq![[2,4,6], array_init![init [i32; 3], "safe", "unsafe_array",
36///     |n| (n as i32 + 1) * 2]];
37/// #[cfg(feature = "alloc")]
38/// assert_eq![Box::new([2,4,6]), array_init![init_heap [i32; 3], "safe", "unsafe_array",
39///     |n| (n as i32 + 1) * 2]];
40///
41/// assert_eq![[7,7,7], array_init![clone [i32; 3], "safe", "unsafe_array", 7]];
42/// assert_eq![[0,0,0], array_init![default [i32; 3], "safe", "unsafe_array"]];
43/// assert_eq![[4,5,6], array_init![iter [i32; 3], "safe", "unsafe_array", vec![4,5,6,7,8]]];
44/// assert_eq![[4,0,0], array_init![iter [i32; 3], "safe", "unsafe_array", vec![4]]];
45///
46/// const fn init(n: usize) -> i32 { (n as i32 + 1) * 4 }
47/// const ARRAY1: [i32; 3] = array_init![const_init [i32; 3], "safe", "unsafe_array", init, 0];
48/// assert_eq![[4, 8, 12], ARRAY1];
49///
50/// const ARRAY2: [i32; 3] = array_init![const_default [i32; 3]];
51/// assert_eq![[0, 0, 0], ARRAY2];
52/// ```
53///
54/// # Features
55/// The unsafe version uses [`MaybeUninit`][crate::MaybeUninit] in the case
56/// of stack allocation or [`Box::from_raw`] in the case of heap allocation.
57///
58/// For the `const_init`, `clone`, `default` and `iter` versions, if the given
59/// `$funsafe` is enabled and the given `$fsafe` is disabled, it will use unsafe
60/// initialization.
61// WAIT [array_repeat](https://github.com/rust-lang/rust/issues/126695)
62// WAIT [array_try_from_fn](https://github.com/rust-lang/rust/issues/89379)
63#[macro_export]
64#[cfg_attr(cargo_primary_package, doc(hidden))]
65macro_rules! array_init {
66    (
67    /* safe initializations */
68
69    // safe array initialization in the stack
70    safe_init [$T:ty; $LEN:expr], $init:expr) => {{
71        #[allow(clippy::redundant_closure_call, reason  = "macro arg isn't redundant")]
72        core::array::from_fn(|i| $init(i))
73    }};
74    (
75    // safe array initialization in the stack, compile-time friendly.
76    safe_const_init [$T:ty; $LEN:expr], $const_init:expr, $copiable:expr) => {{
77        let mut arr: [$T; $LEN] = [$copiable; $LEN];
78        let mut i = 0;
79        while i < $LEN {
80            // WAIT: [const_closures](https://github.com/rust-lang/rust/issues/106003)
81            arr[i] = $const_init(i);
82            i += 1;
83        }
84        arr
85    }};
86    (
87    // safe array initialization in the heap
88    safe_init_heap [$T:ty; $LEN:expr], $init:expr) => {{
89        let mut v = $crate::Vec::<$T>::with_capacity($LEN);
90        for i in 0..$LEN {
91            #[allow(clippy::redundant_closure_call, reason  = "macro arg isn't redundant")]
92            v.push($init(i));
93        }
94        v.into_boxed_slice().try_into().unwrap_or_else(|_| {
95            panic!("Can't turn the boxed slice into a boxed array")
96        })
97    }};
98    (
99
100    /* unsafe initializations */
101
102    // unsafe array initialization in the stack
103    unsafe_init [$T:ty; $LEN:expr], $init:expr) => {{
104        let mut arr: [$crate::MaybeUninit<$T>; $LEN] =
105            // SAFETY: array will be fully initialized in the subsequent loop
106            unsafe { $crate::MaybeUninit::uninit().assume_init() };
107        for (i, e) in &mut arr[..].iter_mut().enumerate() {
108            #[allow(clippy::redundant_closure_call, reason  = "macro arg isn't redundant")]
109            let _ = e.write($init(i)); // NOTE: const since 1.85
110        }
111        // Can't use transmute for now, have to use transmute_copy:
112        // - WAIT: [const generics transmute](https://github.com/rust-lang/rust/issues/61956)
113        //   - https://github.com/rust-lang/rust/issues/62875 (duplicate)
114        // unsafe { $crate::transmute::<_, [T; LEN]>(&arr) }
115        // SAFETY: we've initialized all the elements
116        unsafe { ::core::mem::transmute_copy::<_, [$T; $LEN]>(&arr) }
117    }};
118    (
119    // unsafe array initialization in the stack, compile-time friendly.
120    unsafe_const_init [$T:ty; $LEN:expr], $const_init:expr) => {{
121        // WAIT: [maybe_uninit_uninit_array](https://github.com/rust-lang/rust/issues/96097)
122        let mut arr: [$crate::MaybeUninit<$T>; $LEN] =
123            // SAFETY: array will be fully initialized in the subsequent loop
124            unsafe { $crate::MaybeUninit::uninit().assume_init() };
125        let mut i = 0;
126        while i < $LEN {
127            arr[i] = $crate::MaybeUninit::new($const_init(i));
128            i += 1;
129        }
130        // SAFETY: we've initialized all the elements
131        unsafe { ::core::mem::transmute_copy::<_, [$T; $LEN]>(&arr) }
132    }};
133    (
134    // unsafe array initialization in the heap
135    unsafe_init_heap [$T:ty; $LEN:expr], $init:expr) => {{
136        let mut v = $crate::Vec::<$T>::with_capacity($LEN);
137        #[allow(clippy::redundant_closure_call, reason  = "macro arg isn't redundant")]
138        for i in 0..$LEN { v.push($init(i)); }
139        let slice = v.into_boxed_slice();
140        let raw_slice = Box::into_raw(slice);
141        // SAFETY: pointer comes from using `into_raw`, and capacity is right.
142        unsafe { Box::from_raw(raw_slice as *mut [$T; $LEN]) }
143    }};
144    (
145
146    /* safety-agnostic initializations */
147
148    // initialize an array in the stack
149    init [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $init:expr) => {{
150        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
151        { $crate::array_init![safe_init [$T; $LEN], $init] }
152        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
153        { $crate::array_init![unsafe_init [$T; $LEN], $init] }
154    }};
155    (
156    // initialize an array the stack, compile-time friendly.
157    // $copiable is only used by the safe version as temporary placeholder.
158    const_init
159    [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $const_init:expr, $copiable:expr) => {{
160        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
161        { $crate::array_init![safe_const_init [$T; $LEN], $const_init, $copiable] }
162        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
163        { $crate::array_init![unsafe_const_init [$T; $LEN], $const_init ] }
164    }};
165    (
166    // initialize an array in the heap
167    init_heap [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $init:expr) => {{
168        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
169        { $crate::array_init![safe_init_heap [$T; $LEN], $init] }
170        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
171        { $crate::array_init![unsafe_init_heap [$T; $LEN], $init] }
172    }};
173    (
174
175    // initialize an array in the stack by cloning $clonable
176    clone [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $clonable:expr) => {{
177        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
178        { $crate::array_init![safe_init [$T; $LEN], |_| $clonable.clone()] }
179        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
180        { $crate::array_init![unsafe_init [$T; $LEN], |_| $clonable.clone()] }
181    }};
182    (
183    // initialize an array in the heap, by cloning $clonable
184    clone_heap [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $clonable:expr) => {{
185        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
186        { $crate::array_init![safe_init_heap [$T; $LEN], |_| $clonable.clone()] }
187        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
188        { $crate::array_init![unsafe_init_heap [$T; $LEN], |_| $clonable.clone()] }
189    }};
190    (
191
192    // initialize an array in the stack with $T: Default::default()
193    default [$T:ty; $LEN:expr]) => {{
194        { $crate::array_init![safe_init [$T; $LEN], |_| <$T>::default()] }
195    }};
196    (
197    // initialize an array in the stack with $T: Default::default()
198    default [$T:ty; $LEN:expr], unsafe: $funsafe:literal) => {{
199        #[cfg(not(feature = $funsafe))]
200        { $crate::array_init![safe_init [$T; $LEN], |_| <$T>::default()] }
201        #[cfg(feature = $funsafe)]
202        { $crate::array_init![unsafe_init [$T; $LEN], |_| <$T>::default()] }
203    }};
204    (
205    // initialize an array in the stack with $T: Default::default()
206    default [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal) => {{
207
208        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
209        { $crate::array_init![safe_init [$T; $LEN], |_| <$T>::default()] }
210        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
211        { $crate::array_init![unsafe_init [$T; $LEN], |_| <$T>::default()] }
212    }};
213    (
214    // initialize an array in the stack with $T: ConstDefault::DEFAULT
215    const_default [$T:ty; $LEN:expr]) => {{
216        [<$T as $crate::ConstDefault>::DEFAULT; $LEN]
217    }};
218    (
219    // initialize an array in the heap, with $T: Default::default()
220    default_heap [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal) => {{
221        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
222        { $crate::array_init![safe_init_heap [$T; $LEN], |_| <$T>::default()] }
223        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
224        { $crate::array_init![unsafe_init_heap [$T; $LEN], |_| <$T>::default()] }
225    }};
226    (
227
228    // initialize an array in the stack with an IntoIterator<Item = $T> and with
229    // $T::default() in case the iterator length is < $LEN, for the remaining elements.
230    iter [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $intoiter:expr) => {{
231        let mut iterator = $intoiter.into_iter();
232        let mut init_closure = |_| {
233            if let Some(e) = iterator.next() { e } else { <$T>::default() }
234        };
235        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
236        { $crate::array_init![safe_init [$T; $LEN], init_closure] }
237        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
238        { $crate::array_init![unsafe_init [$T; $LEN], init_closure] }
239    }};
240    (
241    // initialize an array in the heap with an IntoIterator<Item = $T> and with
242    // $T::default() in case the iterator length is < $LEN, for the remaining elements.
243    iter_heap [$T:ty; $LEN:expr], $fsafe:literal, $funsafe:literal, $intoiter:expr) => {{
244
245        let mut iterator = $intoiter.into_iter();
246        let mut init_closure = |_| {
247            if let Some(e) = iterator.next() { e } else { <$T>::default() }
248        };
249        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
250        { $crate::array_init![safe_init_heap [$T; $LEN], init_closure] }
251        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
252        { $crate::array_init![unsafe_init_heap [$T; $LEN], init_closure] }
253    }};
254    (
255
256    // initialize an array by applying $op (in safe mode it first clones $pre)
257    // and propagates errors both from $pre and $op.
258    preop [$T:ty; $LEN:expr]?, $fsafe:literal, $funsafe:literal, $pre:expr, $op:expr) => {{
259
260        #[cfg(any(feature = $fsafe, not(feature = $funsafe)))]
261        {
262            let init_value = $pre?; // error prone initial value
263            let mut arr: [$T; $LEN] = core::array::from_fn(|_| init_value.clone());
264            for (i, data) in $op.enumerate() {
265                arr[i] = data?;
266            }
267            arr
268        }
269        #[cfg(all(not(feature = $fsafe), feature = $funsafe))]
270        {
271            let mut arr: [$crate::MaybeUninit<$T>; $LEN] =
272                // SAFETY: array will be fully initialized in the subsequent loop
273                unsafe { $crate::MaybeUninit::uninit().assume_init() };
274            for (i, data) in $op.enumerate() {
275                arr[i].write(data?);
276            }
277            // SAFETY: we've initialized all the elements
278            unsafe { ::core::mem::transmute_copy::<_, [$T; $LEN]>(&arr) }
279        }
280    }};
281}
