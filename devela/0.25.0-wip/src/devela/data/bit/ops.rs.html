14/// Provides bitwise operations on `T`.
15///
16/// See also [`Bitwise`] for the related const wrapper.
17pub trait BitOps
18where
19    Self: Sized,
20{
21    /// The inner type for the bit representation.
22    type Inner;
23
24    /* new mask */
25
26    /// Returns a bitmask of ones from the `[start..=end]` range.
27    ///
28    /// Sets the rest of the bits to 0.
29    ///
30    /// # Panics
31    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
32    #[doc = include_str!("../../../libs/base/src/data/bit/benches/mask_range.md")]
33    #[must_use]
34    fn bit_mask_range(start: u32, end: u32) -> Self;
35
36    /// Returns a bitmask of ones from the `[start..=end]` checked range.
37    ///
38    /// Sets the rest of the bits to 0.
39    ///
40    /// # Errors
41    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
42    /// and [`MismatchedIndices`] if `start > end`.
43    #[doc = include_str!("../../../libs/base/src/data/bit/benches/mask_checked_range.md")]
44    fn bit_mask_checked_range(start: u32, end: u32) -> Result<Self, MismatchedBounds>;
45
46    /* get */
47
48    /// Gets the bits in `self` from the `[start..=end]` range.
49    ///
50    /// Sets the rest of the bits to 0.
51    /// # Panics
52    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
53    #[must_use]
54    fn bit_get_range(self, start: u32, end: u32) -> Self;
55
56    /// Gets the bits in `self` from the `[start..=end]` checked range.
57    ///
58    /// Sets the rest of the bits to 0.
59    /// # Errors
60    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
61    /// and [`MismatchedIndices`] if `start > end`.
62    fn bit_get_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds>;
63
64    /* get value */
65
66    /// Gets the rightwards shifted bits in `self` from the `[start..=end]` range.
67    ///
68    /// Sets the rest of the bits to 0.
69    ///
70    /// The bits in the specified range are shifted rightwards so that the least
71    /// significant bit (LSB) aligns with the units place, forming the integer value.
72    /// # Panics
73    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
74    #[must_use]
75    fn bit_get_value_range(self, start: u32, end: u32) -> Self;
76
77    /// Gets the rightwards shifted bits in `self` from the `[start..=end]` checked range.
78    ///
79    /// Sets the rest of the bits to 0.
80    ///
81    /// The bits in the specified range are shifted rightwards so that the least
82    /// significant bit (LSB) aligns with the units place, forming the integer value.
83    /// # Errors
84    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
85    /// and [`MismatchedIndices`] if `start > end`.
86    fn bit_get_value_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds>;
87
88    /* set */
89
90    /// Sets the bits in `self` to 1 from the `[start..=end]` range.
91    ///
92    /// Leaves the rest of the bits untouched.
93    /// # Panics
94    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
95    #[must_use]
96    fn bit_set_range(self, start: u32, end: u32) -> Self;
97
98    /// Sets the bits in `self` to 1 from the `[start..=end]` checked range.
99    ///
100    /// Leaves the rest of the bits untouched.
101    /// # Errors
102    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
103    /// and [`MismatchedIndices`] if `start > end`.
104    fn bit_set_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds>;
105
106    /* set value */
107
108    /// Sets the given `value` into the bits from the `[start..=end]` range.
109    ///
110    /// Leaves the rest of the bits unchanged.
111    ///
112    /// The value is first masked to fit the size of the range, and then
113    /// it is inserted into the specified bit range of `self`, replacing
114    /// the existing bits in that range. The rest of the bits in `self` remain unchanged.
115    /// # Panics
116    /// Panics if `start >= BITS || end >= BITS || start > end`.
117    fn bit_set_value_range(self, value: Self::Inner, start: u32, end: u32) -> Self;
118
119    /// Sets the given `value` into the bits from the `[start..=end]` checked range.
120    ///
121    /// Leaves the rest of the bits unchanged.
122    /// # Errors
123    /// Returns [`IndexOutOfBounds`] if `start >= BITS || end >= BITS`
124    /// and [`MismatchedIndices`] if `start > end`.
125    fn bit_set_value_checked_range(
126        self,
127        value: Self::Inner,
128        start: u32,
129        end: u32,
130    ) -> Result<Self, MismatchedBounds>;
131
132    /// Sets the given checked `value` into the bits from the `[start..=end]` checked range.
133    ///
134    /// Leaves the rest of the bits unchanged.
135    /// # Errors
136    /// Returns [`IndexOutOfBounds`] if `start >= BITS || end >= BITS`,
137    /// [`MismatchedIndices`] if `start > end` and
138    /// [`DataOverflow`] if `value` does not fit within the specified bit range.
139    fn bit_set_checked_value_checked_range(
140        self,
141        value: Self::Inner,
142        start: u32,
143        end: u32,
144    ) -> Result<Self, MismatchedBounds>;
145
146    /* unset */
147
148    /// Unsets the bits in `self` to 0 from the `[start..=end]` range.
149    ///
150    /// Leaves the rest of the bits untouched.
151    /// # Panics
152    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
153    #[must_use]
154    fn bit_unset_range(self, start: u32, end: u32) -> Self;
155
156    /// Unsets the bits in `self` to 0 from the `[start..=end]` checked range.
157    ///
158    /// Leaves the rest of the bits untouched.
159    /// # Errors
160    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
161    /// and [`MismatchedIndices`] if `start > end`.
162    fn bit_unset_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds>;
163
164    /* flip */
165
166    /// Flips the bits in `self` from the `[start..=end]` range.
167    ///
168    /// Leaves the rest of the bits untouched.
169    /// # Panics
170    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
171    #[must_use]
172    fn bit_flip_range(self, start: u32, end: u32) -> Self;
173
174    /// Flips the bits in `self` from the `[start..=end]` checked range.
175    ///
176    /// Leaves the rest of the bits untouched.
177    /// # Errors
178    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
179    /// and [`MismatchedIndices`] if `start > end`.
180    fn bit_flip_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds>;
181
182    /* reverse */
183
184    /// Reverses the order of the bits in `self` from the `[start..=end]` range.
185    ///
186    /// Leaves the rest of the bits untouched.
187    /// # Panics
188    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
189    #[must_use]
190    fn bit_reverse_range(self, start: u32, end: u32) -> Self;
191
192    /// Reverses the order of the bits in `self` from the `[start..=end]` checked range.
193    ///
194    /// Leaves the rest of the bits untouched.
195    /// # Errors
196    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
197    /// and [`MismatchedIndices`] if `start > end`.
198    fn bit_reverse_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds>;
199
200    /* count */
201
202    /// Counts the number of 1s in `bits` from the `[start..=end]` range.
203    /// # Panics
204    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
205    #[must_use]
206    fn bit_count_ones_range(self, start: u32, end: u32) -> u32;
207
208    /// Counts the number of 1s in `bits` from the `[start..=end]` checked range.
209    /// # Errors
210    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
211    /// and [`MismatchedIndices`] if `start > end`.
212    fn bit_count_ones_checked_range(self, start: u32, end: u32) -> Result<u32, MismatchedBounds>;
213
214    /// Counts the number of 0s in `bits` from the `[start..=end]` range.
215    /// # Panics
216    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
217    #[must_use]
218    fn bit_count_zeros_range(self, start: u32, end: u32) -> u32;
219
220    /// Counts the number of 0s in `bits` from the `[start..=end]` checked range.
221    /// # Errors
222    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
223    /// and [`MismatchedIndices`] if `start > end`.
224    fn bit_count_zeros_checked_range(self, start: u32, end: u32) -> Result<u32, MismatchedBounds>;
225
226    /* find first */
227
228    /// Finds the index of the first 1 in `bits` from the `[start..=end]` range.
229    ///
230    /// Returns `None` if there are no bits set.
231    ///
232    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
233    /// # Panics
234    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
235    #[must_use]
236    fn bit_find_first_one_range(self, start: u32, end: u32) -> Option<u32>;
237
238    /// Finds the index of the first 1 in `bits` from the `[start..=end]` checked range.
239    ///
240    /// Returns `None` if there are no bits set.
241    ///
242    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
243    /// # Errors
244    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
245    /// and [`MismatchedIndices`] if `start > end`.
246    fn bit_find_first_one_checked_range(
247        self,
248        start: u32,
249        end: u32,
250    ) -> Result<Option<u32>, MismatchedBounds>;
251
252    /// Finds the index of the first 0 in `bits` from the `[start..=end]` range.
253    ///
254    /// Returns `None` if there are no bits unset.
255    ///
256    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
257    /// # Panics
258    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
259    #[must_use]
260    fn bit_find_first_zero_range(self, start: u32, end: u32) -> Option<u32>;
261
262    /// Finds the index of the first 0 in `bits` from the `[start..=end]` checked range.
263    ///
264    /// Returns `None` if there are no bits unset.
265    ///
266    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
267    /// # Errors
268    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
269    /// and [`MismatchedIndices`] if `start > end`.
270    fn bit_find_first_zero_checked_range(
271        self,
272        start: u32,
273        end: u32,
274    ) -> Result<Option<u32>, MismatchedBounds>;
275
276    /* find last */
277
278    /// Finds the index of the last 1 in `bits` from the `[start..=end]` range.
279    ///
280    /// Returns `None` if there are no bits set.
281    ///
282    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
283    /// # Panics
284    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
285    #[must_use]
286    fn bit_find_last_one_range(self, start: u32, end: u32) -> Option<u32>;
287
288    /// Finds the index of the last 1 in `bits` from the `[start..=end]` checked range.
289    ///
290    /// Returns `None` if there are no bits set.
291    ///
292    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
293    /// # Errors
294    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
295    /// and [`MismatchedIndices`] if `start > end`.
296    fn bit_find_last_one_checked_range(
297        self,
298        start: u32,
299        end: u32,
300    ) -> Result<Option<u32>, MismatchedBounds>;
301
302    /// Finds the index of the last 0 in `bits` from the `[start..=end]` range.
303    ///
304    /// Returns `None` if there are no bits unset.
305    ///
306    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
307    /// # Panics
308    /// Panics in debug if `start >= Self::BITS` || `end >= Self::BITS` || `start > end`.
309    #[must_use]
310    fn bit_find_last_zero_range(self, start: u32, end: u32) -> Option<u32>;
311
312    /// Finds the index of the last 0 in `bits` from the `[start..=end]` checked range.
313    ///
314    /// Returns `None` if there are no bits unset.
315    ///
316    /// The index is relative to the entire sequence of `bits`, not to the given `start`.
317    /// # Errors
318    /// Returns [`IndexOutOfBounds`] if `start >= Self::BITS` || `end >= Self::BITS`
319    /// and [`MismatchedIndices`] if `start > end`.
320    fn bit_find_last_zero_checked_range(
321        self,
322        start: u32,
323        end: u32,
324    ) -> Result<Option<u32>, MismatchedBounds>;
325}

327macro_rules! impl_bit_ops {
328    () => {
329        impl_bit_ops![i8, i16, i32, i64, i128, isize, u8, u16, u32, u64, u128, usize];
330    };
331
332    // `$t`: the type to implement the trait for.
333    ($($t:ty),+) => { $( impl_bit_ops![@$t]; )+ };
334    (@$t:ty) => {
335        impl BitOps for $t {
336            type Inner = $t;
337
338            // new mask
339            fn bit_mask_range(start: u32, end: u32) -> Self {
340                Bitwise::<$t>::mask_range(start, end).0
341            }
342            fn bit_mask_checked_range(start: u32, end: u32) -> Result<Self, MismatchedBounds> {
343                Ok(Bitwise::<$t>::mask_checked_range(start, end)?.0)
344            }
345            // get
346            fn bit_get_range(self, start: u32, end: u32) -> Self {
347                Bitwise(self).get_range(start, end).0
348            }
349            fn bit_get_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds> {
350                Ok(Bitwise(self).get_checked_range(start, end)?.0)
351            }
352            // get value
353            fn bit_get_value_range(self, start: u32, end: u32) -> Self {
354                Bitwise(self).get_value_range(start, end).0
355            }
356            fn bit_get_value_checked_range(self, start: u32, end: u32)
357                -> Result<Self, MismatchedBounds> {
358                Ok(Bitwise(self).get_value_checked_range(start, end)?.0)
359            }
360            // set
361            fn bit_set_range(self, start: u32, end: u32) -> Self {
362                Bitwise(self).set_range(start, end).0
363            }
364            fn bit_set_checked_range(self, start: u32, end: u32) -> Result<Self, MismatchedBounds> {
365                Ok(Bitwise(self).set_checked_range(start, end)?.0)
366            }
367            // set value
368            fn bit_set_value_range(self, value: Self::Inner, start: u32, end: u32) -> Self {
369                Bitwise(self).set_value_range(value, start, end).0
370            }
371            fn bit_set_value_checked_range(self, value: Self::Inner, start: u32, end: u32)
372                -> Result<Self, MismatchedBounds> {
373                Ok(Bitwise(self).set_value_checked_range(value, start, end)?.0)
374            }
375            fn bit_set_checked_value_checked_range(self, value: Self::Inner, start: u32, end: u32)
376                -> Result<Self, MismatchedBounds> {
377                Ok(Bitwise(self).set_checked_value_checked_range(value, start, end)?.0)
378            }
379            // unset
380            fn bit_unset_range(self, start: u32, end: u32) -> Self {
381                Bitwise(self).unset_range(start, end).0
382            }
383            fn bit_unset_checked_range(self, start: u32, end: u32)
384                -> Result<Self, MismatchedBounds> {
385                Ok(Bitwise(self).unset_checked_range(start, end)?.0)
386            }
387            // flip
388            fn bit_flip_range(self, start: u32, end: u32) -> Self {
389                Bitwise(self).flip_range(start, end).0
390            }
391            fn bit_flip_checked_range(self, start: u32, end: u32)
392                -> Result<Self, MismatchedBounds> {
393                Ok(Bitwise(self).flip_checked_range(start, end)?.0)
394            }
395            // reverse
396            fn bit_reverse_range(self, start: u32, end: u32) -> Self {
397                Bitwise(self).reverse_range(start, end).0
398            }
399            fn bit_reverse_checked_range(self, start: u32, end: u32)
400                -> Result<Self, MismatchedBounds> {
401                Ok(Bitwise(self).reverse_checked_range(start, end)?.0)
402            }
403            // count
404            fn bit_count_ones_range(self, start: u32, end: u32) -> u32 {
405                Bitwise(self).count_ones_range(start, end)
406            }
407            fn bit_count_ones_checked_range(self, start: u32, end: u32)
408                -> Result<u32, MismatchedBounds> {
409                Bitwise(self).count_ones_checked_range(start, end)
410            }
411            fn bit_count_zeros_range(self, start: u32, end: u32) -> u32 {
412                Bitwise(self).count_zeros_range(start, end)
413            }
414            fn bit_count_zeros_checked_range(self, start: u32, end: u32)
415                -> Result<u32, MismatchedBounds> {
416                Bitwise(self).count_zeros_checked_range(start, end)
417            }
418            // find first
419            fn bit_find_first_one_range(self, start: u32, end: u32) -> Option<u32> {
420                Bitwise(self).find_first_one_range(start, end)
421            }
422            fn bit_find_first_one_checked_range(self, start: u32, end: u32)
423                -> Result<Option<u32>, MismatchedBounds> {
424                Bitwise(self).find_first_one_checked_range(start, end)
425            }
426            fn bit_find_first_zero_range(self, start: u32, end: u32) -> Option<u32> {
427                Bitwise(self).find_first_zero_range(start, end)
428            }
429            fn bit_find_first_zero_checked_range(self, start: u32, end: u32)
430                -> Result<Option<u32>, MismatchedBounds> {
431                Bitwise(self).find_first_zero_checked_range(start, end)
432            }
433            // find last

435                Bitwise(self).find_last_one_range(start, end)
436            }
437            fn bit_find_last_one_checked_range(self, start: u32, end: u32)

440            }
441            fn bit_find_last_zero_range(self, start: u32, end: u32) -> Option<u32> {

443            }
444            fn bit_find_last_zero_checked_range(self, start: u32, end: u32)
445                -> Result<Option<u32>, MismatchedBounds> {
446                Bitwise(self).find_last_zero_checked_range(start, end)
447            }
448        }
449    };
