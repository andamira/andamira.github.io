6use crate::{CodecLen, IoRead, IoResult, IoWrite, TAG_CODEC};
7
8#[doc = TAG_CODEC!()]
9/// A type that can be decoded from an I/O reader.
10///
11/// See also [`Encodable`].
12pub trait Decodable<R: IoRead>: Sized {
13    /// The type produced when decoding.
14    type Output;
15
16    /// Decodes `Self` from the given `reader`, returning the decoded value.
17    ///
18    /// # Errors
19    /// Returns an [`IoError`][crate::IoError] if decoding fails.
20    fn decode(reader: &mut R) -> IoResult<Self::Output>;
21}
22
23#[doc = TAG_CODEC!()]
24/// A type that can be encoded into an I/O writer.
25///
26/// See also [`Decodable`].
27#[doc = crate::_doc!(vendor: "encode")]
28pub trait Encodable<W: IoWrite> {
29    /// Encodes `self` into the given `writer`, returning the bytes written.
30    ///
31    /// # Errors
32    /// Returns [`IoError`][crate::IoError] if encoding fails.
33    fn encode(&self, writer: &mut W) -> IoResult<usize>;
34}
35
36#[doc = TAG_CODEC!()]
37/// A type that can compute the size of its encoded form without actual encoding.
38///
39/// This trait is automatically implemented for all types that implement
40/// [`Encodable`] with [`CodecLen`].
41///
42/// See [`CodecLen`] for details on length-based encoding.
43#[doc = crate::_doc!(vendor: "encode")]
44pub trait EncodableLen: Encodable<CodecLen> {
45    /// Computes the size of `self` when encoded.
46    ///
47    /// This method simulates encoding without writing any data,
48    /// allowing for preallocation or buffer sizing.
49    fn encoded_size(&self) -> IoResult<usize> {
50        let mut encoder = CodecLen::new();
51        self.encode(&mut encoder)?;
52        Ok(encoder.size())
53    }
54}
55impl<T: Encodable<CodecLen>> EncodableLen for T {}
