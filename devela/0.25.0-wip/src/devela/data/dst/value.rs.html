28#[doc = crate::TAG_DATA_STRUCTURE!()]
29/// A statically allocated <abbr title="Dynamically sized type">DST</abbr> value.
30///
31/// `DST` is the unsized type contained.
32/// `BUF` is the buffer used to hold the unsized type (both data and metadata).
33///
34/// # Examples
35/// ```
36/// # use {devela::data::{DstArray, DstValue}, core::fmt::Display};
37/// let val = DstValue::<dyn Display, DstArray<usize, 2>>::new(123456, |v| v as _)
38///     .expect("Insufficient size");
39/// assert_eq!( format!("{}", val), "123456" );
40/// ```
41pub struct DstValue<DST: ?Sized, BUF: DstBuf> {
42    _pd: marker::PhantomData<DST>,
43    // Data contains the object data first, then padding, then the pointer information
44    data: BUF,
45}
46
47impl<DST: ?Sized, BUF: DstBuf> DstValue<DST, BUF> {
48    /// Construct a stack-based DST.
49    ///
50    /// The closure `get_ref` must just convert `&VAL` to `&VAL`
51    /// (if the pointers don't match, an assertion triggers).
52    ///
53    /// Returns `Ok(dst)` if the allocation was successful, or `Err(val)` if it failed.
54    ///
55    /// # Examples
56    /// ```
57    /// # use {devela::data::{DstArray, DstValue}, core::fmt::Display};
58    /// let val = DstValue::<dyn Display, DstArray<usize, 2>>::new(1234, |v| v as _)
59    ///     .expect("Insufficient size");
60    /// assert_eq!( format!("{}", val), "1234" );
61    /// ```
62    pub fn new<VAL, F>(val: VAL, get_ref: F) -> Result<DstValue<DST, BUF>, VAL>
63    where
64        F: FnOnce(&VAL) -> &DST,
65        (VAL, BUF::Inner): MemAligned,
66        BUF: Default,
67    {
68        Self::in_buffer(BUF::default(), val, get_ref)
69    }
70
71    /// Construct a stack-based DST using the given `buffer`.
72    ///
73    /// See `new` for requirements on the `get_ref` closure.
74    ///
75    /// Returns `Ok(dst)` if the allocation was successful, or `Err(val)` if it failed.
76    ///
77    /// # Examples
78    /// ```
79    /// # use {devela::data::DstValue, core::{fmt::Display, mem::MaybeUninit}};
80    /// let val = DstValue::<dyn Display, _>::in_buffer([MaybeUninit::new(0u64); 2], 1234, |v| v)
81    ///     .expect("Insufficient size");
82    /// assert_eq!( format!("{}", val), "1234" );
83    /// ```
84    pub fn in_buffer<VAL, F: FnOnce(&VAL) -> &DST>(
85        buffer: BUF,
86        val: VAL,
87        get_ref: F,
88    ) -> Result<DstValue<DST, BUF>, VAL>
89    where
90        (VAL, BUF::Inner): MemAligned,
91    {
92        <(VAL, BUF::Inner) as MemAligned>::assert_compatibility();
93
94        let rv = unsafe {
95            let ptr: *const _ = check_fat_pointer(&val, get_ref);
96            let (raw_ptr, meta_len, meta) = decompose_pointer(ptr);
97
98            DstValue::new_raw(&meta[..meta_len], raw_ptr.cast_mut(), size_of::<VAL>(), buffer)
99        };
100        match rv {
101            Some(r) => {
102                // Prevent the destructor from running, now that we've copied it away
103                Mem::forget(val);
104                Ok(r)
105            }
106            None => Err(val),
107        }
108    }
109
110    /// Returns a new raw value.
111    ///
112    /// # Safety
113    /// `data` must point to `size` bytes, which shouldn't be freed if `Some` is returned.
114    pub unsafe fn new_raw(
115        info: &[usize],
116        data: *mut (),
117        size: usize,
118        mut buffer: BUF,
119    ) -> Option<DstValue<DST, BUF>> {
120        let req_words = BUF::round_to_words(size_of_val(info)) + BUF::round_to_words(size);
121        if buffer.extend(req_words).is_err() {
122            return None;
123        }
124
125        let mut rv =
126            ManuallyDrop::new(DstValue::<DST, BUF> { _pd: marker::PhantomData, data: buffer });
127        // SAFETY: caller must ensure safety
128        unsafe {
129            rv.write_value(data, size, info);
130        }
131        Some(ManuallyDrop::into_inner(rv))
132    }
133
134    /// Replace the contents without dropping the backing allocation
135    ///
136    /// # Examples
137    /// ```
138    /// # use {devela::data::{DstArray, DstValue}, core::fmt::Display};
139    /// let mut value = DstValue::<dyn Display, DstArray<usize, 2>>::new(1234, |v| v)
140    ///     .unwrap();
141    /// assert_eq!(format!("{}", value), "1234");
142    /// value.replace(1.234, |v| v).unwrap();
143    /// assert_eq!(format!("{}", value), "1.234");
144    /// ```
145    pub fn replace<VAL>(&mut self, val: VAL, get_ref: impl Fn(&VAL) -> &DST) -> Result<(), VAL>
146    where
147        (VAL, BUF::Inner): MemAligned,
148    {
149        <(VAL, BUF::Inner) as MemAligned>::assert_compatibility();
150
151        let size = size_of::<VAL>();
152        let (raw_ptr, meta_len, meta) = decompose_pointer(check_fat_pointer(&val, get_ref));
153        let info = &meta[..meta_len];
154
155        // Check size requirements (allow resizing)
156        let req_words = BUF::round_to_words(size_of_val(info)) + BUF::round_to_words(size);
157        if self.data.extend(req_words).is_err() {
158            return Err(val);
159        }
160        // If met, drop the existing item and move in the new item
161        unsafe {
162            ptr::drop_in_place::<DST>(&mut **self);
163            self.write_value(raw_ptr, size_of::<VAL>(), info);
164        }
165        Ok(())
166    }
167}

169/// # Specialisations for `str` (allowing storage of strings with single-byte alignment)
170impl<BUF: DstBuf> DstValue<str, BUF> {
171    /// Create a new empty string with a default buffer
172    pub fn empty_str() -> Result<Self, ()>
173    where
174        BUF: Default,
175    {
176        Self::empty_str_in_buffer(Default::default())
177    }
178
179    /// Create a new empty string with a provided buffer
180    pub fn empty_str_in_buffer(buffer: BUF) -> Result<Self, ()> {
181        let rv = unsafe {
182            let (raw_ptr, meta_len, meta) = decompose_pointer("");
183
184            DstValue::new_raw(&meta[..meta_len], raw_ptr.cast_mut(), 0, buffer)
185        };
186        match rv {
187            Some(r) => Ok(r),
188            None => Err(()),
189        }
190    }
191
192    /// Construct from a `str` using a default-constructed buffer
193    /// # Examples
194    /// ```
195    /// # use {devela::data::{DstArray, DstValue}, core::fmt::Display};
196    /// let val = DstValue::<str, DstArray<u8, 32>>::new_str("Hello, World")
197    ///     .expect("Insufficient size");
198    /// assert_eq!( &val[..], "Hello, World" );
199    /// ```
200    pub fn new_str(v: &str) -> Result<Self, &str>
201    where
202        BUF: Default,
203    {
204        Self::new_str_in_buffer(Default::default(), v)
205    }
206
207    /// Construct from a `str` using a provided buffer
208    ///
209    /// # Examples
210    /// ```
211    /// # use {devela::data::DstValue, core::{fmt::Display, mem::MaybeUninit}};
212    /// let val = DstValue::new_str_in_buffer([MaybeUninit::new(0u8); 32], "Hello, World")
213    ///     .expect("Insufficient size");
214    /// assert_eq!( &val[..], "Hello, World" );
215    /// ```
216    pub fn new_str_in_buffer(buffer: BUF, val: &str) -> Result<Self, &str> {
217        let rv = unsafe {
218            let (raw_ptr, meta_len, meta) = decompose_pointer(val);
219
220            DstValue::new_raw(&meta[..meta_len], raw_ptr.cast_mut(), size_of_val(val), buffer)
221        };
222        match rv {
223            Some(r) => Ok(r),
224            None => Err(val),
225        }
226    }
227
228    /// Add a string to the end of a string
229    ///
230    /// # Examples
231    /// ```
232    /// # use devela::data::{DstArray, DstValue};
233    /// let mut s = DstValue::<str, DstArray<usize, 8>>::new_str("Foo").unwrap();
234    /// s.append_str("Bar").unwrap();
235    /// assert_eq!(&s[..], "FooBar");
236    /// ```
237    pub fn append_str(&mut self, val: &str) -> Result<(), ()> {
238        let info_words = BUF::round_to_words(size_of::<usize>());
239
240        let ofs = self.len();
241
242        // Check/expand sufficient space
243        let req_words = BUF::round_to_words(ofs + val.len()) + info_words;
244        if self.data.extend(req_words).is_err() {
245            return Err(());
246        }
247
248        // Get the metadata slot
249        let data = self.data.as_mut();
250        let info_ofs = data.len() - info_words;
251
252        unsafe {
253            ptr::copy_nonoverlapping(
254                val.as_ptr(),
255                (data.as_mut_ptr() as *mut u8).add(ofs),
256                val.len(),
257            );
258            store_metadata(&mut data[info_ofs..], &[ofs + val.len()]);
259        }
260
261        Ok(())
262    }
263
264    /// Resize the string (discarding trailing data)
265    ///
266    /// # Examples
267    /// ```
268    /// # use devela::data::{DstArray, DstValue};
269    /// let mut s = DstValue::<str, DstArray<usize, 8>>::new_str("FooBar").unwrap();
270    /// s.truncate(3);
271    /// assert_eq!(&s[..], "Foo");
272    /// ```
273    pub fn truncate(&mut self, len: usize) {
274        if len < self.len() {
275            let _ = &self[..][len..]; // Index to force a panic if the index isn't char-aligned
276
277            let info_words = BUF::round_to_words(size_of::<usize>());
278            let data = self.data.as_mut();
279            let info_ofs = data.len() - info_words;
280            store_metadata(&mut data[info_ofs..], &[len]);
281        }
282    }
283}

285/// # Specialisation for slices (acting like an `ArrayVec`)
286impl<I, BUF: DstBuf> DstValue<[I], BUF>
287where
288    (I, BUF::Inner): MemAligned,
289{
290    /// Create a new zero-sized slice.
291    ///
292    /// # Errors
293    /// Will error only if the metadata doesn't fit.
294    pub fn empty_slice() -> Result<Self, ()>
295    where
296        BUF: Default,
297    {
298        Self::empty_slice_with_buffer(Default::default())
299    }
300    /// Create a new zero-sized slice in the provided buffer.
301    ///
302    /// # Errors
303    /// Will error only if the metadata doesn't fit.
304    pub fn empty_slice_with_buffer(mut buffer: BUF) -> Result<Self, ()> {
305        <(I, BUF::Inner) as MemAligned>::assert_compatibility();
306
307        let info_words = BUF::round_to_words(size_of::<usize>());
308        let req_words = info_words;
309        if buffer.extend(req_words).is_err() {
310            return Err(());
311        }
312        assert!(req_words <= buffer.as_ref().len());
313
314        let mut rv = DstValue { _pd: marker::PhantomData, data: buffer };
315
316        let data = rv.data.as_mut();
317        let info_ofs = data.len() - info_words;
318        let (_data_dst, info_dst) = data.split_at_mut(info_ofs);
319
320        store_metadata(info_dst, &[0]);
321        Ok(rv)
322    }
323
324    /// Append an item to the end of the slice (similar to `Vec::push`)
325    pub fn append(&mut self, v: I) -> Result<(), I> {
326        let info_words = BUF::round_to_words(size_of::<usize>());
327
328        let ofs = self.len();
329
330        // Check/expand sufficient space
331        let req_words = BUF::round_to_words((ofs + 1) * size_of::<I>()) + info_words;
332        if self.data.extend(req_words).is_err() {
333            return Err(v);
334        }
335        let data = self.data.as_mut();
336        assert!(req_words <= data.len());
337        // Write the new value
338        // SAFETY: Alignment is checked, pointer is in-bounds.
339        unsafe {
340            let data_ptr = (data.as_ptr() as *mut I).add(ofs);
341            ptr::write(data_ptr, v);
342        }
343        // Only update item count after the write
344        let info_ofs = data.len() - info_words;
345        store_metadata(&mut data[info_ofs..], &[ofs + 1]);
346
347        Ok(())
348    }
349
350    /// Inline append an item (See [`Self::append`]).
351    pub fn appended(mut self, v: I) -> Result<Self, (Self, I)> {
352        match self.append(v) {
353            Ok(()) => Ok(self),
354            Err(v) => Err((self, v)),
355        }
356    }
357
358    /// Extend a slice with an iterator.
359    pub fn extend<It: Iterator<Item = I>>(&mut self, mut iter: It) -> Result<(), (I, It)> {
360        while let Some(v) = iter.next() {
361            match self.append(v) {
362                Ok(()) => {}
363                Err(v) => return Err((v, iter)),
364            }
365        }
366        Ok(())
367    }
368    /// Helper to extend during construction (See [`Self::extend`]).
369    pub fn extended<It: Iterator<Item = I>>(mut self, iter: It) -> Result<Self, (Self, I, It)> {
370        match self.extend(iter) {
371            Ok(()) => Ok(self),
372            Err((v, iter)) => Err((self, v, iter)),
373        }
374    }
375
376    /// Remove the last item from the slice.
377    pub fn pop(&mut self) -> Option<I> {
378        if !self.is_empty() {
379            let ofs = self.len() - 1;
380            let data = self.data.as_mut();
381            let info_words = BUF::round_to_words(size_of::<usize>());
382            let info_ofs = data.len() - info_words;
383            unsafe {
384                store_metadata(&mut data[info_ofs..], &[ofs]);
385                Some(ptr::read((data.as_ptr() as *const I).add(ofs)))
386            }
387        } else {
388            None
389        }
390    }
391}

395impl<DST: ?Sized, BUF: DstBuf> DstValue<DST, BUF> {
396    unsafe fn write_value(&mut self, data: *const (), size: usize, info: &[usize]) {
397        let info_words = BUF::round_to_words(size_of_val(info));
398        let req_words = info_words + BUF::round_to_words(size);
399        let buf = self.data.as_mut();
400        assert!(req_words <= buf.len());
401
402        // Place pointer information at the end of the region
403        // - Allows the data to be at the start for alignment purposes
404        {
405            let info_ofs = buf.len() - info_words;
406            let info_dst = &mut buf[info_ofs..];
407            store_metadata(info_dst, info);
408        }
409
410        // SAFETY: caller must ensure safety
411        unsafe {
412            ptr::copy_nonoverlapping(data as *const u8, buf.as_mut_ptr() as *mut u8, size);
413        }
414    }
415
416    // Obtain raw pointer to the contained data
417    unsafe fn as_ptr(&self) -> *mut DST {
418        let data = self.data.as_ref();
419        let info_size = size_of::<*mut DST>() / size_of::<usize>() - 1;
420        let info_ofs = data.len() - BUF::round_to_words(info_size * size_of::<usize>());
421        let (data, meta) = data.split_at(info_ofs);
422        // SAFETY: caller must ensure safety
423        unsafe { super::make_fat_ptr(data.as_ptr() as *mut (), meta) }
424    }
425
426    // Obtain raw pointer to the contained data
427    unsafe fn as_ptr_mut(&mut self) -> *mut DST {
428        let data = self.data.as_mut();
429        let info_size = size_of::<*mut DST>() / size_of::<usize>() - 1;
430        let info_ofs = data.len() - BUF::round_to_words(info_size * size_of::<usize>());
431        let (data, meta) = data.split_at_mut(info_ofs);
432        // SAFETY: caller must ensure safety
433        unsafe { super::make_fat_ptr(data.as_mut_ptr() as *mut (), meta) }
434    }
435}
