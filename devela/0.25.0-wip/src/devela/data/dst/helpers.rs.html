8pub(crate) fn decompose_pointer<T: ?Sized>(mut ptr: *const T) -> (*const (), usize, [usize; 3]) {
9    let addr = ptr as *const ();
10    let rv = mem_as_slice(&mut ptr);
11    let mut vals = [0; 3];
12    assert!(rv[0] == addr as usize, "BUG: Pointer layout is not (data_ptr, info...)");
13    vals[..rv.len() - 1].copy_from_slice(&rv[1..]);
14    (addr, rv.len() - 1, vals)
15}

17pub(crate) const fn mem_as_slice<T>(ptr: &mut T) -> &mut [usize] {
18    assert!(size_of::<T>().is_multiple_of(size_of::<usize>()));
19    assert!(align_of::<T>().is_multiple_of(align_of::<usize>()));
20    let words = size_of::<T>() / size_of::<usize>();
21    // SAFETY: Points to valid memory (a raw pointer)
22    unsafe { slice::from_raw_parts_mut(ptr as *mut _ as *mut usize, words) }
23}

25/// Re-construct a fat pointer
26pub(crate) unsafe fn make_fat_ptr<T: ?Sized, W: MemPod>(
27    data_ptr: *mut (),
28    meta_vals: &BufSlice<W>,
29) -> *mut T {
30    #[repr(C)]
31    #[derive(Clone, Copy)]
32    struct Raw {
33        ptr: *const (),
34        meta: [usize; 4],
35    }
36    #[repr(C)]
37    union Inner<T: ?Sized> {
38        ptr: *mut T,
39        raw: Raw,
40    }
41    let mut rv = Inner { raw: Raw { ptr: data_ptr, meta: [0; 4] } };
42    assert!((meta_vals.len() * size_of::<W>()).is_multiple_of(size_of::<usize>()));
43    assert!(meta_vals.len() * size_of::<W>() <= 4 * size_of::<usize>());
44    // SAFETY: caller must ensure safety
45    unsafe {
46        ptr::copy(
47            meta_vals.as_ptr() as *const u8,
48            rv.raw.meta.as_mut_ptr() as *mut u8,
49            meta_vals.len() * size_of::<W>(),
50        );
51    }
52    // SAFETY: FIXME: miri complains
53    // WAIT:FIX: https://github.com/thepowersgang/stack_dst-rs/issues/14
54    let rv = unsafe { rv.ptr };
55    assert_eq!(rv as *const (), data_ptr.cast_const());
56    rv
57}

59/// Write metadata (abstraction around `ptr::copy`)
60pub(crate) fn store_metadata<W: MemPod>(dst: &mut BufSlice<W>, meta_words: &[usize]) {
61    let n_bytes = size_of_val(meta_words);
62    assert!(
63        n_bytes <= dst.len() * size_of::<W>(),
64        "nbytes [{}] <= dst.len() [{}] * sizeof [{}]",
65        n_bytes,
66        dst.len(),
67        size_of::<W>()
68    );
69    unsafe {
70        ptr::copy(meta_words.as_ptr() as *const u8, dst.as_mut_ptr() as *mut u8, n_bytes);
71    }
72}

74pub(crate) const fn round_to_words<T>(len: usize) -> usize {
75    len.div_ceil(size_of::<T>())

78/// Calls a provided function to get a fat pointer version of `v`
79/// (and checks that the returned pointer is sane)
80pub(crate) fn check_fat_pointer<U, T: ?Sized>(v: &U, get_ref: impl FnOnce(&U) -> &T) -> &T {
81    let ptr: &T = get_ref(v);
82    assert_eq!(
83        ptr as *const _ as *const u8, v as *const _ as *const u8,
84        "MISUSE: Closure returned different pointer"
85    );
86    assert_eq!(size_of_val(ptr), size_of::<U>(), "MISUSE: Closure returned a subset pointer");
87    ptr
88}

90/// Push items to a list using a generator function to get the items
91/// - `meta`  - Metadata slot (must be 1 usize long)
92/// - `data`  - Data slot, must be at least `count * sizeof(T)` long
93/// - `count` - Number of items to insert
94/// - `make`  - Generator function (is passed the current index)
95/// - `reset_slot` - A slot updated with `reset_value` when a panic happens before push is complete
96/// - `reset_value` - Value used in `reset_slot`
97///
98/// This provides a panic-safe push as long as `reset_slot` and `reset_value` undo the allocation operation
99pub(crate) unsafe fn list_push_gen<T, W: MemPod>(
100    meta: &mut BufSlice<W>,
101    data: &mut BufSlice<W>,
102    count: usize,
103    mut make: impl FnMut(usize) -> T,
104    reset_slot: &mut usize,
105    reset_value: usize,
106) {
107    /// Helper to drop/zero all pushed items, and reset data structure state if there's a panic
108    struct PanicState<'a, T>(*mut T, usize, &'a mut usize, usize);
109    impl<T> core::ops::Drop for PanicState<'_, T> {
110        fn drop(&mut self) {
111            if self.0.is_null() {
112                return;
113            }
114            // Reset the state of the data structure (leaking items)
115            *self.2 = self.3;
116            // Drop all partially-populated items
117            unsafe {
118                while self.1 != 0 {
119                    ptr::drop_in_place(&mut *self.0);
120                    ptr::write_bytes(self.0 as *mut u8, 0, size_of::<T>());
121                    self.0 = self.0.offset(1);
122                    self.1 -= 1;
123                }
124            }
125        }
126    }
127
128    let mut ptr = data.as_mut_ptr() as *mut T;
129    let mut clr = PanicState(ptr, 0, reset_slot, reset_value);
130    for i in 0..count {
131        let val = make(i);
132        // SAFETY: caller must ensure safety
133        unsafe {
134            ptr::write(ptr, val);
135        }
136        // SAFETY: caller must ensure safety
137        ptr = unsafe { ptr.offset(1) };
138        clr.1 += 1;
139    }
140    // Prevent drops and prevent reset
141    clr.0 = ptr::null_mut();
142    // Save the length once everything has been written
143    store_metadata(meta, &[count]);
144}
