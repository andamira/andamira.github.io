3//! A minimal single-threaded coroutine implementation.

5//! This code demonstrates a basic cooperative multitasking system where tasks

21/// Represents a single-thread stackless coroutine worker.
22///
23/// It has a private status that can be either running or halted.
24#[derive(Clone, Copy, Debug)]
25pub struct CoroWorker<T, E> {
26    status: CoroWorkerStatus,
27    result: OptRes<T, E>,
28}
29
30/// Private coroutine worker status.
31#[derive(Clone, Copy, Debug)]
32enum CoroWorkerStatus {
33    Halted,
34    Running,
35}
36
37impl<T, E> CoroWorker<T, E> {
38    /// Returns a new coroutine worker.
39    #[allow(unused)]
40    const fn new() -> Self {
41        CoroWorker { status: CoroWorkerStatus::Running, result: None }
42    }
43
44    /// Yields an [`Ok`] `value` and returns an awaitable `CoroWork`.
45    pub fn yield_ok(&mut self, value: T) -> CoroWork<'_, T, E> {
46        self.result = sok(value);
47        CoroWork { cor: self }
48    }
49
50    /// Yields an [`Err`] and returns an awaitable `CoroWork`.
51    pub fn yield_err(&mut self, error: E) -> CoroWork<'_, T, E> {
52        self.result = serr(error);
53        CoroWork { cor: self }
54    }
55}

57/* yielder */

59/// A future that alternates between [`Ready`][TaskPoll::Ready] and
60/// [`Pending`][TaskPoll::Pending] status each time it's polled.
61///
62/// This allows the coroutine to yield control back and be resumed later.
63#[derive(Debug)]
64pub struct CoroWork<'a, T, E> {
65    cor: &'a mut CoroWorker<T, E>,
66}
67
68impl<T, E> Future for CoroWork<'_, T, E> {
69    type Output = OptRes<T, E>;
70
71    fn poll(mut self: Pin<&mut Self>, _cx: &mut TaskContext) -> TaskPoll<OptRes<T, E>> {
72        match self.cor.status {
73            CoroWorkerStatus::Halted => {
74                self.cor.status = CoroWorkerStatus::Running;
75                if let Some(result) = self.cor.result.take() {
76                    match result {
77                        Err(error) => TaskPoll::Ready(serr(error)),
78                        Ok(value) => TaskPoll::Ready(sok(value)),
79                    }
80                } else {
81                    unreachable!();
82                }
83            }
84            CoroWorkerStatus::Running => {
85                self.cor.status = CoroWorkerStatus::Halted;
86                TaskPoll::Pending
87            }
88        }
89    }
90}

94/// A managed dynamic collection of single-thread [`CoroWorker`] coroutines.
95///
96/// It maintains a queue of coroutines in the stack, and runs them in a loop
97/// until they are all complete.
98///
99/// When a coroutine is polled and returns [`TaskPoll::Pending`], it is put back
100/// into the queue to be run again later. If it returns [`TaskPoll::Ready`]
101/// it is considered complete and is not put back into the queue.
102///
103/// # Examples
104/// ```
105#[doc = include_str!("../../../../examples/work/coro_manager.rs")]
106/// ```
107/// It outputs:
108/// ```text
109/// Running
110/// > instance 1 NEW
111/// > instance 2 NEW
112/// > instance 3 NEW
113/// > instance 4 NEW
114///   instance 1 A.0 Ok('a'))
115///   instance 2 A.0 Ok('a'))
116///   instance 3 A.0 Ok('a'))
117///   instance 1 B Ok('b')
118///   instance 2 B Ok('b')
119///   instance 3 B Ok('b')
120///   instance 1 A.1 Ok('a'))
121///   instance 2 A.1 Ok('a'))
122///   instance 3 A.1 Ok('a'))
123///   instance 4 BYE!
124///   instance 1 B Ok('b')
125///   instance 2 B Ok('b')
126///   instance 3 B Ok('b')
127///   instance 1 A.2 Ok('a'))
128///   instance 2 A.2 Ok('a'))
129///   instance 3 A.2 Ok('a'))
130///   instance 1 B Ok('b')
131///   instance 2 B Ok('b')
132///   instance 3 B Ok('b')
133///   instance 1 A.3 Ok('a'))
134///   instance 2 A.3 Ok('a'))
135///   instance 3 A.3 Ok('a'))
136///   instance 1 B Ok('b')
137///   instance 2 B Ok('b')
138///   instance 3 B Ok('b')
139/// Done
140/// ```
141#[derive(Default)]
142#[cfg(feature = "alloc")]
143#[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
144#[allow(missing_debug_implementations, reason = "unsatisified trait bounds")]
145pub struct CoroManager<T, E> {
146    #[allow(clippy::type_complexity)]
147    coros: VecDeque<Pin<Box<dyn Future<Output = OptRes<T, E>>>>>,
148}
149
150#[cfg(feature = "alloc")]
151impl<T, E: 'static + Debug> CoroManager<T, E> {
152    /// Returns a new empty manager.
153    pub fn new() -> Self {
154        CoroManager { coros: VecDeque::new() }
155    }
156
157    /// Adds a closure to the manager.
158    pub fn push<C, F>(&mut self, closure: C)
159    where
160        C: FnOnce(CoroWorker<T, E>) -> F,
161        F: Future<Output = OptRes<T, E>> + 'static,
162    {
163        self.coros.push_back(Box::pin(closure(CoroWorker::new())));
164    }
165
166    /// Runs all the coroutines to completion.
167    pub fn run(&mut self) {
168        let waker = TaskWaker::noop();
169        let mut context = TaskContext::from_waker(waker);
170
171        while let Some(mut cor) = self.coros.pop_front() {
172            let polled = cor.as_mut().poll(&mut context);
173            // println!("  coroutine polled:");
174
175            match polled {
176                TaskPoll::Pending => {
177                    // println!("  - pending, push back");
178                    self.coros.push_back(cor);
179                }
180                TaskPoll::Ready(_result) => {
181                    // println!("  - READY");
182                    // if let Some(Err(err)) = result {
183                    //     // eprintln!("    Error in coroutine: {:?}", err);
184                    // }
185                }
186            }
187        }
188    }
189}
