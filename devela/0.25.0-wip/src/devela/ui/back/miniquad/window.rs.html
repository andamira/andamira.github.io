29    /// Returns `true` if the event loop blocks until [`schedule_update`] is called.
30    ///
31    /// See also: `MiniquadService::`[`blocking_event_loop()`], and
32    /// `::miniquad::`[`blocking_event_loop`][blocking_event_loop].
33    ///
34    /// [`schedule_update`]: Self::schedule_update
35    /// [`blocking_event_loop()`]: crate::MiniquadService::blocking_event_loop
36    pub fn blocking_event_loop() -> bool { blocking_event_loop() }
37
38    /// Requests an immediate update, ensuring `update()` and `draw()` are called without waiting.
39    ///
40    /// This must be called from an implementor of [`MiniquadEventHandler`]
41    /// and requires [`conf.platform`]`.`[`blocking_event_loop`] to be enabled.
42    ///
43    /// This can significantly reduce CPU usage while waiting for events.
44    ///
45    /// It is recommended to call this at the end of `resize_event`
46    /// or after relevant mouse/keyboard input.
47    ///
48    /// [`MiniquadEventHandler`]: crate::MiniquadEventHandler
49    /// [`conf.platform`]: crate::MiniquadConf#structfield.platform
50    /// [`blocking_event_loop`]: crate::MiniquadPlatform#structfield.blocking_event_loop
51    pub fn schedule_update() { schedule_update(); }

88    /* quit */
89
90    /// This function simply quits the application without giving the user a chance to intervene.
91    ///
92    /// Usually this might be called when the user clicks the ‘Ok’ button in a
93    /// ‘Really Quit?’ dialog box Window might not be actually closed right away
94    /// (exit(0) might not happen in the order_quit implmentation) and execution
95    /// might continue for some time after But the window is going to be
96    /// inevitably closed at some point.
97    pub fn quit() { order_quit(); }
98
99    /// Triggers the “quit_requested_event” event.
100    ///
101    /// If the event handler callback does nothing, the application will be quit as usual.
102    /// To prevent this, call [`cancel_quit`][Self::cancel_quit] from inside the event handler.
103    pub fn request_quit() { request_quit(); }
104
105    /// Cancels a pending quit request, no matter who it was requested.
106    ///
107    /// The only place where calling this function makes sense is from inside the
108    /// event handler callback when the “quit_requested_event” event has been received.
109    pub fn cancel_quit() { cancel_quit(); }
