56            state: from.state,
57            mods: from.mods,

198    const fn parse_u8_from_bytes(bytes: &[u8]) -> Option<u8> {
199        let (mut num, mut index, mut found_digit) = (0u8, 0usize, false);
200        while index < bytes.len() {
201            let digit = bytes[index];
202            if digit < b'0' || digit > b'9' { break; } // Stop at first invalid character
203            let digit_value = digit - b'0';
204            if num > (u8::MAX / 10) || (num == u8::MAX / 10 && digit_value > u8::MAX % 10) {
205                return None;
206            }
207            num = num * 10 + digit_value;
208            found_digit = true;
209            index += 1;
210        }
211        if found_digit { Some(num) } else { None }
212    }

214    /// Attempts to construct a `KeyMod` from a JavaScript `KeyboardEvent`
215    /// physical [code] + a [location].
216    ///
217    /// [code]: https://developer.mozilla.org/en-US/docs/Web/API/KeyboardEvent/code
218    /// [location]: https://developer.mozilla.org/en-US/docs/Web/API/KeyboardEvent/location
219    pub const fn from_js_code(code: &str, location: WebKeyLocation) -> Option<Self> {
220        use KeyFfi as K;
221        match code.as_bytes() {
222            b"Backspace" => Some(K::Backspace),
223            b"Enter" => Some(K::Enter),
224            b"Tab" => Some(K::Tab),
225            b"Escape" => Some(K::Escape),
226            b"Space" => Some(K::Space),
227            //
228            b"ArrowLeft" => Some(K::Left),
229            b"ArrowRight" => Some(K::Right),
230            b"ArrowUp" => Some(K::Up),
231            b"ArrowDown" => Some(K::Down),
232            //
233            b"Home" => Some(K::Home),
234            b"End" => Some(K::End),
235            b"PageUp" => Some(K::PageUp),
236            b"PageDown" => Some(K::PageDown),
237            //
238            b"Delete" => Some(K::Delete),
239            b"Insert" => Some(K::Insert),
240            //
241            b"CapsLock" => Some(K::CapsLock),
242            b"ScrollLock" => Some(K::ScrollLock),
243            b"NumLock" => Some(K::NumLock),
244            //
245            b"PrintScreen" => Some(K::PrintScreen),
246            b"Pause" => Some(K::Pause),
247            b"ContextMenu" => Some(K::Menu),
248            _ => {
249                // Try converting from inner enums
250                if let Some(alpha) = KeyAlpha::from_js_code(code) {
251                    return Some(K::Alpha(alpha));
252                }
253                if let Some(pad) = KeyPad::from_js_code(code) {
254                    return Some(K::Pad(pad));
255                }
256                if let Some(media) = KeyMedia::from_js_code(code) {
257                    return Some(K::Media(media));
258                }
259                if let Some(mod_key) = KeyMod::from_js_code(code, location) {
260                    return Some(K::Mod(mod_key));
261                }
262                let cbytes = code.as_bytes();
263                // Check if the code starts with "F" (Function keys)
264                if let Some(first) = cbytes.first() {
265                    if *first == b'F' {
266                        if let Some(n) = Self::parse_u8_from_bytes(Slice::range_from(cbytes, 1)) {
267                            if n >= 1 && n <= 48 { return Some(K::F(n)); }
268                        }
269                    }
270                }
271                None
272            }
273        }
274    }
