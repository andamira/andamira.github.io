6/// Creates a `NonExtreme*` [`niche`][crate::num::niche] value with compile-time checking.
7///
8/// # Example
9/// ```
10/// # use devela::ne;
11/// let x = ne!(42u32); // NonExtremeU32
12/// // let y: NonExtremeI32 = ne!(20); // Would fail to compile (needs type suffix)
13/// ```
14#[macro_export]
15#[doc(hidden)]
16macro_rules! ne {
17    (
18    // Only the value (needs the type suffix to aid inference)
19    $num:expr) => {{ $crate::NewNicheHelper($num).to_non_extreme() }};
20    (
21    // Specify the type separately.
22    $num:expr, $T:ty) => {{ $crate::NewNicheHelper::<$T>($num).to_non_extreme() }};
23}
24#[doc(inline)]
25pub use ne;
26
27/// Creates a `NonZero*` [`niche`][crate::num::niche] value with compile-time checking.
28///
29/// # Example
30/// ```
31/// # use devela::nz;
32/// let x = nz!(42u32); // NonZeroU32
33/// // let y: NonZeroI32 = nz!(20); // Would fail to compile (needs type suffix)
34/// ```
35#[macro_export]
36#[doc(hidden)]
37macro_rules! nz {
38    (
39    // Only the value (needs the type suffix to aid inference)
40    $num:expr) => {{ $crate::NewNicheHelper($num).to_non_zero() }};
41    (
42    // Specify the type separately
43    $num:expr, $T:ty) => {{ $crate::NewNicheHelper::<$T>($num).to_non_zero() }};
44}
45#[doc(inline)]
46pub use nz;

50/// Private helper to construct niche types.
51#[doc(hidden)]
52#[derive(Debug)]
53pub struct NewNicheHelper<T>(pub T);
54macro_rules! impl_niche {
55    () => {
56        impl_niche!(U: u8, u16, u32, u64, u128, usize);
57        impl_niche!(I: i8, i16, i32, i64, i128, isize);
58    };
59    ($SIGN:ident: $($prim:ty),+) => { $crate::paste! {
60        $( impl_niche!(@ALL [<NonZero $prim:camel>], $prim); )+
61        $( impl_niche!(@$SIGN [<NonExtreme $prim:camel>], $prim); )+
62    }};
63    (@ALL $nz:ident, $prim:ty) => {
64        impl NewNicheHelper<$prim> {
65            pub const fn to_non_zero(self) -> $crate::$nz {
66                if self.0 == 0 { panic!["value must not be 0"]; }
67                else { $crate::$nz::new(self.0).unwrap() }
68            }
69        }
70    };
71    (@I $ne:ident, $prim:ty) => {
72        impl NewNicheHelper<$prim> {
73            pub const fn to_non_extreme(self) -> $crate::$ne {
74                if self.0 == <$prim>::MIN { panic!["value must not be MIN"]; }
75                else { $crate::$ne::new(self.0).unwrap() }
76            }
77        }
78    };
79    (@U $ne:ident, $prim:ty) => {
80        impl NewNicheHelper<$prim> {
81            pub const fn to_non_extreme(self) -> $crate::$ne {
82                if self.0 == <$prim>::MAX { panic!["value must not be MAX"]; }
83                else { $crate::$ne::new(self.0).unwrap() }
84            }
85        }
86    };
87}
88use impl_niche;
89impl_niche!();

91#[cfg(test)]
92crate::items! {
93    #[test]
94    fn test_nz() {
95        const NZI8: crate::NonZeroI8 = nz![23i8];
96        assert_eq![nz![20i32], crate::NonZeroI32::new(20).unwrap()];
97        assert_eq![nz![20u8], crate::NonZeroU8::new(20).unwrap()];
98        // alternative syntax
99        assert_eq![nz![20, i32], crate::NonZeroI32::new(20).unwrap()];
100        assert_eq![nz![20, u8], crate::NonZeroU8::new(20).unwrap()];
101    }
102    #[test]
103    #[should_panic(expected = "value must not be 0")]
104    // NOTE: using a runtime value so that should_panic catches the panic
105    fn test_nz_panic() { let zero = 0i64; let _panic = nz![zero]; }
106
107    #[test]
108    fn test_ne() {
109        const NEI8: crate::NonExtremeI8 = ne![23i8];
110        assert_eq![ne![20i8], crate::NonExtremeI8::new(20).unwrap()];
111        assert_eq![ne![20u8], crate::NonExtremeU8::new(20).unwrap()];
112        // alternative syntax
113        assert_eq![ne![20, i8], crate::NonExtremeI8::new(20).unwrap()];
114        assert_eq![ne![20, u8], crate::NonExtremeU8::new(20).unwrap()];
115    }
116    #[test]
117    #[should_panic(expected = "value must not be MIN")]
