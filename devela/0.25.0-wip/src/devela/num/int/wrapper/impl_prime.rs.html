19/// Implements prime-related methods for [`Int`].
20///
21/// # Args
22/// $t:   the input/output type
23/// $up:  the upcasted type to do the operations on (for prime_pi)
24/// $cap: the capability feature that enables the given implementation. E.g "_int_i8".
25///
26/// $d:   the doclink suffix for the method name
27macro_rules! impl_prime {
28    () => {
29        impl_prime![signed

42            u128  |u128     :"_int_u128"  | "-10",

54    @signed $t:ty | $up:ty : $cap:literal | $d:literal) => { paste! {
55        #[doc = crate::_doc_availability!(feature = $cap)]
56        ///
57        #[doc = "# Integer prime-related methods for `" $t "`\n\n"]
58        #[doc = "- [is_prime](#method.is_prime" $d ")"]

62        ///
63        #[cfg(feature = $cap )]
64        impl Int<$t> {
65            /// Returns `true` if `n` is prime.
66            ///
67            /// This approach uses optimized trial division, which means it checks
68            /// only odd numbers starting from 3 and up to the square root of the
69            /// given number. This is based on the fact that if a number is
70            /// divisible by a number larger than its square root, the result of the
71            /// division will be smaller than the square root, and it would have
72            /// already been checked in previous iterations.
73            /// # Examples
74            /// ```
75            /// # use devela::Int;
76            #[doc = "assert![Int(127_" $t ").is_prime()];"]
77            #[doc = "assert![Int(2_" $t ").is_prime()];"]
78            #[doc = "assert![!Int(1_" $t ").is_prime()];"]
79            #[doc = "assert![!Int(-2_" $t ").is_prime()];"]
80            /// ```
81            #[must_use]
82            pub const fn is_prime(self) -> bool {
83                match self.0 {
84                    ..=1 =>  false,
85                    2..=3 => true,
86                    _ => {
87                        is![self.0 % 2 == 0; return false];
88                        let limit = is![let Ok(s) = self.sqrt_floor(); s.0; unreachable!()];
89                        let mut i = 3;
90                        while i <= limit { is![self.0 % i == 0; return false]; i += 2; }
91                        true
92                    }
93                }
94            }
95
96            /// Finds the 0-indexed `nth` prime number.
97            ///
98            /// Note: If `nth` is negative, this function treats it as its absolute
99            /// value. For example, a value of `-3` will be treated as `3`, and the
100            /// function will return the 3rd prime number.
101            /// # Errors
102            /// Returns [`Overflow`] if the result can't fit the type.
103            /// # Examples
104            /// ```
105            /// # use devela::Int;
106            #[doc = "assert_eq![Ok(Int(2)), Int(0_" $t ").prime_nth()];"]
107            #[doc = "assert_eq![Ok(Int(3)), Int(1_" $t ").prime_nth()];"]
108            #[doc = "assert_eq![Ok(Int(127)), Int(30_" $t ").prime_nth()];"]
109            #[doc = "assert_eq![Ok(Int(127)), Int(-30_" $t ").prime_nth()];"]
110            /// # #[cfg(feature = "_int_i8")]
111            /// assert![Int(31_i8).prime_nth().is_err()];
112            /// ```
113            pub const fn prime_nth(self) -> Result<Int<$t>> {
114                let [nth, mut count, mut i] = [self.0.abs(), 1, 2];
115                loop {
116                    if Int(i).is_prime() {
117                        is![count - 1 == nth; return Ok(Int(i))];
118                        count += 1;
119                    }
120                    i = is![let Some(i) = i.checked_add(1); i; return Err(Overflow(None))];
121                }
122            }
123
124            /// Counts the number of primes upto and including `n`.
125            ///
126            #[doc = NOTATION_PI!()]
127            ///
128            #[doc = "It upcasts internally to [`" $up "`] for the inner operations."]
129            /// # Panics
130            /// It can panic if `n == i128|u128`, at the last iteration of a loop
131            /// that would take an unfeasable amount of time.
132            ///
133            /// # Examples
134            /// ```
135            /// # use devela::Int;
136            #[doc = "assert_eq![1, Int(2_" $t ").prime_pi()];"]
137            #[doc = "assert_eq![2, Int(3_" $t ").prime_pi()];"]
138            #[doc = "assert_eq![31, Int(127_" $t ").prime_pi()];"]
139            #[doc = "assert_eq![0, Int(-5_" $t ").prime_pi()];"]
140            /// ```
141            /// # Links
142            /// - <https://mathworld.wolfram.com/PrimeCountingFunction.html>.
143            /// - <https://en.wikipedia.org/wiki/Prime-counting_function>.
144            pub const fn prime_pi(self) -> usize {
145                let (mut prime_count, mut i) = (0_usize, 0 as $up);
146                while i <= self.0 as $up {
147                    is![Int(i as $t).is_prime(); prime_count += 1];
148                    i += 1;
149                }
150                prime_count
151            }
152
153            /// Counts the number of integers $<|n|$ that are relatively prime to `n`.
154            ///
155            /// Note: If `n` is negative, this function treats it as its absolute
156            /// value. For example, a value of `-3` will be treated as `3`.
157            ///
158            /// # Formulation
159            /// ## Algorithm
160            #[doc = ALGORITHM_TOTIENT!()]
161            ///
162            /// # Examples
163            /// ```
164            /// # use devela::Int;
165            #[doc = "assert_eq![Int(2), Int(4_" $t ").totient()];"]
166            #[doc = "assert_eq![Int(6), Int(9_" $t ").totient()];"]
167            #[doc = "assert_eq![Int(12), Int(13_" $t ").totient()];"]
168            #[doc = "assert_eq![Int(22), Int(-23_" $t ").totient()];"]
169            #[doc = "assert_eq![Int(2), Int(-3_" $t ").totient()];"]
170            /// ```
171            /// # Links
172            /// - <https://en.wikipedia.org/wiki/Euler%27s_totient_function>.
173            pub const fn totient(self) -> Int<$t> {
174                let (mut n, mut result, mut i) = (self.0.abs(), self.0.abs(), 2);
175                while i * i <= n {
176                    if n % i == 0 {
177                        while n % i == 0 { n /= i; }
178                        result -= result / i;
179                    }
180                    i += 1;
181                }
182                is![n > 1; result -= result / n];
183                Int(result)
184            }
185        }
186    }};
