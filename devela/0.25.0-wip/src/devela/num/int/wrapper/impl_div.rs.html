197            pub const fn div_ties_even(self, b: $t) -> Int<$t> {
198                let a = self.0; let (d, r) = (a / b, a % b);
199                // If the remainder is zero or the |remainder| is less than half of
200                // |b|, return the quotient.
201                is![r == 0 || 2 * r.abs() < b.abs(); Int(d);
202                    // If the |remainder| is greater than half of b,
203                    // return the quotient + the sign of a × the sign of b.
204                    is![2 * r.abs() > b.abs(); Int(d + a.signum() * b.signum());
205                        // If the quotient is even return it, otherwise return
206                        // the quotient + the sign of a × the sign of b.
207                        is![d % 2 == 0; Int(d); Int(d + a.signum() * b.signum())] ] ]
208            }
