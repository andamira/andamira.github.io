24/// Common auto-trait for referenced integer types.
25///
26/// # Notes
27/// - This is automatically implemented for references of types implementing [`NumInt`].
28/// - Mutable operations are only available for exclusive (`&mut`) references.
29//
30// In sync with src/num/int/num_trait/mod.rs (int_ref_* methods)
31#[cfg_attr(nightly_doc, doc(notable_trait))]
32#[rustfmt::skip]
33pub trait NumRefInt<'a>: NumRef<'a>

35    Self: Deref<Target = <Self as NumRef<'a>>::Own>,

164/// Implements the given method.
165macro_rules! impl_int_ref {
166    (
167    // >=0 Num::Rhs args, returns Self::Out
168    $fn:ident(&$self:ident $(, $arg:ident: &Rhs)*) -> Out) => { $crate::paste! {
169        #[doc = own_fn![$fn]]
170        fn $fn(&$self $(, $arg: &<Self::Own as Num>::Rhs)*)
171            -> Result<<Self::Own as Num>::Out> {
172            $self.deref().$fn($($arg),*) }
173    }};
174    (
175    // >=0 Num::Rhs args, returns an array of Self::Out
176    $fn:ident(&$self:ident $(, $arg:ident: &Rhs)*)
177     -> [Out; $LEN:literal]) => { $crate::paste! {
178        #[doc = own_fn![$fn]]
179        fn $fn(&$self $(, $arg: &<Self::Own as Num>::Rhs)*)
180            -> Result<[<Self::Own as Num>::Out; $LEN]> {
181            $self.deref().$fn($($arg),*) }
182    }};
183    (
184    // >=0 Num::Rhs args, returns Valuequant<Self::Out, Self::Out>
185    $fn:ident(&$self:ident $(, $arg:ident: &Rhs)*)
186    -> ValueQuant<Out, Out>) => { $crate::paste! {
187        #[doc = own_fn![$fn]]
188        fn $fn(&$self $(, $arg: &<Self::Own as Num>::Rhs)*)
189            -> Result<ValueQuant<<Self::Own as Num>::Out, <Self::Own as Num>::Out>> {
190            $self.deref().$fn($($arg),*) }
191    }};
192    (
193    // >=0 ty args, returns Self::Out
194    $fn:ident(&$self:ident $(, $arg:ident: $arg_ty:ty)*) -> Out) => { $crate::paste! {
195        #[doc = own_fn![$fn]]
196        fn $fn(&$self $(, $arg: $arg_ty)*) -> Result<<Self::Own as Num>::Out> {
197            $self.deref().$fn($($arg),*) }
198    }};
199    (
200    // 0 args returns ty (bool, usize…)
201    $fn:ident(&$self:ident) -> $out:ty) => { $crate::paste! {
202        #[doc = own_fn![$fn]]
203        fn $fn(&$self) -> Result<$out> {
204            $self.deref().$fn() }
205    }};
206}

209/// Links to the version that operates on values.
210macro_rules! own_fn {
211    ($fn:ident) => { own_fn!(@stringify!($fn)) };
212    ($fn:literal) => { own_fn!(@$fn) };
213    (@$fn_str:expr) => {
214        concat! { "Calls [`NumInt::", $fn_str, "][NumInt::", $fn_str, "]." }
215    };
216}
