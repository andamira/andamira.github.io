12/// A separation between two locations in `D`-dimensional space.
13///
14/// Represents a displacement vector **without an absolute origin**.
15/// It describes the magnitude of separation between positions.
16///
17/// - Unlike [`Position`], `Distance` is **relative**,
18///   and represents how far apart two positions are.
19/// - Unlike [`Orientation`], `Distance` has **magnitude**
20///   but no defined **orientation**.
21#[must_use]
22#[repr(transparent)]
23pub struct Distance<T, const D: usize> {
24    /// The component-wise separation in `D`-dimensional space.
25    pub dim: [T; D],
26}
