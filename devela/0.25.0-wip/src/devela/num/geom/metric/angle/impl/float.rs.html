10/// impl `Angle` methods with a floating-point representation.
11///
12/// # Macro arguments
13/// $f: the inner floating-point type
14/// $cap: the capability that enables the implementation. E.g "_float_f32".
15macro_rules! impl_angle {
16    () => {
17        impl_angle![float f32:"_float_f32", f64:"_float_f64"];
18    };
19    (float $($f:ty : $cap:literal),+ $(,)?) => {
20        $( impl_angle![@float $f:$cap]; )+
21    };
22    (@float $f:ty : $cap:literal) => {
23        #[doc = concat!("# Methods for angles represented using `", stringify!($f), "`.")]
24        #[cfg_attr(nightly_doc, doc(cfg(feature = $cap)))]
25        #[cfg(feature = $cap )]
26        impl Angle<$f> {
27            /* construct */
28
29            /// Creates a normalized full positive angle at 0 degrees.
30            pub const fn new_full() -> Self { Self::new(0.0) }
31
32            /// Creates a normalized right positive angle at 90 degrees (0.25).
33            pub const fn new_right() -> Self { Self::new(0.25) }
34
35            /// Creates a normalized straight positive angle at 180 degrees (0.5).
36            pub const fn new_straight() -> Self { Self::new(0.5) }
37
38            /// Creates a new angle from a `radians` value.
39            pub const fn from_rad(radians: $f) -> Self { Self::new(radians / <$f>::TAU) }
40
41            /// Creates a new angle from a `degrees` value.
42            pub const fn from_deg(degrees: $f) -> Self { Self::new(degrees / 360.0) }
43
44            /// Creates a new angle from a `value` in a `custom_unit` which represents a full turn.
45            pub const fn from_custom(value: $f, custom_unit: $f) -> Self {
46                Self::new(value / custom_unit)
47            }
48
49            /* convert */
50
51            /// Converts the angle to radians.
52            #[must_use]
53            pub const fn to_rad(self) -> $f { self.turn * <$f>::TAU }
54
55            /// Converts the angle to degrees.
56            #[must_use]
57            pub const fn to_deg(self) -> $f { self.turn * 360.0 }
58
59            /// Converts the angle to a `custom_unit` which represents a full turn.
60            #[must_use]
61            pub const fn to_custom(self, custom_unit: $f) -> $f { self.turn * custom_unit }
62
63            /* normalize */
64
65            /// Returns `true` if the angle is between -1 and 1 (non-inclusive).
66            pub const fn is_normalized(self) -> bool {
67                crate::Compare(self.turn).gt(-1.0) && crate::Compare(self.turn).lt(1.0)
68            }
69
70            /// Returns the angle normalized to the non-inclusive range -1 to 1.
71            // BLOCKED: const by fract
72            pub fn normalize(self) -> Self { Self::new(self.turn.fract()) }
73
74            /// Sets the angle normalized to the non-inclusive range -1 to 1.
75            // BLOCKED: const by fract
76            pub fn set_normalized(&mut self) { self.turn = self.turn.fract(); }
77
78            /* direction */
79
80            /// Returns the angle direction.
81            ///
82            /// Since the floating-point representation always maintains the sign
83            /// the direction can't be undefined.
84            pub const fn direction(self) -> AngleDirection {
85                use AngleDirection::{Negative, Positive};
86                if Float(self.turn).is_sign_negative() { Negative } else { Positive }
87            }
88
89            /// Returns `true` if the angle has the given `direction`.
90            ///
91            /// Since the floating-point representation always maintains the sign
92            /// the direction can't be undefined, and it will return `false` in that case.
93            #[must_use]
94            pub const fn has_direction(self, direction: AngleDirection) -> bool {
95                direction as i8 == self.direction() as i8
96            }
97
98            /// Returns a version of the angle with the given `direction`.
99            ///
100            /// An `Undefined` direction will be interpreted as counter-clockwise (positive).
101            pub const fn with_direction(self, direction: AngleDirection) -> Self {
102                use AngleDirection as D;
103                match direction {
104                    D::Positive | D::Undefined => Self::new(Float(self.turn).abs().0),
105                    D::Negative => Self::new(Float(self.turn).neg_abs().0),
106                }
107            }
108
109            /// Sets the angle to the given `direction`.
110            ///
111            /// An `Undefined` direction will be interpreted as counter-clockwise (positive).
112            pub const fn set_direction(&mut self, direction: AngleDirection) {
113                use AngleDirection as D;
114                match direction {
115                    D::Positive | D::Undefined => self.turn = Float(self.turn).abs().0,
116                    D::Negative => self.turn = Float(self.turn).neg_abs().0,
117                }
118            }
119
120            /// Returns a version of the angle with inverted direction.
121            pub const fn invert_direction(self) -> Self {
122                Self::new(Float(self.turn).flip_sign().0)
123            }
124
125            /// Returns the negative version of the angle.
126            pub const fn negative(self) -> Self { Self::new(Float(self.turn).neg_abs().0) }
127
128            /// Sets the angle as negative.
129            pub const fn set_negative(&mut self) { { self.turn = Float(self.turn).neg_abs().0; } }
130
131            /// Returns the positive version of the angle.
132            pub const fn positive(self) -> Self { Self::new(Float(self.turn).abs().0) }
133
134            /// Sets the angle as positive.
135            pub const fn set_positive(&mut self) { self.turn = Float(self.turn).abs().0; }
136
137            /* kind */
138
139            /// Returns the kind of the normalized angle.
140            // BLOCKED: const by normalize
141            pub fn kind(self) -> AngleKind {
142                let angle = crate::Compare(self.normalize().positive().turn);
143                use AngleKind::{Full, Acute, Right, Obtuse, Straight, Reflex};
144                if angle.eq(0.0) { // 1 turn (0' or 360º)
145                    Full
146                } else if angle.eq(0.25) { // 1/4 turn (90º)
147                    Right
148                } else if angle.eq(0.5) { // 1/2 turn (180º)
149                    Straight
150                } else if angle.lt(0.25) { // < 1/4 turn (< 90º)
151                    Acute
152                } else if angle.lt(0.5) { // < 1/2 turn (< 180º)
153                    Obtuse
154                } else { // < 1 turn (< 360º)
155                    Reflex
156                }
157            }
158            /// Returns the kind of the angle using a custom tolerance for approximate matching.
159            // BLOCKED: const by normalize
160            pub fn kind_approx(self, tolerance: $f) -> AngleKind {
161                let angle = self.normalize().positive().turn;
162                use AngleKind::{Full, Acute, Right, Obtuse, Straight, Reflex};
163                if (angle - 0.0).abs() <= tolerance {
164                    Full
165                } else if (angle - 0.25).abs() <= tolerance {
166                    Right
167                } else if (angle - 0.5).abs() <= tolerance {
168                    Straight
169                } else if angle < 0.25 {
170                    Acute
171                } else if angle < 0.5 {
172                    Obtuse
173                } else {
174                    Reflex
175                }
176            }
177
178            /// Returns `true` if the angle is of the given `kind`.
179            #[must_use]
180            // BLOCKED: const by normalize
181            pub fn is_kind(self, kind: AngleKind) -> bool {
182                let angle = crate::Compare(self.normalize().positive().turn);
183                use AngleKind::{Full, Acute, Right, Obtuse, Straight, Reflex};
184                match kind {
185                    Full => angle.eq(0.0),
186                    Right => angle.eq(0.25),
187                    Straight => angle.eq(0.5),
188                    Acute => angle.gt(0.0) && angle.lt(0.25),
189                    Obtuse => angle.gt(0.25) && angle.lt(0.5),
190                    Reflex => angle.gt(0.5) && angle.lt(1.0),
191                }
192            }
193
194            /// Returns `true` if the angle is of the given `kind` using a custom tolerance.
195            #[must_use]
196            // BLOCKED: const by normalize
197            pub fn is_kind_approx(self, kind: AngleKind, tolerance: $f) -> bool {
198                let angle = self.normalize().positive().turn;
199                match kind {
200                    AngleKind::Full => (angle - 0.0).abs() <= tolerance,
201                    AngleKind::Right => (angle - 0.25).abs() <= tolerance,
202                    AngleKind::Straight => (angle - 0.5).abs() <= tolerance,
203                    AngleKind::Acute => angle > 0.0 && angle < 0.25,
204                    AngleKind::Obtuse => angle > 0.25 && angle < 0.5,
205                    AngleKind::Reflex => angle > 0.5 && angle < 1.0,
206                }
207            }
208        }
209    };
