11/// Offers methods to split a primitive into an array of smaller primitives.
12///
13/// See also the [`Cast`] type for the equivalent *const* methods, and the
14/// [`PrimitiveJoin`][super::PrimitiveJoin] trait for the opposite operations.
15pub trait PrimitiveSplit<T, const LEN: usize> {
16    /// Splits `self` into an array of `T` in big-endian order.
17    #[must_use]
18    fn into_array_be(self) -> [T; LEN];
19    /// Splits `self` into an array of `T` in little-endian order.
20    #[must_use]
21    fn into_array_le(self) -> [T; LEN];
22    /// Splits `self` into an array of `T` in native-endian order.
23    #[must_use]
24    fn into_array_ne(self) -> [T; LEN];
25}
26
27// Implements the trait methods
28macro_rules! impl_into_trait {
29    ( $( $P:ident, $T:ident, $LEN:literal );+ $(;)? ) => {
30        $( impl_into_trait![@$P, $T, $LEN]; )+
31    };
32    (@$P:ident, $T:ident, $LEN:literal) => { crate::paste! {
33        impl PrimitiveSplit<$T, $LEN> for $P {
34            fn into_array_be(self) -> [$T; $LEN] { Cast(self).[<into_ $T _be>]() }
35            fn into_array_le(self) -> [$T; $LEN] { Cast(self).[<into_ $T _le>]() }
36            fn into_array_ne(self) -> [$T; $LEN] { Cast(self).[<into_ $T _ne>]() }
37        }
38    }};
39}
