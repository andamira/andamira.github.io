11/// Offers methods to construct a primitive from an array or slice of smaller primitives.
12///
13/// Methods expecting an array are more efficient than the ones expecting an
14/// slice. On the other hand slices of any lenght are supported as follows:
15/// - If the slice contains fewer elements than required, the method will
16///   fill in the missing values with zeros.
17/// - If the slice contains more elements than required, the method will
18///   ignore the extra elements.
19///
20/// See also the [`Cast`] type for the equivalent *const* methods, and the
21/// [`PrimitiveSplit`][super::PrimitiveSplit] trait for the opposite operations.
22pub trait PrimitiveJoin<T, U, const LEN: usize> {
23    /// Constructs a primitive `T` from an array of `U` in big-endian order.
24    #[must_use]
25    fn from_array_be(values: [U; LEN]) -> T;
26    /// Constructs a primitive `T` from an array of `U` in little-endian order.
27    #[must_use]
28    fn from_array_le(values: [U; LEN]) -> T;
29    /// Constructs a primitive `T` from an array of `U` in native-endian order.
30    #[must_use]
31    fn from_array_ne(values: [U; LEN]) -> T;
32
33    /// Constructs a primitive `T` from a slice of `U` in big-endian order.
34    #[must_use]
35    fn from_slice_be(values: &[U]) -> T;
36    /// Constructs a primitive `T` from a slice of `U` in little-endian order.
37    #[must_use]
38    fn from_slice_le(values: &[U]) -> T;
39    /// Constructs a primitive `T` from a slice of `U` in native-endian order.
40    #[must_use]
41    fn from_slice_ne(values: &[U]) -> T;
42}
43
44/// Implements the trait methods.
45macro_rules! impl_from_trait {
46    ( $( $T:ident, $U:ident, $LEN:literal );+ $(;)? ) => {
47        $( impl_from_trait![@$T, $U, $LEN]; )+
48    };
49    (@$T:ident, $U:ident, $LEN:literal) => { crate::paste! {
50        impl PrimitiveJoin<$T, $U, $LEN> for $T {
51            fn from_array_be(values: [$U; $LEN]) -> $T { Cast::<$T>::[<from_ $U _be>](values) }
52            fn from_array_le(values: [$U; $LEN]) -> $T { Cast::<$T>::[<from_ $U _le>](values) }
53            fn from_array_ne(values: [$U; $LEN]) -> $T { Cast::<$T>::[<from_ $U _ne>](values) }
54            fn from_slice_be(values: &[$U]) -> $T {
55                let mut array = [0; $LEN];
56                for (i, &v) in values.iter().enumerate() {
57                    array[i] = v;
58                }
59                Cast::<$T>::[<from_ $U _be>](array)
60            }
61            fn from_slice_le(values: &[$U]) -> $T {
62                let mut array = [0; $LEN];
63                for (i, &v) in values.iter().enumerate() {
64                    array[i] = v;
65                }
66                Cast::<$T>::[<from_ $U _le>](array)
67            }
68            fn from_slice_ne(values: &[$U]) -> $T {
69                let mut array = [0; $LEN];
70                for (i, &v) in values.iter().enumerate() {
71                    array[i] = v;
72                }
73                Cast::<$T>::[<from_ $U _ne>](array)
74            }
75        }
76    }};
77}
