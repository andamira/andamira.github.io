8/// A simple 8-bit <abbr title="Pseudo-Random Number Generator">PRNG</abbr>
9/// with 32-bit of state, based on the *XorShift* algorithm.
10///
11/// It has a 0.8% chance of falling into a poor quality short chain,
12/// a some degree of care is required to seed it. However, the quality of the
13/// random numbers is excellent for such a small state (32 bits), and it passes
14/// almost all of the die hard tests.
15///
16/// Its longest cycle is 4_261_412_736.
17// (== u32::MAX + u16::MAX * 512 + 639).
18#[doc = crate::_doc!(vendor: "8bit_rng")]
19#[must_use]
20#[derive(Clone, Copy, Debug, PartialEq, Eq)]
21pub struct Xyza8a {
22    x: u8,
23    y: u8,
24    z: u8,
25    a: u8,
26}
27
28/// Creates a new PRNG initialized with the default fixed seed.
29impl Default for Xyza8a {
30    fn default() -> Self {
31        Self::DEFAULT
32    }
33}
34/// Creates a new PRNG initialized with the default fixed seed.
35impl ConstDefault for Xyza8a {
36    const DEFAULT: Self = Self::new(Self::DEFAULT_SEED);
37}
38
39// private associated items
40impl Xyza8a {
41    const DEFAULT_SEED: [u8; 4] = [0xDE, 0xFA, 0x00, 0x17];
42}
43
44impl Xyza8a {
45    /// Returns a seeded `Xyza8a` generator from the given 4 × 8-bit seeds.
46    pub const fn new(seeds: [u8; 4]) -> Self {
47        Self { x: seeds[0], y: seeds[1], z: seeds[2], a: seeds[3] }
48    }
49
50    #[must_use]
51    /// Returns the PRNG's inner state as a raw snapshot.
52    pub const fn inner_state(self) -> [u8; 4] {
53        [self.x, self.y, self.z, self.a]
54    }
55    /// Restores the PRNG from the given state.
56    pub const fn from_state(state: [u8; 4]) -> Self {
57        Self { x: state[0], y: state[1], z: state[2], a: state[3] }
58    }
59
60    #[must_use]
61    /// Returns the current random `u8`.
62    pub const fn current_u8(&self) -> u8 {
63        self.a
64    }
65    /// Advances the state and returns the next random `u8`.
66    pub fn next_u8(&mut self) -> u8 {
67        let t = self.x ^ (self.x << 4);
68        self.x = self.y;
69        self.y = self.z;
70        self.z = self.a;
71        self.a = self.z ^ t ^ (self.z >> 1) ^ (t << 1);
72        self.a
73    }
74
75    /// Returns a copy of the next new random state.
76    pub const fn peek_next_state(&self) -> Self {
77        let mut new = *self;
78
79        let t = new.x ^ (new.x << 4);
80        new.x = new.y;
81        new.y = new.z;
82        new.z = new.a;
83        new.a = new.z ^ t ^ (new.z >> 1) ^ (t << 1);
84        new
85    }
86
87    /// Returns both the next random state and the `u8` value.
88    pub const fn own_next_u8(self) -> Own<Self, u8> {
89        let s = self.peek_next_state();
90        let v = s.current_u8();
91        Own::new(s, v)
92    }
93}

123/// A simple 8-bit <abbr title="Pseudo-Random Number Generator">PRNG</abbr>
124/// with 32-bit of state, based on the *XorShift* algorithm.
125///
126/// It has an almost optimal cycle so no real care is required
127/// for seeding except avoiding all zeros, but it fails many of the die hard
128/// random number tests.
129///
130/// Its longest cycle is 4,294,967,294.

156    /// This is the fastest constructor.
157    pub const fn new(seeds: [u8; 4]) -> Self {
158        Self { x: seeds[0], y: seeds[1], z: seeds[2], a: seeds[3] }
159    }

169    }
170
171    /// Returns the current random `u8`.
172    pub const fn current_u8(&self) -> u8 {
173        self.a
174    }
175
176    /// Returns the next random `u8`.
177    pub fn next_u8(&mut self) -> u8 {
178        let t = self.x ^ (self.x >> 1);
179        self.x = self.y;
180        self.y = self.z;
181        self.z = self.a;
182        self.a = self.z ^ t ^ (self.z >> 3) ^ (t << 1);

188        let mut new = *self;
189
190        let t = new.x ^ (new.x >> 1);
191        new.x = new.y;
192        new.y = new.z;
193        new.z = new.a;
194        new.a = new.z ^ t ^ (new.z >> 3) ^ (t << 1);
195        new
196    }

199    pub const fn own_next_u8(self) -> Own<Self, u8> {
200        let s = self.peek_next_state();
201        let v = s.current_u8();
202        Own::new(s, v)
203    }
