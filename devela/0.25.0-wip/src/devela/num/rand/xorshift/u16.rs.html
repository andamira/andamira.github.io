8#[doc = crate::TAG_RAND!()]
9/// The `XorShift16` <abbr title="Pseudo-Random Number Generator">PRNG</abbr>.
10///
11/// It has a 16-bit state and generates 16-bit numbers.
12///
13/// This is [John Metcalf's 16-bit] (7, 9, 8) version of George Marsaglia's
14/// original [`XorShift32`][super::XorShift32].
15///
16/// [John Metcalf's 16-bit]: https://github.com/impomatic/xorshift798
17#[must_use]
18#[derive(Clone, Copy, Debug, PartialEq, Eq)]
19pub struct XorShift16<
20    const BASIS: usize = 0,
21    const A: usize = 7,
22    const B: usize = 9,
23    const C: usize = 8,
24>(u16);
25
26/// Creates a new PRNG initialized with the default fixed seed.
27impl Default for XorShift16 {
28    fn default() -> Self {
29        Self::DEFAULT
30    }
31}
32/// Creates a new PRNG initialized with the default fixed seed.
33impl ConstDefault for XorShift16 {
34    const DEFAULT: Self = Self::new_unchecked(Self::DEFAULT_SEED);
35}

37// private associated items
38impl<const BASIS: usize, const A: usize, const B: usize, const C: usize>
39    XorShift16<BASIS, A, B, C>
40{
41    const DEFAULT_SEED: u16 = 0xDEFA;
42
43    #[cold] #[allow(dead_code)] #[rustfmt::skip]
44    const fn cold_path_default() -> Self { Self::new_unchecked(Self::DEFAULT_SEED) }
45}
46
47impl<const BASIS: usize, const A: usize, const B: usize, const C: usize>
48    XorShift16<BASIS, A, B, C>
49{
50    /// Returns a seeded `XorShift16` generator from the given 16-bit seed.
51    ///
52    /// If the seed is `0`, the default seed is used instead.
53    pub const fn new(seed: u16) -> Self {
54        if seed == 0 { Self::cold_path_default() } else { Self(seed) }
55    }
56
57    /// Returns a seeded `XorShift16` generator from the given 8-bit seed, unchecked.
58    ///
59    /// The seed must not be `0`, otherwise every result will also be `0`.
60    pub const fn new_unchecked(seed: u16) -> Self {
61        debug_assert![seed != 0, "Seed must be non-zero"];
62        Self(seed)
63    }
64
65    #[must_use]
66    /// Returns the PRNG's inner state as a raw snapshot.
67    pub const fn inner_state(self) -> u16 {
68        self.0
69    }
70    /// Restores the PRNG from the given state.
71    pub const fn from_state(state: u16) -> Self {
72        Self(state)
73    }
74
75    /// Returns the current random `u16`.
76    #[must_use]
77    pub const fn current_u16(&self) -> u16 {
78        self.0
79    }
80
81    /// Returns the next random `u16`.
82    ///
83    #[must_use]
84    pub fn next_u16(&mut self) -> u16 {
85        let mut x = self.0;
86        xorshift_basis!(x, BASIS, (A, B, C));
87        self.0 = x;
88        x
89    }
90
91    /// Returns a copy of the next new random state.
92    pub const fn peek_next_state(&self) -> Self {
93        let mut x = self.0;
94        xorshift_basis!(x, BASIS, (A, B, C));
95        Self(x)
96    }
97
98    /// Returns both the next random state and the `u16` value.
99    pub const fn own_next_u16(self) -> Own<Self, u16> {
100        let s = self.peek_next_state();
101        let v = s.current_u16();
102        Own::new(s, v)
103    }
104}

125#[cfg(feature = "dep_rand_core")]
126#[cfg_attr(nightly_doc, doc(cfg(feature = "dep_rand_core")))]
127mod impl_rand {
128    use crate::_dep::rand_core::{RngCore, SeedableRng};
129    use crate::{Cast, XorShift16};

131    impl<const BASIS: usize, const A: usize, const B: usize, const C: usize> RngCore
132        for XorShift16<BASIS, A, B, C>
133    {
134        /// Returns the next 2 × random `u16` combined as a single `u32`.
135        fn next_u32(&mut self) -> u32 {
136            Cast::<u32>::from_u16_le([self.next_u16(), self.next_u16()])
137        }
138        /// Returns the next 4 × random `u16` combined as a single `u64`.
139        fn next_u64(&mut self) -> u64 {
140            Cast::<u64>::from_u16_le([
141                self.next_u16(),
142                self.next_u16(),
143                self.next_u16(),
144                self.next_u16(),
145            ])
146        }
147        fn fill_bytes(&mut self, dest: &mut [u8]) {
148            let mut i = 0;
149            while i < dest.len() {
150                let random_u16 = self.next_u16();
151                let bytes = random_u16.to_le_bytes();
152                let remaining = dest.len() - i;
153
154                if remaining >= 2 {
155                    dest[i] = bytes[0];
156                    dest[i + 1] = bytes[1];
157                    i += 2;
158                } else {
159                    dest[i] = bytes[0];
160                    i += 1;
161                }
162            }
163        }
164    }
165
166    impl<const BASIS: usize, const A: usize, const B: usize, const C: usize> SeedableRng
167        for XorShift16<BASIS, A, B, C>
168    {
169        type Seed = [u8; 2];
170
171        /// When seeded with zero this implementation uses the default seed
172        /// value as the cold path.
173        fn from_seed(seed: Self::Seed) -> Self {
174            if seed == [0; 2] {
175                Self::cold_path_default()
176            } else {
177                Self::new_unchecked(u16::from_le_bytes(seed))
178            }
179        }
180    }
181}
