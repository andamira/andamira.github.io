8#[doc = crate::TAG_RAND!()]
9/// The `XorShift128` <abbr title="Pseudo-Random Number Generator">PRNG</abbr>.
10///
11/// It has a 128-bit state and generates 64-bit numbers.
12///
13/// See also [`XorShift128p`][crate::XorShift128p].
14#[must_use]
15#[derive(Clone, Copy, Debug, PartialEq, Eq)]
16pub struct XorShift128([u32; 4]);
17
18/// Creates a new PRNG initialized with the default fixed seed.
19impl Default for XorShift128 {
20    fn default() -> Self {
21        Self::DEFAULT
22    }
23}
24/// Creates a new PRNG initialized with the default fixed seed.
25impl ConstDefault for XorShift128 {
26    const DEFAULT: Self = Self::new_unchecked(Self::DEFAULT_SEED);
27}
28
29// private associated items
30impl XorShift128 {
31    const DEFAULT_SEED: [u32; 4] = [0xDEFA_0017; 4];
32
33    #[cold] #[allow(dead_code)] #[rustfmt::skip]
34    const fn cold_path_default() -> Self { Self::new_unchecked(Self::DEFAULT_SEED) }
35}
36
37impl XorShift128 {
38    /// Returns a seeded `XorShift128` generator from the given 4 × 32-bit seeds.
39    ///
40    /// If the seeds are all `0`, the default seed is used instead.
41    pub const fn new(seeds: [u32; 4]) -> Self {
42        if (seeds[0] | seeds[1] | seeds[2] | seeds[3]) == 0 {
43            Self::cold_path_default()
44        } else {
45            Self([seeds[0], seeds[1], seeds[2], seeds[3]])
46        }
47    }
48
49    /// Returns a seeded `XorShift128` generator from the given 8-bit seed,
50    /// unchecked.
51    ///
52    /// The seeds must not be all `0`, otherwise every result will also be `0`.
53    ///
54    /// # Panics
55    /// Panics in debug if the seeds are all `0`.
56    pub const fn new_unchecked(seeds: [u32; 4]) -> Self {
57        debug_assert![(seeds[0] | seeds[1] | seeds[2] | seeds[3]) != 0, "Seeds must be non-zero"];
58        Self(seeds)
59    }
60
61    #[must_use]
62    /// Returns the PRNG's inner state as a raw snapshot.
63    pub const fn inner_state(self) -> [u32; 4] {
64        self.0
65    }
66    /// Restores the PRNG from the given state.
67    pub const fn from_state(state: [u32; 4]) -> Self {
68        Self(state)
69    }
70
71    #[must_use]
72    /// Returns the current random `u64`.
73    pub const fn current_u64(&self) -> u64 {
74        ((self.0[0] as u64) << 32) | (self.0[1] as u64)
75    }
76
77    #[must_use]
78    /// Returns the next random `u64`.
79    // Note how the output of the RNG is computed before updating the state,
80    // unlike on Xoroshiro128pp, for example.
81    pub fn next_u64(&mut self) -> u64 {
82        let t = self.0[3];
83        let mut s = self.0[0];
84        self.0[3] = self.0[2];
85        self.0[2] = self.0[1];
86        self.0[1] = s;
87        s ^= s << 11;
88        s ^= s >> 8;
89        self.0[0] = s ^ t ^ (t >> 19);
90
91        ((self.0[0] as u64) << 32) | (self.0[1] as u64)
92    }
93
94    /// Returns a copy of the next new random state.
95    pub const fn peek_next_state(&self) -> Self {
96        let mut x = self.0;
97
98        let t = x[3];
99        let mut s = x[0];
100        x[3] = x[2];
101        x[2] = x[1];
102        x[1] = s;
103        s ^= s << 11;
104        s ^= s >> 8;
105        x[0] = s ^ t ^ (t >> 19);
106
107        Self(x)
108    }
109
110    /// Returns both the next random state and the `u64` value.
111    pub const fn own_next_u64(self) -> Own<Self, u64> {
112        let s = self.peek_next_state();
113        let v = s.current_u64();
114        Own::new(s, v)
115    }
116}

118/// # Extra constructors
119impl XorShift128 {
120    /// Returns a seeded `XorShift128` generator from the given 128-bit seed.
121    ///
122    /// The seeds will be split in little endian order.
123    pub const fn new1_u128(seed: u128) -> Self {
124        Self::new(Cast(seed).into_u32_le())
125    }
126
127    /// Returns a seeded `XorShift128` generator from the given 2 × 64-bit seeds.
128    ///
129    /// The seeds will be split in little endian order.
130    pub const fn new2_u64(seeds: [u64; 2]) -> Self {
131        let [x, y] = Cast(seeds[0]).into_u32_le();
132        let [z, a] = Cast(seeds[1]).into_u32_le();
133        Self::new([x, y, z, a])
134    }
135
136    /// Returns a seeded `XorShift128` generator from the given 4 × 32-bit seeds.
137    ///
138    /// This is an alias of [`new`][Self#method.new].
139    pub const fn new4_u32(seeds: [u32; 4]) -> Self {
140        Self::new(seeds)
141    }
142
143    /// Returns a seeded `XorShift128` generator from the given 8 × 16-bit seeds.
144    ///
145    /// The seeds will be joined in little endian order.
146    pub const fn new8_u16(seeds: [u16; 8]) -> Self {
147        Self::new([
148            Cast::<u32>::from_u16_le([seeds[0], seeds[1]]),
149            Cast::<u32>::from_u16_le([seeds[2], seeds[3]]),
150            Cast::<u32>::from_u16_le([seeds[4], seeds[5]]),
151            Cast::<u32>::from_u16_le([seeds[6], seeds[7]]),
152        ])
153    }
154
155    /// Returns a seeded `XorShift128` generator from the given 16 × 8-bit seeds.
156    ///
157    /// The seeds will be joined in little endian order.
158    pub const fn new16_u8(seeds: [u8; 16]) -> Self {
159        Self::new([
160            u32::from_le_bytes([seeds[0], seeds[1], seeds[2], seeds[3]]),
161            u32::from_le_bytes([seeds[4], seeds[5], seeds[6], seeds[7]]),
162            u32::from_le_bytes([seeds[8], seeds[9], seeds[10], seeds[11]]),
163            u32::from_le_bytes([seeds[12], seeds[13], seeds[14], seeds[15]]),
164        ])
165    }
166}

168#[cfg(feature = "dep_rand_core")]
169#[cfg_attr(nightly_doc, doc(cfg(feature = "dep_rand_core")))]
170mod impl_rand {
171    use super::XorShift128;
172    use crate::_dep::rand_core::{RngCore, SeedableRng};
173
174    impl RngCore for XorShift128 {
175        /// Returns the next random `u32`,
176        /// from the first 32-bits of `next_u64`.
177        fn next_u32(&mut self) -> u32 {
178            self.next_u64() as u32
179        }
180
181        /// Returns the next random `u64`.
182        fn next_u64(&mut self) -> u64 {
183            self.next_u64()
184        }
185
186        fn fill_bytes(&mut self, dest: &mut [u8]) {
187            let mut i = 0;
188            while i < dest.len() {
189                let random_u64 = self.next_u64();
190                let bytes = random_u64.to_le_bytes();
191                let remaining = dest.len() - i;
192
193                if remaining >= 8 {
194                    dest[i..i + 8].copy_from_slice(&bytes);
195                    i += 8;
196                } else {
197                    dest[i..].copy_from_slice(&bytes[..remaining]);
198                    break;
199                }
200            }
201        }
202    }
203
204    impl SeedableRng for XorShift128 {
205        type Seed = [u8; 16];
206
207        /// When seeded with zero this implementation uses the default seed
208        /// value as the cold path.
209        fn from_seed(seed: Self::Seed) -> Self {
210            let mut seed_u32s = [0u32; 4];
211            if seed == [0; 16] {
212                Self::cold_path_default()
213            } else {
214                for i in 0..4 {
215                    seed_u32s[i] = u32::from_le_bytes([
216                        seed[i * 4],
217                        seed[i * 4 + 1],
218                        seed[i * 4 + 2],
219                        seed[i * 4 + 3],
220                    ]);
221                }
222                Self::new_unchecked(seed_u32s)
223            }
224        }
225    }
