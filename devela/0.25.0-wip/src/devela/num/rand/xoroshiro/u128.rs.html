20/// The `Xoroshiro128++`
21/// <abbr title="Pseudo-Random Number Generator">PRNG</abbr>.
22///
23/// It has a 128-bit state and generates 32-bit numbers.
24#[must_use]
25#[derive(Clone, Copy, Debug, PartialEq, Eq)]
26pub struct Xoroshiro128pp([u32; 4]);
27
28/// # Constructors
29impl Xoroshiro128pp {
30    /// Creates a new Xoroshiro128++ PRNG with the given `seed`.
31    ///
32    /// Returns `None` if the seed parts are all zero.
33    #[must_use]
34    pub const fn new(seed: [u32; 4]) -> Option<Self> {
35        if (seed[0] | seed[1] | seed[2] | seed[3]) == 0 {
36            Self::cold_path_result()
37        } else {
38            Some(Self(seed))
39        }
40    }
41
42    /// Creates a new Xoroshiro128++ PRNG with the given seed without any checks.
43    ///
44    /// # Panics
45    /// Panics in debug mode if the seed parts are all `0`.
46    pub const fn new_unchecked(seed: [u32; 4]) -> Self {
47        debug_assert!((seed[0] | seed[1] | seed[2] | seed[3]) != 0, "Seed must be non-zero");
48        Self(seed)
49    }
50
51    /// Creates a new Xoroshiro128++ PRNG, seeded from addresses of stack variables.
52    ///
53    /// This is a very poor source of randomness.
54    #[inline(never)]
55    pub fn from_stack() -> Self {
56        let (a, b, c, d) = (0, 0, 0, 0);
57        let seed: [u32; 4] = [
58            &a as *const _ as u32,
59            &b as *const _ as u32,
60            &c as *const _ as u32,
61            &d as *const _ as u32,
62        ];
63        Self::new_unchecked(seed)
64    }
65
66    /// Creates a new Xoroshiro128++ PRNG, seeded from addresses of heap and stack variables.
67    ///
68    /// This is a very poor source of randomness.
69    #[inline(never)]
70    #[cfg(feature = "alloc")]
71    pub fn from_heap() -> Self {
72        let (a, b, c, d) = (0, Box::new(0), Box::new(0), 0);
73        let seed: [u32; 4] = [
74            &a as *const _ as u32,
75            &b as *const _ as u32,
76            &c as *const _ as u32,
77            &d as *const _ as u32,
78        ];
79        Self::new_unchecked(seed)
80    }
81
82    /// Creates a new Xoroshiro128++ PRNG, seeded from [`RandomState`].
83    #[inline(never)]
84    #[cfg(feature = "std")]
85    pub fn from_randomstate() -> Self {
86        let h = RandomState::new();
87        let (mut hasher1, mut hasher2) = (h.build_hasher(), h.build_hasher());
88        hasher1.write_u32(Self::DEFAULT_SEED[0]);
89        hasher2.write_u32(Self::DEFAULT_SEED[0]);
90        let (hash1, hash2) = (hasher1.finish(), hasher2.finish());
91        let seed = [(hash1 >> 32) as u32, hash1 as u32, (hash2 >> 32) as u32, hash2 as u32];
92        Self::new_unchecked(seed)
93    }
94
95    #[must_use]
96    /// Returns the PRNG's inner state as a raw snapshot.
97    pub const fn inner_state(self) -> [u32; 4] {
98        self.0
99    }
100    /// Restores the PRNG from the given state.
101    pub const fn from_state(state: [u32; 4]) -> Self {
102        Self(state)
103    }
104}

106/// # Methods taking `&mut self`
107impl Xoroshiro128pp {
108    /// The jump function for the generator, equivalent to 2^64 `next_u32` calls.
109    pub fn jump(&mut self) {
110        self.jump_with_constant(Self::JUMP);
111    }
112
113    /// The long jump function for the generator, equivalent to 2^96 `next_u32` calls.
114    pub fn long_jump(&mut self) {
115        self.jump_with_constant(Self::LONG_JUMP);
116    }
117
118    /// Generates the next random `u32` value.
119    // Note how the output of the RNG is computed before updating the state.
120    // unlike on Xorshift128, for example.
121    #[must_use]
122    pub fn next_u32(&mut self) -> u32 {
123        let result = self.current_u32();
124        let t = self.0[1] << 9;
125        self.0[2] ^= self.0[0];
126        self.0[3] ^= self.0[1];
127        self.0[1] ^= self.0[2];
128        self.0[0] ^= self.0[3];
129        self.0[2] ^= t;
130        self.0[3] = Self::rotl(self.0[3], 11);
131        result
132    }
133
134    /// Generates the next 2 random `u32` values.
135    #[must_use]
136    pub fn next2(&mut self) -> [u32; 2] {
137        [self.next_u32(), self.next_u32()]
138    }
139    /// Generates the next 4 random `u32` values.
140    #[must_use]
141    pub fn next4(&mut self) -> [u32; 4] {
142        [self.next_u32(), self.next_u32(), self.next_u32(), self.next_u32()]
143    }
144    /// Generates the next random value split into 4 u8 values.
145    #[must_use]
146    pub fn next4_u8(&mut self) -> [u8; 4] {
147        self.next_u32().to_ne_bytes()
148    }
149    /// Generates the next random value split into 2 u16 values.
150    #[must_use]
151    pub fn next2_u16(&mut self) -> [u16; 2] {
152        Cast(self.next_u32()).into_u16_ne()
153    }
154    /// Returns the next u64, advancing the state 2 times.
155    #[must_use]
156    pub fn next_u64(&mut self) -> u64 {
157        Cast::<u64>::from_u32_ne(self.next2())
158    }
159    /// Returns the next u128, advancing the state 4 times.
160    #[must_use]
161    pub fn next_u128(&mut self) -> u128 {
162        Cast::<u128>::from_u32_ne(self.next4())
163    }
164}

166/// # Methods taking `self`
167impl Xoroshiro128pp {
168    /// Returns the current random `u32`, without updating the state.
169    #[must_use]
170    pub const fn current_u32(self) -> u32 {
171        Self::rotl(self.0[0].wrapping_add(self.0[3]), 7).wrapping_add(self.0[0])
172    }
173
174    /// Returns a copy of the next new random state.
175    pub const fn copy_peek_next_state(self) -> Self {
176        let mut x = self.0;
177        let t = x[1] << 9;
178        x[2] ^= x[0];
179        x[3] ^= x[1];
180        x[1] ^= x[2];
181        x[0] ^= x[3];
182        x[2] ^= t;
183        x[3] = Self::rotl(x[3], 11);
184        Self(x)
185    }
186
187    /// Returns both the next random state and the `u32` value in a tuple.
188    pub const fn own_next_u32(self) -> Own<Self, u32> {
189        let next_state = self.copy_peek_next_state();
190        let next_value = next_state.current_u32();
191        Own::new(next_state, next_value)
192    }
193
194    /// Returns a copy of the state jumped ahead by 2^64 steps.
195    pub const fn copy_jump(self) -> Self {
196        self.copy_jump_with_constant(Self::JUMP)
197    }
198
199    /// Returns a copy of the state long-jumped ahead by 2^96 steps.
200    pub const fn copy_long_jump(self) -> Self {
201        self.copy_jump_with_constant(Self::LONG_JUMP)
202    }
203}

270mod impl_rand {
271    use super::Xoroshiro128pp;
272    use crate::_dep::rand_core::{RngCore, SeedableRng};
273
274    impl RngCore for Xoroshiro128pp {
275        /// Returns the next random `u32`.
276        fn next_u32(&mut self) -> u32 {
277            self.next_u32()
278        }
279        /// Returns the next random `u64`.
280        fn next_u64(&mut self) -> u64 {
281            ((self.next_u32() as u64) << 32) | (self.next_u32() as u64)
282        }
283        fn fill_bytes(&mut self, dest: &mut [u8]) {
284            let mut i = 0;
285            while i < dest.len() {
286                let random_u32 = self.next_u32();
287                let bytes = random_u32.to_le_bytes();
288                let remaining = dest.len() - i;
289
290                if remaining >= 4 {
291                    dest[i..i + 4].copy_from_slice(&bytes);
292                    i += 4;
293                } else {
294                    dest[i..].copy_from_slice(&bytes[..remaining]);
295                    break;
296                }
297            }
298        }
299    }
300
301    impl SeedableRng for Xoroshiro128pp {
302        type Seed = [u8; 16];
303
304        /// When seeded with zero this implementation uses the default seed
305        /// value as the cold path.
306        fn from_seed(seed: Self::Seed) -> Self {
307            let mut seed_u32s = [0u32; 4];
308            if seed == [0; 16] {
309                Self::cold_path_default()
310            } else {
311                for i in 0..4 {
312                    seed_u32s[i] = u32::from_le_bytes([
313                        seed[i * 4],
314                        seed[i * 4 + 1],
315                        seed[i * 4 + 2],
316                        seed[i * 4 + 3],
317                    ]);
318                }
319                Self::new_unchecked(seed_u32s)
320            }
321        }
322    }
323}

327impl Xoroshiro128pp {
328    const DEFAULT_SEED: [u32; 4] = [0xDEFA_0017; 4];
329    const JUMP: [u32; 4] = [0x8764_000b, 0xf542_d2d3, 0x6fa0_35c3, 0x77f2_db5b];
330    const LONG_JUMP: [u32; 4] = [0xb523_952e, 0x0b6f_099f, 0xccf_5a0ef, 0x1c58_0662];
331
332    #[cold] #[rustfmt::skip]
333    const fn cold_path_result() -> Option<Self> { None }
334    #[cold] #[allow(dead_code)] #[rustfmt::skip]
335    const fn cold_path_default() -> Self { Self::new_unchecked(Self::DEFAULT_SEED) }
336
337    // rotates `x` left by `k` bits.
338    const fn rotl(x: u32, k: i32) -> u32 {
339        (x << k) | (x >> (32 - k))
340    }
341
342    fn jump_with_constant(&mut self, jump: [u32; 4]) {
343        let (mut s0, mut s1, mut s2, mut s3) = (0, 0, 0, 0);
344        for &j in jump.iter() {
345            for b in 0..32 {
346                if (j & (1 << b)) != 0 {
347                    s0 ^= self.0[0];
348                    s1 ^= self.0[1];
349                    s2 ^= self.0[2];
350                    s3 ^= self.0[3];
351                }
352                let _ = self.next_u32();
353            }
354        }
355        self.0 = [s0, s1, s2, s3];
356    }
357
358    const fn copy_jump_with_constant(self, jump: [u32; 4]) -> Self {
359        let (mut s0, mut s1, mut s2, mut s3) = (0, 0, 0, 0);
360        let mut state = self;
361        let mut i = 0;
362        while i < jump.len() {
363            let mut b = 0;
364            while b < 32 {
365                if (jump[i] & (1 << b)) != 0 {
366                    s0 ^= state.0[0];
367                    s1 ^= state.0[1];
368                    s2 ^= state.0[2];
369                    s3 ^= state.0[3];
370                }
371                state = state.copy_peek_next_state();
372                b += 1;
373            }
374            i += 1;
375        }
376        Self([s0, s1, s2, s3])
377    }
