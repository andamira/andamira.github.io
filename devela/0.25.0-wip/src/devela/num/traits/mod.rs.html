22/// Common trait for numeric types.
23///
24/// # Notes
25/// - Every method has a default implementation that returns [`NotImplemented`].
26/// - Specific implementations can customize the operations it wants to support.
27/// - Any operations specifically not supported should ideally return [`NotSupported`].
28///
29/// - Most methods come in two variants, starting with different prefixes:
30///   - `num_*` methods take the arguments by value.
31///   - `num_ref_*` methods take the arguments by reference.
32///
33/// - This trait tries to offer the same methods everywhere, giving a result when possible.
34/// - Operations are generally supported if they can be valid for some values.
35/// E.g. `num_abs` for unsigned types is only valid for `0`.
36///
37/// See also [`NumRef`] which is automatically implemented for `Num` references.
38#[rustfmt::skip]
39#[cfg_attr(nightly_doc, doc(notable_trait))]
40#[allow(unused_variables, reason = "default implementation is not implemented")]
41pub trait Num {
42    /// The internal representation of this numeric type.
43    type Inner;
44
45    /// The output type for operations.
46    type Out;
47
48    /// The right hand side type for operations.
49    type Rhs;
50
51    /// Returns the inner `self` representation.
52    #[must_use]
53    fn num_into(self) -> Self::Inner;
54
55    /// Returns `Self` if given a valid `value`.
56    fn num_from(value: Self::Inner) -> Result<Self>
57        where Self: Sized { E::ni() }
58    /// Returns `Self` if given a valid `&value`.
59    fn num_from_ref(value: &Self::Inner) -> Result<Self>
60        where Self: Sized { E::ni() }
61
62    /// Sets `self` to the given valid `value`.
63    fn num_set(&mut self, value: Self::Inner) -> Result<()> { E::ni() }
64    /// Sets `self` to the given valid `&value`.
65    fn num_set_ref(&mut self, value: &Self::Inner) -> Result<()> { E::ni() }
66
67    /* Identities */
68
69    /// Returns `true` if `self` is zero.
70    fn num_is_zero(&self) -> Result<bool> { E::ni() }
71    /// Returns the number zero.
72    fn num_get_zero() -> Result<Self> where Self: Sized { E::ni() }
73    /// Sets `self` to `0`.
74    fn num_set_zero(&mut self) -> Result<()> { E::ni() }
75
76    /// Returns `true` if `self` is one.
77    fn num_is_one(&self) -> Result<bool> { E::ni() }
78    /// Returns the number one.
79    fn num_get_one() -> Result<Self> where Self: Sized { E::ni() }
80    /// Sets the number to one.
81    fn num_set_one(&mut self) -> Result<()> { E::ni() }
82
83    /* Operations */
84
85    /// Computes `self + rhs` (addition).
86    fn num_add(self, rhs: Self::Rhs) -> Result<Self::Out> where Self: Sized { E::ni() }
87    /// *Like [`num_add`][Self::num_add] but takes the arguments by reference.*
88    fn num_ref_add(&self, rhs: &Self::Rhs) -> Result<Self::Out> { E::ni() }
89    /// Computes `&mut self += rhs;` (addition).
90    fn num_ref_add_assign(&mut self, rhs: &Self::Rhs) -> Result<()> { E::ni() }
91
92    /// Computes `self - rhs` (subtraction).
93    fn num_sub(self, rhs: Self::Rhs) -> Result<Self::Out> where Self: Sized { E::ni() }
94    /// *Like [`num_sub`][Self::num_sub] but takes the arguments by reference.*
95    fn num_ref_sub(&self, rhs: &Self::Rhs) -> Result<Self::Out> { E::ni() }
96    /// Computes `&mut self -= rhs;` (subtraction).
97    fn num_ref_sub_assign(&mut self, rhs: &Self::Rhs) -> Result<()> { E::ni() }
98
99    /// Computes `self * rhs` (multiplication).
100    fn num_mul(self, rhs: Self::Rhs) -> Result<Self::Out> where Self: Sized { E::ni() }
101    /// *Like [`num_mul`][Self::num_mul] but takes the arguments by reference.*
102    fn num_ref_mul(&self, rhs: &Self::Rhs) -> Result<Self::Out> { E::ni() }
103    /// Computes `&mut self *= rhs;` (multiplication).
104    fn num_ref_mul_assign(&mut self, rhs: &Self::Rhs) -> Result<()> { E::ni() }
105
106    /// Computes `self / rhs` (division).
107    fn num_div(self, rhs: Self::Rhs) -> Result<Self::Out> where Self: Sized { E::ni() }
108    /// *Like [`num_div`][Self::num_div] but takes the arguments by reference.*
109    fn num_ref_div(&self, rhs: &Self::Rhs) -> Result<Self::Out> { E::ni() }
110    /// Computes `&mut self /= rhs;` (division).
111    fn num_ref_div_assign(&mut self, rhs: &Self::Rhs) -> Result<()> { E::ni() }
112
113    /// Computes `self % rhs` (remainder).
114    fn num_rem(self, rhs: Self::Rhs) -> Result<Self::Out> where Self: Sized { E::ni() }
115    /// *Like [`num_rem`][Self::num_rem] but takes the arguments by reference.*
116    fn num_ref_rem(&self, rhs: &Self::Rhs) -> Result<Self::Out> { E::ni() }
117    /// Computes `&mut self %= rhs;` (remainder).
118    fn num_ref_rem_assign(&mut self, rhs: &Self::Rhs) -> Result<()> { E::ni() }
119
120    /// Computes `-self` (additive inverse).
121    fn num_neg(self) -> Result<Self::Out> where Self: Sized { E::ni() }
122    /// *Like [`num_neg`][Self::num_neg] but takes the arguments by reference.*
123    fn num_ref_neg(&self) -> Result<Self::Out> { E::ni() }
124
125    /// Computes `|self|` (absolute value).
126    fn num_abs(self) -> Result<Self::Out> where Self: Sized { E::ni() }
127    /// *Like [`num_abs`][Self::num_abs] but takes the arguments by reference.*
128    fn num_ref_abs(&self) -> Result<Self::Out> { E::ni() }
129}
