11/// Extension trait for floating-point types. Associated methods.
12///
13/// This trait can be more convenient to use than the [`Float`] struct,
14/// for non-const operations over primitive floating-point types.
15///
16/// `Float` has a few more methods implemented if the `dep_libm` feature is enabled.

434macro_rules! impl_ext_float {

436        impl_ext_float![
437            (f32, u32 | i32),
438            (f64, u32 | i32),
439        ];

442    // $f:   the floating-point type.
443    // $ue:  unsigned int type with the same bit-size.
444    // $ie:  the integer type for integer exponentiation.
445    ($( ($f:ty, $ue:ty|$ie:ty)),+ $(,)?) => {
446        $( impl_ext_float![@$f, $ue|$ie]; )+
447    };
448    (@$f:ty, $ue:ty|$ie:ty) => {
449        impl ExtFloat for $f {
450            fn floor(self) -> Self { Float(self).floor().0 }

452            fn ceil(self) -> Self { Float(self).ceil().0 }

456            fn round_ties_away(self) -> Self { Float(self).round_ties_away().0 }
457
458            fn round_ties_even(self) -> Self { Float(self).round_ties_even().0 }
459
460            fn round_ties_odd(self) -> Self { Float(self).round_ties_odd().0 }
461
462            fn trunc(self) -> Self { Float(self).trunc().0 }

464            fn fract(self) -> Self { Float(self).fract().0 }
465
