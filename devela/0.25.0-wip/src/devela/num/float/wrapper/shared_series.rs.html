39            /// Raises itself to the `y` floating point power using the Taylor series via the
40            /// `exp` and `ln` functions.
41            ///
42            /// # Formulation
43            #[doc = FORMULA_POWF_SERIES!()]
44            ///
45            /// See also [`ln_series_terms`][Self::ln_series_terms].
46            ///
47            /// The terms for the exponential function are calculated using
48            /// [`exp_series_terms`][Self::exp_series_terms] using $y\cdot\ln(x)$.
49            pub const fn powf_series(self, y: $f, ln_x_terms: $ue) -> Float<$f> {
50                let xabs = self.abs().0;
51                if xabs == 0.0 {
52                    is![Float(y).abs().0 == 0.0; Self::ONE; Self::ZERO]
53                } else {
54                    let ln_x = Float(xabs).ln_series(ln_x_terms).0;
55                    let power = Float(y * ln_x);
56                    let exp_y_terms = power.exp_series_terms();
57                    let result = power.exp_series(exp_y_terms);
58                    is![self.is_sign_negative(); Float(-result.0); result]
59                }
60            }

62            /// Computes the exponential function $e^x$ using Taylor series expansion.
63            ///
64            /// # Formulation
65            #[doc = FORMULA_EXP_SERIES!()]
66            ///
67            /// See also [`exp_series_terms`][Self::exp_series_terms].
68            pub const fn exp_series(self, terms: $ue) -> Float<$f> {
69                is![self.0 < 0.0; return Float(1.0 / Float(-self.0).exp_series(terms).0)];
70                let (mut result, mut term) = (1.0, 1.0);
71                let mut i = 1;
72                while i <= terms {
73                    term *= self.0 / i as $f;
74                    result += term;
75                    i += 1;
76                }
77                Float(result)
78            }

87            /// Calculates $ e^x - 1 $ using the Taylor series expansion.
88            ///
89            /// # Formulation
90            #[doc = FORMULA_EXP_M1_SERIES!()]
91            ///
92            /// See also [`exp_series_terms`][Self::exp_series_terms].
93            pub const fn exp_m1_series(self, terms: $ue) -> Float<$f> {
94                if self.0 < 0.0 {
95                    Float(1.0 / Float(-self.0).exp_m1_series(terms).0)
96                } else if self.0 > 0.001 {
97                    Float(self.exp_series(terms).0 - 1.0)
98                } else {
99                    let (mut result, mut term, mut factorial) = (0.0, self.0, 1.0);
100                    let mut i = 1;
101                    while i <= terms {
102                        result += term;
103                        factorial *= (i + 1) as $f;
104                        term *= self.0 / factorial;
105                        i += 1;
106                    }
107                    Float(result)
108                }
109            }

111            /// Calculates $ 2^x $ using the Taylor series expansion.
112            ///
113            /// # Formulation
114            #[doc = FORMULA_EXP2_SERIES!()]
115            ///
116            /// The maximum values with a representable result are:
117            /// 127 for `f32` and 1023 for `f64`.
118            pub const fn exp2_series(self, terms: $ue) -> Float<$f> {
119                let (mut result, mut term) = (1.0, self.0 * Self::LN_2.0);
120                let mut n = 1;
121                while n < terms {
122                    result += term;
123                    term *= self.0 * Self::LN_2.0 / (n as $f + 1.0);
124                    n += 1;
125                }
126                Float(result)
127            }

136            /// Computes the natural logarithm of `self` using a Taylor-Mercator series expansion.
137            ///
138            /// This method is more efficient for values of `self` near 1. Values too
139            /// small or too big could be impractical to calculate with precision.
140            ///
141            /// # Formulation
142            #[doc = FORMULA_LN_SERIES!()]
143            ///
144            /// See also [`ln_series_terms`][Self::ln_series_terms].
145            pub const fn ln_series(self, terms: $ue) -> Float<$f> {
146                if self.0 < 0.0 {
147                    Self::NAN
148                } else if self.0 > 0.0 {
149                    let mut sum = 0.0;
150                    let y = (self.0 - 1.0) / (self.0 + 1.0);
151                    let mut y_pow = y;
152                    cfor![i in 0..terms => {
153                        sum += y_pow / (2 * i + 1) as $f;
154                        y_pow *= y * y;
155                    }];
156                    Float(2.0 * sum)
157                } else {
158                    Self::NEG_INFINITY
159                }
160            }

162            /// Computes the natural logarithm of `1 + self`
163            /// using a Taylor-Mercator series expansion.
164            ///
165            /// This method is more efficient for values of `self` near 0.
166            /// Values too small or too big could be impractical to calculate with precision.
167            ///
168            /// Returns `ln(1+self)` more accurately
169            /// than if the operations were performed separately.
170            ///
171            /// See also [`ln_series_terms`][Self::ln_series_terms].
172            pub const fn ln_1p_series(self, terms: $ue) -> Float<$f> {
173                if self.0 < -1.0 {
174                    Self::NAN
175                } else if self.0 > -1.0 {
176                    let x1 = self.0 + 1.0;
177                    let mut sum = 0.0;
178                    let y = (x1 - 1.0) / (x1 + 1.0);
179                    let mut y_pow = y;
180                    cfor![i in 0..terms => {
181                        sum += y_pow / (2 * i + 1) as $f;
182                        y_pow *= y * y;
183                    }];
184                    Float(2.0 * sum)
185                } else {
186                    Self::NEG_INFINITY
187                }
188            }

190            /// Computes the logarithm to the given `base` using the change of base formula.
191            ///
192            /// # Formulation
193            #[doc = FORMULA_LOG_SERIES!()]
194            ///
195            /// See also [`ln_series_terms`][Self::ln_series_terms].
196            pub const fn log_series(self, base: $f, terms: $ue) -> Float<$f> {
197                if base <= 0.0 {
198                    Self::NAN
199                // The logarithm with a base of 1 is undefined except when the argument is also 1.
200                } else if Float(base - 1.0).abs().0 < Self::MEDIUM_MARGIN.0 { // + robust
201                // } else if base == 1.0 { // good enough for direct input
202                    #[expect(clippy::float_cmp, reason = "we've already checked it with a margin")]
203                    { is![self.0 == 1.0; Self::NAN; Self::NEG_INFINITY] }
204                } else {
205                    Float(self.ln_series(terms).0 / base).ln_series(terms)
206                }
207            }

237            /// The sine calculated using Taylor series expansion.
238            ///
239            /// # Formulation
240            #[doc = FORMULA_SIN_SERIES!()]
241            ///
242            /// This Taylor series converges relatively quickly and uniformly
243            /// over the entire domain.
244            #[doc = TABLE_SIN_SERIES_TERMS!()]
245            // NOTE: OPTIMIZED
246            pub const fn sin_series(self, terms: u32) -> Float<$f> {
247                const PI: $f = Float::<$f>::PI.0;
248                const TAU: $f = Float::<$f>::TAU.0;
249                if terms == 0 { return Self::ZERO; } // Early exit for trivial cases
250                let x = self.0 % TAU; // Reduce angle to [-π, π] while preserving sign
251                let x = is![x > PI; x - TAU; is![x < -PI; x + TAU; x ]];
252                let x_sq = x * x;
253                let (mut sin, mut term, mut factorial) = (x, x, 1.0);
254                let mut i = 1;
255                while i < terms {
256                    term *= -x_sq;
257                    factorial *= ((2*i + 1) * (2*i)) as $f;  // (2n+1)! progression
258                    sin += term / factorial;
259                    i += 1;
260                }
261                Float(sin)
262            }

264            /// Computes the cosine using taylor series expansion.
265            ///
266            /// # Formulation
267            #[doc = FORMULA_COS_SERIES!()]
268            ///
269            /// This Taylor series converges relatively quickly and uniformly
270            /// over the entire domain.
271            #[doc = TABLE_COS_SERIES_TERMS!()]
272            // NOTE: OPTIMIZED
273            pub const fn cos_series(self, terms: $ue) -> Float<$f> {
274                if terms == 0 { return Float::<$f>::ONE; } // Early exit for trivial cases
275                let x = self.0.abs() % Float::<$f>::TAU.0;
276                let x_sq = x * x;
277                let (mut cos, mut term, mut factorial) = (1.0, 1.0, 1.0);
278                let mut i = 1;
279                while i < terms {
280                    term *= -x_sq;
281                    factorial *= ((2*i - 1) * (2*i)) as $f;
282                    cos += term / factorial;
283                    i += 1;
284                }
285                Float(cos)
286            }

290                (self.sin_series(terms), self.cos_series(terms))
291            }

293            /// Computes the tangent using Taylor series expansion of sine and cosine.
294            ///
295            /// # Formulation
296            #[doc = FORMULA_TAN_SERIES!()]
297            ///
298            /// The tangent function has singularities and is not defined for
299            /// `cos(x) = 0`. This function clamps `self` within an appropriate range
300            /// to avoid such issues.
301            ///
302            /// The Taylor series for sine and cosine converge relatively quickly
303            /// and uniformly over the entire domain.
304            #[doc = TABLE_TAN_SERIES_TERMS!()]
305            // NOTE: OPTIMIZED
306            pub const fn tan_series(self, terms: $ue) -> Float<$f> {
307                const PI: $f = Float::<$f>::PI.0;
308                const HALF_PI: $f = Float::<$f>::FRAC_PI_2.0;
309                const THRESHOLD: $f = 1e-6;
310                let x = self.0 % PI; // Reduce angle to [-π/2, π/2]
311                let x = is![x > HALF_PI; x - PI; is![x < -HALF_PI; x + PI; x]];
312                // Handle near-asymptote cases using absolute difference
313                let dist_to_asymptote = (x.abs() - HALF_PI).abs();
314                is![dist_to_asymptote < THRESHOLD; return Float(<$f>::INFINITY.copysign(x))];
315                let (sin, cos) = Float(x).sin_cos_series(terms);
316                Float(sin.0 / cos.0)
317            }

319            /// Computes the arcsine using Taylor series expansion.
320            ///
321            /// # Formulation
322            #[doc = FORMULA_ASIN_SERIES!()]
323            ///
324            /// asin is undefined for $ |x| > 1 $ and in that case returns `NaN`.
325            ///
326            /// The series converges more slowly near the edges of the domain
327            /// (i.e., as `self` approaches -1 or 1). For more accurate results,
328            /// especially near these boundary values, a higher number of terms
329            /// may be necessary.
330            ///
331            /// See also [`asin_series_terms`][Self::asin_series_terms].
332            // NOTE: OPTIMIZED
333            pub const fn asin_series(self, terms: $ue) -> Float<$f> {
334                const HALF_PI: $f = Float::<$f>::FRAC_PI_2.0;
335                const THRESHOLD: $f = 1.0 + Float::<$f>::MEDIUM_MARGIN.0;
336                // Clamp inputs to [-1,1] range (matches std's behavior):
337                let x = is![self.0 >= THRESHOLD; 1.0; is![self.0 <= -THRESHOLD; -1.0; self.0]];
338                // Exact values for endpoints:
339                #[allow(clippy::float_cmp)] {
340                    if x == 1.0 { return Float(HALF_PI); }
341                    if x == -1.0 { return Float(-HALF_PI); }
342                }
343                // Use identity for better convergence near ±1:
344                if x.abs() > 0.9 {
345                    let sign = x.signum();
346                    let x = Float(1.0 - x.abs()).sqrt_hybrid().0;
347                    return Float(sign * (HALF_PI - Self::asin_small_angle(Float(x), terms).0));
348                }
349                // Standard Taylor series for central range
350                Self::asin_small_angle(self, terms)
351            }
352            // Helper for small-angle approximation (|x| ≤ 0.9)
353            const fn asin_small_angle(x: Float<$f>, terms: u32) -> Float<$f> {
354                let x_sq = x.0 * x.0;
355                let (mut result, mut term, mut i) = (x.0, x.0, 1);
356                while i < terms {
357                    term *= x_sq * ((2*i - 1) as $f) / ((2*i) as $f);
358                    result += term / (2*i + 1) as $f;
359                    i += 1;
360                }
361                Float(result)
362            }

394            /// Computes the arctangent using Taylor series expansion.
395            ///
396            /// # Formulation
397            #[doc = FORMULA_ATAN_SERIES!()]
398            ///
399            /// The series converges more slowly near the edges of the domain
400            /// (i.e., as `self` approaches -1 or 1). For more accurate results,
401            /// especially near these boundary values, a higher number of terms
402            /// may be necessary.
403            ///
404            /// See also [`atan_series_terms`][Self::atan_series_terms].
405            pub const fn atan_series(self, terms: $ue) -> Float<$f> {
406                if self.abs().0 > 1.0 {
407                    if self.0 > 0.0 {
408                        Float(Self::FRAC_PI_2.0 - Float(1.0 / self.0).atan_series(terms).0)
409                    } else {
410                        Float(-Self::FRAC_PI_2.0 - Float(1.0 / self.0).atan_series(terms).0)
411                    }
412                } else {
413                    let (mut atan_approx, mut num, mut sign) = (Self::ZERO.0, self.0, Self::ONE.0);
414                    let mut i = 0;
415                    while i < terms {
416                        if i > 0 {
417                            num *= self.0 * self.0;
418                            sign = -sign;
419                        }
420                        atan_approx += sign * num / (2.0 * i as $f + 1.0);
421                        i += 1;
422                    }
423                    Float(atan_approx)
424                }
425            }

435            /// Computes the four quadrant arctangent of `self` and `other`
436            /// using Taylor series expansion.
437            ///
438            /// See also [`atan_series_terms`][Self::atan_series_terms].
439            pub const fn atan2_series(self, other: $f, terms: $ue) -> Float<$f> {
440                if other > 0.0 {
441                    Float(self.0 / other).atan_series(terms)
442                } else if self.0 >= 0.0 && other < 0.0 {
443                    Float(Float(self.0 / other).atan_series(terms).0 + Self::PI.0)
444                } else if self.0 < 0.0 && other < 0.0 {
445                    Float(Float(self.0 / other).atan_series(terms).0 - Self::PI.0)
446                } else if self.0 > 0.0 && other == 0.0 {
447                    Float(Self::PI.0 / 2.0)
448                } else if self.0 < 0.0 && other == 0.0 {
449                    Float(-Self::PI.0 / 2.0)
450                } else {
451                    // self and other are both zero, undefined behavior
452                    Self::NAN
453                }
454            }

520            /// Computes the inverse hyperbolic tangent using the natural logarithm definition.
521            ///
522            /// # Formulation
523            #[doc = FORMULA_ATANH_SERIES!()]
524            ///
525            /// See also [`ln_series_terms`][Self::ln_series_terms].
526            pub const fn atanh_series(self, terms: $ue) -> Float<$f> {
527                if self.0 >= 1.0 {
528                    Self::INFINITY
529                } else if self.0 <= -1.0 {
530                    Self::NEG_INFINITY
531                } else {
532                    Float(Float((self.0 + 1.0) / (1.0 - self.0)).ln_series(terms).0 * 0.5)
533                }
534            }
