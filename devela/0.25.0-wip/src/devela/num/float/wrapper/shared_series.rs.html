30            /// Raises itself to the `y` floating point power using the Taylor series via the
31            /// `exp` and `ln` functions.
32            ///
33            /// # Formulation
34            #[doc = crate::_FLOAT_FORMULA_POWF_SERIES!()]
35            ///
36            /// See also [`ln_series_terms`][Self::ln_series_terms].
37            ///
38            /// The terms for the exponential function are calculated using
39            /// [`exp_series_terms`][Self::exp_series_terms] using $y\cdot\ln(x)$.
40            pub const fn powf_series(self, y: $f, ln_x_terms: $ue) -> Float<$f> {
41                let xabs = self.abs().0;
42                if xabs == 0.0 {
43                    is![Float(y).abs().0 == 0.0; Self::ONE; Self::ZERO]
44                } else {
45                    let ln_x = Float(xabs).ln_series(ln_x_terms).0;
46                    let power = Float(y * ln_x);
47                    let exp_y_terms = power.exp_series_terms();
48                    let result = power.exp_series(exp_y_terms);
49                    is![self.is_sign_negative(); Float(-result.0); result]
50                }
51            }

53            /// Computes the exponential function $e^x$ using Taylor series expansion.
54            ///
55            /// # Formulation
56            #[doc = crate::_FLOAT_FORMULA_EXP_SERIES!()]
57            ///
58            /// See also [`exp_series_terms`][Self::exp_series_terms].
59            pub const fn exp_series(self, terms: $ue) -> Float<$f> {
60                is![self.0 < 0.0; return Float(1.0 / Float(-self.0).exp_series(terms).0)];
61                let (mut result, mut term) = (1.0, 1.0);
62                let mut i = 1;
63                while i <= terms {
64                    term *= self.0 / i as $f;
65                    result += term;
66                    i += 1;
67                }
68                Float(result)
69            }

78            /// Calculates $ e^x - 1 $ using the Taylor series expansion.
79            ///
80            /// # Formulation
81            #[doc = crate::_FLOAT_FORMULA_EXP_M1_SERIES!()]
82            ///
83            /// See also [`exp_series_terms`][Self::exp_series_terms].
84            pub const fn exp_m1_series(self, terms: $ue) -> Float<$f> {
85                if self.0 < 0.0 {
86                    Float(1.0 / Float(-self.0).exp_m1_series(terms).0)
87                } else if self.0 > 0.001 {
88                    Float(self.exp_series(terms).0 - 1.0)
89                } else {
90                    let (mut result, mut term, mut factorial) = (0.0, self.0, 1.0);
91                    let mut i = 1;
92                    while i <= terms {
93                        result += term;
94                        factorial *= (i + 1) as $f;
95                        term *= self.0 / factorial;
96                        i += 1;
97                    }
98                    Float(result)
99                }
100            }

102            /// Calculates $ 2^x $ using the Taylor series expansion.
103            ///
104            /// # Formulation
105            #[doc = crate::_FLOAT_FORMULA_EXP2_SERIES!()]
106            ///
107            /// The maximum values with a representable result are:
108            /// 127 for `f32` and 1023 for `f64`.
109            pub const fn exp2_series(self, terms: $ue) -> Float<$f> {
110                let (mut result, mut term) = (1.0, self.0 * Self::LN_2.0);
111                let mut n = 1;
112                while n < terms {
113                    result += term;
114                    term *= self.0 * Self::LN_2.0 / (n as $f + 1.0);
115                    n += 1;
116                }
117                Float(result)
118            }

127            /// Computes the natural logarithm of `self` using a Taylor-Mercator series expansion.
128            ///
129            /// This method is more efficient for values of `self` near 1. Values too
130            /// small or too big could be impractical to calculate with precision.
131            ///
132            /// # Formulation
133            #[doc = crate::_FLOAT_FORMULA_LN_SERIES!()]
134            ///
135            /// See also [`ln_series_terms`][Self::ln_series_terms].
136            pub const fn ln_series(self, terms: $ue) -> Float<$f> {
137                if self.0 < 0.0 {
138                    Self::NAN
139                } else if self.0 > 0.0 {
140                    let mut sum = 0.0;
141                    let y = (self.0 - 1.0) / (self.0 + 1.0);
142                    let mut y_pow = y;
143                    cfor![i in 0..terms => {
144                        sum += y_pow / (2 * i + 1) as $f;
145                        y_pow *= y * y;
146                    }];
147                    Float(2.0 * sum)
148                } else {
149                    Self::NEG_INFINITY
150                }
151            }

153            /// Computes the natural logarithm of `1 + self`
154            /// using a Taylor-Mercator series expansion.
155            ///
156            /// This method is more efficient for values of `self` near 0.
157            /// Values too small or too big could be impractical to calculate with precision.
158            ///
159            /// Returns `ln(1+self)` more accurately
160            /// than if the operations were performed separately.
161            ///
162            /// See also [`ln_series_terms`][Self::ln_series_terms].
163            pub const fn ln_1p_series(self, terms: $ue) -> Float<$f> {
164                if self.0 < -1.0 {
165                    Self::NAN
166                } else if self.0 > -1.0 {
167                    let x1 = self.0 + 1.0;
168                    let mut sum = 0.0;
169                    let y = (x1 - 1.0) / (x1 + 1.0);
170                    let mut y_pow = y;
171                    cfor![i in 0..terms => {
172                        sum += y_pow / (2 * i + 1) as $f;
173                        y_pow *= y * y;
174                    }];
175                    Float(2.0 * sum)
176                } else {
177                    Self::NEG_INFINITY
178                }
179            }

182            ///
183            /// # Formulation
184            #[doc = crate::_FLOAT_FORMULA_LOG_SERIES!()]
185            ///
186            /// See also [`ln_series_terms`][Self::ln_series_terms].
187            pub const fn log_series(self, base: $f, terms: $ue) -> Float<$f> {
188                if base <= 0.0 {
189                    Self::NAN
190                // The logarithm with a base of 1 is undefined except when the argument is also 1.
191                } else if Float(base - 1.0).abs().0 < Self::MEDIUM_MARGIN.0 { // + robust
192                // } else if base == 1.0 { // good enough for direct input
193                    #[expect(clippy::float_cmp, reason = "we've already checked it with a margin")]
194                    { is![self.0 == 1.0; Self::NAN; Self::NEG_INFINITY] }
195                } else {
196                    Float(self.ln_series(terms).0 / base).ln_series(terms)
197                }
198            }

228            /// The sine calculated using Taylor series expansion.
229            ///
230            /// # Formulation
231            #[doc = crate::_FLOAT_FORMULA_SIN_SERIES!()]
232            ///
233            /// This Taylor series converges relatively quickly and uniformly
234            /// over the entire domain.
235            #[doc = TABLE_SIN_SERIES_TERMS!()]
236            // NOTE: OPTIMIZED
237            pub const fn sin_series(self, terms: u32) -> Float<$f> {
238                const PI: $f = Float::<$f>::PI.0;
239                const TAU: $f = Float::<$f>::TAU.0;
240                if terms == 0 { return Self::ZERO; } // Early exit for trivial cases
241                let x = self.0 % TAU; // Reduce angle to [-π, π] while preserving sign
242                let x = is![x > PI; x - TAU; is![x < -PI; x + TAU; x ]];
243                let x_sq = x * x;
244                let (mut sin, mut term, mut factorial) = (x, x, 1.0);
245                let mut i = 1;
246                while i < terms {
247                    term *= -x_sq;
248                    factorial *= ((2*i + 1) * (2*i)) as $f;  // (2n+1)! progression
249                    sin += term / factorial;
250                    i += 1;
251                }
252                Float(sin)
253            }

255            /// Computes the cosine using taylor series expansion.
256            ///
257            /// # Formulation
258            #[doc = crate::_FLOAT_FORMULA_COS_SERIES!()]
259            ///
260            /// This Taylor series converges relatively quickly and uniformly
261            /// over the entire domain.
262            #[doc = TABLE_COS_SERIES_TERMS!()]
263            // NOTE: OPTIMIZED
264            pub const fn cos_series(self, terms: $ue) -> Float<$f> {
265                if terms == 0 { return Float::<$f>::ONE; } // Early exit for trivial cases
266                let x = self.0.abs() % Float::<$f>::TAU.0;
267                let x_sq = x * x;
268                let (mut cos, mut term, mut factorial) = (1.0, 1.0, 1.0);
269                let mut i = 1;
270                while i < terms {
271                    term *= -x_sq;
272                    factorial *= ((2*i - 1) * (2*i)) as $f;
273                    cos += term / factorial;
274                    i += 1;
275                }
276                Float(cos)
277            }

281                (self.sin_series(terms), self.cos_series(terms))
282            }

284            /// Computes the tangent using Taylor series expansion of sine and cosine.
285            ///
286            /// # Formulation
287            #[doc = crate::_FLOAT_FORMULA_TAN_SERIES!()]
288            ///
289            /// The tangent function has singularities and is not defined for
290            /// `cos(x) = 0`. This function clamps `self` within an appropriate range
291            /// to avoid such issues.
292            ///
293            /// The Taylor series for sine and cosine converge relatively quickly
294            /// and uniformly over the entire domain.
295            #[doc = TABLE_TAN_SERIES_TERMS!()]
296            // NOTE: OPTIMIZED
297            pub const fn tan_series(self, terms: $ue) -> Float<$f> {
298                const PI: $f = Float::<$f>::PI.0;
299                const HALF_PI: $f = Float::<$f>::FRAC_PI_2.0;
300                const THRESHOLD: $f = 1e-6;
301                let x = self.0 % PI; // Reduce angle to [-π/2, π/2]
302                let x = is![x > HALF_PI; x - PI; is![x < -HALF_PI; x + PI; x]];
303                // Handle near-asymptote cases using absolute difference
304                let dist_to_asymptote = (x.abs() - HALF_PI).abs();
305                is![dist_to_asymptote < THRESHOLD; return Float(<$f>::INFINITY.copysign(x))];
306                let (sin, cos) = Float(x).sin_cos_series(terms);
307                Float(sin.0 / cos.0)
308            }

310            /// Computes the arcsine using Taylor series expansion.
311            ///
312            /// # Formulation
313            #[doc = crate::_FLOAT_FORMULA_ASIN_SERIES!()]
314            ///
315            /// asin is undefined for $ |x| > 1 $ and in that case returns `NaN`.
316            ///
317            /// The series converges more slowly near the edges of the domain
318            /// (i.e., as `self` approaches -1 or 1). For more accurate results,
319            /// especially near these boundary values, a higher number of terms
320            /// may be necessary.
321            ///
322            /// See also [`asin_series_terms`][Self::asin_series_terms].
323            // NOTE: OPTIMIZED
324            pub const fn asin_series(self, terms: $ue) -> Float<$f> {
325                const HALF_PI: $f = Float::<$f>::FRAC_PI_2.0;
326                const THRESHOLD: $f = 1.0 + Float::<$f>::MEDIUM_MARGIN.0;
327                // Clamp inputs to [-1,1] range (matches std's behavior):
328                let x = is![self.0 >= THRESHOLD; 1.0; is![self.0 <= -THRESHOLD; -1.0; self.0]];
329                // Exact values for endpoints:
330                #[allow(clippy::float_cmp)] {
331                    if x == 1.0 { return Float(HALF_PI); }
332                    if x == -1.0 { return Float(-HALF_PI); }
333                }
334                // Use identity for better convergence near ±1:
335                if x.abs() > 0.9 {
336                    let sign = x.signum();
337                    let x = Float(1.0 - x.abs()).sqrt_hybrid().0;
338                    return Float(sign * (HALF_PI - Self::asin_small_angle(Float(x), terms).0));
339                }
340                // Standard Taylor series for central range
341                Self::asin_small_angle(self, terms)
342            }
343            // Helper for small-angle approximation (|x| ≤ 0.9)
344            const fn asin_small_angle(x: Float<$f>, terms: u32) -> Float<$f> {
345                let x_sq = x.0 * x.0;
346                let (mut result, mut term, mut i) = (x.0, x.0, 1);
347                while i < terms {
348                    term *= x_sq * ((2*i - 1) as $f) / ((2*i) as $f);
349                    result += term / (2*i + 1) as $f;
350                    i += 1;
351                }
352                Float(result)
353            }

385            /// Computes the arctangent using Taylor series expansion.
386            ///
387            /// # Formulation
388            #[doc = crate::_FLOAT_FORMULA_ATAN_SERIES!()]
389            ///
390            /// The series converges more slowly near the edges of the domain
391            /// (i.e., as `self` approaches -1 or 1). For more accurate results,
392            /// especially near these boundary values, a higher number of terms
393            /// may be necessary.
394            ///
395            /// See also [`atan_series_terms`][Self::atan_series_terms].
396            pub const fn atan_series(self, terms: $ue) -> Float<$f> {
397                if self.abs().0 > 1.0 {
398                    if self.0 > 0.0 {
399                        Float(Self::FRAC_PI_2.0 - Float(1.0 / self.0).atan_series(terms).0)
400                    } else {
401                        Float(-Self::FRAC_PI_2.0 - Float(1.0 / self.0).atan_series(terms).0)
402                    }
403                } else {
404                    let (mut atan_approx, mut num, mut sign) = (Self::ZERO.0, self.0, Self::ONE.0);
405                    let mut i = 0;
406                    while i < terms {
407                        if i > 0 {
408                            num *= self.0 * self.0;
409                            sign = -sign;
410                        }
411                        atan_approx += sign * num / (2.0 * i as $f + 1.0);
412                        i += 1;
413                    }
414                    Float(atan_approx)
415                }
416            }

426            /// Computes the four quadrant arctangent of `self` and `other`
427            /// using Taylor series expansion.
428            ///
429            /// See also [`atan_series_terms`][Self::atan_series_terms].
430            pub const fn atan2_series(self, other: $f, terms: $ue) -> Float<$f> {
431                if other > 0.0 {
432                    Float(self.0 / other).atan_series(terms)
433                } else if self.0 >= 0.0 && other < 0.0 {
434                    Float(Float(self.0 / other).atan_series(terms).0 + Self::PI.0)
435                } else if self.0 < 0.0 && other < 0.0 {
436                    Float(Float(self.0 / other).atan_series(terms).0 - Self::PI.0)
437                } else if self.0 > 0.0 && other == 0.0 {
438                    Float(Self::PI.0 / 2.0)
439                } else if self.0 < 0.0 && other == 0.0 {
440                    Float(-Self::PI.0 / 2.0)
441                } else {
442                    // self and other are both zero, undefined behavior
443                    Self::NAN
444                }
445            }

511            /// Computes the inverse hyperbolic tangent using the natural logarithm definition.
512            ///
513            /// # Formulation
514            #[doc = crate::_FLOAT_FORMULA_ATANH_SERIES!()]
515            ///
516            /// See also [`ln_series_terms`][Self::ln_series_terms].

518                if self.0 >= 1.0 {
519                    Self::INFINITY
520                } else if self.0 <= -1.0 {
521                    Self::NEG_INFINITY
522                } else {
523                    Float(Float((self.0 + 1.0) / (1.0 - self.0)).ln_series(terms).0 * 0.5)
524                }
525            }
