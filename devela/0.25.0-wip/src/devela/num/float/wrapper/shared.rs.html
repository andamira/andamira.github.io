11///
12/// $f:   the floating-point type.
13/// $uf:  unsigned int type with the same bit-size.

15/// $ue:  unsigned int type used for integer exponentiation and number of terms (u32).

83            pub const fn const_round_ties_even(self) -> Float<$f> {
84                let r = self.const_round_ties_away();
85                if r.0 % 2.0 == 0.0 {
86                    r
87                } else {

89                    if Float(self.0 - r.0).abs().0 == 0.5 { // -0.5 < error_margin

97            /// The integral part.
98            /// This means that non-integer numbers are always truncated towards zero.
99            ///
100            /// # Formulation
101            #[doc = FORMULA_TRUNC!()]
102            ///
103            /// This implementation uses bitwise manipulation to remove the fractional part
104            /// of the floating-point number. The exponent is extracted, and a mask is
105            /// created to remove the fractional part. The new bits are then used to create
106            /// the truncated floating-point number.
107            pub const fn const_trunc(self) -> Float<$f> {
108                let bits = self.0.to_bits();
109                const BIAS: $ie = Float::<$f>::EXPONENT_BIAS as $ie;
110                const SIG_BITS: $ie = Float::<$f>::SIGNIFICAND_BITS as $ie;
111                const EXP_MASK: $uf = (1 << Float::<$f>::EXPONENT_BITS) - 1;
112
113                #[allow(clippy::cast_possible_wrap)]
114                let exponent = (((bits >> SIG_BITS) & EXP_MASK) as $ie) - BIAS;
115                if exponent < 0 {
116                    is![self.0.is_sign_positive(); Float(0.0); Float(-0.0)]
117                } else if exponent < SIG_BITS {
118                    let mask = !(((1 as $uf) << (SIG_BITS - exponent)) - 1);
119                    let new_bits = bits & mask;
120                    Float(<$f>::from_bits(new_bits))
121                } else {
122                    self
123                }
124            }

304            /// $ 1 / \sqrt{x} $ the
305            /// [fast inverse square root algorithm](https://en.wikipedia.org/wiki/Fast_inverse_square_root).
306            pub const fn fisr(self) -> Float<$f> {
307                let (mut i, three_halfs, x2) = (self.0.to_bits(), 1.5, self.0 * 0.5);
308                i = Self::FISR_MAGIC - (i >> 1);
309                let y = <$f>::from_bits(i);
310                Float(y * (three_halfs - (x2 * y * y)))
311            }

313            /// $ \sqrt{x} $ The square root calculated calling both [`sqrt_fisr`] and [`sqrt_nr`].
314            ///
315            /// [`sqrt_fisr`]: Self::sqrt_fisr
316            /// [`sqrt_nr`]: Self::sqrt_nr
317            pub const fn sqrt_hybrid(self) -> Float<$f> {
318                is![self.0 < 0.0; return Self::NAN; is![self.0 == 0.0; return Self::ZERO]];
319                let y = self.fisr().0; // fast path using fisr + newton refinement
320                let mut x = self.0 * y; // initial estimate: x ~= sqrt(n)
321                // 1 newton iteration some times can be enough:
322                // Float(0.5 * (x + self.0 / x))
323                // But 2 iterations gives much better precision:
324                cfor! { _ in 0..2 => { x = 0.5 * (x + self.0 / x); }}
325                Float(x)
326            }

328            /// $ \sqrt{x} $ The square root calculated using the
329            /// [Newton-Raphson method](https://en.wikipedia.org/wiki/Newton%27s_method).
330            pub const fn sqrt_nr(self) -> Float<$f> {
331                if self.0 < 0.0 {
332                    Self::NAN
333                } else if self.0 == 0.0 {
334                    Self::ZERO
335                } else {
336                    let mut guess = self.0;
337                    let mut guess_next = 0.5 * (guess + self.0 / guess);
338                    while Self(guess - guess_next).abs().0 > Self::NR_TOLERANCE {
339                        guess = guess_next;
340                        guess_next = 0.5 * (guess + self.0 / guess);
341                    }
342                    Float(guess_next)
343                }
344            }

368            /// $ \sqrt\[3\]{x} $ The cubic root calculated using the
369            /// [Newton-Raphson method](https://en.wikipedia.org/wiki/Newton%27s_method).
370            pub const fn cbrt_nr(self) -> Float<$f> {
371                is![self.0 == 0.0; return self];
372                let mut guess = self.0;
373                loop {
374                    let next_guess = (2.0 * guess + self.0 / (guess * guess)) / 3.0;
375                    if Float(next_guess - guess).abs().0 < Self::NR_TOLERANCE {
376                        break Float(next_guess);
377                    }
378                    guess = next_guess;
379                }
380            }

561            /// Raises itself to the `p` integer power.
562            pub const fn const_powi(self, p: $ie) -> Float<$f> {
563                match p {
564                    0 => Self::ONE,
565                    1.. => {
566                        let mut result = self.0;
567                        cfor! { _ in 1..p => { result *= self.0; }}
568                        Float(result)
569                    }
570                    _ => {
571                        let mut result = self.0;
572                        let abs_p = p.abs();
573                        cfor! { _ in 1..abs_p => { result /= self.0; }}
574                        Float(result)
575                    }
576                }

579            /// Evaluates a polynomial at the `self` point value, using [Horner's method].
580            ///
581            /// Expects a slice of `coefficients` $[a_n, a_{n-1}, ..., a_1, a_0]$
582            /// representing the polynomial $ a_n * x^n + a_{n-1} * x^{(n-1)} + ... + a_1 * x + a_0 $.
583            ///
584            /// # Examples
585            /// ```
586            /// # use devela::Float;
587            /// let coefficients = [2.0, -6.0, 2.0, -1.0];
588            #[doc = cc!["assert_eq![Float(3.0_", sfy![$f], ").eval_poly(&coefficients), 5.0];"]]
589            #[doc = cc!["assert_eq![Float(3.0_", sfy![$f], ").eval_poly(&[]), 0.0];"]]
590            /// ```
591            ///
592            /// [Horner's method]: https://en.wikipedia.org/wiki/Horner%27s_method#Polynomial_evaluation_and_long_division
593            // WAIT: [for-loops in constants](https://github.com/rust-lang/rust/issues/87575)
594            pub const fn eval_poly(self, coefficients: &[$f]) -> Float<$f> {
595                let coef = coefficients;
596                match coef.len() {
597                    0 => Float(0.0),
598                    1 => Float(coef[0]),
599                    _ => {
600                        let mut result = coef[0];
601                        // for &c in &coef[1..] { result = result * self.0 + c; }
602                        cfor! { i in 1..coef.len() => { result = result * self.0 + coef[i]; }}
603                        Float(result)
604                    }
605                }
606            }

608            /// Approximates the derivative of the 1D function `f` at `x` point using step size `h`.
609            ///
610            /// Uses the [finite difference method].
611            ///
612            /// # Formulation
613            #[doc = FORMULA_DERIVATIVE!()]
614            ///
615            // FLAG_DISABLED:nightly_autodiff
616            // See also the [`autodiff`] attr macro, enabled with the `nightly_autodiff` cfg flag.
617            //
618            /// [finite difference method]: https://en.wikipedia.org/wiki/Finite_difference_method
619            // [`autodiff`]: crate::autodiff
620            pub fn derivative<F>(f: F, x: $f, h: $f) -> Float<$f>
621            where
622                F: Fn($f) -> $f,
623            {
624                Float((f(x + h) - f(x)) / h)
625            }

627            /// Approximates the integral of the 1D function `f` over the range `[x, y]`
628            /// using `n` subdivisions.
629            ///
630            /// Uses the [Riemann Sum](https://en.wikipedia.org/wiki/Riemann_sum).
631            ///
632            /// # Formulation
633            #[doc = FORMULA_INTEGRATE!()]
634            pub fn integrate<F>(f: F, x: $f, y: $f, n: usize) -> Float<$f>
635            where
636                F: Fn($f) -> $f,
637            {
638                let dx = (y - x) / n as $f;
639                Float(
640                    (0..n)
641                    .map(|i| { let x = x + i as $f * dx; f(x) * dx })
642                    .sum()
643                )
644            }
