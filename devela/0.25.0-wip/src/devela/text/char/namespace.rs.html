33/// # Methods over `char`
34#[rustfmt::skip]
35impl Char {
36    /// Returns the number of bytes needed to encode the given unicode scalar `code` as UTF-8.
37    ///
38    /// See also [`Char::code_len_utf8`].
39    #[must_use] #[rustfmt::skip]
40    pub const fn len_utf8(code: char) -> usize {
41        let code = code as u32;
42        if code < 0x80 { 1 } else if code < 0x800 { 2 } else if code < 0x10_000 { 3 } else { 4 }
43    }
44    /// Returns the number of bytes needed to encode the given unicode scalar `code` as UTF-8.
45    #[must_use]
46    #[deprecated(since = "0.23.0", note = "Use `len_utf8` instead")]
47    pub const fn len_to_utf8(code: char) -> usize { Self::len_utf8(code) }
48
49    /// Converts the given `char` to a UTF-8 encoded byte sequence.
50    ///
51    /// Always returns a `[u8; 4]` array, with unused bytes set to `0`.
52    ///
53    /// See also [`char::encode_utf8`].
54    #[must_use]
55    pub const fn to_utf8_bytes(c: char) -> [u8; 4] { Char::code_to_utf8_bytes_unchecked(c as u32) }
56
57    /// Returns the ASCII representation as a `&'static str`, or `""` if non-ASCII.
58    #[must_use]
59    pub const fn to_ascii_str(c: char) -> &'static str {
60        if c.is_ascii() { ASCII_TABLE[c as usize] } else { "" }
61    }
62    /// Returns the ASCII representation as a `&'static str`, or panics if non-ASCII.
63    ///
64    /// # Panics
65    /// Panics if the character is not ASCII.
66    #[must_use]
67    pub const fn to_ascii_str_unchecked(c: char) -> &'static str { ASCII_TABLE[c as usize] }
68
69    /// Converts a character to its closest ASCII equivalent, if possible.
70    ///
71    /// This function attempts to replace accented or special characters with
72    /// their ASCII counterparts. If a mapping exists, it returns `Some(char)`,
73    /// otherwise, it returns `None`.
74    #[must_use]
75    pub const fn to_ascii_fold(c: char) -> Option<char> {
76        match c {
77            // ASCII already, return as-is
78            _ if c.is_ascii() => Some(c),
79            // Latin-1 Supplement
80            'À' | 'Á' | 'Â' | 'Ã' | 'Ä' | 'Å' => Some('A'),
81            'à' | 'á' | 'â' | 'ã' | 'ä' | 'å' => Some('a'),
82            'È' | 'É' | 'Ê' | 'Ë' => Some('E'),
83            'è' | 'é' | 'ê' | 'ë' => Some('e'),
84            'Ì' | 'Í' | 'Î' | 'Ï' => Some('I'),
85            'ì' | 'í' | 'î' | 'ï' => Some('i'),
86            'Ò' | 'Ó' | 'Ô' | 'Õ' | 'Ö' => Some('O'),
87            'ò' | 'ó' | 'ô' | 'õ' | 'ö' => Some('o'),
88            'Ù' | 'Ú' | 'Û' | 'Ü' => Some('U'),
89            'ù' | 'ú' | 'û' | 'ü' => Some('u'),
90            'Ý' | 'Ÿ' => Some('Y'),
91            'ý' | 'ÿ' => Some('y'),
92            'Ç' => Some('C'),
93            'ç' => Some('c'),
94            'Ñ' => Some('N'),
95            'ñ' => Some('n'),
96            // Ligatures & Special Cases
97            'Æ' => Some('A'), 'æ' => Some('a'),
98            'Œ' => Some('O'), 'œ' => Some('o'),
99            // Symbols that could be mapped
100            'ß' => Some('s'), // German sharp S → s
101            'Ð' => Some('D'), 'ð' => Some('d'),
102            'Þ' => Some('P'), 'þ' => Some('p'),
103            _ => None, // No reasonable ASCII mapping
104        }
105    }
106    /// Converts a character to its closest ASCII equivalent,
107    /// or returns the input character if no mapping exists.
108    ///
109    /// This function is similar to
110    /// [`to_ascii_fold`][Self::to_ascii_fold], but **never returns `None`**.
111    /// If no ASCII equivalent exists, the input character is returned unchanged.
112    #[must_use]
113    pub const fn to_ascii_fold_unchecked(c: char) -> char {
114        if let Some(m) = Self::to_ascii_fold(c) { m } else { c }
115    }
116}
117/// # Methods over `u32`.
118#[rustfmt::skip]
119impl Char {
120    /// Returns the bytes required to store the given Unicode scalar `code` in a non-UTF encoding.
121    ///
122    /// This function does **not** determine the UTF-8 byte length.
123    /// It assumes a simple encoding where values up to `0xFF` use 1 byte,
124    /// `0x100..=0xFFFF` use 2 bytes, and anything larger uses 3 bytes.
125    #[must_use]
126    pub const fn byte_len(code: u32) -> usize {
127        match code {
128            0x0000..=0x00FF => 1,
129            0x0100..=0xFFFF => 2,
130            _ => 3,
131        }
132    }
133
134    /// Returns the number of bytes required to encode `code` as UTF-8.
135    ///
136    /// Returns `None` if `code` is not a valid Unicode scalar.
137    #[must_use]
138    pub const fn code_len_utf8(code: u32) -> Option<usize> {
139        if Char::is_valid(code) { Some(Char::code_len_utf8_unchecked(code)) } else { None }
140    }
141    /// Returns the UTF-8 byte length of `code` **without validation**.
142    ///
143    /// Assumes `code` is a valid Unicode scalar.
144    /// Use [`code_len_utf8`][Self::code_len_utf8] for a checked version.
145    #[must_use]
146    pub const fn code_len_utf8_unchecked(code: u32) -> usize {
147        match code {
148            0x00_0000..=0x00_007F => 1,
149            0x00_0080..=0x00_07FF => 2,
150            0x00_0800..=0x00_FFFF => 3,
151            _ => 4,
152        }
153    }
154
155    /// Checks if `code` is a valid Unicode scalar (U+0000..=U+10FFFF, excluding surrogates).
156    #[must_use]
157    pub const fn is_valid(code: u32) -> bool {
158        (code <= 0xD7FF) || (code >= 0xE000 && code <= 0x10_FFFF)
159    }
160
161    /// Checks if `code` is a 7-bit ASCII character (U+0000..=U+007F).
162    #[must_use]
163    pub const fn is_7bit(code: u32) -> bool { code <= 0x7F }
164
165    /// Returns `true` if the given unicode scalar `code` is a [noncharacter][0].
166    ///
167    /// [0]: https://www.unicode.org/glossary/#noncharacter
168    #[must_use]
169    // FIXME: make a version that checks for surrogates
170    pub const fn is_noncharacter(code: u32) -> bool {
171        // sub-block of 32 non-characters:
172        (code >= 0xFDD0 && code <= 0xFDEF)
173            // 2× non-characters at the end of each plane:
174            || (code >= 0xFFFE && (code & 0xFF) == 0xFE)
175            || (code >= 0xFFFE && (code & 0xFF) == 0xFF)
176            // unallocated range (16 potential non-characters):
177            || (code >= 0x2FE0 && code <= 0x2FEF)
178    }
179
180    /// Returns the ASCII representation as a `&'static str`, or `""` if non-ASCII.
181    #[must_use]
182    pub const fn code_to_ascii_str(c: u32) -> &'static str {
183        if Char::is_7bit(c) { ASCII_TABLE[c as usize] } else { "" }
184    }
185    /// Returns the ASCII representation as a `&'static str`, or panics if non-ASCII.
186    ///
187    /// # Panics
188    /// Panics if the character is not ASCII.
189    #[must_use]
190    pub const fn code_to_ascii_str_unchecked(c: u32) -> &'static str { ASCII_TABLE[c as usize] }
191
192    /// Converts the Unicode scalar `code` to a UTF-8 encoded byte sequence.
193    ///
194    /// Returns `None` if `code` is not a valid Unicode scalar.
195    /// The result is always a `[u8; 4]` array, with unused bytes set to `0`.
196    ///
197    /// See also [`char::encode_utf8`].
198    pub const fn code_to_utf8_bytes(code: u32) -> Option<[u8; 4]> {
199        if Char::is_valid(code) { Some(Char::code_to_utf8_bytes_unchecked(code)) } else { None }
200    }
201
202    /// Converts the Unicode scalar `code` to a UTF-8 encoded byte sequence **without validation**.
203    ///
204    /// Assumes `code` is a valid Unicode scalar.
205    /// Always returns a `[u8; 4]` array, with unused bytes set to `0`.
206    ///
207    /// See also [`Char::code_to_utf8_bytes`] for a checked version.
208    #[must_use]
209    #[allow(clippy::unusual_byte_groupings)]
210    pub const fn code_to_utf8_bytes_unchecked(code: u32) -> [u8; 4] {
211        match code {
212            // From 0x0000 to 0x007F:
213            // the UTF-8 encoding is the same as the scalar value.
214            0x0000..=0x007F => [code as u8, 0, 0, 0],
215
216            // from 0x0080 to 0x07FF:
217            // the UTF-8 encoding is 110xxxxx 10xxxxxx,
218            // where xxxxx and xxxxxx are the bits of the scalar value.
219            0x0080..=0x07FF => {
220                let y = 0b10_000000 | (0b0011_1111 & (code as u8));
221                let x = 0b110_00000 | ((code >> 6) as u8);
222                [x, y, 0, 0]
223            }
224
225            // From from 0x0800 to 0xFFFF:
226            // the UTF-8 encoding is 1110xxxx 10xxxxxx 10xxxxxx.
227            0x0800..=0xFFFF => {
228                let z = 0b10_000000 | (0b0011_1111 & (code as u8));
229                let y = 0b10_000000 | ((code >> 6) & 0b0011_1111) as u8;
230                let x = 0b1110_0000 | ((code >> 12) as u8);
231                [x, y, z, 0]
232            }
233
234            // From 0x10000 to 0x10FFFF:
235            // the UTF-8 encoding is 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx.
236            _ => {
237                let w = 0b10_000000 | (0b0011_1111 & (code as u8));
238                let z = 0b10_000000 | ((code >> 6) & 0b0011_1111) as u8;
239                let y = 0b10_000000 | ((code >> 12) & 0b0011_1111) as u8;
240                let x = 0b11110_000 | ((code >> 18) as u8);
241                [x, y, z, w]
242            }
243        }
244    }
245}

247/// # Methods over `u8`.
248#[rustfmt::skip]
249impl Char {
250    /// Returns the expected UTF-8 byte length based on the first byte.
251    ///
252    /// This function does **not** validate UTF-8 but determines how many bytes
253    /// a valid sequence **should** occupy based on the leading byte.
254    ///
255    /// - ASCII (0xxxxxxx) → 1 byte
256    /// - 2-byte (110xxxxx) → 2 bytes
257    /// - 3-byte (1110xxxx) → 3 bytes
258    /// - 4-byte (11110xxx) → 4 bytes
259    ///
260    /// ### Caveat
261    /// - If used on malformed UTF-8, it may suggest a length longer than the actual valid sequence.
262    /// - Always use in conjunction with proper UTF-8 validation if handling untrusted input.
263    ///
264    /// For a stricter check, see [`utf8_len_checked`][Self::utf8_len_checked].
265    pub const fn utf8_len(first_byte: u8) -> u8 {
266        match first_byte {
267            0x00..=0x7F => 1, // ASCII (1 byte)
268            0xC0..=0xDF => 2, // 2-byte sequence
269            0xE0..=0xEF => 3, // 3-byte sequence
270            0xF0..=0xF7 => 4, // 4-byte sequence
271            _ => 0,           // Invalid leading byte
272        }
273    }
274
275    /// Returns the UTF-8 byte length or `None` if the first byte is invalid.
276    ///
277    /// This function detects invalid UTF-8 leading bytes and ensures
278    /// they fall within valid Unicode scalar boundaries.
279    ///
280    /// - Returns `Some(len)` for valid leading bytes.
281    /// - Returns `None` for invalid first bytes that cannot start a UTF-8 sequence.
282    ///
283    /// ### Stricter Handling
284    /// - Rejects overlong sequences (C0, C1).
285    /// - Enforces the valid UTF-8 upper bound (max `F4`).
286    /// - Safer for processing untrusted input where malformed UTF-8 must be detected.
287    ///
288    /// For a simpler length-only function, see [`utf8_len`][Self::utf8_len].
289    #[must_use]
290    pub const fn utf8_len_checked(first_byte: u8) -> Option<u8> {
291        match first_byte {
292            0x00..=0x7F => Some(1),
293            0xC2..=0xDF => Some(2),
294            0xE0..=0xEF => Some(3),
295            0xF0..=0xF4 => Some(4),
296            _ => None,
297        }
298    }
299
300    /// Returns the number of bytes needed to store the given unicode scalar `code`,
301    /// already UTF-8 encoded in 2 bytes.
302    #[must_use]
303    #[deprecated(since = "0.23.0", note = "Use `utf8_len` instead")]
304    pub const fn utf8_2bytes_len(code: [u8; 2]) -> u8 {
305        1 + ((code[1] > 0) & (code[1] & 0b1100_0000 != 0b1000_0000)) as u8
306    }
307
308    /// Returns the number of bytes needed to store the given unicode scalar `code`,
309    /// already UTF-8 encoded in 3 bytes.
310    #[must_use]
311    #[deprecated(since = "0.23.0", note = "Use `utf8_len` instead")]
312    pub const fn utf8_3bytes_len(code: [u8; 3]) -> u8 {
313        1 + ((code[1] > 0) & (code[1] & 0b1100_0000 != 0b1000_0000)) as u8
314            + ((code[2] > 0) & (code[2] & 0b1100_0000 != 0b1000_0000)) as u8
315    }
316
317    /// Returns the number of bytes needed to store the given unicode scalar `code`,
318    /// already UTF-8 encoded in 4 bytes.
319    #[must_use]
320    #[deprecated(since = "0.23.0", note = "Use `utf8_len` instead")]
321    pub const fn utf8_4bytes_len(code: [u8; 4]) -> u8 {
322        1 + ((code[1] > 0) & (code[1] & 0b1100_0000 != 0b1000_0000)) as u8
323            + ((code[2] > 0) & (code[2] & 0b1100_0000 != 0b1000_0000)) as u8
324            + ((code[3] > 0) & (code[3] & 0b1100_0000 != 0b1000_0000)) as u8
325    }
326
327    /// Decodes a UTF-8 code point from `bytes`, starting at `index`.
328    ///
329    /// Returns `Some((code, len))` if the input is a valid UTF-8 sequence
330    /// and the decoded code point is a valid Unicode scalar.
331    ///
332    /// Returns `None` if:
333    /// - The index is out of bounds.
334    /// - The bytes do not form a valid UTF-8 sequence.
335    /// - The decoded value is not a valid Unicode scalar.
336    ///
337    /// # Example
338    /// ```
339    /// # use devela::Char;
340    /// assert_eq!(Char::utf8_bytes_to_code("Ħ".as_bytes(), 0), Some((u32::from('Ħ'), 2)));
341    ///
342    /// let invalid = b"\x80"; // Invalid leading byte
343    /// assert_eq!(Char::utf8_bytes_to_code(invalid, 0), None);
344    /// ```
345    #[must_use]
346    pub const fn utf8_bytes_to_code(bytes: &[u8], index: usize) -> Option<(u32, usize)> {
347        if index >= bytes.len() { return None; } // out of bounds
348        let first = bytes[index];
349        let len = unwrap![some? Char::utf8_len_checked(first)]; // invalid leading byte?
350        if index + (len as usize) > bytes.len() { return None; } // not enough bytes
351        Some(Char::utf8_bytes_to_code_unchecked(bytes, index))
352    }
353    /// Decodes a UTF-8 code point from `bytes`, starting at `index`.
354    ///
355    /// Returns `(code, len)`, where `code` is the decoded Unicode scalar,
356    /// and `len` is the number of bytes consumed.
357    ///
358    /// Assumes `bytes[index..]` contains a valid UTF-8 sequence.
359    #[must_use]
360    pub const fn utf8_bytes_to_code_unchecked(bytes: &[u8], index: usize) -> (u32, usize) {
361        let first = bytes[index];
362        match first {
363            0x00..=0x7F => (first as u32, 1), // 1-byte ASCII
364            0xC2..=0xDF => ( // 2-byte UTF-8
365                (((first as u32 & 0b0001_1111) << 6) |
366                 (bytes[index + 1] as u32 & 0b0011_1111)),
367                2
368            ),
369            0xE0..=0xEF => ( // 3-byte UTF-8
370                (((first as u32 & 0b0000_1111) << 12) |
371                 ((bytes[index + 1] as u32 & 0b0011_1111) << 6) |
372                 (bytes[index + 2] as u32 & 0b0011_1111)),
373                3
374            ),
375            0xF0..=0xF4 => ( // 4-byte UTF-8
376                (((first as u32 & 0b0000_0111) << 18) |
377                 ((bytes[index + 1] as u32 & 0b0011_1111) << 12) |
378                 ((bytes[index + 2] as u32 & 0b0011_1111) << 6) |
379                 (bytes[index + 3] as u32 & 0b0011_1111)),
380                4
381            ),
382            _ => (0xFFFD, 1), // Invalid sequence → Return Unicode replacement char (U+FFFD)
383        }
384    }
385}

387#[cfg(test)]
388mod tests {
389    use super::Char;
390
391    #[test]
392    fn utf8_bytes_to_code() {
393        // Single ASCII character
394        assert_eq!(Char::utf8_bytes_to_code(b"a", 0), Some((97, 1))); // 'a' -> U+0061
395
396        // Multi-byte UTF-8 character
397        let bytes = "Ħ".as_bytes(); // 'Ħ' (U+0126) -> [0xC4, 0xA6]
398        assert_eq!(Char::utf8_bytes_to_code(bytes, 0), Some((0x0126, 2)));
399        assert_eq!(char::from_u32(0x0126), Some('Ħ'));
400
401        // 3-byte UTF-8
402        let bytes = "✓".as_bytes(); // '✓' (U+2713) -> [0xE2, 0x9C, 0x93]
403        assert_eq!(Char::utf8_bytes_to_code(bytes, 0), Some((0x2713, 3)));
404
405        // 4-byte UTF-8
406        let bytes = "🚀".as_bytes(); // '🚀' (U+1F680) -> [0xF0, 0x9F, 0x9A, 0x80]
407        assert_eq!(Char::utf8_bytes_to_code(bytes, 0), Some((0x1F680, 4)));
408
409        // Invalid byte sequence
410        let invalid = b"\x80"; // Invalid leading byte
411        assert_eq!(Char::utf8_bytes_to_code(invalid, 0), None);
412
413        let incomplete = b"\xE2\x9C"; // Incomplete 3-byte sequence
414        assert_eq!(Char::utf8_bytes_to_code(incomplete, 0), None);
415    }
