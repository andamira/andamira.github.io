193    /// Returns a [`&str`] backed by a `buffer`, where you always know each
194    /// character's position.
195    ///
196    /// A [*counter string*][0] is a graduated string of arbitrary `length`,
197    /// with a `separator` positioned after the immediately preceding number.
198    /// # Examples
199    /// ```
200    /// # use devela::Str;
201    /// let mut buf = [0; 15];
202    /// assert_eq!("2*4*6*8*11*14*", Str::new_counter(&mut buf, 14, '*'));
203    /// assert_eq!("_3_5_7_9_12_15_", Str::new_counter(&mut buf, 15, '_'));
204    /// ```
205    /// # Panics
206    /// Panics if `buffer.len() < length`, or if `!char.is_ascii()`.
207    ///
208    /// # Features
209    /// Makes use of the `unsafe_str` feature if enabled.
210    ///
211    /// See also [`ExtStr::new_counter`].
212    ///
213    /// [0]: https://www.satisfice.com/blog/archives/22
214    #[cfg(feature = "str")]
215    #[cfg_attr(nightly_doc, doc(cfg(feature = "str")))]
216    pub const fn new_counter(buffer: &mut [u8], length: usize, separator: char) -> &str {
217        assert![buffer.len() >= length];
218        assert![separator.is_ascii()];
219        if length == 0 {
220            Str::new_cold_empty()
221        } else {
222            let separator = separator as u8;
223            let mut index = length - 1; // start writing from the end
224            let mut num = length; // the first number to write is the length
225            let mut separator_turn = true; // start writing the separator
226
227            let mut num_buf = Ascii(num).digits();
228            let mut num_bytes = Slice::trim_leading_bytes(&num_buf, b'0');
229            // IMPROVE:BENCH use NumToStr
230            // let mut num_buf = [0u8; 22];
231            // let mut num_bytes = num.to_bytes_base(10, &mut num_buf);
232
233            let mut num_len = num_bytes.len();
234
235            loop {
236                if separator_turn {
237                    buffer[index] = separator;
238                } else {
239                    is![index > 0; index -= num_len - 1];
240                    // WAIT:1.87 [const_copy_from_slice](https://github.com/rust-lang/rust/issues/131415)
241                    // buffer[index..(num_len + index)].copy_from_slice(&num_bytes[..num_len]);
242                    // Slice::range_mut(buffer, index, num_len + index)
243                    //     .copy_from_slice(Slice::range_to(num_bytes, num_len));
244                    let mut i = 0;
245                    while i < num_len {
246                        buffer[index + i] = num_bytes[i];
247                        i += 1;
248                    }
249
250                    num = index;
251
252                    num_buf = Ascii(num).digits();
253                    num_bytes = Slice::trim_leading_bytes(&num_buf, b'0');
254                    // IMPROVE: use NumToStr
255                    // num_bytes = num.to_bytes_base(10, &mut num_buf);
256
257                    num_len = num_bytes.len();
258                }
259                is![index == 0; break; index -= 1];
260                separator_turn = !separator_turn;
261            }
262
263            #[cfg(any(feature = "safe_text", not(feature = "unsafe_str")))]
264            return unwrap![ok Str::from_utf8(Slice::range_to(buffer, length))];
265            #[cfg(all(not(feature = "safe_text"), feature = "unsafe_str"))]
266            // SAFETY: TODO: since `string` is a valid &str, checks are unneeded.
267            sf! { unsafe { Str::from_utf8_unchecked(Slice::range_to(buffer, length)) }}
268        }
269    }
