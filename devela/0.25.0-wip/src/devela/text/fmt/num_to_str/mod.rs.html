15/// Converts a number into a string representation, storing it into a byte slice.
16///
17/// # Features
18/// It makes use of the `unsafe_str` feature for faster unchecked conversion of
19/// the resulting bytes to a string slice, and of the `dep_simdutf8` dependency.
20///
21#[doc = crate::_doc!(vendor: "numtoa")]
22pub trait NumToStr<T> {
23    /// Given a base for encoding and a mutable byte slice, write the number
24    /// into the byte slice and return the indice where the inner string begins.
25    /// The inner string can be extracted by slicing the byte slice from
26    /// that indice.
27    ///
28    /// # Panics
29    /// If the supplied buffer is smaller than the number of bytes needed to
30    /// write the integer, this will panic. On debug builds, this function will
31    /// perform a check on base 10 conversions to ensure that the input array
32    /// is large enough to hold the largest possible value in digits.
33    ///
34    /// # Example
35    /// ```
36    /// use devela::NumToStr;
37    /// use std::io::{self, Write};
38    ///
39    /// let stdout = io::stdout();
40    /// let stdout = &mut io::stdout();
41    ///
42    /// // Allocate a buffer that will be reused in each iteration.
43    /// let mut buffer = [0u8; 20];
44    ///
45    /// let number = 15325;
46    /// let _ = stdout.write(number.to_bytes_base(10, &mut buffer));
47    ///
48    /// let number = 1241;
49    /// let _ = stdout.write(number.to_bytes_base(10, &mut buffer));
50    ///
51    /// assert_eq!(12345.to_bytes_base(10, &mut buffer), b"12345");
52    /// ```
53    fn to_bytes_base(self, base: T, string: &mut [u8]) -> &[u8];
54
55    /// Convenience method for quickly getting a string from the input's array buffer.
56    fn to_str_base(self, base: T, buf: &mut [u8]) -> &str;
57}

59// A lookup table to prevent the need for conditional branching
60// The value of the remainder of each step will be used as the index
61const LOOKUP: &[u8] = b"0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
62
63// A lookup table optimized for decimal lookups.
64// Each two indices represents one possible number.
65const DEC_LOOKUP: &[u8; 200] = b"0001020304050607080910111213141516171819\
66                                 2021222324252627282930313233343536373839\
67                                 4041424344454647484950515253545556575859\
68                                 6061626364656667686970717273747576777879\
69                                 8081828384858687888990919293949596979899";

71macro_rules! base_10 {
72    ($number:ident, $index:ident, $string:ident) => {
73        // Decode four characters at the same time
74        while $number > 9999 {
75            let rem = ($number % 10000) as u16;
76            let (frst, scnd) = ((rem / 100) * 2, (rem % 100) * 2);
77            $string[$index - 3..$index - 1]
78                .copy_from_slice(&DEC_LOOKUP[frst as usize..frst as usize + 2]);
79            $string[$index - 1..$index + 1]
80                .copy_from_slice(&DEC_LOOKUP[scnd as usize..scnd as usize + 2]);
81            $index = $index.wrapping_sub(4);
82            $number /= 10000;
83        }
84
85        if $number > 999 {
86            let (frst, scnd) = (($number / 100) * 2, ($number % 100) * 2);
87            $string[$index - 3..$index - 1]
88                .copy_from_slice(&DEC_LOOKUP[frst as usize..frst as usize + 2]);
89            $string[$index - 1..$index + 1]
90                .copy_from_slice(&DEC_LOOKUP[scnd as usize..scnd as usize + 2]);
91            $index = $index.wrapping_sub(4);
92        } else if $number > 99 {
93            let section = ($number as u16 / 10) * 2;
94            $string[$index - 2..$index]
95                .copy_from_slice(&DEC_LOOKUP[section as usize..section as usize + 2]);
96            $string[$index] = LOOKUP[($number % 10) as usize];
97            $index = $index.wrapping_sub(3);
98        } else if $number > 9 {
99            $number *= 2;
100            $string[$index - 1..$index + 1]
101                .copy_from_slice(&DEC_LOOKUP[$number as usize..$number as usize + 2]);
102            $index = $index.wrapping_sub(2);
103        } else {
104            $string[$index] = LOOKUP[$number as usize];
105            $index = $index.wrapping_sub(1);
106        }
107    };
108}

110macro_rules! impl_primitive {
111    ( signed $($t:ty),+ ) => { $( impl_primitive![@signed $t]; )+ };
112    ( unsigned $($t:ty),+ ) => { $( impl_primitive![@unsigned $t]; )+ };
113    (@signed $t:ty) => {
114        impl NumToStr<$t> for $t {
115            fn to_bytes_base(mut self, base: $t, string: &mut [u8]) -> &[u8] {
116                if cfg!(debug_assertions) {
117                    if base == 10 {
118                        match size_of::<$t>() {
119                            2 => debug_assert![string.len() >= 6,
120                                "i16 base 10 conversions require at least 6 bytes"],
121                            4 => debug_assert![string.len() >= 11,
122                                "i32 base 10 conversions require at least 11 bytes"],
123                            8 => debug_assert![string.len() >= 20,
124                                "i64 base 10 conversions require at least 20 bytes"],
125                            _ => unreachable![],
126                        }
127                    }
128                }
129                let mut index = string.len() - 1;
130                let mut is_negative = false;
131                if self < 0 {
132                    is_negative = true;
133                    self = match self.checked_abs() {
134                        Some(value) => value,
135                        None => {
136                            let value = <$t>::MAX;
137                            string[index] = LOOKUP[((value % base + 1) % base) as usize];
138                            index -= 1;
139                            value / base + <$t>::from(value % base == base -1)
140                        }
141                    };
142                } else if self == 0 {
143                    string[index] = b'0';
144                    return &string[index..];
145                }
146                if base == 10 {
147                    // Convert using optimized base 10 algorithm
148                    base_10!(self, index, string);
149                } else {
150                    while self != 0 {
151                        let rem = self % base;
152                        string[index] = LOOKUP[rem as usize];
153                        index = index.wrapping_sub(1);
154                        self /= base;
155                    }
156                }
157                if is_negative {
158                    string[index] = b'-';
159                    index = index.wrapping_sub(1);
160                }
161                &string[index.wrapping_add(1)..]
162            }
163            fn to_str_base(self, base: $t, buf: &mut [u8]) -> &str {
164                #[cfg(any(feature = "safe_text", not(feature = "unsafe_str")))]
165                return from_utf8(self.to_bytes_base(base, buf)).unwrap();
166
167                #[cfg(all(not(feature = "safe_text"), feature = "unsafe_str"))]
168                // SAFETY: the bytes are valid utf-8
169                unsafe { from_utf8_unchecked(self.to_bytes_base(base, buf)) }
170            }
171        }
172    };
173    (@unsigned $t:ty) => {
174        impl NumToStr<$t> for $t {
175            fn to_bytes_base(mut self, base: $t, string: &mut [u8]) -> &[u8] {
176                // Check if the buffer is large enough and panic on debug builds if it isn't
177                if cfg!(debug_assertions) {
178                    if base == 10 {
179                        match size_of::<$t>() {
180                            2 => debug_assert![ string.len() >= 5,
181                                "u16 base 10 conversions require at least 5 bytes"],
182                            4 => debug_assert![ string.len() >= 10,
183                                "u32 base 10 conversions require at least 10 bytes"],
184                            8 => debug_assert![ string.len() >= 20,
185                                "u64 base 10 conversions require at least 20 bytes"],
186                            _ => unreachable![],
187                        }
188                    }
189                }
190                let mut index = string.len() - 1;
191                if self == 0 {
192                    string[index] = b'0';
193                    return &string[index..];
194                }
195                if base == 10 {
196                    // Convert using optimized base 10 algorithm
197                    base_10!(self, index, string);
198                } else {
199                    while self != 0 {
200                        let rem = self % base;
201                        string[index] = LOOKUP[rem as usize];
202                        index = index.wrapping_sub(1);
203                        self /= base;
204                    }
205                }
206                &string[index.wrapping_add(1)..]
207            }
208            fn to_str_base(self, base: $t, buf: &mut [u8]) -> &str {
209                #[cfg(any(feature = "safe_text", not(feature = "unsafe_str")))]
210                return from_utf8(self.to_bytes_base(base, buf)).unwrap();
211
212                #[cfg(all(not(feature = "safe_text"), feature = "unsafe_str"))]
213                // SAFETY: the bytes are valid utf-8
214                unsafe { from_utf8_unchecked(self.to_bytes_base(base, buf)) }
215            }
216        }
217    };

222impl NumToStr<i8> for i8 {
223    fn to_bytes_base(mut self, base: i8, string: &mut [u8]) -> &[u8] {
224        if cfg!(debug_assertions) && base == 10 {
225            debug_assert!(string.len() >= 4, "i8 conversions need at least 4 bytes");
226        }
227        let mut index = string.len() - 1;
228        let mut is_negative = false;
229        #[allow(clippy::comparison_chain)]
230        if self < 0 {
231            is_negative = true;
232            self = if let Some(value) = self.checked_abs() {
233                value
234            } else {
235                let value = <i8>::MAX;
236                string[index] = LOOKUP[((value % base + 1) % base) as usize];
237                index -= 1;
238                value / base + ((value % base == base - 1) as i8)
239            };
240        } else if self == 0 {
241            string[index] = b'0';
242            return &string[index..];
243        }
244        if base == 10 {
245            if self > 99 {
246                let section = (self / 10) * 2;
247                string[index - 2..index]
248                    .copy_from_slice(&DEC_LOOKUP[section as usize..section as usize + 2]);
249                string[index] = LOOKUP[(self % 10) as usize];
250                index = index.wrapping_sub(3);
251            } else if self > 9 {
252                self *= 2;
253                string[index - 1..index + 1]
254                    .copy_from_slice(&DEC_LOOKUP[self as usize..self as usize + 2]);
255                index = index.wrapping_sub(2);
256            } else {
257                string[index] = LOOKUP[self as usize];
258                index = index.wrapping_sub(1);
259            }
260        } else {
261            while self != 0 {
262                let rem = self % base;
263                string[index] = LOOKUP[rem as usize];
264                index = index.wrapping_sub(1);
265                self /= base;
266            }
267        }
268        if is_negative {
269            string[index] = b'-';
270            index = index.wrapping_sub(1);
271        }
272        &string[index.wrapping_add(1)..]
273    }

275    fn to_str_base(self, base: Self, buf: &mut [u8]) -> &str {
276        #[cfg(any(feature = "safe_text", not(feature = "unsafe_str")))]
277        return from_utf8(self.to_bytes_base(base, buf)).unwrap();
278
279        #[cfg(all(not(feature = "safe_text"), feature = "unsafe_str"))]
280        // SAFETY: the bytes are valid utf-8
281        unsafe {
282            from_utf8_unchecked(self.to_bytes_base(base, buf))
283        }
284    }
