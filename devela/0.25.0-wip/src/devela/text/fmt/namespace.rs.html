20impl Fmt {
21    /// Takes an [`FmtArguments`] struct and returns the resulting formatted string.
22    ///
23    /// The `FmtArguments` instance can be created with the [`format_args!`] macro.
24    ///
25    /// See `alloc::fmt::`[`format`][fn@format].
26    ///
27    /// [`format_args!`]: crate::format_args
28    #[must_use]
29    #[cfg(feature = "alloc")]
30    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
31    pub fn format(args: FmtArguments<'_>) -> String {
32        format(args)
33    }
34
35    /// Writes formatted output into the given byte buffer.
36    ///
37    /// Returns:
38    /// - `Ok(&str)` if all the formatted data fits into `buf`.
39    /// - `Err(&str)` containing the valid partial result if truncation occurred.
40    ///
41    /// # Example
42    /// ```
43    /// # use devela::Fmt;
44    /// let mut buf = [0u8; 32]; // Big enough to fit everything
45    /// let s = Fmt::format_buf(&mut buf, format_args!["Test: {} {}", "foo", 42]);
46    /// assert_eq!(Ok("Test: foo 42"), s);
47    ///
48    /// let mut buf = [0u8; 9]; // Can't fit everything
49    /// let s = Fmt::format_buf(&mut buf, format_args!["Test: {} {}", "foo", 42]);
50    /// assert_eq!(Err("Test: foo"), s);
51    /// ```
52    pub fn format_buf<'a>(buf: &'a mut [u8], args: FmtArguments) -> Result<&'a str, &'a str> {
53        let mut w = WriteTo::new(buf);
54        let _ = Fmt::write(&mut w, args);
55        if w.truncated { Err(w.as_str()) } else { Ok(w.as_str()) }
56    }
57
58    /// Takes an output stream and an `FmtArguments` struct
59    /// that can be precompiled with the [`format_args!`] macro.
60    ///
61    /// The arguments will be formatted according to the specified format string
62    /// into the output stream provided.
63    ///
64    /// See `core::fmt::`[`write`][fn@write].
65    ///
66    /// [`format_args!`]: crate::format_args
67    pub fn write(output: &mut dyn FmtWrite, args: FmtArguments<'_>) -> Result<(), FmtError> {
68        write(output, args)
69    }
