13/// A utility struct for searching bytes in slices.
14///
15/// # Features
16/// - Supports efficient search for bytes with `dep_memchr` enabled.
17/// - Provides fallback naive implementations otherwise.
18#[derive(Debug)]
19pub struct ByteSearch;
20
21impl ByteSearch {
22    /// Search for the first occurrence of a byte in a slice.
23    ///
24    /// # Features
25    /// Makes use of the `dep_memchr` dependency if enabled. See [memchr],
26    ///
27    /// [memchr]: fn@memchr
28    #[must_use] #[rustfmt::skip]
29    pub fn first1(needle: u8, haystack: &[u8]) -> Option<usize> {
30        #[cfg(feature = "dep_memchr")]
31        return memchr(needle, haystack);
32        #[cfg(not(feature = "dep_memchr"))]
33        haystack.iter().enumerate()
34            .find_map(|(index, &byte)| is![byte == needle; Some(index); None])
35    }
36
37    /// Search for the first occurrence of two possible bytes in a haystack.
38    ///
39    /// # Features
40    /// Makes use of the `dep_memchr` dependency if enabled. See [memchr2],
41    #[must_use] #[rustfmt::skip]
42    pub fn first2(needle1: u8, needle2: u8, haystack: &[u8]) -> Option<usize> {
43        #[cfg(feature = "dep_memchr")]
44        return memchr2(needle1, needle2, haystack);
45        #[cfg(not(feature = "dep_memchr"))]
46        haystack.iter().enumerate()
47            .find_map(|(index, &byte)| is![byte == needle1 || byte == needle2; Some(index); None])
48    }
49
50    /// Search for the first occurrence of three possible bytes in a haystack.
51    ///
52    /// # Features
53    /// Makes use of the `dep_memchr` dependency if enabled. See [memchr3],
54    #[must_use]
55    pub fn first3(needle1: u8, needle2: u8, needle3: u8, haystack: &[u8]) -> Option<usize> {
56        #[cfg(feature = "dep_memchr")]
57        return memchr3(needle1, needle2, needle3, haystack);
58        #[cfg(not(feature = "dep_memchr"))]
59        haystack.iter().enumerate().find_map(|(index, &byte)| {
60            is![byte == needle1 || byte == needle2 || byte == needle3; Some(index); None]
61        })
62    }
63
64    /// Search for the last occurrence of a byte in a slice.
65    ///
66    /// # Features
67    /// Makes use of the `dep_memchr` dependency if enabled. See [memrchr],
68    #[must_use] #[rustfmt::skip]
69    pub fn last1(needle: u8, haystack: &[u8]) -> Option<usize> {
70        #[cfg(feature = "dep_memchr")]
71        return memrchr(needle, haystack);
72        #[cfg(not(feature = "dep_memchr"))]
73        haystack.iter().enumerate().rev()
74            .find_map(|(index, &byte)| is![byte == needle; Some(index); None])
75    }
76
77    /// Search for the last occurrence of two possible bytes in a haystack.
78    ///
79    /// # Features
80    /// Makes use of the `dep_memchr` dependency if enabled. See [memrchr2],
81    #[must_use] #[rustfmt::skip]
82    pub fn last2(needle1: u8, needle2: u8, haystack: &[u8]) -> Option<usize> {
83        #[cfg(feature = "dep_memchr")]
84        return memrchr2(needle1, needle2, haystack);
85        #[cfg(not(feature = "dep_memchr"))]
86        haystack.iter().enumerate().rev()
87            .find_map(|(index, &byte)| is![byte == needle1 || byte == needle2; Some(index); None])
88    }
89
90    /// Search for the last occurrence of three possible bytes in a haystack.
91    ///
92    /// # Features
93    /// Makes use of the `dep_memchr` dependency if enabled. See [memrchr3],
94    #[must_use]
95    pub fn last3(needle1: u8, needle2: u8, needle3: u8, haystack: &[u8]) -> Option<usize> {
96        #[cfg(feature = "dep_memchr")]
97        return memrchr3(needle1, needle2, needle3, haystack);
98        #[cfg(not(feature = "dep_memchr"))]
99        haystack.iter().enumerate().rev()
100            .find_map(|(index, &byte)|
101                is![byte == needle1 || byte == needle2 || byte == needle3; Some(index); None])
102    }
103}
