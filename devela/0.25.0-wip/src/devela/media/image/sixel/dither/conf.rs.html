12/// Configuration for sixel dithering.
13//
14// # Adaptation
15// - Based on `sixel_dither_t` from the `libsixel` C library,
16//   adapted with adjustments for idiomatic Rust usage.
17#[derive(Debug)]
18pub(crate) struct DitherConf {
19    /// Palette definition.
20    pub palette: Vec<u8>,
21    /// Cache table.
22    pub cachetable: Option<Vec<u16>>,
23    /// The number of requested colors.
24    pub reqcolors: i32,
25    /// The number of active colors.
26    pub ncolors: i32,
27    /// The number of original colors.
28    pub origcolors: i32,
29    /// Pixel is 15bpp compressible.
30    pub optimized: bool,
31    /// Minimize palette size.
32    pub optimize_palette: bool,
33    /// For complexion correction.
34    pub complexion: i32,
35    /// Do not output palette section if true.
36    pub bodyonly: bool,
37    /// Method for finding the largest dimention for splitting.
38    pub split_method: SixelSplit,
39    /// Method for choosing a color from the box.
40    pub mean_method: SixelMean,
41    /// Method for diffusing
42    pub dither: Dither,
43    /// Quality of histogram.
44    pub quality_mode: SixelQuality,
45    /// Background color.
46    pub keycolor: i32,
47    /// Pixelformat for internal processing.
48    pub pixel_format: PixelFormat,
49}

51impl DitherConf {
52    /// Creates a new dither configuration with the specified number of colors.
53    pub fn new(mut ncolors: i32) -> SixelResult<Self> {
54        let quality_mode = if ncolors < 0 {
55            ncolors = SIXEL_PALETTE_MAX as i32;
56            SixelQuality::HighColor
57        } else {
58            if ncolors > SIXEL_PALETTE_MAX as i32 {
59                return Err(SixelError::BadInput);
60            }
61            if ncolors < 1 {
62                return Err(SixelError::BadInput);
63                // sixel_helper_set_additional_message(
64                // "DitherConf::new: palette ncolors must be more than 0");
65            }
66            SixelQuality::Low
67        };
68        Ok(Self {
69            palette: vec![0; ncolors as usize * 3],
70            cachetable: None,
71            reqcolors: ncolors,
72            ncolors,
73            origcolors: (-1),
74            keycolor: (-1),
75            optimized: false,
76            optimize_palette: false,
77            complexion: 1,
78            bodyonly: false,
79            split_method: SixelSplit::Norm,
80            mean_method: SixelMean::Center,
81            dither: Dither::FS,
82            quality_mode,
83            pixel_format: PixelFormat::RGB888,
84        })
85    }

126    pub fn initialize(
127        &mut self,
128        data: &[u8],
129        width: i32,
130        height: i32,
131        pixel_format: PixelFormat,
132        split_method: SixelSplit,
133        mean_method: SixelMean,
134        quality_mode: SixelQuality,
135    ) -> SixelResult<()> {
136        self.set_pixel_format(pixel_format);
137        #[expect(clippy::single_match_else, reason = "could be extended")]
138        let input_pixels = match pixel_format {
139            PixelFormat::RGB888 => data.to_vec(),
140            _ => {
141                /* normalize pixel_format */
142                let mut normalized_pixels = vec![0; (width * height * 3) as usize];
143                self.set_pixel_format(pixel_format.normalize(
144                    &mut normalized_pixels,
145                    data,
146                    width,
147                    height,
148                )?);
149                normalized_pixels
150            }
151        };
152
153        self.set_split_method(split_method);
154        self.set_mean_method(mean_method);
155        self.set_quality_mode(quality_mode);
156
157        let buf = sixel_quant_make_palette(
158            &input_pixels,
159            width * height * 3,
160            PixelFormat::RGB888,
161            self.reqcolors,
162            &mut self.ncolors,
163            &mut self.origcolors,
164            self.split_method,
165            self.mean_method,
166            self.quality_mode,
167        )?;
168
169        self.palette = buf;
170        self.optimized = true;
171        if self.origcolors <= self.reqcolors {
172            self.dither = Dither::None;
173        }
174        Ok(())
175    }

236    /* set transparent */
237    pub fn apply_palette(
238        &mut self,
239        pixels: &[u8],
240        width: i32,
241        height: i32,
242    ) -> SixelResult<Vec<u8>> {
243        let bufsize = width * height;
244        let mut dest = vec![0; bufsize as usize];
245
246        /* if quality_mode is full, do not use palette caching */
247        if matches!(self.quality_mode, SixelQuality::Full) {
248            self.optimized = false;
249        }
250
251        if self.cachetable.is_none()
252            && self.optimized
253            && self.palette != SixelPalette::PAL_MONO_DARK
254            && self.palette != SixelPalette::PAL_MONO_LIGHT
255        {
256            self.cachetable = Some(vec![0; 1 << (3 * 5)]);
257        }
258
259        let mut input_pixels = if !matches!(self.pixel_format, PixelFormat::RGB888) {
260            /* normalize pixel_format */
261            let mut normalized_pixels = vec![0; (width * height * 3) as usize];
262            self.pixel_format =
263                self.pixel_format.normalize(&mut normalized_pixels, pixels, width, height)?;
264            normalized_pixels
265        } else {
266            pixels.to_vec()
267        };
268        let ncolors = sixel_quant_apply_palette(
269            &mut dest,
270            &mut input_pixels,
271            width,
272            height,
273            3,
274            &mut self.palette,
275            self.ncolors,
276            self.dither,
277            self.optimized,
278            self.optimize_palette,
279            self.complexion,
280            Some(self.cachetable.as_mut().unwrap()),
281        )?;
282        self.ncolors = ncolors;
283
284        Ok(dest)
285    }
