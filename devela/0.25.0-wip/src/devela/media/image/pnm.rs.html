22/// A collection of methods for encoding and decoding
23/// <abbr title="Portable anymap format">PNM</abbr> bitmap formats.
24///
25/// - <https://en.wikipedia.org/wiki/Netpbm>
26///
27/// The PBM (Portable Bitmap), PGM (Portable Graymap), and PPM (Portable Pixmap)
28/// formats are part of the Netpbm format family. These formats are simple and
29/// straightforward for representing grayscale (PGM), black and white (PBM), and
30/// color (PPM) images in either ASCII or binary modes. Here's a brief overview:
31///
32/// - PBM (Portable Bitmap Format): Used for black and white images. It supports
33/// both ASCII (P1) and binary (P4) representations.
34///
35/// - PGM (Portable Graymap Format): Used for grayscale images. It also supports
36/// ASCII (P2) and binary (P5) representations, with the gray value typically
37/// ranging from 0 (black) to 255 (white) for 8-bit images.
38///
39/// - PPM (Portable Pixmap Format): Used for color images. Similar to PGM,
40/// it supports ASCII (P3) and binary (P6) formats. Each pixel is represented by
41/// three values (red, green, and blue), each in the range of 0 to 255 for 8-bit images.
42#[derive(Debug)]
43pub struct Pnm;
44
45impl Pnm {
46    /// Converts a `bitmap` of 1-bit bytes into PBM ASCII P1 representation.
47    ///
48    /// Expects a slice of width * height bytes equal to `0` or `1`.
49    ///
50    /// # Errors
51    /// Returns [`InvalidImageSize`] if the `bitmap` slice doesn't contain exactly
52    /// `width` * `height` elements, [`InvalidPixel`] if any pixel value is invalid
53    /// and [`FmtError`] if the string writing fails.
54    #[cfg(feature = "alloc")]
55    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
56    pub fn p1_encode_bytes(bitmap: &[u8], width: usize, height: usize) -> Result<String> {
57        if bitmap.len() != width * height {
58            return Err(InvalidImageSize(Some((width, height))));
59        }
60        let mut result = String::new();
61        writeln!(result, "P1\n{width} {height}")?;
62
63        // Convert each byte in `bitmap` to '0' (white) or '1' (black) ASCII
64        for row in 0..height {
65            let first_pixel = bitmap[row * width];
66            result.push(match first_pixel {
67                0 => '0',
68                1 => '1',
69                _ => return invalid_pixel(),
70            });
71
72            for col in 1..width {
73                let pixel = bitmap[row * width + col];
74                result.push(' '); // leading space on non-first-in-row pixels
75                match pixel {
76                    0 => result.push('0'),
77                    1 => result.push('1'),
78                    _ => return invalid_pixel(),
79                }
80            }
81            result.push('\n'); // End of row
82        }
83        Ok(result)
84    }
85
86    /// Converts a `bitmap` of 1-bit bits into PBM ASCII P1 representation.
87    ///
88    /// Each byte in `bitmap` represents 8 pixels, with the most significant bit (MSB)
89    /// of each byte representing the leftmost pixel.
90    ///
91    /// # Errors
92    /// Returns [`InvalidImageSize`] if the `bitmap` slice doesn't contain exactly
93    /// the number of expected bytes `width` * `height` elements
94    /// and [`FmtError`] if the string writing fails.
95    #[cfg(feature = "alloc")]
96    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
97    pub fn p1_encode_bits(bitmap: &[u8], width: usize, height: usize) -> Result<String> {
98        if bitmap.len() != Mem::bytes_from_bits(width * height) {
99            return Err(InvalidImageSize(Some((width, height))));
100        }
101        let mut result = String::new();
102        writeln!(result, "P1\n{width} {height}")?;
103
104        // Convert each bit in `bitmap` to '0' (white) or '1' (black) ASCII
105        for row in 0..height {
106            let first_col = 0;
107            let byte_index = (row * width + first_col) / 8;
108            let bit = 1 << (7 - (first_col % 8));
109            let first_pixel = (bitmap[byte_index] & bit) != 0;
110            result.push(if first_pixel { '1' } else { '0' });
111
112            for col in 1..width {
113                let byte_index = (row * width + col) / 8;
114                let bit = 1 << (7 - (col % 8));
115                let pixel = (bitmap[byte_index] & bit) != 0;
116                result.push(' '); // leading space on non-first-in-row pixels
117                result.push(if pixel { '1' } else { '0' });
118            }
119            result.push('\n'); // end of row
120        }
121        Ok(result)
122    }
123}

125#[cfg(all(test, feature = "alloc"))]
126mod tests_alloc {
127    use super::*;
128    use crate::_dep::_alloc::vec;
129
130    #[test]
131    fn p1_encode_bytes() {
132        let bitmap = vec![
133            1, 0, 1, 0, 1, 0, 1, 0, // First row
134            0, 1, 0, 1, 0, 1, 0, 1, // Second row
135        ];
136        let (w, h) = (8, 2);
137        let expected_output = "P1\n8 2\n1 0 1 0 1 0 1 0\n0 1 0 1 0 1 0 1\n";
138        let result = Pnm::p1_encode_bytes(&bitmap, w, h).expect("PNM P1 encoded");
139        assert_eq!(result, expected_output);
140    }
141    #[test]
142    fn p1_encode_bytes_invalid_size() {
143        let bitmap = vec![1, 0, 1, 0]; // Incorrect size for 2x2 image
144        let (w, h) = (3, 3);
145        let result = Pnm::p1_encode_bytes(&bitmap, w, h);
146        assert_eq!(result, Err(InvalidImageSize(Some((3, 3)))));
147    }
148    #[test]
149    fn p1_encode_bytes_invalid_pixel() {
150        let bitmap = vec![1, 0, 2, 0]; // Invalid pixel value (2)
151        let (w, h) = (2, 2);
152        let result = Pnm::p1_encode_bytes(&bitmap, w, h);
153        assert_eq!(result, Err(InvalidPixel));
154    }
155
156    #[test]
157    fn p1_encode_bits() {
158        let bitmap = vec![0b10101010, 0b01010101]; // Packed bits for 8x2 image
159        let (w, h) = (8, 2);
160        let expected_output = "P1\n8 2\n1 0 1 0 1 0 1 0\n0 1 0 1 0 1 0 1\n";
161        let result = Pnm::p1_encode_bits(&bitmap, w, h).expect("PNM P1 encoded");
162        assert_eq!(result, expected_output);
163    }
164    #[test]
165    fn p1_encode_bits_invalid_size() {
166        let bitmap = vec![0b10101010]; // Incorrect size for 8x2 image
167        let (w, h) = (8, 2);
168        let result = Pnm::p1_encode_bits(&bitmap, w, h);
169        assert_eq!(result, Err(InvalidImageSize(Some((8, 2)))));
170    }
171}
