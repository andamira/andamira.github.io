46mod core_impls {

48    use crate::IoError;

52    impl crate::Error for ImageError {}

54    impl Display for ImageError {
55        fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
56            use ImageError as E;
57            match self {
58                E::InvalidImageSize(o) => write!(f, "InvalidImageSize: {o:?}"),
59                E::InvalidMagicNumber => write!(f, "Invalid magic number."),
60                E::InvalidPixel => write!(f, "Invalid pixel."),
61                //
62                // E::InvalidParsedInteger(k) => write!(f, "Invalid parsed integer: {k:?}."),
63                E::InvalidParsedInteger => write!(f, "Invalid parsed integer."),
64                E::FmtError => write!(f, "A core::fmt::Error."),
65                #[cfg(any(feature = "std", all(not(feature = "std"), feature = "io")))]
66                E::IoError(e) => write!(f, "An I/O Error: {e:?}"),
67            }
68        }
69    }
