7#[rustfmt::skip]
8impl Rgba16 {
9    /// New `Rgba<u16>`.
10    pub const fn new(r: u16, g: u16, b: u16, a: u16) -> Rgba16 { Self { c: [r, g, b, a] } }
11    /// The red component.
12    pub const fn red(self) -> u16 { self.c[0] }
13    pub const fn r(self) -> u16 { self.c[0] }
14    /// The green component.
15    pub const fn green(self) -> u16 { self.c[1] }
16    pub const fn g(self) -> u16 { self.c[1] }
17    /// The blue component.
18    pub const fn blue(self) -> u16 { self.c[2] }
19    pub const fn b(self) -> u16 { self.c[2] }
20    /// The alpha component.
21    pub const fn alpha(self) -> u16 { self.c[3] }
22    pub const fn a(self) -> u16 { self.c[3] }
23}

25/// # General conversions
26impl Rgba16 {
27    /// Create from an array.
28    pub const fn from_array(c: [u16; 4]) -> Rgba16 {
29        Rgba16 { c }
30    }
31    /// Convert to an array.
32    pub const fn as_array(self) -> [u16; 4] {
33        self.c
34    }
35
36    /// Create from a tuple.
37    pub const fn from_tuple(c: (u16, u16, u16, u16)) -> Rgba16 {
38        Rgba16::new(c.0, c.1, c.2, c.3)
39    }
40    /// Convert to a tuple.
41    pub const fn to_tuple(self) -> (u16, u16, u16, u16) {
42        (self.r(), self.g(), self.b(), self.a())
43    }
44
45    /* Rgb16 */
46
47    /// Create from [`Rgb<u16>`].
48    ///
49    /// Adds the given `alpha` channel.
50    pub const fn from_rgb16(c: Rgb16, alpha: u16) -> Rgba16 {
51        Rgba16::new(c.r(), c.g(), c.b(), alpha)
52    }
53    /// Convert to [`Rgb<u16>`].
54    ///
55    /// Loses the alpha channel.
56    pub const fn to_rgb16(self) -> Rgb16 {
57        Rgb16::new(self.r(), self.g(), self.b())
58    }
59
60    /* packed u64 */
61
62    /// Create from a packed `u64` in `0xRRRRGGGGBBBBAAAA` format, with the given `alpha`.
63    ///
64    /// Any bits above `0x0000_FFFF_FFFF` are ignored.
65    pub const fn from_rgb16_packed(packed: u64, alpha: u16) -> Rgba16 {
66        Rgba16::new(
67            ((packed >> 32) & 0xFFFF) as u16,
68            ((packed >> 16) & 0xFFFF) as u16,
69            (packed & 0xFFFF) as u16,
70            alpha,
71        )
72    }
73    /// Convert to a packed `u64` in `0xRRRRGGGGBBBB` format, discarding alpha.
74    pub const fn to_rgb16_packed(self) -> u64 {
75        ((self.r() as u64) << 32) | ((self.g() as u64) << 16) | (self.b() as u64)
76    }
77
78    /// Create from a packed `u64` in `0xRRRRGGGGBBBBAAAA` format (big-endian).
79    pub const fn from_rgba16_packed(packed: u64) -> Rgba16 {
80        Rgba16::new(
81            ((packed >> 48) & 0xFFFF) as u16,
82            ((packed >> 32) & 0xFFFF) as u16,
83            ((packed >> 16) & 0xFFFF) as u16,
84            (packed & 0xFFFF) as u16,
85        )
86    }
87    /// Convert to a packed `u64` in `0xRRRRGGGGBBBBAAAA` format (big-endian).
88    pub const fn to_rgba16_packed(self) -> u64 {
89        ((self.r() as u64) << 48)
90            | ((self.g() as u64) << 32)
91            | ((self.b() as u64) << 16)
92            | (self.a() as u64)
93    }
94}

187#[cfg(test)]
188mod tests {
189    use super::*;
190
191    const C16: Rgb16 = Rgb16::new(2570, 5140, 7710);
192    const CA16: Rgba16 = Rgba16::new(2570, 5140, 7710, 10280);
193    const H16: u64 = 0x_0A0A_1414_1E1E;
194    const HA16: u64 = 0x_0A0A_1414_1E1E_2828;
195
196    #[test]
197    fn general_conversions() {
198        // array/tuple
199        assert_eq![Rgba16::from_array([2570, 5140, 7710, 10280]), CA16];
200        assert_eq![CA16.as_array(), [2570, 5140, 7710, 10280]];
201        assert_eq![Rgba16::from_tuple((2570, 5140, 7710, 10280)), CA16];
202        assert_eq![CA16.to_tuple(), (2570, 5140, 7710, 10280)];
203        // rgb
204        assert_eq![Rgba16::from_rgb16(C16, 10280), CA16];
205        assert_eq![CA16.to_rgb16(), C16];
206        // packed rgba
207        assert_eq![Rgba16::from_rgba16_packed(HA16), CA16];
208        assert_eq![CA16.to_rgba16_packed(), HA16];
209        // packed rgb
210        assert_eq![Rgba16::from_rgb16_packed(H16, 10280), CA16];
211        assert_eq![CA16.to_rgb16_packed(), H16];
212    }
213
214    #[test]
215    fn f32_conversions() {
216        let f = RgbF32::new(0.039215688, 0.078431375, 0.11764706);
217        let fa = RgbaF32::new(0.039215688, 0.078431375, 0.11764706, 0.15686275);
218        assert_eq![Rgba16::from_rgb_f32(f, 0.15686), CA16];
219        assert_eq![CA16.to_rgb_f32(), f];
220        assert_eq![Rgba16::from_rgba_f32(fa), CA16];
221        assert_eq![CA16.to_rgba_f32(), fa];
222    }
223
224    #[test]
225    fn f64_conversions() {
226        let f = RgbF64::new(0.0392156862745098, 0.0784313725490196, 0.11764705882352941);
227        let fa = RgbaF64::new(
228            0.0392156862745098,
229            0.0784313725490196,
230            0.11764705882352941,
231            0.1568627450980392,
232        );
233        assert_eq![Rgba16::from_rgb_f64(f, 0.15686), CA16];
234        assert_eq![CA16.to_rgb_f64(), f];
235        assert_eq![Rgba16::from_rgba_f64(fa), CA16];
236        assert_eq![CA16.to_rgba_f64(), fa];
237    }
