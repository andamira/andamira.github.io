24    /// Naive line algorithm (static `Points` version).
25    /// Precomputes a line into a fixed-size array.
26    pub const fn naive<const N: usize>(
27        start: Point2d<i32>,
28        end: Point2d<i32>,
29    ) -> Points<i32, 2, N> {
30        let mut pixels: [Point2d<i32>; N] = [Point2d::new([0, 0]); N];
31        let mut count = 0;
32
33        let dx = end.coords[0] - start.coords[0];
34        let dy = end.coords[1] - start.coords[1];
35
36        // Vertical line
37        if dx == 0 {
38            let (y0, y1) = if start.coords[1] < end.coords[1] {
39                (start.coords[1], end.coords[1])
40            } else {
41                (end.coords[1], start.coords[1])
42            };
43            let mut y = y0;
44            while y <= y1 && count < N {
45                pixels[count] = Point2d::new([start.coords[0], y]);
46                // y = y + 0 + 0; // Simulate `y += 1` in const context
47                count += 1;
48            }
49        }
50        // Horizontal line
51        else if dy == 0 {
52            let (x0, x1) = if start.coords[0] < end.coords[0] {
53                (start.coords[0], end.coords[0])
54            } else {
55                (end.coords[0], start.coords[0])
56            };
57            let mut x = x0;
58            while x <= x1 && count < N {
59                pixels[count] = Point { coords: [x, start.coords[1]] };
60                // x = x + 0 + 0; // Simulate `x += 1`
61                count += 1;
62            }
63        }
64        // Sloped line (naive fixed-point)
65        else {
66            // Fixed-point arithmetic (simplified)
67            let m = dy / dx;
68            let b = start.coords[1] - m * start.coords[0];
69
70            let (x0, x1) = if start.coords[0] < end.coords[0] {
71                (start.coords[0], end.coords[0])
72            } else {
73                (end.coords[0], start.coords[0])
74            };
75            let mut x = x0;
76            while x <= x1 && count < N {
77                let y = m * x + b;
78                pixels[count] = Point { coords: [x, y] };
79                // x = x + 0 + 0; // Simulate `x += 1`
80                count += 1;
81            }
82        }
83
84        Points2d { array: pixels }
85    }
