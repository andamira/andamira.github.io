10/// The set of enabled cargo features.
11pub(crate) static ENABLED_CARGO_FEATURES: OnceLock<HashSet<String>> = OnceLock::new();
12/// The set of enabled cfg flags.
13pub(crate) static ENABLED_CFG_FLAGS: OnceLock<HashSet<String>> = OnceLock::new();

15pub(crate) fn main() -> Result<(), std::io::Error> {
16    #[cfg(feature = "__dbg")]
17    Build::println_heading("Features & Configuration Flags:");
18
19    /* Collect enabled cargo features from CARGO_FEATURE_* environment variables */
20
21    ENABLED_CARGO_FEATURES.get_or_init(|| {
22        let mut features = HashSet::new();
23        for (key, _) in env::vars_os() {
24            if let Some(feature) = key.to_str().and_then(|k| k.strip_prefix("CARGO_FEATURE_")) {
25                features.insert(feature.to_lowercase());
26            }
27        }
28        features
29    });
30    #[cfg(feature = "__dbg")]
31    if let Some(f) = ENABLED_CARGO_FEATURES.get() {
32        Build::println(&format!("Active cargo features ({}): {:?}", f.len(), f));
33        Build::println("");
34    };
35    // Enable reflection flags based on cargo features
36    let _enabled_flags_from_features = reflection::set_ref_flags_from_cargo_features();
37    #[cfg(feature = "__dbg")]
38    {
39        Build::println(&format!(
40            "Reflection flags auto-enabled by features ({}): {:?}",
41            _enabled_flags_from_features.len(),
42            _enabled_flags_from_features,
43        ));
44        Build::println("");
45    }
46
47    /* Collect enabled cfg flags from both RUSTFLAGS and RUSTDOCFLAGS */
48
49    // WARNING: `RUSTDOCFLAGS` changes may not take effect immediately!
50    // ------------------------------------------------------------------
51    // Unlike `RUSTFLAGS`, changes to `RUSTDOCFLAGS` may not always trigger a rebuild.
52    // Cargo caches documentation builds, so previously compiled flags might still apply.
53    ENABLED_CFG_FLAGS.get_or_init(|| {
54        let mut cfg_flags = HashSet::new();
55        if let Ok(value) = env::var("RUSTDOCFLAGS") {
56            cfg_flags.extend(value.split_whitespace().map(String::from));
57        }
58        if let Ok(value) = env::var("CARGO_ENCODED_RUSTFLAGS") {
59            cfg_flags.extend(value.split('\x1f').map(String::from));
60        }
61        cfg_flags
62    });
63    #[cfg(feature = "__dbg")]
64    if let Some(f) = ENABLED_CFG_FLAGS.get() {
65        // IMPROVE FIXME always shows as ""
66        let filtered_flags: Vec<_> = f.iter().filter(|&f| f != "--cfg" && f != "-C").collect();
67        // let filtered_flags: Vec<_> = f.iter().collect(); // SAME FOR THIS (TEMP)
68        Build::println(&format!(
69            "Active compiler cfg flags ({}): {:?}",
70            filtered_flags.len(),
71            filtered_flags
72        ));
73        Build::println("");
74    }
75    // Enable reflection flags based on cfg flags (e.g., RUSTFLAGS)
76    let _enabled_flags_from_cfg_flags = reflection::set_ref_flags_from_cfg_flags();
77    #[cfg(feature = "__dbg")]
78    {
79        Build::println(&format!(
80            "Flags auto-enabled by cfg flags ({}): {:?}",
81            _enabled_flags_from_cfg_flags.len(),
82            _enabled_flags_from_cfg_flags,
83        ));
84        Build::println("");
85    }
86
87    Ok(())
88}

97    /* FLAGS
98   -------------------------------------------------------------------------
99   - Each `FlagsFlags` entry maps `cfg_flags` (required for activation)
100     to `auto_flags` (which get enabled when any `cfg_flags` is present).
101
102   - Important: Auto-enabled flags **do not propagate recursively**.
103     If a flag (e.g., `nightly_stable`) has its own dependent flags,
104     it must be explicitly included in the `cfg_flags` list of the parent
105     (e.g., `nightly` must list `nightly_stable` if its children should be active).
106    */
107
108    /// Associates a set of automatically enabled flags with a set of `cfg` flags.
109    pub struct FlagsFlags<'a> {
110        /// Flags that are enabled when any of the `cfg_flags` are present.
111        auto_flags: &'a [&'a str],
112        /// `cfg` flags that trigger the corresponding `auto_flags`.
113        cfg_flags: &'a [&'a str],
114    }

154    /* FEATURES
155   -------------------------------------------------------------------------
156   - Each `FlagsFeatures` entry maps `features` (required for activation)
157     to `ref_flags` (which get enabled when any `features` are present).
158
159   - Features propagate **transitively**—enabling a feature activates all its
160     direct and indirect flags. (e.g., enabling `"sys"` also enables `"mem"` and,
161     through it, `"mem··"`, even though `"mem··"` isn't directly enabled in `"sys"`).
162    */
163
164    /// Associates a set of reflection flags with a set of `cfg` features.
165    pub struct FlagsFeatures<'a> {
166        /// Reflection flags enabled when any of the `features` are present.
167        ref_flags: &'a [&'a str],
168        /// Cargo features that trigger the corresponding `ref_flags`.
169        features: &'a [&'a str],
170    }

369    /// Sets the reflection flags for all the corresponding enabled cargo features from the list.
370    ///
371    /// This is the list of the constants defined above.
372    pub(super) fn set_ref_flags_from_cargo_features() -> Vec<String> {
373        let mut enabled_ref_flags = Vec::new();
374
375        for ff in [
376            /* development */
377
378            DEVELOPMENT,
379            ENVIRONMENT,
380            SAFE, SAFEST, UNSAFE,
381            DEPENDENCY,
382
383            /* modules */
384
385            CODE,
386            DATA,
387            GAME,
388            LANG, FFI,
389            MEDIA,
390            NUM, GEOM,
391            PHYS,
392            SYS, MEM,
393            TEXT,
394            UI,
395            WORK,
396
397            /* capabilities */
398
399            // code
400            UNROLL,
401            // data
402            TUPLE,
403            DESTAQUE, GRAPH, NODE, STACK, // collections
404            // text
405            CHAR, STRING, STRING_U,
406            // num
407            FLOAT, INT, UINT, // numbers
408
409        ] { set_flags_dbg_features(ff.ref_flags, ff.features, &mut enabled_ref_flags); }
410
411        enabled_ref_flags
412    }
413    /// Sets reflection flags if some **FEATURES** are enabled.
414    ///
415    /// - flag_names: The name of the reflection flag to set if any feature is enabled.
416    /// - features:   The cargo features names to check.
417    fn set_flags_dbg_features(ref_flags: &[&str], features: &[&str], enabled: &mut Vec<String>) {
418        let is_enabled = features.iter().any(|&f| ENABLED_CARGO_FEATURES.get().unwrap().contains(f));
419        if is_enabled {
420            for flag in ref_flags {
421                println!("cargo:rustc-cfg={flag}");
422                enabled.push(flag.to_string());
423            }
424        }
425    }

429    /// Sets automatic flags, based on enabled cfg flags.
430    pub(super) fn set_ref_flags_from_cfg_flags() -> Vec<String> {
431        let mut enabled_ref_flags = Vec::new();
432        for ff in [
433            FLAGS_NIGHTLY, FLAGS_NIGHTLY_UNSTABLE, FLAGS_NIGHTLY_STABLE, FLAGS_NIGHTLY_REFLECT,
434        ] {
435            set_flags_dbg_flags(ff.auto_flags, ff.cfg_flags, &mut enabled_ref_flags);
436        }
437        enabled_ref_flags
438    }
439    /// Sets automatic flags if some **FLAGS** are enabled.
440    ///
441    /// - flag_names: The name of the reflection flag to set if any cfg flag is enabled.
442    /// - cfg_flags:   The cfg flags names to check.
443    fn set_flags_dbg_flags(ref_flags: &[&str], cfg_flags: &[&str], enabled: &mut Vec<String>) {
444        let is_enabled = cfg_flags.iter().any(|&f| ENABLED_CFG_FLAGS.get().unwrap().contains(f));
445        if is_enabled {
446            for flag in ref_flags {
447                println!("cargo:rustc-cfg={flag}");
448                enabled.push(flag.to_string());
449            }
450        }
451    }
