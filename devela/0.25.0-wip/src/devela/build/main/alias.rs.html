3//! Aliases of combinations of configuration features.

5
6#[cfg(feature = "__dbg")]
7use super::Build;
8use ::std::{env::var, io::Error as IoError};
9
10/// Defines a cfg flag `alias` and saves its name in a `list`.
11fn new_alias(list: &mut Vec<&'static str>, alias: &'static str) {
12    println!("cargo:rustc-cfg={}", alias);
13    list.push(alias);
14}
15
16pub(crate) fn main() -> Result<(), IoError> {
17    #[cfg(feature = "__dbg")]
18    Build::println_heading("Aliases:");
19
20    let mut aliases = Vec::with_capacity(20);
21    arch_aliases(&mut aliases);
22    base_aliases(&mut aliases);
23
24    #[cfg(feature = "__dbg")]
25    Build::println(&format!("Active compiler cfg flag aliases ({}): {:?}", aliases.len(), aliases));
26
27    Ok(())
28}

31// - any_target_arch_linux
32// - any_target_arch_riscv
33#[allow(unused_variables)]
34fn arch_aliases(aliases: &mut Vec<&'static str>) {
35    let target_arch = var("CARGO_CFG_TARGET_ARCH").unwrap();
36
37    if target_arch == "x86"
38        || target_arch == "x86_64"
39        || target_arch == "arm"
40        || target_arch == "aarch64"
41        || target_arch == "riscv32"
42        || target_arch == "riscv64"
43    {
44        new_alias(aliases, "any_target_arch_linux");
45    }
46    if target_arch == "riscv32" || target_arch == "riscv64" {
47        new_alias(aliases, "any_target_arch_riscv");
48    }
49}
