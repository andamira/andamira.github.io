14pub(crate) fn main() -> Result<(), IoError> {
15    // https://doc.rust-lang.org/cargo/reference/build-scripts.html#rerun-if-env-changed
16    println!("cargo:rerun-if-env-changed=CARGO_PRIMARY_PACKAGE");
17    println!("cargo:rerun-if-env-changed=CARGO_TARGET_DIR");
18    println!("cargo:rerun-if-env-changed=CARGO_WORKSPACE_DIR");
19
20    // This environment variable will be set if the package being built is primary.
21    let cargo_primary_package = option_env!("CARGO_PRIMARY_PACKAGE");
22    if cargo_primary_package.is_some() {
23        println!("cargo:rustc-cfg=cargo_primary_package");
24    }
25
26    // Makes sure `CARGO_TARGET_DIR` and `CARGO_WORKSPACE_DIR` are always defined.
27    //
28    // Used in: doclink! (/libs/base_core/src/code/util/doclink
29    // In sync with: /.cargo/config.toml#[env] & /libs/base_core/build/environment.rs
30    let cwd = get_workspace_dir();
31    let ctd = get_target_dir(&cwd);
32    println!("cargo:rustc-env=CARGO_WORKSPACE_DIR={}", cwd);
33    println!("cargo:rustc-env=CARGO_TARGET_DIR={}", ctd);
34
35    #[cfg(feature = "__dbg")]
36    {
37        use super::{Build, CRATE_NAME};
38
39        // https://doc.rust-lang.org/cargo/reference/environment-variables.html
40        Build::println_heading("Environment:");
41
42        // using the global constant because if not, from here we can only get "build_script_mod"
43        Build::println(&format!["· CARGO_CRATE_NAME={CRATE_NAME}"]);
44
45        if let Some(v) = cargo_primary_package {
46            Build::println(&format!["· CARGO_PRIMARY_PACKAGE={v}"]);
47        } else {
48            Build::println("x CARGO_PRIMARY_PACKAGE");
49        }
50
51        // Only show these env vars for the root package
52        // if cargo_primary_package.is_some() { // FIXME: not set for deps
53        if CRATE_NAME == "devela" {
54            Build::println_var("CARGO_WORKSPACE_DIR");
55            Build::println_var("CARGO_TARGET_DIR");
56            Build::println_var("CARGO_MANIFEST_DIR");
57            Build::println_var("CARGO_MANIFEST_PATH");
58            Build::println_var("OUT_DIR");
59
60            Build::println_var("HOST");
61            Build::println_var("TARGET");
62            Build::println_var("PROFILE");
63            Build::println_var("NUM_JOBS");
64
65            Build::println_var("RUSTC");
66            // Build::println_var("RUSTC_WRAPPER");
67            // Build::println_var("RUSTC_LINKER");
68            // Build::println_var("RUSTDOC");
69            // Build::println_var("CARGO");
70            // Build::println_var("CARGO_MAKEFLAGS");
71            // Build::println_var("CARGO_ENCODED_RUSTFLAGS");
72
73            // DEBUG
74            // for (key, value) in std::env::vars() {
75            //     Build::println(&format![">> {key}: {value}"]);
76            // }
77        }
78
79        Build::println_var_encoded("CARGO_ENCODED_RUSTFLAGS", "RUSTFLAGS");
80        Build::println_var("RUSTDOCFLAGS");
81    }
82
83    Ok(())
84}

86fn get_workspace_dir() -> String {
87    option_env!("CARGO_WORKSPACE_DIR")
88        .map(|s| s.to_string())
89        .unwrap_or_else(|| {
90            option_env!("CARGO_MANIFEST_DIR")
91                .and_then(|manifest_dir| {
92                    let path = Path::new(manifest_dir);
93                    path.ancestors()
94                        .find(|p| p.join("Cargo.toml").exists())
95                        .and_then(|p| p.to_str())
96                        .map(|s| s.to_string())
97                })
98                .unwrap_or_else(|| ".".to_string())
99        })
100        .trim_end_matches('/')
101        .to_string()
102        + "/"
103}
104fn get_target_dir(workspace_dir: &str) -> String {
105    option_env!("CARGO_TARGET_DIR")
106        .map(|s| s.to_string())
107        .unwrap_or_else(|| format!("{}/target", workspace_dir))
108        .trim_end_matches('/')
109        .to_string()
110        + "/"
111}
