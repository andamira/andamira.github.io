3//! Defines [`rand_pcg!`].
4//
5
6#[doc = crate::_tags!(construction rand)]
7#[doc = concat!["Defines a canonical ", crate::_ABBR_PCG!(), " (Permuted Congruential Generator) ",
8crate::_ABBR_PRNG!(), " type."]]
9#[doc = crate::_doc_meta!{location("num/prob/rand")}]
10///
11/// This macro generates a concrete, fixed-width PCG PRNG
12/// with a canonical state transition and output permutation.
13///
14/// The output width is selected by specifying one of the supported integer
15/// sizes (`u8`, `u16`, `u32`, `u64`). The appropriate state size, multiplier,
16/// increment, and output permutation are chosen automatically.
17///
18/// The generator stores two internal state values (state and inc), each twice
19/// the output width, for a total internal state of four times the output width.
20///
21/// The 8- and 16-bit PCG variants are classified as weak PRNGs due to their limited
22/// per-stream state and period. The 32- and 64-bit variants use the general PRNG profile.
23///
24/// # Generated items
25/// For the defined PRNG type, this macro generates:
26/// - the PRNG struct with internal state.
27/// - constructors and state accessors.
28/// - step, peek, and jump-ahead operations.
29/// - bounded generation helpers.
30/// - an implementation of [`Rand`][crate::Rand] for generic usage.
31///
32/// # Examples
33/// ```
34// WAIT: [cfg(doctest)](https://github.com/rust-lang/rust/issues/67295)
35/// # #[macro_export]macro_rules!__crate_name{()=>{""};} // doctest stub for _doc_location
36/// # use devela::rand_pcg;
37/// rand_pcg! {
38///     /// Custom attributes
39///     pub(crate) Pcg8: (u8)
40/// }
41/// ```
42/// See also [`Pcg32`][crate::Pcg32].
43///
44/// # Reference
45/// PCG family reference: <https://www.pcg-random.org>
46#[macro_export]
47#[cfg_attr(cargo_primary_package, doc(hidden))]
48macro_rules! rand_pcg {
49    (

54        $crate::rand_pcg!(% $(#[$attrs])* $vis $name, 32+u16, 8+u8, 1,
55            output_xsh_rr_16_8, 0xDEFA, 12829, 47989);
56    };
57    ( // Defines a 16-bit PCG PRNG (`u16` output, 32-bit state)
58      $(#[$attrs:meta])* $vis:vis $name:ident : (u16)) => {
59        $crate::rand_pcg!(% $(#[$attrs])* $vis $name, 64+u32, 16+u16, 2,
60            output_xsh_rr_32_16, 0xDEFA_0017, 747796405, 2891336453);
61    };
62    ( // Defines a 32-bit PCG PRNG (`u32` output, 64-bit state)
63      $(#[$attrs:meta])* $vis:vis $name:ident : (u32)) => {
64        $crate::rand_pcg!(% $(#[$attrs])* $vis $name, 128+u64, 32+u32, 4,
65            output_xsh_rr_64_32, 0xDEFA_0017_DEFA_0017, 6364136223846793005, 1442695040888963407);
66    };
67    ( // Defines a 64-bit PCG PRNG (`u64` output, 128-bit state)
68      $(#[$attrs:meta])* $vis:vis $name:ident : (u64)

71    ) => {

73            output_xsl_rr_128_64, 0xDEFA_0017_DEFA_0017_DEFA_0017_DEFA_0017,
74            47026247687942121848144207491837523525, 117397592171526113268558934119004209487);
75    };

106    /* private arms */
107    (%define $(#[$attrs:meta])*
108     $vis:vis $name:ident, $sbits:literal+$state:ty, $obits:literal+$output:ty,
109     $obytes:literal, $output_fn:ident,
110     $SEED:literal, $MUL:literal, $INC:literal
111     ) => { $crate::paste! {
112        $crate::rand_pcg!(%impls $output $name);
113
114        $(#[$attrs])*
115        #[doc = "It uses a " $sbits
116        "-bit linear congruential state with an independent stream selector, and generates "
117        $obits "-bit values."]
118        ///
119        /// Compared to simple xorshift-style generators of similar size, it
120        /// has stronger statistical properties while remaining small and fast.
121        ///
122        /// This implementation follows a canonical PCG output permutation
123        /// (XSH RR or XSL RR, depending on size), by Melissa O'Neill.
124        ///
125        /// This generator is deterministic and non-cryptographic.
126        ///
127        /// Reference: <https://www.pcg-random.org>
128        #[derive(Clone, Copy, Debug, PartialEq, Eq)]
129        $vis struct $name {
130            state: $state,
131            inc: $state, // must always be odd
132        }
133        /// Creates a new PRNG initialized with the default fixed seed.
134        impl Default for $name {
135            fn default() -> Self {
136                use $crate::ConstInit;
137                Self::INIT
138            }
139        }
140        /// Creates a new PRNG initialized with the default fixed seed.
141        impl $crate::ConstInit for $name {
142            const INIT: Self = Self::new(Self::DEFAULT_SEED, Self::DEFAULT_INC);
143        }
144
145        // Private associated item
146        impl $name {
147            #[doc(hidden)]
148            pub const DEFAULT_SEED: $state = $SEED;
149            #[doc(hidden)]
150            pub const DEFAULT_INC: $state = $INC;
151
152            #[inline]
153            const fn advance_lcg(state: $state,
154                mut delta: $state,
155                mut cur_mult: $state,
156                mut cur_plus: $state,
157            ) -> $state {
158                let mut acc_mult = 1 as $state;
159                let mut acc_plus = 0 as $state;
160                while delta > 0 {
161                    if (delta & 1) != 0 {
162                        acc_mult = acc_mult.wrapping_mul(cur_mult);
163                        acc_plus = acc_plus.wrapping_mul(cur_mult).wrapping_add(cur_plus);
164                    }
165                    cur_plus = cur_mult.wrapping_add(1).wrapping_mul(cur_plus);
166                    cur_mult = cur_mult.wrapping_mul(cur_mult);
167                    delta >>= 1;
168                }
169                acc_mult.wrapping_mul(state).wrapping_add(acc_plus)
170            }
171        }
172
173        impl $name {
174            #[doc = "Creates a new " $name " using the canonical PCG seeding scheme."]
175            ///
176            /// This performs a small warm-up so that closely related seeds produce
177            /// well-scrambled initial outputs.
178            pub const fn new(seed: $state, stream: $state) -> Self {
179                let mut rng = Self::new_fast(0, stream);
180                let _ = rng.[<next_ $output>]();
181                rng.state = rng.state.wrapping_add(seed);
182                let _ = rng.[<next_ $output>]();
183                rng
184            }
185
186            #[doc = "Creates a new " $name " with the given initial state and stream."]
187            ///
188            /// The stream value selects an independent random sequence.
189            pub const fn new_fast(state: $state, stream: $state) -> Self {
190                Self { state, inc: (stream << 1) | 1 }
191            }
192
193            #[doc = "Creates a new " $name " without enforcing invariants."]
194            ///
195            /// # Panics
196            /// Panics in debug if `inc` is not odd.
197            pub const fn new_unchecked(state: $state, inc: $state) -> Self {
198                debug_assert![!inc.is_multiple_of(2), "`inc` must be odd"];
199                Self { state, inc }
200            }
201
202            /// Returns the PRNG's inner state as a raw snapshot.
203            #[must_use]
204            pub const fn inner_state(self) -> ($state, $state) {
205                (self.state, self.inc)
206            }
207
208            /// Restores the PRNG from the given state.
209            #[must_use]
210            pub const fn from_state(state: ($state, $state)) -> Self {
211                Self { state: state.0, inc: state.1 }
212            }
213
214            #[doc = "Returns the current `" $output "` value."]
215            #[must_use]
216            pub const fn [<current_ $output>](&self) -> $output {
217                Self::$output_fn(self.state)
218            }
219
220            #[doc = "Advances to the next random `" $output "` value."]
221            #[must_use]
222            pub const fn [<next_ $output>](&mut self) -> $output {
223                let old = self.state;
224                self.state = old.wrapping_mul($MUL).wrapping_add(self.inc | 1);
225                Self::$output_fn(old)
226            }
227
228            /// Returns a copy of the next PRNG state.
229            #[must_use]
230            pub const fn peek_next_state(&self) -> Self {
231                let next_state = self.state.wrapping_mul($MUL).wrapping_add(self.inc | 1);
232                Self { state: next_state, inc: self.inc }
233            }
234            #[doc = "Returns the next random `" $output "` without advancing the state."]
235            #[must_use]
236            pub const fn [<peek_next_ $output>](&self) -> $output {
237                Self::$output_fn(self.state)
238            }
239
240            #[doc = "Returns both the next random state and the `" $output "` value."]
241            pub const fn [<own_next_ $output>](self) -> $crate::Own<Self, $output> {
242                let value = Self::$output_fn(self.state);
243                let next = self.peek_next_state();
244                $crate::Own::new(next, value)
245            }
246
247            /// Fills the buffer with generated bytes.
248            pub const fn fill_bytes(&mut self, buffer: &mut [u8]) {
249                use $crate::{Slice, slice};
250                let mut i = 0;
251                while i < buffer.len() {
252                    let r = self.[<next_ $output>]();
253                    let bytes = r.to_le_bytes();
254                    let remaining = buffer.len() - i;
255                    if remaining >= $obytes {
256                        // buffer[i..i + $obytes].copy_from_slice(&bytes);
257                        Slice::copy(slice!(mut buffer, i, ..i + $obytes), &bytes);
258                        i += $obytes;
259                    } else {
260                        // buffer[i..].copy_from_slice(&bytes[..remaining]);
261                        Slice::copy(slice!(mut buffer, i, ..), slice!(&bytes, ..remaining));
262                        break;
263                    }
264                }
265            }
266
267            /// Advances the generator by `delta` steps.
268            pub const fn advance(&mut self, delta: $state) {
269                self.state = Self::advance_lcg(self.state, delta, $MUL, self.inc | 1);
270            }
271
272            /// Returns a copy of the state advanced by `delta` steps.
273            pub const fn copy_advance(self, delta: $state) -> Self {
274                let mut s = self;
275                s.advance(delta);
276                s
277            }
278
279            /// Returns a uniformly distributed value in `0..bound`.
280            ///
281            /// Uses rejection sampling to avoid modulo bias.
282            #[must_use]
283            pub const fn next_bounded(&mut self, bound: $output) -> $output {
284                debug_assert!(bound > 0);
285                let threshold = bound.wrapping_neg() % bound;
286                loop {
287                    let r = self.[<next_ $output>]();
288                    if r >= threshold { return r % bound; }
289                }
290            }
291        }

338                debug_assert!(upper > 0);

357            }
