17/// The `Xoroshiro128++`
18/// <abbr title="Pseudo-Random Number Generator">PRNG</abbr>.
19#[doc = crate::_doc_meta!{location("num/prob/rand")}]
20///
21/// It has a 128-bit state and generates 32-bit numbers.
22#[must_use]
23#[derive(Clone, Copy, Debug, PartialEq, Eq)]
24pub struct Xoroshiro128pp([u32; 4]);
25
26/// # Constructors
27impl Xoroshiro128pp {
28    /// Creates a new Xoroshiro128++ PRNG with the given `seed`.
29    ///
30    /// Returns `None` if the seed parts are all zero.
31    #[must_use]
32    pub const fn new(seed: [u32; 4]) -> Option<Self> {
33        if (seed[0] | seed[1] | seed[2] | seed[3]) == 0 {
34            Self::cold_path_result()
35        } else {
36            Some(Self(seed))
37        }
38    }
39
40    /// Creates a new Xoroshiro128++ PRNG with the given seed without any checks.
41    ///
42    /// # Panics
43    /// Panics in debug mode if the seed parts are all `0`.
44    pub const fn new_unchecked(seed: [u32; 4]) -> Self {
45        debug_assert!((seed[0] | seed[1] | seed[2] | seed[3]) != 0, "Seed must be non-zero");
46        Self(seed)
47    }
48
49    /// Creates a new Xoroshiro128++ PRNG, seeded from addresses of stack variables.
50    ///
51    /// This is a very poor source of randomness.
52    #[inline(never)]
53    pub fn from_stack() -> Self {
54        let (a, b, c, d) = (0, 0, 0, 0);
55        let seed: [u32; 4] = [
56            &a as *const _ as u32,
57            &b as *const _ as u32,
58            &c as *const _ as u32,
59            &d as *const _ as u32,
60        ];
61        Self::new_unchecked(seed)
62    }
63
64    #[must_use]
65    /// Returns the PRNG's inner state as a raw snapshot.
66    pub const fn inner_state(self) -> [u32; 4] {
67        self.0
68    }
69    /// Restores the PRNG from the given state.
70    pub const fn from_state(state: [u32; 4]) -> Self {
71        Self(state)
72    }
73}

75/// # Methods taking `&mut self`
76impl Xoroshiro128pp {
77    /// The jump function for the generator, equivalent to 2^64 `next_u32` calls.
78    pub const fn jump(&mut self) {
79        self.jump_with_constant(Self::JUMP);
80    }
81
82    /// The long jump function for the generator, equivalent to 2^96 `next_u32` calls.
83    pub const fn long_jump(&mut self) {
84        self.jump_with_constant(Self::LONG_JUMP);
85    }
86
87    /// Generates the next random `u32` value.
88    // Note how the output of the RNG is computed before updating the state.
89    // unlike on Xorshift128, for example.
90    #[must_use]
91    pub const fn next_u32(&mut self) -> u32 {
92        let result = self.current_u32();
93        let t = self.0[1] << 9;
94        self.0[2] ^= self.0[0];
95        self.0[3] ^= self.0[1];
96        self.0[1] ^= self.0[2];
97        self.0[0] ^= self.0[3];
98        self.0[2] ^= t;
99        self.0[3] = self.0[3].rotate_left(11);
100        result
101    }
102
103    /// Generates the next 2 random `u32` values.
104    #[must_use]
105    pub const fn next2(&mut self) -> [u32; 2] {
106        [self.next_u32(), self.next_u32()]
107    }
108    /// Generates the next 4 random `u32` values.
109    #[must_use]
110    pub const fn next4(&mut self) -> [u32; 4] {
111        [self.next_u32(), self.next_u32(), self.next_u32(), self.next_u32()]
112    }
113    /// Generates the next random value split into 4 u8 values.
114    #[must_use]
115    pub const fn next4_u8(&mut self) -> [u8; 4] {
116        self.next_u32().to_ne_bytes()
117    }
118    /// Generates the next random value split into 2 u16 values.
119    #[must_use]
120    pub const fn next2_u16(&mut self) -> [u16; 2] {
121        Cast(self.next_u32()).into_u16_le()
122    }
123    /// Returns the next u64, advancing the state 2 times.
124    #[must_use]
125    pub const fn next_u64(&mut self) -> u64 {
126        Cast::<u64>::from_u32_le(self.next2())
127    }
128    /// Returns the next u128, advancing the state 4 times.
129    #[must_use]
130    pub const fn next_u128(&mut self) -> u128 {
131        Cast::<u128>::from_u32_le(self.next4())
132    }
133
134    /// Fills the buffer with generated bytes.
135    pub const fn fill_bytes(&mut self, buffer: &mut [u8]) {
136        let mut i = 0;
137        while i < buffer.len() {
138            let random_u32 = self.next_u32();
139            let bytes = random_u32.to_le_bytes();
140            let remaining = buffer.len() - i;
141            if remaining >= 4 {
142                // buffer[i..i + 4].copy_from_slice(&bytes);
143                Slice::copy(slice!(mut buffer, i, ..i + 4), &bytes);
144                i += 4;
145            } else {
146                // buffer[i..].copy_from_slice(&bytes[..remaining]);
147                Slice::copy(slice!(mut buffer, i, ..), slice!(&bytes, ..remaining));
148                break;
149            }
150        }
151    }
152}

154/// # Methods taking `self`
155impl Xoroshiro128pp {
156    /// Returns the current random `u32`, without updating the state.
157    #[must_use]
158    pub const fn current_u32(self) -> u32 {
159        self.0[0].wrapping_add(self.0[3]).rotate_left(7).wrapping_add(self.0[0])
160    }
161
162    /// Returns a copy of the next new random state.
163    pub const fn copy_peek_next_state(self) -> Self {
164        let mut x = self.0;
165        let t = x[1] << 9;
166        x[2] ^= x[0];
167        x[3] ^= x[1];
168        x[1] ^= x[2];
169        x[0] ^= x[3];
170        x[2] ^= t;
171        x[3] = x[3].rotate_left(11);
172        Self(x)
173    }
174
175    /// Returns both the next random state and the `u32` value in a tuple.
176    pub const fn own_next_u32(self) -> Own<Self, u32> {
177        let next_state = self.copy_peek_next_state();
178        let next_value = next_state.current_u32();
179        Own::new(next_state, next_value)
180    }
181
182    /// Returns a copy of the state jumped ahead by 2^64 steps.
183    pub const fn copy_jump(self) -> Self {
184        self.copy_jump_with_constant(Self::JUMP)
185    }
186
187    /// Returns a copy of the state long-jumped ahead by 2^96 steps.
188    pub const fn copy_long_jump(self) -> Self {
189        self.copy_jump_with_constant(Self::LONG_JUMP)
190    }
191}

205        let [x, y] = Cast(seeds[0]).into_u32_le();
206        let [z, a] = Cast(seeds[1]).into_u32_le();
207        Self::new([x, y, z, a])

213        Self::new(seeds)
214    }

216    ///
217    /// The seeds will be joined in little endian order.
218    pub const fn new8_u16(seeds: [u16; 8]) -> Option<Self> {
219        Self::new([
220            Cast::<u32>::from_u16_le([seeds[0], seeds[1]]),
221            Cast::<u32>::from_u16_le([seeds[2], seeds[3]]),
222            Cast::<u32>::from_u16_le([seeds[4], seeds[5]]),
223            Cast::<u32>::from_u16_le([seeds[6], seeds[7]]),
224        ])
225    }
226    /// Returns a seeded `Xoroshiro128pp` generator from the given 16 × 8-bit seeds.
227    ///
228    /// The seeds will be joined in little endian order.
229    pub const fn new16_u8(seeds: [u8; 16]) -> Option<Self> {
230        Self::new([
231            u32::from_le_bytes([seeds[0], seeds[1], seeds[2], seeds[3]]),
232            u32::from_le_bytes([seeds[4], seeds[5], seeds[6], seeds[7]]),
233            u32::from_le_bytes([seeds[8], seeds[9], seeds[10], seeds[11]]),
234            u32::from_le_bytes([seeds[12], seeds[13], seeds[14], seeds[15]]),
235        ])
236    }

280impl Xoroshiro128pp {
281    #[doc(hidden)]
282    pub const DEFAULT_SEED: [u32; 4] = [0xDEFA_0017; 4];
283
284    const JUMP: [u32; 4] = [0x8764_000b, 0xf542_d2d3, 0x6fa0_35c3, 0x77f2_db5b];
285    const LONG_JUMP: [u32; 4] = [0xb523_952e, 0x0b6f_099f, 0xccf_5a0ef, 0x1c58_0662];
286
287    #[cold] #[rustfmt::skip]
288    const fn cold_path_result() -> Option<Self> { None }
289    #[cold] #[allow(dead_code)] #[rustfmt::skip]
290    const fn cold_path_default() -> Self { Self::new_unchecked(Self::DEFAULT_SEED) }
291
292    const fn jump_with_constant(&mut self, jump: [u32; 4]) {
293        let (mut s0, mut s1, mut s2, mut s3) = (0, 0, 0, 0);
294        whilst! { j in 0..4; {
295            whilst! { b in 0..32; {
296                if (jump[j] & (1 << b)) != 0 {
297                    s0 ^= self.0[0];
298                    s1 ^= self.0[1];
299                    s2 ^= self.0[2];
300                    s3 ^= self.0[3];
301                }
302                let _ = self.next_u32();
303            }}
304        }}
305        self.0 = [s0, s1, s2, s3];
306    }
307
308    const fn copy_jump_with_constant(self, jump: [u32; 4]) -> Self {
309        let (mut s0, mut s1, mut s2, mut s3) = (0, 0, 0, 0);
310        let mut state = self;
311        let mut i = 0;
312        while i < jump.len() {
313            let mut b = 0;
314            while b < 32 {
315                if (jump[i] & (1 << b)) != 0 {
316                    s0 ^= state.0[0];
317                    s1 ^= state.0[1];
318                    s2 ^= state.0[2];
319                    s3 ^= state.0[3];
320                }
321                state = state.copy_peek_next_state();
322                b += 1;
323            }
324            i += 1;
325        }
326        Self([s0, s1, s2, s3])
327    }
