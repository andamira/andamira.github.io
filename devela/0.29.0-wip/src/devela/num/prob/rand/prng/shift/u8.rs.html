10/// The `XorShift8` <abbr title="Pseudo-Random Number Generator">PRNG</abbr>.
11#[doc = crate::_doc_meta!{location("num/prob/rand")}]
12///
13/// It has an 8-bit state and generates 8-bit numbers.
14/// It has poor statistical quality and limited entropy.
15#[must_use]
16#[derive(Clone, Copy, Debug, PartialEq, Eq)]
17pub struct XorShift8<const A: usize = 3, const B: usize = 4, const C: usize = 2>(u8);

39impl<const A: usize, const B: usize, const C: usize> XorShift8<A, B, C> {
40    /// Returns a seeded `XorShift8` generator from the given 8-bit seed.
41    ///
42    /// If the seed is `0`, the default seed is used instead.
43    ///
44    /// # Panics
45    /// Panics in debug if either `A`, `B` or `C` are < 1 or > 7.
46    pub const fn new(seed: u8) -> Self {
47        debug_assert![A > 0 && A <= 7];
48        debug_assert![B > 0 && A <= 7];
49        debug_assert![C > 0 && A <= 7];
50        if seed == 0 { Self::cold_path_default() } else { Self(seed) }
51    }
52    /// Returns a seeded `XorShift8` generator from the given 8-bit seed,
53    /// unchecked.
54    ///
55    /// The seed must not be `0`, otherwise every result will also be `0`.
56    ///
57    /// # Panics
58    /// Panics in debug if either `A`, `B` or `C` are < 1 or > 7,
59    /// or if the seed is `0`.
60    pub const fn new_unchecked(seed: u8) -> Self {
61        debug_assert![A > 0 && A <= 7];
62        debug_assert![B > 0 && A <= 7];
63        debug_assert![C > 0 && A <= 7];
64        debug_assert![seed != 0, "Seed must be non-zero"];
65        Self(seed)
66    }
67    #[must_use]
68    /// Returns the PRNG's inner state as a raw snapshot.
69    pub const fn inner_state(self) -> u8 {
70        self.0
71    }
72    /// Restores the PRNG from the given state.
73    pub const fn from_state(state: u8) -> Self {
74        Self(state)
75    }
76    /// Returns the current random `u8`.
77    #[must_use]
78    pub const fn current_u8(&self) -> u8 {
79        self.0
80    }
81    /// Updates the state and returns the next random `u8`.
82    ///
83    pub const fn next_u8(&mut self) -> u8 {
84        let mut x = self.0;
85        _xorshift_basis!(x, 0, (A, B, C));
86        self.0 = x;
87        x
88    }
89    /// Returns a copy of the next new random state.
90    pub const fn peek_next_state(&self) -> Self {
91        let mut x = self.0;
92        _xorshift_basis!(x, 0, (A, B, C));
93        Self(x)
94    }
95    /// Returns both the next random state and the `u8` value.
96    pub const fn own_next_u8(self) -> Own<Self, u8> {
97        let s = self.peek_next_state();
98        let v = s.current_u8();
99        Own::new(s, v)
100    }
101    /// Fills the buffer with generated bytes.
102    pub const fn fill_bytes(&mut self, buffer: &mut [u8]) {
103        whilst! { i in 0..buffer.len(); {
104            buffer[i] = self.next_u8();
105        }}
106    }
107}

109/// # Extras
110impl<const A: usize, const B: usize, const C: usize> XorShift8<A, B, C> {
111    /// Returns a seeded `XorShift8` generator from the given 8-bit seed.
112    ///
113    /// This is an alias of [`new`][Self#method.new].
114    pub const fn new1_u8(seed: u8) -> Self {
115        Self::new(seed)
116    }
117    /// Returns the next 2 × random `u8` combined as a single `u16`.
118    pub const fn next_u16(&mut self) -> u16 {
119        u16::from_le_bytes([self.next_u8(), self.next_u8()])
120    }
121    /// Returns the next 4 × random `u8` combined as a single `u32`.
122    pub const fn next_u32(&mut self) -> u32 {
123        u32::from_le_bytes([self.next_u8(), self.next_u8(), self.next_u8(), self.next_u8()])
124    }
125    /// Returns the next 8 × random `u8` combined as a single `u64`.
126    pub const fn next_u64(&mut self) -> u64 {
127        u64::from_le_bytes([
128            self.next_u8(),
129            self.next_u8(),
130            self.next_u8(),
131            self.next_u8(),
132            self.next_u8(),
133            self.next_u8(),
134            self.next_u8(),
135            self.next_u8(),
136        ])
137    }
138}
