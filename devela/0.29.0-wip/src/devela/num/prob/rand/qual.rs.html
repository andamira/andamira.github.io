3//! Defines [`RandQualities`].
4//
5
6crate::set! {
7    #[doc = crate::_tags!(rand set)]
8    /// Semantic qualities of a random source.
9    #[doc = crate::_doc_meta!{
10        location("num/prob/rand"),
11        test_size_of(RandQualities = 1),
12    }]
13    /// These qualities describe source behavior and suitability independently
14    /// of its output width, state size, and fallibility.
15    ///
16    /// The absence of a quality means that property is not claimed.
17    /// In particular, a source may be neither cryptographic nor known to be weak.
18    pub struct RandQualities(u8) {
19        /// The output stream is determined by the source's current state.
20        DETERMINISTIC = 0;
21
22        /// The same explicit seed material can recreate the same stream.
23        REPRODUCIBLE = 1;
24
25        /// The source is intended for cryptographic use.
26        CRYPTOGRAPHIC = 2;
27
28        /// The source depends on state outside its own value.
29        EXTERNAL = 3;
30
31        /// The source may block while producing random data.
32        MAY_BLOCK = 4;
33
34        /// The source has known generator-side weaknesses, such as
35        /// poor statistical behavior, short period, or limited state capacity.
36        GENERATOR_WEAK = 5;
37
38        /// The source draws unpredictability from weak or low-entropy material.
39        ENTROPY_WEAK = 6;
40    }
41    /// # Common profiles
42    impl {
43        /// Common qualities of reproducible deterministic PRNGs.
44        pub const PRNG: Self = Self::DETERMINISTIC.with_reproducible();
45
46        /// Common qualities of reproducible deterministic PRNGs with known
47        /// generator-side weaknesses.
48        pub const WEAK_PRNG: Self = Self::PRNG.with_generator_weak();
49
50        /// Set containing both broad classes of known weakness:
51        /// generator-side and entropy-source.
52        pub const WEAK: Self = Self::GENERATOR_WEAK.with_entropy_weak();
53
54        /// Returns whether either known form of weakness is present.
55        #[must_use]
56        pub const fn is_weak(self) -> bool {
57            self.intersects(Self::WEAK)
58        }
59    }
60}
