73define_error! { composite: fmt(f)
74    /// All possible integer operation errors.
75    #[doc = crate::_doc_meta!{location("num/error")}]
76    pub enum IntError {
77        +tag: _tags!(no),
78        DOC_NOT_IMPLEMENTED: +const NotImplemented => NotImplemented,
79        +tag: _tags!(no),
80        DOC_NOT_SUPPORTED: +const NotSupported => NotSupported,
81
82        // used by ops: core, combinatoric, modulo, prime, root:
83        DOC_OVERFLOW: +const Overflow(s|0: Option<Sign>) => Overflow(*s),
84        // used by ops: modulo:
85        DOC_NO_INVERSE: +const NoInverse => NoInverse,
86        // used by ops: combinatoric, root:
87        DOC_NON_NEGATIVE_REQUIRED: +const NonNegativeRequired => NonNegativeRequired,
88        // used by ops: division, modulo, root:
89        DOC_NON_ZERO_REQUIRED: +const NonZeroRequired => NonZeroRequired,
90        // used by ops: combinatoric, factor:
91        DOC_MISMATCHED_SIZES: +const MismatchedSizes => MismatchedSizes,
92
93        // DOC_POSITIVE_REQUIRED: +const PositiveRequired => PositiveRequired,
94        // DOC_INCOMPATIBLE_BOUNDS: +const IncompatibleBounds => IncompatibleBounds,
95    }
96}
