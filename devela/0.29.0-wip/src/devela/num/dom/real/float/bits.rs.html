3//! Defines [`f32bits`]|`_niche`, [`f64bits`]|`_niche`.

8use crate::{ConstInit, NonMaxU32, NonMaxU64, niche};

10// Macro helper to implement the types f32bits, f32bits_niche, ...
11macro_rules! _num_dom_real_float_define_fbits {
12    () => {
13        _num_dom_real_float_define_fbits![f32+u32, f64+u64];
14    };
15    ($($float:ident + $bits:ident),+ $(,)?) => { $crate::paste! { $(
16        _num_dom_real_float_define_fbits![
17            @[<$float bits>], [<$float bits_niche>], $float, $bits, [<NonMax $bits:upper>]
18        ];
19    )+ }};
20    (@$non_niche:ident, $niche:ident, $float:ident, $bits:ident, $NM:ident) => { $crate::paste! {
21        /* non-niche */
22
23        #[doc = crate::_tags!(num ffi)]
24        #[doc = "Bitwise wrapper for `" $float "` providing `Eq`, `Ord`, and `Hash`."]
25        #[doc = crate::_doc_meta!{location("num/dom/real")}]
26        ///
27        #[doc = "This stores the raw IEEE-754 bits of a `" $float "` in a `" $bits "`."]
28        /// Ordering and hashing operate on the raw bit pattern, not the numeric value.
29        /// All bit patterns are preserved, including NaNs and signed zeroes.
30        ///
31        #[doc = "For a masked and niche-compressed variant, see [`" $niche "`]."]
32        #[must_use]
33        #[repr(transparent)]
34        #[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
35        pub struct $non_niche($bits);
36
37        impl $non_niche {
38            /// Converts a `$float` into its raw-bit representation.
39            pub const fn new(float: $float) -> Self { Self(float.to_bits()) }
40            /// Wraps an existing raw bit pattern.
41            pub const fn from_bits(bits: $bits) -> Self { Self(bits) }
42
43            /// Reinterprets the stored bits as a `$float`.
44            pub const fn as_float(self) -> $float { <$float>::from_bits(self.0) }
45            /// Returns the underlying raw bits unchanged.
46            pub const fn as_bits(self) -> $bits { self.0 }
47
48            /// Converts to the niche variant.
49            pub const fn to_niche(self) -> $niche { $niche::from_bits(self.as_bits()) }
50        }
51
52        impl Default for $non_niche {
53            /// The default value is the bit pattern of `0.0`.
54            fn default() -> Self { Self::new(0.0) }
55        }
56        impl ConstInit for $non_niche {
57            /// The initialization value is the bit pattern of `0.0`.
58            const INIT: Self = Self::new(0.0);
59        }
60
61        impl From<$float> for $non_niche {
62            fn from(from: $float) -> $non_niche { $non_niche::new(from) }
63        }
64        impl From<$non_niche> for $float {
65            fn from(from: $non_niche) -> $float { from.as_float() }
66        }
67
68        /* niche */
69
70        #[doc = crate::_tags!(num niche)]
71        #[doc = "Bitwise wrapper for `" $float "` stored through a masked [`" $NM "`]."]
72        #[doc = crate::_doc_meta!{location("num/dom/real")}]
73        ///
74        /// This preserves all IEEE-754 bit patterns except the prohibited value.
75        /// Ordering and hashing operate on the masked bit pattern.
76        /// Signed zeroes and all NaNs are represented, with a single NaN payload
77        /// (the prohibited one) mapped to an adjacent payload.
78        ///
79        /// # Reserved NaN Payload
80        #[doc = "The only prohibited bit pattern is `" $bits "::MAX`."]
81        ///
82        /// It is a quiet NaN that also matches the highest value of the
83        /// underlying unsigned integer type. This makes it a good sentinel:
84        /// it never appears in normal floating-point or millisecond values.
85        ///
86        /// If this bit pattern is used, it is automatically replaced with
87        /// the previous representable payload, keeping every other value intact.
88        ///
89        #[doc = "For the unmasked, fully identity-preserving variant, see [`" $non_niche "`]."]
90        #[must_use]
91        #[repr(transparent)]
92        #[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
93        pub struct $niche($NM);
94
95        impl $niche {
96            #[doc = "Converts a `" $float "` into its masked bit representation."]
97            ///
98            /// All bit patterns are preserved except the prohibited one.
99            pub const fn new(float: $float) -> Self {
100                Self(niche![lossy float.to_bits(), $bits; != MAX])
101            }
102            #[doc = "Wraps an existing raw bit pattern, masking it through `" $NM "`."]
103            pub const fn from_bits(bits: $bits) -> Self {
104                Self(niche![lossy bits, $bits; != MAX])
105            }
106
107            #[doc = "Reinterprets the stored bits as a `" $float "`."]
108            pub const fn as_float(self) -> $float { <$float>::from_bits(self.as_bits()) }
109
110            /// Returns the masked raw bits.
111            ///
112            /// The prohibited value is never returned; it is always replaced
113            #[doc = "by the remapped payload defined by `" $NM "`."]
114            pub const fn as_bits(self) -> $bits { self.0.get() }
115
116            /// Converts to the non-niche variant.
117            pub const fn to_non_niche(self) -> $non_niche { $non_niche::from_bits(self.as_bits()) }
118        }
119
120        impl Default for $niche {
121            /// The default value is the bit pattern of `0.0`.
122            fn default() -> Self { Self::new(0.0) }
123        }
124        impl ConstInit for $niche {
125            /// The initialization value is the bit pattern of `0.0`.
126            const INIT: Self = Self::new(0.0);
127        }
128
129        impl From<$float> for $niche {
130            fn from(from: $float) -> $niche { $niche::new(from) }
131        }
132        impl From<$niche> for $float {
133            fn from(from: $niche) -> $float { from.as_float() }
134        }
135    }};
136}
