11/// Macro helper for implementing methods for `Float`, from `std`.
12///
13/// $lib: the library to use.
14/// $f: the floating-point type to support.
15/// $doc: an optional documentation string.
16/// $opfn: the original operation function name.
17/// $op: the new operation function name in Float.
18macro_rules! _num_dom_real_float_impl_std {
19    (
20        // Matches a wildcard floating-point type (f*).
21        // Expands to specific floating-point types (f32, f64).
22        $lib:ident : f* : $($rest:tt)*
23    ) => {
24        _num_dom_real_float_impl_std![$lib : f32 : $($rest)*];
25        _num_dom_real_float_impl_std![$lib : f64 : $($rest)*];
26    };
27    (
28        // Matches a specific floating-point type and any number of operations.
29        // Generates the impl block for Float<$f> and calls the matching implementation.
30        $lib:ident : $f:ty : $($rest:tt)*
31    ) => { $crate::paste! {
32        impl Float<$f> {
33            _num_dom_real_float_impl_std![@$lib : $f : $($rest)*];
34        }
35    }};
36    (
37        // Matches multiple operations and uses recursion to process each one.
38        @$lib:ident : $f:ty : $($doc:literal)? $(+const$($_c:lifetime)?)?
39        $opfn:ident = $op:ident : $($arg:ident),*
40        ; $($rest:tt)*
41    ) => {
42        _num_dom_real_float_impl_std![@$lib : $f :
43            $($doc)? $(+const$($_c)?)? $opfn = $op : $($arg),*];
44        _num_dom_real_float_impl_std![@$lib : $f : $($rest)*];
45    };
46    (
47        // Matches a single operation and implements it using the `std` library.
48        @std : $f:ty : $($doc:literal)? $(+const$($_c:lifetime)?)?
49        $opfn:ident = $op:ident : $($arg:ident),* $(;)?
50    ) => {
51        $(#[doc = $doc])?
52        pub $(const$($_c)?)? fn $op(self, $($arg: $f),*) -> Float<$f> {
53            Float(<$f>::$opfn(self.0, $($arg),*))
54        }
55    };
56}

145    /// $f:   the floating-point type.
146    /// $e:   the integer type for integer exponentiation.
147    macro_rules! _impl_float_std_custom {
148        () => {
149            _impl_float_std_custom![(f32, i32), (f64, i32)];
150        };
151        ($( ($f:ty, $e:ty)),+) => {
152            $( _impl_float_std_custom![@$f, $e]; )+
153        };
154        (@$f:ty, $e:ty) => {
155            /// # *Implementations using the `std` feature*.
156            impl Float<$f> {
157                /// Raises itself to the `p` integer power.
158                pub fn powi(self, p: $e) -> Float<$f> { Float(<$f>::powi(self.0, p)) }
159                /// Both the sine and cosine.
160                pub fn sin_cos(self) -> (Float<$f>, Float<$f>) {
161                    let (sin, cos) = <$f>::sin_cos(self.0);
162                    (Float(sin), Float(cos))
163                }
164                /// The nearest integer to itself, default rounding
165                ///
166                /// This is equivalent to [`round_ties_away`][Self::round_ties_away].
167                pub const fn round(self) -> Float<$f> { Float(self.0.round()) }
168
169                /// The integral part of `x`.
170                /// # Formulation
171                #[doc = crate::_FLOAT_FORMULA_TRUNC!()]
172                pub const fn trunc(self) -> Float<$f> {
173                    Float(self.0.trunc())
174                }
175                /// The fractional part of `x`.
176                /// # Formulation
177                #[doc = crate::_FLOAT_FORMULA_FRACT!()]
178                pub const fn fract(self) -> Float<$f> {
179                    Float(self.0.fract())
180                }
181                /// The integral and fractional parts of `x`.
182                /// # Formulation
183                #[doc = crate::_FLOAT_FORMULA_SPLIT!()]
184                pub const fn split(self) -> (Float<$f>, Float<$f>) {
185                    let trunc = self.trunc();
186                    (trunc, Float(self.0 - trunc.0))
187                }
188            }
189        };
190    }
