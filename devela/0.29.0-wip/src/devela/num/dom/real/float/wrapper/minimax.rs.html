3//! Shared methods implemented using Horner minimax.
4//
5// TODO: f16, 128
6
7use super::definition::Float;
8use crate::{_FloatInternals, is};
9
10/// Implements methods independently of any features
11///
12/// $f:   the floating-point type.
13/// $uf:  unsigned int type with the same bit-size.
14/// $ue:  unsigned int type used for integer exponentiation and number of terms (u32).
15macro_rules! impl_float_shared_minimax {
16    () => {
17        impl_float_shared_minimax![(f32:u32, u32), (f64:u64, u32)];
18    };
19    ($( ($f:ty:$uf:ty, $ue:ty)),+ $(,)?) => {
20        $( impl_float_shared_minimax![@$f:$uf, $ue]; )+
21    };
22    (@$f:ty:$uf:ty, $ue:ty) => {
23        ///
24        /// # *Common methods implemented using Taylor series*
25        impl Float<$f> {
26            /// The sine calculated using a minimax polynomial evaluated by Horner's method.
27            ///
28            /// # Formulation
29            #[doc = crate::_FLOAT_FORMULA_SIN_MINIMAX!()]
30            ///
31            /// This approximation uses a degree-N minimax polynomial over the range
32            /// [-π/2, π/2], combined with argument reduction. It provides higher accuracy
33            /// than the Taylor series for the same number of terms and avoids divisions.
34            pub const fn sin_minimax(self) -> Float<$f> {
35                let (x, _flip) = self.reduce_half_pi();
36                x.sin_minimax_from_reduced()
37            }
38
39            /// The cosine calculated using a minimax polynomial evaluated by Horner's method.
40            ///
41            /// # Formulation
42            #[doc = crate::_FLOAT_FORMULA_COS_MINIMAX!()]
43            ///
44            /// This approximation uses a degree-N minimax polynomial over the range
45            /// [-π/2, π/2], combined with argument reduction. It provides higher accuracy
46            /// than the Taylor series for the same number of terms and avoids divisions.
47            pub const fn cos_minimax(self) -> Float<$f> {
48                let (x, flip) = self.reduce_half_pi();
49                x.cos_minimax_from_reduced(flip)
50            }
51
52            /// Computes the sine and the cosine
53            /// using a minimax polynomial evaluated by Horner's method.
54            ///
55            /// The current implementation computes separately the sine and the cosine.
56            #[inline(always)]
57            pub const fn sin_cos_minimax(self) -> (Float<$f>, Float<$f>) {
58                let (x, flip) = self.reduce_half_pi();
59                (x.sin_minimax_from_reduced(), x.cos_minimax_from_reduced(flip))
60            }
61
62            /* private helpers */
63
64            #[inline(always)]
65            const fn sin_minimax_from_reduced(self) -> Float<$f> {
66                let x2 = self.0 * self.0;
67                let poly = Float(x2).eval_poly_const::<{_FloatInternals::<$f>::SIN_COS_DEGREE}>(
68                    &_FloatInternals::<$f>::SIN_COEFFS
69                ).0;
70                Float(self.0 + self.0 * x2 * poly)
71            }
72
73            #[inline(always)]
74            const fn cos_minimax_from_reduced(self, flip: bool) -> Float<$f> {
75                let x2 = self.0 * self.0;
76                let poly = Float(x2).eval_poly_const::<{_FloatInternals::<$f>::SIN_COS_DEGREE}>(
77                    &_FloatInternals::<$f>::COS_COEFFS
78                ).0;
79                let c = 1.0 + x2 * poly;
80                Float(is![flip, -c, c])
81            }
82
83            #[inline(always)]
84            const fn reduce_half_pi(self) -> (Float<$f>, bool) {
85                use crate::FloatConst;
86                // first stage: reduce into [-PI, PI]
87                let mut xr = self.0 - <$f>::TAU * Float(self.0 / <$f>::TAU).round().0;
88
89                // second stage: fold into [-PI/2, PI/2]
90                let mut flip = false;
91                if xr > <$f>::FRAC_PI_2 {
92                    xr = <$f>::PI - xr;
93                    flip = true;
94                } else if xr < - <$f>::FRAC_PI_2 {
95                    xr = - <$f>::PI - xr;
96                    flip = true;
97                }
98                (Float(xr), flip)
99            }
100
101        }
102    };
103}
104impl_float_shared_minimax!();
