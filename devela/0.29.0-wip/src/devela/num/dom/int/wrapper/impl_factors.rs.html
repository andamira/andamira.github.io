71            /// Returns the count of prime factors.
72            ///
73            /// # Examples
74            /// ```
75            /// # use devela::Int;
76            #[doc = "assert_eq![Int(24_" $t ").factors_prime_count(), 4];"]

82            #[must_use]
83            pub const fn factors_prime_count(self) -> usize {
84                let mut n = self.0.abs();
85                is![n == 0, return 0];
86                let mut count = 0;
87                // Divide by 2 until the number is odd
88                while n % 2 == 0 {
89                    count += 1;
90                    n /= 2;
91                }
92                // Divide by odd numbers starting from 3
93                let mut i = 3;
94                while i * i <= n {
95                    while n % i == 0 {
96                        count += 1;
97                        n /= i;
98                    }
99                    i += 2;
100                }
101                // If the remaining number is greater than 2, it's a prime factor
102                is![n > 2, count += 1];
103                count
104            }

106            /// Returns the count of unique prime factors.
107            ///
108            /// # Examples
109            /// ```
110            /// # use devela::Int;
111            #[doc = "assert_eq![Int(24_" $t ").factors_prime_unique_count(), 2];"]
112            #[doc = "assert_eq![Int(-24_" $t ").factors_prime_unique_count(), 2];"]

114            #[must_use]
115            pub const fn factors_prime_unique_count(self) -> usize {
116                let mut n = self.0.abs();
117                is![n == 0, return 0];
118                let mut count = 0;
119                let mut last = 0;
120
121                // Divide by 2 until the number is odd
122                while n % 2 == 0 {
123                    is![last != 2, { count += 1; last = 2 }];
124                    n /= 2;
125                }
126                // Divide by odd numbers starting from 3
127                let mut i = 3;
128                while i * i <= n {
129                    while n % i == 0 {
130                        is![last != i, { count += 1; last = i }];
131                        n /= i;
132                    }
133                    i += 2;
134                }
135                // If the remaining number is greater than 2,
136                // and not the same as the last factor, it's a prime factor
137                if n > 2 && last != n {
138                    count += 1;
139                }
140                count
141            }

143            /// Writes the factors in `fbuf`, and the unique prime factors in `upfbuf`.
144            ///
145            /// Returns a tuple with the number of factors and unique prime factors found.
146            ///
147            /// # Errors
148            /// Returns [`MismatchedSizes`] if the total number of factors is greater

151            /// # Examples
152            /// ```
153            /// # use devela::Int;
154            /// let (mut fbuf, mut upbuf) = ([0; 20], [0; 20]);

160            pub const fn factors_buf(self, fbuf: &mut [$t], upfbuf: &mut [$t])
161                -> Result<(usize, usize)> {
162                let n = self.0.abs();
163                is![n == 0, return Ok((0, 0))];
164                is![n == 1, { fbuf[0] = 1; return Ok((1, 0)); }];
165                let mut f_count = 0;
166                fbuf[f_count] = 1;
167                f_count += 1;
168                let prime_factors_count = unwrap![ok? self.factors_prime_unique_buf(upfbuf)];
169                let mut i = 2;
170                while i <= n {
171                    if n % i == 0 {
172                        if f_count < fbuf.len() {
173                            fbuf[f_count] = i;
174                            f_count += 1;
175                        } else {
176                            return Err(MismatchedSizes);
177                        }
178                    }
179                    i += 1;
180                }
181                Ok((f_count, prime_factors_count))
182            }

222            /// Writes the prime factors in the given `buffer`.
223            ///
224            /// Returns the number of factors found
225            ///
226            /// # Errors
227            /// Returns [`MismatchedSizes`] if the total number of factors
228            /// is greater than the length of the `buffer`.
229            ///
230            /// # Examples
231            /// ```
232            /// # use devela::Int;
233            /// let mut buf = [0; 5];
234            #[doc = "assert_eq![Int(24_" $t ").factors_prime_buf(&mut buf), Ok(4)];"]
235            ///
236            /// assert_eq![buf[..4], [2, 2, 2, 3]];
237            #[doc = "assert![Int(24_" $t " * 4).factors_prime_buf(&mut buf).is_err()];"]
238            /// assert_eq![buf, [2, 2, 2, 2, 2]]; // the factor of 3 didn't fit
239            ///
240            #[doc = "assert_eq![Int(0_" $t ").factors_prime_buf(&mut buf), Ok(0)];"]
241            #[doc = "assert_eq![Int(1_" $t ").factors_prime_buf(&mut buf), Ok(0)];"]
242            #[doc = "assert_eq![Int(7_" $t ").factors_prime_buf(&mut buf), Ok(1)];"]
243            /// assert_eq![buf[..1], [7]];
244            /// ```
245            pub const fn factors_prime_buf(self, buffer: &mut [$t]) -> Result<usize> {
246                is![self.0 == 0, return Ok(0)];
247                let (mut n, mut idx) = (self.0.abs(), 0);
248                while n % 2 == 0 {
249                    if idx < buffer.len() {
250                        buffer[idx] = 2; idx += 1; n /= 2;
251                    } else {
252                        return Err(MismatchedSizes);
253                    }
254                }
255                let mut i = 3;
256                while i * i <= n {
257                    while n % i == 0 {
258                        if idx < buffer.len() {
259                            buffer[idx] = i; idx += 1; n /= i;
260                        } else {
261                            return Err(MismatchedSizes);
262                        }
263                    }
264                    i += 2;
265                }
266                if n > 2 {
267                    if idx < buffer.len() {
268                        buffer[idx] = n; idx += 1;
269                    } else {
270                        return Err(MismatchedSizes);
271                    }
272                }
273                Ok(idx)
274            }

293            pub const fn factors_prime_unique_buf(self, buffer: &mut [$t]) -> Result<usize> {
294                let prime_factors_count = unwrap![ok? self.factors_prime_buf(buffer)];
295                let mut unique_count = 1;
296                let mut last_unique = buffer[0];
297                whilst! { i in 1..prime_factors_count; {
298                    if buffer[i] != last_unique {
299                        if unique_count < buffer.len() {
300                            buffer[unique_count] = buffer[i];
301                            last_unique = buffer[i];
302                            unique_count += 1;
303                        } else {
304                            return Err(MismatchedSizes);
305                        }
306                    }
307                }}
308                Ok(unique_count)
309            }

349            /// assert_eq![fbuf, [2, 2, 2]]; // the factor of 5 didn't fit

351            ///
352            /// // When `ebuffer` is not large enough:
353            /// let mut fbuf = [0; 4];
354            /// let mut ebuf = [0; 1];
355            #[doc = "assert![Int(24_" $t
356                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf).is_err()];"]
357            /// assert_eq![fbuf[..ebuf.len()], [2]]; // 2^3, Err, …
358            /// assert_eq![ebuf[..], [3]];
359            /// ```
360            // IMPROVE: differenciate between both errors more clearly.
361            pub const fn factors_prime_unique_exp_buf(self, fbuffer: &mut [$t], ebuffer: &mut [u32])
362            -> Result<usize> {
363                let prime_factors_count = unwrap![ok? self.factors_prime_buf(fbuffer)];
364                is![prime_factors_count == 0, return Ok(0)];
365
366                let mut current_factor = fbuffer[0]; // current factor
367                let mut unique_idx = 0; // current unique factor index
368                let mut exp_count = 1; //
369
370                whilst! { i in 1..prime_factors_count; {
371                    // Same factor as before, increment the exponent count
372                    if fbuffer[i] == current_factor {
373                        exp_count += 1;
374                    } else {
375                        // New factor found, store the previous factor and its exp_count
376                        fbuffer[unique_idx] = current_factor;
377                        is![unique_idx >= ebuffer.len(), return Err(MismatchedSizes)];
378                        ebuffer[unique_idx] = exp_count;
379                        unique_idx += 1; // Move to the next unique factor
380
381                        // Reset for the new factor
382                        current_factor = fbuffer[i];
383                        exp_count = 1;
384                    }
385                }}
386                // Store the last factor and its exponent count
387                if unique_idx < fbuffer.len() && unique_idx < ebuffer.len() {
388                    fbuffer[unique_idx] = current_factor;
389                    ebuffer[unique_idx] = exp_count;
390                    unique_idx += 1; // increment the index to represent the unique count
391                } else {
392                    return Err(MismatchedSizes);
393                }
394                Ok(unique_idx)
395            }

414            pub const fn factors_prime_unique_plus_buf(self, pfbuf: &mut [$t], upfbuf: &mut [$t]
415            ) -> Result<(usize, usize)> {
416                let prime_factors_count = unwrap![ok? self.factors_prime_buf(pfbuf)];
417                let mut unique_count = 0;
418                whilst! { i in 0..prime_factors_count; {
419                    let mut unique = true;
420                    whilst! { j in 0..unique_count; {
421                        if pfbuf[i] == upfbuf[j] {
422                            unique = false;
423                            break;
424                        }
425                    }}
426                    if unique {
427                        if unique_count < upfbuf.len() {
428                            upfbuf[unique_count] = pfbuf[i];
429                            unique_count += 1;
430                        } else {
431                            return Err(MismatchedSizes);
432                        }
433                    }
434                }}
435                Ok((prime_factors_count, unique_count))

594                Ok((f_count, prime_factors_count))
595            }

597            /// Writes the prime factors in the given `buffer`.
598            ///
599            /// Returns the number of factors found.
600            ///
601            /// # Errors
602            /// Returns [`MismatchedSizes`] if the total number of factors
603            /// is greater than the length of the `buffer`.
604            ///
605            /// # Examples
606            /// ```
607            /// # use devela::Int;
608            /// let mut buf = [0; 5];
609            #[doc = "assert_eq![Int(24_" $t ").factors_prime_buf(&mut buf), Ok(4)];"]
610            ///
611            /// assert_eq![buf[..4], [2, 2, 2, 3]];
612            #[doc = "assert![Int(24_" $t " * 4).factors_prime_buf(&mut buf).is_err()];"]
613            /// assert_eq![buf, [2, 2, 2, 2, 2]]; // the factor of 3 didn't fit
614            ///
615            #[doc = "assert_eq![Int(0_" $t ").factors_prime_buf(&mut buf), Ok(0)];"]
616            #[doc = "assert_eq![Int(1_" $t ").factors_prime_buf(&mut buf), Ok(0)];"]
617            #[doc = "assert_eq![Int(7_" $t ").factors_prime_buf(&mut buf), Ok(1)];"]
618            /// assert_eq![buf[..1], [7]];
619            /// ```
620            pub const fn factors_prime_buf(self, buffer: &mut [$t]) -> Result<usize> {
621                let n = self.0;
622                is![n == 0, return Ok(0)];
623                let (mut n, mut idx) = (n, 0);
624                while n % 2 == 0 {
625                    if idx < buffer.len() {
626                        buffer[idx] = 2; idx += 1; n /= 2;
627                    } else {
628                        return Err(MismatchedSizes);
629                    }
630                }
631                let mut i = 3;
632                while i * i <= n {
633                    while n % i == 0 {
634                        if idx < buffer.len() {
635                            buffer[idx] = i; idx += 1; n /= i;
636                        } else {
637                            return Err(MismatchedSizes);
638                        }
639                    }
640                    i += 2;
641                }
642                if n > 2 {
643                    if idx < buffer.len() {
644                        buffer[idx] = n; idx += 1;
645                    } else {
646                        return Err(MismatchedSizes);
647                    }
648                }
649                Ok(idx)
650            }

652            /// Writes the prime factors in the given `buffer`.
653            ///
654            /// The `buffer` must be large enough to hold all the non-unique factors of `n`.
655            /// In that case the function will return the number of unique factors found.
656            ///
657            /// # Errors
658            /// Returns [`MismatchedSizes`] otherwise. In that case the buffer
659            /// will only contain the non-unique factors that could fit, same as
660            #[doc = "[`factors_prime_buf`](#method.factors_prime_buf" $d ")."]
661            ///
662            /// # Examples
663            /// ```
664            /// # use devela::Int;
665            /// let mut uniq = [0; 5];
666            #[doc = "assert_eq![Int(24_" $t ").factors_prime_unique_buf(&mut uniq), Ok(2)];"]
667            /// assert_eq![uniq, [2, 3, 2, 3, 0]];
668            /// ```
669            pub const fn factors_prime_unique_buf(self, buffer: &mut [$t]) -> Result<usize> {
670                let prime_factors_count = unwrap![ok? self.factors_prime_buf(buffer)];
671                let mut unique_count = 1;
672                let mut last_unique = buffer[0];
673                whilst! { i in 1..prime_factors_count; {
674                    if buffer[i] != last_unique {
675                        if unique_count < buffer.len() {
676                            buffer[unique_count] = buffer[i];
677                            last_unique = buffer[i];
678                            unique_count += 1;
679                        } else {
680                            return Err(MismatchedSizes);
681                        }
682                    }
683                }}
684                Ok(unique_count)
685            }

687            /// Writes the unique prime factors in the given `fbuffer`, and the
688            /// associated exponent in the given `ebuffer` at the same index.
689            ///
690            /// The `fbuffer` must be large enough to hold all the non-unique factors of `n`.
691            /// In that case the function will return the number of unique factors found.
692            ///
693            /// # Errors
694            /// Returns [`MismatchedSizes`] otherwise. In that case the buffer
695            /// will only contain the non-unique factors that could fit, same as
696            #[doc = "[`factors_prime_buf`](#method.factors_prime_buf" $d ")."]
697            ///
698            /// Returns [`MismatchedSizes`] if `ebuffer` is not large enough as well.
699            /// In that case the number of unique factors will equal `ebuffer.len()`.
700            ///
701            /// # Examples
702            /// ```
703            /// # use devela::Int;
704            /// let mut fbuf = [0; 4];
705            /// let mut ebuf = [0; 2];
706            #[doc = "assert_eq![Int(40_" $t
707                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf), Ok(2)];"]
708            /// assert_eq![fbuf[..2], [2, 5]]; // 2^3, 5^1, …
709            /// assert_eq![ebuf[..2], [3, 1]];
710            ///
711            #[doc = "assert_eq![Int(0_" $t
712                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf), Ok(0)];"]
713            #[doc = "assert_eq![Int(1_" $t
714                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf), Ok(0)];"]
715            #[doc = "assert_eq![Int(7_" $t
716                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf), Ok(1)];"]
717            /// assert_eq![fbuf[..1], [7]]; // 7^1
718            /// assert_eq![ebuf[..1], [1]];
719            ///
720            /// // When `fbuffer` is not large enough:
721            /// let mut fbuf = [0; 3];
722            /// let mut ebuf = [0; 2];
723            #[doc = "assert![Int(24_" $t
724                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf).is_err()];"]
725            /// assert_eq![fbuf, [2, 2, 2]]; // the factor of 5 didn't fit
726            /// assert_eq![ebuf, [0, 0]]; // the exponents didn't get written
727            ///
728            /// // When `ebuffer` is not large enough:
729            /// let mut fbuf = [0; 4];
730            /// let mut ebuf = [0; 1];
731            #[doc = "assert![Int(24_" $t
732                ").factors_prime_unique_exp_buf(&mut fbuf, &mut ebuf).is_err()];"]
733            /// assert_eq![fbuf[..ebuf.len()], [2]]; // 2^3, Err, …
734            /// assert_eq![ebuf[..], [3]];
735            /// ```
736            // IMPROVE: differenciate between both errors more clearly.
737            pub const fn factors_prime_unique_exp_buf(self, fbuffer: &mut [$t], ebuffer: &mut [u32])
738            -> Result<usize> {
739                let prime_factors_count = unwrap![ok? self.factors_prime_buf(fbuffer)];
740                is![prime_factors_count == 0, return Ok(0)];
741
742                let mut current_factor = fbuffer[0]; // current factor
743                let mut unique_idx = 0; // current unique factor index
744                let mut exp_count = 1; //
745
746                whilst! { i in 1..prime_factors_count; {
747                    // Same factor as before, increment the exponent count
748                    if fbuffer[i] == current_factor {
749                        exp_count += 1;
750                    } else {
751                        // New factor found, store the previous factor and its exp_count
752                        fbuffer[unique_idx] = current_factor;
753                        is![unique_idx >= ebuffer.len(), return Err(MismatchedSizes)];
754                        ebuffer[unique_idx] = exp_count;
755                        unique_idx += 1; // Move to the next unique factor
756
757                        // Reset for the new factor
758                        current_factor = fbuffer[i];
759                        exp_count = 1;
760                    }
761                }}
762                // Store the last factor and its exponent count
763                if unique_idx < fbuffer.len() && unique_idx < ebuffer.len() {
764                    fbuffer[unique_idx] = current_factor;
765                    ebuffer[unique_idx] = exp_count;
766                    unique_idx += 1; // increment the index to represent the unique count
767                } else {
768                    return Err(MismatchedSizes);
769                }
770                Ok(unique_idx)
771            }

773            /// Writes the prime factors in `pfbuf`, and the unique factors in `upfbuf`.
774            ///
775            /// Returns the number of factors found.
776            ///
777            /// # Errors
778            /// Returns [`MismatchedSizes`] if the total number of factors
779            /// is greater than the length of the `buffer`.
780            ///
781            /// # Examples
782            /// ```
783            /// # use devela::Int;
784            /// let (mut fac, mut uniq) = ([0; 5], [0; 5]);
785            #[doc = "assert_eq![Int(24_" $t
786                ").factors_prime_unique_plus_buf(&mut fac, &mut uniq), Ok((4, 2))];"]
787            /// assert_eq![fac, [2, 2, 2, 3, 0]];
788            /// assert_eq![uniq, [2, 3, 0, 0, 0]];
789            /// ```
790            pub const fn factors_prime_unique_plus_buf(self, pfbuf: &mut [$t], upfbuf: &mut [$t]
791                ) -> Result<(usize, usize)> {
792                let prime_factors_count = unwrap![ok? self.factors_prime_buf(pfbuf)];
793                let mut unique_count = 0;
794                whilst! { i in 0..prime_factors_count; {
795                    let mut unique = true;
796                    whilst! { j in 0..unique_count; {
797                        if pfbuf[i] == upfbuf[j] {
798                            unique = false;
799                            break;
800                        }
801                    }}
802                    if unique {
803                        if unique_count < upfbuf.len() {
804                            upfbuf[unique_count] = pfbuf[i];
805                            unique_count += 1;
806                        } else {
807                            return Err(MismatchedSizes);
808                        }
809                    }
810                }}
811                Ok((prime_factors_count, unique_count))
812            }
813        }
814    }};
