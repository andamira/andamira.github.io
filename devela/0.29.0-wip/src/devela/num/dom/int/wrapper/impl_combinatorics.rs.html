64            /// Returns the factorial.
65            ///
66            /// Permutations of *n* items, ordered, where $n = r$.
67            ///
68            /// # Formula
69            #[doc = _DOC_INT_FORMULA_FACTORIAL!()]
70            ///
71            /// These are the maximum numbers whose factorials can fit within
72            /// standard signed integer types:
73            ///
74            /// 5 for `i8`, 7 for `i16`, 12 for `i32`, 20 for `i64` and 33 for `i128`.
75            ///
76            /// # Errors
77            /// Returns [`NonNegativeRequired`] if $n<0$,
78            /// and [`Overflow`] if the result can't fit the type.
79            ///
80            /// # Examples
81            /// ```
82            /// # use devela::Int;
83            #[doc = "assert_eq![Ok(Int(120)), Int(5_" $t ").factorial()];"]
84            #[doc = "assert_eq![Ok(Int(6)), Int(3_" $t ").factorial()];"]
85            #[doc = "assert_eq![Ok(Int(1)), Int(0_" $t ").factorial()];"]
86            #[doc = "assert![Int(-3_" $t ").factorial().is_err()];"]
87            #[doc = "assert![Int(" $t "::MAX).factorial().is_err()];"]
88            /// ```
89            pub const fn factorial(self) -> Result<Int<$t>> {
90                let n = self.0;
91                is![n < 0, return Err(NonNegativeRequired)];
92                let (mut n, mut result): ($t, $t) = (n.abs(), 1);
93                while n > 1 {
94                    result = if let Some(res) = result.checked_mul(n) {
95                        res
96                    } else {
97                        return Err(Overflow(None));
98                    };
99                    n -= 1;
100                }
101                Ok(Int(result))
102            }

113            /// ## Closed-Form Formulas

121            /// These are the maximum numbers whose subfactorials can fit within

133            /// # Examples

141            #[doc = "assert![Int(127_" $t ").subfactorial().is_err()];"]
142            /// # }

144            /// # Links
145            /// - The list of subfactorials is available in <https://oeis.org/A000166>.
146            pub const fn subfactorial(self) -> Result<Int<$t>> {
147                let n = self.0;
148                is![n < 0, return Err(NonNegativeRequired)];
149                match n {
150                    0 => Ok(Int(1)),
151                    1 => Ok(Int(0)),
152                    _ => {
153                        let a = match Self::subfactorial(Int(n - 1)) {
154                            Ok(a) => a.0, Err(e) => return Err(e),
155                        };
156                        let b = match Self::subfactorial(Int(n - 2)) {
157                            Ok(b) => b.0, Err(e) => return Err(e),
158                        };
159                        let sum = a + b; // can't overflow
160                        if let Some(res) = (n - 1).checked_mul(sum) {
161                            Ok(Int(res))
162                        } else {
163                            return Err(Overflow(None));
164                        }
165                    }
166                }
167            }

172            #[doc = _DOC_INT_FORMULA_COMBINE!()]
173            ///
174            /// # Errors

176            /// [`MismatchedSizes`] if $r > n$, and

188            pub const fn combine(self, r: $t) -> Result<Int<$t>> {
189                let n = self.0;
190                is![n < 0 || r < 0, return Err(NonNegativeRequired)];
191                is![r > n, return Err(MismatchedSizes)];
192                let (mut num, mut den): ($t, $t) = (1, 1);
193                whilst![i in 0..r; {
194                    num = if let Some(res) = num.checked_mul(n - i) {
195                        res
196                    } else {
197                        return Err(Overflow(None))
198                    };
199                    den = if let Some(res) = den.checked_mul(i + 1) {
200                        res
201                    } else {
202                        return Err(Overflow(None))
203                    };
204                }];
205                Ok(Int(num / den))
206            }
