3//! Absence of niche constraints and commitments.
4//!
5//! This module defines building blocks for working *around* niche-optimized
6//! numeric representations, either by explicitly opting out of layout
7//! optimization or by abstracting over whether such optimization is used.
8//!
9//! - [`MaybeNiche`] represents the **absence of representation commitment**:
10//!   it abstracts over primitive integers, niche-optimized types, and their
11//!   non-optimized counterparts, allowing generic code to remain independent
12//!   of the chosen representation.
13//!
14//! - [`NonNiche`] represents the **absence of niche constraints**: it mirrors
15//!   the API of niche-constrained numeric types while storing values unchanged.
16//!
17//! These types are complementary: one selects a concrete, non-optimized
18//! representation, while the other erases the distinction between optimized
19//! and non-optimized forms.

32#[doc = crate::_tags!(maybe niche)]
33/// A zero-cost wrapper that abstracts over niche and non-niche types.
34#[doc = crate::_doc_meta!{location("num/grain/niche")}]
35///
36/// `MaybeNiche<T>` is a transparent wrapper that preserves the representation
37/// semantics of `T` without imposing a niche choice, and introduces no
38/// invariants of its own.
39///
40/// It enables niche-agnostic generic code over integer-like representations.
41///
42/// See also [`NonNiche`], which provides an explicit non-optimized representation
43/// with the same public API as niche-constrained types.
44///
45/// # Implementations
46///
47/// Implemented for:
48/// - [Primitive integers](#impl-MaybeNiche<u8>):
49///   `u8`, `u16`, `u32`, `u64`, `u128`, `usize`,
50///   `i8`, `i16`, `i32`, `i64`, `i128`, `isize`.
51/// - [`NonNiche<T>`](#impl-MaybeNiche<NonNiche<u8>>) for all supported primitives.
52/// - [`NonZero<T>`](#impl-MaybeNiche<NonZero<u8>>) for all supported primitives.
53/// - [`NonValue*<V>`](#impl-MaybeNiche<NonValueU8<V>>) for all supported primitives.
54///
55/// # Methods and constants
56///
57/// The API is identical across all implementations
58/// (links below point to the `u8` specialization).
59///
60/// - Niche properties: [`IS_NICHE`](#associatedconstant.IS_NICHE),
61///   [`is_niche`](#method.is_niche).
62/// - Contiguity: [`IS_CONTIGUOUS`](#associatedconstant.IS_CONTIGUOUS),
63///   [`is_contiguous`](#method.is_contiguous).
64/// - Signedness: [`HAS_NEGATIVE`](#associatedconstant.HAS_NEGATIVE),
65///   [`has_negative`](#method.has_negative).
66/// - Bounds: [`MIN`](#associatedconstant.MIN), [`MAX`](#associatedconstant.MAX).
67/// - Zero: [`ZERO`](#associatedconstant.ZERO), [`has_zero`](#method.has_zero).
68/// - Construction:
69///   - From `T`:
70///     [`new`](#method.new) / `MaybeNiche(T)`.
71///   - From primitive:
72///     [`try_from_prim`](#method.try_from_prim),
73///     [`from_prim_lossy`](#method.from_prim_lossy).
74///   - From `usize`:
75///     [`try_from_usize`](#method.try_from_usize),
76///     [`from_usize_saturating`](#method.from_usize_saturating),
77///     [`from_usize_wrapping`](#method.from_usize_wrapping).
78/// - Extraction:
79///   - To `T`:
80///     [`get`](#method.get)/[`repr`](#method.repr).
81///   - To primitive:
82///     [`get_prim`](#method.get_prim)/[`prim`](#method.prim).
83///   - To `usize`:
84///     [`try_to_usize`](#method.try_to_usize),
85///     [`to_usize_saturating`](#method.to_usize_saturating),
86///     [`to_usize_wrapping`](#method.to_usize_wrapping).
87#[repr(transparent)]
88#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
89pub struct MaybeNiche<T: Copy>(pub T);

91/// impl helper for [`MaybeNiche`].
92macro_rules! impl_maybe {
93    () => {
94        impl_maybe!(u8, u16, u32, u64, u128, usize, i8, i16, i32, i64, i128, isize);
95    };
96    ($($T:ty),+) => {
97        // % $is_niche, $prim, $T
98        //   $(,<const V: $V>) $(,*$get)? $(,^$new)? $(,@$non0)?
99        // ------------------------------------------------------
100        $( impl_maybe![% false, $T, $T]; )+
101        $( impl_maybe![% false, $T, NonNiche<$T>, *get, ^new]; )+
102        $( impl_maybe![% true,  $T, NonZero<$T>, *get, ^new, @non0]; )+
103        $crate::paste!{ $(
104            impl_maybe![% true, $T, [<NonValue $T:camel>] <V>, <const V: $T>,
105            *get, ^new];
106        )+ }
107    };
108    (%
109     $is_niche:literal, // IS_NICHE
110     $prim:ty,
111     $T:ty $(, <const $V:ident : $v:ty>)?
112     $(, *$get:ident)?  // for: get_prim, prim
113     $(, ^$new:ident)?  // for: from_prim, *_unchecked
114     $(, @$non0:ident)? // identifies nonzero types, for: from_prim_lossy
115    ) => {
116        impl $(<const $V: $v>)? ConstInit for MaybeNiche<$T> where $T: ConstInit {
117            const INIT: Self = Self::new(<$T>::INIT);
118        }
119
120        impl $(<const $V: $v>)? MaybeNiche<$T> {
121            /* constants */
122
123            /// Whether this representation uses a memory niche for layout optimization.
124            pub const IS_NICHE: bool = $is_niche;
125
126            /// Whether the representable domain forms a contiguous interval.
127            ///
128            /// That is, whether every primitive value `v` such that
129            /// `MIN <= v <= MAX` is representable.
130            pub const IS_CONTIGUOUS: bool = {
131                // primitives, NonNiche, NonZero
132                #[crate::compile(none($(<const $V: $v>)?))]
133                const fn _is_contiguous $(<const $V: $v>)? () -> bool { true }
134                // NonValue
135                #[crate::compile(some($(<const $V: $v>)?))]
136                const fn _is_contiguous $(<const $V: $v>)? () -> bool {
137                    V == <$prim>::MIN || V == <$prim>::MAX
138                }
139                _is_contiguous$(::<$V>)?()
140            };
141
142            /// Whether the representable domain includes negative values.
143            #[allow(unused_comparisons, reason = "for unsigned types")]
144            pub const HAS_NEGATIVE: bool = Self::MIN.prim() < 0;
145
146            /// The minimum representable value.
147            pub const MIN: Self = Self(<$T>::MIN);
148            /// The maximum representable value.
149            pub const MAX: Self = Self(<$T>::MAX);
150
151            /// The zero value, if representable by this type.
152            pub const ZERO: Option<Self> = unwrap![ok_some Self::try_from_prim(0)];
153
154            /* constructors */
155
156            /// Creates a new `MaybeNiche` containing `value`.
157            #[must_use] #[inline(always)]
158            pub const fn new(value: $T) -> Self { Self(value) }
159
160            /// Creates a new `MaybeNiche` from a primitive value.
161            /// # Errors
162            /// - [`InvalidValue`] if the value violates the validity invariant of `T`.
163            #[inline(always)]
164            pub const fn try_from_prim(primitive: $prim) -> Result<Self, InvalidValue> {
165                // WAIT: [proc_macro_hygiene](https://github.com/rust-lang/rust/issues/54727)
166                // Can't use `compile` on expr or stmt, e.g.: return Some(Self(primitive));
167                // so we need to leverage defined functions instead.
168
169                // NonNiche, NonValue, NonZero
170                #[crate::compile(some($($new)?))]
171                const fn _new $(<const $V: $v>)? (v: $prim) -> Option<$T> { $( <$T>::$new(v) )? }
172                // primitives
173                #[crate::compile(none($($new)?))]
174                const fn _new $(<const $V: $v>)? (v: $prim) -> Option<$T> { Some(v) }
175
176                Ok(Self(unwrap![some_ok_or? _new(primitive), crate::InvalidValue]))
177            }
178            /// Creates a new `MaybeNiche` without any checks.
179            /// # Safety
180            /// For niche-optimized types, callers must ensure that
181            /// `value` satisfies the variant's validity constraints.
182            #[must_use] #[inline(always)]
183            #[cfg(all(not(feature = "safe_num"), feature = "unsafe_niche"))]
184            #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_niche")))]
185            pub const unsafe fn from_prim_unchecked(primitive: $prim) -> Self {
186                // NonNiche, NonValue, NonZero
187                #[crate::compile(some($($new)?))]
188                const fn _new $(<const $V: $v>)? (v: $prim) -> $T { $crate::paste! {
189                    unsafe { $( <$T>:: [< $new _unchecked >](v) )? }
190                }}
191                // primitives
192                #[crate::compile(none($($new)?))]
193                const fn _new $(<const $V: $v>)? (v: $prim) -> $T { v }
194
195                Self(_new(primitive))
196            }
197
198            /// Creates a new `MaybeNiche` from a primitive value, converting invalid inputs
199            /// into a valid but *approximate* representation.
200            ///
201            /// This constructor performs a **lossy conversion**, applying a best-effort
202            /// fallback when the primitive violates the underlying type's invariant:
203            ///
204            /// - For `NonZero*` types, `0` becomes the smallest valid value (`MIN`).
205            /// - For `NonValue*` types, conversion defers to their own
206            ///   [`new_lossy`](NonValueU8::new_lossy)-style semantics.
207            /// - For `NonNiche` and primitive integers, the value is used as-is.
208            #[must_use] #[inline(always)]
209            pub const fn from_prim_lossy(value: $prim) -> Self {
210                // NonZero converts
211                #[crate::compile(all(some($($new)?), some($($non0)?)))]
212                const fn _lossy $(<const $V: $v>)? (v: $prim) -> $T {
213                    if v == 0 { <$T>::MIN }
214                    else {
215                        cfg_select! { all(feature = "unsafe_niche", not(feature = "safe_num")) => {
216                            unwrap![some_guaranteed_or_ub <$T>::new(v)]
217                        } _ => { unwrap![some <$T>::new(v)] }}
218                    }
219                }
220                // NonNiche, NonValue (has its own rules)
221                #[crate::compile(all(some($($new)?), none($($non0)?)))]
222                const fn _lossy $(<const $V: $v>)? (v: $prim) -> $T { $crate::paste! {
223                    $( <$T>:: [< $new _lossy >](v) )?
224                }}
225                // primitives
226                #[crate::compile(none($($new)?))]
227                const fn _lossy $(<const $V: $v>)? (v: $prim) -> $T { v }
228
229                Self(_lossy(value))
230            }
231
232            /// Tries to create a new `MaybeNiche` from a `usize`.
233            /// # Errors
234            /// - [`NicheValueError::Overflow`] if the value cannot be represented by the
235            ///   underlying primitive type.
236            /// - [`NicheValueError::InvalidValue`] if the value violates the validity
237            ///   invariant of `T`.
238            #[inline(always)]
239            pub const fn try_from_usize(value: usize) -> Result<Self, NicheValueError> {
240                // NonNiche, NonValue, NonZero
241                #[crate::compile(some($($new)?))]
242                const fn _new $(<const $V: $v>)? (v: $prim) -> Option<$T> { $( <$T>::$new(v) )? }
243                // primitives
244                #[crate::compile(none($($new)?))]
245                const fn _new $(<const $V: $v>)? (v: $prim) -> Option<$T> { Some(v) }
246
247                let prim = $crate::paste! { Cast(value).[<checked_cast_to_ $prim>]() };
248                let prim = unwrap![ok_err_map? prim, |e| NicheValueError::from_overflow(e)];
249                Ok(Self(unwrap![some_ok_or? _new(prim), NicheValueError::InvalidValue]))
250            }
251
252            /// Creates a new `MaybeNiche` from a `usize`, saturating at numeric bounds.
253            ///
254            /// The conversion applies the following steps:
255            /// 1. The `usize` value is saturated to the bounds of the primitive type.
256            /// 2. If the resulting value violates the niche invariant of `T`,
257            ///    a best-effort lossy conversion is applied.
258            #[must_use] #[inline(always)]
259            pub const fn from_usize_saturating(value: usize) -> Self {
260                let prim = $crate::paste! { Cast(value).[<saturating_cast_to_ $prim>]() };
261                Self::from_prim_lossy(prim)
262            }
263
264            /// Creates a new `MaybeNiche` from a `usize`, wrapping at numeric bounds.
265            ///
266            /// The conversion applies the following steps:
267            /// 1. The `usize` value is wrapped to the primitive type.
268            /// 2. If the resulting value violates the niche invariant of `T`,
269            ///    a best-effort lossy conversion is applied.
270            #[must_use] #[inline(always)]
271            pub const fn from_usize_wrapping(value: usize) -> Self {
272                let prim = $crate::paste! { Cast(value).[<wrapping_cast_to_ $prim>]() };
273                Self::from_prim_lossy(prim)
274            }
275
276            /* queries */
277
278            /// Returns `true` if this representation uses a memory niche.
279            #[must_use] #[inline(always)]
280            pub const fn is_niche(self) -> bool { Self::IS_NICHE }
281
282            /// Returns `true` if the representable domain is contiguous.
283            ///
284            /// That is, if every primitive value `v` such that `MIN <= v <= MAX` is representable.
285            #[must_use] #[inline(always)]
286            pub const fn is_contiguous(self) -> bool { Self::IS_CONTIGUOUS }
287
288            /// Returns `true` if the representable domain includes negative values.
289            #[must_use] #[inline(always)]
290            pub const fn has_negative(self) -> bool { Self::HAS_NEGATIVE }
291
292            /// Returns `true` if this type can represent zero.
293            #[must_use] #[inline(always)]
294            pub const fn has_zero(self) -> bool { Self::ZERO.is_some() }
295
296            /// Returns `true` if `self == other`.
297            #[must_use] #[inline(always)]
298            pub const fn eq(self, other: Self) -> bool { self.prim() == other.prim() }
299            /// Returns `true` if `self != other`.
300            #[must_use] #[inline(always)]
301            pub const fn ne(self, other: Self) -> bool { self.prim() != other.prim() }
302            /// Returns `true` if `self < other`.
303            #[must_use] #[inline(always)]
304            pub const fn lt(self, other: Self) -> bool { self.prim() < other.prim() }
305            /// Returns `true` if `self <= other`.
306            #[must_use] #[inline(always)]
307            pub const fn le(self, other: Self) -> bool { self.prim() <= other.prim() }
308            /// Returns `true` if `self > other`.
309            #[must_use] #[inline(always)]
310            pub const fn gt(self, other: Self) -> bool { self.prim() > other.prim() }
311            /// Returns `true` if `self >= other`.
312            #[must_use] #[inline(always)]
313            pub const fn ge(self, other: Self) -> bool { self.prim() >= other.prim() }
314
315            /* representation access */
316
317            /// Returns the validated (niche-aware) representation.
318            #[must_use] #[inline(always)]
319            pub const fn get(self) -> $T { self.0 }
320
321            /// Alias of [`get`][Self::get], emphasizing representational access.
322            #[must_use] #[inline(always)]
323            pub const fn repr(self) -> $T { self.get() }
324
325            /* primitive access */
326
327            /// Returns the primitive carrier value.
328            #[must_use] #[inline(always)]
329            pub const fn get_prim(self) -> $prim { self.0 $( . $get() )? }
330
331            /// Alias of [`get_prim`][Self::get_prim], emphasizing primitive access.
332            #[must_use] #[inline(always)]
333            pub const fn prim(self) -> $prim { self.get_prim() }
334
335            /* casts */
336
337            /// Converts the value into a `usize`, returning an error on overflow.
338            ///
339            /// # Errors
340            /// Will return [`Overflow`] if `self` can't fit in a `usize`.
341            #[inline(always)]
342            pub const fn try_to_usize(self) -> Result<usize, Overflow> {
343                Cast(self.get_prim()).checked_cast_to_usize()
344            }
345            /// Converts the value into a `usize`, saturating at the numeric bounds.
346            #[must_use] #[inline(always)]
347            pub const fn to_usize_saturating(self) -> usize {
348                Cast(self.get_prim()).saturating_cast_to_usize()
349            }
350            /// Converts the value into a `usize`, wrapping at the numeric bounds.
351            #[must_use] #[inline(always)]
352            pub const fn to_usize_wrapping(self) -> usize {
353                Cast(self.get_prim()).wrapping_cast_to_usize()
354            }
355        }
356    };

360#[doc = crate::_tags!(no niche)]
361/// A zero-cost wrapper that mimics a niche type without using a niche.
362#[doc = crate::_doc_meta!{location("num/grain/niche")}]
363///
364/// `NonNiche` represents the absence of niche constraints while preserving
365/// API symmetry with niche-optimized numeric types.
366///
367/// Practical note:
368///
369/// `NonNiche<T>` is a concrete representation choice. It gives you a parallel,
370/// non-optimized version of a type (e.g. fast vs compact) while keeping the
371/// same public API and implementation surface.
372///
373/// Used in types like `charu` to provide a non-optimized parallel to
374/// their niche-enabled counterparts.
375///
376/// See also [`MaybeNiche`], which abstracts over primitive, niche-optimized,
377/// and non-optimized integer representations.
378#[repr(transparent)]
379#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
380pub struct NonNiche<T: Copy>(pub T);
381
382#[rustfmt::skip]
383impl<T: Copy> NonNiche<T> {
384    /// Creates a new `NonNiche` with the given value.
385    ///
386    /// This always succeeds, unlike `NonZero*` types.
387    #[must_use] #[inline(always)]
388    pub const fn new(value: T) -> Option<Self> { Some(Self(value)) }
389
390    /// Creates a new `NonNiche` without checking.
391    /// # Safety
392    /// This is always safe since `NonNiche` doesn't have any validity constraints.
393    /// Method provided for API completion.
394    #[must_use] #[inline(always)]
395    #[cfg(all(not(feature = "safe_num"), feature = "unsafe_niche"))]
396    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_niche")))]
397    pub const unsafe fn new_unchecked(value: T) -> Self { Self(value) }
398
399    /// Creates a NonNiche, automatically converting any prohibited values.
400    ///
401    /// There are no prohibited values. Method provided for API completion.
402    #[must_use] #[inline(always)]
403    pub const fn new_lossy(value: T) -> Self { Self(value) }
404
405    /// Extracts the inner value.
406    #[must_use] #[inline(always)]
407    pub const fn get(self) -> T { self.0 }
408}
409
410#[rustfmt::skip]
411impl<T: Copy> From<T> for NonNiche<T> {
412    #[inline(always)]
413    fn from(value: T) -> Self { Self(value) }
414}
415
416impl<T: Copy + ConstInit> ConstInit for NonNiche<T> {
417    const INIT: Self = Self(T::INIT);
418}

436#[cfg(test)]
437mod tests {
438    use super::{MaybeNiche, NonNiche, NonValueU8, NonZero};
439
440    #[test]
441    fn maybe_niche() {
442        let u = MaybeNiche(3_u8);
443        let nn = MaybeNiche(NonNiche::<u8>::new(3).unwrap());
444        let n0 = MaybeNiche(NonZero::<u8>::new(3).unwrap());
445        let nv0 = MaybeNiche(NonValueU8::<0>::new(3).unwrap());
446        let nv1 = MaybeNiche(NonValueU8::<1>::new(3).unwrap());
447
448        // u8
449        assert_eq![u.is_contiguous(), true];
450        assert_eq![u.is_niche(), false];
451        assert_eq![u.has_zero(), true];
452        // NonNiche
453        assert_eq![nn.is_contiguous(), true];
454        assert_eq![nn.is_niche(), false];
455        assert_eq![nn.has_zero(), true];
456        // NonZero
457        assert_eq![n0.is_contiguous(), true];
458        assert_eq![n0.is_niche(), true];
459        assert_eq![n0.has_zero(), false];
460        // NonValue::<0>
461        assert_eq![nv0.is_contiguous(), true];
462        assert_eq![nv0.is_niche(), true];
463        assert_eq![nv0.has_zero(), false];
464        // NonValue::<1>
465        assert_eq![nv1.is_contiguous(), false];
466        assert_eq![nv1.is_niche(), true];
467        assert_eq![nv1.has_zero(), true];
468    }
