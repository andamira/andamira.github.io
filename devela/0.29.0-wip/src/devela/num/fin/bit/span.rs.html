8#[doc = crate::_tags!(bit data_structure)]
9/// A contiguous span of bits in an integer carrier.
10#[doc = crate::_doc_meta!{location("bin")}]
11///
12/// Stores the bounds and derived masks for a packed field.
13#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
14pub struct BitSpan<T> {
15    /// First bit in the span.
16    pub start: u32,
17    /// Last bit in the span.
18    pub end: u32,
19    /// Number of bits in the span.
20    pub width: u32,
21    /// Shifted mask covering the span.
22    pub mask: T,
23    /// Maximum value that fits in the span.
24    pub max: T,
25}
26impl<T> BitSpan<T> {
27    /// Returns a new bit span from precomputed metadata.
28    ///
29    /// # Panics
30    /// Panics if `start > end`.
31    #[must_use]
32    pub const fn from_parts(start: u32, end: u32, mask: T, max: T) -> Self {
33        assert!(start <= end, "BitSpan: start must be <= end");
34        Self { start, end, width: end - start + 1, mask, max }
35    }
36}
37impl<T: Copy> BitSpan<T> {
38    /// Returns whether the span occupies exactly one bit.
39    #[must_use]
40    pub const fn is_single(self) -> bool {
41        self.width == 1
42    }
43    /// Returns whether `bit` is inside the span.
44    #[must_use]
45    pub const fn contains_bit(self, bit: u32) -> bool {
46        self.start <= bit && bit <= self.end
47    }
48    /// Returns whether `other` is fully inside `self`.
49    #[must_use]
50    pub const fn contains_span(self, other: Self) -> bool {
51        self.start <= other.start && other.end <= self.end
52    }
53    /// Returns whether both spans share any bits.
54    #[must_use]
55    pub const fn overlaps(self, other: Self) -> bool {
56        self.start <= other.end && other.start <= self.end
57    }
58    /// Returns whether both spans share no bits.
59    #[must_use]
60    pub const fn is_disjoint(self, other: Self) -> bool {
61        !self.overlaps(other)
62    }
63    /// Returns whether the span fits in a carrier with `bits` bits.
64    #[must_use]
65    pub const fn fits_in(self, bits: u32) -> bool {
66        self.start <= self.end && self.end < bits
67    }
68}

70// methods for unsigned integer primitives
71macro_rules! impl_bit_span {
72    ($($T:ty),+ $(,)?) => { $(
73        impl BitSpan<$T> {
74            /// Returns a new bit span.
75            ///
76            /// # Panics
77            /// Panics if `start > end` or if `end` exceeds the carrier width.
78            #[must_use]
79            pub const fn new(start: u32, end: u32) -> Self {
80                assert!(start <= end, "BitSpan: start must be <= end");
81                assert!(end < <$T>::BITS, "BitSpan: end exceeds carrier width");
82                let width = end - start + 1;
83                let mask = Bitwise::<$T>::mask_range(start, end).0;
84                let max = if width >= <$T>::BITS {
85                    !0 as $T
86                } else {
87                    ((1 as $T) << width) - 1
88                };
89                Self { start, end, width, mask, max }
90            }
91            /// Returns a new checked bit span.
92            ///
93            /// # Errors
94            /// Returns an error if `start > end` or if `end` exceeds the carrier width.
95            pub const fn try_new(start: u32, end: u32)
96                -> Result<Self, MismatchedBounds> {
97                match Bitwise::<$T>::mask_range_checked(start, end) {
98                    Ok(mask) => {
99                        let width = end - start + 1;
100                        let max = if width >= <$T>::BITS {
101                            !0 as $T
102                        } else {
103                            ((1 as $T) << width) - 1
104                        };
105                        Ok(Self { start, end, width, mask: mask.0, max })
106                    }
107                    Err(e) => Err(e),
108                }
109            }
110
111            /// Returns whether `value` fits in this span.
112            #[must_use]
113            pub const fn value_fits(self, value: $T) -> bool { value <= self.max }
114
115            /// Extracts the span value from `bits`.
116            #[must_use]
117            pub const fn get_value(self, bits: $T) -> $T {
118                Bitwise::<$T>(bits).get_value_range(self.start, self.end).0
119            }
120
121            /// Returns `bits` with the span value replaced.
122            ///
123            /// The value is masked to fit the span width.
124            #[must_use]
125            pub const fn set_value(self, bits: $T, value: $T) -> $T {
126                Bitwise::<$T>(bits).set_value_range(value, self.start, self.end).0
127            }
128            /// Returns `bits` with the checked span value replaced.
129            ///
130            /// # Errors
131            /// Returns an error if the span is invalid
132            /// or if `value` does not fit within the span width.
133            pub const fn try_set_value(self, bits: $T, value: $T)
134                -> Result<$T, MismatchedBounds> {
135                match Bitwise::<$T>(bits)
136                    .set_value_range_checked_strict(value, self.start, self.end)
137                {
138                    Ok(bits) => Ok(bits.0),
139                    Err(e) => Err(e),
140                }
141            }
142            /// Clears the span bits in `bits`.
143            #[must_use]
144            pub const fn clear_in(self, bits: $T) -> $T { bits & !self.mask }
145
146            /// Returns whether every span bit is set in `bits`.
147            #[must_use]
148            pub const fn is_full_in(self, bits: $T) -> bool { bits & self.mask == self.mask }
149
150            /// Returns whether no span bit is set in `bits`.
151            #[must_use]
152            pub const fn is_zero_in(self, bits: $T) -> bool { bits & self.mask == 0 }
153        }
154    )+ };
155}

158#[cfg(test)]
159mod tests {
160    use super::BitSpan;
161
162    #[test]
163    fn bit_span_metadata() {
164        let s = BitSpan::<u16>::new(3, 6);
165
166        assert_eq!(s.start, 3);
167        assert_eq!(s.end, 6);
168        assert_eq!(s.width, 4);
169        assert_eq!(s.mask, 0b0000_0000_0111_1000);
170        assert_eq!(s.max, 0b1111);
171
172        assert!(s.contains_bit(3));
173        assert!(s.contains_bit(6));
174        assert!(!s.contains_bit(7));
175        assert!(s.value_fits(15));
176        assert!(!s.value_fits(16));
177    }
178
179    #[test]
180    fn bit_span_value_ops() {
181        let s = BitSpan::<u16>::new(3, 6);
182
183        let bits = s.set_value(0, 0b1010);
184        assert_eq!(bits, 0b0101_0000);
185        assert_eq!(s.get_value(bits), 0b1010);
186
187        assert!(s.try_set_value(0, 0b1111).is_ok());
188        assert!(s.try_set_value(0, 0b1_0000).is_err());
189
190        assert_eq!(s.clear_in(bits), 0);
191    }
