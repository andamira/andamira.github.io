3//! Dense row-major and column-major ordinal encodings.
4//!
5//! Coordinates and dimensions use spatial axis order:
6//! `[x, y, z, ...]` and `[width, height, depth, ...]`.
7//!
8//! Row-major order keeps the first axis contiguous.
9//! Column-major order keeps the last axis contiguous.
10//
11
12use crate::{Order, is, unwrap, whilst};

14/* row-major */

16/// # Row-major ordinal encodings
17///
18/// The first coordinate axis changes fastest.
19impl Order {
20    /// Encodes 2D coordinates in row-major order.
21    ///
22    /// Does not validate coordinates or arithmetic overflow.
23    #[must_use]
24    #[inline]
25    pub const fn row_major_from_2d(x: usize, y: usize, width: usize) -> usize {
26        y * width + x
27    }
28
29    /// Encodes bounded 2D coordinates in row-major order.
30    ///
31    /// Returns `None` if the coordinates are out of bounds or the grid length
32    /// cannot be represented by `usize`.
33    #[must_use]
34    pub const fn row_major_try_from_2d(
35        x: usize,
36        y: usize,
37        width: usize,
38        height: usize,
39    ) -> Option<usize> {
40        is! { x >= width || y >= height, return None }
41        let _len = unwrap![some? width.checked_mul(height)];
42        Some(Self::row_major_from_2d(x, y, width))
43    }
44
45    /// Decodes a row-major ordinal into 2D coordinates.
46    ///
47    /// Does not validate the ordinal. `width` must be nonzero.
48    #[must_use]
49    #[inline]
50    pub const fn row_major_to_2d(i: usize, width: usize) -> (usize, usize) {
51        (i % width, i / width)
52    }
53
54    /// Decodes a bounded row-major ordinal into 2D coordinates.
55    ///
56    /// Returns `None` if the grid is empty, its length overflows, or `i` is
57    /// outside the grid.
58    #[must_use]
59    pub const fn row_major_try_to_2d(
60        i: usize,
61        width: usize,
62        height: usize,
63    ) -> Option<(usize, usize)> {
64        is! { width == 0 || height == 0, return None }
65        let len = unwrap![some? width.checked_mul(height)];
66        is! { i >= len, return None }
67        Some(Self::row_major_to_2d(i, width))
68    }
69
70    /// Encodes 3D coordinates in row-major order.
71    ///
72    /// Does not validate coordinates or arithmetic overflow.
73    #[must_use]
74    #[inline]
75    pub const fn row_major_from_3d(
76        x: usize,
77        y: usize,
78        z: usize,
79        width: usize,
80        height: usize,
81    ) -> usize {
82        (z * height + y) * width + x
83    }
84
85    /// Encodes bounded 3D coordinates in row-major order.
86    ///
87    /// Returns `None` if the coordinates are out of bounds or the grid length
88    /// cannot be represented by `usize`.
89    #[must_use]
90    pub const fn row_major_try_from_3d(
91        x: usize,
92        y: usize,
93        z: usize,
94        width: usize,
95        height: usize,
96        depth: usize,
97    ) -> Option<usize> {
98        is! { x >= width || y >= height || z >= depth,return None }
99        let area = unwrap![some? width.checked_mul(height)];
100        let _len = unwrap![some? area.checked_mul(depth)];
101        Some(Self::row_major_from_3d(x, y, z, width, height))
102    }
103
104    /// Decodes a row-major ordinal into 3D coordinates.
105    ///
106    /// Does not validate the ordinal. `width` and `height` must be nonzero.
107    #[must_use]
108    #[inline]
109    pub const fn row_major_to_3d(i: usize, width: usize, height: usize) -> (usize, usize, usize) {
110        let area = width * height;
111        let z = i / area;
112        let rem = i % area;
113        let y = rem / width;
114        let x = rem % width;
115        (x, y, z)
116    }
117
118    /// Decodes a bounded row-major ordinal into 3D coordinates.
119    #[must_use]
120    pub const fn row_major_try_to_3d(
121        i: usize,
122        width: usize,
123        height: usize,
124        depth: usize,
125    ) -> Option<(usize, usize, usize)> {
126        is! { width == 0 || height == 0 || depth == 0, return None }
127        let area = unwrap![some? width.checked_mul(height)];
128        let len = unwrap![some? area.checked_mul(depth)];
129        is! { i >= len, return None }
130        Some(Self::row_major_to_3d(i, width, height))
131    }
132}

215    /// Decodes a column-major ordinal into 3D coordinates.

221        let area = height * depth;

223        let rem = i % area;

225        let z = rem % depth;
226        (x, y, z)
227    }

242    }

247/// # N-dimensional ordinal encodings
248impl Order {
249    /// Returns the number of coordinates in the grid.
250    ///
251    /// Returns `None` if the product overflows `usize`.
252    ///
253    /// An empty dimension makes the grid empty. A zero-dimensional grid has
254    /// one coordinate, following the empty-product convention.
255    #[must_use]
256    pub const fn grid_volume<const N: usize>(dims: [usize; N]) -> Option<usize> {
257        // The mathematical product is zero regardless of the other dimensions.
258        whilst! { k in 0..N; {
259            is! { dims[k] == 0, return Some(0) }
260        }}
261        let mut volume = 1_usize;
262        whilst! { k in 0..N; {
263            volume = unwrap![some? volume.checked_mul(dims[k])];
264        }}
265        Some(volume)
266    }
267
268    /// Computes row-major strides.
269    ///
270    /// The returned first stride is `1`.
271    ///
272    /// For `[width, height, depth]`, this returns
273    /// `[1, width, width * height]`.
274    #[must_use]
275    pub const fn row_major_strides<const N: usize>(dims: [usize; N]) -> Option<[usize; N]> {
276        let mut strides = [1_usize; N];
277        whilst! { k in 1..N; {
278            strides[k] = unwrap![some? strides[k - 1].checked_mul(dims[k - 1])];
279        }}
280        Some(strides)
281    }
282
283    /// Computes column-major strides.
284    ///
285    /// The returned last stride is `1`.
286    ///
287    /// For `[width, height, depth]`, this returns
288    /// `[height * depth, depth, 1]`.
289    #[must_use]
290    pub const fn col_major_strides<const N: usize>(dims: [usize; N]) -> Option<[usize; N]> {
291        let mut strides = [1_usize; N];
292        whilst! { k in rev 1..N; {
293            strides[k - 1] = unwrap![some? strides[k].checked_mul(dims[k])];
294        }}
295        Some(strides)
296    }
297
298    /// Encodes N-dimensional coordinates in row-major order.
299    ///
300    /// Does not validate coordinates, dimensions, or arithmetic overflow.
301    #[must_use]
302    pub const fn row_major_from<const N: usize>(coords: [usize; N], dims: [usize; N]) -> usize {
303        let mut i = 0;
304        whilst! { k in rev 0..N; {
305            i = i * dims[k] + coords[k];
306        }}
307        i
308    }
309
310    /// Encodes bounded N-dimensional coordinates in row-major order.
311    #[must_use]
312    pub const fn row_major_try_from<const N: usize>(
313        coords: [usize; N],
314        dims: [usize; N],
315    ) -> Option<usize> {
316        whilst! { k in 0..N; {
317            is! { coords[k] >= dims[k], return None }
318        }}
319        let _volume = unwrap![some? Self::grid_volume(dims)];
320        Some(Self::row_major_from(coords, dims))
321    }
322
323    /// Decodes a row-major ordinal into N-dimensional coordinates.
324    ///
325    /// Does not validate the ordinal. Every dimension must be nonzero.
326    #[must_use]
327    pub const fn row_major_to<const N: usize>(mut i: usize, dims: [usize; N]) -> [usize; N] {
328        let mut coords = [0; N];
329        whilst! { k in 0..N; {
330            coords[k] = i % dims[k];
331            i /= dims[k];
332        }}
333        coords
334    }
335
336    /// Decodes a bounded row-major ordinal into N-dimensional coordinates.
337    #[must_use]
338    pub const fn row_major_try_to<const N: usize>(
339        i: usize,
340        dims: [usize; N],
341    ) -> Option<[usize; N]> {
342        let volume = unwrap![some? Self::grid_volume(dims)];
343        is! { i >= volume, return None }
344        Some(Self::row_major_to(i, dims))
345    }
346
347    /// Encodes N-dimensional coordinates in column-major order.
348    ///
349    /// Does not validate coordinates, dimensions, or arithmetic overflow.
350    #[must_use]
351    pub const fn col_major_from<const N: usize>(coords: [usize; N], dims: [usize; N]) -> usize {
352        let mut i = 0;
353        whilst! { k in 0..N; {
354            i = i * dims[k] + coords[k];
355        }}
356        i
357    }
358
359    /// Encodes bounded N-dimensional coordinates in column-major order.
360    #[must_use]
361    pub const fn col_major_try_from<const N: usize>(
362        coords: [usize; N],
363        dims: [usize; N],
364    ) -> Option<usize> {
365        whilst! { k in 0..N; {
366            if coords[k] >= dims[k] {
367                return None;
368            }
369        }}
370        let _volume = unwrap![some? Self::grid_volume(dims)];
371        Some(Self::col_major_from(coords, dims))
372    }
373
374    /// Decodes a column-major ordinal into N-dimensional coordinates.
375    ///
376    /// Does not validate the ordinal. Every dimension must be nonzero.
377    #[must_use]
378    pub const fn col_major_to<const N: usize>(mut i: usize, dims: [usize; N]) -> [usize; N] {
379        let mut coords = [0; N];
380        whilst! { k in rev 0..N; {
381            coords[k] = i % dims[k];
382            i /= dims[k];
383        }}
384        coords
385    }
386
387    /// Decodes a bounded column-major ordinal into N-dimensional coordinates.
388    #[must_use]
389    pub const fn col_major_try_to<const N: usize>(
390        i: usize,
391        dims: [usize; N],
392    ) -> Option<[usize; N]> {
393        let volume = unwrap![some? Self::grid_volume(dims)];
394        is! { i >= volume, return None }
395        Some(Self::col_major_to(i, dims))
396    }

399#[cfg(test)]
400mod tests {
401    use crate::Order;
402    #[test]
403    fn row_major_2d_roundtrip() {
404        let (width, height) = (4, 3);
405        for y in 0..height {
406            for x in 0..width {
407                let i = Order::row_major_from_2d(x, y, width);
408                assert_eq!(Order::row_major_to_2d(i, width), (x, y));
409                assert_eq!(Order::row_major_try_from_2d(x, y, width, height), Some(i),);
410                assert_eq!(Order::row_major_try_to_2d(i, width, height), Some((x, y)),);
411            }
412        }
413    }
414    #[test]
415    fn col_major_2d_roundtrip() {
416        let (width, height) = (4, 3);
417        for x in 0..width {
418            for y in 0..height {
419                let i = Order::col_major_from_2d(x, y, height);
420                assert_eq!(Order::col_major_to_2d(i, height), (x, y));
421                assert_eq!(Order::col_major_try_from_2d(x, y, width, height), Some(i),);
422                assert_eq!(Order::col_major_try_to_2d(i, width, height), Some((x, y)),);
423            }
424        }
425    }
426    #[test]
427    fn major_3d_roundtrip() {
428        let (width, height, depth) = (4, 3, 2);
429        let len = width * height * depth;
430        for i in 0..len {
431            let row = Order::row_major_to_3d(i, width, height);
432            assert_eq!(Order::row_major_from_3d(row.0, row.1, row.2, width, height), i,);
433            let col = Order::col_major_to_3d(i, height, depth);
434            assert_eq!(Order::col_major_from_3d(col.0, col.1, col.2, height, depth), i,);
435        }
436        assert_eq!(Order::row_major_try_from_3d(2, 1, 1, width, height, depth), Some(18),);
437        assert_eq!(Order::col_major_try_from_3d(2, 1, 1, width, height, depth), Some(15),);
438    }
439    #[test]
440    fn generic_matches_fixed() {
441        let dims = [4, 3, 2];
442        assert_eq!(Order::row_major_strides(dims), Some([1, 4, 12]),);
443        assert_eq!(Order::col_major_strides(dims), Some([6, 2, 1]),);
444        assert_eq!(Order::row_major_from([2, 1, 1], dims), Order::row_major_from_3d(2, 1, 1, 4, 3),);
445        assert_eq!(Order::col_major_from([2, 1, 1], dims), Order::col_major_from_3d(2, 1, 1, 3, 2),);
446        for i in 0..24 {
447            let row = Order::row_major_to(i, dims);
448            let col = Order::col_major_to(i, dims);
449            assert_eq!(Order::row_major_from(row, dims), i);
450            assert_eq!(Order::col_major_from(col, dims), i);
451        }
452    }
453    #[test]
454    fn checked_boundaries() {
455        assert_eq!(Order::row_major_try_from_2d(4, 0, 4, 3), None);
456        assert_eq!(Order::row_major_try_to_2d(12, 4, 3), None);
457        assert_eq!(Order::col_major_try_to_2d(0, 4, 0), None);
458        assert_eq!(Order::grid_volume([usize::MAX, 2]), None);
459        assert_eq!(Order::grid_volume([usize::MAX, 2, 0]), Some(0),);
460        assert_eq!(Order::row_major_try_from([], []), Some(0));
461        assert_eq!(Order::row_major_try_to(0, []), Some([]));
462        assert_eq!(Order::row_major_try_to(1, []), None);
463    }
