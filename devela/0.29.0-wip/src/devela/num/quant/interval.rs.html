20#[doc = crate::_tags!(quant)]
21/// Creates an [`Interval`] using extended range notation.
22#[doc = crate::_doc_meta!{location("num/quant")}]
23///
24/// # Syntax
25///
26/// ## Closed Lower Bound `[l, …`
27/// ```text
28/// [l, u]    interval![l, ..= u]    // closed-closed
29/// [l, u)    interval![l, .. u]     // closed-open
30/// [l, ∞)    interval![l, ..]       // closed-unbounded
31/// ```
32///
33/// ## Open Lower Bound `(l, …` *extension syntax*
34/// ```text
35/// (l, u)    interval![l <.. u]     // open-open
36/// (l, u]    interval![l <..= u]    // open-closed
37/// (l, ∞)    interval![l <..]       // open-unbounded
38/// ```
39///
40/// ## Unbounded Lower Bound `(-∞, …`
41/// ```text
42/// (-∞, u]   interval![..= u]       // unbounded-closed
43/// (-∞, u)   interval![.. u]        // unbounded-open
44/// (-∞, ∞)   interval![..]          // fully unbounded (inferred type)
45/// (-∞, ∞)   interval![.., T]       // fully unbounded (explicit type) (e.g., `.., i32`)
46/// ```
47///
48/// # Notes
49/// - Commas required for expressions: `interval![a(), ..= b()]`
50/// - Optional for literals/blocks: `interval![1..=3]` ≡ `interval![1, ..=3]`
51/// - `<..` indicates open-left bounds (non-standard Rust extension)
52///
53/// # Examples
54/// ```
55/// # use devela::interval;
56/// # let (x, y) = (10, 20);
57/// # fn calc() -> i32 { 10 }
58/// # fn other() -> i32 { 20 }
59/// // Literals
60/// interval![1 ..=3];    // [1, 3]
61/// interval![1<.. 3];    // (1, 3)
62///
63/// // Expressions
64/// interval![(x+1), ..= (y*2)];
65/// interval![{ calc() }, <.. { other() }];
66///
67/// // Unbounded
68/// interval![.., i32]; // (-∞, ∞) as i32
69/// interval![..=10];   // (-∞, 10]
70/// interval![1..];     // [1, ∞)
71/// ```
72///
73/// See also [`Interval`].
74#[doc(hidden)]
75#[macro_export]
76#[rustfmt::skip]
77macro_rules! interval· {
78    (
79    /* expressions */
80
81     $l:expr,  ..= $u:expr) => { $crate::Interval::closed($l, $u) }; // lower closed
82    ($l:expr,  ..  $u:expr) => { $crate::Interval::closed_open($l, $u) };
83    ($l:expr,  ..         ) => { $crate::Interval::closed_unbounded($l) };
84    ($l:expr, <..  $u:expr) => { $crate::Interval::open($l, $u) }; // lower open
85    ($l:expr, <..= $u:expr) => { $crate::Interval::open_closed($l, $u) };
86    ($l:expr, <..         ) => { $crate::Interval::open_unbounded($l) };
87    (          ..= $u:expr) => { $crate::Interval::unbounded_closed($u) }; // lower unbounded
88    (          ..  $u:expr) => { $crate::Interval::unbounded_open($u) };
89    (
90    /* blocks */
91
92     $l:block  ..= $u:block) => { $crate::Interval::closed($l, $u) }; // lower closed
93    ($l:block  ..  $u:block) => { $crate::Interval::closed_open($l, $u) };
94    ($l:block  ..          ) => { $crate::Interval::closed_unbounded($l) };
95    ($l:block <..  $u:block) => { $crate::Interval::open($l, $u) }; // lower open
96    ($l:block <..= $u:block) => { $crate::Interval::open_closed($l, $u) };
97    ($l:block <..          ) => { $crate::Interval::open_unbounded($l) };
98    (          ..= $u:block) => { $crate::Interval::unbounded_closed($u) }; // lower unbounded
99    (          ..  $u:block) => { $crate::Interval::unbounded_open($u) };
100    (
101    /* literals */
102
103     $l:literal  ..= $u:literal) => { $crate::Interval::closed($l, $u) }; // lower closed
104    ($l:literal  ..  $u:literal) => { $crate::Interval::closed_open($l, $u) };
105    ($l:literal  ..            ) => { $crate::Interval::closed_unbounded($l) };
106    ($l:literal <..  $u:literal) => { $crate::Interval::open($l, $u) }; // lower open
107    ($l:literal <..= $u:literal) => { $crate::Interval::open_closed($l, $u) };
108    ($l:literal <..            ) => { $crate::Interval::open_unbounded($l) };
109    (            ..= $u:literal) => { $crate::Interval::unbounded_closed($u) }; // lower unbounded
110    (            ..  $u:literal) => { $crate::Interval::unbounded_open($u) };
111    (
112    /* fully unbounded variants (explicit and inferred type) */
113
114     .., $T:ty) => { $crate::Interval::<$T>::unbounded() }; // has to come first
115    (..       ) => { $crate::Interval::unbounded() };
116
117    (
118    /* syntax error msg */
119
120    $($t:tt)*) => {
121        compile_error! { "Invalid interval syntax. Expected forms like: \
122                      l..u, l..=u, l<..u, l<..=u, ..x, ..=x, ..,Type" }
123    };
124}

128#[doc = crate::_tags!(quant)]
129/// A range of values with `lower` and `upper` [`Bound`]s.
130#[doc = crate::_doc_meta!{location("num/quant")}]
131///
132/// The `Interval` type allows modeling ranges of values with optional inclusion
133/// or exclusion at each bound. This is useful for mathematical operations,
134/// range checks, and interval arithmetic.
135///
136/// See also the [`interval!`] macro.
137#[doc(alias = "Range")]
138#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
139pub struct Interval<T> {
140    /// The lower bound (also known as the *start* bound, or the *left* bound).
141    pub lower: Bound<T>,
142    /// The upper bound (also known as the *end*, bound or the *right* bound).
143    pub upper: Bound<T>,
144}
145
146/// Provides a *const* default value for `Interval`, the unbounded interval $(-\infty, \infty)$.
147///
148/// See the [`default`][Self::default] implementation for more information.
149impl<T> ConstInit for Interval<T> {
150    const INIT: Self = Self::unbounded();
151}

153/// # Methodical constructors
154impl<T> Interval<T> {
155    // lower-closed
156
157    /// Creates a closed interval $[l, u]$ `lower..=upper` [`RangeInclusive`].
158    #[must_use]
159    pub const fn closed(lower: T, upper: T) -> Self {
160        Self::new(Bound::Included(lower), Bound::Included(upper))
161    }
162    /// Creates a half-open interval $[l, u)$ `lower..upper` [`Range`].
163    #[must_use]
164    pub const fn closed_open(lower: T, upper: T) -> Self {
165        Self::new(Bound::Included(lower), Bound::Excluded(upper))
166    }
167    /// Creates an interval $[l, ∞)$ `lower..` [`RangeFrom`].
168    #[must_use]
169    pub const fn closed_unbounded(lower: T) -> Self {
170        Self::new(Bound::Included(lower), Bound::Unbounded)
171    }
172
173    // lower-open
174
175    /// Creates an open interval $(l, u)$ *`lower..<upper`*.
176    #[must_use]
177    pub const fn open(lower: T, upper: T) -> Self {
178        Self::new(Bound::Excluded(lower), Bound::Excluded(upper))
179    }
180
181    /// Creates a half-open interval $(l, u]$ *`lower..<=upper`*.
182    #[must_use]
183    pub const fn open_closed(lower: T, upper: T) -> Self {
184        Self::new(Bound::Excluded(lower), Bound::Included(upper))
185    }
186
187    /// Creates an interval $(l, ∞)$ *`lower..<`*.
188    #[must_use]
189    pub const fn open_unbounded(lower: T) -> Self {
190        Self::new(Bound::Excluded(lower), Bound::Unbounded)
191    }
192
193    // lower-unbounded
194
195    /// Creates an unbounded interval $(-∞, ∞)$ `..` [`RangeFull`].
196    #[must_use]
197    pub const fn unbounded() -> Self {
198        Self::new(Bound::Unbounded, Bound::Unbounded)
199    }
200    /// Creates an interval $(-∞, u]$ `..upper` [`RangeTo`].
201    #[must_use]
202    pub const fn unbounded_closed(upper: T) -> Self {
203        Self::new(Bound::Unbounded, Bound::Included(upper))
204    }
205    /// Creates an interval $(-∞, u)$ `..=upper` [`RangeToInclusive`].
206    #[must_use]
207    pub const fn unbounded_open(upper: T) -> Self {
208        Self::new(Bound::Unbounded, Bound::Excluded(upper))
209    }
210}

333impl<T: PartialOrd> Interval<T> {
334    /// Checks if the interval is empty (contains no values).
335    ///
336    /// An interval is empty if:
337    /// - The bounds exclude each other, such as `(x, x)`, `[x, x)`, or `(x, x]`.
338    /// - The `lower` bound is strictly greater than the `upper` bound.
339    ///
340    /// Unbounded intervals are never empty.
341    #[must_use]
342    pub fn is_empty(&self) -> bool {
343        match (&self.lower, &self.upper) {
344            (Bound::Unbounded, _) | (_, Bound::Unbounded) => false,
345            (Bound::Included(l), Bound::Included(u)) => l > u,
346            (Bound::Included(l), Bound::Excluded(u)) => l >= u,
347            (Bound::Excluded(l), Bound::Included(u)) => l >= u,
348            (Bound::Excluded(l), Bound::Excluded(u)) => l >= u,
349        }
350    }
351
352    /// Validates that the interval bounds are ordered correctly.
353    ///
354    /// Returns `true` if the lower bound is less than or equal to the upper bound.
355    /// Unbounded intervals are always considered well ordered.
356    #[must_use]
357    pub fn is_well_ordered(&self) -> bool {
358        match (&self.lower, &self.upper) {
359            (Bound::Unbounded, _) | (_, Bound::Unbounded) => true,
360            (Bound::Included(l), Bound::Included(u)) => l <= u,
361            (Bound::Included(l), Bound::Excluded(u)) => l < u,
362            (Bound::Excluded(l), Bound::Included(u)) => l < u,
363            (Bound::Excluded(l), Bound::Excluded(u)) => l < u,
364        }
365    }
366
367    /// Checks if the interval contains the given value.
368    ///
369    /// ```
370    /// # use devela::Interval;
371    /// let interval = Interval::closed(1, 5);
372    /// assert!(interval.contains(&3));
373    /// assert!(!interval.contains(&6));
374    /// ```
375    #[must_use]
376    pub fn contains(&self, value: &T) -> bool {
377        let lower_check = match &self.lower {
378            Bound::Included(lower) => *lower <= *value,
379            Bound::Excluded(lower) => *lower < *value,
380            Bound::Unbounded => true,
381        };
382        let upper_check = match &self.upper {
383            Bound::Included(upper) => *value <= *upper,
384            Bound::Excluded(upper) => *value < *upper,
385            Bound::Unbounded => true,
386        };
387        lower_check && upper_check
388    }
389
390    /// Returns the size of the interval, if finite.
391    #[must_use]
392    pub fn size(&self) -> Option<T>
393    where
394        T: Clone + core::ops::Sub<Output = T>,
395    {
396        match (&self.lower, &self.upper) {
397            (Bound::Included(l), Bound::Included(u)) => {
398                is![l <= u, Some(u.clone() - l.clone()), None]
399            }
400            (Bound::Included(l), Bound::Excluded(u)) => {
401                is![l < u, Some(u.clone() - l.clone()), None]
402            }
403            (Bound::Excluded(l), Bound::Included(u)) => {
404                is![l < u, Some(u.clone() - l.clone()), None]
405            }
406            (Bound::Excluded(l), Bound::Excluded(u)) => {
407                is![l < u, Some(u.clone() - l.clone()), None]
408            }
409            _ => None, // Unbounded intervals don't have a finite size
410        }
411    }
412}

662    /// Comparison Logic:
663    /// - We compare the lower bounds first.
664    /// - If the lower bounds are equal, we compare the upper bounds.
665    /// - We define Unbounded as less than any bounded value.
666    /// - We define that Included(l) < Excluded(l) at same point l.
667    impl<T: PartialOrd> PartialOrd for Interval<T> {
668        fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
669            match compare_bounds(&self.lower, &other.lower) {
670                Some(Ordering::Equal) => compare_bounds(&self.upper, &other.upper),
671                ord => ord,
672            }
673        }
674    }
675    /// Comparison Logic:
676    /// - We compare the lower bounds first.
677    /// - If the lower bounds are equal, we compare the upper bounds.
678    /// - We define Unbounded as less than any bounded value.
679    /// - We define that Included(l) < Excluded(l) at same point l.
680    impl<T: Ord> Ord for Interval<T> {
681        fn cmp(&self, other: &Self) -> Ordering {
682            match compare_bounds_ord(&self.lower, &other.lower) {
683                Ordering::Equal => compare_bounds_ord(&self.upper, &other.upper),
684                ord => ord,
685            }
686        }
687    }

689    /* helpers */
690
691    fn compare_bounds<T: PartialOrd>(l: &Bound<T>, u: &Bound<T>) -> Option<Ordering> {
692        use Bound::{Excluded, Included, Unbounded};
693        match (l, u) {
694            (Unbounded, Unbounded) => Some(Ordering::Equal),
695            (Unbounded, _) => Some(Ordering::Less),
696            (_, Unbounded) => Some(Ordering::Greater),
697            (Included(a_val), Included(b_val)) => a_val.partial_cmp(b_val),
698            (Excluded(a_val), Excluded(b_val)) => a_val.partial_cmp(b_val),
699            (Included(a_val), Excluded(b_val)) => {
700                match a_val.partial_cmp(b_val) {
701                    Some(Ordering::Equal) => Some(Ordering::Less),
702                    ord => ord,
703                }
704            }
705            (Excluded(a_val), Included(b_val)) => {
706                match a_val.partial_cmp(b_val) {
707                    Some(Ordering::Equal) => Some(Ordering::Greater),
708                    ord => ord,
709                }
710            }
711        }
712    }
713    fn compare_bounds_ord<T: Ord>(l: &Bound<T>, u: &Bound<T>) -> Ordering {
714        use Bound::{Excluded, Included, Unbounded};
715        match (l, u) {
716            (Unbounded, Unbounded) => Ordering::Equal,
717            (Unbounded, _) => Ordering::Less,
718            (_, Unbounded) => Ordering::Greater,
719            (Included(a_val), Included(b_val)) => a_val.cmp(b_val),
720            (Excluded(a_val), Excluded(b_val)) => a_val.cmp(b_val),
721            (Included(a_val), Excluded(b_val)) => {
722                match a_val.cmp(b_val) {
723                    Ordering::Equal => Ordering::Less,
724                    ord => ord,
725                }
726            }
727            (Excluded(a_val), Included(b_val)) => {
728                match a_val.cmp(b_val) {
729                    Ordering::Equal => Ordering::Greater,
730                    ord => ord,
731                }
732            }
733        }
734    }
