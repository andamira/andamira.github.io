6#[cfg(feature = "std")]
7use crate::{IoError, IoErrorKind};
8use crate::{PcmRawError, RiffError};
9
10#[doc = crate::_tags!(audio error)]
11/// WAVE encoding and decoding error.
12#[doc = crate::_doc_meta!{ location("media/audio"), test_size_of(PcmWavError = 4|32) }]
13#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
14pub enum PcmWavError {
15    /// Raw PCM byte/sample conversion failed.
16    Raw(PcmRawError),
17
18    /// RIFF-level parsing failed.
19    Riff(RiffError),
20
21    /// The RIFF form type is not `WAVE`.
22    NotWave,
23
24    /// The required `fmt` chunk was not found.
25    MissingFmt,
26
27    /// The required `data` chunk was not found.
28    MissingData,
29
30    /// The `fmt` chunk is shorter than the base 16-byte PCM format.
31    TruncatedFmt,
32
33    /// The WAVE format code is not supported by this parser.
34    UnsupportedFormat(u16),
35
36    /// The channel count is zero or cannot map to current PCM metadata.
37    UnsupportedChannelCount(u16),
38
39    /// The bits-per-sample field is not supported by current PCM metadata.
40    UnsupportedBitsPerSample(u16),
41
42    /// `block_align` does not match the parsed format.
43    InvalidBlockAlign,
44
45    /// `byte_rate` does not match the parsed format.
46    InvalidByteRate,
47
48    /// The `data` chunk length is not a multiple of `block_align`.
49    InvalidDataLength,
50
51    /// A size computation overflowed.
52    Overflow,
53
54    /// The destination buffer is too small.
55    NotEnoughSpace,
56
57    /// The encoded WAVE file would exceed RIFF's 32-bit size fields.
58    SizeOutOfRange,
59
60    /// The WAVE format metadata cannot be represented by the current writer.
61    UnsupportedEncoding,
62
63    /// The WAVE extensible `fmt` extension is malformed.
64    InvalidExtensibleFormat,
65
66    /// The WAVE extensible subformat GUID is not supported.
67    UnsupportedSubformat,
68
69    /// Reading the file failed.
70    #[cfg(feature = "std")]
71    Io(IoErrorKind),
72}
73crate::impl_trait![fmt::Display+Error for PcmWavError |self, f| match self {
74    Self::Raw(err) => write!(f, "raw PCM operation failed: {err}"),
75    Self::Riff(err) => write!(f, "RIFF parsing failed: {err}"),
76    Self::NotWave => f.write_str("RIFF form type is not `WAVE`"),
77    Self::MissingFmt => f.write_str("missing required `fmt` chunk"),
78    Self::MissingData => f.write_str("missing required `data` chunk"),
79    Self::TruncatedFmt => f.write_str("`fmt` chunk is shorter than the 16-byte PCM base format"),
80    Self::UnsupportedFormat(code) => write!(f, "unsupported WAVE format code: 0x{code:04X}"),
81    Self::UnsupportedChannelCount(chans) => write!(f, "unsupported WAVE channel count: {chans}"),
82    Self::UnsupportedBitsPerSample(bits) => write!(f, "unsupported WAVE bits per sample: {bits}"),
83    Self::InvalidBlockAlign => f.write_str("invalid WAVE block alignment"),
84    Self::InvalidByteRate => f.write_str("invalid WAVE byte rate"),
85    Self::InvalidDataLength => f.write_str("WAVE data length is not a multiple of block alignment"),
86    Self::Overflow => f.write_str("WAVE chunk size computation overflowed"),
87    Self::NotEnoughSpace => f.write_str("not enough space to write WAVE data"),
88    Self::SizeOutOfRange => f.write_str("encoded WAVE data exceeds RIFF size limits"),
89    Self::UnsupportedEncoding => f.write_str("unsupported WAVE encoding"),
90    Self::InvalidExtensibleFormat => f.write_str("invalid WAVE extensible format"),
91    Self::UnsupportedSubformat => f.write_str("unsupported WAVE extensible subformat"),
92    #[cfg(feature = "std")]
93    Self::Io(err) => write!(f, "WAVE file read failed.: {err}"),
94}];
95impl From<PcmRawError> for PcmWavError {
96    fn from(err: PcmRawError) -> Self {
97        Self::Raw(err)
98    }
99}
100impl From<RiffError> for PcmWavError {
101    fn from(err: RiffError) -> Self {
102        Self::Riff(err)
103    }
104}
105#[cfg(feature = "std")]
106impl From<IoError> for PcmWavError {
107    fn from(err: IoError) -> Self {
108        Self::Io(err.kind())
109    }
110}
