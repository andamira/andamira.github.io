15/// RIFF/WAVE operations for PCM-family audio.
16#[doc = crate::_doc_meta!{location("media/audio")}]
17/// `PcmWav` is a namespace for working with WAVE containers whose payload is
18/// raw interleaved PCM-family audio.
19///
20/// It supports:
21/// - parsing borrowed, owned, and file-backed WAVE bytes,
22/// - writing WAVE bytes into caller-provided storage,
23/// - classic integer PCM and IEEE floating-point WAVE,
24/// - WAVE_FORMAT_EXTENSIBLE PCM/float subformats,
25/// - conversion from parsed WAVE data to [`PcmRawBuf`].
26///
27/// The WAVE layer does not reinterpret sample bytes as typed slices. Use the
28/// typed helpers on [`PcmRaw`] or [`PcmRawBuf`] to decode or encode samples.
29///
30/// [`PcmRaw`]: crate::PcmRaw
31/// [`PcmRawBuf`]: crate::PcmRawBuf
32///
33/// # Example
34/// ```
35/// # use devela::{AudioChannels, PcmRaw, PcmSample, PcmSpec, PcmWav, PcmWavError, PcmWavFmt};
36/// # fn main() -> Result<(), PcmWavError> {
37/// let spec = PcmSpec::new(PcmSample::I16, AudioChannels::Stereo, 44_100);
38/// let fmt = PcmWavFmt::from_spec(spec)?;
39///
40/// // Two stereo frames: silence, then max-left/min-right.
41/// let samples = [0i16, 0, i16::MAX, i16::MIN];
42/// let mut pcm = [0u8; 8];
43/// PcmRaw::encode_i16_le_into(&mut pcm, spec, &samples)?;
44///
45/// let len = PcmWav::write_len(fmt, pcm.len())?;
46/// let mut wav_bytes = [0u8; 64];
47/// let written = PcmWav::write_into(&mut wav_bytes, fmt, &pcm)?;
48/// assert_eq!(written, len);
49///
50/// let wav = PcmWav::parse(&wav_bytes[..written])?;
51/// assert_eq!(wav.spec()?, spec);
52/// assert_eq!(wav.frames(), 2);
53///
54/// let raw = wav.raw()?;
55/// let mut decoded = [0i16; 4];
56/// raw.decode_i16_le_into(&mut decoded)?;
57/// assert_eq!(decoded, samples);
58/// # Ok(()) }
59/// ```
60#[derive(Debug)]
61pub struct PcmWav;
62
63/// # Parsing
64impl PcmWav {
65    /// Parses a borrowed WAVE byte region.
66    ///
67    /// This validates the RIFF container, finds the first `fmt ` chunk, finds
68    /// the first `data` chunk, checks the basic PCM shape, and returns a
69    /// byte-backed view into the original input.
70    pub const fn parse(bytes: &[u8]) -> Result<PcmWavBuf<&[u8]>, PcmWavError> {
71        let parts = unwrap![ok? Self::parse_parts(bytes)];
72        Ok(PcmWavBuf::_new(bytes, parts.fmt, parts.data_offset, parts.data_len))
73    }
74    /// Parses owned WAVE bytes.
75    #[cfg(feature = "alloc")]
76    pub fn from_vec(bytes: Vec<u8>) -> Result<PcmWavBuf<Vec<u8>>, PcmWavError> {
77        let parts = Self::parse_parts(bytes.as_slice())?;
78        Ok(PcmWavBuf::_new(bytes, parts.fmt, parts.data_offset, parts.data_len))
79    }
80    /// Reads and parses a WAVE file.
81    #[cfg(feature = "std")]
82    pub fn from_file<P: AsRef<Path>>(path: P) -> Result<PcmWavBuf<Vec<u8>>, PcmWavError> {
83        Self::from_vec(Fs::read(path)?)
84    }

86    /// Parses the base WAVE `fmt` payload.
87    pub const fn parse_fmt(bytes: &[u8]) -> Result<PcmWavFmt, PcmWavError> {
88        is! { bytes.len() < PcmWavFmt::BASE_LEN, return Err(PcmWavError::TruncatedFmt) }
89        let format_tag = u16::from_le_bytes(read_at![bytes, 0, @2]);
90        let extra_len = if bytes.len() >= PcmWavFmt::WAVEFORMATEX_EMPTY_LEN {
91            u16::from_le_bytes(read_at![bytes, 16, @2])
92        } else {
93            0
94        };
95        let mut fmt = PcmWavFmt {
96            format_tag,
97            subformat_tag: format_tag,
98            channels: u16::from_le_bytes(read_at![bytes, 2, @2]),
99            sample_rate: u32::from_le_bytes(read_at![bytes, 4, @4]),
100            byte_rate: u32::from_le_bytes(read_at![bytes, 8, @4]),
101            block_align: u16::from_le_bytes(read_at![bytes, 12, @2]),
102            bits_per_sample: u16::from_le_bytes(read_at![bytes, 14, @2]),
103            valid_bits_per_sample: u16::from_le_bytes(read_at![bytes, 14, @2]),
104            channel_mask: 0,
105            extra_len,
106        };
107        if format_tag == Self::FORMAT_EXTENSIBLE {
108            if bytes.len() < Self::EXTENSIBLE_LEN || extra_len < Self::EXTENSIBLE_EXTRA_LEN {
109                return Err(PcmWavError::InvalidExtensibleFormat);
110            }
111            fmt.valid_bits_per_sample = u16::from_le_bytes(read_at![bytes, 18, @2]);
112            fmt.channel_mask = u32::from_le_bytes(read_at![bytes, 20, @4]);
113            let guid = read_at![bytes, 24, @16];
114            fmt.subformat_tag = match Self::subformat_tag(guid) {
115                Ok(tag) => tag,
116                Err(err) => return Err(err),
117            };
118        }
119        fmt.validate()
120    }

122    /// Returns the packed byte width of a supported WAVE sample encoding.
123    ///
124    /// `format_tag` must be an effective sample format tag, such as
125    /// [`FORMAT_PCM`](Self::FORMAT_PCM) or [`FORMAT_IEEE_FLOAT`](Self::FORMAT_IEEE_FLOAT).
126    /// For WAVE_FORMAT_EXTENSIBLE, pass the resolved subformat tag instead.
127    pub const fn bytes_per_sample(format_tag: u16, bits: u16) -> Result<u16, PcmWavError> {
128        match (format_tag, bits) {
129            (Self::FORMAT_PCM, 8) => Ok(1),
130            (Self::FORMAT_PCM, 16) => Ok(2),
131            (Self::FORMAT_PCM, 24) => Ok(3),
132            (Self::FORMAT_PCM, 32) => Ok(4),
133            (Self::FORMAT_IEEE_FLOAT, 32) => Ok(4),
134            (Self::FORMAT_IEEE_FLOAT, 64) => Ok(8),
135            (_, other) => Err(PcmWavError::UnsupportedBitsPerSample(other)),
136        }
137    }
138    const fn parse_parts(bytes: &[u8]) -> Result<PcmWavParts, PcmWavError> {
139        let root = match Riff::root(bytes) {
140            Ok(root) => root,
141            Err(err) => return Err(PcmWavError::Riff(err)),
142        };
143        match root.container_type() {
144            Some(tag) if tag.eq(Riff::WAVE) => {}
145            _ => return Err(PcmWavError::NotWave),
146        }
147        let mut chunks = match root.subchunks() {
148            Ok(chunks) => chunks,
149            Err(err) => return Err(PcmWavError::Riff(err)),
150        };
151        let mut fmt: Option<PcmWavFmt> = None;
152        let mut data_len = 0usize;
153        let mut data_offset = 0usize;
154        while let Some(next) = chunks.next_chunk() {
155            let chunk = match next {
156                Ok(chunk) => chunk,
157                Err(err) => return Err(PcmWavError::Riff(err)),
158            };
159            if chunk.id().eq(Riff::FMT) {
160                if fmt.is_none() {
161                    fmt = match Self::parse_fmt(chunk.data()) {
162                        Ok(fmt) => Some(fmt),
163                        Err(err) => return Err(err),
164                    };
165                }
166            } else if chunk.id().eq(Riff::DATA) {
167                if data_len == 0 && data_offset == 0 {
168                    data_len = chunk.data().len();
169                    // Full WAVE byte layout:
170                    // - RIFF header:       8 bytes
171                    // - WAVE form tag:     4 bytes
172                    // - subchunk header:   8 bytes
173                    //
174                    // `chunk.offset()` is relative to the subchunk region, which
175                    // starts after the RIFF header and WAVE form tag.
176                    let Some(offset) = 12usize.checked_add(chunk.offset()) else {
177                        return Err(PcmWavError::Overflow);
178                    };
179                    let Some(offset) = offset.checked_add(8) else {
180                        return Err(PcmWavError::Overflow);
181                    };
182                    data_offset = offset;
183                }
184            }
185            is! { fmt.is_some() && (data_len != 0 || data_offset != 0), break }
186        }
187        let Some(fmt) = fmt else {
188            return Err(PcmWavError::MissingFmt);
189        };
190        is! { data_len == 0 && data_offset == 0, return Err(PcmWavError::MissingData) }
191        match Self::validate_data_shape(fmt, data_len) {
192            Ok(()) => Ok(PcmWavParts { fmt, data_offset, data_len }),
193            Err(err) => Err(err),
194        }
195    }
196    /// Checks that the `data` chunk length is frame-aligned.
197    const fn validate_data_shape(fmt: PcmWavFmt, data_len: usize) -> Result<(), PcmWavError> {
198        is! { fmt.block_align == 0, return Err(PcmWavError::InvalidBlockAlign) }
199        if !data_len.is_multiple_of(fmt.block_align as usize) {
200            return Err(PcmWavError::InvalidDataLength);
201        }
202        Ok(())
203    }
204}
205/// # Writing
206impl PcmWav {
207    /// Returns the number of bytes needed to write a minimal WAVE container.
208    pub const fn write_len(fmt: PcmWavFmt, data_len: usize) -> Result<usize, PcmWavError> {
209        let fmt = match fmt.validate() {
210            Ok(fmt) => fmt,
211            Err(err) => return Err(err),
212        };
213        match fmt.frames_for_data_len(data_len) {
214            Ok(_) => {}
215            Err(err) => return Err(err),
216        }
217        is! { !Riff::fits_u32(data_len), return Err(PcmWavError::SizeOutOfRange) }
218        let Some(fmt_chunk_len) = Riff::chunk_len(fmt.encoded_len()) else {
219            return Err(PcmWavError::Overflow);
220        };
221        let fact_chunk_len = if fmt.needs_fact() {
222            match Riff::chunk_len(4) {
223                Some(len) => len,
224                None => return Err(PcmWavError::Overflow),
225            }
226        } else {
227            0
228        };
229        let Some(data_chunk_len) = Riff::chunk_len(data_len) else {
230            return Err(PcmWavError::Overflow);
231        };
232        let Some(subchunks_len) = fmt_chunk_len.checked_add(fact_chunk_len) else {
233            return Err(PcmWavError::Overflow);
234        };
235        let Some(subchunks_len) = subchunks_len.checked_add(data_chunk_len) else {
236            return Err(PcmWavError::Overflow);
237        };
238        // The RIFF size field stores `4 + subchunks_len`.
239        let Some(riff_data_len) = Riff::FORM_TYPE_LEN.checked_add(subchunks_len) else {
240            return Err(PcmWavError::Overflow);
241        };
242        is! { !Riff::fits_u32(riff_data_len), return Err(PcmWavError::SizeOutOfRange) }
243        match Riff::form_len(subchunks_len) {
244            Some(len) => Ok(len),
245            None => Err(PcmWavError::Overflow),
246        }
247    }
248    /// Writes a minimal WAVE container into `dst`.
249    ///
250    /// `data` must already contain raw interleaved PCM-family sample bytes matching `fmt`.
251    /// This method writes RIFF/WAVE headers and copies the payload;
252    /// it does not convert typed samples.
253    pub const fn write_into(
254        dst: &mut [u8],
255        fmt: PcmWavFmt,
256        data: &[u8],
257    ) -> Result<usize, PcmWavError> {
258        use PcmWavError as E;
259        let fmt = unwrap![ok? fmt.validate()];
260        let frames = unwrap![ok? fmt.frames_for_data_len(data.len())];
261        let len = unwrap![ok? Self::write_len(fmt, data.len())];
262        is! { dst.len() < len, return Err(E::NotEnoughSpace) }
263        is! { frames > u32::MAX as usize, return Err(E::SizeOutOfRange) }
264        let fmt_len = fmt.encoded_len();
265        let fmt_chunk_len = unwrap![some_ok_or? Riff::chunk_len(fmt_len), E::Overflow];
266        let fact_chunk_len = if fmt.needs_fact() {
267            unwrap![some_ok_or? Riff::chunk_len(4), E::Overflow]
268        } else {
269            0
270        };
271        let data_chunk_len = unwrap![some_ok_or? Riff::chunk_len(data.len()), E::Overflow];
272        let subchunks_len = unwrap![some_ok_or?
273            fmt_chunk_len.checked_add(fact_chunk_len), E::Overflow];
274        let subchunks_len = unwrap![some_ok_or?
275            subchunks_len.checked_add(data_chunk_len), E::Overflow];
276        let mut offset = unwrap![ok_err_map?
277            Riff::write_form_header(dst, Riff::WAVE, subchunks_len), |e|e.to_wav()];
278        offset += unwrap! { ok_err_map? Riff::write_chunk_header(slice![mut dst, offset,..],
279        Riff::FMT, fmt_len), |e| e.to_wav() };
280        offset += unwrap![ok? Self::write_fmt_payload(slice![mut dst, offset,..], fmt)];
281        if fmt.needs_fact() {
282            offset += unwrap! {ok_err_map? Riff::write_chunk_header(slice![mut dst, offset,..],
283            Riff::FACT, 4), |e| e.to_wav() };
284            offset += unwrap![ok? write_u32_le(slice![mut dst, offset,..], frames as u32)];
285        }
286        offset += unwrap! { ok_err_map? Riff::write_chunk_header(slice![mut dst, offset,..],
287        Riff::DATA, data.len()), |e| e.to_wav() };
288        offset += unwrap![ok? write_bytes(slice![mut dst, offset, ..], data)];
289        is! { Riff::pad_len(data.len()) != 0, { dst[offset] = 0; offset += 1; }}
290        debug_assert!(offset == len);
291        Ok(offset)
292    }
293    /// Encodes a minimal PCM-family WAVE file into an allocated byte vector.
294    #[cfg(feature = "alloc")]
295    pub fn to_vec(fmt: PcmWavFmt, data: &[u8]) -> Result<Vec<u8>, PcmWavError> {
296        let len = Self::write_len(fmt, data.len())?;
297        let mut out = vec_![0u8; len];
298        let written = Self::write_into(&mut out, fmt, data)?;
299        debug_assert_eq!(written, len);
300        Ok(out)
301    }
302    /// Encodes and writes a minimal PCM-family WAVE file.
303    #[cfg(feature = "std")]
304    pub fn to_file<P: AsRef<Path>>(
305        path: P,
306        fmt: PcmWavFmt,
307        data: &[u8],
308    ) -> Result<(), PcmWavError> {
309        let bytes = Self::to_vec(fmt, data)?;
310        Fs::write(path, bytes)?;
311        Ok(())
312    }
313    /// Encodes a minimal PCM-family WAVE file from a [`PcmSpec`].
314    #[cfg(feature = "alloc")]
315    pub fn spec_to_vec(spec: PcmSpec, data: &[u8]) -> Result<Vec<u8>, PcmWavError> {
316        let fmt = PcmWavFmt::from_spec(spec)?;
317        Self::to_vec(fmt, data)
318    }
319    /// Writes the WAVE `fmt ` payload.
320    const fn write_fmt_payload(dst: &mut [u8], fmt: PcmWavFmt) -> Result<usize, PcmWavError> {
321        let len = fmt.encoded_len();
322        is! { dst.len() < len, return Err(PcmWavError::NotEnoughSpace) }
323        let mut off = 0;
324        off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.format_tag)];
325        off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.channels)];
326        off += unwrap![ok? write_u32_le(slice![mut dst, off,..], fmt.sample_rate)];
327        off += unwrap![ok? write_u32_le(slice![mut dst, off,..], fmt.byte_rate)];
328        off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.block_align)];
329        off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.bits_per_sample)];
330        if fmt.format_tag == PcmWav::FORMAT_EXTENSIBLE {
331            is! {
332                fmt.extra_len != PcmWav::EXTENSIBLE_EXTRA_LEN,
333                return Err(PcmWavError::InvalidExtensibleFormat)
334            }
335            off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.extra_len)];
336            off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.valid_bits_per_sample)];
337            off += unwrap![ok? write_u32_le(slice![mut dst, off,..], fmt.channel_mask)];
338            let guid = unwrap![ok? PcmWav::subformat_guid(fmt.subformat_tag)];
339            write_at![dst, +=off, @16 guid];
340            return Ok(off);
341        }
342        if len >= PcmWavFmt::WAVEFORMATEX_EMPTY_LEN {
343            // WAVEFORMATEX extension header. This writer currently supports
344            // only an empty extension for non-extensible formats.
345            off += unwrap![ok? write_u16_le(slice![mut dst, off,..], fmt.extra_len)];
346            is! { fmt.extra_len != 0, return Err(PcmWavError::UnsupportedEncoding) }
347        }
348        Ok(off)
349    }
350}
351/// # Constants
352impl PcmWav {
353    /// Integer PCM.
354    pub const FORMAT_PCM: u16 = 0x0001;
355    /// IEEE floating-point PCM.
356    pub const FORMAT_IEEE_FLOAT: u16 = 0x0003;
357
358    /// Extensible WAVE format code.
359    pub const FORMAT_EXTENSIBLE: u16 = 0xFFFE;
360
361    /// WAVEFORMATEXTENSIBLE extension byte length.
362    pub const EXTENSIBLE_EXTRA_LEN: u16 = 22;
363
364    /// WAVEFORMATEXTENSIBLE `fmt ` payload length.
365    pub const EXTENSIBLE_LEN: usize = 40;
366
367    /// WAVEFORMATEXTENSIBLE PCM subformat GUID bytes.
368    ///
369    /// GUID: `00000001-0000-0010-8000-00AA00389B71`
370    pub const SUBFORMAT_PCM: [u8; 16] = [
371        0x01, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00, 0x80, 0x00, 0x00, 0xAA, 0x00, 0x38, 0x9B,
372        0x71,
373    ];
374
375    /// WAVEFORMATEXTENSIBLE IEEE float subformat GUID bytes.
376    ///
377    /// GUID: `00000003-0000-0010-8000-00AA00389B71`
378    pub const SUBFORMAT_IEEE_FLOAT: [u8; 16] = [
379        0x03, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10, 0x00, 0x80, 0x00, 0x00, 0xAA, 0x00, 0x38, 0x9B,
380        0x71,
381    ];
382
383    /// Returns the WAVE format tag represented by a supported extensible subformat GUID.
384    pub const fn subformat_tag(guid: [u8; 16]) -> Result<u16, PcmWavError> {
385        if Slice::<u8>::eq(&guid, &Self::SUBFORMAT_PCM) {
386            Ok(Self::FORMAT_PCM)
387        } else if Slice::<u8>::eq(&guid, &Self::SUBFORMAT_IEEE_FLOAT) {
388            Ok(Self::FORMAT_IEEE_FLOAT)
389        } else {
390            Err(PcmWavError::UnsupportedSubformat)
391        }
392    }
393    /// Returns the WAVEFORMATEXTENSIBLE GUID for a supported sample format tag.
394    pub const fn subformat_guid(format_tag: u16) -> Result<[u8; 16], PcmWavError> {
395        match format_tag {
396            Self::FORMAT_PCM => Ok(Self::SUBFORMAT_PCM),
397            Self::FORMAT_IEEE_FLOAT => Ok(Self::SUBFORMAT_IEEE_FLOAT),
398            _ => Err(PcmWavError::UnsupportedSubformat),
399        }
400    }

403#[derive(Clone, Copy)]
404struct PcmWavParts {
405    fmt: PcmWavFmt,
406    data_offset: usize,
407    data_len: usize,
408}
409const fn write_bytes(dst: &mut [u8], src: &[u8]) -> Result<usize, PcmWavError> {
410    is! { dst.len() < src.len(), return Err(PcmWavError::NotEnoughSpace) }
411    whilst! { i in 0..src.len(); { dst[i] = src[i]; }}
412    Ok(src.len())
413}
414const fn write_u16_le(dst: &mut [u8], value: u16) -> Result<usize, PcmWavError> {
415    is! { dst.len() < 2, return Err(PcmWavError::NotEnoughSpace) }
416    write_at![dst, 0, @2 value.to_le_bytes()];
417    Ok(2)
418}
419const fn write_u32_le(dst: &mut [u8], value: u32) -> Result<usize, PcmWavError> {
420    is! { dst.len() < 4, return Err(PcmWavError::NotEnoughSpace) }
421    write_at![dst, 0, @4 value.to_le_bytes()];
422    Ok(4)
