6use crate::{AudioChannels, PcmSample, PcmSpec, PcmWav, PcmWavError, is};

8#[doc = crate::_tags!(audio parser)]
9/// Parsed WAVE `fmt` chunk.
10#[doc = crate::_doc_meta!{location("media/audio"), test_size_of(PcmWavFmt = 28|224)}]
11#[must_use]
12#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
13pub struct PcmWavFmt {
14    /// WAVE format code.
15    pub format_tag: u16,
16    /// Effective sample format code.
17    ///
18    /// For classic PCM/float WAVE this equals [`format_tag`](Self::format_tag).
19    /// For WAVE_FORMAT_EXTENSIBLE this is resolved from the subformat GUID.
20    pub subformat_tag: u16,
21    /// Valid precision bits per sample.
22    ///
23    /// For classic WAVE this equals [`bits_per_sample`](Self::bits_per_sample).
24    /// For WAVE_FORMAT_EXTENSIBLE this may be smaller than the container size.
25    pub valid_bits_per_sample: u16,
26    /// WAVE_FORMAT_EXTENSIBLE channel mask.
27    ///
28    /// `0` means absent or unspecified.
29    pub channel_mask: u32,
30    /// Number of interleaved channels.
31    pub channels: u16,
32    /// Samples per second.
33    pub sample_rate: u32,
34    /// Average bytes per second.
35    pub byte_rate: u32,
36    /// Bytes per interleaved frame.
37    pub block_align: u16,
38    /// Meaningful bits per sample.
39    pub bits_per_sample: u16,
40    /// Number of extension bytes after the base 16-byte format.
41    ///
42    /// This is `0` for the classic 16-byte PCM form. When serialized as a
43    /// WAVEFORMATEX chunk, the `cbSize` field itself occupies two bytes and
44    /// contains this value.
45    pub extra_len: u16,
46}
47///
48impl PcmWavFmt {
49    /// Creates validated WAVE format metadata from a PCM stream specification.
50    pub const fn from_spec(spec: PcmSpec) -> Result<Self, PcmWavError> {
51        let format_tag = if spec.sample.is_float() {
52            PcmWav::FORMAT_IEEE_FLOAT
53        } else if spec.sample.is_int() {
54            PcmWav::FORMAT_PCM
55        } else {
56            return Err(PcmWavError::UnsupportedBitsPerSample(spec.sample.bits() as u16));
57        };
58        Self::new(
59            format_tag,
60            spec.channel_count() as u16,
61            spec.sample_rate,
62            spec.sample.bits() as u16,
63            0, // TEMP
64               // if format_tag == PcmWav::FORMAT_IEEE_FLOAT { 0 } else { 0 }, // IMPROVE
65        )
66    }
67    /// Creates validated WAVE format metadata.
68    pub const fn new(
69        format_tag: u16,
70        channels: u16,
71        sample_rate: u32,
72        bits_per_sample: u16,
73        extra_len: u16,
74    ) -> Result<Self, PcmWavError> {
75        match format_tag {
76            PcmWav::FORMAT_PCM | PcmWav::FORMAT_IEEE_FLOAT => {}
77            PcmWav::FORMAT_EXTENSIBLE => return Err(PcmWavError::InvalidExtensibleFormat),
78            other => return Err(PcmWavError::UnsupportedFormat(other)),
79        }
80        is! { channels == 0, return Err(PcmWavError::UnsupportedChannelCount(channels)) }
81        let bytes_per_sample = match PcmWav::bytes_per_sample(format_tag, bits_per_sample) {
82            Ok(bytes) => bytes,
83            Err(err) => return Err(err),
84        };
85        let Some(block_align) = channels.checked_mul(bytes_per_sample) else {
86            return Err(PcmWavError::InvalidBlockAlign);
87        };
88        let Some(byte_rate) = sample_rate.checked_mul(block_align as u32) else {
89            return Err(PcmWavError::InvalidByteRate);
90        };
91        Ok(Self {
92            format_tag,
93            subformat_tag: format_tag,
94            channels,
95            sample_rate,
96            byte_rate,
97            block_align,
98            bits_per_sample,
99            valid_bits_per_sample: bits_per_sample,
100            channel_mask: 0,
101            extra_len,
102        })
103    }
104    /// Creates validated extended WAVE format metadata.
105    pub const fn new_extensible(
106        subformat_tag: u16,
107        channels: u16,
108        sample_rate: u32,
109        bits_per_sample: u16,
110        valid_bits_per_sample: u16,
111        channel_mask: u32,
112    ) -> Result<Self, PcmWavError> {
113        match subformat_tag {
114            PcmWav::FORMAT_PCM | PcmWav::FORMAT_IEEE_FLOAT => {}
115            other => return Err(PcmWavError::UnsupportedFormat(other)),
116        }
117        if valid_bits_per_sample == 0 || valid_bits_per_sample > bits_per_sample {
118            return Err(PcmWavError::InvalidExtensibleFormat);
119        }
120        let mut fmt = match Self::new(
121            subformat_tag,
122            channels,
123            sample_rate,
124            bits_per_sample,
125            PcmWav::EXTENSIBLE_EXTRA_LEN,
126        ) {
127            Ok(fmt) => fmt,
128            Err(err) => return Err(err),
129        };
130        fmt.format_tag = PcmWav::FORMAT_EXTENSIBLE;
131        fmt.subformat_tag = subformat_tag;
132        fmt.valid_bits_per_sample = valid_bits_per_sample;
133        fmt.channel_mask = channel_mask;
134        fmt.extra_len = PcmWav::EXTENSIBLE_EXTRA_LEN;
135
136        Ok(fmt)
137    }
138    /// Returns the encoded `fmt ` chunk payload length.
139    ///
140    /// PCM is written as the classic 16-byte `fmt ` form.
141    /// IEEE float is written as WAVEFORMATEX with `cbSize = 0`.
142    pub const fn encoded_len(self) -> usize {
143        if self.format_tag == PcmWav::FORMAT_EXTENSIBLE {
144            PcmWav::EXTENSIBLE_LEN
145        } else if self.format_tag == PcmWav::FORMAT_PCM && self.extra_len == 0 {
146            Self::BASE_LEN
147        } else {
148            Self::WAVEFORMATEX_EMPTY_LEN + self.extra_len as usize
149        }
150    }
151    /// Returns whether this format should be accompanied by a `fact` chunk.
152    #[must_use]
153    pub const fn needs_fact(self) -> bool {
154        self.subformat_tag != PcmWav::FORMAT_PCM
155    }
156    /// Returns the number of frames in `data_len`, if frame-aligned.
157    pub const fn frames_for_data_len(self, data_len: usize) -> Result<usize, PcmWavError> {
158        let spec = match self.spec() {
159            Ok(spec) => spec,
160            Err(err) => return Err(err),
161        };
162        match spec.frames_for_data_len(data_len) {
163            Some(frames) => Ok(frames),
164            None => Err(PcmWavError::InvalidDataLength),
165        }
166    }
167    /// Validates this format against its derived fields.
168    pub const fn validate(self) -> Result<Self, PcmWavError> {
169        is![self.channels == 0, return Err(PcmWavError::UnsupportedChannelCount(self.channels))];
170        match self.format_tag {
171            PcmWav::FORMAT_PCM | PcmWav::FORMAT_IEEE_FLOAT => {
172                if self.subformat_tag != self.format_tag {
173                    return Err(PcmWavError::InvalidExtensibleFormat);
174                }
175                if self.valid_bits_per_sample != self.bits_per_sample {
176                    return Err(PcmWavError::InvalidExtensibleFormat);
177                }
178            }
179            PcmWav::FORMAT_EXTENSIBLE => {
180                if self.extra_len < PcmWav::EXTENSIBLE_EXTRA_LEN {
181                    return Err(PcmWavError::InvalidExtensibleFormat);
182                }
183                if self.valid_bits_per_sample == 0
184                    || self.valid_bits_per_sample > self.bits_per_sample
185                {
186                    return Err(PcmWavError::InvalidExtensibleFormat);
187                }
188            }
189            other => return Err(PcmWavError::UnsupportedFormat(other)),
190        }
191        let bytes_per_sample =
192            match PcmWav::bytes_per_sample(self.subformat_tag, self.bits_per_sample) {
193                Ok(bytes) => bytes,
194                Err(err) => return Err(err),
195            };
196        let Some(expected_align) = self.channels.checked_mul(bytes_per_sample) else {
197            return Err(PcmWavError::InvalidBlockAlign);
198        };
199        if self.block_align != expected_align {
200            return Err(PcmWavError::InvalidBlockAlign);
201        }
202        let Some(expected_rate) = self.sample_rate.checked_mul(self.block_align as u32) else {
203            return Err(PcmWavError::InvalidByteRate);
204        };
205        is![self.byte_rate != expected_rate, return Err(PcmWavError::InvalidByteRate)];
206        Ok(self)
207    }
208    /// Maps this WAVE format metadata to current [`PcmSpec`] metadata when unambiguous.
209    pub const fn spec(self) -> Result<PcmSpec, PcmWavError> {
210        let sample = match (self.subformat_tag, self.bits_per_sample) {
211            (PcmWav::FORMAT_PCM, 8) => PcmSample::U8,
212            (PcmWav::FORMAT_PCM, 16) => PcmSample::I16,
213            (PcmWav::FORMAT_PCM, 24) => PcmSample::I24,
214            (PcmWav::FORMAT_PCM, 32) => PcmSample::I32,
215            (PcmWav::FORMAT_IEEE_FLOAT, 32) => PcmSample::F32,
216            (PcmWav::FORMAT_IEEE_FLOAT, 64) => PcmSample::F64,
217            (_, bits) => return Err(PcmWavError::UnsupportedBitsPerSample(bits)),
218        };
219        let channels = match self.channels {
220            1 => AudioChannels::Mono,
221            2 => AudioChannels::Stereo,
222            other => return Err(PcmWavError::UnsupportedChannelCount(other)),
223        };
224        Ok(PcmSpec::new(sample, channels, self.sample_rate))
225    }
226}
227/// # Constants
228impl PcmWavFmt {
229    /// The base PCM `fmt ` payload length.
230    pub const BASE_LEN: usize = 16;
231    /// The WAVEFORMATEX `fmt ` payload length with `cbSize = 0`.
232    pub const WAVEFORMATEX_EMPTY_LEN: usize = 18;
233}
