10#[doc = crate::_tags!(audio data)]
11/// Parsed WAVE byte buffer over borrowed or owned storage.
12#[doc = crate::_doc_meta!{
13    location("media/audio"),
14    #[cfg(target_pointer_width = "64")]
15    test_size_of(PcmWavBuf_Slice: PcmWavBuf<&[u8]> = 64|512),
16    #[cfg(target_pointer_width = "64")]
17    test_size_of(PcmWavBuf_Vec: PcmWavBuf<Vec<u8>> = 72|576),
18}]
19///
20/// This stores the full WAVE byte region together with the parsed `fmt ` metadata
21/// and the byte span of the `data` payload.
22///
23/// The storage type decides ownership:
24/// - `PcmWavBuf<&[u8]>` borrows existing WAVE bytes.
25/// - `PcmWavBuf<Vec<u8>>` owns allocated WAVE bytes.
26///
27/// The audio payload is still raw bytes. Typed PCM decoding should be explicit,
28/// because WAVE integer samples are little-endian and may not be naturally aligned.
29#[must_use]
30#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
31pub struct PcmWavBuf<B> {
32    /// Full WAVE byte region.
33    bytes: B,
34    /// Parsed `fmt ` chunk.
35    fmt: PcmWavFmt,
36    /// Byte offset of the `data` payload within `bytes`.
37    data_offset: usize,
38    /// Byte length of the `data` payload.
39    data_len: usize,
40}
41#[rustfmt::skip]
42impl<B> PcmWavBuf<B> {
43    /// Creates a parsed WAVE buffer from already-validated parts.
44    ///
45    /// This is crate-private because the offset and length must remain valid
46    /// for `bytes`, and the `fmt ` metadata must have already been checked.
47    pub(crate) const fn _new(bytes: B, fmt: PcmWavFmt, data_offset: usize, data_len: usize)
48        -> Self { Self { bytes, fmt, data_offset, data_len } }
49
50    /// Returns the parsed `fmt ` metadata.
51    pub const fn fmt(&self) -> PcmWavFmt { self.fmt }
52    /// Returns the byte offset of the `data` payload in the full WAVE byte region.
53    #[must_use]
54    pub const fn data_offset(&self) -> usize { self.data_offset }
55    /// Returns the byte length of the `data` payload.
56    #[must_use]
57    pub const fn data_len(&self) -> usize { self.data_len }
58    /// Returns the byte offset and length of the `data` payload.
59    #[must_use]
60    pub const fn data_span(&self) -> (usize, usize) { (self.data_offset, self.data_len) }
61    /// Returns the number of interleaved PCM frames.
62    #[must_use]
63    pub const fn frames(&self) -> usize { self.data_len / self.fmt.block_align as usize }
64    /// Maps this WAVE metadata to current [`PcmSpec`] metadata when unambiguous.
65    pub const fn spec(&self) -> Result<PcmSpec, PcmWavError> { self.fmt.spec() }
66}
67#[rustfmt::skip]
68impl<B: AsRef<[u8]>> PcmWavBuf<B> {
69    /// Returns the full WAVE byte region.
70    #[must_use]
71    pub fn bytes(&self) -> &[u8] { self.bytes.as_ref() }
72
73    /// Returns the borrowed `data` chunk payload.
74    ///
75    /// This slices the original WAVE byte region using the span found during
76    /// parsing. The fields are private so safe constructors preserve the span.
77    #[must_use]
78    pub fn data_bytes(&self) -> &[u8] {
79        let start = self.data_offset;
80        let end = start + self.data_len;
81        &self.bytes.as_ref()[start..end]
82    }
83
84    /// Returns whether the `data` chunk payload is empty.
85    #[must_use]
86    pub fn is_empty(&self) -> bool { self.data_len == 0 }
87
88    /// Returns an iterator over the WAVE subchunks.
89    pub fn chunks(&self) -> Result<RiffChunkIter<'_>, PcmWavError> {
90        let root = Riff::root(self.bytes()).map_err(PcmWavError::Riff)?;
91        root.subchunks().map_err(PcmWavError::Riff)
92    }
93
94    /// Returns the WAVE `data` chunk as headerless raw PCM.
95    pub fn raw(&self) -> Result<PcmRawBuf<&[u8]>, PcmWavError> {
96        let spec = self.spec()?;
97        Ok(PcmRawBuf::_new(self.data_bytes(), spec))
98    }
99}
100#[rustfmt::skip]
101impl<'a> PcmWavBuf<&'a [u8]> {
102    /// Returns the full borrowed WAVE byte region.
103    ///
104    /// This exists as a const-friendly alternative to [`bytes`](Self::bytes).
105    #[must_use]
106    pub const fn bytes_const(&self) -> &'a [u8] { self.bytes }
107}
108#[rustfmt::skip]
109#[cfg(feature = "alloc")]
110impl PcmWavBuf<Vec<u8>> {
111    /// Returns this owned WAVE buffer as a borrowed WAVE buffer.
112    pub fn as_borrowed(&self) -> PcmWavBuf<&[u8]> {
113        PcmWavBuf::_new(self.bytes.as_slice(), self.fmt, self.data_offset, self.data_len)
114    }
115    /// Returns the owned full WAVE byte region.
116    #[must_use]
117    pub fn into_bytes(self) -> Vec<u8> { self.bytes }
118}
