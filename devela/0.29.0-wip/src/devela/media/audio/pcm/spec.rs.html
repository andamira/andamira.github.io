6use crate::{_impl_init, AudioChannels, PcmSample, impl_trait};
7
8#[doc = crate::_tags!(audio)]
9/// Essential metadata describing a PCM audio stream.
10#[doc = crate::_doc_meta!{
11    location("media/audio"),
12    test_size_of(PcmSpec = 8|64),
13}]
14#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
15pub struct PcmSpec {
16    /// bit-depth + numeric type
17    pub sample: PcmSample,
18    /// layout (Mono, Stereo, 5.1…)
19    pub channels: AudioChannels,
20    /// Hz
21    pub sample_rate: u32,
22}
23_impl_init![Self::new(PcmSample::INIT, AudioChannels::INIT, 0) => PcmSpec];
24impl_trait![fmt::Display for PcmSpec |self, f| {
25    write!(f, "{}/{}@{}Hz", self.sample, self.channels, self.sample_rate)
26}];
27
28impl PcmSpec {
29    /// Creates a PCM stream specification.
30    #[must_use]
31    pub const fn new(sample: PcmSample, channels: AudioChannels, sample_rate: u32) -> Self {
32        Self { sample, channels, sample_rate }
33    }
34    /// Returns the number of channels.
35    #[must_use]
36    pub const fn channel_count(self) -> usize {
37        self.channels.channels() as usize
38    }
39    /// Returns the byte size of one interleaved frame.
40    #[must_use]
41    pub const fn frame_bytes(self) -> usize {
42        self.sample.bytes() * self.channel_count()
43    }
44    /// Returns whether the sample rate is non-zero.
45    #[must_use]
46    pub const fn has_valid_rate(self) -> bool {
47        self.sample_rate != 0
48    }
49    /// Returns whether this stream has a non-zero channel count and sample rate.
50    #[must_use]
51    pub const fn is_valid(self) -> bool {
52        self.channel_count() != 0 && self.sample_rate != 0 && self.frame_bytes() != 0
53    }
54    /// Returns the number of frames in `data_len`, if frame-aligned.
55    pub const fn frames_for_data_len(self, data_len: usize) -> Option<usize> {
56        let frame_bytes = self.frame_bytes();
57        if frame_bytes == 0 || !data_len.is_multiple_of(frame_bytes) {
58            None
59        } else {
60            Some(data_len / frame_bytes)
61        }
62    }
63    /// Returns whether `data_len` contains complete interleaved frames.
64    #[must_use]
65    pub const fn has_complete_frames_for_data_len(self, data_len: usize) -> bool {
66        self.frames_for_data_len(data_len).is_some()
67    }
68}
