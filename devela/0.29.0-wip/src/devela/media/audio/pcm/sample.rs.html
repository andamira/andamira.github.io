11/// Numeric encoding of a single PCM sample.
12#[doc = crate::_doc_meta!{
13    location("media/audio"),
14    test_size_of(PcmSample = 1|8),
15}]
16#[allow(missing_docs)]
17#[must_use]
18#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
19pub enum PcmSample {
20    I8,
21    U8,
22    #[default]
23    I16,
24    I24,
25    I32,
26    F32,
27    F64,
28}
29_impl_init![Self::I16 => PcmSample];
30impl_trait![fmt::Display for PcmSample |self, f| f.write_str(self.as_code())];
31
32impl PcmSample {
33    /// Returns the number of meaningful sample bits.
34    #[must_use]
35    pub const fn bits(self) -> u8 {
36        match self {
37            Self::I8 => 8,
38            Self::U8 => 8,
39            Self::I16 => 16,
40            Self::I24 => 24,
41            Self::I32 | Self::F32 => 32,
42            Self::F64 => 64,
43        }
44    }
45    /// Returns the canonical byte width for packed raw PCM.
46    #[must_use]
47    pub const fn bytes(self) -> usize {
48        match self {
49            Self::I8 => 1,
50            Self::U8 => 1,
51            Self::I16 => 2,
52            Self::I24 => 3,
53            Self::I32 | Self::F32 => 4,
54            Self::F64 => 8,
55        }
56    }
57    /// Checks the equality of two PCM samples.
58    #[must_use]
59    pub const fn eq(self, other: Self) -> bool {
60        self as u8 == other as u8
61    }
62    /// Returns whether this is an integer sample encoding.
63    #[must_use]
64    pub const fn is_int(self) -> bool {
65        matches!(self, Self::I8 | Self::U8 | Self::I16 | Self::I24 | Self::I32)
66    }
67    /// Returns whether this is a signed integer sample encoding.
68    #[must_use]
69    pub const fn is_signed_int(self) -> bool {
70        matches!(self, Self::I8 | Self::I16 | Self::I24 | Self::I32)
71    }
72
73    /// Returns whether this is an unsigned integer sample encoding.
74    #[must_use]
75    pub const fn is_unsigned_int(self) -> bool {
76        matches!(self, Self::U8)
77    }
78    /// Returns whether this is a floating-point sample encoding.
79    #[must_use]
80    pub const fn is_float(self) -> bool {
81        matches!(self, Self::F32 | Self::F64)
82    }
83    /// Returns whether this sample encoding can represent negative values.
84    #[must_use]
85    pub const fn has_negative(self) -> bool {
86        matches!(self, Self::I8 | Self::I16 | Self::I24 | Self::I32 | Self::F32 | Self::F64)
87    }
88    /// Returns a compact encoding label.
89    #[must_use]
90    pub const fn as_code(self) -> &'static str {
91        match self {
92            Self::I8 => "i8",
93            Self::U8 => "u8",
94            Self::I16 => "i16",
95            Self::I24 => "i24",
96            Self::I32 => "i32",
97            Self::F32 => "f32",
98            Self::F64 => "f64",
99        }
100    }
101
102    // Converts the sample to ALSA format.
103    #[cfg(feature = "alsa")]
104    #[cfg_attr(not(ffi_alsa··), allow(dead_code))]
105    pub(crate) const fn to_alsa(self) -> _alsa_raw::snd_pcm_format_t {
106        match self {
107            Self::I8 => _alsa_raw::SND_PCM_FORMAT_S8,
108            Self::U8 => _alsa_raw::SND_PCM_FORMAT_U8,
109            Self::I16 => _alsa_raw::SND_PCM_FORMAT_S16_LE,
110            Self::I24 => _alsa_raw::SND_PCM_FORMAT_S24_3LE,
111            Self::I32 => _alsa_raw::SND_PCM_FORMAT_S32_LE,
112            Self::F32 => _alsa_raw::SND_PCM_FORMAT_FLOAT_LE,
113            Self::F64 => _alsa_raw::SND_PCM_FORMAT_FLOAT64_LE,
114        }
115    }
116}

119/// Rust sample type with a fixed PCM sample encoding.
120#[doc = crate::_doc_meta!{location("media/audio")}]
121///
122/// This trait connects typed sample buffers such as `&[i16]` or `&mut [f32]`
123/// to their corresponding [`PcmSample`] encoding.
124///
125/// It is sealed because backend I/O depends on the Rust type layout matching
126/// the declared PCM sample representation.
127#[expect(private_bounds, reason = "Sealed")]
128pub trait PcmSampleType: Copy + Sealed {
129    /// The PCM sample encoding represented by this Rust type.
130    const SAMPLE: PcmSample;
131
132    /// The silent value for this PCM sample type.
133    ///
134    /// This is the value that represents zero amplitude in the corresponding
135    /// PCM encoding.
136    const SILENCE: Self;
137}
138
139/// Marker trait to prevent downstream implementations of the [`PcmSampleType`] trait.
140trait Sealed {}
141macro_rules! _pcm_sample_type {
142    () => {
143        _pcm_sample_type!(u8,U8,128; i8,I8,0; i16,I16,0; i32,I32,0; f32,F32,0.0; f64,F64,0.0);
144    };
145    ($($T:ty, $sample:ident, $SI:literal);+ $(;)?) => {
146        $( _pcm_sample_type!(% $T, $sample, $SI);)+
147    };
148    (% $T:ty, $sample:ident, $SI:literal) => {
149        impl Sealed for $T {}
150        impl PcmSampleType for $T {
151            const SAMPLE: PcmSample = PcmSample::$sample;
152            const SILENCE: Self = $SI;
153        }
154    };
155}
156_pcm_sample_type!();
