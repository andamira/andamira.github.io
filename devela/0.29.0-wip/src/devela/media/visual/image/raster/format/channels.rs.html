3//! Defines [`RasterChannels`].
4//
5
6use crate::{_impl_init, RasterPackedChannels, RasterSampleFormat};
7
8/// Channel order, packing, or indexing model of raster elements.
9#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
10#[allow(dead_code, reason = "Not exposed: Xrgb, Xbgr")]
11pub(crate) enum RasterChannels {
12    /// Unknown or unspecified channel model.
13    #[default]
14    Unknown,
15
16    /* gray */
17    /// Single luminance-like channel.
18    Gray,
19    /// Gray channel followed by alpha.
20    GrayAlpha,
21
22    /* rgb */
23    /// Red, green, blue.
24    Rgb,
25    /// Red, green, blue, alpha.
26    Rgba,
27    /// Alpha, red, green, blue.
28    Argb,
29    /// Padding, red, green, blue.
30    Xrgb,
31    /// Red, green, blue, padding.
32    Rgbx,
33
34    /* bgr */
35    /// Blue, green, red.
36    Bgr,
37    /// Blue, green, red, alpha.
38    Bgra,
39    /// Alpha, blue, green, red.
40    Abgr,
41    /// Padding, blue, green, red.
42    Xbgr,
43    /// Blue, green, red, padding.
44    Bgrx,
45
46    /* indexed */
47    /// Palette index instead of direct color channels.
48    Indexed,
49
50    /// Channel fields packed inside one scalar raster element.
51    ///
52    /// The packed format describes the bit layout of that scalar element,
53    /// not the byte order of a borrowed byte slice.
54    Packed(RasterPackedChannels),
55}
56_impl_init![Self::Unknown => RasterChannels];
57
58impl RasterChannels {
59    /// Returns whether this channel model is unknown or incomplete.
60    pub const fn is_unknown(self) -> bool {
61        match self {
62            Self::Unknown => true,
63            Self::Packed(packed) => packed.is_unknown(),
64            _ => false,
65        }
66    }
67
68    /// Returns the number of stored channel fields.
69    ///
70    /// Padding fields such as `X` in `Xrgb` are counted here.
71    pub const fn channel_count(self) -> Option<u8> {
72        match self {
73            Self::Unknown => None,
74
75            Self::Gray | Self::Indexed => Some(1),
76            Self::GrayAlpha => Some(2),
77
78            Self::Rgb | Self::Bgr => Some(3),
79
80            Self::Rgba
81            | Self::Bgra
82            | Self::Argb
83            | Self::Abgr
84            | Self::Xrgb
85            | Self::Xbgr
86            | Self::Rgbx
87            | Self::Bgrx => Some(4),
88
89            Self::Packed(packed) => packed.channel_count(),
90        }
91    }
92
93    /// Returns the number of direct color channel fields.
94    ///
95    /// Alpha and padding fields are not counted. Indexed formats return
96    /// `None`, since the direct element is a palette index, not a color field.
97    pub const fn color_channel_count(self) -> Option<u8> {
98        match self {
99            Self::Unknown | Self::Indexed => None,
100
101            Self::Gray | Self::GrayAlpha => Some(1),
102
103            Self::Rgb
104            | Self::Bgr
105            | Self::Rgba
106            | Self::Bgra
107            | Self::Argb
108            | Self::Abgr
109            | Self::Xrgb
110            | Self::Xbgr
111            | Self::Rgbx
112            | Self::Bgrx => Some(3),
113
114            Self::Packed(packed) => packed.color_channel_count(),
115        }
116    }
117
118    /// Returns whether the channel model has an alpha field.
119    ///
120    /// This only describes the stored channel topology. Semantic alpha
121    /// interpretation is still controlled by `RasterAlpha`.
122    pub const fn has_alpha_field(self) -> bool {
123        match self {
124            Self::GrayAlpha | Self::Rgba | Self::Bgra | Self::Argb | Self::Abgr => true,
125
126            Self::Packed(packed) => packed.has_alpha(),
127
128            Self::Unknown
129            | Self::Gray
130            | Self::Rgb
131            | Self::Bgr
132            | Self::Xrgb
133            | Self::Xbgr
134            | Self::Rgbx
135            | Self::Bgrx
136            | Self::Indexed => false,
137        }
138    }
139
140    /// Returns whether the channel model has a padding field.
141    pub const fn has_padding_channel(self) -> bool {
142        match self {
143            Self::Xrgb | Self::Xbgr | Self::Rgbx | Self::Bgrx => true,
144            Self::Packed(packed) => packed.has_padding_channel(),
145
146            Self::Unknown
147            | Self::Gray
148            | Self::GrayAlpha
149            | Self::Rgb
150            | Self::Bgr
151            | Self::Rgba
152            | Self::Bgra
153            | Self::Argb
154            | Self::Abgr
155            | Self::Indexed => false,
156        }
157    }
158
159    /// Returns whether this channel model stores palette indices.
160    pub const fn is_indexed(self) -> bool {
161        matches!(self, Self::Indexed)
162    }
163
164    /// Returns whether this channel model uses packed scalar pixels.
165    pub const fn is_packed(self) -> bool {
166        matches!(self, Self::Packed(_))
167    }
168
169    /// Returns the number of meaningful bits per pixel for this channel model.
170    ///
171    /// Padding bits are excluded. Alpha bits are included when present.
172    pub const fn depth_bits(self, sample: RasterSampleFormat) -> Option<u16> {
173        match self {
174            Self::Unknown => None,
175            Self::Packed(packed) => packed.depth_bits(),
176            Self::Xrgb | Self::Xbgr | Self::Rgbx | Self::Bgrx => match sample.bits() {
177                Some(bits) => Some(3 * bits),
178                None => None,
179            },
180            _ => match (self.channel_count(), sample.bits()) {
181                (Some(channels), Some(bits)) => Some(channels as u16 * bits),
182                _ => None,
183            },
184        }
185    }
186
187    /// Returns the number of stored bits per pixel for this channel model.
188    ///
189    /// Padding bits are included.
190    pub const fn bits_per_pixel(self, sample: RasterSampleFormat) -> Option<u16> {
191        match self {
192            Self::Unknown => None,
193            Self::Packed(packed) => packed.bits_per_pixel(),
194
195            _ => match (self.channel_count(), sample.bits()) {
196                (Some(channels), Some(bits)) => Some(channels as u16 * bits),
197                _ => None,
198            },
199        }
200    }
201}
