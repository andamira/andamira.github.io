9/// A sixel character.
10#[doc = crate::_doc_meta!{location("media/visual/image")}]
11#[must_use]
12#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
13pub struct SixelChar(u8);
14
15impl SixelChar {
16    const MASK: u8 = 0b111_111; // == 63
17
18    /// Create a sixel character from the first 6 bits of the given `mask`.
19    ///
20    /// Bit 1 is the bottom pixel and bit 6 (== 32) is the top pixel.
21    /// Bits 7 and 8 are ignored.
22    #[inline(always)] #[rustfmt::skip]
23    pub const fn from_bitmask(mask: u8) -> Self { Self(mask & Self::MASK) }
24
25    /// Get the 6-bit bitmask representation.
26    #[must_use] #[inline(always)] #[rustfmt::skip]
27    pub const fn as_bitmask(self) -> u8 { self.0 }
28
29    /// Get the sixel byte value.
30    #[must_use] #[inline(always)] #[rustfmt::skip]
31    pub const fn as_byte(self) -> u8 { self.0 + Self::MASK }
32
33    /// Create a sixel character from a valid Unicode scalar, from '@' to '~'.
34    pub const fn from_char(c: char) -> Option<Self> {
35        let byte = c as u32;
36        if byte >= Self::MASK as u32 && byte <= (Self::MASK + Self::MASK) as u32 {
37            Some(Self((byte - Self::MASK as u32) as u8))
38        } else {
39            None
40        }
41    }
42    /// Returns the corresponding Unicode scalar.
43    #[must_use] #[inline(always)] #[rustfmt::skip]
44    pub const fn as_char(self) -> char { self.as_byte() as char }
45
46    /// Checks the equality of two chars.
47    #[must_use] #[inline(always)] #[rustfmt::skip]
48    pub const fn eq(self, other: Self) -> bool { self.0 == other.0 }
49
50    /// Returns `true` if all the pixels are unset.
51    #[must_use] #[inline(always)] #[rustfmt::skip]
52    pub const fn is_empty(self) -> bool { self.eq(Self::EMPTY) }
53
54    /// Returns `true` if all the pixels are set.
55    #[must_use] #[inline(always)] #[rustfmt::skip]
56    pub const fn is_full(self) -> bool { self.eq(Self::FULL) }
57
58    /// Returns the next sixel character in sequence, wrapping around from 63 to 0.
59    pub const fn next(self) -> Self {
60        is![self.eq(Self::FULL), Self::EMPTY, Self(self.0 - 1)]
61    }
62
63    /// Returns the previous sixel character in sequence, wrapping around from 0 to 63.
64    pub const fn prev(self) -> Self {
65        is![self.eq(Self::EMPTY), Self::FULL, Self(self.0 - 1)]
66    }
67
68    /// Returns the next sixel character in sequence, returning None if at maximum.
69    #[must_use]
70    pub const fn next_checked(self) -> Option<Self> {
71        is![self.eq(Self::FULL), None, Some(Self(self.0 + 1))]
72    }
73
74    /// Returns the previous sixel character in sequence, returning None if at minimum.
75    #[must_use]
76    pub const fn prev_checked(self) -> Option<Self> {
77        is![self.eq(Self::EMPTY), None, Some(Self(self.0 - 1))]
78    }
79}

81/// # Other conversions
82impl SixelChar {
83    /* bools */
84
85    /// Create a sixel character from 6 pixels [top, 2, 3, 4, 5].
86    #[allow(clippy::identity_op)]
87    pub const fn from_bools(bools: [bool; 6]) -> Self {
88        let value = (bools[0] as u8) * 32  // Top pixel = MSB
89            + (bools[1] as u8) * 16
90            + (bools[2] as u8) * 8
91            + (bools[3] as u8) * 4
92            + (bools[4] as u8) * 2
93            + (bools[5] as u8) * 1; // Bottom pixel = LSB
94        Self(value)
95    }
96    /// Extract the sixel bits as [top, 2, 3, 4, 5, bottom] bools.
97    #[must_use]
98    pub const fn to_bools(self) -> [bool; 6] {
99        let value = self.as_bitmask();
100        [
101            value & 32 != 0, // Top pixel (MSB)
102            value & 16 != 0,
103            value & 8 != 0,
104            value & 4 != 0,
105            value & 2 != 0,
106            value & 1 != 0, // Bottom pixel (LSB)
107        ]
108    }
109
110    /* braille */
111
112    /// Attempts to create a `SixelChar` from a Braille pattern Unicode character.
113    ///
114    /// Returns `Some(SixelChar)` if the character is in the Braille block (U+2800..=U+28FF), where
115    /// dots 1-6 map to the sixel bits (dots 7-8 are ignored). Returns `None` for other characters.
116    pub const fn from_braille(braille: char) -> Option<Self> {
117        let b = braille as u32;
118        if b >= 0x2800 && b <= 0x28FF {
119            let bits = (b & Self::MASK as u32) as u8;
120            Some(Self::from_bitmask(bits))
121        } else {
122            None
123        }
124    }
125    /// Creates a `SixelChar` from any character, using the lower 6 bits of its codepoint.
126    ///
127    /// For Braille patterns (U+2800..=U+28FF), this preserves dots 1-6 and ignores dots 7-8.
128    /// For other characters, the result may not be visually meaningful.
129    pub const fn from_braille_unchecked(braille: char) -> Self {
130        Self::from_bitmask((braille as u32 & Self::MASK as u32) as u8)
131    }
132    /// Converts this sixel to a Braille pattern character.
133    ///
134    /// The sixel bits map directly to Braille dots 1-6:
135    /// - Bit 0 (top) → Dot 1
136    /// - Bit 5 (bottom) → Dot 6
137    #[must_use]
138    pub const fn to_braille(self) -> char {
139        unwrap![some char::from_u32(0x2800 + self.as_bitmask() as u32)]
140    }
141
142    /* hexagram */
143
144    /// Creates a sixel from an I Ching hexagram character.
145    ///
146    /// - Returns `None` if the character is not in the hexagram range (U+4DC0..=U+4DFF).
147    /// - Standard mapping: filled pixels represent broken lines.
148    pub const fn from_hexagram(hexagram: char) -> Option<Self> {
149        let h = hexagram as usize;
150        if h >= 0x4DC0 && h <= 0x4DFF {
151            Some(Self::from_bitmask(AsciiLut::CHAR_HEXAGRAM_TO_SIXEL[h - 0x4DC0]))
152        } else {
153            None
154        }
155    }
156    /// Converts to an I Ching hexagram character using standard mapping.
157    ///
158    /// - Standard mapping: filled pixels represent broken lines.
159    pub const fn to_hexagram(self) -> char {
160        AsciiLut::SIXEL_TO_CHAR_HEXAGRAM[self.as_bitmask() as usize]
161    }
162
163    /// Creates a sixel from an I Ching hexagram character.
164    ///
165    /// - Returns `None` if the character is not in the hexagram range (U+4DC0..=U+4DFF).
166    /// - Inverted mapping: filled pixels represent unbroken lines.
167    pub const fn from_hexagram_inverted(hexagram: char) -> Option<Self> {
168        let h = hexagram as usize;
169        if h >= 0x4DC0 && h <= 0x4DFF {
170            Some(Self::from_bitmask(!AsciiLut::CHAR_HEXAGRAM_TO_SIXEL[h - 0x4DC0]))
171        } else {
172            None
173        }
174    }
175    /// Converts to an I Ching hexagram character using inverted mapping.
176    ///
177    /// - Inverted mapping: filled pixels represent unbroken lines.
178    pub const fn to_hexagram_inverted(self) -> char {
179        AsciiLut::SIXEL_TO_CHAR_HEXAGRAM[(!self.as_bitmask() & Self::MASK) as usize]
180    }

184    /// Converts this sixel to an ANSI-colored string showing multiple representations.
185    ///
186    /// The output displays:
187    /// - The raw character (red)
188    /// - Braille pattern equivalent (green)
189    /// - Box representation (blue)
190    /// - Hexagram character (cyan)
191    /// - Binary bitmask representation
192    ///
193    /// Uses ANSI escape codes for coloring and resets formatting at the end.
194    /// # Examples
195    /// ```ignore
196    /// # use devela::{Ansi, SixelChar, write_at};
197    /// let (mut offset, mut result) = (0, [0; 65]);
198    /// write_at![result, +=offset, @Ansi::BLACK_BG, @Ansi::RED, b'@', @'⠁'];
199    /// assert_eq![
200    ///     SixelChar::TOP.to_string_ansi(), // == "@⠁■□□□□□|000001"
201    ///     "\u{1b}[40m\u{1b}[31m@\u{1b}[32m⠁\u{1b}[34m■□□□□□\u{1b}[0m\u{1b}[36m\u{1b}[0m|000001"
202    /// ];
203    /// ```
204    #[allow(non_snake_case)]
205    // TODO: make const to_string_ansi
206    pub fn to_string_ansi(&self) -> StringU8<65> {
207        use crate::{Ansi, lets};
208        let mut buf = [0; 65];
209        let c = self.as_char();
210        let m = self.as_bitmask();
211        let b = self.to_braille();
212        // let h = self.to_hexagram(); // double-width
213        let bx = self.to_string_box();
214        lets![res = "\x1b[0m", @Ansi::{R=RED, B=BLUE, G=GREEN, W=CYAN, KB=BLACK_BG}];
215        let args = format_args!["{KB}{R}{c}{G}{b}{B}{bx}{res}{W}{res}|{m:06b}"];
216        let len = FmtWriter::format_len_unchecked(&mut buf, args);
217        StringU8::<65>::_from_array_len_trusted(buf, len as u8)
218    }
219
220    /// Converts this sixel to a box representation showing pixel states.
221    ///
222    /// Each of the six pixels is represented as either:
223    /// - `■` for filled/true pixels
224    /// - `□` for empty/false pixels
225    ///
226    /// The output shows pixels from top to bottom.
227    /// # Examples
228    /// ```
229    /// # use devela::SixelChar;
230    /// assert_eq![SixelChar::from_bitmask(0b111001).to_string_box(), "■□□■■■"];
231    /// ```
232    pub const fn to_string_box(&self) -> StringU8<20> {
233        let mut string = StringU8::<20>::new();
234        let mask = self.as_bitmask();
235        let mut i = 0;
236        while i < 6 {
237            is![mask & (1 << i) != 0, string.push('■'), string.push('□')];
238
239            i += 1;
240        }
241        assert![string.len() == 18];
242        string
243    }
244}

246/// # Constants
247impl SixelChar {
248    /// No pixels set. (?)
249    pub const EMPTY: Self = Self::from_bitmask(0b000_000);
250    /// All the six pixels set ().
251    pub const FULL: Self = Self::from_bitmask(0b111_111);
252
253    /// The bottom pixel (_).
254    pub const BOT: Self = Self::from_bitmask(0b100_000); // MSB
255    /// The 2 bottom pixels (o).
256    pub const BOT2: Self = Self::from_bitmask(0b110_000);
257    /// The 3 bottom pixels (w).
258    pub const BOT3: Self = Self::from_bitmask(0b111_000);
259    /// The 3 bottom pixels ({).
260    pub const BOT4: Self = Self::from_bitmask(0b111_100);
261    /// The 5 bottom pixels (}).
262    pub const BOT5: Self = Self::from_bitmask(0b111_110);
263
264    /// The top pixel (@).
265    pub const TOP: Self = Self::from_bitmask(0b000_001); // LSB
266    /// The 2 top pixels (B).
267    pub const TOP2: Self = Self::from_bitmask(0b000_011);
268    /// The 3 top pixels (F).
269    pub const TOP3: Self = Self::from_bitmask(0b000_111);
270    /// The 4 top pixels (N).
271    pub const TOP4: Self = Self::from_bitmask(0b001_111);
272    /// The 5 top pixels (^).
273    pub const TOP5: Self = Self::from_bitmask(0b011_111);
274}

276impl Display for SixelChar {
277    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
278        let mut buf = [0u8; 16];
279        let c = self.as_char();
280        let m = self.as_bitmask();
281        let b = self.to_braille();
282        f.write_str(format_buf!(&mut buf, "[{c}{b}0b_{m:06b}]").unwrap())
283    }
284}

322    #[cfg(test)]
323    const _SIXEL_HEXAGRAM_CORRESPONDANCE: () = {
324        let mut index = 0;
325        while index < 64 {
326            let bitmask = AsciiLut::CHAR_HEXAGRAM_TO_SIXEL[index];
327            let char_from_bitmask = AsciiLut::SIXEL_TO_CHAR_HEXAGRAM[bitmask as usize];
328            let char_code = char_from_bitmask as u32;
329            let index_from_char = (char_code - 0x4DC0) as usize;
330
331            // Verify that SIXEL_TO_CHAR_HEXAGRAM[bitmask] gives a character
332            // whose code point corresponds to the original index
333            assert!(index_from_char == index);
334
335            // Verify that CHAR_HEXAGRAM_TO_SIXEL[index_from_char] gives back the original bitmask
336            let roundtrip_bitmask = AsciiLut::CHAR_HEXAGRAM_TO_SIXEL[index_from_char];
337            assert!(roundtrip_bitmask == bitmask);
338
339            index += 1;
340        }
341    };
