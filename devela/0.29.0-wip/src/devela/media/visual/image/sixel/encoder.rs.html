63    /// Sixel encoding with RLE compression for repeated characters.
64    ///
65    /// - RLE encoding is used when the same sixel character repeats 4 or more times.
66    #[rustfmt::skip]
67    pub const fn encode_rgb(&mut self, rgb: &[u8], width: usize, height: usize, buf: &mut [u8])
68        -> Result<usize, NotEnoughSpace> {
69        __dbg![slog!{clear sixel_encoder:64+64}]; // slog point of entry
70
71        let mut off = 0;
72        off += unwrap![ok? Self::write_sixel_start_simple(slice![mut buf, off,..])];
73
74        self.width = width;
75        self.height = height;
76
77        unwrap![ok? self.build_palette(rgb, width, height)];
78        off += self.palette.write_definitions(slice![mut buf, off, ..]);
79
80        // process each horizontal band
81        let mut band_y = 0;
82        while band_y < height {
83            let band_height = Cmp(height - band_y).min(6);
84            __dbg![slog!{sixel_encoder:64+64
85                "Processing band at row ", %band_y, " (height: ", %band_height, ")"}];
86
87            // for each color, render with RLE compression
88            let mut iter = self.palette.defined_colors();
89            while let Some((idx, palette_color)) = iter.next() {
90                is![idx == 0, continue]; // Skip background
91
92                __dbg![slog!{sixel_encoder:64+64 "  Rendering color #", %idx, " in band."}];
93                off += unwrap![ok? Self::write_color_reference(slice![mut buf, off,..], idx)];
94
95                // build and compress sixel pattern for this color
96                lets![mut current_char = None, mut repeat_count = 0];
97                whilst! { x in 0..width; {
98                    let sixel_bits = self.build_sixel_bits_for_color(rgb, width, x, band_y,
99                        band_height, palette_color);
100                    let sixel_char = SixelChar::from_bitmask(sixel_bits);
101
102                    // RLE compression logic
103                    // Handles lines longer than 255 by splitting long runs,
104                    // outputting the current run and starting a new one with the same character
105                    let is_same_char = is![let Some(c) = current_char, sixel_char.eq(c), false];
106                    match (is_same_char, repeat_count == 255) {
107                        (true, false) => {
108                            repeat_count += 1;
109                        }
110                        (true, true) => { // Split run: output but keep same character
111                            if let Some(char) = current_char {
112                                off += unwrap![ok? Self::write_rle_run(slice![mut buf, off,..],
113                                    char, repeat_count)];
114                            }
115                            repeat_count = 1;
116                        }
117                        (false, _) => { // New run: output and switch character
118                            if let Some(char) = current_char {
119                                off += unwrap![ok? Self::write_rle_run(slice![mut buf, off,..],
120                                    char, repeat_count)];
121                            }
122                            current_char = Some(sixel_char);
123                            repeat_count = 1;
124                        }
125                    }
126                }}
127                // output the final run
128                if let Some(char) = current_char {
129                    off += unwrap![ok?
130                        Self::write_rle_run(slice![mut buf, off,..], char, repeat_count)];
131                }
132                // return to start for next color
133                is! { idx != self.palette.len() as u16 - 1, write_at![buf, +=off, b'$'];}
134            }
135            band_y += 6;
136            is! { band_y < height, write_at![buf, +=off, b'-'];} // move to next band?
137        }
138        off += unwrap![ok? Self::write_sixel_end(slice![mut buf, off,..])];
139        Ok(off)
140    }
141}

151impl<const MAX_COLORS: usize> SixelEncoder<MAX_COLORS> {
152    /// Builds the palette from the given rgb byte buffer.
153    const fn build_palette(&mut self, rgb: &[u8], w: usize, h: usize) -> Result<(), NotEnoughSpace> {
154        self.palette = SixelPalette::new();
155        unwrap![ok? self.palette.add_color(SixelColor::BLACK)];
156        whilst! { i in 0..(w * h); {
157            // is![self.palette.is_full(), break]; // early termination MAYBE for another version
158            let idx = i * 3;
159            let color = SixelColor::from_rgb888(rgb[idx], rgb[idx + 1], rgb[idx + 2]);
160            if !color.is_black() { let _ = self.palette.find_or_add(color); }
161        }}
162        Ok(())
163    }
164
165    /// Build sixel bits for a specific color in a column.
166    const fn build_sixel_bits_for_color(&self, rgb: &[u8], width: usize, x: usize, band_y: usize,
167        band_height: usize, target_color: SixelColor) -> u8 {
168        let mut sixel_bits = 0u8;
169        whilst! { dy in 0..band_height; {
170            let y = band_y + dy;
171            let idx = (y * width + x) * 3;
172            // only process if within bounds
173            if idx + 2 < rgb.len() {
174                let pixel_color = SixelColor::from_rgb888(rgb[idx], rgb[idx + 1], rgb[idx + 2]);
175                if pixel_color.is_similar_to(target_color) { sixel_bits |= 1 << dy }
176            }
177        }}
178        sixel_bits
179    }
180}

188impl<const MAX_COLORS: usize> SixelEncoder<MAX_COLORS> {
189    /// Write sixel sequence start with full raster attributes.
190    ///
191    /// Returns an error if the buffer doesn't have at least 21 bytes, for good measure.
192    pub const fn write_sixel_start(
193        buf: &mut [u8],
194        width: usize,
195        height: usize,
196    ) -> Result<usize, NotEnoughSpace> {
197        let needed = 5 + 6 + 5 + 1 + 5; // assume we require 5+5 digits for raster dimensions
198        is![buf.len() < needed, return Err(NotEnoughSpace(Some(needed)))];
199
200        let (width_digits, height_digits) = (Digits(width), Digits(height));
201        let mut offset = 0;
202
203        // initial sequence (6 bytes)
204        write_at!(
205            buf, +=offset, b'\x1b', b'P', // CIS
206            b';', b'1', b';', // p2 = 1 (transparent background)
207            b'q', // sixel ID
208        );
209        // raster attributes opening sequence (5 bytes)
210        // https://vt100.net/docs/vt3xx-gp/chapter14.html#S14.3.2
211        write_at!(buf, +=offset, b'"', b'1', b';', b'1', b';'); // pixel aspect ratio
212        offset += width_digits.write_digits10(buf, offset);
213        write_at!(buf, +=offset, b';'); // + 1 byte
214        offset += height_digits.write_digits10(buf, offset);
215        Ok(offset)
216    }
217    /// Write sixel sequence start with full raster attributes
218    pub const fn write_sixel_start_simple(buf: &mut [u8]) -> Result<usize, NotEnoughSpace> {
219        is![buf.len() < 6, return Err(NotEnoughSpace(Some(6)))];
220        write_at!(
221            buf, 0, b'\x1b', b'P', // CIS
222            b';', b'1', b';', // p2 = 1 (transparent background)
223            b'q', // sixel ID
224        );
225        Ok(6)
226    }
227
228    /// Write sixel sequence end.
229    pub const fn write_sixel_end(buf: &mut [u8]) -> Result<usize, NotEnoughSpace> {
230        is![buf.len() < 2, return Err(NotEnoughSpace(Some(2)))];
231        write_at!(buf, 0, b'\x1b', b'\\'); // string terminator C0 sequence
232        Ok(2)
233    }
234
235    /// Write color reference (#NN)
236    const fn write_color_reference(buffer: &mut [u8], index: u16) -> Result<usize, NotEnoughSpace> {
237        is![buffer.len() < 2, return Err(NotEnoughSpace(Some(2)))];
238        buffer[0] = b'#';
239        let digits_written = Digits(index).write_digits10_nonzero(buffer, 1);
240        is![digits_written == 0, return Err(NotEnoughSpace(Some(2)))];
241        Ok(1 + digits_written)
242    }
243
244    /// Write an RLE run: either compressed `!NNN` or repeated characters.
245    const fn write_rle_run(
246        buffer: &mut [u8],
247        sc: SixelChar,
248        count: u8,
249    ) -> Result<usize, NotEnoughSpace> {
250        let mut off = 0;
251        if count >= 4 {
252            let digits = Digits(count);
253            let dcount = digits.count_digits10();
254            let needed = 1 + dcount as usize;
255            is![buffer.len() < needed, return Err(NotEnoughSpace(Some(needed)))];
256
257            write_at![buffer, +=off, b'!'];
258            off += digits.write_digits10(buffer, 1);
259            write_at![buffer, +=off, sc.as_byte()];
260            __dbg![slog! {sixel_encoder:64+64 "  RLE: !", %count, ":", @sc.to_string_box(),
261            " (saved ", %count-(2+dcount), " bytes)"}];
262        } else {
263            __dbg![slog! {sixel_encoder:64+64
264            "  NO RLE: ", %count, " times ", @sc.to_string_box()}];
265            whilst! { _n in 0..count; {
266                is![off >= buffer.len(), return Err(NotEnoughSpace(Some(buffer.len())))];
267                write_at![buffer, +=off, sc.as_byte()];
268            }}
269        }
270        Ok(off)
271    }
