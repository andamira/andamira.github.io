3//! Defines [`ColorDepth`].

6#[doc = crate::_tags!(color)]
7/// Color vocabulary depth.
8#[doc = crate::_doc_meta!{location("media/visual/color")}]
9#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
10#[repr(u8)]
11pub enum ColorDepth {
12    /// No selectable color.
13    #[default]
14    Mono = 0,
15    /// 2-color palette, 1 bit.
16    Palette2 = 1,
17    /// 8-color palette, 3 bits.
18    Palette8 = 3,
19    /// 16-color palette, 4 bits.
20    Palette16 = 4,
21    /// 256-color palette, 8 bits.
22    Palette256 = 8,
23    /// 16-bit RGB color, commonly RGB565.
24    Rgb16 = 16,
25    /// 24-bit RGB color.
26    Rgb24 = 24,
27}
28#[allow(non_upper_case_globals)]
29impl ColorDepth {
30    /// An alias of `Palette8`, matching 3-bit ANSI color.
31    pub const Ansi3: Self = Self::Palette8;
32    /// An alias of `Palette16`, matching 4-bit ANSI color.
33    pub const Ansi4: Self = Self::Palette16;
34    /// An alias of `Palette256`, matching 8-bit ANSI color.
35    pub const Ansi8: Self = Self::Palette256;
36    /// An alias of `Rgb16`, commonly called high color.
37    pub const High16: Self = Self::Rgb16;
38    /// An alias of `Rgb24`, commonly called true color.
39    pub const True24: Self = Self::Rgb24;
40}
41impl ColorDepth {
42    /// Returns `true` if this supports any selectable color.
43    #[must_use]
44    pub const fn has_color(self) -> bool {
45        !matches!(self, Self::Mono)
46    }
47    /// Returns `true` if this supports direct RGB color.
48    #[must_use]
49    pub const fn has_rgb(self) -> bool {
50        matches!(self, Self::Rgb16 | Self::Rgb24)
51    }
52    /// Returns whether this depth can represent at least `other`.
53    #[must_use]
54    pub const fn supports(self, other: Self) -> bool {
55        self as u8 >= other as u8
56    }
57    /// Returns the greatest supported depth not exceeding `bits`.
58    ///
59    /// For example, `7` truncates to [`Palette16`][Self::Palette16],
60    /// and `23` truncates to [`Rgb16`][Self::Rgb16].
61    #[must_use]
62    pub const fn from_bits_trunc(bits: u8) -> Self {
63        match bits {
64            24..=u8::MAX => Self::Rgb24,
65            16..=23 => Self::Rgb16,
66            8..=15 => Self::Palette256,
67            4..=7 => Self::Palette16,
68            3 => Self::Palette8,
69            1..=2 => Self::Palette2,
70            _ => Self::Mono,
71        }
72    }
73    /// Returns the nominal bit depth.
74    #[must_use]
75    pub const fn bits(self) -> u8 {
76        self as u8
77    }
78}
