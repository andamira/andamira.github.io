9#[doc = crate::_tags!(color)]
10/// A generic luminance-like component.
11#[doc = crate::_doc_meta!{location("media/visual/color")}]
12///
13/// Represents either physical luminance, gamma-encoded luma, or perceptual lightness,
14/// depending on the `LINEAR` and `LIGHTNESS` flags.
15///
16/// Variants (in order of typical usage):
17/// - [`Luminance<T>`]:        linear, physical
18/// - [`Lightness<T>`]:        non-linear, perceptual
19/// - [`Luma<T>`]:             non-linear, technical
20/// - [`LinearLightness<T>`]:  linear, perceptual (experimental hybrid)
21#[must_use]
22#[repr(C)]
23#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, PartialOrd, Ord)]
24pub struct Lum<T, const LINEAR: bool = true, const LIGHTNESS: bool = false> {
25    /// The luminance-like channel value.
26    pub c: [T; 1],
27}
28
29/* aliases*/
30
31#[doc = crate::_tags!(color)]
32/// Physical [luminance].
33#[doc = crate::_doc_meta!{location("media/visual/color")}]
34///
35/// Linear light intensity, measured in cd/m² or normalized to [0.0, 1.0].
36///
37/// [luminance]: https://en.wikipedia.org/wiki/Luminance
38pub type Luminance<T> = Lum<T, true, false>;
39
40#[doc = crate::_tags!(color)]
41/// Perceptual [lightness] (L*).
42#[doc = crate::_doc_meta!{location("media/visual/color")}]
43///
44/// Non-linear encoding of luminance,
45/// normalized to [0.0, 1.0] for floats or 0..=MAX for integers.
46///
47/// [lightness]: https://en.wikipedia.org/wiki/Lightness
48pub type Lightness<T> = Lum<T, false, true>;
49
50#[doc = crate::_tags!(color)]
51/// Gamma-encoded [luma] (Y′).
52#[doc = crate::_doc_meta!{location("media/visual/color")}]
53///
54/// A non-linear approximation of luminance, typically used in video systems.
55///
56/// [luma]: https://en.wikipedia.org/wiki/Luma_(video)
57pub type Luma<T> = Lum<T, false, false>;
58
59#[doc = crate::_tags!(color)]
60/// Linearized perceptual lightness (L* in linear space).
61#[doc = crate::_doc_meta!{location("media/visual/color")}]
62///
63/// Use cases include:
64/// - Combining linear luminance (for precise computations)
65///   and perceptual lightness (for display scaling).
66/// - Tone mapping in HDR imaging, where linear data is scaled to a perceptual range.
67/// - Representing raw radiometric data (e.g., watts/sr/m²) prior to photometric weighting (CIE Y).
68pub type LinearLightness<T> = Lum<T, true, true>;
69
70/// ## Args
71/// `$T`   : the type used to represent the main value. (u8, u16, f32, f64)
72/// `$f`   : associated floating-point type for operations. (f32|f64)
73/// `$BITS`: the number of bits of each inner component.
74/// `$INT` : a boolean indicating whether the components are integers.
75macro_rules! impl_lum {
76    () => {
77        impl_lum![common u8|f32, u16|f32];
78        impl_lum![common f32|f32];
79        impl_lum![common f64|f64];
80
81        impl_lum![lumina u8|f32:8+true, u16|f32:16+true];
82        impl_lum![lumina f32|f32:32+false];
83        impl_lum![lumina f64|f64:64+false];
84
85        impl_lum![light u8|f32:8+true, u16|f32:16+true];
86        impl_lum![light f32|f32:32+false];
87        impl_lum![light f64|f64:64+false];
88
89        impl_lum![luma u8|f32:8+true, u16|f32:16+true];
90        impl_lum![luma f32|f32:32+false];
91        impl_lum![luma f64|f64:64+false];
92
93        impl_lum![lumi_light u8|f32:8+true, u16|f32:16+true];
94        impl_lum![lumi_light f32|f32:32+false];
95        impl_lum![lumi_light f64|f64:64+false];
96    };
97    ( // Methods common to all types.
98      common  $( $T:ty | $f:ty ),+) => { $( impl_lum![@common $T|$f]; )+ };
99    (@common     $T:ty | $f:ty    ) => {
100        impl<const LINEAR: bool, const LIGHTNESS: bool> Lum<$T, LINEAR, LIGHTNESS> {
101            /// New `Luminance` with the given channel.
102            pub const fn new(c: $T) -> Self { Self { c: [c] } }
103
104            /// Returns the raw channel value, regardless of interpretation.
105            ///
106            /// Prefer type-specific methods like [`luminance()`](Luminance::luminance) or
107            /// [`lightness()`](Lightness::lightness) where possible.
108            pub const fn l(self) -> $T { self.c[0] }
109
110            /// Returns a mutable reference to the raw channel value.
111            pub const fn l_mut(&mut self) -> &mut $T { &mut self.c[0] }
112
113            /// Converts an `Rgb` into unweighted brightness by averaging the R'G'B' components.
114            ///
115            /// May be useful for quick approximations.
116            /// Not correct for perceptual brightness (luma) or physical light (luminance).
117            pub const fn brightness_from_rgb(rgb: Rgb<$T>) -> Self {
118                Self::new((rgb.r() + rgb.g() + rgb.b()) / <$T as NumConst>::NUM_THREE.unwrap())
119            }
120        }
121    };
122    ( // Methods for Luminance: (linear non-lightness)
123      lumina $( $T:ty | $f:ty : $BITS:literal + $INT:literal),+) => {
124        $( impl_lum![@lumina $T|$f : $BITS+$INT]; )+
125    };
126    (@lumina    $T:ty | $f:ty : $BITS:literal + $INT:literal   ) => {
127        // impl_color![lum: $T, $BITS, $INT, true, false]; // TEMP
128
129        impl Luminance<$T> {
130            /// Returns the **linear luminance** (physical light intensity, Y).
131            ///
132            /// Measured in cd/m² (floats) or normalized (integers).
133            pub const fn luminance(self) -> $T { self.c[0] }
134            /// Returns a mutable reference to the **linear luminance**.
135            pub const fn luminance_mut(&mut self) -> &mut $T { &mut self.c[0] }
136
137            /* gamma conversion */
138        }
139    };
140    ( // Methods for Lightness: (non-linear, lightness)
141      light $( $T:ty | $f:ty : $BITS:literal + $INT:literal),+) => {
142        $( impl_lum![@light $T|$f : $BITS+$INT]; )+
143    };
144    (@light    $T:ty | $f:ty : $BITS:literal + $INT:literal   ) => {
145        // impl_color![lum: $T, $BITS, $INT, false, true]; // TEMP
146
147        impl Lightness<$T> {
148            /// Returns the **perceptual lightness** (CIE L\*).
149            ///
150            /// Normalized to `0.0..=1.0` (floats) or `0..=MAX` (integers).
151            pub const fn lightness(self) -> $T { self.c[0] }
152            /// Returns a mutable reference to the **perceptual lightness**.
153            pub const fn lightness_mut(&mut self) -> &mut $T { &mut self.c[0] }
154
155            /* gamma conversion */

162            // pub const fn to_luma() -> Luma<$T> {

164            // }
165        }
166    };
167    ( // Methods for Luma: (non-linear, non-lightness)
168      luma $( $T:ty | $f:ty : $BITS:literal + $INT:literal),+) => {
169        $( impl_lum![@luma $T|$f : $BITS+$INT]; )+
170    };
171    (@luma    $T:ty | $f:ty : $BITS:literal + $INT:literal    ) => {
172        // impl_color![lum: $T, $BITS, $INT, false, false]; // TEMP
173
174        impl Luma<$T> {
175            /// Returns the **gamma-encoded luma** (non-linear Y′).
176            ///
177            /// Compatible with sRGB/Rec. 709 for display.
178            pub const fn lightness(self) -> $T { self.c[0] }
179            /// Returns a mutable reference to the **gamma-encoded luma**.
180            pub const fn lightness_mut(&mut self) -> &mut $T { &mut self.c[0] }
181
182            /* gamma conversion */

189            // pub const fn to_lightness() -> Luma<$T> {
190            //     todo![]
191            // }
192        }
193    };
194    ( // Methods for LinearLuminance: (linear, lightness)
195      lumi_light $( $T:ty | $f:ty : $BITS:literal + $INT:literal),+) => {
196          $( impl_lum![@lumi_light $T|$f : $BITS+$INT]; )+
197    };
198    (@lumi_light    $T:ty | $f:ty : $BITS:literal + $INT:literal   ) => {
199        // impl_color![lum: $T, $BITS, $INT, true, true]; TEMP

202            /// Returns the **linear-light perceptual** value (experimental).

206            /// Returns a mutable reference to the **linear-light perceptual** value.
