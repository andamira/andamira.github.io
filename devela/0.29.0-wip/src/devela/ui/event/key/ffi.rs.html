9#[doc = crate::_tags!(interaction ffi)]
10/// An FFI-safe version of [`Key`], used in [`EventKeyFfi`][crate::EventKeyFfi].
11// (The main difference is in the Char variant.)
12#[doc = crate::_doc_meta!{
13    location("ui/event"),
14    test_size_of(KeyFfi = 8|64; niche Option),
15}]
16#[repr(C)]
17#[non_exhaustive]
18#[allow(missing_docs)] #[rustfmt::skip]
19#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, Hash)]
20pub enum KeyFfi {
21    // common control keys
22    Backspace, Enter, Tab, Escape, Space,
23    // navigation keys
24    Left, Right, Up, Down,
25    Home, End, PageUp, PageDown,
26    // editing keys
27    Delete, Insert,
28    // lock keys
29    CapsLock, ScrollLock, NumLock,
30    // special system kesy
31    PrintScreen, Pause, Menu,
32
33    /// A keypad key.
34    Pad(KeyPad),
35    /// A multimedia key.
36    Media(KeyMedia),
37    /// A modifier key.
38    Mod(KeyMod),
39
40    /// Function keys (F1-F48).
41    Fn(u8),
42
43    #[doc = crate::_tags!(ffi)]
44    /// A unicode character (FFI-safe, stored as `u32`).
45    Char(u32),
46
47    /// A dead key (accent prefix, composition).
48    Dead(KeyDead),
49
50    // Letters
51    A, B, C, D, E, F, G, H, I, J,
52    K, L, M, N, O, P, Q, R, S, T,
53    U, V, W, X, Y, Z,
54
55    // Digits (E row)
56    Digit1, Digit2, Digit3, Digit4, Digit5,
57    Digit6, Digit7, Digit8, Digit9, Digit0,
58
59    // ANSI punctuation / symbol positions
60    // E row
61    /// Physical key left of '1'
62    Backquote,
63    /// Physical `-` key.
64    Minus,
65    /// Physical `=` key.
66    Equal,
67    // D row
68    /// Physical `[` key.
69    BracketLeft,
70    /// Physical `]` key.
71    BracketRight,
72    // C (home) row
73    /// Physical `;` key.
74    Semicolon,
75    /// Physical `'` key.
76    Quote,
77    // B row
78    /// Physical `\` key.
79    Backslash,
80    /// ISO-102 extra key.
81    IntlBackslash,
82    // A row
83    /// Physical `,` key.
84    Comma,
85    /// Physical `.` key.
86    Dot,
87    /// Physical `/` key.
88    Slash,
89
90    /// A physical scancode key.
91    Scancode(u16),
92
93    #[default]
94    Unknown,
95}
96impl ConstInit for KeyFfi {
97    const INIT: Self = Self::Unknown;
98}

103        use {Key as K, KeyFfi as F};
104        match self {
105            // common control
106            K::Backspace => F::Backspace,

109            K::Escape => F::Escape,

111            // navigation
112            K::Left => F::Left,
113            K::Right => F::Right,
114            K::Up => F::Up,
115            K::Down => F::Down,
116            K::Home => F::Home,
117            K::End => F::End,
118            K::PageUp => F::PageUp,

144            K::F => F::F,
