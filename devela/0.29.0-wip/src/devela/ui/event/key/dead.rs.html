82impl ConstInit for KeyDead {
83    const INIT: Self = KeyDead::Unknown;
84}
85
86/// Aliases.
87#[allow(non_upper_case_globals)]
88impl KeyDead {
89    pub const Perispomeni: Self = Self::Tilde;
90    pub const Psili: Self = Self::CommaAbove;
91    pub const Dasia: Self = Self::ReversedCommaAbove;
92}
93
94impl KeyDead {
95    // Valid dead-key range: FE50 ..= FE93  (offset 0 ..= 67)
96    const BASE: u32 = 0xFE50;
97
98    /// Converts a dead-key keysym into a `KeyDead`.
99    pub const fn from_keysym(sym: u32) -> KeyDead {
100        if sym.wrapping_sub(Self::BASE) < DEAD_TO_SYM.len() as u32 {
101            DEAD_TO_SYM[(sym - Self::BASE) as usize] // safe: offset in [0..67]
102        } else {
103            KeyDead::Unknown
104        }
105    }
106
107    /// Returns the canonical dead-key keysym for this variant.
108    #[inline(always)]
109    pub const fn to_keysym(kind: KeyDead) -> u32 {
110        match kind {
111            KeyDead::Unknown => 0,
112            _ => 0xFE50 + (kind as u8 as u32),
113        }
114    }
115}
