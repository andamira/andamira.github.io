11/// Identifies what an [`Event`][crate::Event] is conceptually directed to.
12#[doc = crate::_doc_meta!{location("ui/event")}]
13///
14/// - Some backends (X11, Wayland, Win32, macOS) generate window–scoped events.
15///   These are represented as [`EventTarget::Window`], carrying a window
16///   identifier meaningful to the backend.
17///
18/// - Device–level sources (gamepads, MIDI devices, libinput devices, sensors)
19///   do not report a window association. These use [`EventTarget::Device`] with
20///   a device identifier meaningful to the backend or driver.
21///
22/// - Some events conceptually apply to the whole application or have no
23///   specific target (timers, system notifications, compositor signals).
24///   These use [`EventTarget::Global`].
25///
26/// Backends are free to choose their own identifier space, but values must be
27/// stable enough that the caller can reliably distinguish sources over time.
28///
29/// This allows a unified cross-backend event pipeline without leaking backend-
30/// specific handle types into the UI layer.
31#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
32pub enum EventTarget {
33    /// Event originated from a specific window or surface.
34    ///
35    /// The `u32` is a backend-defined identifier (e.g. an X11 `Window`,
36    /// a Wayland `wl_surface` ID, a Win32 `HWND` translated to an integer,
37    /// or an application-generated index when using a custom windowing layer).
38    Window(WindowId),
39
40    /// Event originated from a hardware or virtual device.
41    ///
42    /// The `u32` identifies the device within its backend:
43    /// - libinput device ID,
44    /// - MIDI device ID,
45    /// - joystick/gamepad index,
46    /// - virtual device channel,
47    /// - etc.
48    Device(DeviceId),
49
50    /// Event is not associated with any specific window or device.
51    ///
52    /// Examples:
53    /// - timers,
54    /// - compositor-level notifications,
55    /// - global shortcuts,
56    /// - system–wide state changes.
57    #[default]
58    Global,
59}
60
61#[rustfmt::skip]
62impl EventTarget {
63    /// Returns the `WindowId` if there's some.
64    pub fn some_window(&self) -> Option<WindowId> {
65        match self { Self::Window(id) => Some(*id), _ => None }
66    }
67
68    /// Returns the `DeviceId` if there's some.
69    pub fn some_device(&self) -> Option<DeviceId> {
70        match self { Self::Device(id) => Some(*id), _ => None }
71    }
72}
