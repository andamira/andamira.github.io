37        /// # Stack resize.
38        // T, S: Bare
39        #[cfg_attr(nightly_doc, doc(cfg(feature = $cap)))]
40        impl<T: Default, const CAP: usize> Stack<T, CAP, $IDX, Bare> {
41            /// Converts the current stack to a different capacity
42            /// while preserving all existing elements.
43            ///
44            /// This method creates a new stack with the specified new capacity and moves the
45            /// current elements into it. The operation ensures that the new stack can accommodate
46            /// the number of elements currently held in the stack. It is designed to work with
47            /// both increases and decreases in capacity, as long as the new capacity can fit the
48            /// current number of elements.
49            ///
50            /// # Errors
51            /// Returns [`IndexOutOfBounds`]`(Some(NEW_CAP))` if `NEW_CAP < self.len()`,
52            #[doc = "if `CAP > `[`" $IDX "::MAX`]"]
53            /// or if `CAP > isize::MAX / size_of::<T>()`.
54            ///
55            /// # Examples
56            /// ```
57            #[doc = "# use devela::Stack" $IDX:camel ";"]
58            #[doc = "let s = Stack" $IDX:camel "::<_, 8>::from([1, 2, 3, 4]);"]
59            #[doc = "let less_cap: Stack" $IDX:camel "::<_, 4> = s.resize_default().unwrap();"]
60            /// assert_eq![s.as_slice(), less_cap.as_slice()];
61            #[doc = "let more_cap: Stack" $IDX:camel "::<_, 12> = s.resize_default().unwrap();"]
62            /// assert_eq![s.as_slice(), more_cap.as_slice()];
63            /// assert![s.resize_default::<2>().is_err()]; // too small
64            /// ```
65            pub fn resize_default<const NEW_CAP: usize>(self)
66                -> Result<Stack<T, NEW_CAP, $IDX, Bare>, IndexOutOfBounds> {
67                if NEW_CAP < (self.len() as usize) ||
68                    NEW_CAP > $IDX::MAX as usize ||
69                    NEW_CAP > isize::MAX as usize / size_of::<T>() {
70                    Err(IndexOutOfBounds(Some(NEW_CAP)))
71                } else {
72                    let old_arr: [T; CAP] = self.data.into_array();
73                    let mut new_arr =
74                        init_array![default [T; NEW_CAP], "safe_data", "unsafe_array"];
75                    for (i, item) in old_arr.into_iter().enumerate().take(NEW_CAP) {
76                        new_arr[i] = item;
77                    }
78                    Ok(Stack {
79                        data: Array::new(new_arr),
80                        len: self.len,
81                    })
82                }
83            }
84            /// Converts the current stack to a different capacity, dropping elements if needed.
85            ///
86            /// This method creates a new stack with the specified new capacity and moves the
87            /// current elements into it. The operation will drop any elements that can't fit
88            /// in the new capacity, starting with the first ones (from the front of the stack).
89            ///
90            /// # Examples
91            /// ```
92            #[doc = "# use devela::Stack" $IDX:camel ";"]
93            #[doc = "let s = Stack" $IDX:camel "::<_, 8>::from([1, 2, 3, 4]);"]
94            #[doc = "let less_cap: Stack" $IDX:camel "::<_, 4> = s.resize_default_truncate();"]
95            /// assert_eq![less_cap.as_slice(), s.as_slice()];
96            #[doc = "let more_cap: Stack" $IDX:camel "::<_, 12> = s.resize_default_truncate();"]
97            /// assert_eq![more_cap.as_slice(), s.as_slice()];
98            #[doc = "let drop_cap: Stack" $IDX:camel "::<_, 2> = s.resize_default_truncate();"]
99            /// assert_eq![drop_cap.as_slice(), &[3, 4]];
100            /// ```
101            pub fn resize_default_truncate<const NEW_CAP: usize>(self)
102                -> Stack<T, NEW_CAP, $IDX, Bare> {
103                let start_idx = if self.len() as usize > NEW_CAP {
104                    self.len() as usize - NEW_CAP
105                } else {
106                    0
107                };
108                let new_len = core::cmp::min(self.len(), NEW_CAP as $IDX);
109                let old_arr: [T; CAP] = self.data.into_array();
110                let mut new_arr = init_array![default [T; NEW_CAP], "safe_data", "unsafe_array"];
111                for (i, item) in old_arr.into_iter().enumerate().skip(start_idx).take(NEW_CAP) {
112                    new_arr[i - start_idx] = item;
113                }
114                Stack {
115                    data: Array::new(new_arr),
116                    len: new_len,
117                }
118            }
119        }

122        #[cfg(feature = "alloc")]
123        #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
124        #[cfg_attr(nightly_doc, doc(cfg(feature = $cap)))]
125        impl<T: Default, const CAP: usize> Stack<T, CAP, $IDX, Boxed> {
126            /// Converts the current stack to a different capacity while preserving all existing
127            /// elements.
128            ///
129            /// This method creates a new stack with the specified new capacity and moves the
130            /// current elements into it. The operation ensures that the new stack can accommodate
131            /// the number of elements currently held in the stack. It is designed to work with
132            /// both increases and decreases in capacity, as long as the new capacity can fit the
133            /// current number of elements.
134            ///
135            /// # Errors
136            /// Returns [`IndexOutOfBounds`]`(Some(NEW_CAP))` if `NEW_CAP < self.len()`,
137            #[doc = "if `CAP > `[`" $IDX "::MAX`]"]
138            /// or if `CAP > isize::MAX / size_of::<T>()`.
139            ///
140            /// # Examples
141            /// ```
142            #[doc = "# use devela::{Boxed, Stack" $IDX:camel "};"]
143            #[doc = "let s = Stack" $IDX:camel "::<_, 8, Boxed>::from([1, 2, 3, 4]);"]
144            #[doc = "let less_cap: Stack" $IDX:camel
145                "::<_, 4, Boxed> = s.clone().resize_default().unwrap();"]
146            /// assert_eq![s.as_slice(), less_cap.as_slice()];
147            #[doc = "let more_cap: Stack" $IDX:camel
148                "::<_, 12, Boxed> = s.clone().resize_default().unwrap();"]
149            /// assert_eq![s.as_slice(), more_cap.as_slice()];
150            /// assert![s.resize_default::<2>().is_err()]; // too small
151            /// ```
152            pub fn resize_default<const NEW_CAP: usize>(self)
153            -> Result<Stack<T, NEW_CAP, $IDX, Boxed>, IndexOutOfBounds> {
154                if NEW_CAP < (self.len() as usize) ||
155                    NEW_CAP > $IDX::MAX as usize ||
156                    NEW_CAP > isize::MAX as usize / size_of::<T>() {
157                    Err(IndexOutOfBounds(Some(NEW_CAP)))
158                } else {
159                    let old_arr = self.data.into_vec();
160                    let mut v = Vec::with_capacity(NEW_CAP);
161                    for item in old_arr.into_iter().take(self.len as usize) {
162                        v.push(item);
163                    }
164                    v.resize_with(NEW_CAP, Default::default);
165
166                    let new_arr: Box<[T; NEW_CAP]> =
167                        v.into_boxed_slice().try_into().unwrap_or_else(|_| {
168                        panic!("Failed to convert Box<[T]> to Box<[T; NEW_CAP={}]>", NEW_CAP)
169                    });
170                    Ok(Stack {
171                        data: Array::new_boxed(new_arr),
172                        len: self.len,
173                    })
174                }
175            }
176            /// Converts the current stack to a different capacity
177            /// while preserving all existing elements.
178            ///
179            /// This method creates a new stack with the specified new capacity and moves the
180            /// current elements into it. The operation will drop any elements that can't fit
181            /// in the new capacity, starting with the first ones (from the front of the stack).
182            ///
183            /// # Examples
184            /// ```
185            #[doc = "# use devela::{Boxed, Stack" $IDX:camel "};"]
186            #[doc = "let s = Stack" $IDX:camel "::<_, 8, Boxed>::from([1, 2, 3, 4]);"]
187            #[doc = "let less_cap: Stack" $IDX:camel
188                "::<_, 4, Boxed> = s.clone().resize_default_truncate();"]
189            /// assert_eq![less_cap.as_slice(), s.as_slice()];
190            #[doc = "let more_cap: Stack" $IDX:camel
191                    "::<_, 12, Boxed> = s.clone().resize_default_truncate();"]
192            /// assert_eq![more_cap.as_slice(), s.as_slice()];
193            #[doc = "let drop_cap: Stack" $IDX:camel
194                    "::<_, 2, Boxed> = s.resize_default_truncate();"]
195            /// assert_eq![drop_cap.as_slice(), &[3, 4]];
196            /// ```
197            pub fn resize_default_truncate<const NEW_CAP: usize>(self)
198            -> Stack<T, NEW_CAP, $IDX, Boxed> {
199                let mut old_vec = self.data.into_vec();
200
201                // When reducing capacity, truncate elements from the front.
202                if NEW_CAP < self.len as usize {
203                    let excess = self.len as usize - NEW_CAP;
204                    old_vec.drain(0..excess);
205                }
206                // Ensure the vector's length matches NEW_CAP before conversion.
207                old_vec.resize_with(NEW_CAP, Default::default);
208
209                let new_arr: Box<[T; NEW_CAP]> =
210                    old_vec.into_boxed_slice().try_into().unwrap_or_else(|_| {
211                    panic!("Failed to convert Box<[T]> to Box<[T; NEW_CAP={}]>", NEW_CAP)
212                });
213                Stack {
214                    data: Array::new_boxed(new_arr),
215                    len: core::cmp::min(self.len as usize, NEW_CAP) as $IDX,
216                }
217            }
218        }

223            /// Converts the current stack to a different capacity
224            /// while preserving all existing elements.
225            ///
226            /// This method creates a new stack with the specified new capacity and moves the
227            /// current elements into it. The operation ensures that the new stack can accommodate
228            /// the number of elements currently held in the stack. It is designed to work with
229            /// both increases and decreases in capacity, as long as the new capacity can fit the
230            /// current number of elements.
231            ///
232            /// # Errors
233            /// Returns [`IndexOutOfBounds`]`(Some(NEW_CAP))` if `NEW_CAP < self.len()`,
234            #[doc = "if `CAP > `[`" $IDX "::MAX`]"]
235            /// or if `CAP > isize::MAX / size_of::<T>()`.
236            ///
237            /// # Examples
238            /// ```
239            #[doc = "# use devela::Stack" $IDX:camel ";"]
240            #[doc = "const S: Stack" $IDX:camel "<i32, 8> = Stack" $IDX:camel "::own_new(0)"]
241            ///     .s_const_unwrap().s.own_push(1).s.own_push(2).s.own_push(3).s;
242            #[doc = "const T: Stack" $IDX:camel
243                "<i32, 4> = S.own_resize_default().s_const_unwrap().s;"]
244            /// assert_eq![S.as_slice(), T.as_slice()];
245            /// let _ = S.own_resize_default::<2>().s_assert_err(); // too small
246            /// ```
247            pub const fn own_resize_default<const NEW_CAP: usize>(self)
248                -> Own<Result<Stack<T, NEW_CAP, $IDX, Bare>, IndexOutOfBounds>, ()> {
249                if NEW_CAP < (self.len as usize) ||
250                    NEW_CAP > $IDX::MAX as usize ||
251                    NEW_CAP > isize::MAX as usize / size_of::<T>() {
252                    Own::empty(Err(IndexOutOfBounds(Some(NEW_CAP))))
253                } else {
254                    let old_arr: [T; CAP] = self.data.into_array_copy();
255                    let mut new_arr = init_array![INIT in ConstInit [T; NEW_CAP]];
256
257                    let mut i = 0;
258                    while i < self.len as usize {
259                        new_arr[i] = old_arr[i];
260                        i += 1;
261                    }
262
263                    Own::empty(Ok(Stack {
264                        data: Array::new_bare(new_arr),
265                        len: self.len,
266                    }))
267                }
268            }
269            /// Converts the current stack to a different capacity, dropping elements if needed.
270            ///
271            /// This method creates a new stack with the specified new capacity and moves the
272            /// current elements into it. The operation will drop any elements that can't fit
273            /// in the new capacity, starting with the first ones (from the front of the stack).
274            ///
275            /// # Examples
276            /// ```
277            #[doc = "# use devela::Stack" $IDX:camel ";"]
278            #[doc = "const S: Stack" $IDX:camel "<i32, 8> = Stack" $IDX:camel "::own_new(0)"]
279            ///     .s_const_unwrap().s.own_push(1).s.own_push(2).s.own_push(3).s;
280            #[doc = "const T: Stack" $IDX:camel "<i32, 4> = S.own_resize_default_truncate().s;"]
281            /// assert_eq![S.as_slice(), T.as_slice()];
282            #[doc = "const U: Stack" $IDX:camel "<i32, 2> = S.own_resize_default_truncate().s;"]
283            /// assert_eq![U.as_slice(), &[2, 3]];
284            /// ```
285            pub const fn own_resize_default_truncate<const NEW_CAP: usize>(self)
286                -> Own<Stack<T, NEW_CAP, $IDX, Bare>, ()> {
287                let start_idx = if self.len as usize > NEW_CAP {
288                    self.len as usize - NEW_CAP
289                } else {
290                    0
291                };
292                let new_len = Cmp(NEW_CAP).min(self.len as usize);
293                let old_arr: [T; CAP] = self.data.into_array_copy();
294                let mut new_arr = init_array![INIT in ConstInit [T; NEW_CAP]];
295
296                let mut i = 0;
297                while i < new_len {
298                    new_arr[i] = old_arr[i + start_idx];
299                    i += 1;
300                }
301
302                Own::empty(Stack {
303                    data: Array::new_bare(new_arr),
304                    len: new_len as $IDX,
305                })
306            }
307        }
