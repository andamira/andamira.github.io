4/// Defines a ring buffer type over contiguous storage backends.
5#[doc = crate::_doc_meta!{location("data/layout")}]
6///
7/// The generated type represents a fixed-capacity
8/// circular logical range over contiguous storage.
9///
10/// A ring tracks:
11/// - `head`: the physical start of the logical range.
12/// - `len`: the number of live elements.
13///
14/// The tail is not stored. It is derived as `(head + len) % capacity`.
15///
16/// ## Implemented backends
17///
18/// Currently implemented:
19/// - **static `array`**
20///   Fully initialized array storage (`[T; CAP]`).
21///   All slots always contain a valid `T`; `len` controls which slots are
22///   logically visible.
23/// - **static `option`**
24///   Array of options (`[Option<T>; CAP]`).
25///   Occupied logical slots are `Some`; unused physical slots are `None`.
26///
27/// Reserved for later:
28/// - **static `uninit`**
29/// - **view backends**
30/// - **alloc backends**
31///
32/// ## Index type requirements
33///
34/// The index type must:
35/// - Be non-negative.
36/// - Represent zero.
37/// - Form a contiguous integer range.
38/// - Be able to represent the full capacity.
39///
40/// Primitive unsigned integers and supported niche wrappers are accepted
41/// through [`MaybeNiche`][crate::MaybeNiche].
42///
43///
44/// ## Storage backends
45///
46/// Backends are opt-in and selected after the struct declaration.
47///
48/// - **`array`**
49///   Fully initialized array storage (`[T; CAP]`).
50///
51///   This backend separates *initialization* from *logical membership*:
52///   every physical slot stores a valid `T`, while `len` determines which
53///   elements are visible through the ring API.
54///
55///   Operations that move values out of the array require either:
56///   - copying (`*_copy`),
57///   - an explicit replacement (`*_with`),
58///   - or a convenience replacement (`*_default`, `*_init`).
59///
60/// - **`option`**
61///   Array of options (`[Option<T>; CAP]`).
62///
63///   This backend stores occupancy explicitly: occupied logical slots are
64///   `Some(T)`, and unused physical slots are `None`.
65///
66///   It supports moving arbitrary `T` values in and out
67///   without requiring replacement values.
68///
69///
70/// ## Examples
71/// ```
72/// # use devela::buffer_ring;
73/// buffer_ring!(
74///     /// Static ring buffer.
75///     pub struct RingU8: (u8);
76///     array, option
77/// );
78///
79/// let mut array_ring = RingU8::<i32, [i32; 4]>::new_init();
80/// array_ring.push_back(10).unwrap();
81/// array_ring.push_back(20).unwrap();
82///
83/// assert_eq!(array_ring.pop_front_copy(), Some(10));
84/// assert_eq!(array_ring.peek_front(), Some(&20));
85///
86/// let mut option_ring = RingU8::<i32, [Option<i32>; 4]>::new();
87/// option_ring.push_back(10).unwrap();
88/// option_ring.push_back(20).unwrap();
89///
90/// assert_eq!(option_ring.pop_front(), Some(10));
91/// assert_eq!(option_ring.peek_front(), Some(&20));
92/// ```
93///
94/// See also:
95/// [`BufferRingStaticExample`][crate::BufferRingStaticExample],
96//
97// NOTE: The index type is passed as a token group to allow complex or path-qualified types.
98#[doc(hidden)]
99#[macro_export]
100macro_rules! buffer_ring {
101    (
102    // STATIC (option)
103    // struct definition + optional implementations
104
105        $(#[$attr:meta])*
106        $vis:vis struct $name:ident : $(static)? ($($I:tt)+);
107        $($rest:tt)*
108    ) => {
109        $(#[$attr])*
110        #[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
111        $vis struct $name<T, S> {
112            storage: S,
113            head: $crate::MaybeNiche<$($I)+>,
114            len: $crate::MaybeNiche<$($I)+>,
115            _m: $crate::PhantomData<T>,
116        }
117
118        $crate::buffer_ring!(%impl_common_static $name, $($I)+, $crate::niche_prim![$($I)+]);
119        $crate::buffer_ring!(%impls_static
120            $name : $($I)+, $crate::niche_prim![$($I)+] ; $($rest)*);
121    };
122    ( // implementations only
123        impl $name:ident : $(static)? ($($I:tt)+); $($rest:tt)*
124    ) => {
125        $crate::buffer_ring!(%impls_static
126            $name : $($I)+, $crate::niche_prim![$($I)+] ; $($rest)*);
127    };
128    (
129    // VIEW:  TODO: (slice_mut, slice)
130    // struct definition + optional implementations
131
132        $(#[$_attr:meta])*
133        $_vis:vis struct $_name:ident : view ($($_I:tt)+);
134        $($_rest:tt)*
135    ) => {
136        compile_error!("buffer_ring!: view backends are not implemented yet");
137    };
138    ( // implementations only
139        impl $_name:ident : view ($($_I:tt)+);
140        $($_rest:tt)*
141    ) => {
142        compile_error!("buffer_ring!: view backends are not implemented yet");
143    };
144
145    (
146    // ALLOC TODO: (vec)
147    // struct definition + optional implementations
148
149        $(#[$_attr:meta])*
150        $_vis:vis struct $_name:ident : alloc ($($_I:tt)+);
151        $($_rest:tt)*
152    ) => {
153        compile_error!("buffer_ring!: alloc backends are not implemented yet");
154    };
155    ( // implementations only
156        impl $_name:ident : alloc ($($_I:tt)+);
157        $($_rest:tt)*
158    ) => {
159        compile_error!("buffer_ring!: alloc backends are not implemented yet");
160    };
161
162    // --------------------------------------------------------------------------------------------
163    // static dispatch
164    (
165
166    /* internals */
167    %impls_static $name:ident : $I:ty, $P:ty ;) => {}; // no impls
168    (%impls_static $name:ident : $I:ty, $P:ty ; $(#[$i:meta])* $impl:ident) => { // last impl
169        $crate::buffer_ring!(%impl1_static $(#[$i])* $name : $I, $P ; $impl);
170    };
171    // static: array
172    (%impls_static $name:ident : $I:ty, $P:ty ; $(#[$i:meta])* array , $($rest:tt)*) => {
173        $crate::__buffer_ring_impl_array!($(#[$i])* $name, $I, $P);
174        $crate::buffer_ring!(%impls_static $name : $I, $P ; $($rest)*);
175    };
176    (%impl1_static $(#[$i:meta])* $name:ident : $I:ty, $P:ty ; array) => {
177        $crate::__buffer_ring_impl_array!($(#[$i])* $name, $I, $P);
178    };
179    // static: option
180    (%impls_static $name:ident : $I:ty, $P:ty ; $(#[$i:meta])* option , $($rest:tt)*) => {
181        $crate::__buffer_ring_impl_option!($(#[$i])* $name, $I, $P);
182        $crate::buffer_ring!(%impls_static $name : $I, $P ; $($rest)*);
183    };
184    (%impl1_static $(#[$i:meta])* $name:ident : $I:ty, $P:ty ; option) => {
185        $crate::__buffer_ring_impl_option!($(#[$i])* $name, $I, $P);
186    };
187
188    // safe-guards
189    (%impls_static $_name:ident : $_I:ty, $_P:ty ; $(#[$_i:meta])* $impl:ident , $($_r:tt)*) => {
190        compile_error!(concat!("buffer_ring!: unknown static impl `", stringify!($impl), "`"));
191    };
192    (%impl1_static $(#[$_i:meta])* $_name:ident : $_I:ty, $_P:ty ; $impl:ident) => {
193        compile_error!(concat!("buffer_ring!: unknown static impl `", stringify!($impl), "`"));
194    };
195
196    /* blocks for common private associated items */
197
198    (%impl_common_static $name:ident, $I:ty, $P:ty) => {
199        /// Common methods.
200        impl<T, S> $name<T, S> {
201            $crate::buffer_ring!(%common_tracked $name, $I, $P);
202        }
203    };
204    // common items for tracked rings
205    (%common_tracked $name:ident, $I:ty, $P:ty) => {
206        $crate::buffer_ring!(%guard_index_repr $I);
207
208        /// Constructs a ring from raw components, assuming all invariants hold.
209        #[inline(always)]
210        const fn _new(storage: S,
211            head: $crate::MaybeNiche<$I>, len: $crate::MaybeNiche<$I>) -> Self {
212            Self { storage, head, len, _m: $crate::PhantomData }
213        }
214
215        /* idx */
216
217        /// Returns the zero value as a MaybeNiche wrapped index type.
218        #[inline(always)]
219        const fn _idx_zero() -> $crate::MaybeNiche<$I> {
220            // SAFETY-INVARIANT: checked above; buffer indices must represent zero.
221            $crate::unwrap![some_guaranteed_or_ub $crate::MaybeNiche::<$I>::ZERO]
222        }
223
224        /// `a == b`
225        #[inline(always)]
226        const fn _idx_eq(a: $I, b: $I) -> bool {
227            let (a, b) = ($crate::MaybeNiche(a).prim(), $crate::MaybeNiche(b).prim()); a == b
228        }
229        /// `a <= b`
230        #[inline(always)]
231        const fn _idx_le(a: $I, b: $I) -> bool {
232            let (a, b) = ($crate::MaybeNiche(a).prim(), $crate::MaybeNiche(b).prim()); a <= b
233        }
234        /// `a >= b`
235        #[inline(always)]
236        const fn _idx_ge(a: $I, b: $I) -> bool {
237            let (a, b) = ($crate::MaybeNiche(a).prim(), $crate::MaybeNiche(b).prim()); a >= b
238        }
239
240        /* prim */
241
242        /// Returns the given index-typed value as a primitive.
243        #[inline(always)]
244        const fn _idx_to_prim(from: $I) -> $P { $crate::MaybeNiche(from).prim() }
245
246        /// Returns the given primitive value as an index type.
247        #[inline(always)]
248        const fn _prim_to_idx(from: $P) -> Result<$I, $crate::InvalidValue> {
249            $crate::unwrap![ok_map? $crate::MaybeNiche::<$I>::try_from_prim(from), |v| v.repr()]
250        }
251        /// Returns the given primitive value as an index type,
252        /// converting invalid inputs to the closest valid number.
253        #[inline(always)]
254        const fn _prim_to_idx_lossy(from: $P) -> $I {
255            $crate::MaybeNiche::<$I>::from_prim_lossy(from).repr()
256        }
257
258        /* usize */
259
260        /// The maximum representable value of the index type, as a usize.
261        const _IDX_MAX_USIZE: usize = $crate::MaybeNiche(<$I>::MAX).to_usize_saturating();
262
263        /// Returns the current logical length as a `usize`, saturating if necessary.
264        #[inline(always)]
265        const fn _len_usize(&self) -> usize { self.len.to_usize_saturating()
266        }
267        /// Returns the current physical head as a `usize`, saturating if necessary.
268        #[inline(always)]
269        const fn _head_usize(&self) -> usize { self.head.to_usize_saturating() }
270
271        /// Returns the given usize value as a MaybeNiche wrapped saturated index type.
272        #[inline(always)]
273        const fn _usize_to_idx_sat(from: usize) -> $crate::MaybeNiche<$I> {
274            $crate::MaybeNiche::<$I>::from_usize_saturating(from)
275        }
276
277        /// Returns the given usize value as a MaybeNiche wrapped index type.
278        #[inline(always)]
279        const fn _usize_to_idx(from: usize) -> $crate::MaybeNiche<$I> {
280            $crate::unwrap![ok $crate::MaybeNiche::<$I>::try_from_usize(from)]
281        }
282        /// Returns the given index value as a usize.
283        #[inline(always)]
284        const fn _idx_to_usize(from: $I) -> usize {
285            $crate::unwrap![ok $crate::MaybeNiche(from).try_to_usize()]
286        }
287
288        /* state */
289
290        /// Sets the physical head without checking invariants.
291        #[inline(always)]
292        const fn _set_head(&mut self, head: $I) { self.head = $crate::MaybeNiche(head); }
293
294        /// Sets the logical length without checking invariants.
295        #[inline(always)]
296        const fn _set_len(&mut self, len: $I) { self.len = $crate::MaybeNiche(len); }
297
298        /// Returns the next logical length.
299        ///
300        /// Caller must guarantee `len < capacity`.
301        #[inline(always)]
302        const fn _len_inc(&self) -> $crate::MaybeNiche<$I> {
303            $crate::unwrap![ok $crate::MaybeNiche::<$I>::try_from_prim(self.len.prim() + 1)]
304        }
305        /// Returns the previous logical length.
306        ///
307        /// Caller must guarantee `len > 0`.
308        #[inline(always)]
309        const fn _len_dec(&self) -> $crate::MaybeNiche<$I> {
310            $crate::unwrap![ok $crate::MaybeNiche::<$I>::try_from_prim(self.len.prim() - 1)]
311        }
312
313        /* public methods */
314
315        /// Returns the number of elements currently stored in the ring.
316        pub const fn len(&self) -> $I { self.len.repr() }
317
318        /// Returns the number of elements currently stored in the ring.
319        pub const fn len_prim(&self) -> $P { self.len.prim() }
320
321        /// Returns `true` if the ring contains no elements.
322        pub const fn is_empty(&self) -> bool { self.len.prim() == 0 }
323    };
324
325    // common items for static rings
326    (%common_static $name:ident, $I:ty, $P:ty) => {
327        const _CHECK_INVARIANTS: () = {
328            assert!(!$crate::MaybeNiche::<$I>::HAS_NEGATIVE,
329                "buffer_ring! index type must be non-negative");
330            assert!($crate::MaybeNiche::<$I>::ZERO.is_some(),
331                "buffer_ring! index type cannot represent zero");
332            assert!($crate::MaybeNiche::<$I>::IS_CONTIGUOUS,
333                "buffer_ring! index type must be contiguous");
334            assert!($crate::MaybeNiche::<$I>::try_from_usize(CAP).is_ok(),
335                "buffer_ring! capacity does not fit in index type");
336        };
337
338        /// The fixed capacity of the ring as the index type.
339        pub const CAP: $I = {
340            let _ = Self::_CHECK_INVARIANTS; // ensure proper eval order
341            Self::_usize_to_idx(CAP).repr()
342        };
343        /// The fixed capacity of the ring as the primitive type.
344        pub const CAP_PRIM: $P = Self::_idx_to_prim(Self::CAP);
345
346        /// Returns the fixed capacity of the ring.
347        pub const fn capacity(&self) -> $I { Self::CAP }
348        /// Returns the fixed capacity of the ring.
349        pub const fn capacity_prim(&self) -> $P { Self::CAP_PRIM }
350
351        /// Returns the number of additional elements that fit within the current capacity.
352        pub const fn remaining_capacity(&self) -> $I {
353            $crate::unwrap![ok_guaranteed_or_ub Self::_prim_to_idx(self.remaining_capacity_prim())]
354        }
355        /// Returns the number of additional elements that fit within the current capacity.
356        pub const fn remaining_capacity_prim(&self) -> $P { self.capacity_prim() - self.len_prim() }
357
358        /// Returns `true` if the ring has reached its capacity.
359        pub const fn is_full(&self) -> bool { Self::_idx_eq(self.len(), self.capacity()) }
360
361        /* internal methods */
362
363        /// Wraps a physical index into the fixed ring capacity.
364        ///
365        /// Caller should only pass values smaller than `2 * CAP`.
366        #[inline(always)]
367        const fn _wrap_usize(index: usize) -> usize {
368            if CAP == 0 { 0 } else if index >= CAP { index - CAP } else { index }
369        }
370
371        /// Returns the physical index for a logical index.
372        ///
373        /// Caller must guarantee `logical < len`.
374        #[inline(always)]
375        const fn _physical_usize(&self, logical: usize) -> usize {
376            Self::_wrap_usize(self._head_usize() + logical)
377        }
378        /// Returns the physical insertion index at the back.
379        ///
380        /// This is the derived tail.
381        #[inline(always)]
382        const fn _tail_usize(&self) -> usize {
383            Self::_wrap_usize(self._head_usize() + self._len_usize())
384        }
385        /// Returns the physical index of the current back element.
386        ///
387        /// Caller must guarantee `len > 0`.
388        #[inline(always)]
389        const fn _back_usize(&self) -> usize {
390            Self::_wrap_usize(self._head_usize() + self._len_usize() - 1)
391        }
392        /// Returns the physical index before the current head.
393        ///
394        /// Caller must guarantee `CAP > 0`.
395        #[inline(always)]
396        const fn _prev_head_usize(&self) -> usize {
397            let head = self._head_usize();
398            if head == 0 { CAP - 1 } else { head - 1 }
399        }
400    };
401    // only allow implementations over unsigned integers of size <= pointer-width
402    (%guard_index_repr $I:ty) => {
403        const __GUARD_INDEX_REPR: () = {
404            const fn __index_repr<I: $crate::IndexRepr>() {}
405            __index_repr::<$I>();
406        };
407    };
408}
