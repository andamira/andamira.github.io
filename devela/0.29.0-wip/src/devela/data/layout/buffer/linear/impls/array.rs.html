4#[macro_export]
5macro_rules! __buffer_linear_impl_array {
6    ($(#[$impl_attr:meta])* $name:ident, $I:ty, $P:ty) => {
7        $(#[$impl_attr])*
8        ///
9        /// Fully initialized linear array.
10        ///
11        /// # Invariants
12        /// - All CAP slots always contain a valid `T`.
13        /// - `len` controls logical membership, not initialization.
14        /// - Dropping the array drops all `CAP` elements.
15        ///
16        /// Consequences
17        /// - Cannot move out `T` safely.
18        /// - Popping by value needs `Copy` or `Clone`.
19        /// - Shrinking `len` does not affect drop behavior.
20        #[rustfmt::skip]
21        impl<T, const CAP: usize> $name<T, [T; CAP]> {
22            /* construct */
23
24            /// Creates a buffer from an already initialized array, with logical length 0.
25            pub const fn new(array: [T; CAP]) -> Self {
26                Self::_new(array, Self::_idx_zero())
27            }
28            /// Creates a new fully initialized buffer with logical length 0.
29            pub const fn new_init() -> Self where T: $crate::ConstInit {
30                Self::_new([T::INIT; CAP], Self::_idx_zero())
31            }
32
33            /// Creates a buffer from an already initialized array,
34            /// limiting the logical length to `max_len`.
35            pub fn from_array_clamped(array: [T; CAP], max_len: $I) -> Self {
36                let a = $crate::MaybeNiche(max_len).prim();
37                let b = $crate::MaybeNiche(Self::CAP).prim();
38                let min = $crate::Cmp(a).min(b);
39                // SAFETY: both are already validated
40                let len = $crate::unwrap![ok_guaranteed_or_ub
41                    $crate::MaybeNiche::<$I>::try_from_prim(min)];
42                Self::_new(array, len)
43            }
44            /// Primitive-index variant of [`from_array_clamped`][Self::from_array_clamped].
45            #[inline(always)]
46            pub fn from_array_clamped_prim(array: [T; CAP], max_len: $P) -> Self {
47                Self::from_array_clamped(array, Self::_prim_to_idx_lossy(max_len))
48            }
49
50            /// Creates a new buffer by cloning all the possible elements from `src`,
51            /// after initializing the capacity with the `init` value.
52            pub fn from_slice_clone(src: &[T], init: T) -> Option<Self> where T: Clone {
53                if src.len() > CAP { return None; }
54                let mut storage = $crate::array_from_fn(|_| init.clone());
55                $crate::whilst! { i in 0..src.len(); { storage[i] = src[i].clone(); }}
56                Some(Self::_new(storage, Self::_usize_to_idx(src.len())))
57            }
58
59            /// Creates a new buffer by copying all the possible elements from `src`,
60            /// after initializing the capacity with the `init` value.
61            pub const fn from_slice_copy(src: &[T], init: T) -> Option<Self> where T: Copy {
62                if src.len() > CAP { return None; }
63                let mut storage = [init; CAP];
64                $crate::whilst! { i in 0..src.len(); { storage[i] = src[i]; }}
65                Some(Self::_new(storage, Self::_usize_to_idx(src.len())))
66            }
67
68            /// Creates a new buffer by moving all the possible elements from `src`,
69            /// and replacing them with the default value,
70            /// after initializing the capacity with the default value.
71            pub fn from_slice_move_default(src: &mut [T]) -> Option<Self> where T: Default {
72                if src.len() > CAP { return None; }
73                let mut storage = $crate::array_from_fn(|_| T::default());
74                $crate::whilst! { i in 0..src.len(); {
75                    storage[i] = $crate::Mem::take(&mut src[i]);
76                }}
77                Some(Self::_new(storage, Self::_usize_to_idx(src.len())))
78            }
79
80            /* size & capacity */
81
82            $crate::buffer_linear!(%common_static $name, $I, $P);
83
84            /* logical range control */
85
86            /// Sets the logical length to zero.
87            ///
88            /// Does not drop elements.
89            pub const fn clear(&mut self) { self.len = Self::_idx_zero(); }
90
91            /// Sets the logical length to `min(new_len, len)`.
92            ///
93            /// If `new_len >= len`, this is a no-op.
94            pub const fn truncate(&mut self, new_len: $I) {
95                if Self::_idx_le(new_len, self.len()) { self._set_len(new_len); }
96            }
97            /// Primitive-index variant of [`truncate`][Self::truncate],
98            #[inline(always)]
99            pub const fn truncate_prim(&mut self, new_len: $P) -> Result<(), $crate::InvalidValue> {
100                self.truncate($crate::unwrap![ok? Self::_prim_to_idx(new_len)]);
101                Ok(())
102            }
103
104            /* push */
105
106            /// Appends a value to the back of the buffer.
107            ///
108            /// Returns `Err(value)` if the buffer is full.
109            pub fn push_back(&mut self, value: T) -> Result<(), T> {
110                if self.is_full() { return Err(value); }
111                self.storage[self._len_usize()] = value;
112                self.len = self._len_inc();
113                Ok(())
114            }
115            /// Appends a copy of `value` to the back of the buffer.
116            ///
117            /// Returns `Err(value)` if the buffer is full.
118            pub const fn push_back_copy(&mut self, value: T) -> Result<(), T> where T: Copy {
119                if self.is_full() { return Err(value); }
120                self.storage[self._len_usize()] = value;
121                self.len = self._len_inc();
122                Ok(())
123            }
124
125            /// Appends as many cloned elements from `src` as fit.
126            ///
127            /// Returns the number of elements appended.
128            pub fn push_slice(&mut self, src: &[T]) -> usize where T: Clone {
129                let len = self._len_usize();
130                let count = usize::min(src.len(), CAP - len);
131                self.storage[len..len + count].clone_from_slice(&src[..count]);
132                self.len = Self::_usize_to_idx(len + count);
133                count
134            }
135            /// Appends as many copied elements from `src` as fit.
136            ///
137            /// Returns the number of elements appended.
138            pub const fn push_slice_copy(&mut self, src: &[T]) -> usize where T: Copy {
139                let len = self._len_usize();
140                let count = $crate::cmp!(min src.len(), CAP - len);
141                let dst_slice = $crate::Slice::range_mut(&mut self.storage, len, len + count);
142                let src_slice = $crate::Slice::range_to(&src, count);
143                dst_slice.copy_from_slice(src_slice);
144                self.len = Self::_usize_to_idx(len + count);
145                count
146            }
147            /// Appends all copied elements from `src`,
148            /// or returns the number appended before running out of space.
149            pub const fn push_slice_copy_exact(&mut self, src: &[T]) -> Result<(), usize>
150            where T: Copy {
151                let rem = CAP - self._len_usize();
152                if src.len() > rem { return Err(rem); }
153                let _ = self.push_slice_copy(src);
154                Ok(())
155            }
156
157            /* pop */
158
159            /// Removes and returns a cloned value from the back of the buffer.
160            pub fn pop_back_clone(&mut self) -> Option<T> where T: Clone {
161                if self.is_empty() { return None; }
162                self.len = self._len_dec();
163                Some(self.storage[self._len_usize()].clone())
164            }
165            /// Removes and returns a copied value from the back of the buffer.
166            pub const fn pop_back_copy(&mut self) -> Option<T> where T: Copy {
167                if self.is_empty() { return None; }
168                self.len = self._len_dec();
169                Some(self.storage[self._len_usize()])
170            }
171
172            /* peek */
173
174            /// Returns a shared reference to the last element without removing it.
175            pub const fn peek_back(&self) -> Option<&T> {
176                if self.is_empty() { return None; }
177                Some(&self.storage[self._len_dec().to_usize_saturating()])
178            }
179            /// Returns an exclusive reference to the last element without removing it.
180            pub const fn peek_mut_back(&mut self) -> Option<&mut T> {
181                if self.is_empty() { return None; }
182                Some(&mut self.storage[self._len_dec().to_usize_saturating()])
183            }
184
185            /* get */
186
187            /// Returns a shared reference to the element at `index`, or `None` if out of bounds.
188            pub const fn get(&self, index: $I) -> Option<&T> {
189                if Self::_idx_ge(index, self.len()) { return None; }
190                Some(&self.storage[Self::_idx_to_usize(index)])
191            }
192
193            /// Returns an exclusive reference to the element at `index`,
194            /// or `None` if out of bounds.
195            pub const fn get_mut(&mut self, index: $I) -> Option<&mut T> {
196                if Self::_idx_ge(index, self.len()) { return None; }
197                Some(&mut self.storage[Self::_idx_to_usize(index)])
198            }
199
200            /* take */
201
202            /// Takes the value at `index`, replacing it with `T::default()`.
203            pub fn take_default(&mut self, index: $I) -> Option<T>
204            where T: Default {
205                if index >= self.len() { return None; }
206                let index_usize = Self::_idx_to_usize(index);
207                Some($crate::Mem::replace(&mut self.storage[index_usize], T::default()))
208            }
209            /// Takes the value at `index`, replacing it with `T::INIT`.
210            pub const fn take_init(&mut self, index: $I) -> Option<T>
211            where T: $crate::ConstInit {
212                if Self::_idx_ge(index, self.len()) { return None; }
213                let index_usize = Self::_idx_to_usize(index);
214                Some($crate::Mem::replace(&mut self.storage[index_usize], T::INIT))
215            }
216            /// Takes the value at `index`, replacing it with `other`.
217            pub fn take_with(&mut self, index: $I, other: T) -> Option<T> {
218                if Self::_idx_ge(index, self.len()) { return None; }
219                let index_usize = Self::_idx_to_usize(index);
220                Some($crate::Mem::replace(&mut self.storage[index_usize], other))
221            }
222            /// Takes the value at `index`, replacing it with a copy of `other`.
223            pub const fn take_with_copy(&mut self, index: $I, other: T) -> Option<T>
224            where T: Copy {
225                if Self::_idx_ge(index, self.len()) { return None; }
226                let index_usize = Self::_idx_to_usize(index);
227                Some($crate::Mem::replace(&mut self.storage[index_usize], other))
228            }
229
230            /* views */
231
232            /// Returns the active logical range as a slice.
233            pub const fn as_slice(&self) -> &[T] {
234                let len = self._len_usize(); $crate::Slice::range_to(&self.storage, len)
235            }
236            /// Returns the active logical range as an exclusive slice.
237            pub const fn as_mut_slice(&mut self) -> &mut [T] {
238                let len = self._len_usize(); $crate::Slice::range_to_mut(&mut self.storage, len)
239            }
240
241            /* iteration & visitation */
242
243            $crate::buffer_linear!(%common_iter_visit $name, $I, $P);
244        }
245    };
