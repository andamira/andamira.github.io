3//! Defines [`buffer_linear!`].

6#[doc = crate::_tags!(construction data_structure)]
7/// Defines a linear buffer type over contiguous storage backends.
8#[doc = crate::_doc_meta!{location("data/layout")}]
9///
10/// The generated type represents a logical range over contiguous
11/// storage, using an index type to track length and bounds.
12///
13/// ## Ownership regimes
14///
15/// The macro supports three regimes:
16///
17/// - **static** (default)
18///   The buffer owns its fixed-capacity storage backend.
19///   The optional `static` specifier can be omitted.
20///
21/// - **View**
22///   The buffer is a non-owning view over externally provided storage.
23///   The `view` specifier is required.
24///
25/// - **Alloc**
26///   The buffer owns heap-allocated storage.
27///   The `alloc` specifier is required.
28///
29/// ## Index type requirements
30///
31/// The index type must:
32/// - Be non-negative
33/// - Represent zero
34/// - Form a contiguous integer range
35/// - Be able to represent all valid buffer positions
36///
37/// Primitive unsigned integers and supported niche wrappers are accepted
38/// (see [`MaybeNiche`][crate::MaybeNiche]).
39///
40/// ## Storage backends
41///
42/// Backends are opt-in and selected by listing them after the struct declaration.
43///
44/// ### Fixed-capacity backends
45///
46/// - **`array`**
47///   Fully initialized fixed-size array (`[T; CAP]`).
48///   Logical length is tracked separately from capacity.
49///
50/// - **`uninit`**
51///   Partially initialized array (`[MaybeUninit<T>; CAP]`).
52///   Logical length tracks initialized elements and controls drop.
53///
54/// - **`option`**
55///   Array of options (`[Option<T>; CAP]`).
56///   `Some` marks initialized elements; `None` marks unused slots.
57///
58/// ### View backends
59///
60/// - **`slice_mut`**
61///   Exclusive slice (`&mut [T]`).
62///   The slice provides backing storage and determines capacity.
63///
64/// - **`slice`**
65///   Shared slice (`&[T]`).
66///   Read-only view over a contiguous logical range.
67///
68/// ### Alloc backends
69///
70/// Requires the `alloc` crate to be linked:
71/// ```
72/// # #[cfg(feature = "__std")]
73/// extern crate alloc;
74/// ```
75///
76/// - **`vec`**
77///   Growable `Vec<T>` storage.
78///
79/// ---
80/// Only explicitly enabled backends generate implementations.
81/// Constructors and methods depend on the selected backends.
82///
83/// ## Examples
84///
85/// ### Static buffer (default)
86/// ```
87/// # use devela::buffer_linear;
88/// buffer_linear!(
89///     /// Static linear buffer.
90///     pub struct BufferU8: (u8); array
91/// );
92/// let mut buf = BufferU8::<i32, [i32; 8]>::new_init();
93/// buf.push_back(10).unwrap();
94/// buf.push_back(20).unwrap();
95/// assert_eq!(buf.as_slice(), &[10, 20]);
96/// ```
97///
98/// ### View buffer
99/// ```
100/// # use devela::buffer_linear;
101/// buffer_linear!(
102///     /// Read-only linear view.
103///     pub struct BufferViewU8: view (u8); slice
104/// );
105/// let storage = [1u8, 2, 3, 4];
106/// let buf = BufferViewU8::try_from_slice(&storage).unwrap();
107/// assert_eq!(buf.len_prim(), 4);
108/// assert_eq!(buf.peek_back(), Some(&4));
109/// ```
110///
111/// ### Alloc buffer
112/// ```
113/// # #[cfg(feature = "__std")] extern crate alloc;
114/// # #[cfg(feature = "__std")] {
115/// # use devela::buffer_linear;
116/// buffer_linear!(
117///     /// Dynamic linear buffer.
118///     pub struct BufferU8: alloc (u8); vec
119/// );
120/// let mut buf = BufferU8::new();
121/// buf.push_back(10);
122/// buf.push_back(20);
123/// assert_eq!(buf.as_slice(), &[10, 20]);
124/// # }
125/// ```
126///
127/// ### Separate implementation blocks
128/// ```
129/// # use devela::buffer_linear;
130/// buffer_linear!(pub struct BufferSplit: view (u8););
131/// buffer_linear!(impl BufferSplit: view (u8); slice, slice_mut);
132/// ```
133/// See also:
134/// [`BufferLinearStaticExample`][crate::BufferLinearStaticExample],
135/// [`BufferLinearViewExample`][crate::BufferLinearViewExample],
136/// [`BufferLinearAllocExample`][crate::BufferLinearAllocExample].
137//

305    (%common_tracked $name:ident, $I:ty, $P:ty) => {
306        $crate::buffer_linear!(%guard_index_repr $I);
307
308        /// Constructs a buffer from raw components, assuming all invariants hold.
309        #[inline(always)]
310        const fn _new(storage: S, len: $crate::MaybeNiche<$I>) -> Self {
311            Self { storage, len, _m: $crate::PhantomData }
312        }
313
314        /* idx */
315
316        /// Returns the zero value as a MaybeNiche wrapped index type.
317        // It should not panic since we've already checked the invariants.
318        #[inline(always)]
319        const fn _idx_zero() -> $crate::MaybeNiche<$I> {
320            // SAFETY-INVARIANT: checked above; buffer indices must represent zero.
321            $crate::unwrap![some_guaranteed_or_ub $crate::MaybeNiche::<$I>::ZERO]
322        }
323
324        /// `a == b`
325        #[inline(always)]
326        const fn _idx_eq(a: $I, b: $I) -> bool {
327            let (a, b) = ($crate::MaybeNiche(a).prim(), $crate::MaybeNiche(b).prim()); a == b
328        }
329        /// `a <= b`
330        #[inline(always)]
331        const fn _idx_le(a: $I, b: $I) -> bool {
332            let (a, b) = ($crate::MaybeNiche(a).prim(), $crate::MaybeNiche(b).prim()); a <= b
333        }
334        /// `a >= b`
335        #[inline(always)]
336        const fn _idx_ge(a: $I, b: $I) -> bool {
337            let (a, b) = ($crate::MaybeNiche(a).prim(), $crate::MaybeNiche(b).prim()); a >= b
338        }
339
340        /* prim */
341
342        /// Returns the given index-typed value as a primitive.
343        #[inline(always)]
344        const fn _idx_to_prim(from: $I) -> $P { $crate::MaybeNiche(from).prim() }
345
346        /// Returns the given primitive value as an index type.
347        #[inline(always)]
348        // IMPROVE:MAYBE make an unchecked version leveraging unsafe_niche.
349        const fn _prim_to_idx(from: $P) -> Result<$I, $crate::InvalidValue> {
350            $crate::unwrap![ok_map? $crate::MaybeNiche::<$I>::try_from_prim(from), |v| v.repr()]
351        }
352        /// Returns the given primitive value as an index type,
353        /// converting invalid inputs to the closest valid number.
354        #[inline(always)]
355        const fn _prim_to_idx_lossy(from: $P) -> $I {
356            $crate::MaybeNiche::<$I>::from_prim_lossy(from).repr()
357        }
358
359        /* usize */
360
361        /// The maximum representable value of the index type, as a usize.
362        const _IDX_MAX_USIZE: usize = $crate::MaybeNiche(<$I>::MAX).to_usize_saturating();
363
364        /// Returns the current logical length as a `usize`, saturating if necessary.
365        #[inline(always)]
366        const fn _len_usize(&self) -> usize { self.len.to_usize_saturating() }
367
368        /// Returns the given usize value as a MaybeNiche wrapped saturated index type.
369        #[inline(always)]
370        const fn _usize_to_idx_sat(from: usize) -> $crate::MaybeNiche<$I> {
371            $crate::MaybeNiche::<$I>::from_usize_saturating(from)
372        }
373        /// Returns the given usize value as a MaybeNiche wrapped index type.
374        // It should not panic since we've already checked the invariants.
375        #[inline(always)]
376        const fn _usize_to_idx(from: usize) -> $crate::MaybeNiche<$I> {
377            $crate::unwrap![ok $crate::MaybeNiche::<$I>::try_from_usize(from)]
378        }
379        /// Returns the given index value as a usize.
380        // It should not panic unless we're using 128-bit values!
381        #[inline(always)]
382        const fn _idx_to_usize(from: $I) -> usize {
383            $crate::unwrap![ok $crate::MaybeNiche(from).try_to_usize()]
384        }
385
386        /* len */
387
388        /// Sets the logical length without checking invariants.
389        #[inline(always)]
390        const fn _set_len(&mut self, len: $I) {
391            self.len = $crate::MaybeNiche(len);
392        }
393        /// Returns the next logical length (len + 1).
394        ///
395        /// Caller must guarantee `len < capacity`.
396        #[inline(always)]
397        const fn _len_inc(&self) -> $crate::MaybeNiche<$I> {
398            $crate::unwrap![ok $crate::MaybeNiche::<$I>::try_from_prim(self.len.prim() + 1)]
399        }
400        /// Returns the previous logical length (len - 1).
401        ///
402        /// Caller must guarantee `len > 0`.
403        #[inline(always)]
404        const fn _len_dec(&self) -> $crate::MaybeNiche<$I> {
405            $crate::unwrap![ok $crate::MaybeNiche::<$I>::try_from_prim(self.len.prim() - 1)]
406        }

413        pub const fn len_prim(&self) -> $P { self.len.prim() }
414        /// Returns `true` if the buffer contains no elements.
415        pub const fn is_empty(&self) -> bool { self.len.prim() == 0 }

461    (%common_static $name:ident, $I:ty, $P:ty) => {
462        const _CHECK_INVARIANTS: () = {
463            assert!(!$crate::MaybeNiche::<$I>::HAS_NEGATIVE,
464                "buffer_linear! index type must be non-negative");
465            assert!($crate::MaybeNiche::<$I>::ZERO.is_some(),

467            assert!($crate::MaybeNiche::<$I>::IS_CONTIGUOUS,

469            assert!($crate::MaybeNiche::<$I>::try_from_usize(CAP).is_ok(),
470                "buffer_linear! capacity does not fit in index type");
471        };

473        /// The fixed capacity of the buffer as the index type.
474        pub const CAP: $I = {
475            let _ = Self::_CHECK_INVARIANTS; // ensure proper eval order
476            Self::_usize_to_idx(CAP).repr()
477        };

483        /// Returns the fixed capacity of the buffer.
484        pub const fn capacity_prim(&self) -> $P { Self::CAP_PRIM }
485        /// Returns the number of additional elements that fit within the current capacity.
486        pub const fn remaining_capacity(&self) -> $I {
487            $crate::unwrap![ok_guaranteed_or_ub Self::_prim_to_idx(self.remaining_capacity_prim())]
488        }
489        /// Returns the number of additional elements that fit within the current capacity.
490        pub const fn remaining_capacity_prim(&self) -> $P {
491            self.capacity_prim() - self.len_prim()
492        }
493        /// Returns `true` if the buffer has reached its capacity.
494        pub const fn is_full(&self) -> bool { Self::_idx_eq(self.len(), self.capacity()) }
495    };

513        /// Returns the number of additional elements that fit within the current capacity.
514        pub const fn remaining_capacity(&self) -> $I {
515            $crate::unwrap![ok_guaranteed_or_ub Self::_prim_to_idx(self.remaining_capacity_prim())]
516        }
517        /// Returns the number of additional elements that fit within the current capacity.
518        pub const fn remaining_capacity_prim(&self) -> $P {
519            self.capacity_prim() - self.len_prim()
520        }
521        /// Returns `true` if the buffer has reached its capacity.
522        pub const fn is_full(&self) -> bool { Self::_idx_eq(self.len(), self.capacity()) }
523    };
524    // common items for variants: array, uninit, slice_mut & vec.
525    (%common_iter_visit $name:ident, $I:ty, $P:ty) => {
526        /* iteration */
527
528        /// Iterates over the initialized elements.
529        pub fn iter(&self) -> impl Iterator<Item = &T> {
530            self.as_slice().iter()
531        }
532        /// Iterates mutably over the initialized elements.
533        pub fn iter_mut(&mut self) -> impl Iterator<Item = &mut T> {
534            self.as_mut_slice().iter_mut()
535        }
536
537        /* visitation */
538
539        /// Visits each initialized element without exposing borrow identity.
540        pub fn visit_each<F>(&self, f: F) where for<'v> F: Fn(&'v T) {
541            for x in self.as_slice() { f(x); }
542        }
543        /// Visits each initialized element mutably without exposing borrow identity.
544        pub fn visit_each_mut<F>(&mut self, f: F) where for<'v> F: Fn(&'v mut T) {
545            for x in self.as_mut_slice() { f(x); }
546        }
547
548        /// Visits the active logical range as a shared slice of `Some(T)`,
549        /// without exposing borrow identity.
550        pub fn visit_slice<F, R>(&self, f: F)
551            -> R where for<'v> F: FnOnce(&'v [T]) -> R { f(self.as_slice())
552        }
553        /// Visits the active logical range as an exclusive slice of `Some(T)`,
554        /// without exposing borrow identity.
555        pub fn visit_mut_slice<F, R>(&mut self, f: F)
556            -> R where for<'v> F: FnOnce(&'v mut [T]) -> R { f(self.as_mut_slice())
557        }
558    };
559    // only allow implementions over unsigned integers of size <= pointer-width
560    (%guard_index_repr $I:ty) => {
