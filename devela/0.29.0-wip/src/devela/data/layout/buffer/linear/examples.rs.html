306#[cfg(test)]
307mod tests {
308    use super::*;
309
310    #[test]
311    fn array() {
312        let mut buf = BufferLinearStaticExample::<i32, [i32; 8]>::new_init();
313        buf.push_back(10).unwrap();
314        buf.push_back(20).unwrap();
315        assert_eq!(buf.as_slice(), &[10, 20]);
316    }
317    #[test]
318    fn option() {
319        let mut buf = BufferLinearStaticExample::<i32, [Option<i32>; 8]>::new();
320        buf.push_back(10).unwrap();
321        buf.push_back(20).unwrap();
322        assert_eq!(buf.as_slice(), &[Some(10), Some(20)]);
323    }
324    #[test]
325    #[cfg(all(not(feature = "safe_data"), feature = "unsafe_array"))]
326    fn uninit() {
327        let mut buf = BufferLinearStaticExample::<i32, [crate::MaybeUninit<i32>; 8]>::new();
328        buf.push_back(10).unwrap();
329        buf.push_back(20).unwrap();
330        assert_eq!(buf.as_slice(), &[10, 20]);
331    }
