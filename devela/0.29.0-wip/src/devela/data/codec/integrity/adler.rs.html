13#[doc = crate::_tags!(hash)]
14/// Adler-32 checksum.
15#[doc = crate::_doc_meta!{location("data/codec")}] // integrity
16///
17/// A lightweight checksum consisting of two 16-bit accumulators (`a` and `b`)
18/// updated modulo 65521, the largest prime below 2^16.
19/// The final checksum packs `b` in the high 16 bits and `a` in the low 16 bits.
20///
21/// Adler-32 is **byte-order sensitive**: the checksum reflects the exact
22/// sequence of input bytes. When used together with Rust's `Hash` trait,
23/// results may differ between little-endian and big-endian targets unless
24/// the input is serialized in a fixed byte order.
25///
26/// This type supports incremental updates via [`write_bytes`][Self::write_bytes],
27/// making it suitable for checksumming streaming data.
28///
29/// For details, see <https://en.wikipedia.org/wiki/Adler-32>.
30#[derive(Debug, Copy, Clone)]
31pub struct Adler32 {
32    /// Running sum of bytes.
33    a: u16,
34    /// Running sum of all previous a values.
35    b: u16,
36}
37
38#[rustfmt::skip]
39impl ConstInit for Adler32 { const INIT: Self = Self::new(); }
40#[rustfmt::skip]
41impl Default for Adler32 { fn default() -> Self { Self::new() } }
42#[rustfmt::skip]
43impl Hasher for Adler32 {
44    fn finish(&self) -> u64 { u64::from(self.checksum()) }
45    fn write(&mut self, bytes: &[u8]) { self.write_bytes(bytes); }
46}
47
48/// Compile-time friendly API.
49#[rustfmt::skip]
50impl Adler32 {
51    /* constructors */
52
53    /// Creates a new Adler-32 state with the initial value `1`.
54    ///
55    /// This corresponds to the standard Adler-32 initial state `(a=1, b=0)`.
56    pub const fn new() -> Self { Self { a: 1, b: 0 } }
57
58    /// Builds a checksum state from a previously computed Adler-32 value.
59    pub const fn from_checksum(sum: u32) -> Self {
60        Adler32 { a: sum as u16, b: (sum >> 16) as u16 }
61    }
62
63    /* */
64
65    /// Returns the current Adler-32 checksum as a packed `u32`.
66    pub const fn checksum(&self) -> u32 {
67        ((self.b as u32) << 16) | self.a as u32
68    }
69
70    /// Computes the Adler-32 checksum of a byte slice.
71    ///
72    /// This is a convenience wrapper that does not require
73    /// constructing an [`Adler32`] value explicitly.
74    pub const fn checksum_bytes(bytes: &[u8]) -> u32 {
75        let mut new = Adler32::new();
76        new.write_bytes(bytes);
77        new.checksum()
78    }
79
80    /// Updates the checksum with the given bytes.
81    ///
82    /// This method may be called multiple times to checksum data streams.
83    pub const fn write_bytes(&mut self, bytes: &[u8]) {
84        self.update_chunked(bytes);
85    }
86
87    /// Internal Adler-32 update routine.
88    ///
89    /// Implements the standard Adler-32 update step:
90    /// ```text
91    /// a = (a + data[i]) mod 65521
92    /// b = (b + a)        mod 65521
93    /// ```
94    /// using chunked reduction to avoid accumulator overflow.
95    const fn update_chunked(&mut self, bytes: &[u8]) {
96        const MOD: u32 = 65521;
97        const OUTER_CHUNK: usize = 5552 * INNER_CHUNK;
98        const INNER_CHUNK: usize = 4;
99        let mut a = self.a as u32;
100        let mut b = self.b as u32;
101        let mut a_vec = Lane4::<u32>([0; INNER_CHUNK]);
102        let mut b_vec = a_vec;
103
104        let (bytes, final_remainder) = bytes.split_at(bytes.len() - bytes.len() % INNER_CHUNK);
105
106        // vectorized outer processing (full OUTER_CHUNK blocks)
107        let (outer_count, remainder_outer_chunk) = Slice::chunks_exact::<OUTER_CHUNK>(bytes);
108        whilst! { i in 0..outer_count; {
109            let outer_chunk = unwrap![some Slice::chunk::<OUTER_CHUNK>(bytes, i)];
110
111            // vectorized inner accumulation over 4-byte groups
112            let (inner_count, _) = Slice::chunks_exact::<INNER_CHUNK>(outer_chunk);
113            whilst! { j in 0..inner_count; {
114                let byte_vec = Slice::chunk::<INNER_CHUNK>(outer_chunk, j).unwrap();
115                let val = Lane4::<u32>::from_byte_values(byte_vec);
116                a_vec.add_assign_plain(val);
117                b_vec.add_assign_plain(a_vec);
118            }}
119            b += OUTER_CHUNK as u32 * a;
120            a_vec.rem_scalar_assign_plain(MOD);
121            b_vec.rem_scalar_assign_plain(MOD);
122            b %= MOD;
123        }}
124        // vectorized remainder after the outer loop
125        let (inner_count, _) = Slice::chunks_exact::<INNER_CHUNK>(remainder_outer_chunk);
126        whilst! { i in 0..inner_count; {
127            let byte_vec = unwrap![some Slice::chunk::<4>(remainder_outer_chunk, i)];
128            let val = Lane4::<u32>::from_byte_values(byte_vec);
129            a_vec.add_assign_plain(val);
130            b_vec.add_assign_plain(a_vec);
131        }}
132
133        // combine vector lanes into scalar a and b
134        b += remainder_outer_chunk.len() as u32 * a;
135        a_vec.rem_scalar_assign_plain(MOD);
136        b_vec.rem_scalar_assign_plain(MOD);
137        b %= MOD;
138        b_vec.mul_scalar_assign_plain(4);
139        b_vec.0[1] += MOD - a_vec.0[1];
140        b_vec.0[2] += (MOD - a_vec.0[2]) * 2;
141        b_vec.0[3] += (MOD - a_vec.0[3]) * 3;
142        whilst! { i in 0..a_vec.0.len(); { a += a_vec.0[i]; }}
143        whilst! { i in 0..b_vec.0.len(); { b += b_vec.0[i]; }}
144
145        // final scalar remainder (bytes < 4)
146        whilst! { i in 0..final_remainder.len(); {
147            a += final_remainder[i] as u32;
148            b += a;
149        }}
150        // reduce to 16-bit result
151        self.a = (a % MOD) as u16;
152        self.b = (b % MOD) as u16;
153    }
154}

156#[cfg(test)]
157#[rustfmt::skip]
158mod tests {
159    use crate::{Adler32, Hash};
160
161    #[test]
162    fn checksum_bytes() {
163        assert_eq![0x00000001, Adler32::checksum_bytes(b"")];
164        assert_eq![0x00620062, Adler32::checksum_bytes(b"a")];
165        assert_eq![0x024D0127, Adler32::checksum_bytes(b"abc")];
166        assert_eq![0x29750586, Adler32::checksum_bytes(b"message digest")];
167        assert_eq![0x64A607E0, Adler32::checksum_bytes(b"ABCDEFGHIJKLMNOPQRSTUVWXYZ")];
168        assert_eq![0x11E60398, Adler32::checksum_bytes(b"Wikipedia")];
169    }
170
171    #[test]
172    fn write_bytes() {
173        let full = Adler32::checksum_bytes(b"HelloWorld");
174        let mut st = Adler32::new();
175        st.write_bytes(b"Hello");
176        st.write_bytes(b"World");
177        assert_eq!(full, st.checksum());
178    }
179
180    #[test]
181    fn many_small_updates() {
182        let input = b"abcdefghij1234567890";
183        let mut a = Adler32::new();
184        a.write_bytes(input);
185        let full = a.checksum();
186        let mut b = Adler32::new();
187        for byte in input {
188            b.write_bytes(&[*byte]);
189        }
190        assert_eq!(full, b.checksum());
191    }
192
193    #[test]
194    fn hash_struct() {
195        #[derive(Hash)]
196        struct Packet { id: u16, flag: bool, value: u32, }
197        let pkt = Packet { id: 0x1234, flag: true, value: 0xCAFEBABE };
198        let mut ad = Adler32::new();
199        pkt.hash(&mut ad);
200        if cfg!(target_endian = "little") { assert_eq!(ad.checksum(), 0x09D00388); }
201        if cfg!(target_endian = "big") { assert_eq!(ad.checksum(), 0x0A160388); }
202    }
