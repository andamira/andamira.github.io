95            /// Updates the digest state with `bytes`.
96            ///
97            /// # Errors
98            /// Returns [`LengthOverflow`][crate::CryptoError::LengthOverflow]
99            /// if the total message length would exceed MD5's 64-bit bit-length field.
100            pub const fn update(&mut self, bytes: &[u8]) -> Result<(), $crate::CryptoError> {
101                if bytes.len() > (u64::MAX / 8) as usize {
102                    return Err($crate::CryptoError::LengthOverflow)
103                }
104                let add_bits = (bytes.len() as u64) * 8;
105                let Some(new_len_bits) = self.len_bits.checked_add(add_bits) else {
106                    return Err($crate::CryptoError::LengthOverflow);
107                };
108                self.len_bits = new_len_bits;
109                self.write_blocks(bytes);
110                Ok(())
111            }
112            /// Computes the MD5 digest of `bytes`.

119                let mut md5 = $Self::new();
120                $crate::unwrap![ok? md5.update(bytes)];
121                Ok(md5.finalize())
122            }

126            /// Finalizes the digest and consumes the state.
127            ///
128            /// This method cannot fail.
129            pub const fn finalize(mut self) -> $crate::Digest<{ $Self::DIGEST_LEN }> {
130                let len_bits = self.len_bits;
131                self.push_padding_byte(0x80);
132                while self.block_len != 56 {
133                    self.push_padding_byte(0);
134                }
135                let len = len_bits.to_le_bytes();
136                $crate::whilst! { i in 0..len.len(); { self.push_padding_byte(len[i]); }}
137                let mut out = [0u8; $Self::DIGEST_LEN];
138                $crate::whilst! { i in 0..self.state.len(); {
139                    let bytes = self.state[i].to_le_bytes();
140                    let j = i * 4;
141                    out[j] = bytes[0];
142                    out[j + 1] = bytes[1];
143                    out[j + 2] = bytes[2];
144                    out[j + 3] = bytes[3];
145                }}
146                $crate::Digest(out)
147            }
148            const fn write_blocks(&mut self, mut bytes: &[u8]) {
149                if self.block_len != 0 {
150                    let used = self.block_len as usize;
151                    let free = $Self::BLOCK_LEN - used;
152                    let take = $crate::cmp![min free, bytes.len()];
153                    $crate::Slice::range_mut(&mut self.block, used, used + take)
154                        .copy_from_slice($crate::Slice::range_to(&bytes, take));
155                    self.block_len += take as u8;
156                    bytes = $crate::Slice::range_from(&bytes, take);
157                    if self.block_len as usize == $Self::BLOCK_LEN {
158                        let block = self.block;
159                        self.process_block(&block);
160                        self.block_len = 0;
161                    }
162                }
163                while bytes.len() >= $Self::BLOCK_LEN {
164                    let mut block = [0u8; $Self::BLOCK_LEN];
165                    block.copy_from_slice($crate::Slice::range_to(&bytes, $Self::BLOCK_LEN));
166                    self.process_block(&block);
167                    bytes = $crate::Slice::range_from(&bytes, $Self::BLOCK_LEN);
168                }
169                if !bytes.is_empty() {
170                    $crate::Slice::range_to_mut(&mut self.block, bytes.len()).copy_from_slice(bytes);
171                    self.block_len = bytes.len() as u8;
172                }
173            }
174            const fn push_padding_byte(&mut self, byte: u8) {
175                self.block[self.block_len as usize] = byte;
176                self.block_len += 1;
177                if self.block_len as usize == $Self::BLOCK_LEN {
178                    let block = self.block;
179                    self.process_block(&block);
180                    self.block = [0; $Self::BLOCK_LEN];
181                    self.block_len = 0;
182                }
183            }
184            const fn process_block(&mut self, block: &[u8; $Self::BLOCK_LEN]) {
185                let mut m = [0u32; 16];
186                $crate::whilst! { i in 0..16; {
187                    let j = i * 4;
188                    m[i] = u32::from_le_bytes([
189                        block[j],
190                        block[j + 1],
191                        block[j + 2],
192                        block[j + 3],
193                    ]);
194                }}
195                let [mut a, mut b, mut c, mut d] = self.state;
196                $crate::whilst! { i in 0..64; {
197                    let (f, g) = if i < 16 {
198                        ((b & c) | ((!b) & d), i)
199                    } else if i < 32 {
200                        ((d & b) | ((!d) & c), (5 * i + 1) % 16)
201                    } else if i < 48 {
202                        (b ^ c ^ d, (3 * i + 5) % 16)
203                    } else {
204                        (c ^ (b | (!d)), (7 * i) % 16)
205                    };
206                    let next = b.wrapping_add(
207                        a.wrapping_add(f)
208                            .wrapping_add(Self::K[i])
209                            .wrapping_add(m[g])
210                            .rotate_left(Self::S[i]),
211                    );
212                    a = d;
213                    d = c;
214                    c = b;
215                    b = next;
216                }}
217                self.state[0] = self.state[0].wrapping_add(a);
218                self.state[1] = self.state[1].wrapping_add(b);
219                self.state[2] = self.state[2].wrapping_add(c);
220                self.state[3] = self.state[3].wrapping_add(d);
221            }
222        }
223    };
