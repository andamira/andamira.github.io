71macro_rules! _assert_digest_vectors {
72    ($Alg:ident; $($input:expr => $hex:literal),+ $(,)?) => {
73        $( assert_eq!($Alg::digest_bytes($input).unwrap(), $crate::Digest($crate::_hex($hex))); )+
74    };
75}
76macro_rules! _assert_hmac_vectors {
77    ($Alg:ident; $($key:expr, $data:expr => $hex:literal),+ $(,)?) => {
78        $( assert_eq!($Alg::hmac($key, $data).unwrap(), $crate::Digest($crate::_hex($hex))); )+
79    };
80}
81macro_rules! _assert_hotp_codes {
82    ($Alg:ident, $key:expr, $digits:expr; $($counter:expr => $code:expr),+ $(,)?) => {
83        $( assert_eq!($Alg::hotp($key, $counter, $digits).unwrap().code(), $code); )+
84    };
85}
86macro_rules! _assert_totp_codes {
87    ($Alg:ident, $key:expr; $($time:expr => $code:expr),+ $(,)?) => {
88        $( assert_eq!($Alg::totp_with($key, $time, 0, 30, 8).unwrap().code(), $code); )+
89    };
90}
91macro_rules! _digest_common_tests {
92    ($Alg:ident;
93        abc: $abc:literal,
94        block: $block:literal,
95        pad_at: $pad_at:literal,
96        million_a: $million:literal $(,)?
97    ) => {
98        #[test]
99        #[cfg(not(miri))] // too slow for miri
100        fn million_a() {
101            let mut digest = $Alg::new();
102            for _ in 0..1_000_000 {
103                digest.update(b"a").unwrap();
104            }
105            assert_eq!(digest.finalize(), $crate::Digest($crate::_hex($million)));
106        }
107        #[test]
108        fn chunked_updates_match_one_shot() {
109            let input = b"the quick brown fox jumps over the lazy dog";
110            let full = $Alg::digest_bytes(input).unwrap();
111            let mut chunked = $Alg::new();
112            chunked.update(b"the quick ").unwrap();
113            chunked.update(b"brown fox ").unwrap();
114            chunked.update(b"jumps over ").unwrap();
115            chunked.update(b"the lazy dog").unwrap();
116            assert_eq!(full, chunked.finalize());
117        }
118        #[test]
119        fn boundary_lengths() {
120            let lengths = [
121                0,
122                1,
123                $pad_at - 1,
124                $pad_at,
125                $pad_at + 1,
126                $block - 1,
127                $block,
128                $block + 1,
129                $block * 2 - 9,
130                $block * 2 - 8,
131            ];
132            let mut bytes = [0u8; 256];
133            for len in lengths {
134                crate::whilst! { i in 0..len; { bytes[i] = i as u8; }}
135                let full = $Alg::digest_bytes(&bytes[..len]).unwrap();
136                let mut chunked = $Alg::new();
137                for chunk in bytes[..len].chunks(3) {
138                    chunked.update(chunk).unwrap();
139                }
140                assert_eq!(full, chunked.finalize());
141            }
142        }
143        #[test]
144        fn reset() {
145            let mut digest = $Alg::new();
146            digest.update(b"abc").unwrap();
147            digest.reset();
148            digest.update(b"abc").unwrap();
149            assert_eq!(digest.finalize(), $crate::Digest($crate::_hex($abc)));
150        }
151    };
152}
