9/// A fixed-size message digest.
10#[doc = crate::_doc_meta!{location("data/codec/crypto")}]
11#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
12#[must_use]
13pub struct Digest<const N: usize>(pub [u8; N]);
14
15#[rustfmt::skip]
16impl<const N: usize> Digest<N> {
17    /// The digest size in bytes.
18    pub const LEN: usize = N;
19
20    /// Creates a digest from bytes.
21    pub const fn new(bytes: [u8; N]) -> Self { Self(bytes) }
22
23    /// Returns the digest bytes.
24    #[must_use] pub const fn as_array(&self) -> &[u8; N] { &self.0 }
25    /// Returns the digest bytes by value.
26    #[must_use] pub const fn into_array(self) -> [u8; N] { self.0 }
27    /// Returns the digest bytes.
28    #[must_use] pub const fn as_slice(&self) -> &[u8; N] { &self.0 }
29    /// Returns the digest size in bytes.
30    #[must_use] pub const fn len(&self) -> usize { N }
31    /// Returns `true` if the digest has zero length.
32    #[must_use] pub const fn is_empty(&self) -> bool { N == 0 }
33
34    /// Returns `true` if both digests contain the same bytes.
35    ///
36    /// This is compile-time friendly.
37    /// It does not promise hardened side-channel behavior at runtime.
38    #[must_use]
39    pub const fn eq(&self, other: &Self) -> bool {
40        let mut diff = 0u8;
41        whilst! { i in 0..N; { diff |= self.0[i] ^ other.0[i]; }}
42        diff == 0
43    }
44    /// Returns `true` if this digest equals `bytes`.
45    ///
46    /// This is compile-time friendly.
47    /// It does not promise hardened side-channel behavior at runtime.
48    #[must_use]
49    pub const fn eq_slice(&self, bytes: &[u8]) -> bool {
50        is! { bytes.len() != N, return false }
51        let mut diff = 0u8;
52        whilst! { i in 0..N; { diff |= self.0[i] ^ bytes[i]; }}
53        diff == 0
54    }
55}
56#[rustfmt::skip]
57mod impls_traits {
58    use crate::{ConstInit, Digest};
59    impl<const N: usize> ConstInit for Digest<N> {
60        const INIT: Self = Self([0; N]);
61    }
62    impl<const N: usize> From<[u8; N]> for Digest<N> {
63        fn from(bytes: [u8; N]) -> Self { Self(bytes) }
64    }
65    impl<const N: usize> From<Digest<N>> for [u8; N] {
66        fn from(digest: Digest<N>) -> Self { digest.0 }
67    }
68    impl<const N: usize> AsRef<[u8]> for Digest<N> {
69        fn as_ref(&self) -> &[u8] { &self.0 }
70    }
71}

74/// Defines a selected cryptographic message-digest state type.
75#[doc = crate::_doc_meta!{location("data/codec/crypto")}]
76///
77/// Generates a concrete allocation-free, const-friendly digest type from a
78/// whitelisted algorithm selector.
79///
80/// The generated type includes streaming update/finalize methods, one-shot
81/// digesting, HMAC support when applicable, and OTP helpers when the digest
82/// output is suitable for HOTP truncation.
83///
84/// # Examples
85/// ```
86/// use devela::{Digest, digest};
87/// digest! {
88///     /// Incremental SHA-256 state for this module.
89///     pub struct MySha256: Sha256
90/// }
91/// let digest = MySha256::digest_bytes(b"abc").unwrap();
92/// assert_eq!(
93///     digest,
94///     Digest([
95///         0xba, 0x78, 0x16, 0xbf, 0x8f, 0x01, 0xcf, 0xea,
96///         0x41, 0x41, 0x40, 0xde, 0x5d, 0xae, 0x22, 0x23,
97///         0xb0, 0x03, 0x61, 0xa3, 0x96, 0x17, 0x7a, 0x9c,
98///         0xb4, 0x10, 0xff, 0x61, 0xf2, 0x00, 0x15, 0xad,
99///     ]),
100/// );
101/// ```
