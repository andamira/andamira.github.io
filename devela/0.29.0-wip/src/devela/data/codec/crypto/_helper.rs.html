6pub(crate) const fn _hex<const N: usize>(s: &str) -> [u8; N] {
7    use devela::{Base, Rfc4648};
8    type Hex = Base<16, false, false, true, Rfc4648>;
9    let input = s.as_bytes();
10    assert!(input.len() == N * 2);
11    let mut out = [0u8; N];
12    let Some(written) = Hex::decode_from_slice(input, &mut out) else {
13        panic!("invalid hex")
14    };
15    assert!(written == N);
16    out
17}

19/// Implements the standard HMAC construction for a concrete digest type.
20///
21/// The hash type must provide `BLOCK_LEN`, `digest_bytes`, `new`, `update`, and `finalize`.
22/// The generated method is allocation-free and const-friendly.
23#[doc(hidden)]
24#[macro_export]
25macro_rules! __crypto_impl_hmac {
26    ($Self:ident) => {
27        $crate::paste! {
28            /// Computes the HMAC of `message` using the given `key`.
29            ///
30            /// If the key is longer than the block size, it is first hashed.
31            ///
32            /// # Errors
33            /// Returns [`LengthOverflow`] if key hashing or the
34            /// inner/outer digest computation exceeds the hash input length limit.
35            ///
36            /// [`LengthOverflow`]: crate::CryptoError::LengthOverflow
37            pub const fn hmac(key: &[u8], message: &[u8])
38                -> Result<$crate::Digest<{ $Self::DIGEST_LEN }>, $crate::CryptoError> {
39                // 1. Prepare a key block.
40                let mut key_block = [0u8; $Self::BLOCK_LEN];
41                if key.len() > $Self::BLOCK_LEN {
42                    // hash the key and pad the rest with zeros
43                    let hashed = $crate::unwrap![ok? $Self::digest_bytes(key)].into_array();
44                    $crate::whilst! { i in 0..hashed.len(); { key_block[i] = hashed[i]; }}
45                } else {
46                    // copy the key and pad with zeros
47                    $crate::whilst! { i in 0..key.len(); { key_block[i] = key[i]; }}
48                }
49                // 2. Inner and outer padding
50                const IPAD: u8 = 0x36;
51                const OPAD: u8 = 0x5c;
52                let mut ipad = [0u8; $Self::BLOCK_LEN];
53                let mut opad = [0u8; $Self::BLOCK_LEN];
54                $crate::whilst! { i in 0..$Self::BLOCK_LEN; {
55                    ipad[i] = key_block[i] ^ IPAD;
56                    opad[i] = key_block[i] ^ OPAD;
57                }}
58                // 3. Inner hash: H(ipad || message)
59                let mut inner = $Self::new();
60                $crate::unwrap![ok? inner.update(&ipad)];
61                $crate::unwrap![ok? inner.update(message)];
62                let inner_hash = inner.finalize().into_array();
63                // 4. Outer hash: H(opad || inner_hash)
64                let mut outer = $Self::new();
65                $crate::unwrap![ok? outer.update(&opad)];
66                $crate::unwrap![ok? outer.update(&inner_hash)];
67                Ok(outer.finalize())
68            }
69        }
70    };
71}
72#[doc(hidden)]
73pub use __crypto_impl_hmac;

75#[doc(hidden)]
76#[macro_export]
77macro_rules! __crypto_impl_otp {
78    (hash: $name:ident, otp: $otp:path, doc: $doc:literal) => { $crate::paste! {
79        #[doc = $doc " impls"]
80        impl $name {
81            #[doc = "Generates an HOTP code using HMAC-" $doc "."]
82            ///
83            /// Computes `HOTP(K, C)` from the shared secret `key` and 8-byte big-endian
84            /// counter, then applies dynamic truncation.
85            ///
86            /// The returned code is numeric and may have fewer visible digits than
87            /// [`digits`][crate::Otp::digits]; format it with leading zeroes for display.
88            ///
89            /// # Errors
90            /// - Returns [`InvalidLength`][crate::CryptoError::InvalidLength]
91            ///   if `digits` is outside `MIN_DIGITS..=MAX_DIGITS`.
92            /// - Returns [`LengthOverflow`][crate::CryptoError::LengthOverflow] if the
93            #[doc = "underlying HMAC-" $doc " computation exceeds " $doc "'s input length limit."]
94            pub const fn hotp(key: &[u8], counter: u64, digits: u32)
95                -> Result<$otp, $crate::CryptoError> {
96                let mac = $crate::unwrap![ok? <$name>::hmac(key, &counter.to_be_bytes())];
97                let offset = (mac.0[<$name>::DIGEST_LEN - 1] & 0x0f) as usize;
98                let code = u32::from_be_bytes([
99                    mac.0[offset] & 0x7f,
100                    mac.0[offset + 1],
101                    mac.0[offset + 2],
102                    mac.0[offset + 3],
103                ]);
104                $otp::new_reduced(code as u64, digits)
105            }
106
107            #[doc = "Generates a TOTP code using HMAC-" $doc " and the default TOTP parameters."]
108            ///
109            /// Uses [`DEFAULT_EPOCH`][crate::Otp::DEFAULT_EPOCH]
110            /// and [`DEFAULT_PERIOD`][crate::Otp::DEFAULT_PERIOD].
111            ///
112            /// # Errors
113            /// - Returns [`InvalidLength`][crate::CryptoError::InvalidLength]
114            ///   if `digits` is outside `MIN_DIGITS..=MAX_DIGITS`.
115            /// - Returns [`LengthOverflow`][crate::CryptoError::LengthOverflow]
116            #[doc = "underlying HMAC-" $doc " computation exceeds " $doc "'s input length limit."]
117            pub const fn totp(key: &[u8], unix_seconds: u64, digits: u32)
118                -> Result<$otp, $crate::CryptoError> {
119                $name::totp_with(key, unix_seconds,
120                    $otp::DEFAULT_EPOCH, $otp::DEFAULT_PERIOD, digits)
121            }
122            #[doc = "Generates a TOTP code using HMAC-" $doc " and explicit TOTP parameters."]
123            ///
124            /// Derives the moving counter from `unix_seconds`, `epoch`, and `period`,
125            #[doc = "then computes HOTP-" $doc " with that counter."]
126            ///
127            /// # Errors
128            /// - Returns [`InvalidParameter`][crate::CryptoError::InvalidParameter]
129            ///   if `period == 0` or if `unix_seconds < epoch`.
130            /// - Returns [`InvalidLength`][crate::CryptoError::InvalidLength]
131            ///   if `digits` is outside `MIN_DIGITS..=MAX_DIGITS`.
132            /// - Returns [`LengthOverflow`][crate::CryptoError::LengthOverflow]
133            #[doc = "underlying HMAC-" $doc " computation exceeds " $doc "'s input length limit."]
134            pub const fn totp_with(key: &[u8], unix_seconds: u64, epoch: u64,
135                period: u64, digits: u32) -> Result<$otp, $crate::CryptoError> {
136                let counter = $crate::unwrap![ok? $otp::time_counter(unix_seconds, epoch, period)];
137                $name::hotp(key, counter, digits)
138            }
139        }
140    }};
141}
