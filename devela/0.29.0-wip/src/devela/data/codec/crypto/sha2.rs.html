7/// Implements SHA-2 digest methods for a concrete hash type.
8#[doc = crate::_doc_meta!{location("data/codec/crypto")}]
9///
10/// Supports the SHA-2 32-bit-word core used by SHA-224 and SHA-256, and the
11/// SHA-2 64-bit-word core used by SHA-384, SHA-512, and SHA-512/t variants.
12///
13/// The selected word type determines the machine shape: word size, block size,
14/// length-field size, schedule size, round constants, and compression functions.
15/// The invocation supplies the concrete digest size, output truncation shape,
16/// display name, and initial hash state.
17//
18// References:
19// - FIPS 180-4, Secure Hash Standard.
20// - RFC 6234, US Secure Hash Algorithms.
21// - RFC 2104, Keyed-Hash Message Authentication Code.
22// - RFC 4231, Test Vectors for HMAC-SHA-224, HMAC-SHA-256, HMAC-SHA-384, and HMAC-SHA-512.

164            ///
165            /// # Errors
166            /// Returns [`LengthOverflow`] if the total message length would exceed
167            #[doc = concat!($DOC, "'s ", stringify!($len_bits), " bit-length field.")]
168            ///
169            /// [`LengthOverflow`]: crate::CryptoError::LengthOverflow
170            pub const fn update(&mut self, bytes: &[u8]) -> Result<(), $crate::CryptoError> {
171                let Some(add_bits) = (bytes.len() as $len_bits).checked_mul(8) else {
172                    return Err($crate::CryptoError::LengthOverflow);
173                };
174                let Some(new_len_bits) = self.len_bits.checked_add(add_bits) else {
175                    return Err($crate::CryptoError::LengthOverflow);
176                };
177                self.len_bits = new_len_bits;
178                self.write_blocks(bytes);
179                Ok(())
180            }

197            /// Finalizes the digest and consumes the state.
198            ///
199            /// This method cannot fail.
200            pub const fn finalize(mut self) -> $crate::Digest<{ $Self::DIGEST_LEN }> {
201                let len_bits = self.len_bits;
202                self.push_padding_byte(0x80);
203                while self.block_len != $pad_at { self.push_padding_byte(0); }
204                let len = len_bits.to_be_bytes();
205                $crate::whilst! { i in 0..len.len(); { self.push_padding_byte(len[i]); }}
206                let mut out = [0u8; $digest_len];
207                $crate::__crypto_impl_sha2! { %write_digest
208                    out: out, state: self.state,
209                    word_bytes: $word_bytes,
210                    output_words: $output_words,
211                    output_tail_bytes: $output_tail_bytes
212                }
213                $crate::Digest(out)
214            }

219            const fn write_blocks(&mut self, mut bytes: &[u8]) {
220                if self.block_len != 0 {
221                    let used = self.block_len as usize;
222                    let free = $Self::BLOCK_LEN - used;
223                    let take = $crate::cmp![min free, bytes.len()];
224                    $crate::Slice::range_mut(&mut self.block, used, used + take)
225                        .copy_from_slice($crate::Slice::range_to(&bytes, take));
226                    self.block_len += take as u8;
227                    bytes = $crate::Slice::range_from(&bytes, take);
228                    if self.block_len as usize == $Self::BLOCK_LEN {
229                        let block = self.block;
230                        self.process_block(&block);
231                        self.block_len = 0;
232                    }
233                }
234                while bytes.len() >= $Self::BLOCK_LEN {
235                    let mut block = [0u8; $Self::BLOCK_LEN];
236                    block.copy_from_slice($crate::Slice::range_to(&bytes, $Self::BLOCK_LEN));
237                    self.process_block(&block);
238                    bytes = $crate::Slice::range_from(&bytes, $Self::BLOCK_LEN);
239                }
240                if !bytes.is_empty() {
241                    $crate::Slice::range_to_mut(&mut self.block, bytes.len())
242                        .copy_from_slice(bytes);
243                    self.block_len = bytes.len() as u8;
244                }
245            }
246            const fn push_padding_byte(&mut self, byte: u8) {
247                self.block[self.block_len as usize] = byte;
248                self.block_len += 1;
249                if self.block_len as usize == $Self::BLOCK_LEN {
250                    let block = self.block;
251                    self.process_block(&block);
252                    self.block = [0; $Self::BLOCK_LEN];
253                    self.block_len = 0;
254                }
255            }
256            const fn process_block(&mut self, block: &[u8; $Self::BLOCK_LEN]) {
257                let mut w = [0; $schedule_len];
258                $crate::whilst! { t in 0..16; {
259                    let i = t * $word_bytes;
260                    w[t] = $crate::__crypto_impl_sha2![%read_word word: $word, block: block, at: i];
261                }}
262                $crate::whilst! { t in 16..$schedule_len; {
263                    w[t] = Self::small_sigma1(w[t - 2])
264                        .wrapping_add(w[t - 7])
265                        .wrapping_add(Self::small_sigma0(w[t - 15]))
266                        .wrapping_add(w[t - 16]);
267                }}
268                let [mut a, mut b, mut c, mut d, mut e, mut f, mut g, mut h] = self.state;
269                $crate::whilst! { t in 0..$schedule_len; {
270                    let t1 = h
271                        .wrapping_add(Self::big_sigma1(e))
272                        .wrapping_add(Self::ch(e, f, g))
273                        .wrapping_add(Self::K[t])
274                        .wrapping_add(w[t]);
275                    let t2 = Self::big_sigma0(a).wrapping_add(Self::maj(a, b, c));
276                    h = g; g = f; f = e; e = d.wrapping_add(t1);
277                    d = c; c = b; b = a; a = t1.wrapping_add(t2);
278                }}
279                self.state[0] = self.state[0].wrapping_add(a);
280                self.state[1] = self.state[1].wrapping_add(b);
281                self.state[2] = self.state[2].wrapping_add(c);
282                self.state[3] = self.state[3].wrapping_add(d);
283                self.state[4] = self.state[4].wrapping_add(e);
284                self.state[5] = self.state[5].wrapping_add(f);
285                self.state[6] = self.state[6].wrapping_add(g);
286                self.state[7] = self.state[7].wrapping_add(h);
287            }
288        }
289        $crate::__crypto_impl_otp! { hash: $Self, otp: $crate::Otp, doc: $DOC }
290    };
291    (%write_digest
292        out: $out:ident,
293        state: $state:expr,
294        word_bytes: $word_bytes:tt,
295        output_words: $output_words:literal,
296        output_tail_bytes: $output_tail_bytes:tt
297    ) => {
298        $crate::whilst! { wi in 0..$output_words; {
299            let bytes = ($state)[wi].to_be_bytes();
300            let j = wi * $word_bytes;
301            $crate::punroll! { $word_bytes |bi|
302                $out[j + bi] = bytes[bi]
303            }
304        }}
305        $crate::__crypto_impl_sha2! { %write_digest_tail
306            out: $out,
307            state: $state,
308            word_bytes: $word_bytes,
309            output_words: $output_words,
310            output_tail_bytes: $output_tail_bytes
311        }
312    };
313    (%write_digest_tail
314        out: $out:ident,
315        state: $state:expr,
316        word_bytes: $word_bytes:tt,
317        output_words: $output_words:literal,
318        output_tail_bytes: 0
319    ) => {};
320    (%write_digest_tail
321        out: $out:ident,
322        state: $state:expr,
323        word_bytes: $word_bytes:tt,
324        output_words: $output_words:literal,
325        output_tail_bytes: $tail:tt
326    ) => {
327        let bytes = ($state)[$output_words].to_be_bytes();
328        let j = $output_words * $word_bytes;
329        $crate::punroll! { $tail |bi| $out[j + bi] = bytes[bi] }
330    };
331    (%read_word word: u32, block: $block:expr, at: $i:expr) => {
332        u32::from_be_bytes([
333            $block[$i + 0], $block[$i + 1], $block[$i + 2], $block[$i + 3],
334        ])
335    };
336    (%read_word word: u64, block: $block:expr, at: $i:expr) => {
337        u64::from_be_bytes([
338            $block[$i + 0], $block[$i + 1], $block[$i + 2], $block[$i + 3],
339            $block[$i + 4], $block[$i + 5], $block[$i + 6], $block[$i + 7],
340        ])
341    };
