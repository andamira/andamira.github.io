8#[doc = crate::_tags!(data codec)]
9/// A borrowed RIFF chunk.
10#[doc = crate::_doc_meta!{location("data/codec/pack")}]
11///
12/// This is a view into an existing byte slice. The chunk owns no data and does
13/// not interpret its payload unless asked for container-specific information
14/// such as its RIFF form type or LIST type.
15#[derive(Clone, Copy, Debug, PartialEq, Eq)]
16pub struct RiffChunk<'a> {
17    pub(super) id: BinTag4,
18    pub(super) size: u32,
19    pub(super) data: &'a [u8],
20    pub(super) offset: usize,
21}
22
23#[rustfmt::skip]
24impl<'a> RiffChunk<'a> {
25    /// Returns the chunk identifier.
26    #[inline(always)]
27    pub const fn id(self) -> BinTag4 { self.id }
28
29    /// Returns the declared chunk data size.
30    #[must_use] #[inline(always)]
31    pub const fn size(self) -> u32 { self.size }
32    /// Returns the chunk data length.
33    #[must_use] #[inline(always)]
34    pub const fn len(self) -> usize { self.data.len() }
35    /// Returns whether the chunk data is empty.
36    #[must_use] #[inline(always)]
37    pub const fn is_empty(self) -> bool { self.data.is_empty() }
38
39    /// Returns the chunk data.
40    #[must_use] #[inline(always)]
41    pub const fn data(self) -> &'a [u8] { self.data }
42    /// Returns the chunk offset relative to the parsed byte region.
43    #[must_use] #[inline(always)]
44    pub const fn offset(self) -> usize { self.offset }
45
46    /// Returns whether this chunk is `RIFF`.
47    #[must_use] #[inline(always)]
48    pub const fn is_riff(self) -> bool { self.id.eq_bytes(*b"RIFF") }
49    /// Returns whether this chunk is `RIFX`.
50    #[must_use] #[inline(always)]
51    pub const fn is_rifx(self) -> bool { self.id.eq_bytes(*b"RIFX") }
52    /// Returns whether this chunk is `LIST`.
53    #[must_use] #[inline(always)]
54    pub const fn is_list(self) -> bool { self.id.eq_bytes(*b"LIST") }
55
56    /// Returns whether this chunk can contain subchunks.
57    #[must_use] #[inline(always)]
58    pub const fn is_container(self) -> bool {
59        self.is_riff() || self.is_rifx() || self.is_list()
60    }
61    /// Returns the RIFF form type or LIST type.
62    #[must_use]
63    pub const fn container_type(self) -> Option<BinTag4> {
64        is! { !self.is_container() || self.data.len() < 4, return None }
65        Some(BinTag4::new([self.data[0], self.data[1], self.data[2], self.data[3]]))
66    }
67
68    /// Returns the subchunk byte region of a `RIFF`, `RIFX`, or `LIST` chunk.
69    #[must_use]
70    pub const fn subchunk_data(self) -> Option<&'a [u8]> {
71        is! { !self.is_container() || self.data.len() < 4, return None }
72        Some(self.data.split_at(4).1)
73    }
74    /// Iterates over this chunk's subchunks.
75    pub const fn subchunks(self) -> Result<RiffChunkIter<'a>, RiffError> {
76        let Some(data) = self.subchunk_data() else { return Err(RiffError::MissingContainerType); };
77        Ok(Riff::chunks(data))
78    }
79}

81#[doc = crate::_tags!(data codec iterator)]
82/// An iterator over borrowed RIFF chunks.
83#[doc = crate::_doc_meta!{location("data/codec/pack")}]
84#[must_use]
85#[derive(Clone, Debug, PartialEq, Eq)]
86pub struct RiffChunkIter<'a> {
87    bytes: &'a [u8],
88    offset: usize,
89}
90#[rustfmt::skip]
91impl<'a> RiffChunkIter<'a> {
92    pub(crate) const fn _new(bytes: &'a [u8], offset: usize) -> Self { Self { bytes, offset } }
93
94    /// Returns a new iterator over the chunks in `bytes`.
95    #[inline(always)]
96    pub const fn new(bytes: &'a [u8]) -> Self { Self::_new(bytes, 0) }
97
98    /// Returns the remaining byte region.
99    #[must_use] #[inline(always)]
100    pub const fn remaining(self) -> &'a [u8] { slice![self.bytes, self.offset, ..] }
101    /// Returns the current byte offset.
102    #[must_use] #[inline(always)]
103    pub const fn offset(self) -> usize { self.offset }
104    /// Returns whether no bytes remain.
105    #[must_use] #[inline(always)]
106    pub const fn is_empty(self) -> bool { self.offset >= self.bytes.len() }
107
108    /// Returns the next RIFF chunk.
109    ///
110    /// Returns `None` once there are no bytes left.
111    ///
112    /// On error, the iterator is exhausted.
113    #[must_use]
114    pub const fn next_chunk(&mut self) -> Option<Result<RiffChunk<'a>, RiffError>> {
115        is![self.offset >= self.bytes.len(), return None];
116        match Riff::chunk_at(self.bytes, self.offset) {
117            Ok((chunk, next)) => { self.offset = next; Some(Ok(chunk)) }
118            Err(err) => { self.offset = self.bytes.len(); Some(Err(err)) }
119        }
120    }
121    /// Counts remaining valid chunks, consuming the iterator.
122    ///
123    /// # Errors
124    /// Returns the first parsing error encountered.
125    pub const fn count_chunks(mut self) -> Result<usize, RiffError> {
126        let mut count = 0;
127        while let Some(result) = self.next_chunk() {
128            match result { Ok(_) => count += 1, Err(err) => return Err(err) }
129        }
130        Ok(count)
131    }
132}
133#[rustfmt::skip]
134impl<'a> Iterator for RiffChunkIter<'a> {
135    type Item = Result<RiffChunk<'a>, RiffError>;
136    #[inline(always)]
137    fn next(&mut self) -> Option<Self::Item> { self.next_chunk() }
138    fn size_hint(&self) -> (usize, Option<usize>) {
139        let remaining = self.bytes.len().saturating_sub(self.offset);
140        // Every valid chunk needs at least 8 bytes of header.
141        // If trailing invalid bytes remain, `next_chunk` can still yield one error.
142        let max = if remaining == 0 { 0 } else { remaining / 8 + 1 };
143        (0, Some(max))
144    }
145}
146impl<'a> crate::IteratorFused for RiffChunkIter<'a> {}
