9#[doc = crate::_tags!(data lifetime)]
10/// A linked list node in a `ConstList`.
11#[doc = crate::_doc_meta!{location("data/topol")}]
12#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
13struct ConstListItem<'a, T: 'a> {
14    /// The item represented by this node.
15    first: T,
16    /// The rest of the list.
17    rest: &'a ConstList<'a, T>,
18}
19
20#[doc = crate::_tags!(data_structure lifetime)]
21/// An immutable, append-only, linear, functional, non-contiguous, list.
22#[doc = crate::_doc_meta!{location("data/topol")}]
23///
24/// A safe, predictable, and lightweight structure, suitable where immutability
25/// is an asset and compile-time guarantees matter more than list manipulation.
26///
27/// # Examples
28/// ```
29/// # use devela::ConstList;
30/// const MY_LIST: ConstList<'static, i32> = ConstList::new()
31///     .push(2)
32///     .push(4)
33///     .push(8);
34///
35/// assert_eq!(8, *MY_LIST.pop().0.unwrap());
36/// ```
37#[doc = crate::_doc_vendor!("const_list")]
38#[must_use]
39#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, Hash)]
40pub struct ConstList<'a, T: 'a>(Option<ConstListItem<'a, T>>);
41
42crate::_impl_init![<T> Self::new() => ConstList<'_, T>];
43
44impl<'a, T: 'a> ConstList<'a, T> {
45    /// Creates a new, empty list.
46    pub const fn new() -> Self {
47        Self(None)
48    }
49
50    /// Gets a reference to the item at the provided index in this list, if any.
51    #[must_use]
52    pub const fn get(&self, index: usize) -> Option<&T> {
53        if let Some(value) = &self.0 {
54            if index == 0 { Some(&value.first) } else { value.rest.get(index - 1) }
55        } else {
56            None
57        }
58    }
59
60    /// Determines the length of this list.
61    #[must_use]
62    pub const fn len(&self) -> usize {
63        if let Some(value) = &self.0 { value.rest.len() + 1 } else { 0 }
64    }
65
66    /// Whether the list is empty.
67    #[must_use]
68    pub const fn is_empty(&self) -> bool {
69        self.0.is_none()
70    }
71
72    /// Pushes a new item onto the beginning of this list,
73    /// producing a new list head.
74    pub const fn push(&'a self, value: T) -> Self {
75        ConstList(Some(ConstListItem { first: value, rest: self }))
76    }
77
78    /// Removes the first item (if any) from this list, and produces
79    /// the rest of the list.
80    pub const fn pop(&'a self) -> (Option<&'a T>, &'a Self) {
81        if let Some(value) = &self.0 {
82            (Some(&value.first), value.rest)
83        } else {
84            (None, self)
85        }
86    }
87
88    /// Creates an iterator over the contents of the list.
89    pub const fn iter(&self) -> ConstListIter<'_, T> {
90        ConstListIter { target: self }
91    }
92}
93
94impl<'a, T> IntoIterator for &'a ConstList<'a, T> {
95    type Item = &'a T;
96    type IntoIter = ConstListIter<'a, T>;
97
98    fn into_iter(self) -> Self::IntoIter {
99        ConstListIter { target: self }
100    }
101}

103#[doc = crate::_tags!(iterator lifetime)]
104/// Iterates over the contents of a [`ConstList`].
105#[doc = crate::_doc_meta!{location("data/topol")}]
106#[must_use]
107#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
108pub struct ConstListIter<'a, T> {
109    /// The current list head.
110    target: &'a ConstList<'a, T>,
111}
112impl<'a, T> ConstListIter<'a, T> {
113    /// Returns the next item from the list.
114    pub const fn next(&mut self) -> Option<&'a T> {
115        let (first, rest) = self.target.pop();
116        self.target = rest;
117        first
118    }
119}
120impl<'a, T> Iterator for ConstListIter<'a, T> {
121    type Item = &'a T;
122    fn next(&mut self) -> Option<Self::Item> {
123        self.next()
124    }
