11#[doc = crate::_tags!(data_structure platform set)]
12/// A Linux mmap-backed sparse set with lazily zeroed storage.
13#[doc = crate::_doc_meta!{location("data/store/key")}]
14///
15/// The backing arrays are anonymous memory mappings, so Linux provides
16/// zero-initialized virtual pages without Rust clearing every element first.
17#[derive(Debug)]
18pub struct LinuxSparseSet {
19    dense: LinuxMmapUsize,
20    sparse: LinuxMmapUsize,
21    count: usize,
22}
23#[rustfmt::skip]
24impl LinuxSparseSet {
25    /// Creates a new Linux mmap-backed sparse set.
26    ///
27    /// `dense` is the maximum number of stored values.
28    /// `sparse` is the size of the value universe.
29    pub fn new(dense: usize, sparse: usize) -> SparseSetResult<Self> {
30        Ok(Self {
31            dense: LinuxMmapUsize::new_zeroed(dense, false)?,
32            sparse: LinuxMmapUsize::new_zeroed(sparse, false)?,
33            count: 0,
34        })
35    }
36    /// Creates a new sparse set using `MAP_NORESERVE`.
37    ///
38    /// This can reserve a large virtual range, but writes may fail at the OS
39    /// level if memory cannot be committed later.
40    pub fn new_noreserve(dense: usize, sparse: usize) -> SparseSetResult<Self> {
41        Ok(Self {
42            dense: LinuxMmapUsize::new_zeroed(dense, true)?,
43            sparse: LinuxMmapUsize::new_zeroed(sparse, true)?,
44            count: 0,
45        })
46    }
47    /// Returns the maximum number of values this set can store.
48    #[must_use]
49    pub const fn capacity(&self) -> usize { self.dense.len() }
50
51    /// Returns the size of the value universe.
52    #[must_use]
53    pub const fn universe(&self) -> usize { self.sparse.len() }
54
55    /// Returns the number of stored values.
56    #[must_use]
57    pub const fn len(&self) -> usize { self.count }
58
59    /// Returns `true` if the set contains no values.
60    #[must_use]
61    pub const fn is_empty(&self) -> bool { self.count == 0 }
62
63    /// Returns `true` if the dense storage is full.
64    #[must_use]
65    pub const fn is_full(&self) -> bool { self.count == self.dense.len() }
66
67    /// Clears the set in constant time.
68    pub const fn clear(&mut self) { self.count = 0; }
69
70    /// Returns the live dense values.
71    #[must_use]
72    pub fn as_slice(&self) -> &[usize] { &self.dense.as_slice()[..self.count] }
73
74    /// Returns an iterator over the stored values.
75    pub fn iter(&self) -> IterSlice<'_, usize> { self.as_slice().iter() }
76
77    /// Returns `true` if `value` is present.
78    #[must_use]
79    pub fn contains(&self, value: usize) -> bool {
80        contains_raw(self.dense.as_slice(), self.sparse.as_slice(), self.count, value)
81    }
82    /// Inserts `value`.
83    pub fn insert(&mut self, value: usize) -> SparseSetResult<bool> {
84        insert_raw(self.dense.as_mut_slice(), self.sparse.as_mut_slice(), &mut self.count, value)
85    }
86    /// Removes `value`.
87    pub fn remove(&mut self, value: usize) -> SparseSetResult<bool> {
88        remove_raw(self.dense.as_mut_slice(), self.sparse.as_mut_slice(), &mut self.count, value)
89    }
90    /// Checks the internal dense/sparse consistency.
91    pub fn check_consistency(&self) {
92        check_consistency_raw(self.dense.as_slice(), self.sparse.as_slice(), self.count);
93    }
94}

96/// Owned Linux mmap storage for `usize` values.
97#[derive(Debug)]
98struct LinuxMmapUsize {
99    ptr: PtrNonNull<usize>,
100    len: usize,
101    bytes: usize,
102}
103
104impl LinuxMmapUsize {
105    fn new_zeroed(len: usize, noreserve: bool) -> SparseSetResult<Self> {
106        let bytes =
107            len.checked_mul(size_of::<usize>()).ok_or(SparseSetError::AllocationTooLarge)?;
108        is! { bytes == 0, return Ok(Self { ptr: PtrNonNull::dangling(), len, bytes }) }
109        let mut flags = Libc::MAP_PRIVATE | Libc::MAP_ANONYMOUS;
110        is! { noreserve, flags |= Libc::MAP_NORESERVE }
111        // SAFETY:
112        // - `addr = null` lets the kernel choose the address.
113        // - `bytes > 0`.
114        // - `fd = -1` and `offset = 0` are correct for anonymous mappings.
115        // - On success, the returned mapping is readable, writable, and suitably aligned.
116        let raw = unsafe {
117            Libc::mmap(
118                core::ptr::null_mut(),
119                bytes,
120                Libc::PROT_READ | Libc::PROT_WRITE,
121                flags,
122                -1,
123                0,
124            )
125        };
126        is! { Libc::is_map_failed(raw), return Err(SparseSetError::MapFailed) }
127        let ptr = PtrNonNull::new(raw.cast::<usize>()).ok_or(SparseSetError::MapFailed)?;
128        Ok(Self { ptr, len, bytes })
129    }
130    const fn len(&self) -> usize {
131        self.len
132    }
133    fn as_slice(&self) -> &[usize] {
134        // SAFETY:
135        // - `ptr` points to a valid mmap allocation for `len * size_of::<usize>()` bytes,
136        //   or is dangling with `len == 0`.
137        // - The mapping is initialized to zero by Linux anonymous mmap.
138        unsafe { Slice::from_raw_parts(self.ptr.as_ptr(), self.len) }
139    }
140    fn as_mut_slice(&mut self) -> &mut [usize] {
141        // SAFETY:
142        // - same as `as_slice`.
143        // - `&mut self` guarantees exclusive access to this owner.
144        unsafe { Slice::from_raw_parts_mut(self.ptr.as_ptr(), self.len) }
145    }
146}
147
148impl Drop for LinuxMmapUsize {
149    fn drop(&mut self) {
150        if self.bytes != 0 {
151            // SAFETY:
152            // - `ptr` and `bytes` come from a successful `mmap`.
153            // - this owner calls `munmap` at most once, on drop.
154            let _ = unsafe { Libc::munmap(self.ptr.as_ptr().cast::<c_void>(), self.bytes) };
155        }
156    }
157}

159#[cfg(test)]
160mod tests {
161    use super::*;
162
163    #[test]
164    fn linux_sparse_set_basic() {
165        let mut set = LinuxSparseSet::new(5, 10_000).unwrap();
166        assert_eq!(set.insert(3), Ok(true));
167        assert_eq!(set.insert(5), Ok(true));
168        assert_eq!(set.insert(5), Ok(false));
169        assert!(set.contains(3));
170        assert!(set.contains(5));
171        assert!(!set.contains(2));
172        assert!(!set.contains(11_000));
173        assert_eq!(set.remove(3), Ok(true));
174        assert!(!set.contains(3));
175        assert_eq!(set.remove(500), Ok(false));
176        set.check_consistency();
177    }
178    #[test]
179    fn linux_sparse_set_clear() {
180        let mut set = LinuxSparseSet::new(8, 1024).unwrap();
181        assert_eq!(set.insert(10), Ok(true));
182        assert_eq!(set.insert(20), Ok(true));
183        set.clear();
184        assert!(!set.contains(10));
185        assert!(!set.contains(20));
186        assert_eq!(set.len(), 0);
187        assert_eq!(set.insert(20), Ok(true));
188        assert!(set.contains(20));
189        set.check_consistency();
190    }
191}
