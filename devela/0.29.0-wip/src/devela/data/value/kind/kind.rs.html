7#[doc = crate::_tags!(data value)]
8/// The semantic category of a value word.
9#[doc = crate::_doc_meta!{location("data/value")}]
10#[repr(u8)]
11#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
12pub enum ValueKind {
13    /* 4-bit compact universal band 0..=15 */
14    /// No value.
15    #[default]
16    Nil = 0,
17    /// A boolean truth value.
18    Bool = 1,
19    /// A signed integer value of unspecified width.
20    Int = 2,
21    /// An unsigned integer value of unspecified width.
22    UInt = 3,
23    /// A floating-point numeric value of unspecified width.
24    Float = 4,
25    /// A character-like scalar value.
26    Char = 5,
27    /// An interned or externally resolved symbolic identifier.
28    Symbol = 6,
29    /// A closed local choice from a finite set.
30    Enum = 7,
31    //
32    /// A generic reference into another memory domain.
33    Ref = 8,
34    /// A byte sequence or byte-oriented value.
35    Bytes = 9,
36    /// A text or string-like value.
37    Text = 10,
38    /// An ordered aggregate.
39    List = 11,
40    /// A membership aggregate.
41    Set = 12,
42    /// A relational, record-like, or columnar aggregate.
43    Table = 13,
44    /// A callable, closure, function, command, or executable binding.
45    Callable = 14,
46    /// A value not directly expressible by the local layout or profile.
47    Escape = 15,
48
49    /* 6-bit band */
50    /// A domain-specific or externally interpreted body.
51    Object = 32,
52    /// A resource-like body managed outside the value itself.
53    Resource = 33,
54    /// A time-shaped value such as a point, span, date, duration, or tick.
55    Temporal = 34,
56    /// A color value or visual scalar.
57    Color = 35,
58    /// A minimal two-part composition.
59    Pair = 36,
60
61    /* 8-bit band */
62    /// An unknown, invalid, or profile-unmapped kind.
63    Unknown = 255,
64}
65impl ValueKind {
66    /// Returns the raw 8-bit kind code.
67    #[must_use]
68    #[inline(always)]
69    pub const fn code(self) -> u8 {
70        self as u8
71    }
72    /// Returns a kind from its raw 8-bit code.
73    ///
74    /// Unknown codes map to [`Unknown`][Self::Unknown].
75    #[must_use]
76    #[inline(always)]
77    pub const fn from_code(code: u8) -> Self {
78        match code {
79            0 => Self::Nil,
80            1 => Self::Bool,
81            2 => Self::Int,
82            3 => Self::UInt,
83            4 => Self::Float,
84            5 => Self::Char,
85            6 => Self::Symbol,
86            7 => Self::Enum,
87            //
88            8 => Self::Ref,
89            9 => Self::Bytes,
90            10 => Self::Text,
91            11 => Self::List,
92            12 => Self::Set,
93            13 => Self::Table,
94            14 => Self::Callable,
95            15 => Self::Escape,
96            //
97            32 => Self::Object,
98            33 => Self::Resource,
99            34 => Self::Temporal,
100            35 => Self::Color,
101            36 => Self::Pair,
102            //
103            255 => Self::Unknown,
104            _ => Self::Unknown,
105        }
106    }
107}
108impl ValueKind {
109    /// Returns whether this kind belongs to the compact universal band.
110    #[must_use]
111    #[inline(always)]
112    pub const fn is_compact(self) -> bool {
113        self.code() <= 15
114    }
115    /// Returns whether this kind represents absence.
116    #[must_use]
117    #[inline(always)]
118    pub const fn is_nil(self) -> bool {
119        matches!(self, Self::Nil)
120    }
121    /// Returns whether this kind represents a numeric scalar.
122    #[must_use]
123    #[inline(always)]
124    pub const fn is_number(self) -> bool {
125        matches!(self, Self::Int | Self::UInt | Self::Float)
126    }
127    /// Returns whether this kind represents a scalar-like atom.
128    #[must_use]
129    #[inline(always)]
130    pub const fn is_scalar(self) -> bool {
131        matches!(
132            self,
133            Self::Nil
134                | Self::Bool
135                | Self::Int
136                | Self::UInt
137                | Self::Float
138                | Self::Char
139                | Self::Symbol
140                | Self::Enum
141        )
142    }
143    /// Returns whether this kind points outside the immediate value body.
144    #[must_use]
145    #[inline(always)]
146    pub const fn is_ref_like(self) -> bool {
147        matches!(self, Self::Ref)
148    }
149    /// Returns whether this kind describes aggregate data.
150    #[must_use]
151    #[inline(always)]
152    pub const fn is_aggregate(self) -> bool {
153        matches!(self, Self::Bytes | Self::Set | Self::Text | Self::List | Self::Table)
154    }
155    /// Returns whether this kind describes callable behavior.
156    #[must_use]
157    #[inline(always)]
158    pub const fn is_callable(self) -> bool {
159        matches!(self, Self::Callable)
160    }
161    /// Returns whether this kind describes a domain-specific body.
162    #[must_use]
163    #[inline(always)]
164    pub const fn is_object(self) -> bool {
165        matches!(self, Self::Object)
166    }
167    /// Returns whether this kind describes an externally managed resource.
168    #[must_use]
169    #[inline(always)]
170    pub const fn is_resource(self) -> bool {
171        matches!(self, Self::Resource)
172    }
173    /// Returns whether this kind needs profile-specific decoding or widening.
174    #[must_use]
175    #[inline(always)]
176    pub const fn is_escape(self) -> bool {
177        matches!(self, Self::Escape)
178    }
179    /// Returns whether this kind is unknown or unmapped.
180    #[must_use]
181    #[inline(always)]
182    pub const fn is_unknown(self) -> bool {
183        matches!(self, Self::Unknown)
184    }
185}
