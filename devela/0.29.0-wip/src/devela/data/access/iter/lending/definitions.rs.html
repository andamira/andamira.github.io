13/// A lending iterator using a generic associated lifetime.
14#[doc = crate::_doc_meta!{location("data/access/iter")}]
15///
16/// A lending iterator yields items borrowed from its own internal state.
17/// Each call to [`next`][Self::next] creates a temporary borrow of `self`
18/// and returns an item tied to that borrow. This enables iteration over
19/// memory-backed structures without copying or owning the underlying data.
20///
21/// This is the borrowing analogue of [`Iterator`][crate::Iterator].
22///
23/// See also:
24/// - [`IteratorLendingDoubleEnded`]
25/// - [`IteratorLendingExactSize`]
26/// - [`IteratorLendingPeek`]
27/// - [`IteratorLendingPeekDoubleEnded`]
28#[rustfmt::skip]
29pub trait IteratorLending {
30    /// The item yielded by this iterator.
31    ///
32    /// The lifetime `'a` corresponds to the borrow of `self` required to produce the item.
33    type Item<'a> where Self: 'a;
34
35    /// Advances the iterator and returns the next item.
36    ///
37    /// The lifetime of the returned reference is scoped to the borrow
38    /// of `self` established during this call.
39    fn next<'a>(&'a mut self) -> Option<Self::Item<'a>>;
40
41    /* non-required methods */
42
43    /// Returns a size hint for the remaining items.
44    ///
45    /// The default implementation returns `(0, None)`, which is correct but
46    /// conservative. Implementors may override this for improved accuracy.
47    fn size_hint(&self) -> (usize, Option<usize>) {
48        (0, None)
49    }
50
51    /// Consumes the iterator, returning the number of remaining items.
52    fn count(&mut self) -> usize {
53        let mut n = 0;
54        while self.next().is_some() { n += 1; }
55        n
56    }
57
58    // ✗ last
59
60    /// Advances the iterator by `n` steps.
61    ///
62    /// Returns `Ok(())` if `n` items were successfully skipped.
63    /// Returns `Err(remaining)` if the iterator ended early, where
64    /// `remaining` is the number of items that were *not* skipped.
65    fn advance_by(&mut self, mut n: usize) -> Result<(), usize> {
66        while n > 0 {
67            match self.next() {
68                Some(_) => n -= 1,
69                None => return Err(n),
70            }
71        }
72        Ok(())
73    }
74
75    /// Advances the iterator by `n` items and returns the next one.
76    fn nth<'a>(&'a mut self, mut n: usize) -> Option<Self::Item<'a>> {
77        while n > 0 {
78            is![self.next().is_none(), return None];
79            n -= 1;
80        }
81        self.next()
82    }
83
84    // ✗ step_by
85    // ✗ chain
86    // ✗ zip
87    // ✗ intersperse
88    // ✗ intersperse_with
89    // ✗ map
90
91    /// Applies `f` to each remaining item.
92    fn for_each<F>(&mut self, mut f: F)
93    where F: FnMut(Self::Item<'_>) {
94        while let Some(item) = self.next() { f(item); }
95    }
96
97    // ✗ filter
98    // ✗ filter_map
99    // ✗ enumerate
100    // ✗ peekable
101    // ✗ skip_while
102    // ✗ take_while
103    // ✗ map_while
104    // ✓ skip
105    // ✓ take
106    // ✗ scan
107    // ✗ flatmap
108    // ✗ flatten
109    // ✗ map_windows
110    // ✗ fuse
111
112    /// Calls `f` for each item before yielding it.
113    ///
114    /// This is useful for debugging or side effects. The iterator
115    /// behavior is otherwise unchanged.
116    fn inspect<F>(&mut self, mut f: F)
117    where
118        F: FnMut(&Self::Item<'_>),
119    {
120        while let Some(item) = self.next() {
121            f(&item);
122        }
123    }
124
125    // ✓ by_ref
126    // ✗ collect
127    // ✗ try_collect
128    // ✗ collect_into
129    // ✗ partition
130    // ✗ partition_in_place
131
132    /// Returns `true` if the iterator is partitioned according to `pred`.
133    ///
134    /// All items for which `pred` returns `true` must come before all
135    /// items for which it returns `false`. Once the predicate becomes
136    /// `false`, it must remain `false` for all subsequent items.
137    fn is_partitioned<P>(&mut self, mut pred: P) -> bool
138    where
139        P: FnMut(&Self::Item<'_>) -> bool,
140    {
141        let mut seen_false = false;
142        while let Some(item) = self.next() {
143            if pred(&item) {
144                if seen_false { return false; }
145            } else {
146                seen_false = true;
147            }
148        }
149        true
150    }
151
152    /// Applies `f` to each item, accumulating results in `acc`.
153    ///
154    /// Iteration stops on the first `Err`. Returns `Ok(acc)` when all
155    /// applications succeed.
156    ///
157    /// This mirrors `Iterator::try_fold` but works with borrowed items.
158    fn try_fold<B, E, F>(&mut self, mut acc: B, mut f: F) -> Result<B, E>
159    where
160        F: FnMut(B, Self::Item<'_>) -> Result<B, E>,
161    {
162        while let Some(item) = self.next() {
163            acc = f(acc, item)?;
164        }
165        Ok(acc)
166    }
167
168    /// Applies `f` to each remaining item until an error occurs.
169    fn try_for_each<F, E>(&mut self, mut f: F) -> Result<(), E>
170    where F: FnMut(Self::Item<'_>) -> Result<(), E> {
171        while let Some(item) = self.next() { f(item)?; }
172        Ok(())
173    }
174
175    // ✗ fold
176    // ✓ reduce
177    // ✓ try_reduce
178
179    /// Returns `true` if all items satisfy the predicate.
180    fn all<P>(&mut self, mut pred: P) -> bool
181    where
182        P: FnMut(&Self::Item<'_>) -> bool,
183    {
184        while let Some(item) = self.next() { is![!pred(&item), return false]; }
185        true
186    }
187    /// Returns `true` if any item satisfies the predicate.
188    fn any<P>(&mut self, mut pred: P) -> bool
189    where
190        P: FnMut(&Self::Item<'_>) -> bool,
191    {
192        while let Some(item) = self.next() { is![pred(&item), return true]; }
193        false
194    }
195
196    // ✓ find
197    // ✓ find_map
198    // ✓ try_find
199
200    /// Returns the index of the first item for which `pred` returns `true`,
201    /// or `None` if no matching item is found.
202    fn position<P>(&mut self, mut pred: P) -> Option<usize>
203    where
204        P: FnMut(&Self::Item<'_>) -> bool,
205    {
206        let mut index = 0;
207        while let Some(item) = self.next() {
208            is![pred(&item), return Some(index)];
209            index += 1;
210        }
211        None
212    }
213
214    // ✓ rposition
215    // ✓ max
216    // ✓ min
217    // ✓ max_by_key
218    // ✓ max_by
219    // ✓ min_by_key
220    // ✓ min_by
221    // ✓ rev (in double-ended)
222    // ✗ unzip
223    // ✗ copied
224    // ✗ cloned
225    // ✗ cycle
226    // ✗ array_chunks
227
228    // ~ sum
229    // ~ product
230    // ✓ cmp
231    // ✓ cmp_by
232    // ✓ partial_cmp
233    // ✓ partial_cmp_by
234    // ✓ eq
235    // ✓ eq_by
236    // ✓ ne
237    // ✓ lt
238    // ✓ le
239    // ✓ gt
240    // ✓ ge
241    // ✓ is_sorted
242    // ✓ is_sorted_by
243    // ✓ is_sorted_by_key
244}

247/// A lending iterator that can yield items from the back.
248#[doc = crate::_doc_meta!{location("data/access/iter")}]
249///
250/// This is the borrowing analogue of : [`IteratorDoubleEnded`][crate::IteratorDoubleEnded].
251pub trait IteratorLendingDoubleEnded: IteratorLending {
252    /// Returns the next item from the back of the iterator.
253    fn next_back<'a>(&'a mut self) -> Option<Self::Item<'a>>;
254
255    /* non-required methods*/
256
257    /// Skips `n` items from the back and returns the next one.
258    ///
259    /// Repeatedly calls [`next_back`][Self::next_back] `n + 1` times.
260    /// Returns the final yielded item, or `None` if the iterator ends before advancing `n` steps.
261    ///
262    /// This is the reverse analogue of [`IteratorLending::nth`].
263    fn nth_back<'a>(&'a mut self, mut n: usize) -> Option<Self::Item<'a>> {
264        while n > 0 {
265            is![self.next_back().is_none(), return None];
266            n -= 1;
267        }
268        self.next_back()
269    }
270
271    // ✓ rposition
272
273    /// Reverse-direction equivalent of [`IteratorLending::try_fold`].
274    ///
275    /// Applies `f` starting from the back of the iterator, folding
276    /// items in reverse order. Stops on first `Err`.
277    fn try_rfold<B, E, F>(&mut self, mut acc: B, mut f: F) -> Result<B, E>
278    where
279        F: FnMut(B, Self::Item<'_>) -> Result<B, E>,
280    {
281        while let Some(item) = self.next_back() {
282            acc = f(acc, item)?;
283        }
284        Ok(acc)
285    }
286}

288#[doc = crate::_tags!(iterator lifetime)]
289/// A lending iterator with a known remaining length.
290#[doc = crate::_doc_meta!{location("data/access/iter")}]
291///
292/// This is the borrowing analogue of [`IteratorExactSize`][crate::IteratorExactSize].
293///
294/// # Contract
295/// Implementors must return a [`size_hint`][IteratorLending::size_hint] where:
296/// - the lower bound equals the upper bound, and
297/// - both equal the exact number of items remaining.
298///
299/// In other words, `size_hint()` must return `(len, Some(len))` at all times.
300pub trait IteratorLendingExactSize: IteratorLending {
301    /* non-required methods */
302
303    /// Returns the exact number of items remaining.
304    ///
305    /// The default implementation derives this from
306    /// [`size_hint`][IteratorLending::size_hint], matching the standard library.
307    ///
308    /// # Panics
309    /// In debug builds, panics if the upper bound returned by `size_hint` is not `Some(lower)`.
310    fn len(&self) -> usize {
311        let (lower, upper) = self.size_hint();
312        debug_assert!(upper == Some(lower));
313        lower
314    }
315
316    /// Returns `true` when no more items can be produced.
317    fn is_empty(&self) -> bool {
318        self.len() == 0
319    }
320}

323/// A lending iterator that can inspect the next item without advancing.
324#[doc = crate::_doc_meta!{location("data/access/iter")}]
325///
326/// The returned reference is tied to the temporary mutable borrow of `self`
327/// created by this call. Implementations must not advance the iteration state
328/// (i.e., the next call to [`IteratorLending::next`] yields the same item, if any).
329///
330/// Implementations may internally buffer the next item to support peeking,
331/// but the externally observable iteration order must be unchanged.
332pub trait IteratorLendingPeek: IteratorLending {
333    /// Returns the next item without advancing the iterator.
334    ///
335    /// Borrows `self` mutably so that both shared and exclusive lending
336    /// iterators can expose a peek. The reference remains valid until
337    /// `self` is borrowed again.
338    fn peek<'a>(&'a mut self) -> Option<Self::Item<'a>>;
339
340    /* non-required */
341
342    /// Returns the next item if it matches the predicate.
343    ///
344    /// The iterator is advanced only when `pred` returns `true` for the
345    /// next item. Otherwise, the iterator remains unchanged.
346    fn next_if<F>(&mut self, pred: F) -> Option<Self::Item<'_>>
347    where
348        for<'a> F: FnOnce(&Self::Item<'a>) -> bool,
349    {
350        is![self.peek().is_some_and(|i| pred(&i)), self.next(), None]
351    }
352
353    /// Returns the next item if it is equal to `expected`.
354    ///
355    /// This is a convenience wrapper around [`next_if`][Self::next_if] using `PartialEq`.
356    fn next_if_eq<Q>(&mut self, expected: &Q) -> Option<Self::Item<'_>>
357    where
358        for<'a> Self::Item<'a>: PartialEq<Q>,
359        Q: ?Sized,
360    {
361        self.next_if(|item| item == expected)
362    }
363
364    /// Applies `f` to the next item without advancing the iterator unless `f` returns `Ok`.
365    ///
366    /// If `f` returns:
367    /// - `Ok(r)` — the iterator is advanced and `Ok(Some(r))` is returned.
368    /// - `Err(e)` — the iterator is left unchanged and `Err(e)` is returned.
369    /// If the iterator is exhausted, `Ok(None)` is returned.
370    ///
371    /// `f` receives a temporary borrow of the next item.
372    /// The borrow ends before this method calls [`IteratorLending::next`],
373    /// so advancing the iterator is always sound.
374    fn next_if_map<R, E, F>(&mut self, f: F) -> Result<Option<R>, E>
375    where
376        F: FnOnce(&Self::Item<'_>) -> Result<R, E>,
377    {
378        match self.peek().map(|item| f(&item)) {
379            None => Ok(None),
380            Some(Ok(r)) => {
381                self.next();
382                Ok(Some(r))
383            }
384            Some(Err(e)) => Err(e),
385        }
386    }
387}

390/// A lending iterator that can inspect the next item from the back, without advancing.
391#[doc = crate::_doc_meta!{location("data/access/iter")}]
392///
393/// The returned reference is tied to the temporary mutable borrow of `self`
394/// created by this call. Implementations must not modify the iteration state
395/// (i.e., the next call to [`IteratorLendingDoubleEnded::next_back`]
396/// will yield the same item as it would have without peeking).
397///
398/// Implementations may internally buffer the back item to support peeking,
399/// but the externally observable iteration order must be unchanged.
400pub trait IteratorLendingPeekDoubleEnded: IteratorLendingPeek + IteratorLendingDoubleEnded {
401    /// Returns the next item from the back, without advancing the iterator.
402    ///
403    /// Borrows `self` mutably so that both shared and exclusive lending
404    /// iterators can expose a peek. The reference remains valid until
405    /// `self` is borrowed again.
406    fn peek_back<'a>(&'a mut self) -> Option<Self::Item<'a>>;
407
408    /* non-required */
409
410    /// Returns the next item from the back if it matches the predicate.
411    ///
412    /// The iterator is advanced only when `pred` returns `true` for the
413    /// next item. Otherwise, the iterator remains unchanged.
414    fn next_back_if<F>(&mut self, pred: F) -> Option<Self::Item<'_>>
415    where
416        for<'a> F: FnOnce(&Self::Item<'a>) -> bool,
417    {
418        is![self.peek_back().is_some_and(|i| pred(&i)), self.next_back(), None]
419    }
420
421    /// Returns the next item from the back if it is equal to `expected`.
422    ///
423    /// This is a convenience wrapper around [`next_back_if`][Self::next_back_if] using `PartialEq`.
424    fn next_back_if_eq<Q>(&mut self, expected: &Q) -> Option<Self::Item<'_>>
425    where
426        for<'a> Self::Item<'a>: PartialEq<Q>,
427        Q: ?Sized,
428    {
429        self.next_back_if(|item| item == expected)
430    }
431
432    /// Applies `f` to the next item from the back
433    /// without advancing the iterator unless `f` returns `Ok`.
434    ///
435    /// If `f` returns:
436    /// - `Ok(r)` — the iterator is advanced and `Ok(Some(r))` is returned.
437    /// - `Err(e)` — the iterator is left unchanged and `Err(e)` is returned.
438    /// If the iterator is exhausted, `Ok(None)` is returned.
439    ///
440    /// `f` receives a temporary borrow of the next item from the back.
441    /// The borrow ends before this method calls [`IteratorLendingDoubleEnded::next_back`],
442    /// so advancing the iterator is always sound.
443    fn next_back_if_map<R, E, F>(&mut self, f: F) -> Result<Option<R>, E>
444    where
445        F: FnOnce(&Self::Item<'_>) -> Result<R, E>,
446    {
447        match self.peek_back().map(|item| f(&item)) {
448            None => Ok(None),
449            Some(Ok(r)) => {
450                self.next_back();
451                Ok(Some(r))
452            }
453            Some(Err(e)) => Err(e),
454        }
455    }
