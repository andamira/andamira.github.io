6/// # Write methods
7impl<'a> ByteCursor<&'a mut [u8]> {
8    /// Creates a write cursor at position `0`.
9    #[must_use]
10    #[inline(always)]
11    pub const fn new(storage: &'a mut [u8]) -> Self {
12        Self { storage, pos: 0 }
13    }
14
15    /// Creates a write cursor at position `0`.
16    ///
17    /// This is an explicit alias for cases where `ByteCursor::new` inference
18    /// would be less clear.
19    #[must_use]
20    #[inline(always)]
21    pub const fn writer(storage: &'a mut [u8]) -> Self {
22        Self::new(storage)
23    }
24    /// Creates a write cursor at `pos`.
25    ///
26    /// The position is not rejected here. Write methods return
27    /// `NotEnoughSpace` when the position cannot satisfy the requested
28    /// operation.
29    #[must_use]
30    #[inline(always)]
31    pub const fn at(storage: &'a mut [u8], pos: usize) -> Self {
32        Self { storage, pos }
33    }
34
35    /// Returns the underlying byte slice as shared.
36    #[must_use]
37    #[inline(always)]
38    pub const fn as_slice(&self) -> &[u8] {
39        self.storage
40    }
41    /// Returns the underlying byte slice as mutable.
42    ///
43    /// The returned borrow is tied to `&mut self`, not to the original storage
44    /// lifetime. This keeps the cursor from lending a long-lived mutable window
45    /// while it still owns the exclusive storage capability.
46    #[must_use]
47    #[inline(always)]
48    pub const fn as_mut_slice(&mut self) -> &mut [u8] {
49        self.storage
50    }
51
52    /// Returns the total storage length.
53    #[must_use]
54    #[inline(always)]
55    pub const fn len(&self) -> usize {
56        self.storage.len()
57    }
58    /// Returns whether the storage is empty.
59    #[must_use]
60    #[inline(always)]
61    pub const fn is_empty(&self) -> bool {
62        self.storage.is_empty()
63    }
64    /// Returns the remaining writable byte count, saturating at `0`.
65    ///
66    /// If `pos > len`, this returns `0`.
67    #[must_use]
68    #[inline(always)]
69    pub const fn remaining_len(&self) -> usize {
70        let len = self.storage.len();
71        len.saturating_sub(self.pos)
72    }
73
74    /// Returns whether exactly `len` bytes can be written at the current
75    /// position.
76    #[must_use]
77    #[inline(always)]
78    pub const fn can_write(&self, len: usize) -> bool {
79        self.can_write_at(self.pos, len)
80    }
81    /// Returns whether exactly `len` bytes can be written at `pos`.
82    #[must_use]
83    #[inline(always)]
84    pub const fn can_write_at(&self, pos: usize, len: usize) -> bool {
85        let total = self.storage.len();
86        pos <= total && len <= total - pos
87    }
88
89    /// Sets the position if it is inside the slice bounds.
90    #[inline(always)]
91    pub const fn try_set_pos(&mut self, pos: usize) -> Result<(), IndexOutOfBounds> {
92        if pos <= self.storage.len() {
93            self.pos = pos;
94            Ok(())
95        } else {
96            Err(IndexOutOfBounds(Some(pos)))
97        }
98    }
99    /// Sets the position, clamped to the slice length.
100    ///
101    /// Returns the new position.
102    #[inline(always)]
103    pub const fn set_pos_clamped(&mut self, pos: usize) -> usize {
104        let len = self.storage.len();
105        self.pos = if pos <= len { pos } else { len };
106        self.pos
107    }
108
109    /// Advances by up to `len` bytes, saturating at the end.
110    ///
111    /// Returns the new position.
112    ///
113    /// Use [`skip_exact`](#method.skip_exact)
114    /// when insufficient space should be treated as an error.
115    #[inline(always)]
116    pub const fn advance(&mut self, len: usize) -> usize {
117        let total = self.storage.len();
118        if self.pos > total {
119            self.pos = total;
120            return self.pos;
121        }
122        let rem = total - self.pos;
123        self.pos += if len <= rem { len } else { rem };
124        self.pos
125    }
126    /// Advances by exactly `len` bytes.
127    #[inline(always)]
128    pub const fn skip_exact(&mut self, len: usize) -> Result<(), NotEnoughSpace> {
129        if self.can_write(len) {
130            self.pos += len;
131            Ok(())
132        } else {
133            Err(NotEnoughSpace(Some(len)))
134        }
135    }

139    /// Writes a dynamic byte slice and advances by its length.
140    #[inline]
141    pub const fn write(&mut self, bytes: &[u8]) -> Result<(), NotEnoughSpace> {
142        let len = bytes.len();
143        is! { !self.can_write(len), return Err(NotEnoughSpace(Some(len))) }
144        let start = self.pos;
145        whilst! { i in 0..len; { self.storage[start + i] = bytes[i]; }}
146        self.pos += len;
147        Ok(())
148    }
149    /// Writes one byte and advances by `1`.
150    #[inline(always)]
151    pub const fn write_u8(&mut self, byte: u8) -> Result<(), NotEnoughSpace> {
152        if self.can_write(1) {
153            write_at![self.storage, +=self.pos, byte];
154            Ok(())
155        } else {
156            Err(NotEnoughSpace(Some(1)))
157        }
158    }
159    /// Writes two bytes and advances by `2`.
160    #[inline(always)]
161    pub const fn write_2(&mut self, bytes: [u8; 2]) -> Result<(), NotEnoughSpace> {
162        unwrap![ok? self.write_at_2(self.pos, bytes)];
163        self.pos += 2;
164        Ok(())
165    }
166    /// Writes four bytes and advances by `4`.
167    #[inline(always)]
168    pub const fn write_4(&mut self, bytes: [u8; 4]) -> Result<(), NotEnoughSpace> {
169        unwrap![ok? self.write_at_4(self.pos, bytes)];
170        self.pos += 4;
171        Ok(())
172    }
173    /// Writes eight bytes and advances by `8`.
174    #[inline(always)]
175    pub const fn write_8(&mut self, bytes: [u8; 8]) -> Result<(), NotEnoughSpace> {
176        unwrap![ok? self.write_at_8(self.pos, bytes)];
177        self.pos += 8;
178        Ok(())
179    }
180
181    /// Writes a little-endian `u16`.
182    #[inline(always)]
183    pub const fn write_u16_le(&mut self, value: u16) -> Result<(), NotEnoughSpace> {
184        self.write_2(value.to_le_bytes())
185    }
186    /// Writes a big-endian `u16`.
187    #[inline(always)]
188    pub const fn write_u16_be(&mut self, value: u16) -> Result<(), NotEnoughSpace> {
189        self.write_2(value.to_be_bytes())
190    }
191    /// Writes a little-endian `u32`.
192    #[inline(always)]
193    pub const fn write_u32_le(&mut self, value: u32) -> Result<(), NotEnoughSpace> {
194        self.write_4(value.to_le_bytes())
195    }
196    /// Writes a big-endian `u32`.
197    #[inline(always)]
198    pub const fn write_u32_be(&mut self, value: u32) -> Result<(), NotEnoughSpace> {
199        self.write_4(value.to_be_bytes())
200    }
201    /// Writes a little-endian `u64`.
202    #[inline(always)]
203    pub const fn write_u64_le(&mut self, value: u64) -> Result<(), NotEnoughSpace> {
204        self.write_8(value.to_le_bytes())
205    }
206    /// Writes a big-endian `u64`.
207    #[inline(always)]
208    pub const fn write_u64_be(&mut self, value: u64) -> Result<(), NotEnoughSpace> {
209        self.write_8(value.to_be_bytes())
210    }

214    /// Writes a dynamic byte slice at `pos` without advancing.
215    #[inline]
216    pub const fn write_at(&mut self, pos: usize, bytes: &[u8]) -> Result<(), NotEnoughSpace> {
217        let len = bytes.len();
218        is! { !self.can_write_at(pos, len), return Err(NotEnoughSpace(Some(len))) }
219        whilst! { i in 0..len; { self.storage[pos + i] = bytes[i]; }}
220        Ok(())
221    }
222    /// Writes two bytes at `pos` without advancing.
223    #[inline(always)]
224    pub const fn write_at_2(&mut self, pos: usize, bytes: [u8; 2]) -> Result<(), NotEnoughSpace> {
225        if self.can_write_at(pos, 2) {
226            write_at![self.storage, pos, @2 bytes];
227            Ok(())
228        } else {
229            Err(NotEnoughSpace(Some(2)))
230        }
231    }
232    /// Writes four bytes at `pos` without advancing.
233    #[inline(always)]
234    pub const fn write_at_4(&mut self, pos: usize, bytes: [u8; 4]) -> Result<(), NotEnoughSpace> {
235        if self.can_write_at(pos, 4) {
236            write_at![self.storage, pos, @4 bytes];
237            Ok(())
238        } else {
239            Err(NotEnoughSpace(Some(4)))
240        }
241    }
242    /// Writes eight bytes at `pos` without advancing.
243    #[inline(always)]
244    pub const fn write_at_8(&mut self, pos: usize, bytes: [u8; 8]) -> Result<(), NotEnoughSpace> {
245        if self.can_write_at(pos, 8) {
246            write_at![self.storage, pos, @8 bytes];
247            Ok(())
248        } else {
249            Err(NotEnoughSpace(Some(8)))
250        }
251    }

253    /// Writes a little-endian `u32` at `pos` without advancing.
254    ///
255    /// Useful for patching size fields after writing a payload.
256    #[inline(always)]
257    pub const fn write_at_u32_le(&mut self, pos: usize, value: u32) -> Result<(), NotEnoughSpace> {
258        self.write_at_4(pos, value.to_le_bytes())
259    }

262    /// Useful for patching size fields after writing a payload.
263    #[inline(always)]
264    pub const fn write_at_u32_be(&mut self, pos: usize, value: u32) -> Result<(), NotEnoughSpace> {
265        self.write_at_4(pos, value.to_be_bytes())
