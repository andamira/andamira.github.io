6use super::raw;

9#[doc = crate::_tags!(platform namespace)]
10/// Raw C/POSIX system interfaces exposed through the platform's libc.
11#[doc = crate::_doc_meta!{location("sys/os")}]
12///
13/// Thin, unsafe bindings that provide direct access to C ABI functions
14/// for memory mapping, shared memory, file descriptors, and other
15/// low-level operations. These are unmodified system calls as provided
16/// by the host libc.
17///
18/// - <https://www.gnu.org/software/libc/manual/html_node>
19#[derive(Debug)]
20pub struct Libc;
21
22/// Constants from POSIX and libc.
23impl Libc {
24    /// POSIX open flag: open for read-only access.
25    pub const O_RDONLY: c_int = 0o0;
26    /// POSIX open flag: open for write-only access.
27    pub const O_WRONLY: c_int = 0o1;
28    /// POSIX open flag: open for reading and writing.
29    pub const O_RDWR: c_int = 0o2;
30    /// POSIX open flag: create the file if it does not exist.
31    pub const O_CREAT: c_int = 0o100;
32
33    /// POSIX mmap protection: pages are readable.
34    pub const PROT_READ: c_int = 0x1;
35    /// POSIX mmap protection: pages are writable.
36    pub const PROT_WRITE: c_int = 0x2;
37
38    /// POSIX mmap flag: mapping is shared with other processes.
39    pub const MAP_SHARED: c_int = 0x01;
40    /// POSIX mmap failure sentinel returned on error.
41    pub const MAP_FAILED: *mut c_void = !0 as *mut c_void;
42    /// POSIX mmap flag: mapping is private to this process.
43    pub const MAP_PRIVATE: c_int = 0x02;
44    /// Linux mmap flag: mapping is not backed by a file.
45    pub const MAP_ANONYMOUS: c_int = 0x20;
46    /// Linux mmap flag: do not reserve swap space for this mapping.
47    pub const MAP_NORESERVE: c_int = 0x4000;
48}
49
50/// Convenience helpers for libc return values.
51impl Libc {
52    /// Returns `true` if `ptr` matches the POSIX `MAP_FAILED` sentinel.
53    #[inline(always)]
54    pub fn is_map_failed(ptr: *mut c_void) -> bool {
55        ptr == Self::MAP_FAILED
56    }
57}
58
59/// Direct wrappers around libc/POSIX calls.
60#[allow(clippy::missing_safety_doc)]
61impl Libc {
62    /// Create/open POSIX shared memory.
63    /// - <https://www.man7.org/linux/man-pages/man3/shm_open.3.html>
64    #[inline(always)]
65    pub unsafe fn shm_open(name: *const c_char, oflag: c_int, mode: c_mode_t) -> c_int {
66        unsafe { raw::shm_open(name, oflag, mode) }
67    }
68
69    /// Unlink POSIX shared memory.
70    /// - <https://www.man7.org/linux/man-pages/man3/shm_open.3.html>
71    #[inline(always)]
72    pub unsafe fn shm_unlink(name: *const c_char) -> c_int {
73        unsafe { raw::shm_unlink(name) }
74    }
75
76    /// Truncate a file to a specified length.
77    /// - <https://man7.org/linux/man-pages/man3/ftruncate.3p.html>
78    #[inline(always)]
79    pub unsafe fn ftruncate(fd: c_int, length: c_off_t) -> c_int {
80        unsafe { raw::ftruncate(fd, length) }
81    }
82
83    /// Map pages of memory.
84    /// - <https://man7.org/linux/man-pages/man3/mmap.3p.html>
85    // /// - <https://man7.org/linux/man-pages/man2/mmap.2.html>
86    #[inline(always)]
87    pub unsafe fn mmap(
88        addr: *mut c_void,
89        length: usize,
90        prot: c_int,
91        flags: c_int,
92        fd: c_int,
93        offset: c_off_t,
94    ) -> *mut c_void {
95        unsafe { raw::mmap(addr, length, prot, flags, fd, offset) }
96    }
97
98    ///  Unmap pages of memory.
99    /// - <https://man7.org/linux/man-pages/man3/munmap.3p.html>
100    // /// - <https://man7.org/linux/man-pages/man2/mmap.2.html>
101    #[inline(always)]
102    pub unsafe fn munmap(addr: *mut c_void, length: usize) -> c_int {
103        unsafe { raw::munmap(addr, length) }
104    }
105
106    /// Free allocated memory.
107    /// - <https://man7.org/linux/man-pages/man3/free.3p.html>
108    #[inline(always)]
109    pub unsafe fn free(ptr: *mut c_void) {
110        unsafe { raw::free(ptr) }
111    }
112
113    /// Closes a file descriptor.
114    /// - <https://man7.org/linux/man-pages/man3/close.3p.html>
115    // /// - <https://man7.org/linux/man-pages/man2/close.2.html>
116    #[inline(always)]
117    pub unsafe fn close(fd: c_int) -> c_int {
118        unsafe { raw::close(fd) }
119    }
120}
