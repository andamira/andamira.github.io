6use crate::{Cmp, Position2, RegionS2, TermGrid, whilst};

8#[inline] #[rustfmt::skip]
9const fn clip_region(region: RegionS2<usize>, width: usize, height: usize)
10    -> Option<(usize, usize, usize, usize)> {
11    let [x, y] = region.pos.dim;
12    let [w, h] = region.ext.dim;
13    if x >= width || y >= height || w == 0 || h == 0 { return None; }
14    let x1 = Cmp(x.saturating_add(w)).min(width);
15    let y1 = Cmp(y.saturating_add(h)).min(height);
16    Some((x, y, x1, y1))
17}

19/// # Region drawing
20#[rustfmt::skip]
21impl<E: Copy, S: AsRef<[E]> + AsMut<[E]>> TermGrid<E, S> {
22    /// Fills the part of `region` lying within the grid.
23    ///
24    /// Regions extending beyond the grid are clipped.
25    pub fn fill_region(&mut self, region: RegionS2<usize>, element: impl Into<E>) {
26        let Some((x0, y0, x1, y1)) = clip_region(region, self.width(), self.height())
27            else { return; };
28        let element = element.into();
29        whilst! { y in y0,..y1; {
30            self.row_mut(y).unwrap()[x0..x1].fill(element);
31        }}
32    }
33    /// Fills a clipped horizontal span.
34    pub fn hline(&mut self, x: usize, y: usize, len: usize, element: impl Into<E>) {
35        self.fill_region(RegionS2::from(((x, y), (len, 1))), element);
36    }
37    /// Fills a clipped vertical span.
38    pub fn vline(&mut self, x: usize, y: usize, len: usize, element: impl Into<E>) {
39        self.fill_region(RegionS2::from(((x, y), (1, len))), element);
40    }
41    /// Draws a clipped one-cell frame around `region`.
42    pub fn frame(&mut self, region: RegionS2<usize>, element: impl Into<E>) {
43        let [x, y] = region.pos.dim;
44        let [w, h] = region.ext.dim;
45        if w == 0 || h == 0 { return; }
46        let element = element.into();
47        self.hline(x, y, w, element);
48        if h > 1 { self.hline(x, y.saturating_add(h - 1), w, element); }
49        if h > 2 {
50            let inner_y = y.saturating_add(1);
51            self.vline(x, inner_y, h - 2, element);
52            if w > 1 { self.vline( x.saturating_add(w - 1), inner_y, h - 2, element); }
53        }
54    }
55}

57/// # Grid transfer
58#[rustfmt::skip]
59impl<E: Copy, D> TermGrid<E, D> where D: AsRef<[E]> + AsMut<[E]> {
60    /// Copies `source` into this grid at `destination`.
61    ///
62    /// Cells extending beyond the destination grid are clipped.
63    pub fn blit_at<S>(&mut self, source: &TermGrid<E, S>, destination: Position2<usize>)
64    where S: AsRef<[E]> {
65        self.blit_region_at(source,
66            RegionS2::from(((0, 0), (source.width(), source.height()))), destination);
67    }
68    /// Copies a source region into this grid at `destination`.
69    ///
70    /// Both the source region and destination grid are clipped.
71    pub fn blit_region_at<S>(&mut self, source: &TermGrid<E, S>,
72        source_region: RegionS2<usize>, destination: Position2<usize>) where S: AsRef<[E]> {
73        let [src_x, src_y] = source_region.pos.dim;
74        let [req_w, req_h] = source_region.ext.dim;
75        let [dst_x, dst_y] = destination.dim;
76        if src_x >= source.width() || src_y >= source.height()
77            || dst_x >= self.width() || dst_y >= self.height() { return; }
78        let width = req_w.min(source.width() - src_x).min(self.width() - dst_x);
79        let height = req_h.min(source.height() - src_y).min(self.height() - dst_y);
80        if width == 0 || height == 0 { return; }
81        whilst! { y in 0..height; {
82            let src = &source.row(src_y + y).unwrap()[src_x..src_x + width];
83            let dst = &mut self.row_mut(dst_y + y).unwrap()[dst_x..dst_x + width];
84            dst.copy_from_slice(src);
85        }}
86    }
87}
