3//! Defines [`TermelMeta`] and [`TermelOccupancy`].

6use TermelOccupancy::{Continuation, Origin, Reserved, Unoccupied};
7
8#[doc = crate::_tags!(term text layout bit)]
9/// The cell-space occupancy represented by terminal-element metadata.
10#[doc = crate::_doc_meta!{location("sys/os/term/grid")}]
11#[repr(u8)]
12#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
13pub enum TermelOccupancy {
14    /// No terminal element occupies this position.
15    #[default]
16    Unoccupied = 0,
17    /// The position begins a terminal element.
18    Origin = 1,
19    /// The position continues an earlier terminal element.
20    Continuation = 2,
21    /// Reserved for future extension.
22    Reserved = 3,
23}
24impl TermelOccupancy {
25    /// Creates an occupancy kind from its packed representation.
26    #[must_use]
27    pub const fn from_u8(value: u8) -> Self {
28        match value {
29            0 => Self::Unoccupied,
30            1 => Self::Origin,
31            2 => Self::Continuation,
32            _ => Self::Reserved,
33        }
34    }
35}
36
37crate::bitfield! {
38    #[doc = crate::_tags!(term text layout bit)]
39    /// Compact terminal-cell occupancy metadata for a [`Termel`][crate::Termel].
40    #[doc = crate::_doc_meta!{
41        location("sys/os/term/grid"),
42        test_size_of(TermelMeta = 1),
43    }]
44    ///
45    /// An origin stores its occupied column width.
46    /// A continuation stores its backward offset to the origin.
47    ///
48    /// Widths and offsets range from 1 to 15 columns.
49    #[must_use]
50    pub struct TermelMeta(u8) {
51        /// The [`TermelOccupancy`] discriminant.
52        KIND = 0..=1;
53        /// The origin width or continuation offset.
54        VALUE = 2..=5;
55    }
56    impl {
57        /// The maximum representable width or backward offset.
58        pub const MAX_SPAN: u8 = 15;
59
60        /// Creates metadata for an element origin.
61        ///
62        /// Returns `None` when `width` is zero or exceeds [`Self::MAX_SPAN`].
63        #[must_use]
64        pub const fn origin(width: u8) -> Option<Self> {
65            if width == 0 || width > Self::MAX_SPAN { None }
66            else { Some( Self::new().with_kind(Origin as u8).with_value(width)) }
67        }
68        /// Creates metadata for an element continuation.
69        ///
70        /// `origin_offset` is the number of columns back to the origin.
71        ///
72        /// Returns `None` when it is zero or exceeds [`Self::MAX_SPAN`].
73        #[must_use]
74        pub const fn continuation(origin_offset: u8) -> Option<Self> {
75            if origin_offset == 0 || origin_offset > Self::MAX_SPAN { None }
76            else { Some( Self::new().with_kind(Continuation as u8).with_value(origin_offset)) }
77        }
78        /// Returns the occupancy kind.
79        #[must_use]
80        pub const fn occupancy(self) -> TermelOccupancy {
81            TermelOccupancy::from_u8(self.get_kind())
82        }
83
84        /// Returns whether this position is occupied.
85        pub const fn is_occupied(self) -> bool {
86            matches!( self.occupancy(), Origin | Continuation)
87        }
88        /// Returns whether this position is empty.
89        #[must_use]
90        pub const fn is_unoccupied(self) -> bool { matches!(self.occupancy(), Unoccupied) }
91
92        /// Returns whether this position begins an element.
93        #[must_use]
94        pub const fn is_origin(self) -> bool { matches!(self.occupancy(), Origin) }
95        /// Returns whether this position continues an earlier element.
96        #[must_use]
97        pub const fn is_continuation(self) -> bool { matches!(self.occupancy(), Continuation) }
98
99        /// Returns the occupied width when this is an origin.
100        #[must_use]
101        pub const fn width(self) -> Option<u8> {
102            if self.is_origin() { Some(self.get_value()) } else { None }
103        }
104        /// Returns the backward distance to the origin when this is a continuation.
105        #[must_use]
106        pub const fn origin_offset(self) -> Option<u8> {
107            if self.is_continuation() { Some(self.get_value()) } else { None }
108        }
109        /// Returns whether this value has a valid canonical representation.
110        #[must_use]
111        pub const fn is_canonical(self) -> bool {
112            if self.bits() & !Self::mask() != 0 { return false; }
113            match self.occupancy() {
114                Unoccupied => self.get_value() == 0,
115                Origin | Continuation => { self.get_value() != 0 }
116                Reserved => false,
117            }
118        }
119    }
120}

122#[cfg(test)]
123mod tests {
124    use super::*;
125
126    #[test]
127    fn size() {
128        assert_eq!(size_of::<TermelMeta>(), 1);
129    }
130    #[test]
131    fn empty() {
132        let meta = TermelMeta::new();
133        assert!(meta.is_empty());
134        assert_eq!(meta.width(), None);
135        assert_eq!(meta.origin_offset(), None);
136        assert!(meta.is_canonical());
137    }
138    #[test]
139    fn origin() {
140        let meta = TermelMeta::origin(4).unwrap();
141        assert!(meta.is_origin());
142        assert_eq!(meta.width(), Some(4));
143        assert_eq!(meta.origin_offset(), None);
144        assert!(meta.is_canonical());
145    }
146    #[test]
147    fn continuation() {
148        let meta = TermelMeta::continuation(3).unwrap();
149        assert!(meta.is_continuation());
150        assert_eq!(meta.width(), None);
151        assert_eq!(meta.origin_offset(), Some(3));
152        assert!(meta.is_canonical());
153    }
154    #[test]
155    fn span_limits() {
156        assert_eq!(TermelMeta::origin(0), None);
157        assert_eq!(TermelMeta::origin(16), None);
158        assert_eq!(TermelMeta::continuation(0), None);
159        assert_eq!(TermelMeta::continuation(16), None);
160        assert_eq!(TermelMeta::origin(TermelMeta::MAX_SPAN).unwrap().width(), Some(15),);
161    }
162    #[test]
163    fn occupied_sequence() {
164        let cells = [
165            TermelMeta::origin(4).unwrap(),
166            TermelMeta::continuation(1).unwrap(),
167            TermelMeta::continuation(2).unwrap(),
168            TermelMeta::continuation(3).unwrap(),
169        ];
170        assert_eq!(cells[0].width(), Some(4));
171        assert_eq!(cells[1].origin_offset(), Some(1));
172        assert_eq!(cells[2].origin_offset(), Some(2));
173        assert_eq!(cells[3].origin_offset(), Some(3));
174    }
175}
