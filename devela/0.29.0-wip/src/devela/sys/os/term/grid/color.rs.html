9/// The stored representation of a terminal color.
10#[doc = crate::_doc_meta!{location("sys/os/term")}]
11#[repr(u8)]
12#[must_use]
13#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
14pub enum TermColorKind {
15    /// Uses the terminal's default color.
16    #[default]
17    Default = 0,
18    /// Uses an indexed terminal palette color.
19    Indexed = 1,
20    /// Uses a 24-bit RGB color.
21    Rgb = 2,
22    /// Reserved for future extension.
23    Reserved = 3,
24}
25impl TermColorKind {
26    /// Creates a color kind from its packed representation.
27    pub const fn from_u8(value: u8) -> Self {
28        match value {
29            0 => Self::Default,
30            1 => Self::Indexed,
31            2 => Self::Rgb,
32            _ => Self::Reserved,
33        }
34    }
35}

38/// The composition mode of one terminal color.
39#[doc = crate::_doc_meta!{location("sys/os/term")}]
40#[repr(u8)]
41#[must_use]
42#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
43pub enum TermColorMode {
44    /// Replaces the color beneath it.
45    #[default]
46    Opaque = 0,
47    /// Blends with the color beneath it.
48    Blend = 1,
49    /// Preserves the color beneath it.
50    Transparent = 2,
51    /// Selects a contrasting color from the resolved background.
52    Contrast = 3,
53}
54impl TermColorMode {
55    /// Creates a color mode from its packed representation.
56    pub const fn from_u8(value: u8) -> Self {
57        match value {
58            0 => Self::Opaque,
59            1 => Self::Blend,
60            2 => Self::Transparent,
61            _ => Self::Contrast,
62        }
63    }
64}

66crate::bitfield! {
67    #[doc = crate::_tags!(term color bit)]
68    /// A packed terminal color and its composition mode.
69    #[doc = crate::_doc_meta!{
70        location("sys/os/term"),
71        test_size_of(TermColor = 4|32),
72    }]
73    ///
74    /// The low 24 bits store either an RGB value or an indexed palette value.
75    /// The remaining defined bits select the representation and composition mode.
76    #[must_use]
77    pub struct TermColor(u32) {
78        /// RGB or indexed-color payload.
79        VALUE = 0..=23;
80        /// [`TermColorKind`] discriminant.
81        KIND = 24..=25;
82        /// [`TermColorMode`] discriminant.
83        MODE = 26..=27;
84    }
85    impl {
86        /// Uses the terminal's default color.
87        pub const DEFAULT: Self = Self::new();
88
89        /// Creates an opaque indexed terminal color.
90        pub const fn indexed(index: u8) -> Self {
91            Self::new().with_value(index as u32).with_kind(TermColorKind::Indexed as u32)
92        }
93        /// Creates an opaque 24-bit RGB terminal color.
94        pub const fn rgb(rgb: [u8; 3]) -> Self {
95            let [r, g, b] = rgb;
96            let value = ((r as u32) << 16) | ((g as u32) << 8) | b as u32;
97            Self::new().with_value(value).with_kind(TermColorKind::Rgb as u32)
98        }
99        /// Creates a color using the terminal default and the given mode.
100        pub const fn default_with_mode(mode: TermColorMode) -> Self {
101            Self::new().with_mode(mode as u32)
102        }
103        /// Returns the stored color representation.
104        pub const fn kind(self) -> TermColorKind {
105            TermColorKind::from_u8(self.get_kind() as u8)
106        }
107        /// Returns the color's composition mode.
108        pub const fn mode(self) -> TermColorMode {
109            TermColorMode::from_u8(self.get_mode() as u8)
110        }
111        /// Returns this color with the given composition mode.
112        pub const fn with_color_mode(self, mode: TermColorMode) -> Self {
113            self.with_mode(mode as u32)
114        }
115        /// Returns whether this uses the terminal's default color.
116        #[must_use]
117        pub const fn is_default(self) -> bool { matches!(self.kind(), TermColorKind::Default) }
118
119        /// Returns whether this uses an indexed terminal color.
120        #[must_use]
121        pub const fn is_indexed(self) -> bool { matches!(self.kind(), TermColorKind::Indexed) }
122
123        /// Returns whether this uses a 24-bit RGB color.
124        #[must_use]
125        pub const fn is_rgb(self) -> bool { matches!(self.kind(), TermColorKind::Rgb) }
126
127        /// Returns whether this color is opaque.
128        #[must_use]
129        pub const fn is_opaque(self) -> bool { matches!(self.mode(), TermColorMode::Opaque) }
130
131        /// Returns whether this color preserves the color beneath it.
132        #[must_use]
133        pub const fn is_transparent(self) -> bool {
134            matches!(self.mode(), TermColorMode::Transparent)
135        }
136        /// Returns the indexed palette value, when applicable.
137        #[must_use]
138        pub const fn index(self) -> Option<u8> {
139            if self.is_indexed() { Some(self.get_value() as u8) } else { None }
140        }
141        /// Returns the RGB components, when applicable.
142        #[must_use]
143        pub const fn rgb_components(self) -> Option<[u8; 3]> {
144            if self.is_rgb() {
145                let value = self.get_value();
146                Some([(value >> 16) as u8, (value >> 8) as u8, value as u8])
147            } else {
148                None
149            }
150        }
151        /// Returns whether all currently reserved bits are clear.
152        #[must_use]
153        pub const fn is_canonical(self) -> bool {
154            self.bits() & !Self::mask() == 0 && !matches!(self.kind(), TermColorKind::Reserved)
155        }
156        /// Returns this value with currently reserved bits cleared.
157        pub const fn canonicalized(self) -> Self { Self::from_bits(self.bits() & Self::mask()) }
158
159        /// Converts an ANSI color into a packed terminal color.
160        ///
161        /// Returns `None` for [`AnsiColor::None`], which represents absence
162        /// rather than a resolved terminal color.
163        #[must_use]
164        pub const fn from_ansi(color: AnsiColor) -> Option<Self> {
165            match color {
166                AnsiColor::None => None,
167                AnsiColor::Default => Some(Self::DEFAULT),
168                AnsiColor::Dark(color) => Some(Self::indexed(color as u8)),
169                AnsiColor::Bright(color) => Some(Self::indexed(color as u8 + 8)),
170                AnsiColor::Palette(color) => Some(Self::indexed(color.0)),
171                AnsiColor::Rgb(rgb) => Some(Self::rgb(rgb)),
172            }
173        }
174        /// Converts a 3-bit ANSI color into a packed terminal color.
175        pub const fn from_ansi3_dark(color: AnsiColor3) -> Self {
176            Self::indexed(color as u8)
177        }
178        /// Converts a 3-bit ANSI color into a packed terminal color.
179        pub const fn from_ansi3_bright(color: AnsiColor3) -> Self {
180            Self::indexed(color as u8 + 8)
181        }
182
183        /// Converts a 8-bit ANSI color into a packed terminal color.
184        pub const fn from_ansi8(color: AnsiColor8) -> Self { Self::indexed(color.0) }
185
186        /// Converts an opaque color into its ANSI representation.
187        ///
188        /// Non-opaque and reserved colors return `None`.
189        #[must_use]
190        pub const fn to_ansi(self) -> Option<AnsiColor> {
191            if !self.is_opaque() { return None; }
192            match self.kind() {
193                TermColorKind::Default => Some(AnsiColor::Default),
194                TermColorKind::Indexed => {
195                    let index = self.get_value() as u8;
196                    match index {
197                        0..=7 => Some(AnsiColor::Dark(AnsiColor3::from_u8(index))),
198                        8..=15 => {
199                            Some(AnsiColor::Bright(AnsiColor3::from_u8(index - 8)))
200                        }
201                        _ => Some(AnsiColor::Palette(AnsiColor8(index))),
202                    }
203                }
204                TermColorKind::Rgb => match self.rgb_components() {
205                    Some(rgb) => Some(AnsiColor::Rgb(rgb)),
206                    None => None,
207                },
208                TermColorKind::Reserved => None,
209            }
210        }
211    }
212}

214#[doc = crate::_tags!(term color)]
215/// Packed foreground and background terminal colors.
216#[doc = crate::_doc_meta!{
217    location("sys/os/term"),
218    test_size_of(TermColors = 8|64),
219}]
220///
221/// The foreground occupies the low 32 bits and the background the high 32 bits.
222#[must_use]
223#[repr(transparent)]
224#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
225pub struct TermColors {
226    bits: u64,
227}
228#[rustfmt::skip]
229impl TermColors {
230    /// Terminal-default foreground and background colors.
231    pub const DEFAULT: Self = Self::new(TermColor::DEFAULT, TermColor::DEFAULT);
232
233    /// Creates a foreground and background color pair.
234    pub const fn new(fg: TermColor, bg: TermColor) -> Self {
235        Self { bits: fg.bits() as u64 | ((bg.bits() as u64) << 32) }
236    }
237    /// Creates a color pair from its raw representation.
238    pub const fn from_bits(bits: u64) -> Self { Self { bits } }
239
240    /// Creates a foreground and background color pair from a pair of ANSI colors.
241    ///
242    /// Returns `None` for [`AnsiColor::None`], which represents absence
243    /// rather than a resolved terminal color.
244    pub const fn from_ansi(fg: AnsiColor, bg: AnsiColor) -> Option<Self> {
245        let (fg, bg) = (TermColor::from_ansi(fg), TermColor::from_ansi(bg));
246        Some(Self::new(unwrap![some? fg], unwrap![some? bg]))
247    }
248    /// Creates a foreground and background color pair from a pair of 3-bit ANSI colors.
249    pub const fn from_ansi3_dark(fg: AnsiColor3, bg: AnsiColor3) -> Self {
250        Self::new(TermColor::from_ansi3_dark(fg), TermColor::from_ansi3_dark(bg))
251    }
252    /// Creates a foreground and background color pair from a pair of 3-bit ANSI colors.
253    pub const fn from_ansi3_bright(fg: AnsiColor3, bg: AnsiColor3) -> Self {
254        Self::new(TermColor::from_ansi3_bright(fg), TermColor::from_ansi3_bright(bg))
255    }
256    /// Creates a foreground and background color pair from a pair of 8-bit ANSI colors.
257    pub const fn from_ansi8(fg: AnsiColor8, bg: AnsiColor8) -> Self {
258        Self::new(TermColor::from_ansi8(fg), TermColor::from_ansi8(bg))
259    }
260
261    /// Returns the raw packed representation.
262    #[must_use]
263    pub const fn bits(self) -> u64 { self.bits }
264
265    /// Returns the foreground color.
266    pub const fn fg(self) -> TermColor { TermColor::from_bits(self.bits as u32) }
267    /// Returns the background color.
268    pub const fn bg(self) -> TermColor { TermColor::from_bits((self.bits >> 32) as u32) }
269
270    /// Returns this pair with a new foreground color.
271    pub const fn with_fg(self, fg: TermColor) -> Self { Self::new(fg, self.bg()) }
272
273    /// Returns this pair with a new background color.
274    pub const fn with_bg(self, bg: TermColor) -> Self { Self::new(self.fg(), bg) }
275
276    /// Replaces the foreground color.
277    pub const fn set_fg(&mut self, fg: TermColor) { self.bits = Self::new(fg, self.bg()).bits; }
278    /// Replaces the background color.
279    pub const fn set_bg(&mut self, bg: TermColor) { self.bits = Self::new(self.fg(), bg).bits; }
280
281    /// Returns this pair with foreground and background exchanged.
282    pub const fn swapped(self) -> Self { Self::new(self.bg(), self.fg()) }
283
284    /// Exchanges the foreground and background colors.
285    pub const fn swap(&mut self) { self.bits = self.bits.rotate_left(32); }
286
287    /// Returns whether both colors are canonical.
288    #[must_use]
289    pub const fn is_canonical(self) -> bool { self.fg().is_canonical() && self.bg().is_canonical() }
290
291    /// Returns this pair with both colors canonicalized.
292    pub const fn canonicalized(self) -> Self {
293        Self::new(self.fg().canonicalized(), self.bg().canonicalized())
294    }
295}

297#[cfg(test)]
298mod tests {
299    use super::*;
300
301    #[test]
302    fn default_representation_is_zero() {
303        assert_eq!(TermColor::DEFAULT.bits(), 0);
304        assert_eq!(TermColors::DEFAULT.bits(), 0);
305        assert!(TermColor::DEFAULT.is_default());
306        assert!(TermColor::DEFAULT.is_opaque());
307    }
308    #[test]
309    fn indexed_roundtrip() {
310        let color = TermColor::indexed(173);
311        assert!(color.is_indexed());
312        assert_eq!(color.index(), Some(173));
313        assert_eq!(color.rgb_components(), None);
314        assert!(color.is_canonical());
315    }
316    #[test]
317    fn rgb_roundtrip() {
318        let color = TermColor::rgb([0x12, 0x34, 0x56]);
319        assert!(color.is_rgb());
320        assert_eq!(color.rgb_components(), Some([0x12, 0x34, 0x56]));
321        assert_eq!(color.get_value(), 0x12_34_56);
322        assert!(color.is_canonical());
323    }
324    #[test]
325    fn composition_mode_preserves_color() {
326        let color = TermColor::rgb([10, 20, 30]).with_color_mode(TermColorMode::Transparent);
327        assert_eq!(color.rgb_components(), Some([10, 20, 30]));
328        assert!(color.is_transparent());
329    }
330    #[test]
331    fn paired_colors() {
332        let fg = TermColor::indexed(7);
333        let bg = TermColor::rgb([1, 2, 3]);
334        let colors = TermColors::new(fg, bg);
335        assert_eq!(colors.fg(), fg);
336        assert_eq!(colors.bg(), bg);
337        assert_eq!(colors.swapped(), TermColors::new(bg, fg));
338    }
339    #[test]
340    fn ansi_roundtrip() {
341        let values = [
342            AnsiColor::Default,
343            AnsiColor::Dark(AnsiColor3::Red),
344            AnsiColor::Bright(AnsiColor3::Blue),
345            AnsiColor::Palette(AnsiColor8(203)),
346            AnsiColor::Rgb([10, 20, 30]),
347        ];
348        for ansi in values {
349            let color = TermColor::from_ansi(ansi).unwrap();
350            assert_eq!(color.to_ansi(), Some(ansi));
351        }
352        assert_eq!(TermColor::from_ansi(AnsiColor::None), None);
353    }
354    #[test]
355    fn reserved_bits_are_not_canonical() {
356        let color = TermColor::from_bits(0xF000_0000);
357        assert!(!color.is_canonical());
358        assert_eq!(color.canonicalized(), TermColor::DEFAULT);
359    }
