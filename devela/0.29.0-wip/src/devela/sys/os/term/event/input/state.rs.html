10/// Internal parser state.
11#[derive(Clone, Debug, Default)]
12pub(crate) enum TermInputState {
13    /// No partial sequence is active.
14    #[default]
15    Ground,
16    /// A single `ESC` byte has been received.
17    Esc,
18    /// An SS3 sequence introduced by `ESC O` is waiting for its final byte.
19    Ss3,
20    /// A CSI sequence introduced by `ESC [` is being collected.
21    Csi { buf: [u8; 16], len: u8 },
22    /// A UTF-8 scalar is being collected.
23    Utf8 { buf: [u8; 4], len: u8, need: u8 },
24}
25_impl_init! { Self::Ground => TermInputState }

45/// Const-safe CSI parser result.
46///
47/// Keeps CSI dispatch free of drop-bearing event types
48/// until the final conversion to [`TermParsed`].
49#[non_exhaustive]
50#[derive(Clone, Copy, Debug, PartialEq, Eq)]
51pub(crate) enum TermParsedCsi {
52    /// No CSI match; continue with the next parser layer.
53    Continue,
54    /// The sequence was handled, but produced no user-facing event.
55    Pending,
56    /// A keyboard event.
57    Key(Key),
58    /// A mouse event.
59    Mouse(EventMouse),
60    /// A wheel event.
61    Wheel(EventWheel),
62    /// Terminal focus was gained.
63    FocusGained,
64    /// Terminal focus was lost.
65    FocusLost,
66    /// An internal terminal reply.
67    Reply(TermReply),
68    /// A complete but unsupported CSI sequence.
69    Unknown,
70}
71impl TermParsedCsi {
72    /// Converts this handled CSI result into the general parser result.
73    ///
74    /// `Continue` maps to `Unknown`; callers should normally handle it before conversion.
75    pub(super) const fn to_term_parsed(self) -> TermParsed {
76        match self {
77            TermParsedCsi::Continue => TermParsed::Unknown,
78            TermParsedCsi::Pending => TermParsed::Pending,
79            TermParsedCsi::Key(key) => TermParsed::Event(super::TermInputParser::key(key)),
80            TermParsedCsi::Mouse(mouse) => TermParsed::Event(EventKind::Mouse(mouse)),
81            TermParsedCsi::Wheel(wheel) => TermParsed::Event(EventKind::Wheel(wheel)),
82            TermParsedCsi::FocusGained => {
83                TermParsed::Event(EventKind::Window(EventWindow::FocusGained))
84            }
85            TermParsedCsi::FocusLost => {
86                TermParsed::Event(EventKind::Window(EventWindow::FocusLost))
87            }
88            TermParsedCsi::Reply(reply) => TermParsed::Reply(reply),
89            TermParsedCsi::Unknown => TermParsed::Unknown,
90        }
91    }
92}

95/// Terminal reply parsed from the input stream.
96#[derive(Clone, Copy, Debug, PartialEq, Eq)]
97pub(crate) enum TermReply {
98    /// Cursor-position report: `ESC [ row ; col R`.
99    ///
100    /// The position is in terminal cells, meaning x = column and y = row.
101    CursorPosition(Position2<u16>),
102    /// Device-attributes reply: `ESC [ ... c`.
103    DeviceAttributes,
104    /// DEC private mode report: `ESC [ ? mode ; status $ y`.
105    DecPrivateMode { mode: u16, status: TermDecModeStatus },
106}
107
108///
109#[derive(Clone, Copy, Debug, PartialEq, Eq)]
110#[repr(u8)]
111pub(crate) enum TermDecModeStatus {
112    NotRecognized = 0,
113    Set = 1,
114    Reset = 2,
115    PermanentlySet = 3,
116    PermanentlyReset = 4,
117}
118impl TermDecModeStatus {
119    pub(crate) const fn from_u16(value: u16) -> Option<Self> {
120        match value {
121            0 => Some(Self::NotRecognized),
122            1 => Some(Self::Set),
123            2 => Some(Self::Reset),
124            3 => Some(Self::PermanentlySet),
125            4 => Some(Self::PermanentlyReset),
126            _ => None,
127        }
128    }
129    pub(crate) const fn is_supported(self) -> bool {
130        !matches!(self, Self::NotRecognized)
131    }
