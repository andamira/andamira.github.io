6use crate::Ansi;
7
8/// The state of an Ansi code for [`Ansi::strip_codes`].
9#[derive(Clone, Copy, PartialEq)]
10enum State {
11    Normal, // Copying text to output
12    Escape, // Saw ESC, determining sequence type
13    Csi,    // In CSI sequence (ESC[)
14    Osc,    // In OSC sequence (ESC])
15}
16
17impl Ansi {
18    /// Removes ANSI escape sequences from a byte slice and writes the clean text to a buffer.
19    ///
20    /// This function processes well-formed ANSI escape sequences as defined by ECMA-48,
21    /// including CSI (Control Sequence Introducer) sequences like `ESC [ ... m` and
22    /// OSC (Operating System Command) sequences like `ESC ] ... BEL`.
23    ///
24    /// # Behavior
25    ///
26    /// - **Strips valid sequences**: Complete, well-formed ANSI sequences are removed.
27    /// - **Preserves malformed content**: Invalid sequences or partial sequences are
28    ///   passed through as-is to avoid data loss.
29    /// - **Handles partial buffers**: Works correctly if buffers end mid-sequence.
30    /// - **No allocations**: Uses only the provided buffers.
31    ///
32    /// # Supported Sequences
33    ///
34    /// - CSI sequences: `ESC [ P... I... F` where:
35    ///   - `P...` = parameters (digits, semicolons)
36    ///   - `I...` = intermediate bytes (rare)
37    ///   - `F` = final byte (command: 0x40-0x7E)
38    /// - OSC sequences: `ESC ] ... (BEL or ESC \)`
39    /// - Single-character escapes (stripped)
40    ///
41    /// # Returns
42    /// The number of bytes written to `dest`. This will always be ≤ the original text length.
43    ///
44    /// # Examples
45    /// ```
46    /// # use devela::Ansi;
47    /// let input = b"Hello \x1B[32mWorld\x1B[0m!";
48    /// let mut output = [0u8; 20];
49    /// let len = Ansi::strip_codes(input, &mut output);
50    /// assert_eq!(&output[0..len], b"Hello World!");
51    /// ```
52    ///
53    /// # Notes
54    /// - Malformed sequences (e.g., `ESC` in middle of parameters) are preserved as text.
55    /// - Lone `ESC` bytes (0x1B) without valid sequence syntax are stripped.
56    /// - If `dest` is too small, output is truncated (no panic).
57    pub const fn strip_codes(src: &[u8], dest: &mut [u8]) -> usize {
58        let mut src_pos = 0;
59        let mut dest_pos = 0;
60        let mut state = State::Normal;
61
62        while src_pos < src.len() && dest_pos < dest.len() {
63            let byte = src[src_pos];
64
65            match state {
66                State::Normal => {
67                    if byte == 0x1B {
68                        // ESC character
69                        state = State::Escape;
70                    } else {
71                        dest[dest_pos] = byte;
72                        dest_pos += 1;
73                    }
74                    src_pos += 1;
75                }
76                State::Escape => {
77                    if byte == b'[' {
78                        state = State::Csi;
79                        src_pos += 1;
80                    } else if byte == b']' {
81                        state = State::Osc;
82                        src_pos += 1;
83                    } else if (byte >= b'0' && byte <= b'9') || byte == b';' {
84                        // Parameter in progress, keep consuming
85                        src_pos += 1;
86                    } else {
87                        // Single-character escape or invalid sequence.
88                        // Strip the ESC and process current byte.
89                        state = State::Normal;
90                    }
91                }
92                State::Csi => {
93                    // Consume until command character
94                    if byte >= 0x40 && byte <= 0x7E {
95                        // Command character
96                        state = State::Normal;
97                    }
98                    src_pos += 1;
99                }
100                State::Osc => {
101                    // OSC sequences end with BEL or ESC\
102                    if byte == 0x07 {
103                        // BEL
104                        state = State::Normal;
105                    } else if byte == 0x1B && src_pos + 1 < src.len() && src[src_pos + 1] == b'\\' {
106                        state = State::Normal;
107                        src_pos += 1; // Skip the backslash
108                    }
109                    src_pos += 1;
110                }
111            }
112        }
113        dest_pos
114    }
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120
121    #[test]
122    fn strip_codes() {
123        // Test 1: Basic sequence
124        let input = b"Hello \x1B[32mWorld\x1B[0m!";
125        let mut output = [0u8; 50];
126        let len = Ansi::strip_codes(input, &mut output);
127        assert_eq!(&output[0..len], b"Hello World!");
128
129        // Test 2: Consecutive escape sequences
130        let input = b"\x1B[1m\x1B[32mBold Green\x1B[0mText";
131        let len = Ansi::strip_codes(input, &mut output);
132        assert_eq!(&output[0..len], b"Bold GreenText");
133
134        // Test 3: CSI sequences with parameters
135        let input = b"Normal\x1B[1;33;44mText\x1B[0mEnd";
136        let len = Ansi::strip_codes(input, &mut output);
137        assert_eq!(&output[0..len], b"NormalTextEnd");
138
139        // Test 4: OSC sequences
140        let input = b"Start\x1B]0;Window Title\x07End";
141        let len = Ansi::strip_codes(input, &mut output);
142        assert_eq!(&output[0..len], b"StartEnd");
143
144        // Test 5: Incomplete escape sequence at end
145        let input = b"Text\x1B[";
146        let len = Ansi::strip_codes(input, &mut output);
147        assert_eq!(&output[0..len], b"Text");
148
149        // Test 6: Real ESC character (0x1B) in normal text (should be stripped)
150        let input = b"Has\x1B ESC char";
151        let len = Ansi::strip_codes(input, &mut output);
152        assert_eq!(&output[0..len], b"Has ESC char");
153
154        // Test 7: Empty input
155        let input = b"";
156        let len = Ansi::strip_codes(input, &mut output);
157        assert_eq!(&output[0..len], b"");
158
159        // Test 8: Only escape sequences
160        let input = b"\x1B[1m\x1B[32m\x1B[0m";
161        let len = Ansi::strip_codes(input, &mut output);
162        assert_eq!(&output[0..len], b"");
163
164        // Test 9: Mixed content with various sequence types
165        let input = b"Start\x1B[32mGreen\x1B]0;Title\x07\x1B[0mEnd";
166        let len = Ansi::strip_codes(input, &mut output);
167        assert_eq!(&output[0..len], b"StartGreenEnd");
168
169        // Test 10: Buffer too small for output
170        let input = b"Hello World";
171        let mut small_output = [0u8; 5];
172        let len = Ansi::strip_codes(input, &mut small_output);
173        assert_eq!(&small_output[0..len], b"Hello");
174
175        // Test 11: Escape sequences that look like text
176        let input = b"Text\x1B[0;1;mMore[Text]";
177        let len = Ansi::strip_codes(input, &mut output);
178        assert_eq!(&output[0..len], b"TextMore[Text]");
179
180        // Test 12: Binary data with ESC characters
181        let input = &[b'A', 0x1B, b'[', b'1', b'm', b'B', 0x00, b'C'];
182        let len = Ansi::strip_codes(input, &mut output);
183        assert_eq!(&output[0..len], &[b'A', b'B', 0x00, b'C']);
184
185        // Test 13: Multiple ESC characters in a row
186        let input = b"\x1B\x1B\x1B[32mText";
187        let len = Ansi::strip_codes(input, &mut output);
188        assert_eq!(&output[0..len], b"Text");
189    }
