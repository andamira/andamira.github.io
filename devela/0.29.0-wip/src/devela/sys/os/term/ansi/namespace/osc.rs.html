33    ///
34    /// The returned value formats `text` as an OSC 8 hyperlink pointing to `uri`.

81#[must_use]
82#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
83pub struct AnsiOsc<'a, const N: usize> {
84    /// The OSC command number, encoded as ASCII digits.
85    pub command: &'static str,
86    /// The semicolon-separated OSC fields.
87    pub fields: [&'a str; N],
88}
89impl<'a, const N: usize> AnsiOsc<'a, N> {
90    /// Creates a new OSC sequence.
91    pub const fn new(command: &'static str, fields: [&'a str; N]) -> Self {
92        Self { command, fields }
93    }
94    /// Returns the number of bytes needed to encode this OSC sequence.
95    #[must_use]
96    pub const fn len(self) -> usize {
97        let mut len = 2 + self.command.len() + 2; // ESC ] + command + ST
98        whilst! { i in 0..N; {
99            len += 1 + self.fields[i].len(); // ;field
100        }}
101        len
102    }
103    /// Returns whether the encoded OSC sequence is empty.
104    #[must_use]
105    pub const fn is_empty(self) -> bool {
106        self.len() == 0
107    }
108    /// Writes this OSC sequence into `dst`, returning the final offset.
109    ///
110    /// Returns `Err(needed)` if the slice is too small.
111    pub const fn write_to(self, dst: &mut [u8]) -> Result<usize, usize> {
112        let needed = self.len();
113        let mut offset = 0;
114        _put!(dst, offset, needed, b"\x1b]");
115        _put!(dst, offset, needed, self.command.as_bytes());
116        whilst! { i in 0..N; {
117            _put!(dst, offset, needed, b";");
118            _put!(dst, offset, needed, self.fields[i].as_bytes());
119        }}
120        _put!(dst, offset, needed, b"\x1b\\");
121        Ok(offset)
122    }
123    fn fmt_streamed(self, f: &mut Formatter<'_>) -> FmtResult<()> {
124        f.write_str("\x1b]")?;
125        f.write_str(self.command)?;
126        whilst! { i in 0..N; {
127            f.write_str(";")?;
128            f.write_str(self.fields[i])?;
129        }}
130        f.write_str("\x1b\\")
131    }
132}
133impl<const N: usize> Display for AnsiOsc<'_, N> {
134    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
135        self.fmt_streamed(f)
136    }
137}

157///
158/// Formatting this value emits:
159///
160/// ```text
161/// OSC 8 ; params ; URI ST
162/// text
163/// OSC 8 ; ; ST
164/// ```
165///
166/// # Notes
167/// Support depends on the terminal emulator. Unsupported terminals usually
168/// display only the visible text or ignore the sequence.
169#[must_use]
170#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
171pub struct AnsiLink<'a> {
172    /// The visible terminal text.
173    pub text: &'a str,
174    /// The hyperlink target URI.
175    pub uri: &'a str,
176    /// Optional OSC 8 hyperlink id.
177    ///
178    /// Terminals may use this to associate separated spans with the same link.
179    pub id: Option<&'a str>,
180}
181impl<'a> AnsiLink<'a> {
182    /// Creates a new terminal hyperlink.
183    pub const fn new(text: &'a str, uri: &'a str) -> Self {
184        Self { text, uri, id: None }
185    }
186    /// Creates a new terminal hyperlink with an explicit id.
187    pub const fn with_id(text: &'a str, uri: &'a str, id: &'a str) -> Self {
188        Self { text, uri, id: Some(id) }
189    }
190    /// Returns the number of bytes needed to encode this hyperlink.
191    #[must_use]
192    pub const fn len(self) -> usize {
193        let id_len = match self.id {
194            Some(id) => 3 + id.len(), // "id=" + id
195            None => 0,
196        };
197        // ESC ] 8 ; params ; uri ST + text + close
198        4 + id_len + 1 + self.uri.len() + 2 + self.text.len() + Ansi::HYPERLINK_CLOSE_B.len()
199    }
200    /// Returns whether this hyperlink encodes to no bytes.
201    ///
202    /// This is always `false` for the current OSC 8 encoding, but is useful
203    /// for API symmetry with byte-oriented values.
204    #[must_use]
205    pub const fn is_empty(self) -> bool {
206        self.len() == 0
207    }
208    /// Writes this hyperlink into `dst`, returning the final offset.
209    ///
210    /// Returns `Err(needed)` if the slice is too small.
211    ///
212    /// The text, URI and id are emitted as-is.
213    pub const fn write_to(self, dst: &mut [u8]) -> Result<usize, usize> {
214        let needed = self.len();
215        let mut offset = 0;
216        _put!(dst, offset, needed, b"\x1b]8;");
217        if let Some(id) = self.id {
218            _put!(dst, offset, needed, b"id=");
219            _put!(dst, offset, needed, id.as_bytes());
220        }
221        _put!(dst, offset, needed, b";");
222        _put!(dst, offset, needed, self.uri.as_bytes());
223        _put!(dst, offset, needed, b"\x1b\\");
224        _put!(dst, offset, needed, self.text.as_bytes());
225        _put!(dst, offset, needed, &Ansi::HYPERLINK_CLOSE_B);
226        Ok(offset)
227    }
228    fn fmt_streamed(self, f: &mut Formatter<'_>) -> FmtResult<()> {
229        f.write_str("\x1b]8;")?;
230        if let Some(id) = self.id {
231            f.write_str("id=")?;
232            f.write_str(id)?;
233        }
234        f.write_str(";")?;
235        f.write_str(self.uri)?;
236        f.write_str("\x1b\\")?;
237        f.write_str(self.text)?;
238        f.write_str(Ansi::HYPERLINK_CLOSE)
239    }
240}
241#[rustfmt::skip]
242impl Display for AnsiLink<'_> {
243    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> { self.fmt_streamed(f) }
244}

248#[inline(always)] #[rustfmt::skip]
249const fn _put_bytes(dst: &mut [u8], offset: usize, src: &[u8], needed: usize)
250    -> Result<usize, usize> {
251    is! { offset > dst.len() || src.len() > dst.len() - offset, return Err(needed) }
252    whilst! { i in 0..src.len(); { dst[offset + i] = src[i]; }}
253    Ok(offset + src.len())
254}
255macro_rules! _put {
256    ($dst:expr, $offset:ident, $needed:expr, $src:expr) => {
257        match _put_bytes($dst, $offset, $src, $needed) {
258            Ok(new_offset) => $offset = new_offset,
259            Err(needed) => return Err(needed),
260        }
261    };
262}
