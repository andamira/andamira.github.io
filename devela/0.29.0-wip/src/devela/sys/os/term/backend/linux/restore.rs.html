10    /// Terminal escape-sequence restoration flags.
11    ///
12    /// These flags only track ANSI/reporting/presentation cleanup.
13    /// Termios state is restored separately by the stored line-discipline guard.
14    struct TermLinuxRestoreFlags(u16) {
15        RESET_STYLE              = 0;
16        SHOW_CURSOR              = 1;
17        LEAVE_ALT_SCREEN         = 2;
18        DISABLE_BRACKETED_PASTE  = 3;
19        DISABLE_FOCUS_EVENTS     = 4;
20
21        DISABLE_MOUSE            = 5;
22        DISABLE_MOUSE_DRAG       = 6;
23        DISABLE_MOUSE_MOTION     = 7;
24        DISABLE_MOUSE_SGR        = 8;
25        DISABLE_MOUSE_SGR_PIXELS = 9;
26
27        DISABLE_SYNC_UPDATE      = 10;
28        CLEAR_ON_DROP            = 11;
29    }

33/// Linux terminal session restore payload.
34///
35/// This is the synchronization point between [`TermMode`] and Linux-specific
36/// terminal state changes. It applies termios line mode changes and ANSI
37/// reporting/presentation requests, recording the corresponding restoration
38/// work as each step succeeds.
39#[derive(Debug)]
40pub(crate) struct TermLinuxRestore {
41    guard: Option<LinuxTermModeGuard>,
42    flags: TermLinuxRestoreFlags,
43}
44impl TermLinuxRestore {
45    pub(crate) fn none() -> Self {
46        Self { guard: None, flags: TermLinuxRestoreFlags::new() }
47    }
48    pub(crate) fn raw() -> LinuxResult<Self> {
49        Self::enter(TermMode::raw())
50    }
51    pub(crate) fn enter(mode: TermMode) -> LinuxResult<Self> {
52        let mut restore = Self::none();
53
54        /* line discipline */
55        match mode.line_mode() {
56            TermLineMode::Line => {}
57            TermLineMode::Event => {
58                restore.guard = Some(Linux::scoped_event_mode()?);
59            }
60            TermLineMode::Raw => {
61                restore.guard = Some(Linux::scoped_termios_update(|state| {
62                    state.make_raw();
63
64                    if mode.has_processed_output() {
65                        state.set_output_processing(true);
66                    }
67                })?);
68            }
69        }
70
71        /* presentation */
72        if mode.has_alt_screen() {
73            Linux::print_bytes(&Ansi::ENABLE_ALT_SCREEN_B)?;
74            restore.flags = restore.flags.with_leave_alt_screen();
75        }
76        if mode.has_hide_cursor() {
77            Linux::print_bytes(&Ansi::CURSOR_INVISIBLE_B)?;
78            restore.flags = restore.flags.with_show_cursor();
79        }
80
81        /* input reporting */
82        if mode.has_bracketed_paste() {
83            Linux::print_bytes(&Ansi::ENABLE_BRACKETED_PASTE_B)?;
84            restore.flags = restore.flags.with_disable_bracketed_paste();
85        }
86        if mode.has_focus_events() {
87            Linux::print_bytes(&Ansi::ENABLE_FOCUS_EVENTS_B)?;
88            restore.flags = restore.flags.with_disable_focus_events();
89        }
90
91        /* mouse: enable encoding before reporting */
92        // with fallback:
93        if mode.has_mouse_sgr() {
94            Linux::print_bytes(&Ansi::ENABLE_MOUSE_SGR_B)?;
95            restore.flags = restore.flags.with_disable_mouse_sgr();
96        }
97        if mode.has_mouse_sgr_pixels() {
98            Linux::print_bytes(&Ansi::ENABLE_MOUSE_SGR_PIXELS_B)?;
99            restore.flags = restore.flags.with_disable_mouse_sgr_pixels();
100        }
101
102        if mode.has_mouse_motion() {
103            Linux::print_bytes(&Ansi::ENABLE_MOUSE_MOTION_B)?;
104            restore.flags = restore.flags.with_disable_mouse_motion();
105        } else if mode.has_mouse_drag() {
106            Linux::print_bytes(&Ansi::ENABLE_MOUSE_DRAG_B)?;
107            restore.flags = restore.flags.with_disable_mouse_drag();
108        } else if mode.has_mouse() {
109            Linux::print_bytes(&Ansi::ENABLE_MOUSE_B)?;
110            restore.flags = restore.flags.with_disable_mouse();
111        }
112
113        /* screen clearing */
114        if mode.has_clear_on_enter() {
115            Linux::print_bytes(&Ansi::ERASE_SCREEN_B)?;
116            Linux::print_bytes(&Ansi::CURSOR_HOME_B)?;
117        }
118        if mode.has_clear_on_drop() {
119            restore.flags = restore.flags.with_clear_on_drop();
120        }
121
122        /* output transaction */
123        if mode.has_sync_update() {
124            Linux::print_bytes(&Ansi::ENABLE_SYNC_UPDATE_B)?;
125            restore.flags = restore.flags.with_disable_sync_update();
126        }
127
128        Ok(restore)
129    }
130}
131impl Drop for TermLinuxRestore {
132    fn drop(&mut self) {
133        /* style reset */
134        if self.flags.has_reset_style() {
135            let _ = Linux::print_bytes(&Ansi::RESET_B);
136        }
137
138        /* input reporting */
139        if self.flags.has_disable_bracketed_paste() {
140            let _ = Linux::print_bytes(&Ansi::DISABLE_BRACKETED_PASTE_B);
141        }
142        if self.flags.has_disable_focus_events() {
143            let _ = Linux::print_bytes(&Ansi::DISABLE_FOCUS_EVENTS_B);
144        }
145
146        /* mouse: disable reporting before encoding */
147        if self.flags.has_disable_mouse_motion() {
148            let _ = Linux::print_bytes(&Ansi::DISABLE_MOUSE_MOTION_B);
149        }
150        if self.flags.has_disable_mouse_drag() {
151            let _ = Linux::print_bytes(&Ansi::DISABLE_MOUSE_DRAG_B);
152        }
153        if self.flags.has_disable_mouse() {
154            let _ = Linux::print_bytes(&Ansi::DISABLE_MOUSE_B);
155        }
156
157        if self.flags.has_disable_mouse_sgr_pixels() {
158            let _ = Linux::print_bytes(&Ansi::DISABLE_MOUSE_SGR_PIXELS_B);
159        }
160        if self.flags.has_disable_mouse_sgr() {
161            let _ = Linux::print_bytes(&Ansi::DISABLE_MOUSE_SGR_B);
162        }
163
164        /* screen clearing */
165        if self.flags.has_clear_on_drop() {
166            let _ = Linux::print_bytes(&Ansi::ERASE_SCREEN_B);
167            let _ = Linux::print_bytes(&Ansi::CURSOR_HOME_B);
168        }
169        /* presentation */
170        if self.flags.has_show_cursor() {
171            let _ = Linux::print_bytes(&Ansi::CURSOR_VISIBLE_B);
172        }
173        if self.flags.has_leave_alt_screen() {
174            let _ = Linux::print_bytes(&Ansi::DISABLE_ALT_SCREEN_B);
175        }
176        /* output transaction */
177        if self.flags.has_disable_sync_update() {
178            let _ = Linux::print_bytes(&Ansi::DISABLE_SYNC_UPDATE_B);
179        }
180        // `guard` drops after this and restores termios.
181    }
182}
