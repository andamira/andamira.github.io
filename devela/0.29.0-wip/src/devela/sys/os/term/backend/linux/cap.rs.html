19    /// Probes terminal protocol capabilities exposed by the current terminal endpoint.
20    ///
21    /// When running inside a terminal multiplexer such as `tmux`, this probes the
22    /// multiplexer pseudo-terminal, not necessarily the outer terminal emulator.
23    // IMPROVE: add more queries
24    pub fn probe_term_capabilities(&mut self) -> LinuxResult<TermCaps> {
25        let _guard = Linux::scoped_event_mode()?;
26        let mut caps = TermCaps::new();
27        /* input */
28        // Keyboard
29        is! { self.query_dec_private_mode_supported(1000)?, caps.set_cap(TermCap::Mouse) }
30        is! { self.query_dec_private_mode_supported(1002)?, caps.set_cap(TermCap::MouseDrag) }
31        is! { self.query_dec_private_mode_supported(1003)?, caps.set_cap(TermCap::MouseMotion) }
32        is! { self.query_dec_private_mode_supported(1006)?, caps.set_cap(TermCap::MouseSgr); }
33        is! { self.query_dec_private_mode_supported(1016)?, caps.set_cap(TermCap::MouseSgrPixels) }
34        is! { self.query_dec_private_mode_supported(1004)?, caps.set_cap(TermCap::Focus) }
35        is! { self.query_dec_private_mode_supported(2004)?, caps.set_cap(TermCap::BracketedPaste) }
36        // resize
37        /* output */
38        // Ansi
39        // Cursor
40        // Style
41        // x scroll 1007
42        // x screen_only 1047
43        is! { self.query_dec_private_mode_supported(1049)?, caps.set_cap(TermCap::AltScreen) }
44        is! { self.query_dec_private_mode_supported(2026)?, caps.set_cap(TermCap::SyncUpdate) }
45        /* image */
46        is! { self.query_dec_private_mode_supported(80)?, caps.set_cap(TermCap::Sixel) }
47        // KittyImage
48        // ItermImage
49        /* query */
50        // QueryDeviceAttrs,
51        // QueryCursorPos,
52        // QueryColor,
53        self.term_caps = caps;
54        Ok(caps)
55    }

57    /// Returns the last known runtime capabilities.
58    pub const fn run_capabilities(&self) -> RunCap {
59        self.run_cap
60    }
61    /// Refreshes runtime capabilities from cached terminal capabilities and current size.
62    pub fn refresh_run_capabilities(&mut self) -> LinuxResult<RunCap> {
63        self.refresh_size()?;
64        self.run_cap = Self::derive_run_capabilities(self.term_caps, self.size);
65        Ok(self.run_cap)
66    }
67    /// Probes terminal capabilities, refreshes size, and recomputes runtime capabilities.
68    pub fn probe_run_capabilities(&mut self) -> LinuxResult<RunCap> {
69        let _ = self.probe_term_capabilities()?;
70        self.refresh_run_capabilities()
71    }

74// Private helpers
75impl TermLinux {
76    ///
77    fn query_dec_private_mode_supported(&mut self, mode: u16) -> LinuxResult<bool> {
78        Ok(self.query_dec_private_mode(mode)?.is_some_and(|s| s.is_supported()))
79    }
80    ///
81    fn query_dec_private_mode(&mut self, mode: u16) -> LinuxResult<Option<TermDecModeStatus>> {
82        let mut query = [0u8; 10];
83        let query = Ansi::QUERY_DEC_PRIVATE_MODE_N_B(&mut query, mode);
84        self.print(query)?;
85        self.read_dec_private_mode_reply(mode)
86    }
87
88    /// Reads a DEC private mode reply for `mode`, if the terminal answers.
89    ///
90    /// Expects a DECRPM reply of the form:
91    /// `ESC [ ? mode ; status $ y`.
92    ///
93    /// Returns `Ok(None)` on timeout or when another reply is received.
94    fn read_dec_private_mode_reply(&mut self, mode: u16) -> LinuxResult<Option<TermDecModeStatus>> {
95        // Conservative probe timeout:
96        // 40 × 1ms keeps startup responsive while giving terminal replies time to arrive.
97        //
98        // This avoids blocking forever on terminals that do not implement DECRQM.
99        const TRIES: usize = 40;
100        let mut tries = 0;
101        while tries < TRIES {
102            if let Some(byte) = Linux::try_get_byte() {
103                match self.parser.feed_parsed(byte) {
104                    TermParsed::Reply(TermReply::DecPrivateMode { mode: reply_mode, status })
105                        if reply_mode == mode =>
106                    {
107                        return Ok(Some(status));
108                    }
109                    // Other terminal replies may be from previous/proximate probes.
110                    // For now we ignore them.
111                    TermParsed::Reply(_) => {}
112                    // User-facing input during probing.
113                    //
114                    // If TermLinux eventually has an event queue, this is the place
115                    // to preserve it instead of dropping it.
116                    TermParsed::Event(_) => {}
117                    TermParsed::Pending | TermParsed::Unknown => {}
118                }
119            } else {
120                tries += 1;
121                Linux::sleep_ms(1)?;
122            }
123        }
124        Ok(None)
125    }

127    /// Projects terminal capabilities into runtime capabilities.
128    pub(super) const fn derive_run_capabilities(caps: TermCaps, _size: Option<TermSize>) -> RunCap {
129        let depth = caps.color_depth();
130        RunCap {
131            color: if depth.has_color() {
132                Some(RunCapColor {
133                    depth,
134                    palette_size: Self::palette_size(depth),
135                    palette_set: caps.get_query_color() != 0,
136                })
137            } else {
138                None
139            },
140            image: if caps.get_sixel() != 0
141                || caps.get_kitty_image() != 0
142                || caps.get_iterm_image() != 0
143            {
144                Some(RunCapImage {
145                    max_bitmap_extent: None,
146                    pixel_native: caps.get_kitty_image() != 0 || caps.get_iterm_image() != 0,
147                })
148            } else {
149                None
150            },
151            input: Some(
152                RunCapInput::new()
153                    .with_if(caps.get_keyboard() != 0, RunCapInput::KEYBOARD)
154                    .with_if(caps.get_mouse() != 0, RunCapInput::MOUSE),
155            ),
156            text: Some(
157                RunCapText::new()
158                    .with(RunCapText::WRITE)
159                    .with(RunCapText::CELL_GRID)
160                    .with_if(caps.get_cursor() != 0, RunCapText::CURSOR)
161                    .with_if(caps.get_style() != 0, RunCapText::STYLE)
162                    .with_if(caps.get_keyboard() != 0, RunCapText::EDIT)
163                    .with_if(depth.has_color(), RunCapText::COLOR_PAIR),
164            ),
165            audio: None,
166            system: None,
167            window: if caps.get_resize() != 0 { Some(RunCapWindow { multi: false }) } else { None },
168        }
169    }
170    const fn palette_size(depth: ColorDepth) -> Option<NonMaxU16> {
171        match depth {
172            ColorDepth::Palette2 => NonMaxU16::new(2),
173            ColorDepth::Palette8 => NonMaxU16::new(8),
174            ColorDepth::Palette16 => NonMaxU16::new(16),
175            ColorDepth::Palette256 => NonMaxU16::new(256),
176            _ => None,
177        }
178    }
