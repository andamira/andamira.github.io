1// devela/src/sys/os/browser/web/api/canvas.rs

4//! Implements the web canvas API.
5//
6
7use crate::Web;
8use crate::{_js_doc, _js_extern, JsTextRenderMetrics, JsTextRenderMetricsFull, js_number};
9
10/// # Web API canvas
11///
12/// - <https://developer.mozilla.org/en-US/docs/Web/API/CanvasRenderingContext2D>
13/// - <https://html.spec.whatwg.org/multipage/canvas.html>
14#[rustfmt::skip]
15impl Web {
16    /* misc. */
17    /// Sets the active canvas using a CSS `selector`.
18    pub fn set_canvas(selector: &str) { unsafe { set_canvas(selector.as_ptr(), selector.len()); } }
19
20    /* color settings */
21    #[doc = _js_doc!(canvas "fillStyle")]
22    /// Sets the color or style for filling shapes.
23    pub fn fill_style(r: u8, g: u8, b: u8) { fill_style(r, g, b); }
24    #[doc = _js_doc!(canvas "strokeStyle")]
25    /// Sets the color or style for lines.
26    pub fn stroke_style(r: u8, g: u8, b: u8) { stroke_style(r, g, b); }
27
28    /* drawing rectangles */
29    #[doc = _js_doc!(canvas "fillRect")]
30    /// Draws a filled rectangle.
31    pub fn fill_rect(x: js_number, y: js_number, w: js_number, h: js_number)
32        { fill_rect(x, y, w, h); }
33    #[doc = _js_doc!(canvas "strokeRect")]
34    /// Draws a rectangular outline.
35    pub fn stroke_rect(x: js_number, y: js_number, w: js_number, h: js_number)
36        { stroke_rect(x, y, w, h); }
37    #[doc = _js_doc!(canvas "clearRect")]
38    /// Clears the specified rectangular area, making it fully transparent.
39    pub fn clear_rect(x: js_number, y: js_number, w: js_number, h: js_number)
40        { clear_rect(x, y, w, h); }
41
42    /* drawing shapes */
43    /// Draws a line.
44    pub fn draw_line(x1: js_number, y1: js_number, x2: js_number, y2: js_number)
45        { draw_line(x1, y1, x2, y2); }
46    /// Draws a circle.
47    pub fn draw_circle(x: js_number, y: js_number, radius: js_number)
48        { draw_circle(x, y, radius); }
49
50    /* drawing text */
51    #[doc = _js_doc!(canvas "fillText")]
52    /// Draws filled text at the specified position.
53    pub fn fill_text(text: &str, x: js_number, y: js_number) {
54        unsafe { fill_text(text.as_ptr(), text.len(), x, y); }
55    }
56    #[doc = _js_doc!(canvas "strokeText")]
57    /// Draws text outline at the specified position.
58    pub fn stroke_text(text: &str, x: js_number, y: js_number) {
59        unsafe { stroke_text(text.as_ptr(), text.len(), x, y); }
60    }
61    #[doc = _js_doc!(canvas "measureText")]
62    /// Measures the essential properties of text.
63    pub fn measure_text(text: &str) -> JsTextRenderMetrics {
64        let (mut metrics, ptr, len) = (JsTextRenderMetrics::default(), text.as_ptr(), text.len());
65        unsafe { measure_text(ptr, len, &mut metrics as *mut JsTextRenderMetrics); }
66        metrics
67    }
68    #[doc = _js_doc!(canvas "measureTextFull")]
69    /// Measures all available text metrics.
70    pub fn measure_text_full(text: &str) -> JsTextRenderMetricsFull {
71        let (mut metrics, ptr, len) =
72            (JsTextRenderMetricsFull::default(), text.as_ptr(), text.len());
73        unsafe { measure_text_full(ptr, len, &mut metrics as *mut JsTextRenderMetricsFull); }
74        metrics
75    }
76}
77_js_extern! {
78    [ module: "api_canvas" ]
79    /* misc. */
80    unsafe fn set_canvas(str_ptr: *const u8, str_len: usize);
81    /* color settings */
82    safe fn "fillStyle" fill_style(r: u8, g: u8, b: u8);
83    safe fn "strokeStyle" stroke_style(r: u8, g: u8, b: u8);
84    /* drawing rectangles */
85    safe fn "fillRect" fill_rect(x: js_number, y: js_number, w: js_number, h: js_number);
86    safe fn "strokeRect" stroke_rect(x: js_number, y: js_number, w: js_number, h: js_number);
87    safe fn "clearRect" clear_rect(x: js_number, y: js_number, w: js_number, h: js_number);
88    /* drawing paths */
89    //
90    /* drawing shapes */
91    safe fn draw_line(x1: js_number, y1: js_number, x2: js_number, y2: js_number);
92    safe fn draw_circle(x: js_number, y: js_number, radius: js_number);
93
94    /* drawing text */
95    unsafe fn "fillText" fill_text(str_ptr: *const u8, str_len: usize, x: js_number, y: js_number);
96    unsafe fn "strokeText" stroke_text(str_ptr: *const u8, str_len: usize,
97        x: js_number, y: js_number);
98    unsafe fn "measureText" measure_text(text_ptr: *const u8, text_len: usize,
99        out_metrics: *mut JsTextRenderMetrics);
100    unsafe fn "measureTextFull" measure_text_full(text_ptr: *const u8, text_len: usize,
101        fut_metrics: *mut JsTextRenderMetricsFull);
102}
