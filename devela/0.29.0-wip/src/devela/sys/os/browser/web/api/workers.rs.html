13/// Web API workers
14#[rustfmt::skip]
15impl Web {
16    /// Spawns a Web Worker and returns its ID.
17    pub fn worker_spawn(script: &str) -> Result<WebWorker, Error> {
18        let id = unsafe { worker_spawn(script.as_ptr(), script.len()) };
19        if id == 0 { Err(Error::InvalidScript) } else { Ok(WebWorker { id }) }
20    }
21    /// Checks if this worker is still active by querying JavaScript.
22    pub fn worker_is_active(worker: WebWorker) -> js_bool { worker_is_active(worker.id()) }
23    /// Stops a specific Web Worker by ID.
24    pub fn worker_stop(worker: WebWorker) { worker_stop(worker.id()); }
25    /// Stops all Web Workers.
26    pub fn worker_stop_all() { worker_stop_all(); }
27    /// Returns the number of active workers.
28    pub fn worker_list_len() -> usize { worker_list_len() as usize }
29    /// Returns the list of active worker IDs.
30    #[cfg(feature = "alloc")]
31    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
32    pub fn worker_list() -> Vec<WebWorker> {
33        let len = worker_list_len() as usize;
34        let mut workers = vec![WebWorker::default(); len];
35        let count = Web::worker_list_buf(&mut workers);
36        workers.truncate(count);
37        workers
38    }
39    /// Writes active worker handles into a buffer and returns the number written.
40    pub fn worker_list_buf(buffer: &mut [WebWorker]) -> usize {
41        let len = worker_list_len() as usize;
42        let count = len.min(buffer.len());
43        unsafe { worker_list(buffer.as_mut_ptr() as *mut js_uint32, count as js_uint32); }
44        count
45    }
46    /// Sends a message to a specific Web Worker.
47    pub fn worker_send_message(worker: WebWorker, msg: &str) {
48        unsafe { worker_send_message(worker.id(), msg.as_ptr(), msg.len()); }
49    }
50    /// Requests execution of JavaScript inside a worker.
51    pub fn worker_eval(worker: WebWorker, js_code: &str) -> WebWorkerJob {
52        let id = unsafe { worker_eval(worker.id(), js_code.as_ptr(), js_code.len()) };
53        WebWorkerJob { worker, id }
54    }
55    /// Polls for the result of a JavaScript execution in a worker, and writes it into `buffer`.
56    ///
57    /// If ready, returns the number of bytes written to the buffer.
58    pub fn worker_poll(job: WebWorkerJob, buffer: &mut [u8])
59        -> AsyncPoll<Result<usize, Error>> {
60        if !job.worker().is_active() { return AsyncPoll::Ready(Err(Error::WorkerNotFound)); }
61        let written = unsafe { worker_poll(job.id(), buffer.as_mut_ptr(), buffer.len()) };
62        match written {
63            0 => AsyncPoll::Pending,
64            js_uint32::MAX => AsyncPoll::Ready(Err(Error::JobNotFound)),
65            _ => AsyncPoll::Ready(Ok(written as usize)),
66        }
67    }
68    /// Polls for the result of a JavaScript execution in a worker.
69    #[cfg(feature = "alloc")]
70    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
71    pub fn worker_poll_alloc(job: WebWorkerJob) -> AsyncPoll<Result<String, Error>> {
72        if !job.worker().is_active() { return AsyncPoll::Ready(Err(Error::WorkerNotFound)); }
73        let mut first_check = true;
74        let result = Js::read_string_capped(128, false, |ptr, cap| {
75            let res = unsafe { worker_poll(job.id(), ptr, cap as usize) as js_int32 };
76            if first_check {
77                first_check = false; // Intercept status codes before bothering with decoding
78                if res == 0 { return 0; } else if res == -1 { return -1; } // pending or not found
79            }
80            res
81        });
82        match result.as_str() {
83            "" => AsyncPoll::Pending, // Covers 0 and -1 (mapped above)
84            _ => AsyncPoll::Ready(Ok(result)),
85        }
86    }
87    /// Cancels an ongoing JavaScript evaluation.
88    pub fn worker_eval_cancel(job: WebWorkerJob) { worker_cancel_eval(job.id()); }
89}
