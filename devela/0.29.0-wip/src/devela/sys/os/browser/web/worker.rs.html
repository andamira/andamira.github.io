38#[rustfmt::skip]
39#[cfg(not(feature = "safe_lang"))]
40#[cfg(all(feature = "unsafe_ffi", not(windows)))]
41#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_ffi")))]
42#[cfg_attr(nightly_doc, doc(cfg(target_arch = "wasm32")))]
43impl WebWorker {
44    /// Spawns a new JavaScript Web Worker from a script.
45    pub fn spawn(script: &str) -> Result<Self, WebWorkerError> { Web::worker_spawn(script) }
46    /// Stops this Web Worker.
47    pub fn stop(self) { Web::worker_stop(self); }
48    /// Checks if this worker is still active by querying JavaScript.
49    pub fn is_active(self) -> bool { Web::worker_is_active(self) }
50    /// Sends a message to this Web Worker.
51    pub fn send_message(self, message: &str) { Web::worker_send_message(self, message); }
52    /// Executes JavaScript inside this worker, returning a `WebWorkerJob`.
53    pub fn eval(self, js_code: &str) -> Result<WebWorkerJob, WebWorkerError> {
54        if !self.is_active() { return Err(WebWorkerError::WorkerNotFound); }
55        let job = Web::worker_eval(self, js_code);
56        if job.id == 0 { return Err(WebWorkerError::WorkerNotFound); }
57        Ok(job)
58    }
59}
