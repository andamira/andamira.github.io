9/// A typed selector for web event names used by the web API browser bridge.
10#[doc = crate::_doc_meta!{
11    location("sys/os/browser/web"),
12    test_size_of(WebEventKind = 4|32; niche Option),
13}]
14///
15/// # Role
16/// - Selects the exact web event name passed to `addEventListener`.
17/// - Bridges raw browser events into web API callbacks and normalized events.
18/// - Keeps the web adapter typed without leaking stringly-typed event names into user code.
19///
20/// ---
21/// - <https://developer.mozilla.org/en-US/docs/Web/API/Event>
22/// - <https://developer.mozilla.org/en-US/docs/Web/API/EventTarget>
23// NOTE: In sync with ../js/events.js::getEventKind()
24#[repr(C)]
25#[non_exhaustive]
26#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, Hash)]
27pub enum WebEventKind {
28    /// Unknown event. Default case.
29    #[default]
30    Unknown = 0,
31
32    /// Fires when an element is clicked.
33    Click = 1,
34
35    /* keyboard */
36    /// Fires when a key is pressed down.
37    KeyDown = 2,
38    /// Fires when a key is released.
39    KeyUp = 3,
40    // no KeyPress (deprecated)
41
42    /* mouse */
43    /// Fires when the mouse button is pressed down.
44    MouseDown = 4,
45    /// Fires when the mouse button is released.
46    MouseUp = 5,
47    /// Fires when the mouse moves over an element.
48    MouseMove = 6,
49
50    /* pointer */
51    /// Fires when the pointer is pressed down.
52    PointerDown = 7,
53    /// Fires when the pointer is released.
54    PointerUp = 8,
55    /// Fires when the pointer is moved.
56    PointerMove = 9,
57
58    /// Fires when the wheel is moved.
59    Wheel = 10,
60
61    ///
62    GamepadPoll = 11,
63
64    /// Fires when the window is resized.
65    Resize = 12,
66}
67impl WebEventKind {
68    /// Constructs a `WebEventKind` from its representation.
69    pub const fn from_repr(from: u8) -> Self {
70        use WebEventKind as E;
71        match from {
72            1 => E::Click,
73            2 => E::KeyDown,
74            3 => E::KeyUp,
75            4 => E::MouseDown,
76            5 => E::MouseUp,
77            6 => E::MouseMove,
78            7 => E::PointerDown,
79            8 => E::PointerUp,
80            9 => E::PointerMove,
81            10 => E::Wheel,
82            11 => E::GamepadPoll,
83            12 => E::Resize,
84            _ => E::Unknown,
85        }
86    }
87    /// Returns the event name as a string.
88    pub const fn as_str(self) -> &'static str {
89        use WebEventKind as E;
90        match self {
91            E::Click => "click",
92            E::KeyDown => "keydown",
93            E::KeyUp => "keyup",
94            E::MouseDown => "mousedown",
95            E::MouseUp => "mouseup",
96            E::MouseMove => "mousemove",
97            E::PointerDown => "pointerdown",
98            E::PointerUp => "pointerup",
99            E::PointerMove => "pointermove",
100            E::Wheel => "wheel",
101            E::GamepadPoll => "gamepadpoll",
102            E::Resize => "resize",
103            E::Unknown => "none",
104        }
105    }
106}
107
108#[rustfmt::skip]
109impl WebEventKind {
110    /// Converts a `WebEventKind` to `KeyState`, if applicable.
111    // https://developer.mozilla.org/en-US/docs/Web/API/KeyboardEvent/repeat
112    #[must_use]
113    pub const fn to_key_state(self, repeat: bool) -> Option<KeyState> {
114        match self {
115            WebEventKind::KeyDown => Some(is![repeat, KeyState::Repeat, KeyState::Press]),
116            WebEventKind::KeyUp => Some(KeyState::Release),
117            _ => None,
118        }
119    }
120    /// Converts a `KeyState` to `WebEventKind`.
121    #[must_use]
122    pub const fn from_key_state(from: KeyState) -> WebEventKind {
123        match from {
124            KeyState::Press => WebEventKind::KeyDown,
125            KeyState::Release => WebEventKind::KeyUp,
126            KeyState::Repeat => WebEventKind::KeyDown,
127        }
128    }
