23///
24/// Represents a JavaScript `KeyboardEvent` in a compact callback-friendly form.
25///
26/// `key` is the Unicode scalar value of `KeyboardEvent.key` when it contains
27/// exactly one Unicode scalar, or `0` for non-text keys such as `Enter`.
28#[repr(C)]
29#[derive(Copy, Clone, Debug, PartialEq)]
30pub struct WebEventKey {
31    /// Unicode scalar from `KeyboardEvent.key`, or `0` for non-text keys.
32    pub key: js_uint32, // 4 bytes
33    /// The key transition state.
34    pub state: KeyState, // 1 byte
35    /// Which part of the keyboard produced the event.
36    pub location: WebKeyLocation, // 1 byte
37    /// Active keyboard modifiers.
38    pub mods: KeyMods, // 2 bytes
39
40    /// The JavaScript event timestamp.
41    pub timestamp: JsInstant, // 8 bytes
42}
43
44impl WebEventKey {
45    /// Returns a new keyboard event.
46    pub const fn new(
47        key: js_uint32,
48        state: KeyState,
49        location: WebKeyLocation,
50        mods: KeyMods,
51        timestamp: JsInstant,
52    ) -> Self {
53        Self { key, location, mods, state, timestamp }
54    }
55
56    /// Returns whether this event carries a text scalar.
57    pub const fn has_key_scalar(self) -> bool {
58        self.key != 0
59    }
60
61    /// Converts `WebEventKey` to a normalized `EventKey`.
62    ///
63    /// Returns `None` if the web event kind is not a keyboard event.
64    pub const fn to_event_key(self) -> EventKey {
65        let sem = if let Some(c) = char::from_u32(self.key) { Key::Char(c) } else { Key::Unknown };
66        EventKey::new(sem, Key::Unknown, self.mods, self.state)
67    }
68    /// Converts `WebEventKey` to `EventKindTimed`.
69    ///
70    /// Returns `None` if the web event kind is not a keyboard event.
71    pub const fn to_kind_timed(self) -> EventKindTimed {
72        EventKindTimed::new(
73            EventKind::Key(self.to_event_key()),
74            Some(EventTimestamp::from_js(self.timestamp)),
75        )
76    }
77
78    /// Converts a normalized `EventKey` back to `WebEventKey`.
79    ///
80    /// This is necessarily partial because `WebEventKey` stores only a
81    /// scalar key, not the browser `KeyboardEvent.key` and
82    /// `KeyboardEvent.code` strings.
83    pub const fn from_event_key(from: EventKey) -> WebEventKey {
84        let key = Self::scalar_from_semantic_key(from.semantic);
85        let location = Self::location_from_event_key(from);
86        WebEventKey::new(key, from.state, location, from.mods, JsInstant { ms: 0.0 })
87    }
88    /// Converts a timed normalized `EventKey` back to `WebEventKey`.
89    pub const fn from_event_key_timed(
90        from: Timed<EventKey, Option<EventTimestamp>>,
91    ) -> WebEventKey {
92        let timestamp = is![let Some(t) = from.time, t.to_js(), JsInstant { ms: 0.0 }];
93        let mut event = Self::from_event_key(from.value);
94        event.timestamp = timestamp;
95        event
96    }
97
98    /* helpers */
99
100    /// Converts a semantic `Key` into the scalar part of `KeyboardEvent.key`.
101    const fn scalar_from_semantic_key(key: Key) -> u32 {
102        match key {
103            Key::Char(c) => c as u32,
104            Key::Space => ' ' as u32,
105            _ => 0,
106        }
107    }
108    /// Infers the browser key location from the normalized key event.
109    const fn location_from_event_key(event: EventKey) -> WebKeyLocation {
110        match event.physical {
111            Key::Unknown => event.semantic.to_web_key().1,
112            physical => physical.to_web_code().1,
113        }
114    }
115}
116impl From<WebEventKey> for EventKindTimed {
117    fn from(from: WebEventKey) -> Self {
118        from.to_kind_timed()
119    }
120}
121impl From<Timed<EventKey, Option<EventTimestamp>>> for WebEventKey {
122    fn from(from: Timed<EventKey, Option<EventTimestamp>>) -> Self {
123        Self::from_event_key_timed(from)
124    }
125}

141    //     let (mut num, mut valid) = (0u8, false);
142    //     while let Some((&digit, rest)) = bytes.split_first() {
143    //         if digit < b'0' || digit > b'9' { return (0, false); }

150    const fn parse_u8_from_bytes(bytes: &[u8]) -> Option<u8> {
151        let (mut num, mut index, mut found_digit) = (0u8, 0usize, false);
152        while index < bytes.len() {
153            let digit = bytes[index];
154            if digit < b'0' || digit > b'9' { break; } // Stop at first invalid character
155            let digit_value = digit - b'0';
156            if num > (u8::MAX / 10) || (num == u8::MAX / 10 && digit_value > u8::MAX % 10) {
157                return None;
158            }
159            num = num * 10 + digit_value;
160            found_digit = true;
161            index += 1;
162        }
163        if found_digit { Some(num) } else { None }
164    }
