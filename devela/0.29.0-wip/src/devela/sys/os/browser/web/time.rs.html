6#[rustfmt::skip]
7impl JsInstant {
8    /// Returns the current instant using `performance.now()`.
9    pub fn now() -> Self { Web::performance_now() }
10    /// Returns the time origin using `performance.timeOrigin()`.
11    pub fn origin() -> Self { Web::performance_time_origin() }
12
13    /// Resets this instant to the current time.
14    pub fn reset(&mut self) { *self = Web::performance_now(); }
15    /// Returns the elapsed time since this instant.
16    pub fn elapsed(self) -> Self { Self::from_millis_f64(Web::performance_now().ms - self.ms) }
17    /// Returns the elapsed time since this instant as a `TimeDelta`.
18    pub fn delta_elapsed(self) -> TimeDelta { TimeDelta::from_js(self.elapsed()) }
19}

21#[rustfmt::skip]
22impl JsTimeout {
23    #[doc = _js_doc!("Window", "setTimeout")]
24    /// Calls a function after a delay in milliseconds.
25    pub fn timeout(callback: extern "C" fn(), delay_ms: js_uint32) -> Self {
26        WebWindow::set_timeout(callback, delay_ms) }
27    #[doc = _js_doc!("Window", "setInterval")]
28    /// Calls a function repeatedly at a fixed interval in milliseconds.
29    pub fn interval(callback: extern "C" fn(), interval_ms: js_uint32) -> Self {
30        WebWindow::set_timeout(callback, interval_ms) }
31
32    /// Executes JavaScript code immediately.
33    /// ## Security Warning
34    /// - Avoid passing untrusted input, as this executes arbitrary JS.
35    /// - Ensure all evaluated code is **safe and controlled**.
36    pub fn eval(js_code: &str) { WebWindow::eval(js_code) }
37    #[doc = _js_doc!("Window", "setTimeout")]
38    /// Executes JavaScript code after a delay in milliseconds.
39    pub fn eval_timeout(js_code: &str, delay_ms: js_uint32) -> Self {
40        WebWindow::eval_timeout(js_code, delay_ms) }
41    #[doc = _js_doc!("Window", "setInterval")]
42    /// Executes JavaScript code repeatedly at a fixed interval in milliseconds.
43    pub fn eval_interval(js_code: &str, interval_ms: js_uint32) -> Self {
44        WebWindow::eval_interval(js_code, interval_ms) }
45
46    #[doc = _js_doc!("Window", "clearTimeout")]
47    #[doc = _js_doc!("Window", "clearInterval")]
48    /// Cancels a timeout or interval.
49    pub fn clear(self) { WebWindow::clear_timeout(self); }
