25/// Represents a JavaScript pointer event containing relevant properties.
26///
27/// - <https://developer.mozilla.org/en-US/docs/Web/API/PointerEvent>
28///
29/// # Compatibility
30/// Some Firefox/Linux configurations may report stylus input as
31/// `pointerType = "mouse"` with `pressure = 0.5` and zero tilt.
32///
33/// That is the browser/platform fallback stream.
34//
35// related to MOZ_USE_XINPUT2 in X11:
36// DONE? [pen recognized as mouse in firefox](https://bugzilla.mozilla.org/show_bug.cgi?id=1606832)
37// WAIT: - [Use XInput2 manager on GTK3](https://bugzilla.mozilla.org/show_bug.cgi?id=1207700)
38// WAIT:   - [repeat always false in X11](https://bugzilla.mozilla.org/show_bug.cgi?id=1594003)
39//
40// WAIT: [META stylus/pen handling issues](https://bugzilla.mozilla.org/show_bug.cgi?id=1631377)
41// WAIT: - [spurious pen pointer evens](https://bugzilla.mozilla.org/show_bug.cgi?id=1906968)
42// WAIT: - […incorrect pointerType](https://bugzilla.mozilla.org/show_bug.cgi?id=1449660)
43// WAIT: - […events broken for pen devices](https://bugzilla.mozilla.org/show_bug.cgi?id=1487509)
44// WAIT: - […impl has too many errors](https://bugzilla.mozilla.org/show_bug.cgi?id=1583519)
45// WAIT: - […button state incorrect value](https://bugzilla.mozilla.org/show_bug.cgi?id=1425384)
46// WAIT: - [sActivePointersIds gets 0](https://bugzilla.mozilla.org/show_bug.cgi?id=1904865)
47// WAIT: - [pen input is misrepresented](https://bugzilla.mozilla.org/show_bug.cgi?id=1953665)
48#[repr(C)]
49#[derive(Copy, Clone, PartialEq)]
50pub struct WebEventPointer {
51    //: 24 bytes
52    /// The X-coordinate of the pointer event relative to the viewport.
53    pub x: js_number, // 8 bytes
54    /// The Y-coordinate of the pointer event relative to the viewport.
55    pub y: js_number, // 8 bytes
56    /// The pressure applied to the pointer (0.0 to 1.0 for most devices).
57    pub pressure: js_number, // 8 bytes
58
59    //: 8 bytes
60    /// Unique identifier for the pointer device.
61    pub id: js_int32, // 4 bytes
62    /// The tilt of the stylus along the X-axis (-90° to 90°).
63    pub tilt_x: i8, // 1 bytes
64    /// The tilt of the stylus along the Y-axis (-90° to 90°).
65    pub tilt_y: i8, // 1 bytes
66    /// The rotation of the stylus around its own axis (0° to 359°).
67    pub twist: u16, // 2 bytes, 9 bits
68
69    //: 16 bytes
70    // Encodes pointer kind and triggering button in one byte.
71    // This is a compact internal representation, not the raw DOM shape.
72    code: WebPointerCode, // 1 byte
73    /// A bitmask of buttons currently being held down during the pointer event.
74    pub buttons: u8, // 1 byte
75    /// A bitmask of active keyboard modifiers during the pointer event.
76    pub mods: KeyMods, // 2 bytes
77    /// The type of pointer event (PointerDown, PointerMove, etc.).
78    pub etype: WebEventKind, // 4 bytes
79    /// The JavaScript event timestamp.
80    pub timestamp: JsInstant, // 8 bytes
81}
82impl_trait! { fmt::Debug for WebEventPointer |self, f| {

98}}
99impl WebEventPointer {
100    /// Returns a new [`WebEventPointer`].
101    #[allow(clippy::too_many_arguments)] #[rustfmt::skip]
102    pub const fn new(
103        x: js_number, y: js_number,
104        pressure: js_number,
105        id: js_int32,
106        tilt_x: i8, tilt_y: i8,
107        twist: u16,
108        kind: EventPointerKind,
109        button: u8, buttons: u8,
110        mods: KeyMods,
111        etype: WebEventKind,
112        timestamp: JsInstant,
113    ) -> Self {
114        let code = WebPointerCode::new()
115            .with_kind(kind.to_web())
116            .with_button(Self::encode_web_button(button));
117        Self { x, y, pressure, id, tilt_x, tilt_y, twist, code, buttons, mods, etype, timestamp }
118    }
119    /// Returns the pointer-device kind.
120    pub const fn kind(self) -> EventPointerKind {
121        EventPointerKind::from_web(self.code.get_kind())
122    }
123    /// Returns the compact web pointer-kind code.
124    pub const fn web_kind(self) -> u8 {
125        self.code.get_kind()
126    }
127    /// Returns `self` with the pointer-device kind replaced.
128    pub const fn with_kind(mut self, kind: EventPointerKind) -> Self {
129        self.code.set_kind(kind.to_web());
130        self
131    }
132
133    /// Returns the normalized triggering button, or `None` when no button changed.
134    pub const fn button(self) -> Option<EventButton> {
135        EventButton::from_web(self.web_button())
136    }
137    /// Returns the raw DOM-compatible `button` value.
138    ///
139    /// Returns `255` for DOM `button = -1`, meaning no button changed.
140    pub const fn web_button(self) -> u8 {
141        Self::decode_web_button(self.code.get_button())
142    }
143    /// Returns `self` with the normalized triggering button replaced.
144    pub const fn with_button(mut self, button: Option<EventButton>) -> Self {
145        let web_button = match button {
146            Some(button) => button.to_web(),
147            None => 255,
148        };
149        self.code.set_button(Self::encode_web_button(web_button));
150        self
151    }
152    /// Returns `self` with the raw DOM-compatible `button` value replaced.
153    pub const fn with_web_button(mut self, button: u8) -> Self {
154        self.code.set_button(Self::encode_web_button(button));
155        self
156    }
157
158    /* helpers */
159    const WEB_BUTTON_NONE: u8 = 15;
160    const fn encode_web_button(button: u8) -> u8 {
161        match button {
162            255 => Self::WEB_BUTTON_NONE,
163            0..=14 => button,
164            _ => Self::WEB_BUTTON_NONE,
165        }
166    }
167    const fn decode_web_button(code: u8) -> u8 {
168        match code {
169            Self::WEB_BUTTON_NONE => 255,
170            n => n,
171        }
172    }
173}

175impl WebEventPointer {
176    /// Converts `WebEventPointer` to `EventKindTimed`.
177    pub const fn to_kind_timed(self) -> EventKindTimed {
178        let kind = EventKind::Pointer(EventPointer::new(
179            self.kind(),
180            self.id as u32,
181            self.x as i32,
182            self.y as i32,
183            0,
184            0,
185            f32bits_niche::new(self.pressure as f32),
186            self.tilt_x,
187            self.tilt_y,
188            self.twist,
189            self.button(),
190            EventButtonState::from_web(self.etype),
191            EventButtons::from_bits(self.buttons),
192            self.mods,
193        ));
194        let timestamp = Some(EventTimestamp::from_js(self.timestamp));
195        EventKindTimed::new(kind, timestamp)
196    }
197    /// Converts a timed normalized `EventPointer` back to `WebEventPointer`.
198    pub const fn from_event_pointer_timed(
199        from: Timed<EventPointer, Option<EventTimestamp>>,
200    ) -> WebEventPointer {
201        let timestamp = is![let Some(t) = from.time, t.to_js(), JsInstant { ms: 0.0 }];
202        WebEventPointer::new(
203            from.value.x as js_number,
204            from.value.y as js_number,
205            from.value.get_pressure() as js_number,
206            from.value.id as js_int32,
207            from.value.tilt_x,
208            from.value.tilt_y,
209            from.value.twist,
210            from.value.kind,
211            is![let Some(b) = from.value.button, b.to_web(), 255],
212            from.value.buttons.bits(),
213            from.value.mods,
214            from.value.state.to_web_as_pointer(),
215            timestamp,
216        )
217    }
218}
219impl From<WebEventPointer> for EventKindTimed {
220    fn from(from: WebEventPointer) -> Self {
221        from.to_kind_timed()
222    }
223}
224impl From<Timed<EventPointer, Option<EventTimestamp>>> for WebEventPointer {
225    fn from(from: Timed<EventPointer, Option<EventTimestamp>>) -> Self {
226        Self::from_event_pointer_timed(from)
227    }
228}

230impl EventPointerKind {
231    /// Converts a compact web pointer kind into `EventPointerKind`.
232    pub const fn from_web(kind: u8) -> Self {
233        match kind {
234            1 => EventPointerKind::Touch,
235            2 => EventPointerKind::Pen,
236            _ => EventPointerKind::Mouse,
237        }
238    }
239    /// Converts `EventPointerKind` into a compact web pointer kind.
240    pub const fn to_web(self) -> u8 {
241        match self {
242            EventPointerKind::Mouse => 0,
243            EventPointerKind::Touch => 1,
244            EventPointerKind::Pen => 2,
245        }
246    }
247}

249crate::bitfield! {
250    /// Internal compact encoding for pointer kind and triggering button.
251    struct WebPointerCode(u8) {
252        /// Compact pointer kind: `0` mouse, `1` touch, `2` pen.
253        KIND = 0..=1;
254        /// Compact triggering-button code.
255        ///
256        /// This is not the raw DOM `button` value. In particular,
257        /// DOM `button = -1` is encoded with a private sentinel.
258        BUTTON = 2..=5;
259        /// Reserved bits for future pointer metadata.
260        RESERVED = 6..=7;
261    }
