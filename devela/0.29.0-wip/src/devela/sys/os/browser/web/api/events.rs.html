108    /// Callback dispatcher for WebAssembly events without args.
109    ///
110    /// - This function is used by JavaScript to invoke a Rust function pointer.
111    /// - It allows Rust event listeners to execute when triggered by JS.
112    /// - The `callback_ptr` is a function pointer cast from JS,
113    ///   which is then transmuted into a callable Rust function.
114    ///
115    /// # Safety
116    /// `callback_ptr` must be a valid function pointer.
117    ///
118    /// # Examples
119    /// ```ignore
120    /// #[unsafe(no_mangle)]
121    /// pub extern "C" fn my_callback() { Web::console_log("Button clicked!"); }
122    /// Web::event_add_listener("#my_button", WebEventKind::Click, my_callback);
123    /// ```
124    #[unsafe(no_mangle)]
125    pub unsafe extern "C" fn wasm_callback(callback_ptr: usize) {
126        let callback = callback_ptr as *const ();
127        let callback: extern "C" fn() = unsafe { transmute(callback) };
128        callback();
129    }
130    /// WebAssembly keyboard event callback dispatcher.
131    ///
132    /// - Called from JavaScript when a keyboard event is fired.
133    /// - Passes the `WebEventKey` struct to the Rust callback.
134    ///
135    /// # Safety
136    /// - `callback_ptr` must be a valid function pointer.
137    #[unsafe(no_mangle)]
138    pub unsafe extern "C" fn wasm_callback_key(callback_ptr: usize,
139        key: js_int32, location: js_int32, repeat: js_int32, mods: js_int32,
140        etype: js_int32, timestamp: js_number) {
141        let callback = callback_ptr as *const ();
142        let callback: extern "C" fn(WebEventKey) = unsafe { transmute(callback) };
143        let location = WebKeyLocation::from_repr(location as u8);
144        let repeat = repeat != 0;
145        let mods = KeyMods::from_web(mods as u8);
146        let etype = WebEventKind::from_repr(etype as u8);
147        let Some(state) = etype.to_key_state(repeat) else { return };
148        let timestamp = JsInstant::from_millis_f64(timestamp);
149        callback(WebEventKey::new(key as u32, state, location, mods, timestamp));
150    }

182        kind: js_int32, x: js_number, y: js_number, pressure: js_number,
183        tilt_x: js_int32, tilt_y: js_int32, twist: js_int32, button: js_int32, buttons: js_int32,
184        mods: js_int32, etype: js_int32, timestamp: js_number) {

191        callback(WebEventPointer::new(x, y, pressure, id, tilt_x as i8, tilt_y as i8,
192            twist as u16, kind, button as u8, buttons as u8, mods, etype, timestamp));
193    }
