12#[doc = crate::_tags!(event web)]
13/// A web API Wheel Event.
14#[doc = crate::_doc_meta!{
15    location("sys/os/browser/web"),
16    test_size_of(WebEventWheel = 48|384; niche Option),
17}]
18///
19/// Represents a JavaScript wheel event with browser-native deltas and unit.
20///
21/// This is a raw web-side event carrier.
22/// Use [`to_kind_timed`](Self::to_kind_timed) to normalize it into
23/// [`EventKindTimed`].
24///
25/// # Notes
26/// - `delta_x` and `delta_y` preserve the browser-provided wheel deltas.
27/// - `unit` preserves the browser-reported delta unit.
28/// - `buttons` preserves the DOM buttons bitmask as observed during the wheel event.
29/// - `x` and `y` are viewport-relative coordinates.
30///
31/// See also [`EventWheel`], [`EventWheelUnit`].
32///
33/// - <https://developer.mozilla.org/en-US/docs/Web/API/WheelEvent>
34//
35// WAIT: [Wheel-scrolling missing 1st event](https://bugzilla.mozilla.org/show_bug.cgi?id=1969373)
36// WAIT: […stops working with XInput2](https://bugzilla.mozilla.org/show_bug.cgi?id=1182700)
37#[repr(C)]
38#[derive(Copy, Clone, PartialEq)]
39pub struct WebEventWheel {
40    /// The X-coordinate of the wheel event relative to the viewport.
41    pub x: js_number, // 8 bytes
42    /// The Y-coordinate of the wheel event relative to the viewport.
43    pub y: js_number, // 8 bytes
44    /// The horizontal wheel delta reported by the browser.
45    pub delta_x: js_number, // 8 bytes
46    /// The vertical wheel delta reported by the browser.
47    pub delta_y: js_number, // 8 bytes
48
49    /// A bitmask of currently held buttons during the wheel event.
50    pub buttons: u8, // 1 byte
51    /// A bitmask of active keyboard modifiers during the mouse event.
52    pub mods: KeyMods, // 2 bytes
53    /// The browser-reported unit associated to `delta_x` and `delta_y`.
54    pub unit: EventWheelUnit, // 1 byte
55    // 4 byte gap
56    /// The JavaScript event timestamp.
57    pub timestamp: JsInstant, // 8 bytes
58}
59impl_trait! { fmt::Debug for WebEventWheel |self, f| {
60    f.debug_struct("WebEventWheel")
61        .field("x", &JsNumFmt::<2>(self.x))
62        .field("y", &JsNumFmt::<2>(self.y))
63        .field("delta_x", &JsNumFmt::<2>(self.delta_x))
64        .field("delta_y", &JsNumFmt::<2>(self.delta_y))
65        .field("buttons", &EventButtons::from_bits(self.buttons))
66        .field("mods", &self.mods)
67        .field("unit", &self.unit)
68        .field("timestamp", &self.timestamp)
69        .finish()
70}}
71impl WebEventWheel {
72    /// Returns a new [`WebEventWheel`].
73    #[allow(clippy::too_many_arguments)]
74    pub const fn new(
75        x: js_number,
76        y: js_number,
77        delta_x: js_number,
78        delta_y: js_number,
79        buttons: u8,
80        mods: KeyMods,
81        unit: EventWheelUnit,
82        timestamp: JsInstant,
83    ) -> Self {
84        Self {
85            x,
86            y,
87            delta_x,
88            delta_y,
89            buttons,
90            mods,
91            unit,
92            timestamp,
93        }
94    }
95
96    /// Returns the normalized held-button set.
97    pub const fn buttons(self) -> EventButtons {
98        EventButtons::from_bits(self.buttons)
99    }
100    /// Returns the raw DOM `buttons` bitmask.
101    pub const fn web_buttons(self) -> u8 {
102        self.buttons
103    }
104
105    /// Converts `WebEventWheel` to `EventKindTimed`.
106    pub const fn to_kind_timed(self) -> EventKindTimed {
107        let kind = EventKind::Wheel(EventWheel {
108            delta_x: self.delta_x as i32,
109            delta_y: self.delta_y as i32,
110            unit: self.unit,
111            x: self.x as i32,
112            y: self.y as i32,
113            buttons: EventButtons::from_bits(self.buttons),
114            mods: self.mods,
115        });
116        let timestamp = Some(EventTimestamp::from_js(self.timestamp));
117        EventKindTimed::new(kind, timestamp)
118    }
119
120    /// Converts a timed normalized `EventWheel` back to `WebEventWheel`.
121    pub const fn from_event_wheel_timed(
122        from: Timed<EventWheel, Option<EventTimestamp>>,
123    ) -> WebEventWheel {
124        let timestamp = is![let Some(t) = from.time, t.to_js(), JsInstant { ms: 0.0 }];
125        WebEventWheel {
126            x: from.value.x as js_number,
127            y: from.value.y as js_number,
128            delta_x: from.value.delta_x as js_number,
129            delta_y: from.value.delta_y as js_number,
130            buttons: from.value.buttons.bits(),
131            mods: from.value.mods,
132            unit: from.value.unit,
133            timestamp,
134        }
135    }
136}
137
138impl From<WebEventWheel> for EventKindTimed {
139    fn from(from: WebEventWheel) -> Self {
140        from.to_kind_timed()
141    }
142}
143impl From<Timed<EventWheel, Option<EventTimestamp>>> for WebEventWheel {
144    fn from(from: Timed<EventWheel, Option<EventTimestamp>>) -> Self {
145        Self::from_event_wheel_timed(from)
146    }
