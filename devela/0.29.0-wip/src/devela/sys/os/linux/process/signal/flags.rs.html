3//! Defines [`LinuxSigactionFlags`].
4//
5
6#[cfg(doc)]
7use crate::LinuxSigaction;
8use crate::{LINUX_SIGACTION as A, c_size_t, set};
9
10set! {
11    #[doc = crate::_tags!(linux signal set)]
12    /// Input flags for [`LinuxSigaction`].
13    #[doc = crate::_doc_meta!{
14        location("sys/os/term/session"),
15        test_size_of(LinuxSigactionFlags = 1|8),
16    }]
17    pub struct LinuxSigactionFlags(u8) {
18        /// Do not notify when child processes stop or resume.
19        SA_NOCLDSTOP = 0;
20        /// Do not create zombies when child processes terminate.
21        SA_NOCLDWAIT = 1;
22        /// Do not mask the signal while its handler is running.
23        SA_NODEFER = 2;
24        /// Run the handler on an alternate signal stack.
25        SA_ONSTACK = 3;
26        /// Reset the signal action to the default on handler entry.
27        SA_RESETHAND = 4;
28        /// Restart certain interrupted system calls.
29        SA_RESTART = 5;
30        /// Use the explicit Linux signal-restorer trampoline.
31        SA_RESTORER = 6;
32        /// Use a three-argument `SA_SIGINFO` handler.
33        SA_SIGINFO = 7;
34    }
35    impl {
36        /// Returns flags from the raw Linux `sa_flags` word.
37        #[must_use]
38        pub const fn from_c_size_t(bits: c_size_t) -> Self {
39            Self::new()
40                .with_if(bits & (A::SA_NOCLDSTOP as c_size_t) != 0, Self::SA_NOCLDSTOP)
41                .with_if(bits & (A::SA_NOCLDWAIT as c_size_t) != 0, Self::SA_NOCLDWAIT)
42                .with_if(bits & (A::SA_NODEFER as c_size_t) != 0, Self::SA_NODEFER)
43                .with_if(bits & (A::SA_ONSTACK as c_size_t) != 0, Self::SA_ONSTACK)
44                .with_if(bits & (A::SA_RESETHAND as c_size_t) != 0, Self::SA_RESETHAND)
45                .with_if(bits & (A::SA_RESTART as c_size_t) != 0, Self::SA_RESTART)
46                .with_if(bits & (A::SA_RESTORER as c_size_t) != 0, Self::SA_RESTORER)
47                .with_if(bits & (A::SA_SIGINFO as c_size_t) != 0, Self::SA_SIGINFO)
48        }
49        /// Returns flags as a raw Linux `sa_flags` word.
50        #[must_use]
51        pub const fn as_c_size_t(self) -> c_size_t {
52            let mut bits = 0;
53            if self.has(Self::SA_NOCLDSTOP) { bits |= A::SA_NOCLDSTOP as c_size_t; }
54            if self.has(Self::SA_NOCLDWAIT) { bits |= A::SA_NOCLDWAIT as c_size_t; }
55            if self.has(Self::SA_NODEFER)   { bits |= A::SA_NODEFER as c_size_t; }
56            if self.has(Self::SA_ONSTACK)   { bits |= A::SA_ONSTACK as c_size_t; }
57            if self.has(Self::SA_RESETHAND) { bits |= A::SA_RESETHAND as c_size_t; }
58            if self.has(Self::SA_RESTART)   { bits |= A::SA_RESTART as c_size_t; }
59            if self.has(Self::SA_RESTORER)  { bits |= A::SA_RESTORER as c_size_t; }
60            if self.has(Self::SA_SIGINFO)   { bits |= A::SA_SIGINFO as c_size_t; }
61            bits
62        }
63        const _ASSERT_RAW_VALUES: () = {
64            assert!(Self::SA_NOCLDSTOP.as_c_size_t() == A::SA_NOCLDSTOP);
65            assert!(Self::SA_NOCLDWAIT.as_c_size_t() == A::SA_NOCLDWAIT);
66            assert!(Self::SA_NODEFER.as_c_size_t()   == A::SA_NODEFER);
67            assert!(Self::SA_ONSTACK.as_c_size_t()   == A::SA_ONSTACK);
68            assert!(Self::SA_RESETHAND.as_c_size_t() == A::SA_RESETHAND);
69            assert!(Self::SA_RESTART.as_c_size_t()   == A::SA_RESTART);
70            assert!(Self::SA_RESTORER.as_c_size_t()  == A::SA_RESTORER);
71            assert!(Self::SA_SIGINFO.as_c_size_t()   == A::SA_SIGINFO);
72        };
73    }
74}
75impl From<Option<LinuxSigactionFlags>> for LinuxSigactionFlags {
76    fn from(flags: Option<LinuxSigactionFlags>) -> Self {
77        flags.unwrap_or_default()
78    }
