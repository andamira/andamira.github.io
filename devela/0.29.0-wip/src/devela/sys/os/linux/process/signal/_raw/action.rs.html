6use crate::c_size_t;

10#[doc = crate::_tags!(linux signal abi)]
11/// Linux flag constants for [`LinuxSigaction`].
12#[doc = crate::_doc_meta!{location("sys/os/linux/process")}]
13#[allow(non_camel_case_types)]
14#[derive(Debug)]
15pub(crate) struct LINUX_SIGACTION;
16
17// 8 flags
18impl LINUX_SIGACTION {
19    /// If signum is [`SIGCHLD`], do not receive notification when child processes
20    /// stop (i.e., when they receive one of [`SIGSTOP`], [`SIGTSTP`], [`SIGTTIN`],
21    /// or [`SIGTTOU`]) or resume (i.e., they receive [`SIGCONT`])
22    /// (see [`wait(2)`]).
23    ///
24    /// This flag is meaningful only when establishing a handler for `SIGCHLD`.
25    ///
26    /// [`SIGCHLD`]: LINUX_SIGNAL::SIGCHLD
27    /// [`SIGSTOP`]: LINUX_SIGNAL::SIGSTOP
28    /// [`SIGTSTP`]: LINUX_SIGNAL::SIGTSTP
29    /// [`SIGTTIN`]: LINUX_SIGNAL::SIGTTIN
30    /// [`SIGTTOU`]: LINUX_SIGNAL::SIGTTOU
31    /// [`SIGCONT`]: LINUX_SIGNAL::SIGCONT
32    /// [`wait(2)`]: https://man7.org/linux/man-pages/man2/wait.2.html
33    pub const SA_NOCLDSTOP: c_size_t = 0x0000_0001;
34
35    /// If signum is [`SIGCHLD`], do not transform children into
36    /// zombies when they terminate.
37    ///
38    /// See also [`waitpid(2)`].
39    ///
40    /// This flag is meaningful only when establishing a handler for `SIGCHLD`,
41    /// or when setting that signal's disposition to [`SIG_DFL`].
42    ///
43    /// (since Linux 2.6)
44    ///
45    /// [`SIGCHLD`]: LINUX_SIGNAL::SIGCHLD
46    /// [`SIG_DFL`]: LinuxSigaction::SIG_DFL
47    /// [`waitpid(2)`]: https://man7.org/linux/man-pages/man2/wait.2.html
48    pub const SA_NOCLDWAIT: c_size_t = 0x0000_0002;
49
50    /// If the [`SA_NOCLDWAIT`] flag is set when establishing a handler for
51    /// [`SIGCHLD`], POSIX.1 leaves it unspecified whether a `SIGCHLD` signal is
52    /// generated when a child process terminates.
53    ///
54    /// On Linux, a `SIGCHLD` signal is generated in this case; on some other
55    /// implementations, it is not.
56    ///
57    /// Do not add the signal to the thread's signal mask while the handler is
58    /// executing, unless the signal is specified in act.sa_mask.  Consequently,
59    /// a further instance of the signal may be delivered to the thread while it
60    /// is executing the handler.  This flag is meaningful only when establishing
61    /// a signal handler.
62    ///
63    /// `SA_NOMASK` is an obsolete, nonstandard synonym for this flag.
64    ///
65    /// [`SIGCHLD`]: LINUX_SIGNAL::SIGCHLD
66    /// [`SA_NOCLDWAIT`]: Self::SA_NOCLDWAIT
67    pub const SA_NODEFER: c_size_t = 0x4000_0000;
68
69    /// Call the signal handler on an alternate signal stack provided by
70    /// [`sigaltstack(2)`].
71    ///
72    /// If an alternate stack is not available, the default stack will be used.
73    /// This flag is meaningful only when establishing a signal handler.
74    ///
75    /// [`sigaltstack(2)`]: https://man7.org/linux/man-pages/man2/sigaltstack.2.html
76    pub const SA_ONSTACK: c_size_t = 0x0800_0000;
77
78    /// Restore the signal action to the default upon entry to the signal handler.
79    ///
80    /// This flag is meaningful only when establishing a signal handler.
81    ///
82    /// `SA_ONESHOT` is an obsolete, nonstandard synonym for this flag.
83    pub const SA_RESETHAND: c_size_t = 0x8000_0000;
84
85    /// Provide behavior compatible with BSD signal semantics by making certain
86    /// system calls restartable across signals. This flag is meaningful only
87    /// when establishing a signal handler.  See [`signal(7)`] for a discussion of
88    /// system call restarting.
89    ///
90    /// [`signal(7)`]: https://man7.org/linux/man-pages/man7/signal.7.html
91    pub const SA_RESTART: c_size_t = 0x1000_0000;
92
93    /// Not intended for application use.
94    ///
95    /// This flag is used by C libraries to indicate that the `sa_restorer` field contains
96    /// the address of a "signal trampoline".  See [`sigreturn(2)`] for more details.
97    ///
98    /// [`sigreturn(2)`]: https://man7.org/linux/man-pages/man2/sigreturn.2.html
99    pub const SA_RESTORER: c_size_t = 0x0400_0000;
100
101    /// The signal handler takes three arguments, not one.
102    ///
103    /// In this case, `sa_sigaction` should be set instead of `sa_handler`.
104    ///
105    /// This flag is meaningful only when establishing a signal handler.
106    ///
107    /// (since Linux 2.2)
108    pub const SA_SIGINFO: c_size_t = 0x0000_0004;
109}
