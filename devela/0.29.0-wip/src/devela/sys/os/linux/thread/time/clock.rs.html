13/// [`Linux`][crate::Linux] clock identifiers.
14///
15/// Used for [`sys_clock_gettime`][crate::Linux::sys_clock_gettime] and related time functions.
16#[doc = crate::_doc_meta!{location("sys/os/linux/thread")}]
17///
18/// See [clock_gettime(2)] for detailed information about each clock type.
19///
20/// [clock_gettime(2)]: https://www.man7.org/linux/man-pages/man2/clock_gettime.2.html
21#[repr(i32)] // == c_int
22#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
23pub enum LinuxClock {
24    /// System-wide realtime clock (wall clock). Adjustable via NTP.
25    Realtime = 0,
26
27    /// Monotonic clock (steady since boot). Unaffected by NTP.
28    Monotonic = 1,
29
30    /// A clock that measures CPU time consumed by this process.
31    // since Linux 2.6.12
32    ProcessCpuTimeId = 2,
33
34    /// A clock that measures CPU time consumed by this thread.
35    // since Linux 2.6.12
36    ThreadCpuTimeId = 3,
37
38    /// Monotonic, raw hardware time (TSC-based where available).
39    // since Linux 2.6.28
40    MonotonicRaw = 4,
41
42    /// A faster but less precise version of [`Realtime`][Self::Realtime].
43    // since Linux 2.6.32
44    RealtimeCoarse = 5,
45
46    /// A faster but less precise version of [`Monotonic`][Self::Monotonic].
47    // since Linux 2.6.32
48    MonotonicCoarse = 6,
49
50    /// Like [`Monotonic`][Self::Monotonic] but includes suspend time.
51    // since Linux 2.6.39
52    Boottime = 7,
53
54    /// Like [`Realtime`][Self::Realtime], but not settable.
55    // since Linux 3.0
56    RealtimeAlarm = 8,
57
58    /// Like [`Boottime`][Self::Boottime], but not settable.
59    BoottimeAlarm = 9,
60
61    // The driver implementing this got removed
62    // SgiCycle = 10,
63    /// International Atomic Time (ignores leap seconds).
64    // since Linux 3.10
65    Tai = 11,
66}
67
68impl TryFrom<c_int> for LinuxClock {
69    type Error = LinuxError;
70    fn try_from(value: c_int) -> Result<Self> {
71        Self::from_raw(value).ok_or(LinuxError::Sys(LINUX_ERRNO::EINVAL))
72    }
73}
74impl From<LinuxClock> for c_int {
75    fn from(clock: LinuxClock) -> Self {
76        clock.as_raw()
77    }
78}

80impl LinuxClock {
81    /// Converts to the raw integer value for syscalls.
82    pub const fn as_raw(self) -> c_int {
83        self as c_int
84    }
85    /// Converts from a raw integer (for FFI or error handling).
86    pub const fn from_raw(value: c_int) -> Option<Self> {
87        match value {
88            0 => Some(Self::Realtime),
89            1 => Some(Self::Monotonic),
90            2 => Some(Self::ProcessCpuTimeId),
91            3 => Some(Self::ThreadCpuTimeId),
92            4 => Some(Self::MonotonicRaw),
93            5 => Some(Self::RealtimeCoarse),
94            6 => Some(Self::MonotonicCoarse),
95            7 => Some(Self::Boottime),
96            8 => Some(Self::RealtimeAlarm),
97            9 => Some(Self::BoottimeAlarm),
98            11 => Some(Self::Tai),
99            _ => None,
100        }
101    }
102}
103
104impl LinuxClock {
105    /// Is this clock adjusted by NTP or system time changes?
106    pub fn is_ntp_adjusted(self) -> bool {
107        matches!(self, Self::Realtime | Self::RealtimeCoarse)
108        // Note: TAI is not NTP-adjusted but can be manually set
109    }
110
111    /// Is this clock based on raw hardware (completely unadjusted)?
112    pub fn is_raw_hardware(self) -> bool {
113        matches!(self, Self::MonotonicRaw)
114        // TAI is also "raw" in a sense but can be set, so excluded
115    }
116
117    /// Is this clock settable via `clock_settime()`?
118    ///
119    /// Note that only alarm clocks are specifically NOT settable.
120    pub fn is_settable(self) -> bool {
121        !self.is_alarm()
122    }
123
124    /// Is this a monotonic clock? (never goes backwards)
125    ///
126    /// Note that CPU time clocks are monotonic for their process/thread.
127    pub fn is_monotonic(self) -> bool {
128        matches!(
129            self,
130            Self::Monotonic
131                | Self::MonotonicRaw
132                | Self::MonotonicCoarse
133                | Self::Boottime
134                | Self::BoottimeAlarm
135                | Self::ProcessCpuTimeId
136                | Self::ThreadCpuTimeId
137        )
138    }
139
140    /// Is this a realtime (wall clock) type? (represents calendar time)
141    pub fn is_realtime(self) -> bool {
142        matches!(self, Self::Realtime | Self::RealtimeCoarse | Self::RealtimeAlarm | Self::Tai)
143    }
144
145    /// Is this a CPU time measurement clock? (measures actual CPU usage)
146    pub fn is_cpu_time(self) -> bool {
147        matches!(self, Self::ProcessCpuTimeId | Self::ThreadCpuTimeId)
148    }
149
150    /// Is this an alarm clock? (for timerfd/signals, not settable)
151    pub fn is_alarm(self) -> bool {
152        matches!(self, Self::RealtimeAlarm | Self::BoottimeAlarm)
153    }
154
155    /// Does this clock include time during system suspend?
156    pub fn includes_suspend_time(self) -> bool {
157        matches!(self, Self::Boottime | Self::BoottimeAlarm)
158    }
159}

161#[crate::macro_apply(crate::_unsafe_syscall_not_miri)]
162impl LinuxClock {
163    /// Gets the current time for this clock
164    pub fn get_time(self) -> Result<LinuxTimespec> {
165        Linux::clock_gettime(self)
166    }
167    /// Gets the current time for this clock
168    pub fn get_duration(self) -> Result<Duration> {
169        Linux::clock_gettime(self).and_then(|ts| ts.try_to_duration().map_err(Into::into))
170    }
171
172    /// Gets the resolution for this clock
173    pub fn get_resolution(self) -> Result<LinuxTimespec> {
174        Linux::clock_getres(self)
175    }
176}
