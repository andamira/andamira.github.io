6use crate::{Linux, LinuxClock, LinuxTimespec, TimeScale, TimeSource, TimeSourceCfg};
7
8#[doc = crate::_tags!(linux time)]
9/// A fast monotonic Linux time source.
10#[doc = crate::_doc_meta!{location("sys/os/linux/thread")}]
11///
12/// This is the canonical fixed Linux numeric source.
13/// It uses `CLOCK_MONOTONIC` and projects time as `u64` nanoseconds.
14#[derive(Debug)]
15pub struct LinuxInstant;
16
17#[inline(always)]
18fn linux_ts_nanos(ts: LinuxTimespec) -> u64 {
19    (ts.tv_sec as u64) * 1_000_000_000 + (ts.tv_nsec as u64)
20}
21
22#[rustfmt::skip]
23/// Relative `u64` projection of [`LinuxInstant`] in nanoseconds.
24///
25/// This is the canonical wide numeric projection for `CLOCK_MONOTONIC`.
26/// At nanosecond resolution, `u64` spans about 584 years.
27impl TimeSource<u64> for LinuxInstant {
28    fn time_is_monotonic() -> bool { true }
29    fn time_is_absolute() -> bool { false }
30    fn time_scale() -> TimeScale { TimeScale::Nanos }
31    fn time_now() -> u64 {
32        // SAFETY: CLOCK_MONOTONIC is always valid.
33        let ts = unsafe {
34            let mut ts = LinuxTimespec::default();
35            let _ = Linux::sys_clock_gettime(LinuxClock::Monotonic, ts.as_mut_ptr());
36            ts
37        };
38        linux_ts_nanos(ts)
39    }
40    fn time_point_value(point: u64) -> u64 { point }
41    fn time_elapsed_value(elapsed: u64) -> u64 { elapsed }
42}
43
44/// Relative `u32` projection of [`LinuxInstant`] in microseconds.
45///
46/// This compact projection trades range for storage size while keeping
47/// sub-millisecond precision useful for short and medium-lived runtimes.
48///
49/// At microsecond resolution, `u32` spans about 71.6 minutes.
50#[rustfmt::skip]
51impl TimeSource<u32> for LinuxInstant {
52    fn time_is_monotonic() -> bool { true }
53    fn time_is_absolute() -> bool { false }
54    fn time_scale() -> TimeScale { TimeScale::Micros }
55    fn time_now() -> u32 {
56        // SAFETY: CLOCK_MONOTONIC is always valid.
57        let ts = unsafe {
58            let mut ts = LinuxTimespec::default();
59            let _ = Linux::sys_clock_gettime(LinuxClock::Monotonic, ts.as_mut_ptr());
60            ts
61        };
62        let ts_micros = (ts.tv_sec as u64) * 1_000_000 + (ts.tv_nsec as u64) / 1_000;
63        u32::try_from(ts_micros).expect("LinuxInstant u32 microsecond projection overflow")
64    }
65    fn time_point_value(point: u32) -> u64 { point.into() }
66    fn time_elapsed_value(elapsed: u32) -> u64 { elapsed.into() }
67}

69#[doc = crate::_tags!(linux time)]
70/// A configurable family of Linux clocks.
71#[doc = crate::_doc_meta!{location("sys/os/linux/thread")}]
72///
73/// Each [`LinuxClock`] selects a concrete Linux timeline and exposes it
74/// through the common `TimeSourceCfg<u64>` interface using nanosecond values.
75#[derive(Debug)]
76pub struct LinuxTime;
77
78#[rustfmt::skip]
79/// Configurable `u64` projection of [`LinuxTime`] in nanoseconds.
80///
81/// This is the canonical configurable wide numeric projection for Linux clocks.
82/// At nanosecond resolution, `u64` spans about 584 years.
83///
84/// Whether the selected timeline is monotonic or absolute depends on the chosen [`LinuxClock`].
85impl TimeSourceCfg<u64> for LinuxTime {
86    type Config = LinuxClock;
87
88    fn time_is_monotonic(clock: LinuxClock) -> bool {
89        matches!(
90            clock,
91            LinuxClock::Monotonic
92                | LinuxClock::MonotonicRaw
93                | LinuxClock::MonotonicCoarse
94                | LinuxClock::Boottime
95                | LinuxClock::BoottimeAlarm
96                | LinuxClock::Tai
97        )
98    }
99    fn time_is_absolute(clock: LinuxClock) -> bool {
100        matches!(
101            clock,
102            LinuxClock::Realtime
103                | LinuxClock::RealtimeCoarse
104                | LinuxClock::RealtimeAlarm
105                | LinuxClock::Tai
106        )
107    }
108    #[inline(always)]
109    fn time_scale(_: LinuxClock) -> TimeScale { TimeScale::Nanos }
110    #[inline(always)]
111    fn time_now(clock: LinuxClock) -> u64 {
112        let ts = Linux::clock_gettime(clock).expect("clock_gettime failed");
113        linux_ts_nanos(ts)
114    }
115    #[inline(always)]
116    fn time_point_value(_: LinuxClock, point: u64) -> u64 { point }
117    #[inline(always)]
118    fn time_elapsed_value(_: LinuxClock, elapsed: u64) -> u64 { elapsed }
119}
