24/// Represents the [`termios`] structure from libc, used to control terminal I/O.
25#[doc = crate::_doc_meta!{
26    location("sys/os/linux/io"),
27    test_size_of(LinuxTermios = 36|288),
28}]
29///
30/// It has fields for input, output, control, and local modes,
31/// as well as a line discipline and control characters.
32///
33/// [`termios`]: https://man7.org/linux/man-pages/man3/termios.3.html
34#[repr(C)]
35#[derive(Clone, Copy, Debug, Default, PartialEq)]
36pub struct LinuxTermios {
37    /// Input flags.
38    pub c_iflag: c_uint,
39
40    /// Output flags.
41    pub c_oflag: c_uint,
42
43    /// Control flags.
44    pub c_cflag: c_uint,
45
46    /// Local flags.
47    pub c_lflag: c_uint,
48
49    /// …
50    pub c_line: u8,
51
52    /// Special control characters.
53    pub c_cc: [u8; 19],
54}

82/// # Linux terminal state syscalls
83#[cfg(any_target_arch_linux)]
84#[crate::macro_apply(crate::_unsafe_syscall_not_miri)]
85impl LinuxTermios {
86    /// Reads the current terminal termios state.
87    pub fn read_state() -> Result<LinuxTermios> {
88        let mut state = LinuxTermios::new();
89        let res = unsafe {
90            Linux::sys_ioctl(LINUX_FILENO::STDIN, LINUX_IOCTL::TCGETS, state.as_mut_bytes_ptr())
91        };
92        is![res >= 0, Ok(state), Err(LinuxError::Sys(res))]
93    }
94    /// Writes this termios state as the current terminal state.
95    pub fn write_state(mut self) -> Result<()> {
96        let res = unsafe {
97            Linux::sys_ioctl(LINUX_FILENO::STDIN, LINUX_IOCTL::TCSETS, self.as_mut_bytes_ptr())
98        };
99        is![res >= 0, Ok(()), Err(LinuxError::Sys(res))]
100    }
101    /// Reads the current termios state, mutates it, and writes it back.
102    pub fn update_state(f: impl FnOnce(&mut Self)) -> Result<()> {
103        let mut state = Self::read_state()?;
104        f(&mut state);
105        state.write_state()
106    }
107
108    /// Returns `true` if standard input is attached to a terminal.
109    #[must_use]
110    pub fn is_terminal() -> bool {
111        match Self::read_state() {
112            Ok(_) => true,
113            Err(LinuxError::Sys(err)) => err != -LINUX_ERRNO::ENOTTY && err != -LINUX_ERRNO::EINVAL,
114            Err(_) => false, // Other errors are not related to terminal checks
115        }
116    }
117    /// Reads the terminal window size, in cells and pixels.
118    pub fn read_window_size() -> Result<TermSize> {
119        let mut winsize = TermSize::default();
120        let res = unsafe {
121            Linux::sys_ioctl(
122                LINUX_FILENO::STDIN,
123                LINUX_IOCTL::TIOCGWINSZ,
124                &mut winsize as *mut TermSize as *mut u8,
125            )
126        };
127        is![res >= 0, Ok(winsize), Err(LinuxError::Sys(res))]
128    }
129
130    /// Enables event-oriented terminal mode.
131    pub fn enable_event_mode() -> Result<()> {
132        Self::update_state(|state| state.make_event())
133    }
134    /// Enables raw terminal mode.
135    pub fn enable_raw_mode() -> Result<()> {
136        Self::update_state(|state| state.make_raw())
137    }
138    /// Best-effort reset to ordinary line-buffered terminal behavior.
139    ///
140    /// This does not restore the exact previous terminal state.
141    /// Prefer a scoped terminal session for exact restoration.
142    pub fn reset_cooked_mode() -> Result<()> {
143        Self::update_state(|state| state.make_cooked_reset())
144    }
145}

147/// # Terminal state transforms
148#[rustfmt::skip]
149impl LinuxTermios {
150    /// Makes this state event-oriented.
151    ///
152    /// Event mode disables canonical line buffering and echo, allowing input to
153    /// be read immediately while preserving terminal-generated signals.
154    pub const fn make_event(&mut self) {
155        self.remove_input_flags(I::IXON);
156        self.remove_local_flags(L::ECHO);
157        self.remove_local_flags(L::ICANON);
158        // self.remove_local_flags(L::IEXTEN);
159        self.set_read_min_timeout(1, 0);
160    }
161    /// Makes this state raw.
162    ///
163    /// Raw mode disables canonical input, echo, software flow control, input
164    /// translations, output processing, extensions, and terminal-generated
165    /// signal characters.
166    ///
167    /// In this mode, control keys such as Ctrl-C and Ctrl-Q are delivered as
168    /// input bytes instead of being handled by the terminal line discipline.
169    pub const fn make_raw(&mut self) {
170        self.remove_input_flags(I::BRKINT);
171        self.remove_input_flags(I::ICRNL);
172        self.remove_input_flags(I::INPCK);
173        self.remove_input_flags(I::ISTRIP);
174        self.set_software_flow(false);
175
176        self.remove_output_flags(O::OPOST);
177
178        self.remove_local_flags(L::ECHO);
179        self.remove_local_flags(L::ICANON);
180        self.remove_local_flags(L::IEXTEN);
181        self.remove_local_flags(L::ISIG);
182
183        self.set_char_size(LinuxTermiosCharSize::Bits8);
184        self.set_read_min_timeout(1, 0);
185    }
186    /// Makes this state conventionally line-buffered.
187    ///
188    /// This is a best-effort convenience reset, not an exact restoration of the
189    /// previous terminal state. For exact restoration, use a scoped terminal session.
190    pub const fn make_cooked_reset(&mut self) {
191        // input
192        self.set_software_flow(true);
193        self.set_input_cr_to_lf(true);
194        // local
195        self.set_canonical(true);
196        self.set_echo(true);
197        self.set_signals(true);
198        self.set_extensions(true);
199        self.set_output_processing(true);
200        self.set_read_min_timeout(1, 0);
201    }
202
203    /* local */
204    /// Enables or disables echoing typed input.
205    pub const fn set_echo(&mut self, enabled: bool) {
206        if enabled { self.insert_local_flags(L::ECHO); }
207        else { self.remove_local_flags(L::ECHO); }
208    }
209    /// Enables or disables canonical line buffering.
210    pub const fn set_canonical(&mut self, enabled: bool) {
211        if enabled { self.insert_local_flags(L::ICANON); }
212        else { self.remove_local_flags(L::ICANON); }
213    }
214    /// Enables or disables terminal-generated signal characters.
215    ///
216    /// When enabled, characters such as Ctrl-C and Ctrl-Z are handled by the
217    /// terminal line discipline instead of being delivered as ordinary input.
218    pub const fn set_signals(&mut self, enabled: bool) {
219        if enabled { self.insert_local_flags(L::ISIG); }
220        else { self.remove_local_flags(L::ISIG); }
221    }
222    /// Enables or disables implementation-defined line discipline extensions.
223    pub const fn set_extensions(&mut self, enabled: bool) {
224        if enabled { self.insert_local_flags(L::IEXTEN); }
225        else { self.remove_local_flags(L::IEXTEN); }
226    }
227
228    /* input */
229    /// Enables or disables interrupt generation on BREAK.
230    pub const fn set_break_interrupt(&mut self, enabled: bool) {
231        if enabled { self.insert_input_flags(I::BRKINT); }
232        else { self.remove_input_flags(I::BRKINT); }
233    }
234    /// Enables or disables translating carriage return to newline on input.
235    pub const fn set_input_cr_to_lf(&mut self, enabled: bool) {
236        if enabled { self.insert_input_flags(I::ICRNL); }
237        else { self.remove_input_flags(I::ICRNL); }
238    }
239    /// Enables or disables XON/XOFF software flow control.
240    pub const fn set_software_flow(&mut self, enabled: bool) {
241        if enabled {
242            // decides whether Ctrl-S/Ctrl-Q are intercepted for output flow control
243            self.insert_input_flags(I::IXON);
244            // sends start/stop control characters to the other side when input queues fill
245            self.insert_input_flags(I::IXOFF);
246        } else {
247            self.remove_input_flags(I::IXON);
248            self.remove_input_flags(I::IXOFF);
249        }
250    }
251
252    /* output */
253    /// Enables or disables terminal output post-processing.
254    pub const fn set_output_processing(&mut self, enabled: bool) {
255        if enabled { self.insert_output_flags(O::OPOST); }
256        else { self.remove_output_flags(O::OPOST); }
257    }
258
259    /* control characters */
260    /// Sets noncanonical read blocking behavior.
261    ///
262    /// - `min` is the minimum number of bytes required to satisfy a read.
263    /// - `timeout_deciseconds` is the inter-byte timeout in tenths of a second.
264    pub const fn set_read_min_timeout(&mut self, min: u8, timeout_deciseconds: u8) {
265        self.set_cc(Cc::VMIN, min);
266        self.set_cc(Cc::VTIME, timeout_deciseconds);
267    }
268}

338/// # Typed mode-field accessors
339impl LinuxTermios {
340    /// Returns the configured character size.
341    #[must_use]
342    pub const fn char_size(&self) -> LinuxTermiosCharSize {
343        let bits = self.c_cflag & C::CSIZE.as_c_uint();
344        if bits == C::CS6.as_c_uint() {
345            LinuxTermiosCharSize::Bits6
346        } else if bits == C::CS7.as_c_uint() {
347            LinuxTermiosCharSize::Bits7
348        } else if bits == C::CS8.as_c_uint() {
349            LinuxTermiosCharSize::Bits8
350        } else {
351            LinuxTermiosCharSize::Bits5
352        }
353    }
354    /// Sets the configured character size.
355    pub const fn set_char_size(&mut self, size: LinuxTermiosCharSize) {
356        let mask = C::CSIZE.as_c_uint();
357        self.c_cflag = (self.c_cflag & !mask) | size.to_control_flags().as_c_uint();
358    }
359}

378/// A Linux termios character size.
379#[doc = crate::_doc_meta!{location("sys/os/linux/io/term")}]
380#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, PartialOrd, Ord, Hash)]
381#[repr(u8)]
382pub enum LinuxTermiosCharSize {
383    /// 5-bit characters.
384    Bits5 = 5,
385    /// 6-bit characters.
386    Bits6 = 6,
387    /// 7-bit characters.
388    Bits7 = 7,
389    /// 8-bit characters. (This the default)
390    #[default]
391    Bits8 = 8,
392}
393impl LinuxTermiosCharSize {
394    /// Returns the corresponding control-flag field value.
395    #[must_use]
396    pub const fn to_control_flags(self) -> C {
397        match self {
398            Self::Bits5 => C::CS5,
399            Self::Bits6 => C::CS6,
400            Self::Bits7 => C::CS7,
401            Self::Bits8 => C::CS8,
402        }
403    }
