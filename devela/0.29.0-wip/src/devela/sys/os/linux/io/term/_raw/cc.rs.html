3use crate::c_uchar;
4
5#[doc = crate::_tags!(linux term)]
6/// [`LinuxTermios`][crate::LinuxTermios] special characters symbolic indices.
7#[doc = crate::_doc_meta!{location("sys/os/linux/io")}]
8#[derive(Debug)]
9pub struct LINUX_TERMIOS_CC;
10
11impl LINUX_TERMIOS_CC {
12    /// Interrupt character (INTR).  Send a SIGINT signal.
13    ///
14    /// (003, ETX, Ctrl-C, or also 0177, DEL, rubout)
15    ///
16    /// Recognized when ISIG is set, and then not passed as input.
17    pub const VINTR: c_uchar = 0;
18
19    /// (034, FS, Ctrl-\) Quit character (QUIT).
20    ///
21    /// Send SIGQUIT signal.
22    ///
23    /// Recognized when ISIG is set, and then not passed as input.
24    pub const VQUIT: c_uchar = 1;
25
26    /// Erase character (ERASE).
27    ///
28    /// This erases the previous not-yet-erased character,
29    /// but does not erase past EOF or beginning-of- line.
30    ///
31    /// (0177, DEL, rubout, or 010, BS, Ctrl-H, or also #)
32    ///
33    /// Recognized when ICANON is set, and then not passed as input.
34    pub const VERASE: c_uchar = 2;
35
36    /// Kill character (KILL).
37    ///
38    /// This erases the input since the last EOF or beginning-of-line.
39    ///
40    /// (025, NAK, Ctrl-U, or Ctrl-X, or also @)
41    ///
42    /// Recognized when ICANON is set, and then not passed as input.
43    pub const VKILL: c_uchar = 3;
44
45    /// End-of-file character (EOF).
46    ///
47    /// More precisely: this character causes the pending tty buffer to be sent
48    /// to the waiting user program without waiting for end-of-line.
49    ///
50    /// If it is the first character of the line, the read(2) in the user
51    /// program returns 0, which signifies end- of-file.
52    ///
53    /// (004, EOT, Ctrl-D)
54    ///
55    /// Recognized when ICANON is set, and then not passed as input.
56    pub const VEOF: c_uchar = 4;
57
58    /// Timeout in deciseconds for noncanonical read (TIME).
59    pub const VTIME: c_uchar = 5;
60
61    /// Minimum number of characters for noncanonical read (MIN).
62    pub const VMIN: c_uchar = 6;
63
64    /// Switch character (SWTCH).
65    ///
66    /// (not in POSIX; not supported under Linux; 0, NUL)
67    ///
68    /// Used in System V to switch shells in shell layers, a predecessor to shell job control.
69    pub const VSWTC: c_uchar = 7;
70    // pub const VSWTCH : c_uchar = 7;
71
72    /// Start character (START).
73    ///
74    /// (021, DC1, Ctrl-Q)
75    ///
76    /// Restarts output stopped by the Stop character.
77    ///
78    /// Recognized when IXON is set, and then not passed as input.
79    pub const VSTART: c_uchar = 8;
80
81    /// Stop character (STOP).
82    ///
83    /// (023, DC3, Ctrl-S)
84    /// Stop output until Start character typed.
85    ///
86    /// Recognized when IXON is set, and then not passed as input.
87    pub const VSTOP: c_uchar = 9;
88
89    /// Suspend character (SUSP).
90    ///
91    /// Send SIGTSTP signal.
92    ///
93    /// (032, SUB, Ctrl-Z)
94    ///
95    /// Recognized when ISIG is set, and then not passed as input.
96    pub const VSUSP: c_uchar = 10;
97
98    /// Additional end-of-line character (EOL).
99    ///
100    /// (0, NUL)
101    ///
102    /// Recognized when ICANON is set.
103    pub const VEOL: c_uchar = 11;
104
105    /// Reprint unread characters (REPRINT).
106    ///
107    /// (not in POSIX; 022, DC2, Ctrl-R)
108    ///
109    /// Recognized when ICANON and IEXTEN are set, and then not passed as input.
110    pub const VREPRINT: c_uchar = 12;
111
112    /// (not in POSIX; not supported under Linux; 017, SI, Ctrl-O)
113    /// Toggle: start/stop discarding pending output.  Recognized
114    /// when IEXTEN is set, and then not passed as input.
115    pub const VDISCARD: c_uchar = 13;
116
117    /// Word erase (WERASE).
118    ///
119    /// (not in POSIX; 027, ETB, Ctrl-W)
120    ///
121    /// Recognized when ICANON and IEXTEN are set, and then not passed as input.
122    pub const VWERASE: c_uchar = 14;
123
124    /// Literal next (LNEXT).
125    ///
126    /// Quotes the next input character, depriving it of a possible special meaning.
127    ///
128    /// (not in POSIX; 026, SYN, Ctrl-V)
129    ///
130    /// Recognized when IEXTEN is set, and then not passed as input.
131    pub const VLNEXT: c_uchar = 15;
132
133    /// Yet another end-of-line character (EOL2).
134    ///
135    /// (not in POSIX; 0, NUL)
136    ///
137    /// Recognized when ICANON is set.
138    pub const VEOL2: c_uchar = 16;
139
140    // Status character (STATUS).
141    //
142    // Display status information at terminal, including state of
143    // foreground process and amount of CPU time it has consumed.
144    // Also sends a SIGINFO signal (not supported on Linux) to the
145    // foreground process group.
146    //
147    // (not in POSIX; not supported under Linux; status request: 024, DC4, Ctrl-T).
148}
