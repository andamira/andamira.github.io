19    /// Registers multiple signals using a simple handler function.
20    ///
21    /// This is a convenience wrapper over [`sig_handler_info`][Self::sig_handler_info].
22    /// Internally it installs an `SA_SIGINFO` handler and discards the extra signal
23    /// information before calling `handler`.
24    ///
25    /// # Arguments
26    /// - `handler`: Function called with the received signal number.
27    /// - `signals`: Signals to handle.
28    /// - `flags`: Additional signal action flags.
29    ///
30    /// # Notes
31    /// - `SA_RESTORER` and `SA_SIGINFO` are always included internally.
32    /// - `SIGKILL` and `SIGSTOP` are ignored because they cannot be caught.
33    /// - Signal handlers run asynchronously; keep them small and prefer atomics.
34    ///
35    /// # Examples
36    /// ```no_run
37    /// # #[cfg(feature = "std")] {
38    /// use devela::{AtomicBool, AtomicI32, AtomicOrdering::SeqCst, Thread, ThreadExt};
39    /// use devela::{Linux, LinuxSignal, LinuxSignalSet};
40    ///
41    /// static GOT_SIGNAL: AtomicBool = AtomicBool::new(false);
42    /// static LAST_SIGNAL: AtomicI32 = AtomicI32::new(0);
43    ///
44    /// fn handler(sig: i32) {
45    ///     LAST_SIGNAL.store(sig, SeqCst);
46    ///     GOT_SIGNAL.store(true, SeqCst);
47    /// }
48    ///
49    /// fn main() {
50    ///     Linux::sig_handler(handler, LinuxSignalSet::SIGINT | LinuxSignalSet::SIGWINCH, None);
51    ///
52    ///     println!("press Ctrl+C or resize the terminal");
53    ///     loop {
54    ///         if GOT_SIGNAL.swap(false, SeqCst) {
55    ///             let sig = LAST_SIGNAL.load(SeqCst);
56    ///             println!("received signal {sig}");
57    ///             if sig == LinuxSignal::SIGINT.as_c_int() {
58    ///                 break;
59    ///             }
60    ///         }
61    ///         Thread::sleep_ms(50);
62    ///     }
63    ///     println!("bye");
64    /// }
65    /// # }
66    /// ```
67    pub fn sig_handler(
68        handler: fn(c_int),
69        signals: LinuxSignalSet,
70        flags: impl Into<LinuxSigactionFlags>,
71    ) {
72        fn simple_adapter(sig: c_int, _info: &LinuxSiginfo, _context: *mut c_void) {
73            is! { LinuxSignal::from_c_int(sig).is_none(), return }
74            let handler = LINUX_SIG_HANDLERS[sig as usize].load(SeqCst);
75            if !handler.is_null() {
76                // SAFETY: we control both storage and retrieval and check for null.
77                let handler: fn(c_int) = unsafe { transmute(handler) };
78                handler(sig);
79            }
80        }
81        signals.for_each_catchable(|signal| {
82            LINUX_SIG_HANDLERS[signal.as_c_int() as usize].store(handler as *mut (), SeqCst);
83        });
84        Self::sig_handler_info(simple_adapter, signals, flags);
85    }

87    /// Registers multiple signals using a handler that receives signal information.
88    ///
89    /// The handler receives the signal number, a borrowed [`LinuxSiginfo`], and the
90    /// raw signal context pointer supplied by the kernel.
91    ///
92    /// # Arguments
93    /// - `handler`: Function called with the signal number, signal info, and context.
94    /// - `signals`: Signals to handle.
95    /// - `flags`: Additional signal action flags.
96    ///
97    /// # Notes
98    /// - `SA_RESTORER` and `SA_SIGINFO` are always included internally.
99    /// - `SIGKILL` and `SIGSTOP` are ignored because they cannot be caught.
100    /// - Signal handlers run asynchronously; keep them small and prefer atomics.
101    /// - The `LinuxSiginfo` borrow is valid only during the handler call.
102    ///
103    /// # Examples
104    /// ```no_run
105    /// # #[cfg(feature = "std")] {
106    /// use devela::{AtomicBool, AtomicI32, AtomicOrdering::SeqCst, Thread, ThreadExt, c_void};
107    /// use devela::{Linux, LinuxSigactionFlags, LinuxSiginfo, LinuxSignalSet};
108    ///
109    /// static GOT_SIGNAL: AtomicBool = AtomicBool::new(false);
110    /// static LAST_SIGNAL: AtomicI32 = AtomicI32::new(0);
111    /// static LAST_PID: AtomicI32 = AtomicI32::new(0);
112    ///
113    /// fn handler(sig: i32, info: &LinuxSiginfo, _context: *mut c_void) {
114    ///     LAST_SIGNAL.store(sig, SeqCst);
115    ///     LAST_PID.store(info.pid(), SeqCst);
116    ///     GOT_SIGNAL.store(true, SeqCst);
117    /// }
118    ///
119    /// fn main() {
120    ///     Linux::sig_handler_info(
121    ///         handler,
122    ///         LinuxSignalSet::SIGINT,
123    ///         LinuxSigactionFlags::SA_RESTART,
124    ///     );
125    ///
126    ///     println!("press Ctrl+C");
127    ///     loop {
128    ///         if GOT_SIGNAL.swap(false, SeqCst) {
129    ///             let sig = LAST_SIGNAL.load(SeqCst);
130    ///             let pid = LAST_PID.load(SeqCst);
131    ///             println!("received signal {sig} from pid {pid}");
132    ///             break;
133    ///         }
134    ///         Thread::sleep_ms(50);
135    ///     }
136    ///     println!("bye");
137    /// }
138    /// # }
139    /// ```
140    pub fn sig_handler_info(
141        handler: fn(i32, &LinuxSiginfo, *mut c_void),
142        signals: LinuxSignalSet,
143        flags: impl Into<LinuxSigactionFlags>,
144    ) {
145        extern "C" fn c_handler_siginfo(sig: c_int, info: *mut LinuxSiginfo, context: *mut c_void) {
146            is! { LinuxSignal::from_c_int(sig).is_none() || info.is_null(), return }
147            let handler = LINUX_SIGINFO_HANDLERS[sig as usize].load(SeqCst);
148            if !handler.is_null() {
149                // SAFETY: we control both storage and retrieval and check for null.
150                let handler: fn(c_int, &LinuxSiginfo, *mut c_void) = unsafe { transmute(handler) };
151
152                // SAFETY: kernel-provided for this SA_SIGINFO handler call.
153                let info = unsafe { &*info };
154                handler(sig, info, context);
155            }
156        }
157        // let signals = signals.catchable_only();
158        let flags = flags.into().with_sa_restorer().with_sa_siginfo();
159        let sigaction = LinuxSigaction::new_siginfo(
160            c_handler_siginfo,
161            flags,
162            LinuxSigset::empty(),
163            Some(__devela_linux_restore_rt),
164        );
165        signals.for_each_catchable(|signal| {
166            let raw = signal.as_c_int();
167            LINUX_SIGINFO_HANDLERS[raw as usize].store(handler as *mut (), SeqCst);
168            let _res = unsafe {
169                Linux::sys_rt_sigaction(raw, &sigaction, Ptr::null_mut(), LinuxSigset::size())
170            };
171        });
172    }
