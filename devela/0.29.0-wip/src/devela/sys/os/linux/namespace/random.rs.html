6/// Generates a couple of `random_*` functions for each given integer primitive
7macro_rules! _sys_os_linux_impl_random_fns {
8    ($($prim:ident : $len:literal),+) => { $crate::paste! { $(
9        #[doc = "Generates a random `" $prim "` using the default Linux random mode."]
10        pub fn [<random_ $prim>]() -> Result<$prim> {
11            Linux::[<random_ $prim _with>](Linux::RANDOM_MODE)
12        }
13        #[doc = "Generates a random `" $prim "` using the selected Linux random mode."]
14        pub fn [<random_ $prim _with>](mode: LinuxRandomMode) -> Result<$prim> {
15            let mut bytes = [0; $len];
16            Linux::random_bytes_with(&mut bytes, mode)?;
17            Ok($prim::from_ne_bytes(bytes))
18        }
19    )+ }};
20}

22/// # Randomness-related functionality.
23impl Linux {
24    /// Default Linux random mode.
25    ///
26    /// This is [`LinuxRandomMode::SecureNonblock`]: cryptographic randomness
27    /// requested without blocking. If the kernel randomness source is not ready,
28    /// random operations return an error instead of blocking.
29    pub const RANDOM_MODE: LinuxRandomMode = LinuxRandomMode::SecureNonblock;
30
31    /// Fills the whole buffer with random bytes using [`Linux::RANDOM_MODE`].
32    ///
33    /// This either fills the entire buffer or returns an error.
34    pub fn random_bytes(buffer: &mut [u8]) -> Result<()> {
35        Self::random_bytes_with(buffer, Self::RANDOM_MODE)
36    }
37
38    /// Fills the whole buffer with random bytes using `mode`.
39    ///
40    /// This either fills the entire buffer or returns an error.
41    pub fn random_bytes_with(buffer: &mut [u8], mode: LinuxRandomMode) -> Result<()> {
42        Self::random_bytes_flags(buffer, mode.flags())
43    }

45    _sys_os_linux_impl_random_fns![u8:1, u16:2, u32:4, u64:8, u128:16];

47impl Linux {
48    // from `sys/random.h`
49    pub(crate) const GRND_NONBLOCK: c_uint = 0x0001;
50    pub(crate) const GRND_INSECURE: c_uint = 0x0004;
51
52    /// Maximum number of retry attempts after `getrandom` returns `EAGAIN`.
53    pub(crate) const RAND_MAX_ATTEMPTS: usize = 15;
54
55    #[cold]
56    #[must_use]
57    fn getrandom_try_again_cold(attempts: &mut usize) -> bool {
58        *attempts += 1;
59        *attempts <= Linux::RAND_MAX_ATTEMPTS
60    }
61
62    fn random_bytes_flags(buffer: &mut [u8], flags: c_uint) -> Result<()> {
63        #[cold]
64        fn getrandom_failed_cold(n: isize) -> Result<()> {
65            Err(LinuxError::Sys(n))
66        }
67        #[cold]
68        fn getrandom_incomplete_cold() -> Result<()> {
69            Err(LinuxError::Sys(-LINUX_ERRNO::EAGAIN))
70        }
71
72        debug_assert![buffer.len() <= isize::MAX as usize];
73        let (mut attempts, mut offset) = (0, 0);
74
75        while offset < buffer.len() {
76            let n = unsafe {
77                Linux::sys_getrandom(buffer[offset..].as_mut_ptr(), buffer.len() - offset, flags)
78            };
79            if n == -LINUX_ERRNO::EAGAIN {
80                is! {
81                    !Linux::getrandom_try_again_cold(&mut attempts),
82                    return getrandom_incomplete_cold()
83                }
84            } else if n < 0 {
85                return getrandom_failed_cold(n);
86            } else if n == 0 {
87                return getrandom_incomplete_cold();
88            } else {
89                offset += n as usize;
90            }
91        }
92        Ok(())
93    }
94}
