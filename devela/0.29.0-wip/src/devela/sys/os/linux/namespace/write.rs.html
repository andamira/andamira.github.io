3use crate::{LINUX_FILENO as FILENO, Linux, LinuxError, LinuxResult as Result, MaybeUninit, c_int};
4
5/// # Write-related methods.
6#[rustfmt::skip]
7impl Linux {
8    /* core helpers */
9
10    const STACK_BUFFER_LEN: usize = 512; // PIPE_BUF is typically 4096 but 512 is safe
11
12    /// Writes all bytes to a file descriptor, handling partial writes.
13    fn write_all(fd: c_int, mut buf: &[u8]) -> Result<()> {
14        while !buf.is_empty() {
15            let n = unsafe { Linux::sys_write(fd, buf.as_ptr(), buf.len()) };
16            if n < 0 { return Err(LinuxError::Sys(n)); }
17            buf = &buf[n as usize..];
18        }
19        Ok(())
20    }
21    /// Writes all bytes to stdout. Returns error on syscall failure.
22    fn write_all_unchecked(fd: c_int, mut buf: &[u8]) {
23        while !buf.is_empty() {
24            let n = unsafe { Linux::sys_write(fd, buf.as_ptr(), buf.len()) };
25            if n <= 0 { panic!("write failed with return value {n}"); }
26            buf = &buf[n as usize..];
27        }
28    }
29
30    /* public methods */
31
32    /// Writes string + newline to stdout. Optimized for small strings.
33    pub fn print(s: &str) -> Result<()> {
34        Self::write_all(FILENO::STDOUT, s.as_bytes())
35    }
36    /// Like `print`, but panics on failure instead of returning errors.
37    pub fn print_unchecked(s: &str) { Self::write_all_unchecked(FILENO::STDOUT, s.as_bytes()); }
38
39    /// Writes string + newline to stdout. Optimized for small strings.
40    pub fn println(s: &str) -> Result<()> {
41        if s.len() <= Self::STACK_BUFFER_LEN { // small string optimization:
42            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
43            let bytes = s.as_bytes();
44            buf[..bytes.len()].copy_from_slice(bytes);
45            buf[bytes.len()] = b'\n';
46            Self::write_all(FILENO::STDOUT, &buf[..bytes.len() + 1])
47        } else { // fallback for large strings:
48            Self::write_all(FILENO::STDOUT, s.as_bytes())?;
49            Self::write_all(FILENO::STDOUT, b"\n")
50        }
51    }
52    /// Like `println`, but panics on failure instead of returning errors.
53    pub fn println_unchecked(s: &str) {
54        if s.len() <= Self::STACK_BUFFER_LEN {
55            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
56            let bytes = s.as_bytes();
57            buf[..bytes.len()].copy_from_slice(bytes);
58            buf[bytes.len()] = b'\n';
59            Self::write_all_unchecked(FILENO::STDOUT, &buf[..bytes.len() + 1]);
60        } else {
61            Self::write_all_unchecked(FILENO::STDOUT, s.as_bytes());
62            Self::write_all_unchecked(FILENO::STDOUT, b"\n");
63        }
64    }
65    /// Ultra-fast stdout write. Panics if not all bytes are written at once.
66    /// Only use for small strings (<512 bytes).
67    pub fn print_unchecked_fast(s: &str) {
68        let bytes = s.as_bytes();
69        let n = unsafe { Linux::sys_write(FILENO::STDOUT, bytes.as_ptr(), bytes.len()) };
70        if n != bytes.len() as isize {
71            panic!("write failed with return value {n}");
72        }
73    }
74    /// Ultra-fast stdout write with newline. Panics if write isn't atomic.
75    /// Only use for small strings (<511 bytes).
76    pub fn println_unchecked_fast(s: &str) {
77        let bytes = s.as_bytes();
78        let mut buf = MaybeUninit::<[u8; Self::STACK_BUFFER_LEN]>::uninit();
79        // SAFETY: we're about to initialize all accessed bytes
80        unsafe {
81            let buf_ptr = buf.as_mut_ptr() as *mut u8;
82            core::ptr::copy_nonoverlapping(bytes.as_ptr(), buf_ptr, bytes.len());
83            *buf_ptr.add(bytes.len()) = b'\n'; // Add newline
84            let n = Linux::sys_write(FILENO::STDOUT, buf_ptr, bytes.len() + 1); // single syscall
85            if n != (bytes.len() + 1) as isize { panic!("write failed with return value {n}"); }
86        }
87    }
88    /// Writes bytes to stdout. Returns error on syscall failure.
89    pub fn print_bytes(b: &[u8]) -> Result<()> { Self::write_all(FILENO::STDOUT, b) }
90    /// Like `print_bytes`, but panics on failure instead of returning errors.
91    pub fn print_bytes_unchecked(b: &[u8]) { Self::write_all_unchecked(FILENO::STDOUT, b); }
92
93    /// Writes all bytes to stderr. Returns error on syscall failure.
94    pub fn eprint(s: &str) -> Result<()> { Self::write_all(FILENO::STDERR, s.as_bytes()) }
95    /// Writes string + newline to stderr. Optimized for small strings.
96    pub fn eprintln(s: &str) -> Result<()> {
97        if s.len() <= Self::STACK_BUFFER_LEN { // small string optimization:
98            let mut buf = [0u8; Self::STACK_BUFFER_LEN];
99            let bytes = s.as_bytes();
100            buf[..bytes.len()].copy_from_slice(bytes);
101            buf[bytes.len()] = b'\n';
102            Self::write_all(FILENO::STDERR, &buf[..bytes.len() + 1])
103        } else { // fallback for large strings:
104            Self::write_all(FILENO::STDERR, s.as_bytes())?;
105            Self::write_all(FILENO::STDERR, b"\n")
106        }
107    }
108    /// Writes bytes to stderr. Returns error on syscall failure.
109    pub fn eprint_bytes(b: &[u8]) -> Result<()> {
110        Self::write_all(FILENO::STDERR, b)
111    }
