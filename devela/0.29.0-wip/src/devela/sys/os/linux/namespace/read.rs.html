11    /// Checks if there is input available to read from stdin.
12    pub fn has_input() -> bool { Self::available_bytes().unwrap_or(0) > 0 }

14    /// Returns the number of bytes available to be read from *stdin*.
15    pub fn available_bytes() -> LinuxResult<usize> {
16        let mut n = 0;
17        let result = unsafe {
18            Linux::sys_ioctl(FILENO::STDIN, IOCTL::FIONREAD, &mut n as *mut i32 as *mut u8)
19        };
20        if result < 0 { Err(LinuxError::Sys(result)) } else { Ok(n as usize) }
21    }
22
23    /// Gets a single byte from *stdin*.
24    pub fn get_byte() -> LinuxResult<u8> {
25        let mut c = 0;
26        loop {
27            let n = unsafe { Linux::sys_read(FILENO::STDIN, &mut c as *mut u8, 1) };
28            if n < 0 { return Err(LinuxError::Sys(n)); }
29            if n == 1 { break; }
30        }
31        Ok(c)
32    }

34    /// Attempts to get a single byte from stdin without blocking.
35    ///
36    /// Checks if there is input available using `has_input`,
37    /// and if so, reads a byte using `get_byte`.
38    ///
39    /// # Returns
40    /// - `Some(u8)` if a byte is successfully read.
41    /// - `None` if no data is available.
42    pub fn try_get_byte() -> Option<u8> {
43        if Linux::has_input() { Linux::get_byte().ok() } else { None }
44    }

46    /// Pauses execution until receiving from *stdin* any `char` in the `list`.
47    pub fn pause_until_char(list: &[char]) {
48        loop { if list.contains(&Linux::get_dirty_char()) { break; } }
49    }

51    /// Gets a single `char` from *stdin*.
52    ///
53    /// # Returns
54    /// - `Ok(char)` if a valid UTF-8 character is read.
55    /// - `Err(LinuxError::InvalidUtf8)` if the bytes are not valid UTF-8.
56    /// - `Err(LinuxError::NoInput)` if the UTF-8 sequence is valid but empty.
57    pub fn get_char() -> LinuxResult<char> {
58        let bytes = Linux::get_utf8_bytes()?;
59        let s = unsafe { Str::from_utf8_unchecked(&bytes) };
60        s.chars().next().ok_or(LinuxError::NoInput)
61    }

63    /// Gets either a single `char` from *stdin*, or the replacement character.
64    ///
65    /// If the bytes received doesn't form a valid unicode scalar then the
66    /// [replacement character (�)][char::REPLACEMENT_CHARACTER] will be returned.
67    pub fn get_dirty_char() -> char {
68        match Linux::get_utf8_bytes() {
69            Ok(bytes) => {
70                let s = unsafe { Str::from_utf8_unchecked(&bytes) };
71                s.chars().next().unwrap()
72            }
73            Err(_) => char::REPLACEMENT_CHARACTER,
74        }
75    }

77    /// Gets a UTF-8 encoded byte sequence from *stdin* representing a `char`.
78    pub fn get_utf8_bytes() -> LinuxResult<[u8; 4]> {
79        let mut bytes = [0u8; 4];
80        let len;
81        // read the first byte to determine the length of the character
82        bytes[0] = Linux::get_byte()?;
83        if bytes[0] & 0x80 == 0 { return Ok([bytes[0], 0, 0, 0]); } // ASCII char
84        // IMPROVE: use text module functionality
85        else if bytes[0] & 0xE0 == 0xC0 { len = 2; }
86        else if bytes[0] & 0xF0 == 0xE0 { len = 3; }
87        else if bytes[0] & 0xF8 == 0xF0 { len = 4; }
88        else { return Err(LinuxError::InvalidUtf8) }
89        // read the remaining bytes of the character
90        for i in 1..len {
91            bytes[i as usize] = Linux::get_byte()?;
92            if bytes[i as usize] & 0xC0 != 0x80 { return Err(LinuxError::InvalidUtf8); }
93        }
94        Ok(bytes)
95    }

97    /// Reads immediately available stdin bytes into `buf`.
98    ///
99    /// Returns `0` if no bytes are currently available.
100    pub fn read_available(buf: &mut [u8]) -> LinuxResult<usize> {
101        is! { buf.is_empty(), return Ok(0) }
102        let count = Linux::available_bytes()?.min(buf.len());
103        is! { count == 0, return Ok(0) }
104        let n = unsafe { Linux::sys_read(FILENO::STDIN, buf.as_mut_ptr(), count) };
105        is! { n < 0, Err(LinuxError::Sys(n)), Ok(n as usize) }
106    }

108    /// Reads all available bytes from stdin into an allocated buffer.
109    ///
110    /// # Returns
111    /// - `Ok(Vec<u8>)` containing the bytes read.
112    /// - `Err(isize)` if the read fails, returning the error code.
113    #[cfg(feature = "alloc")]
114    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
115    pub fn read_available_alloc() -> LinuxResult<Vec<u8>> {
116        let count = Linux::available_bytes()?;
117        if count == 0 { return Ok(Vec::new()); }
118        let mut buffer = crate::vec_![0u8; count];
119        let n = unsafe { Linux::sys_read(FILENO::STDIN, buffer.as_mut_ptr(), count) };
120        if n < 0 { Err(LinuxError::Sys(n)) } else { buffer.truncate(n as usize); Ok(buffer) }
121    }

123    /// Prompts the user for a string from *stdin*, backed by a `buffer`.
124    ///
125    /// # Examples
126    /// ```no_run
127    /// # use devela::Linux;
128    /// let mut name_buffer = [0_u8; 32];
129    /// let name: &str = Linux::prompt::<32>("Enter your name: ", &mut name_buffer).unwrap();

131    pub fn prompt<'i, const CAP: usize>(text: &str, buffer: &'i mut [u8; CAP])
132        -> LinuxResult<&'i str> {
133        Linux::print(text)?; Linux::get_line(buffer)
134    }

136    /// Gets a string from *stdin* backed by a `buffer`, until a newline.

148    /// Gets a string from *stdin* backed by a `buffer`, until the `stop` char is received.
149    ///
150    /// # Examples
151    /// ```no_run
152    /// # use devela::Linux;
153    /// let mut buf = [0_u8; 32];
154    /// let name: &str = Linux::get_str::<32>(&mut buf, '\n').unwrap();
155    /// ```
156    pub fn get_str<const CAP: usize>(buffer: &mut [u8; CAP], stop: char) -> LinuxResult<&str> {
157        let mut index = 0;
158        loop {
159            if let Ok(c) = Linux::get_char() {
160                let mut c_buf = [0; 4];
161                let c_str = c.encode_utf8(&mut c_buf);
162                if c == stop {
163                    break;
164                } else if index + c_str.len() <= CAP {
165                    // Linux::print(c_str)?;
166                    for &b in c_str.as_bytes() {
167                        buffer[index] = b;
168                        index += 1;
169                    }
170                }
171            }
172        }
173        Ok(unsafe { Str::from_utf8_unchecked(&buffer[..index]) })
174    }
