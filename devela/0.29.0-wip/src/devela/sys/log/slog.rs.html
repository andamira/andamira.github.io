15/// Internal helper to set the re-entrancy guard with a custom panic message.
16///
17/// Provides `enter` and `leave` arms for setting and clearing the guard.
18///
19/// # Panics
20/// Panics on `enter` if the logger is already in use,
21/// reporting the calling method's name in the message.
22macro_rules! _slog_guard {
23    ($self:ident enter $msg:literal) => {
24        is![$self.guard != 0, panic!($msg)];
25        $self.guard = 1;
26    };
27    ($self:ident leave) => {
28        $self.guard = 0;
29    };
30}

33/// Fixed-capacity static logger with owned byte buffers.
34#[doc = crate::_doc_meta!{location("sys/log")}]
35///
36/// Each logger is identified by its capacity (`CAP`) as the number of stored messages,
37/// and each message's maximum length (`MSG_LEN`).
38///
39/// Use the [`slog!`] macro to define, log, clear, or iterate its messages.
40///
41/// # Safety
42/// Employs a single-byte, non-atomic guard as a const-safe soft check against
43/// re-entrant or concurrent access.
44///
45/// While not providing full thread safety, it can detect most accidental overlaps
46/// in single-threaded or unsynchronized contexts, serving as a last-resort diagnostic
47/// aid under the assumption of disciplined single-threaded use.
48#[derive(Clone, Debug, PartialEq, Eq, Hash)]
49pub struct LoggerStatic<const CAP: usize, const MSG_LEN: usize> {
50    buf: [[u8; MSG_LEN]; CAP],
51    lens: [usize; CAP],
52    leftover: [usize; CAP], // 0 = no truncation
53    len: usize,
54    guard: i8, // 0 = free, 1 = in use
55}
56
57impl<const CAP: usize, const MSG_LEN: usize> Default for LoggerStatic<CAP, MSG_LEN> {
58    fn default() -> Self {
59        Self::new()
60    }
61}
62
63impl<const CAP: usize, const MSG_LEN: usize> LoggerStatic<CAP, MSG_LEN> {
64    /// Constructs a new static logger.
65    #[inline(always)]
66    pub const fn new() -> Self {
67        Self {
68            buf: [[0; MSG_LEN]; CAP],
69            lens: [0; CAP],
70            leftover: [0; CAP],
71            len: 0,
72            guard: 0,
73        }
74    }
75
76    /// Clears all the stored messages.
77    #[inline(always)]
78    pub const fn clear(&mut self) {
79        _slog_guard![self enter "LoggerStatic::clear()"];
80        self.len = 0;
81        _slog_guard![self leave];
82    }
83
84    /// Returns the number of logged messages.
85    #[inline(always)]
86    pub const fn count(&mut self) -> usize {
87        _slog_guard![self enter "LoggerStatic::len()"];
88        let len = self.len;
89        _slog_guard![self leave];
90        len
91    }
92
93    /// Whether the logger is full.
94    #[must_use]
95    #[inline(always)]
96    pub const fn is_full(&mut self) -> bool {
97        _slog_guard![self enter "LoggerStatic::is_full()"];
98        let full = self.len == CAP;
99        _slog_guard![self leave];
100        full
101    }
102
103    /// Logs bytes from `msg`, truncating if longer than `MSG_LEN`.
104    pub const fn log_bytes(&mut self, msg: &[u8]) {
105        _slog_guard![self enter "LoggerStatic::log_bytes()"];
106        if self.len < CAP {
107            let written = Slice::copy_utf8_into(&mut self.buf[self.len], 0, msg);
108            let remaining = msg.len().saturating_sub(written);
109            self.lens[self.len] = written;
110            self.leftover[self.len] = remaining;
111            self.len += 1;
112        }
113        _slog_guard![self leave];
114    }
115
116    /// Returns the message and truncation count at index `i`, or `None` if out of bounds.
117    ///
118    /// # Features
119    /// Uses the `unsafe_str` feature to skip duplicated UTF-8 validation.
120    #[must_use]
121    pub const fn get(&mut self, i: usize) -> Option<(&str, usize)> {
122        _slog_guard![self enter "LoggerStatic::get()"];
123        is! { i >= self.len, { _slog_guard![self leave]; return None; }}
124
125        let len = self.lens[i];
126        let leftover = self.leftover[i];
127        let msg = Slice::range_to(&self.buf[i], len);
128
129        cfg_select! { all(feature = "unsafe_str", not(feature = "safe_text")) => {
130            // SAFETY: we ensure to always contain valid UTF-8
131            let s = unsafe { Str::from_utf8_unchecked(msg) };
132        } _ => {
133            let s = crate::unwrap![ok Str::from_utf8(msg)];
134        }}
135        _slog_guard![self leave];
136        Some((s, leftover))
137    }
138
139    /// Runs the provided closure for each message in run-time.
140    ///
141    /// The closure receives:
142    /// - `i`: index of the message
143    /// - `s`: string slice (truncated if necessary)
144    /// - `truncated`: number of bytes that did not fit
145    ///
146    /// # Features
147    /// Uses the `unsafe_str` feature to skip duplicated validation checks.
148    pub fn for_each<F: FnMut(usize, &str, usize)>(&mut self, mut f: F) {
149        _slog_guard![self enter "LoggerStatic::log_bytes()"];
150        for (i, msg) in self.buf[..self.len].iter().enumerate() {
151            let len = self.lens[i];
152            let leftover = self.leftover[i];
153
154            cfg_select! { all(feature = "unsafe_str", not(feature = "safe_text")) => {
155                // SAFETY: we ensure to always contain valid UTF-8
156                let s = unsafe { Str::from_utf8_unchecked(&msg[..len]) };
157            } _ => {
158                let s = crate::unwrap![ok Str::from_utf8(&msg[..len])];
159            }}
160            f(i, s, leftover);
161        }
162        _slog_guard![self leave];
163    }
164
165    /// Returns `true` if any logged message was truncated.
166    #[must_use]
167    pub const fn any_truncated(&mut self) -> bool {
168        _slog_guard![self enter "LoggerStatic::any_truncated()"];
169        whilst! { i in 0..self.len; {
170            is![self.leftover[i] > 0, { _slog_guard![self leave]; return true; }];
171        }}
172        _slog_guard![self leave];
173        false
174    }
175
176    /// Returns `(count, total_lost_bytes)` for truncated messages.
177    #[must_use]
178    pub const fn truncation_stats(&mut self) -> (usize, usize) {
179        _slog_guard![self enter "LoggerStatic::truncation_stats()"];
180        lets![mut count = 0,  mut total = 0];
181        whilst! { i in 0..self.len; {
182            let lost = self.leftover[i];
183            is! { lost > 0, { count += 1; total += lost; }}
184        }}
185        _slog_guard![self leave];
186        (count, total)
187    }
188}

209#[doc = crate::_tags!(log)]
210/// Static global logger macro, compile-time friendly.
211#[doc = crate::_doc_meta!{location("sys/log")}]
212///
213/// Defines and operates on [`LoggerStatic`] instances for fixed-capacity logging.
214/// Each logger is identified by its `(CAP, MSG_LEN)` pair and may optionally have
215/// an extra identifier for multiple independent instances.
216///
217/// A `new` static logger can be given custom attributes and visibility.
218///
219/// See `LoggerStatic` for details on storage, safety, and const behavior.
220///
221/// # Overview
222/// - `slog!([vis] new [id:]CAP+LEN)` — define a static logger.
223/// - `slog!([id:]CAP+LEN "...")` — log a formatted message.
224/// - `slog!(clear [id:]CAP+LEN)` — clear all messages.
225/// - `slog!(get [id:]CAP+LEN, idx)` — get some message at idx.
226/// - `slog!(for_each [id:]CAP+LEN |idx, msg, trunc| { ... })` — iterate logged messages.
227/// - `slog!(any_truncated [id:]CAP+LEN)` — check if any message was truncated.
228/// - `slog!(truncation_stats [id:]CAP+LEN)` — return truncation count and total lost bytes.
229///
230/// # Examples
231/// ```
232/// # use devela::slog;
233/// slog!(new 4+32);
234/// slog!(4+32 "init ok");
235/// slog!(4+32 "processing step ", %2u8, ".");
236/// # #[cfg(feature = "__std")]
237/// slog!(for_each 4+32 |idx, msg, _trunc| println!("[{idx}] {msg}"));
238/// slog!(clear 4+32);
239///
240/// // with an extra identifier
241/// slog!(#[doc(hidden)] pub new log1:4+32); // with custom attributes & visibility
242/// slog!(log1:4+32 "hello world");
243/// assert_eq![slog!(get log1:4+32, 0), Some(("hello world", 0))];
244/// slog!(clear log1:4+32);
245/// ```
246#[macro_export]
247#[cfg_attr(cargo_primary_package, doc(hidden))]
248#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_sync")))]
249#[cfg(feature = "unsafe_sync")]
250macro_rules! slog {
251    (
252    /* public API*/
253
254    // Define a static logger.
255    $(#[$attrs:meta])*
256    $vis:vis new $($id:ident :)? $CAP:literal + $LEN:literal) => { $crate::paste! {
257        $crate::slog![@$(#[$attrs])* $vis new
258            $($id:)? $CAP+$LEN,
259            // NOTE: getting the ident using the macro doesn't seem compatible with paste!
260            // slog![@static_ident $($id:)?$CAP+$LEN],
261            // slog![@fn_ident $($id:)?$CAP+$LEN],
262            [<__LOGGER_ $($id _)?$CAP _$LEN>], // $static
263            [<__logger_ $($id _)?$CAP _$LEN>], // $fn
264        ];
265    }};
266    (
267    // Clear all logs.
268    clear $($id:ident :)? $CAP:literal + $LEN:literal) => {
269        $crate::slog!(@get $($id:)? $CAP + $LEN).clear();
270    };
271    (
272    // Return the number of messages.
273    count $($id:ident :)? $CAP:literal + $LEN:literal) => {
274        $crate::slog!(@get $($id:)? $CAP + $LEN).count();
275    };
276    (
277    // Whether the logger is full.
278    is_full $($id:ident :)? $CAP:literal + $LEN:literal) => {
279        $crate::slog!(@get $($id:)? $CAP + $LEN).is_full();
280    };
281    (
282    // Log message with formatted arguments.
283    $($id:ident :)? $CAP:literal + $LEN:literal $($fmt:tt)+) => {{
284        let mut buf = [0u8; $LEN];
285        let mut pos = 0;
286        $crate::fmtcat!(buf, pos, $($fmt)+);
287        let slice = $crate::Slice::range_to(&buf, pos);
288        $crate::slog!(@get $($id:)? $CAP + $LEN).log_bytes(slice);
289    }};
290    (
291    // Retrieve a specific message by index.
292    get $($id:ident :)? $CAP:literal + $LEN:literal, $index:expr) => {
293        $crate::slog!(@get $($id :)? $CAP + $LEN).get($index)
294    };
295    (
296    // Run a closure for each log message.
297    for_each $($id:ident :)? $CAP:literal + $LEN:literal $closure:expr) => {
298        $crate::slog!(@get $($id:)? $CAP + $LEN).for_each($closure);
299    };
300    (
301    // Whether if any message was truncated.
302    any_truncated $($id:ident :)? $CAP:literal + $LEN:literal) => {
303        $crate::slog!(@get $($id:)? $CAP + $LEN).any_truncated();
304    };
305    (
306    // Returns `(count, total_lost_bytes)` for truncated messages.
307    truncation_stats $($id:ident :)? $CAP:literal + $LEN:literal) => {
308        $crate::slog!(@get $($id:)? $CAP + $LEN).truncation_stats();
309    };
310    (
311    // Returns the name name of the global static as a static string slice.
312    static_name $($id:ident :)? $CAP:literal + $LEN:literal) => { $crate::paste! {
313        stringify!{ $crate::slog![@static_ident $($id :)? $CAP+$LEN] }
314    }};
315    // (
316    // // Returns name of the global fn as a static string slice.
317    // fn_name $($id:ident :)? $CAP:literal + $LEN:literal) => { $crate::paste! {
318    //     stringify!{ $crate::slog![@fn_ident $($id :)? $CAP+$LEN] }
319    // }};
320    (
321    /* private API */
322    @$(#[$attrs:meta])*
323    $vis:vis new $($id:ident :)? $CAP:literal + $LEN:literal,
324    $static:ident, $fn:ident $(,)?) => { $crate::paste! {
325        $(#[$attrs])*
326        #[doc = "\n\nA single-thread global static logger `" $($id ":" )? $CAP "+" $LEN "`."]
327        ///
328        /// Query it with the [`slog!`] macro, like this:
329        /// ```
330        /// # use devela::*;
331        /// # #[cfg(feature = "__std")]
332        #[doc = "slog![for_each " $($id ":" )? $CAP "+" $LEN " |i, s, _| println!(\"[{i}] {s}\")];"]
333        /// ```
334        // #[doc = "See also the fn `" $fn "()`."]
335        #[allow(non_upper_case_globals, reason = "case-sensitive $id")]
336        $vis static mut $static: $crate::LoggerStatic<$CAP, $LEN> = $crate::LoggerStatic::new();
337
338        #[doc(hidden)] // the unsafe fn accessor is always hidden
339        $(#[$attrs])*
340        #[doc = "Returns a mutable reference to the global static [`" $static "`]."]
341        ///
342        /// # Safety
343        /// The caller must ensure single-threaded discipline when mutating the returned reference.
344        #[inline(always)]
345        $vis const unsafe fn $fn() -> &'static mut $crate::LoggerStatic<$CAP, $LEN> {
346            #[allow(static_mut_refs, reason = "accessing the single-thread static logger instance")]
347            // SAFETY: user upholds single-threaded access to this static instance.
348            unsafe { &mut $static }
349        }
350    }};
351    // returns the global static reference
352    (@get $($id:ident :)? $CAP:literal + $LEN:literal) => {{
353        unsafe { $crate::slog![@fn_ident $($id :)? $CAP+$LEN]() }
354    }};
355    // returns the identifier of the global static
356    (@static_ident $($id:ident :)? $CAP:literal + $LEN:literal) => { $crate::paste! {
357        [<__LOGGER_ $($id _)? $CAP _ $LEN>]
358    }};
359    // returns the identifier of the global fn
360    (@fn_ident $($id:ident :)? $CAP:literal + $LEN:literal) => { $crate::paste! {
361        [<__logger_ $($id _)? $CAP _ $LEN>]
362    }};
363    // import the static
364    (@$(#[$attr:meta])*
365     $vis:vis use static: $($id:ident :)? $CAP:literal+$LEN:literal in $path:path) => {
366        // NOTE: we can't call a macro to get the ident at the end of the path.
367        $crate::paste! { $(#[$attr])* $vis use $path::[<__LOGGER_ $($id _)? $CAP _ $LEN>]; }
368    };
369    // import the fn
370    (@$(#[$attr:meta])*
371     $vis:vis use fn: $($id:ident :)? $CAP:literal+$LEN:literal in $path:path) => {
372        $crate::paste! { $(#[$attr])* $vis use $path::[<__logger_ $($id _)? $CAP _ $LEN>]; }
373    };
374}
