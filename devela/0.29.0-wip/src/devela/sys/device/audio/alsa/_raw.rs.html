154    pub(crate) fn snd_pcm_writen(pcm: *mut snd_pcm_t, bufs: *mut *mut c_void,

157    pub(crate) fn snd_pcm_readn(pcm: *mut snd_pcm_t, bufs: *mut *mut c_void,

160    pub(crate) fn snd_pcm_prepare(pcm: *mut snd_pcm_t) -> c_int;

165    /// Initialize hardware parameters with full configuration space.

169    pub(crate) fn snd_pcm_hw_params_set_access(pcm: *mut snd_pcm_t,
