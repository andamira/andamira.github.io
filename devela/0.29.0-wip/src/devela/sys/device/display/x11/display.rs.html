59    /// Opens the default X11 display and retrieves its first screen.
60    pub fn open() -> Result<Self, XError> {
61        // open connection
62        let mut screen_num = 0;
63        let conn = unsafe { _raw::xcb_connect(Ptr::null(), &mut screen_num as *mut c_int) };
64        is! { conn.is_null(), return Err(XError::ConnectionFailed) }
65        if unsafe { _raw::xcb_connection_has_error(conn) } != 0 {
66            unsafe { _raw::xcb_disconnect(conn) };
67            return Err(XError::ConnectionFailed);
68        }
69        // get setup
70        let setup = unsafe { _raw::xcb_get_setup(conn) };
71        if setup.is_null() {
72            unsafe { _raw::xcb_disconnect(conn) };
73            return Err(XError::SetupFailed);
74        }
75        let iter = unsafe { _raw::xcb_setup_roots_iterator(setup) };
76        if iter.data.is_null() {
77            unsafe { _raw::xcb_disconnect(conn) };
78            return Err(XError::NoScreensFound);
79        }
80        let screen: *const _raw::xcb_screen_t = iter.data;
81        let depth = unsafe { (*screen).root_depth };
82        let image_format = Self::query_image_format(setup, depth)?;
83        #[cfg(ffi_xcb_shm··)]
84        let shm_caps = Self::query_shm_caps(conn);
85
86        // extension setup hand-shake
87        lets![mut major = 0, mut minor = 0, mut ev = 0, mut err = 0];
88        let ok = unsafe {
89            _raw::xkb_x11_setup_xkb_extension(
90                conn, 1, 0, 0, &mut major, &mut minor, &mut ev, &mut err,
91            )
92        };
93        if ok <= 0 {
94            unsafe { _raw::xcb_disconnect(conn) };
95            return Err(XError::ExtensionUnavailable("xkb-setup"));
96        }
97        let xkb = XkbState::new(conn)?;
98        let atoms = XAtoms::new(conn);
99
100        // repeat
101        let pending = None;
102        let repeat_filter = KeyRepeatFilter::new();
103
104        let windows = vec![];
105        let queue = EventQueue::new();
106
107        Ok(Self { conn, screen, screen_num, depth, image_format,
108            #[cfg(ffi_xcb_shm··)]
109            shm_caps,
110            xkb, pending, repeat_filter, atoms, windows, queue, })
111    }

130    /// Polls the next event without blocking.
131    ///
132    /// If a raw event is available, it is processed through `handle_raw_event`
133    /// and returned as a high-level [`Event`].
134    /// If no event is pending, returns [`Event::None`].
135    ///
136    /// This is suitable for rendering loops or cooperative schedulers where
137    /// the program must remain responsive without waiting.
138    #[inline(always)]
139    pub fn poll_event(&mut self) -> Event {
140        is![let Some(ev) = self.queue.pop(), return ev]; // return pending events first
141        is![let Some(raw) = self.poll_raw_event(), return self.handle_raw_event(raw)];
142        Event::None
143    }
144
145    /// Waits for the next event, blocking until one is available.
146    ///
147    /// The raw event is passed to `handle_raw_event` and returned as a high-level [`Event`].
148    /// If the X connection enters an error state, returns [`Event::None`].
149    ///
150    /// This is the blocking counterpart of [`poll_event`][Self::poll_event], used when the
151    /// application should sleep until user input or a window notification arrives.
152    #[inline(always)]
153    pub fn wait_event(&mut self) -> Event {
154        is![let Some(ev) = self.queue.pop(), return ev]; // return pending events first
155        is![let Some(raw) = self.wait_raw_event(), self.handle_raw_event(raw), Event::None]
156    }
157
158    /// Flushes pending XCB commands.
159    #[inline(always)]
160    pub fn flush(&self) { unsafe { _raw::xcb_flush(self.conn); } }
161}

189        self.windows.iter().find(|&&(id, _)| id == window_id).map(|(_, state)| state)

196    /// Registers a window.
197    pub(crate) fn window_register(&mut self, window_id: u32, state: XWindowState) {
198        debug_assert!(!self.has_window(window_id), "window already registered: {window_id}");
199        self.windows.push((window_id, state));
200    }
201    /// Unregisters a window.
202    fn window_unregister(&mut self, window_id: u32) -> Option<XWindowState> {
203        let index = self.windows.iter().position(|&(id, _)| id == window_id)?;
204        Some(self.windows.swap_remove(index).1)
205    }
206    /// Destroys a registered window and removes its cached state.
207    ///
208    /// This sends a destroy request to the X server, frees the associated
209    /// graphics context, unregisters the window from this display, and flushes
210    /// the connection.
211    ///
212    /// Returns `true` if the window was registered and a destroy request was sent.
213    pub(crate) fn window_destroy(&mut self, window_id: u32, gc: u32) -> bool {
214        let Some(_) = self.window_unregister(window_id) else {
215            return false;
216        };
217        unsafe {
218            _raw::xcb_free_gc(self.conn, gc);
219            _raw::xcb_destroy_window(self.conn, window_id);
220        }
221        self.flush();
222        true
223    }

268    /// Updates the requested window size and/or position and returns its delta.
269    fn window_update_configure(
270        &mut self,
271        window_id: u32,
272        x: i16,
273        y: i16,
274        width: u16,
275        height: u16,
276    ) -> Option<XWindowConfigureDelta> {
277        let state = self.window_state_mut(window_id)?;
278        let moved = state.x != x || state.y != y;
279        let resized = state.width != width || state.height != height;
280        state.x = x;
281        state.y = y;
282        state.width = width;
283        state.height = height;
284        is! { resized, state.needs_redraw = true }
285        Some(XWindowConfigureDelta { moved, resized })
286    }

290    fn query_image_format(
291        setup: *const _raw::xcb_setup_t,
292        depth: u8,
293    ) -> Result<XImageFormat, XError> {
294        let mut it = unsafe { _raw::xcb_setup_pixmap_formats_iterator(setup) };
295        while it.rem > 0 && !it.data.is_null() {
296            let fmt = unsafe { &*it.data };
297            if fmt.depth == depth {
298                return Ok(XImageFormat::new(fmt.depth, fmt.bits_per_pixel, fmt.scanline_pad));
299            }
300            unsafe { _raw::xcb_format_next(&mut it) };
301        }
302        Err(XError::Other("no pixmap format for root depth"))
303    }
304    #[cfg(ffi_xcb_shm··)]
305    fn query_shm_caps(conn: *mut _raw::xcb_connection_t) -> Option<XShmCaps> {
306        unsafe {
307            let cookie = _raw::xcb_shm_query_version(conn);
308            let mut err = core::ptr::null_mut();
309            let reply = _raw::xcb_shm_query_version_reply(conn, cookie, &mut err);
310            if !err.is_null() || reply.is_null() {
311                return None;
312            }
313            let caps = XShmCaps {
314                major_version: (*reply).major_version,
315                minor_version: (*reply).minor_version,
316                shared_pixmaps: (*reply).shared_pixmaps != 0,
317                pixmap_format: (*reply).pixmap_format,
318            };
319            Libc::free(reply.cast());
320            Some(caps)
321        }
322    }

326    /// Interprets a raw X11 event and converts it into a high-level [`Event`].
327    ///
328    /// This handles:
329    /// - key translation (press/release, repeat filtering, semantic mapping),
330    /// - basic pointer and window-related notifications,
331    /// - special cases such as synthetic key releases.
332    ///
333    /// It never performs I/O. The caller supplies the raw event pointer; the method returns
334    /// the corresponding high-level `Event` or `Event::None` if the event is ignored.
335    #[rustfmt::skip]
336    fn handle_raw_event(&mut self, xev: XEvent) -> Event {
337        use {EventKind as Kind, EventWindow as Win};
338        if let Some(ev) = xev.as_raw_key() {
339            let keycode = unsafe { ev.detail };
340            let time_ms = unsafe { ev.time };
341            if xev.is_key_release() {
342                let real = self.classify_release(keycode, time_ms);
343                is! { !real, return Event::default() } // skip fake release
344            }
345            let key = xev.to_event_key(&self.xkb, &mut self.repeat_filter);
346            return Event::from_window(ev.event, Kind::Key(key), xev.timestamp());
347        } else if let Some(ev) = xev.as_raw_button() {
348            let x = ev.event_x.into();
349            let y = ev.event_y.into();
350            let timestamp = xev.timestamp();
351            let buttons = XEvent::map_button_mask(ev.state);
352            let mods = self.xkb.key_mods(ev.state);
353            let unit = EventWheelUnit::Step;
354            // X11 raw button 4..7 become EventWheel { unit: Step, ... }
355            match ev.detail {
356                4 => { return Event::from_window(ev.event, Kind::Wheel(
357                        EventWheel::new(0, -1, unit, x, y, buttons, mods)), timestamp); }
358                5 => { return Event::from_window(ev.event, Kind::Wheel(
359                        EventWheel::new(0, 1, unit, x, y, buttons, mods)), timestamp); }
360                6 => { return Event::from_window(ev.event, Kind::Wheel(
361                        EventWheel::new(-1, 0, unit, x, y, buttons, mods)), timestamp); }
362                7 => { return Event::from_window(ev.event, Kind::Wheel(
363                        EventWheel::new(1, 0, unit, x, y, buttons, mods)), timestamp); }
364                _ => {
365                    let button = XEvent::map_button(ev.detail);
366                    let state = xev.map_button_state();
367                    return Event::from_window(ev.event, Kind::Mouse(
368                        EventMouse { x, y, button: Some(button), state, buttons, mods }),
369                        timestamp,
370                    );
371                }
372            }
373        } else if let Some(ev) = xev.as_raw_motion() {
374            let buttons = XEvent::map_button_mask(ev.state);
375            let button = EventButton::from_one_bit_mask(buttons);
376            let mods = self.xkb.key_mods(ev.state);
377            return Event::from_window(
378                ev.event,
379                Kind::Mouse(EventMouse {
380                    x: ev.event_x.into(),
381                    y: ev.event_y.into(),
382                    button,
383                    state: EventButtonState::Moved,
384                    buttons,
385                    mods,
386                }),
387                xev.timestamp(),
388            );
389        } else if let Some(cm) = xev.as_raw_client_message() {
390            let proto = unsafe { cm.data.data32[0] };
391            if proto == self.atoms.wm_delete_window {
392                return Event::from_window(
393                    cm.window,
394                    Kind::Window(Win::CloseRequested),
395                    xev.timestamp(),
396                );
397            }
398        } else if let Some(ev) = xev.as_raw_enter() {
399            return Event::from_window(ev.event, Kind::Window(Win::Entered), xev.timestamp());
400        } else if let Some(ev) = xev.as_raw_leave() {
401            return Event::from_window(ev.event, Kind::Window(Win::Left), xev.timestamp());
402        } else if let Some(ev) = xev.as_raw_expose() {
403            self.window_mark_redraw(ev.window);
404            return Event::from_window(
405                ev.window,
406                Kind::Window(Win::RedrawRequested),
407                xev.timestamp(),
408            );
409        } else if let Some(ev) = xev.as_raw_configure() {
410            let window_id = ev.window;
411            let (x, y, w, h) = (ev.x, ev.y, ev.width, ev.height);
412            let ts = xev.timestamp();
413            if let Some(delta) = self.window_update_configure(window_id, x, y, w, h) {
414                if delta.moved {
415                    let position = Position::<i32, 2>::new([x as i32, y as i32]);
416                    self.queue.push(Event::from_window(
417                        window_id,
418                        EventKind::Window(EventWindow::Moved(Some(position))),
419                        ts,
420                    ));
421                }
422                if delta.resized {
423                    let extent = Extent::<u32, 2>::new([w as u32, h as u32]);
424                    self.queue.push(Event::from_window(
425                        window_id,
426                        EventKind::Window(EventWindow::Resized(Some(extent))),
427                        ts,
428                    ));
429                }
430            }
431            return Event::None;
432        } else {
433            // eprintln!("(OTHER): {} {:?}", xev.response_type(), xev.timestamp());
434        }
435        Event::None
436    }

438    /// Returns the next raw X11 event pointer, if available.
439    ///
440    /// Consumes a previously buffered event if one exists; otherwise polls
441    /// the connection with `xcb_poll_for_event`. Returns `None` if no event
442    /// is available at this time.
443    fn next_raw_event(&mut self) -> Option<*mut _raw::xcb_generic_event_t> {
444        is! { let Some(ev) = self.pending.take(), return Some(ev) }
445        let ev = unsafe { _raw::xcb_poll_for_event(self.conn) };
446        if ev.is_null() { None } else { Some(ev) }
447    }
448
449    /// Peeks at the next raw X11 event without consuming it.
450    ///
451    /// If no event is currently buffered, attempts a non-blocking poll and,
452    /// if successful, stores the event in the internal `pending` slot.
453    /// Returns the buffered pointer, or `None` if no event is available.
454    fn peek_raw_event(&mut self) -> Option<*mut _raw::xcb_generic_event_t> {
455        if self.pending.is_none() {
456            let ev = unsafe { _raw::xcb_poll_for_event(self.conn) };
457            is! { !ev.is_null(), self.pending = Some(ev) }
458        }
459        self.pending
460    }
461
462    /// Retrieves the next X11 event, if any.
463    ///
464    /// Returns an `XEvent` wrapper around a raw XCB event pointer.
465    /// Uses any internally buffered event before polling the X server.
466    fn poll_raw_event(&mut self) -> Option<XEvent> {
467        let ev = self.next_raw_event()?;
468        Some(XEvent { raw: ev })
469    }
470
471    /// Blocks until the next raw X11 event is available.
472    ///
473    /// Returns a pointer to the raw `xcb_generic_event_t`.
474    /// If a previously buffered event exists, it is returned first.
475    /// Otherwise the call blocks until the X server sends a new event.
476    ///
477    /// Only returns `None` if the connection is in an error state.
478    fn wait_raw_event(&mut self) -> Option<XEvent> {
479        is! { let Some(ev) = self.pending.take(), return Some(XEvent { raw: ev }) }
480        let ev = unsafe { _raw::xcb_wait_for_event(self.conn) };
481        if ev.is_null() { None } else { Some(XEvent { raw: ev }) }
482    }

484    /// Determines whether a key release is genuine or part of an auto-repeat sequence.
485    ///
486    /// X11 synthesizes a "fake" release immediately followed by a press when a key auto-repeats.
487    /// This method looks ahead at the next pending event and classifies the release as:
488    /// - `false`: synthetic release (auto-repeat).
489    /// - `true`:  real physical release.
490    ///
491    /// The classification is based on matching keycode and timestamp.
492    fn classify_release(&mut self, keycode: u8, timestamp: u32) -> bool {
493        if let Some(next) = self.peek_raw_event() {
494            unsafe {
495                let ty = (*next).response_type & 0x7F;
496                if ty == _raw::xcb_event_code::XCB_KEY_PRESS as u8 {
497                    let kpev = next as *const _raw::xcb_key_press_event_t;
498                    is! { (*kpev).detail == keycode && (*kpev).time == timestamp, return false }
499                }
500            }
501        }
502        true
503    }
