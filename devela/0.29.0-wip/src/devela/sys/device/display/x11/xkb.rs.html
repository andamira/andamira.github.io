13/// Tracks the minimal state needed to classify `Press` vs `Repeat`.
14#[doc = crate::_doc_meta!{location("sys/device/display/x11")}]

23}

27    pub const fn new() -> Self { Self { last_keycode: 0, repeating: false } }

29    /// Classifies an incoming `(keycode, is_press)` pair as `Press`, `Repeat`, or `Release`.
30    ///
31    /// Assumes synthetic X11 `(Release, Press)` autorepeat pairs have
32    /// already been removed by [`classify_release`][super::XDisplay::classify_release].
33    ///
34    /// Rules:
35    /// - A press of the same key while `repeating == true` → `Repeat`
36    /// - Any new key press → `Press` and begin repeating for that key
37    /// - Any release → `Release` and stop repeating
38    #[inline(always)]
39    pub fn filter(&mut self, keycode: u8, is_press: bool) -> KeyState {
40        if is_press {
41            // is this a repeated press of the same key?
42            is![self.repeating && self.last_keycode == keycode, return KeyState::Repeat];
43            // new press: start repeating for this key
44            self.last_keycode = keycode;
45            self.repeating = true;
46            KeyState::Press
47        } else {
48            // a real release resets the repeating state
49            self.repeating = false;
50            KeyState::Release
51        }
52    }
53}

158    /// Creates an XKB interpreter for the given X11 connection.
159    ///
160    /// Loads the keymap and constructs an internal XKB state for:
161    /// - keysym translation,
162    /// - physical scancode lookup,
163    /// - modifier interpretation.
164    ///
165    /// Returns an [`XError`] if the XKB extension cannot be used.
166    pub fn new(conn: *mut _raw::xcb_connection_t) -> Result<Self, XError> {
167        // context
168        let ctx = unsafe { _raw::xkb_context_new(0) };
169        is![ctx.is_null(), return Err(XError::ExtensionUnavailable("xkb-context"))];
170
171        // find core keyboard device
172        let device_id = unsafe { _raw::xkb_x11_get_core_keyboard_device_id(conn) };
173        is![device_id < 0, return Err(XError::ExtensionUnavailable("xkb-core-keyboard"))];
174
175        // load keymap from device
176        let keymap = unsafe { _raw::xkb_x11_keymap_new_from_device(ctx, conn, device_id, 0) };
177        is![keymap.is_null(), return Err(XError::ExtensionUnavailable("xkb-keymap"))];
178
179        // create XKB state
180        let state = unsafe { _raw::xkb_x11_state_new_from_device(keymap, conn, device_id) };
181        if state.is_null() {
182            unsafe { _raw::xkb_keymap_unref(keymap) };
183            return Err(XError::ExtensionUnavailable("xkb-state"));
184        }
185
186        // WAIT: until libxkbcommon ≥ 1.12 becomes widely deployed.
187        // cache indices of key modifiers and leds
188        // let mod_idx = XkbModIndices::new(keymap);
189        // let led_idx = XkbLedIndices::new(keymap);
190        // Ok(Self { ctx, keymap, state, mod_idx, led_idx })
191
192        Ok(Self { ctx, keymap, state })
193    }

261    /// Converts an X11 core modifier bitmask into [`KeyMods`].
262    ///
263    /// This is a compatibility path. It interprets the conventional X11
264    /// modifier slots directly (`Mod1` ≈ Alt, `Mod4` ≈ Super, `Mod5` ≈ AltGr).
265    ///
266    /// Prefer the XKB effective-modifier path
267    /// once the minimum supported libxkbcommon version is high enough.
268    pub fn key_mods(&self, xcb_modifiers: u16) -> KeyMods {
269        let (x, mut m) = (Bitwise(xcb_modifiers), KeyMods::new());
270        is![x.is_set_mask(_raw::XCB_MOD_MASK_SHIFT), m.set_shift()];
271        is![x.is_set_mask(_raw::XCB_MOD_MASK_CONTROL), m.set_control()];
272        is![x.is_set_mask(_raw::XCB_MOD_MASK_LOCK), m.set_caps_lock()];
273        // X11 conventional modifier slots:
274        is![x.is_set_mask(_raw::XCB_MOD_MASK_1), m.set_alt()];
275        is![x.is_set_mask(_raw::XCB_MOD_MASK_2), m.set_num_lock()];
276        is![x.is_set_mask(_raw::XCB_MOD_MASK_3), m.set_scroll_lock()];
277        is![x.is_set_mask(_raw::XCB_MOD_MASK_4), m.set_super()];
278        is![x.is_set_mask(_raw::XCB_MOD_MASK_5), m.set_alt_gr()];
279        m
280    }

282    /// Returns the logical (layout-dependent) key for the given X11 keycode.
283    ///
284    /// This is the "semantic" key: the character or meaning according to the
285    /// active keyboard layout (e.g., `'q'`, `'Q'`, `'é'`, `Left`, `F5`, etc.).
286    #[inline(always)]
287    pub fn key_semantic(&self, keycode: u8) -> Key {
288        // try to get special keys via keysym (arrows, F1, Shift, etc.)
289        let sym = unsafe { _raw::xkb_state_key_get_one_sym(self.state, keycode as u32) };
290        if let Some(key) = Self::map_special_keys(sym) { return key; }
291
292        // try to get Unicode (ñ, ç, á, € …)
293        let utf32 = unsafe { _raw::xkb_state_key_get_utf32(self.state, keycode as u32) };
294        if utf32 != 0 {
295            cfg_select! { all(feature = "unsafe_str", not(feature = "safe_sys")) => {
296                // SAFETY: we trust the fn _raw::xkb_state_key_get_utf32
297                unsafe { Key::Char(char::from_u32_unchecked(utf32)) }
298            } _ => {
299                Key::Char(char::from_u32(utf32).expect("valid unicode scalar"))
300            }}
301        } else {
302            // fallback (dead keys etc.)
303            Key::Unknown
304        }
305    }

307    /// Returns the physical key for the given X11 keycode.
308    ///
309    /// This is the "layout-independent" key: the actual physical scancode
310    /// (e.g., "key in row 2, column 1"), stable across layouts.
311    ///
312    /// Useful for games and universal keybindings.
313    #[inline(always)]
314    pub fn key_physical(&self, keycode: u8) -> Key {
315        // X11 core keycodes on Linux are usually evdev + 8.
316        // We subtract 8 to get a layout-independent "physical" scancode.
317        let scancode = keycode.saturating_sub(8) as u32;
318        Self::map_scancode_to_key(scancode)
319    }

324    /// Maps special keys before converting to UTF32 in [`XkbState::key_semantic`].
325    #[inline(always)]
326    const fn map_special_keys(sym: u32) -> Option<Key> {
327        let k = match sym {
328            /* control keys */
329            _raw::XKB_KEY_Return       => Key::Enter,
330            _raw::XKB_KEY_Tab          => Key::Tab,
331            _raw::XKB_KEY_BackSpace    => Key::Backspace,
332            _raw::XKB_KEY_Escape       => Key::Escape,
333            _raw::XKB_KEY_Delete       => Key::Delete,
334            _raw::XKB_KEY_KP_Delete    => Key::Delete,
335            _raw::XKB_KEY_Insert       => Key::Insert,
336            _raw::XKB_KEY_KP_Insert    => Key::Insert,
337
338            /* navigation */
339            _raw::XKB_KEY_Left         => Key::Left,
340            _raw::XKB_KEY_Right        => Key::Right,
341            _raw::XKB_KEY_Up           => Key::Up,
342            _raw::XKB_KEY_Down         => Key::Down,
343            _raw::XKB_KEY_Home         => Key::Home,
344            _raw::XKB_KEY_End          => Key::End,
345            _raw::XKB_KEY_Page_Up      => Key::PageUp,
346            _raw::XKB_KEY_Page_Down    => Key::PageDown,
347
348            /* lock keys */
349            _raw::XKB_KEY_Num_Lock     => Key::NumLock,
350            _raw::XKB_KEY_Caps_Lock    => Key::CapsLock,
351            _raw::XKB_KEY_Scroll_Lock  => Key::ScrollLock,
352
353            /* dead keys */
354            0xfe50 ..= 0xfe93 => Key::Dead(KeyDead::from_keysym(sym)),
355
356            /* modifiers */
357            _raw::XKB_KEY_Shift_L      => Key::Mod(KeyMod::LeftShift),
358            _raw::XKB_KEY_Shift_R      => Key::Mod(KeyMod::RightShift),
359            _raw::XKB_KEY_Control_L    => Key::Mod(KeyMod::LeftControl),
360            _raw::XKB_KEY_Control_R    => Key::Mod(KeyMod::RightControl),
361            _raw::XKB_KEY_Alt_L        => Key::Mod(KeyMod::LeftAlt),
362            _raw::XKB_KEY_Alt_R        => Key::Mod(KeyMod::RightAlt),
363            _raw::XKB_KEY_Super_L      => Key::Mod(KeyMod::LeftSuper),
364            _raw::XKB_KEY_Super_R      => Key::Mod(KeyMod::RightSuper),
365            0xfe03..=0xfe05           => Key::Mod(KeyMod::IsoLevel3Shift),
366            _raw::XKB_KEY_Mode_switch  => Key::Mod(KeyMod::AltGr),
367            0xfe11..=0xfe13           => Key::Mod(KeyMod::IsoLevel5Shift),
368
369            /* function keys */
370            sym if (sym >= _raw::XKB_KEY_F1 && sym <= _raw::XKB_KEY_F35) => {
371                let n = (sym - _raw::XKB_KEY_F1 + 1) as u8;
372                Key::Fn(n)
373            }
374
375            /* keypad numeric keys */
376            _raw::XKB_KEY_KP_0         => Key::Pad(KeyPad::Num0),
377            _raw::XKB_KEY_KP_1         => Key::Pad(KeyPad::Num1),
378            _raw::XKB_KEY_KP_2         => Key::Pad(KeyPad::Num2),
379            _raw::XKB_KEY_KP_3         => Key::Pad(KeyPad::Num3),
380            _raw::XKB_KEY_KP_4         => Key::Pad(KeyPad::Num4),
381            _raw::XKB_KEY_KP_5         => Key::Pad(KeyPad::Num5),
382            _raw::XKB_KEY_KP_6         => Key::Pad(KeyPad::Num6),
383            _raw::XKB_KEY_KP_7         => Key::Pad(KeyPad::Num7),
384            _raw::XKB_KEY_KP_8         => Key::Pad(KeyPad::Num8),
385            _raw::XKB_KEY_KP_9         => Key::Pad(KeyPad::Num9),
386            _raw::XKB_KEY_KP_Add       => Key::Pad(KeyPad::Add),
387            _raw::XKB_KEY_KP_Subtract  => Key::Pad(KeyPad::Subtract),
388            _raw::XKB_KEY_KP_Multiply  => Key::Pad(KeyPad::Multiply),
389            _raw::XKB_KEY_KP_Divide    => Key::Pad(KeyPad::Divide),
390            _raw::XKB_KEY_KP_Enter     => Key::Pad(KeyPad::Enter),
391            _raw::XKB_KEY_KP_Equal     => Key::Pad(KeyPad::Equal),
392            _raw::XKB_KEY_KP_Separator => Key::Pad(KeyPad::Comma),
393            _raw::XKB_KEY_KP_Decimal   => Key::Pad(KeyPad::Decimal),
394
395            // no match => not a special key
396            _ => return None,
397        };
398        Some(k)
399    }
