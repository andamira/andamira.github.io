12/// Borrowed X11 backend context for one runtime frame.

15/// Provides temporary access to the display connection and target window

18    pub(crate) display: &'a mut XDisplay,
19    pub(crate) window: &'a mut XWindow,
20}

22#[doc = crate::_tags!(unix runtime)]
23/// A single-window X11 runtime backend.
24#[doc = crate::_doc_meta!{location("sys/device/display/x11")}]
25///
26/// `XBackend` owns the X11 display connection and window used for event
27/// collection and per-frame backend access.
28#[derive(Debug)]
29pub(crate) struct XBackend {
30    display: XDisplay,
31    window: XWindow,
32}
33impl XBackend {
34    /// Opens a single-window X11 backend.
35    pub fn open(x: i16, y: i16, width: u16, height: u16) -> Result<Self, XError> {
36        let mut display = XDisplay::open()?;
37        let window = XWindow::new(&mut display, x, y, width, height, 1)?;
38        Ok(Self { display, window })
39    }
40}
41impl RunBackend for XBackend {
42    type Event = Event;
43    type Error = XError;
44    type Context<'a>
45        = XFrameCtx<'a>
46    where
47        Self: 'a;
48    fn collect_events(&mut self, out: &mut [Self::Event]) -> Result<usize, Self::Error> {
49        whilst! { written in 0..out.len(); {
50            let ev = self.display.poll_event();
51            is! { ev.is_none(), break }
52            out[written] = ev;
53        }}
54        Ok(written)
55    }
56    fn context(&mut self) -> Self::Context<'_> {
57        XFrameCtx {
58            display: &mut self.display,
59            window: &mut self.window,
60        }
61    }
62}

64#[doc = crate::_tags!(unix runtime)]
65/// An X11 frontend for frame-driven runtimes.
66#[doc = crate::_doc_meta!{location("sys/device/display/x11")}]
67///
68/// Owns the current X11 runtime backend and its presentation finalizer,
69/// while leaving runtime control, app logic, and rendering policy external.
70///
71/// The current implementation manages one window.
72/// That is an implementation detail, not the intended long-term boundary.
73#[derive(Debug)]
74pub struct XFrontend {
75    backend: XBackend,
76    presenter: XPresenter,
77}
78#[rustfmt::skip]
79impl XFrontend {
80    const fn _new(backend: XBackend, presenter: XPresenter) -> Self { Self { backend, presenter } }
81
82    /// Opens an X11 frontend.
83    pub fn open(x: i16, y: i16, width: u16, height: u16) -> Result<Self, XError> {
84        Self::open_with(XImageMode::Auto, x, y, width, height)
85    }
86    /// Opens an X11 frontend with the given image `mode`.
87    pub fn open_with(mode: XImageMode, x: i16, y: i16, width: u16, height: u16)
88        -> Result<Self, XError> {
89        Ok(Self::_new(XBackend::open(x, y, width, height)?, XPresenter::new(mode)))
90    }
91
92    /// Returns a shared reference to the X11 Display server.
93    pub const fn display(&self) -> &XDisplay { &self.backend.display }
94    /// Returns an exclusive reference to the X11 Display server.
95    pub const fn display_mut(&mut self) -> &mut XDisplay { &mut self.backend.display }
96
97    /// Returns whether MIT-SHM is available on this display.
98    #[cfg(ffi_xcb_shm··)]
99    pub const fn has_shm(&self) -> bool { self.backend.display.has_shm() }
100
101    /// Returns the configured X11 image presentation mode.
102    ///
103    /// This is the mode requested when opening the frontend.
104    /// It may still be [`XImageMode::Auto`] before any presentation occurs.
105    pub const fn mode(&self) -> XImageMode { self.presenter.mode() }
106
107    /// Returns the currently resolved X11 image presentation mode, if any.
108    ///
109    /// This resolves the active backing chosen after presentation begins.
110    ///
111    /// It returns `None` while no presentation surface has been created yet.
112    pub fn active_mode(&self) -> Option<XImageMode> { self.presenter.active_mode() }
113
114    /// Polls the next event without blocking.
115    ///
116    /// See [`XDisplay::poll_event`] for more information.
117    pub fn poll_event(&mut self) -> Event { self.backend.display.poll_event() }
118
119    /// Waits for the next event, blocking until one is available.
120    ///
121    /// See [`XDisplay::wait_event`] for more information.
122    pub fn wait_event(&mut self) -> Event { self.backend.display.wait_event() }
123}
124impl XFrontend {
125    /// Drives one X11 runtime iteration including rendering and presentation.
126    #[allow(private_bounds, reason = "private XFrameCtx")]
127    pub fn step_frame<T, A, R, S, RE>(
128        &mut self,
129        driver: &mut RunDriver<T>,
130        app: &mut A,
131        renderer: &mut R,
132        scene: &S,
133        events: &mut [A::Event],
134    ) -> Result<RunControl, RunDriverFrameError<XError, A::Error, RE, XError>>
135    where
136        A: RunApp<Event = Event>,
137        for<'a> R: RunRender<S, Event, XFrameCtx<'a>, Error = RE, Output<'a> = XPresent<'a>>,
138    {
139        // NOTE: because of compiler limitations we can't do just this:
140        // driver.step_frame(&mut self.backend, app, renderer, &mut self.presenter, scene, events)
141        _run_driver_step_run_frame_body!(@check
142            driver, self.backend, app, renderer, self.presenter, scene(scene), events, control,
143            driver.runtime_mut().tick_once(), driver.runtime_mut().transition(RunPhase::Stopped));
144        Ok(control)
145    }
146
147    /// Runs until the app stops or an error occurs,
148    /// including rendering and presentation.
149    #[allow(private_bounds, reason = "private XFrameCtx")]
150    pub fn run_frame<T, A, R, S, RE>(
151        &mut self,
152        driver: &mut RunDriver<T>,
153        app: &mut A,
154        renderer: &mut R,
155        scene: &S,
156        events: &mut [A::Event],
157    ) -> Result<(), RunDriverFrameError<XError, A::Error, RE, XError>>
158    where
159        A: RunApp<Event = Event>,
160        for<'a> R: RunRender<S, Event, XFrameCtx<'a>, Error = RE, Output<'a> = XPresent<'a>>,
161    {
162        // NOTE: because of compiler limitations we can't do just this:
163        // driver.run_frame(&mut self.backend, app, renderer, &mut self.presenter, scene, events)
164        driver.start();
165        while driver.can_advance() {
166            _run_driver_step_run_frame_body!(
167                driver,
168                self.backend,
169                app,
170                renderer,
171                self.presenter,
172                scene(scene),
173                events,
174                control,
175                driver.runtime_mut().tick_once(),
176                driver.runtime_mut().transition(RunPhase::Stopped),
177                break
178            );
179        }
180        Ok(())
181    }
182}
183impl XFrontend {
184    /// Renders directly into the retained X11 surface and presents it.
185    ///
186    /// This bypasses [`XRasterRenderer`][crate::XRasterRenderer]
187    /// and avoids the scene-to-surface copy.
188    pub fn with_surface_frame<F, T>(
189        &mut self,
190        width: u16,
191        height: u16,
192        depth: u8,
193        clear_redraw: bool,
194        render: F,
195    ) -> Result<T, XError>
196    where
197        F: FnOnce(&mut XSurfaceFrame<'_>) -> Result<T, XError>,
198    {
199        let result = {
200            let mut frame =
201                self.presenter.surface_frame(&self.backend.display, width, height, depth)?;
202            render(&mut frame)?
203        };
204        self.presenter.present_surface(
205            &mut self.backend.display,
206            &mut self.backend.window,
207            clear_redraw,
208        )?;
209        Ok(result)
210    }
211
212    /// Advances one runtime frame and renders directly into the retained X11 surface.
213    ///
214    /// This is the X11-specific zero-copy rendering path. It bypasses
215    /// [`XRasterRenderer`][crate::XRasterRenderer] and exposes the
216    /// retained surface bytes for the duration of one frame.
217    ///
218    /// This avoids the intermediate artifact-to-surface copy,
219    /// but it also gives up backend independence.
220    //
221    // BENCH NOTE: On local X11 SHM tests at 2560x1440-like scale, direct surface
222    // rendering was ~5-20% faster than the artifact path depending on workload.
223    // Presentation remained the dominant cost. Keeping both paths.
224    #[allow(clippy::too_many_arguments)]
225    #[allow(private_bounds, reason = "private XFrameCtx")]
226    pub fn step_frame_surface<R, A, F, RE>(
227        &mut self,
228        driver: &mut RunDriver<R>,
229        app: &mut A,
230        width: u16,
231        height: u16,
232        depth: u8,
233        clear_redraw: bool,
234        events: &mut [Event],
235        render: F,
236    ) -> Result<RunControl, RunDriverFrameError<XError, A::Error, RE, XError>>
237    where
238        A: RunApp<Event = Event>,
239        F: FnOnce(&mut XSurfaceFrame<'_>) -> Result<(), RE>,
240    {
241        // 1. Collect backend events.
242        let written = self.backend.collect_events(events).map_err(RunDriverFrameError::Backend)?;
243        let events = &events[..written];
244        // 2. Build the runtime step.
245        let step = RunStep::new(driver.runtime_mut().tick(), driver.phase(), events);
246        // 3. Advance app logic.
247        let control = app.run_step(step).map_err(RunDriverFrameError::App)?;
248        // 4B. Stop before rendering/presenting.
249        if matches!(control, RunControl::Stop) {
250            driver.runtime_mut().transition(RunPhase::Stopped);
251            return Ok(control);
252        }
253        // 5B. Ensure the retained surface and render directly into it.
254        let mut surface = self
255            .presenter
256            .surface_frame(&self.backend.display, width, height, depth)
257            .map_err(RunDriverFrameError::Present)?;
258        render(&mut surface).map_err(RunDriverFrameError::Render)?;
259        // 6B. Present after the surface borrow ends.
260        self.presenter
261            .present_surface(&mut self.backend.display, &mut self.backend.window, clear_redraw)
262            .map_err(RunDriverFrameError::Present)?;
263        // 7B. Advance runtime.
264        driver.runtime_mut().tick_once();
265        Ok(control)
266    }
