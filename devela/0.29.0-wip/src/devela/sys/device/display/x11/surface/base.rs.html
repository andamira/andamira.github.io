10#[doc = crate::_tags!(unix runtime)]
11/// Pixel-backed X11 drawing surface.
12#[doc = crate::_doc_meta!{location("sys/device/display/x11")}]
13///
14/// Owns raster pixels independently of any one rendering policy, and can present
15/// them to an [`XWindow`] using either shared memory or plain client-side upload.
16#[derive(Debug)]
17pub(crate) struct XSurface {
18    width: u16,
19    height: u16,
20    depth: u8,
21    storage: XSurfaceStorage,
22}
23impl XSurface {
24    fn _new(width: u16, height: u16, depth: u8, storage: XSurfaceStorage) -> Self {
25        Self { width, height, depth, storage }
26    }
27    /// Creates the best available X11 pixel surface for the display.
28    pub fn new(display: &XDisplay, width: u16, height: u16, depth: u8) -> Result<Self, XError> {
29        #[cfg(not(ffi_xcb_shm··))]
30        return Self::new_cpu(display, width, height, depth);
31        #[cfg(ffi_xcb_shm··)]
32        match XShmBuffer::new(display, width, height, depth) {
33            Ok(shm) => Ok(Self::_new(width, height, depth, XSurfaceStorage::Shm(shm))),
34            Err(XError::ExtensionUnavailable(_)) => Self::new_cpu(display, width, height, depth),
35            Err(err) => Err(err),
36        }
37    }
38    /// Creates a CPU-backed X11 pixel surface.
39    pub fn new_cpu(display: &XDisplay, width: u16, height: u16, depth: u8) -> Result<Self, XError> {
40        let cpu = XCpuBuffer::new(display, width, height, depth)?;
41        Ok(Self::_new(width, height, depth, XSurfaceStorage::Cpu(cpu)))
42    }
43    /// Creates an SHM-backed X11 pixel surface.
44    #[cfg(ffi_xcb_shm··)]
45    pub fn new_shm(display: &XDisplay, width: u16, height: u16, depth: u8) -> Result<Self, XError> {
46        let shm = XShmBuffer::new(display, width, height, depth)?;
47        Ok(Self::_new(width, height, depth, XSurfaceStorage::Shm(shm)))
48    }
49    ///
50    pub fn present(&self, display: &mut XDisplay, window: &XWindow) -> Result<(), XError> {
51        self.storage.present(display, window, self.width, self.height, self.depth)
52    }
53    /// Returns the current X11 image presentation mode.
54    pub const fn mode(&self) -> XImageMode {
55        match self.storage {
56            XSurfaceStorage::Cpu(_) => XImageMode::Cpu,
57            #[cfg(ffi_xcb_shm··)]
58            XSurfaceStorage::Shm(_) => XImageMode::Shm,
59        }
60    }
61}
62#[rustfmt::skip]
63impl XImageStore for XSurface {
64    fn size(&self) -> (u16, u16) { (self.width, self.height) }
65    fn depth(&self) -> u8 { self.depth }
66    fn bytes(&self) -> &[u8] { self.storage.bytes() }
67    fn bytes_mut(&mut self) -> &mut [u8] { self.storage.bytes_mut() }
68    fn resize(&mut self, display: &XDisplay, width: u16, height: u16, depth: u8)
69        -> Result<(), XError> {
70        self.storage.resize(display, width, height, depth)
71    }
72}

75/// Borrowed mutable X11 surface for direct frame rendering.
76#[doc = crate::_doc_meta!{
77    location("sys/device/display/x11"),
78    #[cfg(target_pointer_width = "64")]
79    test_size_of(__: XSurfaceFrame<'_> = 16|128; niche Option),
80}]
81///
82/// This exposes the retained X11 presentation surface for one frame.
83///
84/// Drawing into this surface avoids the intermediate scene-to-surface copy
85/// used by [`XRasterRenderer`][crate::XRasterRenderer].
86///
87/// `XSurfaceFrame` is the X11 direct-surface path.
88///
89/// It exposes the retained X11 surface for one frame, allowing callers to render
90/// directly into the CPU/SHM presentation buffer.
91///
92/// Use it when X11-specific performance or surface control matters.
93pub struct XSurfaceFrame<'a> {
94    surface: &'a mut XSurface,
95    bytes_per_line: u32,
96}
97#[rustfmt::skip]
98impl<'a> XSurfaceFrame<'a> {
99    pub(crate) const fn _new(surface: &'a mut XSurface, bytes_per_line: u32) -> Self {
100        Self { surface, bytes_per_line }
101    }
102    /// Returns the surface width in pixels.
103    #[must_use]
104    pub const fn width(&self) -> u16 { self.surface.width }
105    /// Returns the surface height in pixels.
106    #[must_use]
107    pub const fn height(&self) -> u16 { self.surface.height }
108    /// Returns the surface pixel depth in bits.
109    #[must_use]
110    pub const fn depth(&self) -> u8 { self.surface.depth }
111    /// Returns the byte stride between consecutive rows.
112    #[must_use]
113    pub const fn bytes_per_line(&self) -> u32 { self.bytes_per_line }
114    /// Returns the active surface backing mode.
115    pub const fn mode(&self) -> XImageMode { self.surface.mode() }
116    /// Returns whether each row has no backend padding.
117    #[must_use]
118    pub const fn is_tight_rows(&self) -> bool {
119        let bits = self.width() as u32 * self.depth() as u32;
120        bits.div_ceil(8) == self.bytes_per_line
121    }
122    /// Returns the mutable surface bytes for direct rendering.
123    pub fn bytes_mut(&mut self) -> &mut [u8] { self.surface.bytes_mut() }
124}

126/// Retained X11 image backing, either CPU-owned or MIT-SHM-backed.
127#[derive(Debug)]
128enum XSurfaceStorage {
129    Cpu(XCpuBuffer),
130    #[cfg(ffi_xcb_shm··)]
131    Shm(XShmBuffer),
132}
133#[rustfmt::skip]
134impl XSurfaceStorage {
135    #[allow(unused_variables, reason = "display gated by ffi_xcb_shm··")]
136    pub fn present(&self, display: &mut XDisplay, window: &XWindow,
137        width: u16, height: u16, depth: u8) -> Result<(), XError> {
138        match self {
139            Self::Cpu(cpu) => {
140                window.put_image_bytes(width, height, depth, cpu.bytes());
141                Ok(())
142            }
143            #[cfg(ffi_xcb_shm··)]
144            Self::Shm(shm) => shm.present(display, window),
145        }
146    }
147}
148impl XImageStore for XSurfaceStorage {
149    fn size(&self) -> (u16, u16) {
150        match self {
151            Self::Cpu(cpu) => cpu.size(),
152            #[cfg(ffi_xcb_shm··)]
153            Self::Shm(shm) => shm.size(),
154        }
155    }
156    fn depth(&self) -> u8 {
157        match self {
158            Self::Cpu(cpu) => cpu.depth(),
159            #[cfg(ffi_xcb_shm··)]
160            Self::Shm(shm) => shm.depth(),
161        }
162    }
163    fn bytes(&self) -> &[u8] {
164        match self {
165            Self::Cpu(cpu) => cpu.bytes(),
166            #[cfg(ffi_xcb_shm··)]
167            Self::Shm(shm) => shm.bytes(),
168        }
169    }
170    fn bytes_mut(&mut self) -> &mut [u8] {
171        match self {
172            Self::Cpu(cpu) => cpu.bytes_mut(),
173            #[cfg(ffi_xcb_shm··)]
174            Self::Shm(shm) => shm.bytes_mut(),
175        }
176    }
177    fn resize(
178        &mut self,
179        display: &XDisplay,
180        width: u16,
181        height: u16,
182        depth: u8,
183    ) -> Result<(), XError> {
184        match self {
185            Self::Cpu(cpu) => cpu.resize(display, width, height, depth),
186            #[cfg(ffi_xcb_shm··)]
187            Self::Shm(shm) => shm.resize(display, width, height, depth),
188        }
189    }
