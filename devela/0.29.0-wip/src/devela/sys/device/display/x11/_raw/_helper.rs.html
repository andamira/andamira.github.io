9/// Convenience wrapper for setting a 32-bit property.
10///
11/// `data` is interpreted as an array of 32-bit values (`format = 32`).
12/// Common use cases include: `_NET_WM_STATE`, `_NET_WM_WINDOW_TYPE`,
13/// `WM_NORMAL_HINTS`, and other ICCCM/EWMH integer properties.
14#[inline(always)]
15pub(crate) fn change_property_u32(
16    conn: *mut xcb_connection_t,
17    win: u32,
18    property: u32,
19    type_atom: u32,
20    data: &[u32],
21) {
22    let bytes =
23        unsafe { ::core::slice::from_raw_parts(data.as_ptr().cast::<u8>(), data.len() * 4) };
24    change_property_u8(conn, win, property, type_atom, 32, bytes);
25}
26
27/// Convenience wrapper for setting UTF-8 string properties.
28///
29/// Used by EWMH (`_NET_WM_NAME`, `_NET_WM_ICON_NAME`) and ICCCM (`WM_NAME`)
30/// when providing a UTF-8 encoded window title.
31#[inline(always)]
32pub(crate) fn change_property_str(
33    conn: *mut xcb_connection_t,
34    win: u32,
35    property: u32,
36    type_atom: u32,
37    s: &str,
38) {
39    change_property_u8(conn, win, property, type_atom, 8, s.as_bytes());
40}
41
42/// Thin safe wrapper over `xcb_change_property` for all formats.
43/// `format` must be 8, 16, or 32. `data` is raw bytes.
44///
45/// `data_len_units` is automatically derived from format.
46#[inline(always)]
47pub(crate) fn change_property_u8(
48    conn: *mut xcb_connection_t,
49    win: u32,
50    property: u32,
51    type_atom: u32,
52    format: u8,
53    data: &[u8],
54) {
55    debug_assert!(format == 8 || format == 16 || format == 32);
56
57    let unit = (format as usize) / 8;
58    let len_units = is![unit == 0, 0, data.len() as u32 / unit as u32];
59
60    unsafe {
61        xcb_change_property(
62            conn,
63            xcb_prop_mode::XCB_PROP_MODE_REPLACE as u8,
64            win,
65            property,
66            type_atom,
67            format,
68            len_units,
69            data.as_ptr().cast(),
70        );
71    }
72}
