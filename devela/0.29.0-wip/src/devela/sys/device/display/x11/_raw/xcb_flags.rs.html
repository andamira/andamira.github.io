103/* window attribute value-mask bits (`xcb_cw_t`). */
104
105/**< Overrides the default background-pixmap. The background pixmap and window must
106have the same root and same depth. Any size pixmap can be used, although some
107sizes may be faster than others.
108
109If `XCB_BACK_PIXMAP_NONE` is specified, the window has no defined background.
110The server may fill the contents with the previous screen contents or with
111contents of its own choosing.
112
113If `XCB_BACK_PIXMAP_PARENT_RELATIVE` is specified, the parent's background is
114used, but the window must have the same depth as the parent (or a Match error
115results).   The parent's background is tracked, and the current version is
116used each time the window background is required. */
117pub(crate) const XCB_CW_BACK_PIXMAP: u32 = 1 << 0;
118
119/// Overrides `BackPixmap`.
120///
121/// A pixmap of undefined size filled with the specified background pixel is used for the
122/// background. Range-checking is not performed, the background pixel is truncated
123/// to the appropriate number of bits.
124pub(crate) const XCB_CW_BACK_PIXEL: u32 = 1 << 1;
125
126/// Overrides the default border-pixmap.
127///
128/// The border pixmap and window must have the same root and the same depth.
129/// Any size pixmap can be used, although some sizes may be faster than others.
130///
131/// The special value `XCB_COPY_FROM_PARENT` means the parent's border pixmap is copied
132/// (subsequent changes to the parent's border attribute do not affect the child),
133/// but the window must have the same depth as the parent.
134pub(crate) const XCB_CW_BORDER_PIXMAP: u32 = 1 << 2;
135
136/// Overrides `BorderPixmap`.
137///
138/// A pixmap of undefined size filled with the specified border pixel is used for the border.
139/// Range checking is not performed on the border-pixel value,
140/// it is truncated to the appropriate number of bits.
141pub(crate) const XCB_CW_BORDER_PIXEL: u32 = 1 << 3;
142
143/// Defines which region of the window should be retained if the window is resized.
144pub(crate) const XCB_CW_BIT_GRAVITY: u32 = 1 << 4;
145
146/// Defines how the window should be repositioned if the parent is resized.
147/// (see `ConfigureWindow`)
148pub(crate) const XCB_CW_WIN_GRAVITY: u32 = 1 << 5;
149
150/**< A backing-store of `WhenMapped` advises the server that maintaining contents of
151obscured regions when the window is mapped would be beneficial. A backing-store
152of `Always` advises the server that maintaining contents even when the window
153is unmapped would be beneficial. In this case, the server may generate an
154exposure event when the window is created. A value of `NotUseful` advises the
155server that maintaining contents is unnecessary, although a server may still
156choose to maintain contents while the window is mapped. Note that if the server
157maintains contents, then the server should maintain complete contents not just
158the region within the parent boundaries, even if the window is larger than its
159parent. While the server maintains contents, exposure events will not normally
160be generated, but the server may stop maintaining contents at any time. */
161pub(crate) const XCB_CW_BACKING_STORE: u32 = 1 << 6;
162
163/// The backing-planes indicates (with bits set to 1) which bit planes of the window
164/// hold dynamic data that must be preserved in backing-stores and during save-unders.
165pub(crate) const XCB_CW_BACKING_PLANES: u32 = 1 << 7;
166
167/// The backing-pixel specifies what value to use in planes not covered by backing-planes.
168///
169/// The server is free to save only the specified bit planes in the backing-store or save-under
170/// and regenerate the remaining planes with the specified pixel value.
171/// Any bits beyond the specified depth of the window in these values are simply ignored.
172pub(crate) const XCB_CW_BACKING_PIXEL: u32 = 1 << 8;
173
174/// The override-redirect specifies whether map and configure requests

176/// typically to inform a window manager not to tamper with the window.
177pub(crate) const XCB_CW_OVERRIDE_REDIRECT: u32 = 1 << 9;
178
179/// If 1, the server is advised that when this window is mapped,
180/// saving the contents of windows it obscures would be beneficial.
181pub(crate) const XCB_CW_SAVE_UNDER: u32 = 1 << 10;

183/// The event-mask defines which events the client is interested in for this window

187/// The do-not-propagate-mask defines which events should not be propagated to

189pub(crate) const XCB_CW_DONT_PROPAGATE: u32 = 1 << 12;

191/// The colormap specifies the colormap that best reflects the true colors of the window.
192///
193/// Servers capable of supporting multiple hardware colormaps may use this information,
194/// and window managers may use it for InstallColormap requests.
195///
196/// The colormap must have the same visual type and root as the window (or a Match error results).
197/// If CopyFromParent is specified, the parent's colormap is copied (subsequent changes to the
198/// parent's colormap attribute do not affect the child).
199///
200/// However, the window must have the same visual type as the parent (or a Match error results),
201/// and the parent must not have a colormap of None (or a Match error results). For an explanation
202/// of None, see FreeColormap request. The colormap is copied by sharing the colormap object
203/// between the child and the parent, not by making a complete copy of the colormap contents.
204pub(crate) const XCB_CW_COLORMAP: u32 = 1 << 13;
205
206/// If a cursor is specified, it will be used whenever the pointer is in the window.
207///
208/// If None is specified, the parent's cursor will be used when the pointer is in the window,
209/// and any change in the parent's cursor will cause an immediate change in the displayed cursor.
210pub(crate) const XCB_CW_CURSOR: u32 = 1 << 14;
