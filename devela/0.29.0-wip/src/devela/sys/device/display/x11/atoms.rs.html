3//! Defines [`XAtoms`].

6use super::_raw;
7use crate::{Libc, Ptr, is};
8
9#[doc = crate::_tags!(linux uid runtime)]
10/// Cached atoms required for high–level event interpretation.
11#[doc = crate::_doc_meta!{location("sys/device/display/x11")}]
12///
13/// X11 atoms are integer identifiers allocated by the server and
14/// used as symbolic names for properties and protocols.
15///
16/// This struct caches only the atoms that are relevant for the current minimal X11 backend.
17#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
18pub(crate) struct XAtoms {
19    /// The `WM_NORMAL_HINTS atom.
20    pub wm_normal_hints: u32,
21    /// The `WM_PROTOCOLS` atom.
22    pub wm_protocols: u32,
23    /// The `WM_DELETE_WINDOW` protocol atom.
24    pub wm_delete_window: u32,
25}
26impl XAtoms {
27    /// Returns a new [`XAtoms`] by resolving and caching the relevant atoms.
28    pub(crate) fn new(conn: *mut _raw::xcb_connection_t) -> Self {
29        let wm_normal_hints = XAtoms::get_atom(conn, b"WM_NORMAL_HINTS");
30        let wm_protocols = XAtoms::get_atom(conn, b"WM_PROTOCOLS");
31        let wm_delete_window = XAtoms::get_atom(conn, b"WM_DELETE_WINDOW");
32        Self { wm_normal_hints, wm_protocols, wm_delete_window }
33    }
34
35    /// Resolves an X11 atom by name.
36    ///
37    /// Returns `0` if the atom does not exist or an error occurs.
38    pub(crate) fn get_atom(conn: *mut _raw::xcb_connection_t, name: &[u8]) -> u32 {
39        unsafe {
40            let cookie = _raw::xcb_intern_atom(conn, 0, name.len() as u16, name.as_ptr());
41            let mut err: *mut _raw::xcb_generic_error_t = Ptr::null_mut();
42            let reply = _raw::xcb_intern_atom_reply(conn, cookie, &mut err);
43            is![reply.is_null(), return 0];
44            let atom = (*reply).atom;
45            Libc::free(reply as *mut _);
46            atom
47        }
48    }
49
50    /// Sets a window property that stores a single atom value.
51    ///
52    /// This is mainly used to advertise support for a protocol such as `WM_DELETE_WINDOW`.
53    pub(crate) fn set_property_atom(
54        &self,
55        conn: *mut _raw::xcb_connection_t,
56        window: u32,
57        property: u32,
58        atom_value: u32,
59    ) {
60        let data = atom_value.to_ne_bytes();
61        unsafe {
62            _raw::xcb_change_property(
63                conn,
64                _raw::xcb_prop_mode::XCB_PROP_MODE_REPLACE as u8,
65                window,
66                property,
67                _raw::xcb_atom_enum_t::XCB_ATOM_ATOM as u32,
68                32,
69                1,
70                data.as_ptr(),
71            );
72        }
73    }
74}
