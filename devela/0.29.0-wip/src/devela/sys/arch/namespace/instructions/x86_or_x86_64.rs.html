1// devela/src/sys/arch/namespace/instructions/x86_or_x86_64.rs

3//! Implements processor instructions for both x86 and x86_64.
4//
5
6use crate::{_ARCH, Arch};
7
8#[cfg(target_arch = "x86")]
9use ::core::arch::x86::{__rdtscp, _rdtsc};
10#[cfg(target_arch = "x86_64")]
11use ::core::arch::x86_64::{__rdtscp, _rdtsc};
12
13/// # X86 or X86_64 instructions
14impl Arch {
15    /// Reads the current value of the processor's time-stamp counter.
16    ///
17    #[doc = concat!("See `::core::arch::", _ARCH!(), "::`[`_rdtsc()`].")]
18    #[must_use]
19    #[inline(always)]
20    pub fn rdtsc() -> u64 {
21        unsafe { _rdtsc() }
22    }
23
24    /// Reads the current value of the processor's time-stamp counter and the processor ID.
25    ///
26    /// The processor ID:
27    /// - is a CPU core identifier (logical/core number), not a process ID.
28    /// - is tipically a small u32 (e.g., 0 to N-1 for N cores).
29    #[doc = concat!("See `::core::arch::", _ARCH!(), "::`[`__rdtscp()`].")]
30    #[must_use]
31    #[inline(always)]
32    pub fn rdtscp() -> (u64, u32) {
33        let mut aux: u32 = 0;
34        let res = unsafe { __rdtscp(&mut aux as *mut u32) };
35        (res, aux)
36    }
37}
