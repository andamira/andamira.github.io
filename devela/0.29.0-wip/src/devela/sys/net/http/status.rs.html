6#[doc = crate::_tags!(network protocol)]
7/// The response class of an HTTP status code.
8#[doc = crate::_doc_meta!{location("sys/net/http")}]
9#[repr(u8)]
10#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
11pub enum HttpStatusClass {
12    /// `1xx`: processing continues.
13    Informational = 1,
14    /// `2xx`: successful result.
15    Success = 2,
16    /// `3xx`: redirection or related action.
17    Redirection = 3,
18    /// `4xx`: client error.
19    ClientError = 4,
20    /// `5xx`: server error.
21    ServerError = 5,
22}

25/// HTTP response status code.
26#[doc = crate::_doc_meta!{location("sys/net/http")}]

29#[repr(transparent)]
30#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]

171/// # Fundamental methods
172impl HttpStatus {
173    /// Creates a valid HTTP status code.
174    #[must_use]
175    pub const fn from_code(code: u16) -> Option<Self> {
176        if code >= 100 && code <= 599 { Some(Self(code)) } else { None }
177    }
178
179    /// Returns the three-digit status code.
180    #[must_use]
181    pub const fn code(self) -> u16 {
182        self.0
183    }
184
185    /// Returns the status-code class.
186    #[must_use]
187    pub const fn class(self) -> HttpStatusClass {
188        match self.0 / 100 {
189            1 => HttpStatusClass::Informational,
190            2 => HttpStatusClass::Success,
191            3 => HttpStatusClass::Redirection,
192            4 => HttpStatusClass::ClientError,
193            5 => HttpStatusClass::ServerError,
194            _ => unreachable!(),
195        }
196    }
197
198    /// Returns whether this is a `1xx` informational status.
199    #[must_use]
200    pub const fn is_informational(self) -> bool {
201        self.0 >= 100 && self.0 < 200
202    }
203    /// Returns whether this is a `2xx` successful status.
204    #[must_use]
205    pub const fn is_success(self) -> bool {
206        self.0 >= 200 && self.0 < 300
207    }
208    /// Returns whether this is a `3xx` redirection status.
209    #[must_use]
210    pub const fn is_redirection(self) -> bool {
211        self.0 >= 300 && self.0 < 400
212    }
213    /// Returns whether this is a `4xx` client-error status.
214    #[must_use]
215    pub const fn is_client_error(self) -> bool {
216        self.0 >= 400 && self.0 < 500
217    }
218    /// Returns whether this is a `5xx` server-error status.
219    #[must_use]
220    pub const fn is_server_error(self) -> bool {
221        self.0 >= 500 && self.0 < 600
222    }
223    /// Returns whether this is a final, non-informational status.
224    #[must_use]
225    pub const fn is_final(self) -> bool {
226        self.0 >= 200
227    }
228    /// Returns whether this is a client or server error.
229    #[must_use]
230    pub const fn is_error(self) -> bool {
231        self.0 >= 400
232    }
233}
