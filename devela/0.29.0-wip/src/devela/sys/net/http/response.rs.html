9#[doc = crate::_tags!(network protocol)]
10/// HTTP response head.
11#[doc = crate::_doc_meta!{location("sys/net/http")}]
12///
13/// Represents an HTTP status line and basic representation headers.
14///
15/// The current encoder writes the textual HTTP/1 wire format.
16#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
17pub struct HttpResponseHead<'a> {
18    version: HttpVersion,
19    status: HttpStatus,
20    content_type: Option<&'a str>,
21    content_length: Option<usize>,
22}
23impl<'a> HttpResponseHead<'a> {
24    /// Creates an HTTP/1.1 response head with the given status.
25    #[must_use]
26    pub const fn new(status: HttpStatus) -> Self {
27        Self {
28            version: HttpVersion::Http11,
29            status,
30            content_type: None,
31            content_length: None,
32        }
33    }
34    /// Creates a `200 OK` HTTP/1.1 response head.
35    #[must_use]
36    pub const fn ok() -> Self {
37        Self::new(HttpStatus::OK)
38    }
39    /// Creates a `404 Not Found` HTTP/1.1 response head.
40    #[must_use]
41    pub const fn not_found() -> Self {
42        Self::new(HttpStatus::NOT_FOUND)
43    }
44    /// Sets the HTTP version.
45    #[must_use]
46    pub const fn with_version(mut self, version: HttpVersion) -> Self {
47        self.version = version;
48        self
49    }
50    /// Sets the `Content-Type` header.
51    ///
52    /// The value is validated when the response head is encoded.
53    #[must_use]
54    pub const fn with_content_type(mut self, content_type: &'a str) -> Self {
55        self.content_type = Some(content_type);
56        self
57    }
58    /// Sets the `Content-Length` header.
59    #[must_use]
60    pub const fn with_content_length(mut self, len: usize) -> Self {
61        self.content_length = Some(len);
62        self
63    }
64    /// Returns the HTTP version.
65    #[must_use]
66    pub const fn version(self) -> HttpVersion {
67        self.version
68    }
69    /// Returns the response status.
70    #[must_use]
71    pub const fn status(self) -> HttpStatus {
72        self.status
73    }
74    /// Returns the `Content-Type` value, when set.
75    #[must_use]
76    pub const fn content_type(self) -> Option<&'a str> {
77        self.content_type
78    }
79    /// Returns the `Content-Length` value, when set.
80    #[must_use]
81    pub const fn content_length(self) -> Option<usize> {
82        self.content_length
83    }
84    /// Returns the encoded HTTP/1 response-head length.
85    ///
86    /// Returns an error when the configured version has no HTTP/1 textual
87    /// representation or a configured header value is invalid.
88    pub const fn http1_encoded_len(self) -> Result<usize, HttpError> {
89        let Some(version) = self.version.http1_token() else {
90            return Err(HttpError::InvalidVersion);
91        };
92        let reason = match self.status.reason() {
93            Some(reason) => reason,
94            None => "",
95        };
96        let mut len = version.len()
97            + 1 // SP
98            + 3 // status code
99            + 1 // SP
100            + reason.len()
101            + 2; // CRLF
102        if let Some(content_type) = self.content_type {
103            if !Self::is_valid_header_value(content_type.as_bytes()) {
104                return Err(HttpError::InvalidHeaderValue);
105            }
106            len += b"Content-Type: ".len() + content_type.len() + 2; // CRLF
107        }
108        if let Some(content_length) = self.content_length {
109            len += b"Content-Length: ".len() + Digits(content_length).count_digits10() as usize + 2; // CRLF
110        }
111        Ok(len + 2) // final CRLF
112    }
113    /// Writes the textual HTTP/1 response head into `dst`.
114    ///
115    /// Returns the number of bytes written.
116    ///
117    /// The operation is atomic: if an error is returned, `dst` is unchanged.
118    pub const fn write_http1_into(self, dst: &mut [u8]) -> Result<usize, HttpError> {
119        let needed = match self.http1_encoded_len() {
120            Ok(needed) => needed,
121            Err(error) => return Err(error),
122        };
123        if dst.len() < needed {
124            return Err(HttpError::NotEnoughSpace(needed));
125        }
126        let Some(version) = self.version.http1_token() else {
127            return Err(HttpError::InvalidVersion);
128        };
129        let reason = match self.status.reason() {
130            Some(reason) => reason,
131            None => "",
132        };
133        let mut pos = 0;
134        /* status line */
135        write_at!(dst, +=pos, @version.as_bytes(), b' ');
136        pos += Digits(self.status.code()).write_digits10(dst, pos);
137        write_at!(dst, +=pos, b' ', @reason.as_bytes(), @b"\r\n");
138        /* headers */
139        if let Some(content_type) = self.content_type {
140            write_at!(dst, +=pos, @b"Content-Type: ", @content_type.as_bytes(), @b"\r\n");
141        }
142        if let Some(content_length) = self.content_length {
143            write_at!(dst, +=pos, @b"Content-Length: ");
144            pos += Digits(content_length).write_digits10(dst, pos);
145            write_at!(dst, +=pos, @b"\r\n");
146        }
147        /* end of head */
148        write_at!(dst, +=pos, @b"\r\n");
149        Ok(pos)
150    }
151    /// Returns whether `bytes` form a safe HTTP field value.
152    const fn is_valid_header_value(bytes: &[u8]) -> bool {
153        whilst! { i in 0..bytes.len(); {
154            let byte = bytes[i];
155            // Permit HTAB, visible ASCII and non-ASCII opaque bytes.
156            // Reject CR, LF, DEL and the remaining control bytes.
157            if (byte < b' ' && byte != b'\t') || byte == 0x7f { return false; }
158        }}
159        true
160    }
161}
