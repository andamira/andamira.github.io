11/// A duplex byte I/O primitive.
12#[doc = crate::_doc_meta!{location("sys/io")}]
13///
14/// This trait is blanket-implemented for any type that implements both
15/// [`IoRead`] and [`IoWrite`]. Duplex capability does not imply simultaneous
16/// readiness for reading and writing.
17///
18/// It provides fundamental duplex semantics without imposing buffering,
19/// allocation, or higher-level protocols.
20///
21/// Convenience methods offer safe, low-overhead patterns for partial I/O,
22/// full-buffer writes, and direct stream-to-stream copying.
23pub trait IoDuplex: IoRead + IoWrite {
24    /// Reads until at least `min` bytes are read into `buf`, unless EOF occurs first.
25    fn read_at_least(&mut self, buf: &mut [u8], min: usize) -> IoResult<usize> {
26        let mut filled = 0;
27        while filled < min {
28            let n = self.read(&mut buf[filled..])?;
29            is![n == 0, break];
30            filled += n;
31        }
32        Ok(filled)
33    }
34
35    /// Copies all available data from this duplex into the given writer.
36    fn copy_to<W: IoWrite>(&mut self, out: &mut W) -> IoResult<u64> {
37        const BUF: usize = 8 * 1024;
38        let mut tmp = [0u8; BUF];
39        let mut total = 0u64;
40        loop {
41            let n = self.read(&mut tmp)?;
42            is![n == 0, break];
43            out.write_all(&tmp[..n])?;
44            total += n as u64;
45        }
46        Ok(total)
47    }
48
49    /// Reads bytes into `out` until `delim` is found or no more space remains.
50    ///
51    /// The delimiter is included in the output when encountered. Stops early on EOF.
52    /// This method performs no allocation.
53    fn read_until_byte(&mut self, delim: u8, out: &mut [u8]) -> IoResult<usize> {
54        let mut count = 0;
55        for slot in out.iter_mut() {
56            let mut b = [0];
57            let n = self.read(&mut b)?;
58            is![n == 0, break];
59            *slot = b[0];
60            count += 1;
61            is![b[0] == delim, break];
62        }
63        Ok(count)
64    }
65}
