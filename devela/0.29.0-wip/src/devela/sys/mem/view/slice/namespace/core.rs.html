3use crate::{Slice, is};
4use ::core::slice::{from_mut, from_ref};
5#[allow(unused_imports, reason = "unsafe feature-gated")]
6use ::core::slice::{from_raw_parts, from_raw_parts_mut};

8/// # `core::slice` namespaced methods.
9#[rustfmt::skip]
10impl<T> Slice<T> {
11    /// Clones all elements from `src` into `dst`.
12    ///
13    /// # Panics
14    /// Panics if `src` and `dst` slices have different lengths.
15    ///
16    /// See `core::slice::`[`clone_from_slice`][slice#method.clone_from_slice].
17    #[inline(always)]
18    pub fn clone(dst: &mut [T], src: &[T]) where T: Clone { dst.clone_from_slice(src); }
19
20    /// Copies all elements from `src` into `dst` using a memcpy.
21    ///
22    /// # Panics
23    /// Panics if `src` and `dst` slices have different lengths.
24    ///
25    /// See `core::slice::`[`copy_from_slice`][slice#method.copy_from_slice].
26    #[inline(always)]
27    pub const fn copy(dst: &mut [T], src: &[T]) where T: Copy { dst.copy_from_slice(src); }
28
29    /// Returns a shared reference to the element at `index`, or `None` if out of bounds.
30    ///
31    /// This is a stable const analogue just for single-element checked access.
32    ///
33    /// See `core::slice::`[`get`][slice#method.get].
34    #[must_use]
35    #[inline(always)]
36    pub const fn get(slice: &[T], index: usize) -> Option<&T> {
37        is! { index < slice.len(), Some(&slice[index]), None }
38    }
39    /// Returns an exclusive reference to the element at `index`, or `None` if out of bounds.
40    ///
41    /// This is a stable const analogue just for single-element checked access.
42    ///
43    /// See `core::slice::`[`get_mut`][slice#method.get_mut].
44    #[must_use]
45    #[inline(always)]
46    pub const fn get_mut(slice: &mut [T], index: usize) -> Option<&mut T> {
47        is! { index < slice.len(), Some(&mut slice[index]), None }
48    }
49
50    /// Converts a reference to `T` into a slice of length 1 (without copying).
51    ///
52    /// See `core::slice::`[`from_ref`].
53    #[must_use]
54    #[inline(always)]
55    pub const fn from_ref(s: &T) -> &[T] { from_ref(s) }
56
57    /// Converts a reference to `T` into a slice of length 1 (without copying).
58    ///
59    /// See `core::slice::`[`from_mut`].
60    #[must_use]
61    #[inline(always)]
62    pub const fn from_mut(s: &mut T) -> &mut [T] { from_mut(s) }
63
64    /// Forms a shared slice from a pointer and a length.
65    ///
66    /// # Safety
67    /// See `core::slice::`[`from_raw_parts`]
68    ///
69    /// See also `Ptr::`[`slice_from_raw_parts`][crate::Ptr::slice_from_raw_parts].
70    #[inline(always)]
71    #[cfg_attr(nightly_doc, doc(cfg(unsafe··)))]
72    #[cfg(all(not(feature = "safe_mem"), unsafe··))]
73    pub const unsafe fn from_raw_parts<'a>(data: *const T, len: usize) -> &'a [T] {
74        // SAFETY: Caller must uphold the safety contract.
75        unsafe { from_raw_parts(data, len) }
76    }
77
78    /// Forms an exclusive slice from a pointer and a length.
79    ///
80    /// # Safety
81    /// See `core::slice::`[`from_raw_parts_mut`].
82    ///
83    /// See also `Ptr::`[`slice_from_raw_parts_mut`][crate::Ptr::slice_from_raw_parts_mut].
84    #[inline(always)]
85    #[cfg_attr(nightly_doc, doc(cfg(unsafe··)))]
86    #[cfg(all(not(feature = "safe_mem"), unsafe··))]
87    pub const unsafe fn from_raw_parts_mut<'a>(data: *mut T, len: usize) -> &'a mut [T] {
88        // SAFETY: Caller must uphold the safety contract.
89        unsafe { from_raw_parts_mut(data, len) }
90    }
91}
