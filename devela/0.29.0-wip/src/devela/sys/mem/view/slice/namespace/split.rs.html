5/// # `*split*` API methods for subslicing.
6impl<T> Slice<T> {
7    /* left split */
8
9    /// Returns the leftmost sub-`slice` with the given maximum `len`.
10    ///
11    /// If `len > self.len()` it returns the full slice.
12    ///
13    /// # Examples
14    /// ```
15    /// # use devela::Slice;
16    /// let v = [1, 2, 3, 4, 5, 6];
17    /// assert_eq!(Slice::lsplit(&v, 3), &[1, 2, 3]);
18    /// assert_eq!(Slice::lsplit(&v, 0), &[] as &[i32]);
19    /// assert_eq!(Slice::lsplit(&v, 10), &[1, 2, 3, 4, 5, 6]);
20    /// ```
21    pub const fn lsplit(slice: &[T], len: usize) -> &[T] {
22        let end_idx = Cmp(len).clamp(0, slice.len());
23        let (left, _) = slice.split_at(end_idx);
24        left
25    }
26
27    /// Returns the leftmost exclusive sub-`slice` with the given maximum `len`.
28    ///
29    /// If `left_len > slice.len()` it returns the full slice.
30    ///
31    /// # Examples
32    /// ```
33    /// # use devela::Slice;
34    /// let mut v = [1, 2, 3, 4, 5, 6];
35    /// assert_eq!(Slice::lsplit_mut(&mut v, 3), &mut [1, 2, 3]);
36    /// assert_eq!(Slice::lsplit_mut(&mut v, 0), &mut [] as &mut [i32]);
37    /// assert_eq!(Slice::lsplit_mut(&mut v, 10), &mut [1, 2, 3, 4, 5, 6]);
38    /// ```
39    /// See also [`Slice::lsplit_mut`].
40    pub const fn lsplit_mut(slice: &mut [T], len: usize) -> &mut [T] {
41        let end_idx = Cmp(len).clamp(0, slice.len());
42        let (left, _) = slice.split_at_mut(end_idx);
43        left
44    }

48    /// Returns the rightmost sub-`slice` with the given maximum `len`.
49    ///
50    /// If `left_len > slice.len()` it returns the full slice.
51    ///
52    /// # Examples
53    /// ```
54    /// # use devela::Slice;
55    /// let v = [1, 2, 3, 4, 5, 6];
56    /// assert_eq!(Slice::rsplit(&v, 3), &[4, 5, 6]);
57    /// assert_eq!(Slice::rsplit(&v, 0), &[] as &[i32]);
58    /// assert_eq!(Slice::rsplit(&v, 10), &[1, 2, 3, 4, 5, 6]);
59    /// ```
60    #[must_use]
61    pub const fn rsplit(slice: &[T], len: usize) -> &[T] {
62        let start_idx = slice.len().saturating_sub(len);
63        let (_, right) = slice.split_at(start_idx);
64        right
65    }

87    /* middle split left biased */

89    /// Returns the middle sub-`slice` with the given maximum `len` and a left bias.
90    ///
91    /// In case of a non-perfect middle split, it will have one element more on the left.
92    ///
93    /// If `len > slice.len()` returns the full `slice`.
94    ///
95    /// # Examples
96    /// ```
97    /// # use devela::Slice;
98    /// let v = [1, 2, 3, 4, 5, 6];
99    /// assert_eq!(Slice::msplit_left(&v, 0), &[] as &[i32]);
100    /// assert_eq!(Slice::msplit_left(&v, 1), &[3]);
101    /// assert_eq!(Slice::msplit_left(&v, 2), &[3, 4]);
102    /// assert_eq!(Slice::msplit_left(&v, 3), &[2, 3, 4]);
103    /// assert_eq!(Slice::msplit_left(&v, 4), &[2, 3, 4, 5]);
104    /// assert_eq!(Slice::msplit_left(&v, 5), &[1, 2, 3, 4, 5]);
105    /// assert_eq!(Slice::msplit_left(&v, 10), &[1, 2, 3, 4, 5, 6]);
106    /// ```
107    /// See also [`Slice::msplit_right`].
108    #[must_use]
109    pub const fn msplit_left(slice: &[T], len: usize) -> &[T] {
110        let mid_idx = slice.len() / 2;
111        let half_len = len / 2;
112        let start_idx = mid_idx.saturating_sub(half_len + (len % 2));
113        let end_idx = Cmp(mid_idx + half_len).min(slice.len());
114        let (_, right) = slice.split_at(start_idx);
115        let (middle, _) = right.split_at(end_idx - start_idx);
116        middle
117    }

172    pub const fn msplit_right(slice: &[T], len: usize) -> &[T] {
173        let mid_idx = slice.len() / 2;
174        let half_len = len / 2;
175        let start_idx = mid_idx.saturating_sub(half_len);
176        let end_idx = Cmp(mid_idx + half_len + (len % 2)).min(slice.len());
177        let (_, right) = slice.split_at(start_idx);
178        let (middle, _) = right.split_at(end_idx - start_idx);
179        middle
180    }
