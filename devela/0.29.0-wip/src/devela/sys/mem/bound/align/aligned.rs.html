13#[doc = crate::_tags!(mem)]
14/// Marker trait to verify memory alignment between two types.
15#[doc = crate::_doc_meta!{location("sys/mem")}]
16///
17/// This trait ensures that a value of the `Candidate` type can be safely placed
18/// into a storage medium designed for the `Requirement` type, adhering to
19/// alignment requirements.
20///
21/// For runtime alignment adjustments, see [`Mem::align_down`] and [`Mem::align_up`].
22//
23// WAIT:DONE:1.79 [ptr.is_aligned](https://github.com/rust-lang/rust/pull/121948)
24// WAIT: [pointer_is_aligned_to](https://github.com/rust-lang/rust/issues/96284)
25// WAIT: [const_pointer_is_aligned](https://github.com/rust-lang/rust/issues/104203)
26#[cfg_attr(nightly_doc, doc(notable_trait))]
27#[expect(private_bounds, reason = "Sealed")]
28pub trait MemAligned: Sealed {
29    /// Checks if the `Candidate`'s alignment is compatible with the `Requirement`'s alignment.
30    ///
31    /// Returns `true` if `Candidate` type's alignment is less than or equal to the
32    /// `Requirement` type's alignment.
33    ///
34    /// Alignment represents the byte boundary to which a type must be aligned
35    /// in memory. It is crucial to ensure that types are placed in memory
36    /// locations that respect their alignment requirements.
37    #[must_use]
38    fn is_compatible() -> bool;
39
40    /// Asserts that the memory alignment of the `Candidate` type meets or
41    /// exceeds the `Requirement` type's alignment.
42    ///
43    /// # Panics
44    /// Panics if the `Candidate` type's alignment is greater than the
45    /// `Requirement` type's alignment, indicating an incompatibility that could
46    /// lead to undefined behavior or performance penalties.
47    fn assert_compatibility();
48}
49
50// #[cfg(not(feature = "nightly_const"))]
51impl<Candidate, Requirement> MemAligned for (Candidate, Requirement) {
52    fn is_compatible() -> bool {
53        align_of::<Candidate>() <= align_of::<Requirement>()
54    }
55
56    fn assert_compatibility() {
57        assert!(
58            Self::is_compatible(),
59            "`Candidate`'s alignment ({}) exceeds `Requirement`'s alignment ({}).",
60            align_of::<Candidate>(),
61            align_of::<Requirement>(),
62        );
63    }
64}
