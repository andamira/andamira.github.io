7/// Emits a compile-time warning with a provided message.
8#[doc = crate::_doc_meta!{location("code/util")}]
9///
10/// # Example
11/// ```
12/// # use devela::const_warn;
13/// fn main() {
14///     const_warn!("hello warning!");
15///     const_warn!(if size_of::<u8>() == 1, "byte alert!!");
16/// }
17/// ```
18/// That may look as follows:
19/// ```text
20/// warning: use of deprecated associated constant `main::_::__::<true>::WARNING`: hello warning!
21///  --> src/main.rs:3:5
22///   |
23/// 3 |     const_warn!("hello warning!");
24///   |     ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
25///   |
26///   = note: `#[warn(deprecated)]` on by default
27///   = note: this warning originates in the macro `$crate::const_warn` which comes from the expansion of the macro `const_warn` (in Nightly builds, run with -Z macro-backtrace for more info)
28///
29/// warning: use of deprecated associated constant `main::_::__::<true>::WARNING`: byte alert!!
30///  --> src/main.rs:4:5
31///   |
32/// 4 |     const_warn!(if size_of::<u8>() == 1, "byte alert!!");
33///   |     ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
34///   ...
35/// ```
36// IMPROVE: use a proc-macro leveraging nightly feature proc_macro_diagnostics
37#[macro_export]
38#[cfg_attr(cargo_primary_package, doc(hidden))]
39macro_rules! const_warn {
40    ($msg:literal $(,)?) => {
41        $crate::const_warn!(if true, $msg);
42    };
43    (if $cond:expr, $msg:literal $(,)?) => {
44        $crate::const_warn!(%if $cond, WARNING, $msg);
45    };
46    (%if $cond:expr, $name:ident, $msg:literal $(,)?) => {
47        #[allow(dead_code)]
48        const _: () = {
49            struct __<const B: bool>;
50            impl __<false> {
51                const $name: () = ();
52            }
53            impl __<true> {
54                #[deprecated(note = $msg)]
55                const $name: () = ();
56            }
57            let _ = __::<{ $cond }>::$name;
58        };
59    };
60}
61#[doc(inline)]
62pub use const_warn;

65/// Returns a best-effort name of the enclosing function.
66#[doc = crate::_doc_meta!{location("code/util")}]
67///
68/// This is a diagnostic helper based on [`core::any::type_name_of_val`].
69/// The exact format is not guaranteed by Rust, so this macro must not be
70/// used for semantic program behavior.
71///
72/// ## Examples
73/// ```
74/// mod bar {
75///     pub fn sample_function() {
76///         # use devela::fn_name;
77///         assert!(fn_name!().ends_with("bar::sample_function"));
78///     }
79/// }
80/// bar::sample_function();
81/// ```
82#[macro_export]
83#[cfg_attr(cargo_primary_package, doc(hidden))]
84macro_rules! fn_name {
85    () => {{
86        fn f() {}
87        let name = ::core::any::type_name_of_val(&f);
88        name.strip_suffix("::f").unwrap_or(name)
89    }};
90}
91#[doc(inline)]
92pub use fn_name;
