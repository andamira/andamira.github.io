6#[doc = crate::_tags!(assert)]
7/// Asserts the equality of a series of expressions.
8#[doc = crate::_doc_meta!{location("code/util")}]
9///
10/// Similar to [`assert_eq`] but supports more than 2 terms to test for equality.
11///
12/// # Panics
13// TODO
14#[macro_export]
15#[cfg_attr(cargo_primary_package, doc(hidden))]
16macro_rules! assert_eq_all {
17    ($first:expr, $($rest:expr),+ $(,)?) => {{
18        let first_val = &$first;
19        $(
20            let rest_val = &$rest;
21            assert!(
22                first_val == rest_val,
23                "Assertion failed: ({}) is not equal to ({})\n  left: {}\n right: {}",
24                stringify!($first),
25                stringify!($rest),
26                first_val,
27                rest_val,
28            );
29        )+
30    }};
31}
32#[doc(inline)]
33pub use assert_eq_all;

35#[doc = crate::_tags!(assert)]
36/// Asserts the approximate equality of a series of expressions within `tolerance`.
37#[doc = crate::_doc_meta!{location("code/util")}]
38///
39/// This macro should work with any numeric type that supports comparison and
40/// subtraction, including signed and unsigned integers and floating-point numbers.
41///
42/// The given `$tolerance` must be a non-negative number.
43///
44/// # Panics
45// TODO
46///
47/// # Examples
48/// The following examples compile:
49/// ```
50/// # use devela::assert_approx_eq_all;
51/// assert_approx_eq_all![tolerance: 0.01, 1.0, 1.001, 0.999]; // up to 0.01 away from 1.0
52/// assert_approx_eq_all![tolerance: 1_u32, 4, 3, 5]; // up to 1 away from 4
53///
54/// assert_approx_eq_all![tolerance: 0.01, -1.0, -1.001, -0.999];
55/// assert_approx_eq_all![tolerance: 1_i32, -4, -3, -5];
56/// assert_approx_eq_all![tolerance: 0_i32, 3, 3];
57/// ```
58/// The following examples panic:
59/// ```should_panic
60/// # use devela::assert_approx_eq_all;
61/// assert_approx_eq_all![tolerance: 0.01, 1.0, 1.001, 0.989]; // |0.989 - 1.0| > |0.01|
62/// ```
63/// ```should_panic
64/// # use devela::assert_approx_eq_all;
65/// assert_approx_eq_all![tolerance: 1_u32, 4, 3, 5, 6]; // |6 - 4| > |1|
66/// ```
67/// ```should_panic
68/// # use devela::assert_approx_eq_all;
69/// assert_approx_eq_all![tolerance: 1_u32, 3, 4, 5]; // |5 - 3| > |1|
70/// ```
71/// ```should_panic
72/// # use devela::assert_approx_eq_all;
73/// assert_approx_eq_all![tolerance: -0.01, 1.0, 1.001, 0.999]; // tolerance: -0.01 < 0
74/// ```
75/// ```should_panic
76/// # use devela::assert_approx_eq_all;
77/// assert_approx_eq_all![tolerance: -1_i32, 4, 3, 5]; // tolerance: -1 < 0
78/// ```
79#[macro_export]
80#[cfg_attr(cargo_primary_package, doc(hidden))]
81macro_rules! assert_approx_eq_all {
82    (tolerance: $tolerance:expr, $first:expr, $($rest:expr),+ $(,)?) => {{
83        let first_val = $first;
84        $(
85            let rest_val = $rest;
86            // WIP_: handle infinities first
87
88            // Calculate the absolute difference without relying on `abs`:
89            let difference =
90                if first_val > rest_val { first_val - rest_val } else { rest_val - first_val };
91            assert!(
92                difference <= $tolerance,
93                "Assertion failed: ({}) is not approximately equal to ({})
94      left: {}\n     right: {}\n tolerance: {}",
95                stringify!($first), stringify!($rest), first_val, rest_val, $tolerance,
96            );
97        )+
98    }};
99}
100#[doc(inline)]
101pub use assert_approx_eq_all;
