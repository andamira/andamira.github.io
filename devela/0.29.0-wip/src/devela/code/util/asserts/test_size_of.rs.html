4/// Tests the stack size of a type.
5#[doc = crate::_doc_meta!{location("code/util")}]
6///
7/// Optionally checks the matching bit count
8/// and whether `Option<T>` keeps the same stack size as `T`.
9///
10/// This is a convenience macro around [`core::mem::size_of`] for probing,
11/// checking, or locking the expected size of a type.
12///
13/// It supports both panicking test-oriented modes and caller-buffer modes for
14/// fallible diagnostics without allocation.
15///
16/// # Modes
17/// - `assert T = N`: asserts the byte-size inside an existing test.
18/// - `assert T = N|B`: also checks the corresponding bit count.
19/// - `assert T = N; niche Option`: also checks `Option<T>` has the same size.
20/// - `assert { T = N; ... }`: asserts several type sizes.
21/// - `check_into buf; T = N`: returns `(ok, type_name, message)`.
22/// - `line_into buf; T = N`: returns `(ok, line)` ready to print.
23/// - `probe T`: panics with the measured size.
24/// - `name: T = N`: generates a named unit test.
25/// - `T = N`: generates `test_size_of_T` for simple identifier types.
26///
27/// The `; niche Option` suffix checks size preservation:
28/// `size_of::<Option<T>>() == size_of::<T>()`.
29///
30/// # Buffer modes
31/// The `check_into` and `line_into` modes write diagnostics into caller-provided
32/// byte storage and never panic for size mismatches.
33///
34/// `check_into` returns:
35/// - `true`, the type name, and an empty message on success.
36/// - `false`, the type name, and a diagnostic message on failure.
37///
38/// `line_into` returns:
39/// - `true` and a printable success line on success.
40/// - `false` and the diagnostic message on failure.
41///
42/// Messages are truncated to fit the provided buffer.
43///
44/// # Panics
45/// The `assert` modes panic when the measured byte-size does not match the
46/// expected value.
47///
48/// They panic when an explicit bit count does not match the expected byte-size
49/// converted to bits.
50///
51/// They panic when `; niche Option` is used and `Option<T>` does not have the
52/// same stack size as `T`.
53///
54/// The `probe` mode always panics, intentionally, and prints the measured size.
55///
56/// # Examples
57/// Assertion checks:
58/// ```
59/// # use devela::test_size_of;
60/// test_size_of![assert u8 = 1];
61/// test_size_of![assert u16 = 2|16];
62/// test_size_of![assert [u8; 4] = 4];
63/// test_size_of![assert core::num::NonZeroU8 = 1; niche Option];
64///
65/// test_size_of![assert {
66///     u16 = 2;
67///     u32 = 4|32;
68///     [u8; 8] = 8;
69///     core::num::NonZeroU8 = 1; niche Option;
70/// }];
71/// ```
72///
73/// Fallible checks:
74/// ```
75/// # use devela::test_size_of;
76/// let mut buf = [0; 128];
77///
78/// let (ok, ty, msg) = test_size_of![check_into &mut buf; u16 = 2|16];
79/// assert!(ok);
80/// assert_eq!(ty, "u16");
81/// assert_eq!(msg, "");
82///
83/// let (ok, ty, msg) = test_size_of![check_into &mut buf; u16 = 1];
84/// assert!(!ok);
85/// assert_eq!(ty, "u16");
86/// assert!(msg.contains("size_of::<u16>() byte mismatch"));
87/// ```
88///
89/// Printable checks:
90/// ```
91/// # use devela::test_size_of;
92/// let mut buf = [0; 128];
93///
94/// let (ok, line) = test_size_of![line_into &mut buf; u16 = 2|16];
95/// assert!(ok);
96/// assert_eq!(line, "ok: size_of::<u16>() = 2 bytes (16 bits)");
97/// ```
98///
99/// Generated tests:
100/// ```
101/// # use devela::test_size_of;
102/// test_size_of![u32 = 4];
103/// test_size_of![core_num_nonzero_u8: core::num::NonZeroU8 = 1; niche Option];
104/// test_size_of![option_nonzero_u8: Option<core::num::NonZeroU8> = 1];
105/// ```
106#[macro_export]
107#[cfg_attr(cargo_primary_package, doc(hidden))]
108macro_rules! test_size_of {
109    // Panicking check for use inside tests.
110    (assert $ty:ty = $bytes:literal $(| $bits:literal)? $(; niche $wrap:ident)? $(,)?) => {{
111        let (actual, expected): (usize, usize) = (::core::mem::size_of::<$ty>(), $bytes);
112        if actual != expected {
113            ::core::panic!(
114    "size_of::<{}>() mismatch:\n  actual:   {} bytes ({} bits)\n  expected: {} bytes ({} bits)",
115                ::core::any::type_name::<$ty>(),
116                actual, actual.saturating_mul(8),
117                expected, expected.saturating_mul(8),
118            );
119        }
120        $(
121            let (actual_bits, expected_bits): (usize, usize) = (expected.saturating_mul(8), $bits);
122            if actual_bits != expected_bits {
123                ::core::panic!(
124    "size_of::<{}>() bit mismatch:\n  bytes:    {} bytes\n  actual:   {} bits\n  expected: {} bits",
125                    ::core::any::type_name::<$ty>(),
126                    expected, actual_bits, expected_bits,
127                );
128            }
129        )?
130        $( $crate::test_size_of![%assert_niche $ty, $wrap]; )?
131    }};
132    // Panicking batch check.
133    (assert { $($rest:tt)* }) => {{
134        $crate::test_size_of![%assert_batch $($rest)*];
135    }};
136    // Fallible buffer check; returns `(ok, type_name, message)`.
137    (check_into $buf:expr; $ty:ty = $bytes:literal $(| $bits:literal)? $(; niche $wrap:ident)?
138    $(,)?) => {{
139        let __buf: &mut [u8] = $buf;
140        let __ty: &'static str = ::core::any::type_name::<$ty>();
141        let (__actual, __expected): (usize, usize) = (::core::mem::size_of::<$ty>(), $bytes);
142        if __actual != __expected {
143            (false, __ty, $crate::__test_size_of_report(__buf, __ty, "byte", __actual, __expected))
144        } else {
145            let (__ok, __msg) =
146                $crate::test_size_of![%check_into_bits __buf; $ty, __expected $(, $bits)?];
147            if !__ok { (__ok, __ty, __msg) } else {
148                let (__ok, __msg) = $crate::test_size_of![%check_into_niche __buf; $ty $(, $wrap)?];
149                if !__ok { (__ok, __ty, __msg) } else { (true, __ty, "") }
150            }
151        }
152    }};
153    // Printable buffer check; returns `(ok, line)`.
154    (line_into $buf:expr; $ty:ty = $bytes:literal $(| $bits:literal)? $(; niche $wrap:ident)?
155    $(,)?) => {{
156        let __buf: &mut [u8] = $buf;
157        let __bytes: usize = $bytes;
158        let (__ok, __ty, __msg) =
159            $crate::test_size_of![check_into __buf; $ty = $bytes $(| $bits)? $(; niche $wrap)?];
160        if __ok { (true, $crate::format_buf![? __buf,
161            "ok: size_of::<{}>() = {} bytes ({} bits)", __ty, __bytes, __bytes.saturating_mul(8)])
162        } else { (false, __msg) }
163    }};
164    // Probe mode; always panics with the measured size.
165    (probe $ty:ty $(,)?) => {{
166        let actual: usize = ::core::mem::size_of::<$ty>();
167        ::core::panic!("size_of::<{}>() = {} bytes ({} bits)",
168            ::core::any::type_name::<$ty>(), actual, actual.saturating_mul(8),
169        );
170    }};
171    // Defines a named unit test for an expected size.
172    ($name:ident : $ty:ty = $bytes:literal $(| $bits:literal)? $(; niche $wrap:ident)? $(,)?) => {
173        #[cfg(test)] #[test] #[allow(nonstandard_style)]
174        fn $name() { $crate::test_size_of![assert $ty = $bytes $(| $bits)? $(; niche $wrap)?]; }
175    };
176    // Defines a named unit test that probes the measured size.
177    ($name:ident : probe $ty:ty $(,)?) => {
178        #[cfg(test)] #[test] #[allow(nonstandard_style)]
179        fn $name() { $crate::test_size_of![probe $ty]; }
180    };
181    // Defines `test_size_of_T` for simple identifier types.
182    ($ty:ident = $bytes:literal $(| $bits:literal)? $(; niche $wrap:ident)? $(,)?
183        ) => { $crate::paste! {
184        $crate::test_size_of!([<test_size_of_ $ty>] : $ty = $bytes $(| $bits)? $(; niche $wrap)?);
185    }};
186    // Defines `test_size_of_T` probe for simple identifier types.
187    (probe $ty:ident $(,)?) => { $crate::paste! {
188        $crate::test_size_of!([<test_size_of_ $ty>] : probe $ty);
189    }};
190    /* private arms*/
191    // Ends batch parsing.
192    (%assert_batch) => {};
193    // Parses one batch entry with a niche check.
194    (%assert_batch
195        $ty:ty = $bytes:literal $(| $bits:literal)? ; niche $wrap:ident ; $($rest:tt)*) => {{
196        $crate::test_size_of![assert $ty = $bytes $(| $bits)?; niche $wrap];
197        $crate::test_size_of![%assert_batch $($rest)*];
198    }};
199    // Parses one plain batch entry.
200    (%assert_batch $ty:ty = $bytes:literal $(| $bits:literal)? ; $($rest:tt)*) => {{
201        $crate::test_size_of![assert $ty = $bytes $(| $bits)?];
202        $crate::test_size_of![%assert_batch $($rest)*];
203    }};
204    // Panicking niche-size check.
205    (%assert_niche $ty:ty, Option) => {{
206        let (plain, wrapped): (usize, usize) =
207            (::core::mem::size_of::<$ty>(), ::core::mem::size_of::<Option<$ty>>());
208        if wrapped != plain {
209            ::core::panic!(
210    "Option niche size mismatch for `{}`:\n  size_of::<{}>()         = \
211{} bytes ({} bits)\n  size_of::<Option<{}>>() = {} bytes ({} bits)\n  expected: same stack size",
212                ::core::any::type_name::<$ty>(), ::core::any::type_name::<$ty>(),
213                plain, plain.saturating_mul(8), ::core::any::type_name::<$ty>(),
214                wrapped, wrapped.saturating_mul(8),
215            );
216        }
217    }};
218    // Rejects unsupported niche wrappers.
219    (%assert_niche $ty:ty, $wrap:ident) => {
220        ::core::compile_error!( concat!( "unsupported `test_size_of!` niche wrapper `",
221                stringify!($wrap), "`; currently supported: `Option`"));
222    };
223    // No bit check requested.
224    (%check_into_bits $buf:expr; $ty:ty, $bytes:expr) => {{ (true, "") }};
225    // Fallible bit-count check.
226    (%check_into_bits $buf:expr; $ty:ty, $bytes:expr, $bits:literal) => {{
227        let __actual_bits = $bytes.saturating_mul(8);
228        let __expected_bits: usize = $bits;
229        if __actual_bits != __expected_bits {
230            (false, $crate::__test_size_of_report(
231                $buf, ::core::any::type_name::<$ty>(), "bit", __actual_bits, __expected_bits))
232        } else { (true, "") }
233    }};
234    // No niche check requested.
235    (%check_into_niche $buf:expr; $ty:ty) => {{ (true, "") }};
236    // Fallible `Option<T>` niche-size check.
237    (%check_into_niche $buf:expr; $ty:ty, Option) => {{
238        let (__plain, __wrapped): (usize, usize) =
239            (::core::mem::size_of::<$ty>(), ::core::mem::size_of::<Option<$ty>>());
240        if __wrapped != __plain {
241            (false, $crate::format_buf![? $buf,
242                "Option niche size mismatch for `{}`:\n  size_of::<{}>()         = \
243{} bytes ({} bits)\n  size_of::<Option<{}>>() = {} bytes ({} bits)\n  expected: same stack size",
244                    ::core::any::type_name::<$ty>(), ::core::any::type_name::<$ty>(),
245                    __plain, __plain.saturating_mul(8), ::core::any::type_name::<$ty>(),
246                    __wrapped, __wrapped.saturating_mul(8)])
247        } else { (true, "") }
248    }};
249    // Rejects unsupported niche wrappers.
250    (%check_into_niche $buf:expr; $ty:ty, $wrap:ident) => {
251        ::core::compile_error!(concat!(
252            "unsupported `test_size_of!` niche wrapper `",
253            stringify!($wrap),
254            "`; currently supported: `Option`"
255        ));
256    };
257}

261/// Formats a byte/bit size mismatch into caller-provided storage.
262#[doc(hidden)] #[rustfmt::skip]
263pub fn __test_size_of_report<'a>(buf: &'a mut [u8],
264    ty: &str, kind: &str, actual: usize, expected: usize,) -> &'a str {
265    match kind {
266        "bit" => crate::format_buf![? buf,
267            "size_of::<{}>() bit mismatch:\n  actual:   {} bits\n  expected: {} bits",
268            ty, actual, expected],
269        _ => crate::format_buf![? buf,
270            "size_of::<{}>() byte mismatch:\n  actual:   {} bytes ({} bits)\n  expected: {} bytes ({} bits)",
271            ty, actual, actual.saturating_mul(8), expected, expected.saturating_mul(8)],
272    }
