6#[doc = crate::_tags!(code construction member set bit)]
7/// Defines an enum and an associated type set of its variants.
8#[doc = crate::_doc_meta!{location("code/util")}]
9///
10/// It uses the [`set!`][crate::set] macro to create the associated set.
11///
12/// You have to give unique names both to the enum and to the associated set.
13///
14/// One or more custom `impl enum { ... }` and `impl set { ... }` blocks
15/// may be provided after the variant block. They are emitted
16/// before all generated associated constants and methods,
17/// and may still refer to them through `Self`.
18///
19/// # Limitations
20/// - Deriving [`Default`] on the generated enum is not currently supported,
21///   and has to be manually implemented instead.
22/// - Outer attributes for custom impl blocks are placed
23///   after the `impl enum` or `impl set` marker.
24///
25/// # Examples
26/// See also [`EnumExample`][crate::EnumExample] and [`EnumSetExample`][crate::EnumSetExample],
27/// generated in the [enumset][crate::_doc::examples::enumset] example.
28///
29/// ```
30/// # use devela::enumset;
31/// enumset! {
32///     #[derive(Debug)]
33///     pub enum MyEnum(pub MyEnumSet: u8) {
34///         Variant1,
35///         Variant2(bool),
36///         Variant3{a: u8, b: u16},
37///     }
38///    impl enum
39///    /// Extra methods for [`MyEnum`].
40///    {
41///        /// Returns whether this variant belongs to [`MyEnumSet::DATA`].
42///        pub const fn is_data(&self) -> bool {
43///            self.is_in(MyEnumSet::DATA)
44///        }
45///    }
46///    impl set
47///    /// Extra constants for [`MyEnumSet`].
48///    {
49///        /// Variants carrying payload data.
50///        pub const DATA: Self = Self::Variant2.with(Self::Variant3);
51///    }
52/// }
53/// assert_eq![3, MyEnum::VARIANTS];
54/// let mut es = MyEnumSet::default();
55/// assert![es.is_empty()];
56/// es.insert(MyEnumSet::Variant1);
57/// assert![es.contains(MyEnumSet::Variant1)];
58/// ```
59#[doc(hidden)]
60#[macro_export]
61macro_rules! enumset· {
62    (
63        // $enum_attr: the attributes of the enum.
64        // $enum_vis:  the visibility of the enum.
65        // $enum_name: the name of the new enum.
66        // $set_attr:  the attributes of the set.
67        // $set_vis:   the visibility of the set.
68        // $set_name:  the name of the associated set.
69        // $set_ty:    the inner unsigned integer primitive for the bit set (u8, u16, u32…).
70        $( #[$enum_attr:meta] )*
71        $enum_vis:vis enum $enum_name:ident
72            $( < $($gen:tt),* $(,)? > )? // optional generics and lifetimes
73            // attributes, visibility, name and inner type of the set, between ():
74            ( $( #[$set_attr:meta] )* $set_vis:vis $set_name:ident: $set_ty:ty )
75            $([where $($where:tt)+ $(,)? ] $(,)? )? // optional where clauses, between []
76        {
77            $(
78                $( #[$variant_attr:meta] )*
79                $variant_name:ident
80                $(( $($tuple_type:ty),* $(,)? ))?
81                $({ $( $( #[$field_attr:meta] )* $field_name:ident : $field_type:ty),* $(,)? })?
82                $(= $discriminant:expr)?
83                $(,)?
84            )*
85        }
86        $( // optional impl blocks for the enum
87            impl enum $(#[$impl_enum_attrs:meta])* { $($impl_enum:item)* }
88        )*
89        $( // optional impl blocks for the set
90            impl set $(#[$impl_set_attrs:meta])* { $($impl_set:item)* }
91        )*
92    ) => {
93        /* define enum */
94
95        $( #[$enum_attr] )*
96        // #[doc = "\n\nSee also the associated type set of variants [`" $set_name "`]."]
97        #[allow(missing_debug_implementations, reason = "custom impl")]
98        $enum_vis enum $enum_name $( < $($gen),* > )? $(where $($where)+)? {
99            $(
100                $( #[$variant_attr] )*
101                $variant_name
102                $(( $($tuple_type),* ))?
103                $({ $( $( #[$field_attr] )* $field_name : $field_type),* })?
104                $(= $discriminant)?
105            ),*
106        }
107        /* impl enum methods */
108        $crate::__enumset_impl_enum_blocks! {
109            [$( < $($gen),* > )?] [$enum_name] [$( < $($gen),* > )?] [$( where $($where)+ )?]
110            $(
111                [$( #[$impl_enum_attrs] )*]
112                { $($impl_enum)* }
113            )*
114        }
115
116        /// # Associated set methods
117        #[allow(dead_code)]
118        impl $( < $($gen),* > )? $enum_name $( < $($gen),* > )? $( where $($where)* )? {
119            /// The total number of *declared* variants.
120            $enum_vis const VARIANTS: usize = $crate::paste! { [<_$enum_name _private>]::VARIANTS };
121
122            /// Returns the total number of *declared* variants.
123            #[must_use]
124            $enum_vis const fn variants(&self) -> usize { Self::VARIANTS }
125            /// Returns the associated empty set.
126            #[must_use]
127            $enum_vis const fn empty_set() -> $set_name { $set_name::new() }
128            /// Returns the associated full set.
129            #[must_use]
130            $enum_vis const fn full_set() -> $set_name { $set_name::all() }
131            /// Returns the singleton set for this variant.
132            #[must_use]
133            $enum_vis const fn to_set(&self) -> $set_name {
134                #[allow(unused_doc_comments, reason = "attributes could be comments")]
135                match self {
136                    $(
137                        $( #[$variant_attr] )*
138                        $crate::__enumset_to_set_pattern!(
139                            $enum_name, $variant_name
140                            $(( $($tuple_type),* ))?
141                            $({ $( $(#[$field_attr])* $field_name : $field_type),* })?
142                        ) => $set_name::$variant_name,
143                    )*
144                }
145            }
146            /// Returns whether this variant belongs to `set`.
147            #[must_use]
148            $enum_vis const fn is_in(&self, set: $set_name) -> bool {
149                set.contains(self.to_set())
150            }
151        }
152
153        /* define the associated bit set */
154        $crate::paste! {
155            #[allow(non_snake_case)]
156            mod [<_$enum_name _private>] {
157                pub(super) const VARIANTS: usize = $crate::ident_total!($($variant_name),*);
158                $crate::ident_const_index!(pub(super), VARIANTS; $($variant_name),*);
159            }
160            $crate::data::set! {
161                $( #[$set_attr] )*
162                $set_vis struct $set_name($set_ty) {
163                    $(
164                        #[doc = "The bit index that corresponds to `" $enum_name "::`[`"
165                            $variant_name "`][" $enum_name "::" $variant_name "]."]
166                        #[allow(non_upper_case_globals)]
167                        $variant_name = ([<_$enum_name _private>]::$variant_name);
168                    )*
169                }
170                $( // optional set impl blocks
171                    $( #[$impl_set_attrs] )*
172                    impl { $($impl_set)* }
173                )*
174            }
175        }
176    };
177}
178#[doc(inline)]
179pub use enumset· as enumset;
180
181#[doc(hidden)]
182#[macro_export]
183macro_rules! __enumset_to_set_pattern {
184    ($enum_name:ident, $variant_name:ident) => {
185        $enum_name::$variant_name // Unit variant
186    };
187    ($enum_name:ident, $variant_name:ident ( $($field:tt)*) ) => {
188        $enum_name::$variant_name(..) // Tuple variant
189    };
190    ($enum_name:ident, $variant_name:ident { $($field:tt)*} ) => {
191        $enum_name::$variant_name { .. } // Struct variant
192    };
193}
194#[doc(hidden)]
195#[macro_export]
196macro_rules! __enumset_impl_enum_blocks {
197    ( // End
198        [$($impl_gen:tt)*][$enum_name:ident][$($ty_gen:tt)*][$($where:tt)*] ) => {};
199    ( // Emit one impl block, then recurse
200        [$($impl_gen:tt)*][$enum_name:ident][$($ty_gen:tt)*][$($where:tt)*]
201        [$($impl_attrs:tt)*] { $($impl_item:item)* } $($rest:tt)*
202    ) => {
203        $($impl_attrs)*
204        impl $($impl_gen)* $enum_name $($ty_gen)* $($where)* { $($impl_item)* }
205        $crate::__enumset_impl_enum_blocks! {
206            [$($impl_gen)*] [$enum_name] [$($ty_gen)*] [$($where)*] $($rest)*
207        }
208    };

211#[cfg(test)]
212mod tests {
213    #![allow(unused)]
214
215    crate::enumset! {
216        enum TestEnum(pub TestEnumSet: u8) {
217            A,
218            B(bool),
219            C { x: u8 },
220        }
221    }
222    #[test]
223    fn enumset_uses_set_macro() {
224        assert_eq!(TestEnum::VARIANTS, 3);
225
226        assert_eq!(TestEnumSet::A.bits(), 0b001);
227        assert_eq!(TestEnumSet::B.bits(), 0b010);
228        assert_eq!(TestEnumSet::C.bits(), 0b100);
229
230        assert_eq!(TestEnum::empty_set().bits(), 0);
231        assert_eq!(TestEnum::full_set().bits(), 0b111);
232
233        let ab = TestEnumSet::A.union(TestEnumSet::B);
234        assert!(ab.contains(TestEnumSet::A));
235        assert!(ab.contains(TestEnumSet::B));
236        assert!(!ab.contains(TestEnumSet::C));
237
238        let mut s = TestEnumSet::A;
239
240        s.insert(TestEnumSet::B);
241        assert_eq!(s.bits(), 0b011);
242
243        s.remove(TestEnumSet::A);
244        assert_eq!(s.bits(), 0b010);
245
246        s.toggle(TestEnumSet::C);
247        assert_eq!(s.bits(), 0b110);
248
249        s.clear();
250        assert_eq!(s.bits(), 0);
251        assert!(s.is_empty());
252
253        assert_eq!((TestEnumSet::A | TestEnumSet::B).bits(), 0b011);
254        assert_eq!((TestEnumSet::A | TestEnumSet::B) & TestEnumSet::B, TestEnumSet::B);
255        assert_eq!((TestEnumSet::A | TestEnumSet::B) - TestEnumSet::A, TestEnumSet::B);
256        assert_eq!((TestEnumSet::A | TestEnumSet::B) ^ TestEnumSet::B, TestEnumSet::A);
257        assert_eq!((!TestEnumSet::A).bits(), 0b110);
258    }
259    #[test]
260    fn enumset_is_in_set() {
261        let a = TestEnum::A;
262        let b = TestEnum::B(true);
263        let c = TestEnum::C { x: 7 };
264        assert_eq!(a.to_set(), TestEnumSet::A);
265        assert_eq!(b.to_set(), TestEnumSet::B);
266        assert_eq!(c.to_set(), TestEnumSet::C);
267        assert!(a.is_in(TestEnumSet::A));
268        assert!(b.is_in(TestEnumSet::A | TestEnumSet::B));
269        assert!(c.is_in(TestEnumSet::C));
270        assert!(!c.is_in(TestEnumSet::A | TestEnumSet::B));
271    }
272    #[test]
273    fn enumset_generics() {
274        crate::enumset! {
275            enum GenEnum<'a, T>(pub GenEnumSet: u8) {
276                A(T),
277                B(&'a str),
278            }
279            impl enum {
280                pub fn is_a(&self) -> bool {
281                    self.is_in(GenEnumSet::A)
282                }
283            }
284        }
285        let a = GenEnum::A(());
286        assert!(a.is_a());
287    }
