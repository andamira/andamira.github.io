6use crate::{Display, FmtResult, Formatter, Slice};

8#[doc = crate::_tags!(code)]
9/// A precise location in the source code.
10#[doc = crate::_doc_meta!{location("code")}]
11///
12/// Captures the module path, file name, line, and column of a specific
13/// invocation site. Intended as a lightweight, zero-cost provenance
14/// primitive usable across diagnostics, logging, errors, and profiling.
15///
16/// Instances are typically created via [`CodeLocation::here`], which
17/// records the current call site using compile-time macros.
18#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
19pub struct CodeLocation {
20    /// The module path.
21    pub module: &'static str,
22    /// The file name.
23    pub file: &'static str,
24    /// The line number.
25    pub line: u32,
26    /// The column number.
27    pub column: u32,
28}
29
30impl Display for CodeLocation {
31    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
32        write!(f, "{} ({}:{}:{})", self.module, self.file, self.line, self.column)
33    }
34}
35
36impl CodeLocation {
37    /* constructors */
38
39    /// Returns the code location of the current invocation site.
40    #[inline(always)]
41    pub const fn here() -> Self {
42        CodeLocation {
43            module: crate::code_module!(),
44            file: crate::code_file!(),
45            line: crate::code_line!(),
46            column: crate::code_column!(),
47        }
48    }
49
50    /* deconstructors */
51
52    /// Returns the file name and line number.
53    pub const fn file_line(&self) -> (&'static str, u32) {
54        (self.file, self.line)
55    }
56
57    /// Returns the file name, and line and column numbers.
58    pub const fn file_line_column(&self) -> (&'static str, u32, u32) {
59        (self.file, self.line, self.column)
60    }
61
62    /* */
63
64    /// Compile-time equality comparison.
65    pub const fn eq(&self, other: &Self) -> bool {
66        self.line == other.line
67            && self.column == other.column
68            && Slice::<&str>::eq(self.file, other.file)
69            && Slice::<&str>::eq(self.module, other.module)
70    }
71
72    /// Short display helper.
73    pub fn fmt_short(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
74        write!(f, "{}:{}:{}", self.file, self.line, self.column)
75    }
76}

78#[doc = crate::_tags!(code)]
79/// A contiguous span between two code locations.
80#[doc = crate::_doc_meta!{location("code")}]
81///
82/// Represents a range in the source code, typically describing where a construct,
83/// operation, or effect originates. The span is inclusive of both endpoints
84/// and carries no semantic meaning beyond positional ordering.
85///
86/// `CodeSpan` is purely structural and does not assume ordering across different files or modules.
87#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
88pub struct CodeSpan {
89    /// The start code location.
90    pub start: CodeLocation,
91    /// The end code location.
92    pub end: CodeLocation,
93}
94
95impl CodeSpan {
96    /// Creates a span from two code locations.
97    pub const fn new(start: CodeLocation, end: CodeLocation) -> Self {
98        Self { start, end }
99    }
100
101    /// Creates a zero-length span at the current invocation site.
102    pub const fn here() -> Self {
103        let loc = CodeLocation::here();
104        Self { start: loc, end: loc }
105    }
106
107    /// Returns true if the span represents a single point.
108    pub const fn is_point(&self) -> bool {
109        self.start.eq(&self.end)
110    }
111}
112
113impl Display for CodeSpan {
114    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
115        if self.is_point() {
116            write!(f, "{}", self.start)
117        } else {
118            write!(
119                f,
120                "{}:{}:{} – {}:{}:{}",
121                self.start.file,
122                self.start.line,
123                self.start.column,
124                self.end.line,
125                self.end.column,
126                self.end.module
127            )
128        }
129    }
130}
