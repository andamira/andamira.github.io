19/// A trait for giving a type a useful const-friendly initial value.
20#[doc = crate::_doc_meta!{location("code")}]
21///
22/// # Comparison with `Default`
23///
24/// `Default` represents the type's natural baseline, usually a neutral or zero-like state.
25/// `ConstInit` simply provides a valid const-time initializer, without requiring that meaning.
26///
27/// When a type has a clear `Default`, these traits should return the same value to keep behavior
28/// consistent. They only diverge for types that cannot offer a safe or meaningful `Default`, but
29/// still need a guaranteed const initializer.
30///
31/// - Use `Default` for semantic baselines.
32/// - Use `ConstInit` for invariant-safe const initialization.
33pub trait ConstInit {
34    /// Returns the compile-time "initial value" for a type.
35    const INIT: Self;
36}

39/// A macro helper to implement [`ConstInit`]. Supports generics.
40macro_rules! _impl_init {
41    // <A> (e.g: pointers and reference)
42    (<$A:ident> $def:expr => $($t:ty),+) => {
43        $( $crate::_impl_init![@<$A> $def => $t]; )+
44    };
45    (@<$A:ident> $def:expr => $t:ty) => {
46        impl<$A> $crate::ConstInit for $t {
47            #[allow(clippy::declare_interior_mutable_const)]
48            const INIT: Self = $def;
49        }
50    };
51
52    // <A: A_> (bounded) (e.g. Option, Cell, Range…)
53    (<$A:ident:$A_:ident> $def:expr => $($t:ty),+) => {
54        $( $crate::_impl_init![@<$A:$A_> $def => $t]; )+
55    };
56    (@<$A:ident:$A_:ident> $def:expr => $t:ty) => {
57        impl<$A: $crate::ConstInit> $crate::ConstInit for $t {
58            #[allow(clippy::declare_interior_mutable_const, reason = "FIXME?")]
59            const INIT: Self = $def;
60        }
61    };
62
63    // <A, B>
64    (<$A:ident, $B:ident> $def:expr => $($t:ty),+) => {
65        $( $crate::_impl_init![@<$A, $B> $def => $t]; )+
66    };
67    (@<$A:ident, $B:ident> $def:expr => $t:ty) => {
68        impl<$A, $B> $crate::ConstInit for $t {
69            #[allow(clippy::declare_interior_mutable_const)] //
70            const INIT: Self = $def;
71        }
72    };
73    // <A: A_, B: B_> (bounded) (e.g. CycleCount)
74    (<$A:ident:$A_:ident, $B:ident:$B_:ident> $def:expr => $($t:ty),+) => {
75        $( $crate::_impl_init![@<$A:$A_, $B:$B_> $def => $t]; )+ };
76    (@<$A:ident:$A_:ident, $B:ident:$B_:ident> $def:expr => $t:ty) => {
77        impl<$A:$A_, $B:$B_> $crate::ConstInit for $t {
78            #[allow(clippy::declare_interior_mutable_const)] //
79            const INIT: Self = $def;
80        }
81    };
82
83    // <A, B, C>
84    (<$A:ident, $B:ident, $C:ident> $def:expr => $($t:ty),+) => {
85        $( $crate::_impl_init![@<$A, $B, $C> $def => $t]; )+
86    };
87    (@<$A:ident, $B:ident, $C:ident> $def:expr => $t:ty) => {
88        impl<$A, $B, $C> $crate::ConstInit for $t {
89            #[allow(clippy::declare_interior_mutable_const)] //
90            const INIT: Self = $def;
91        }
92    };
93    // <> (e.g.: bool, char, integers, floats, NonZero…) (supports attributes)
94    ($def:expr => $( $(#[$attr:meta])* $t:ty ),+ $(,)?) => {
95        $( $crate::_impl_init![@$def => $(#[$attr])* $t]; )+
96    };
97    (@$def:expr => $(#[$attr:meta])* $t:ty) => {
98        $(#[$attr])*
99        impl $crate::ConstInit for $t {
100            #[allow(clippy::declare_interior_mutable_const)]
101            const INIT: Self = $def;
102        }
103    };
104
105    // impl for arrays of the given $LEN lenghts
106    (arrays <$A:ident:$BOUND:ident> $($LEN:literal),+) => {
107        $( $crate::_impl_init![@array:$LEN <$A:$BOUND>]; )+
108    };
109    (@array:$LEN:literal <$A:ident:$BOUND:ident>) => {
110        impl<$A: $crate::ConstInit> $crate::ConstInit for [$A; $LEN] {
111            #[allow(clippy::declare_interior_mutable_const)] //
112            const INIT: Self = [$A::INIT; $LEN];
113        }
114    };
115
116    // impl for tuples of lenghts from 1 to 12
117    (tuples <$A:ident:$BOUND:ident>) => {
118        $crate::_impl_init![@tuple <$A:$BOUND> ($A,) => // 1
119            ($A::INIT,)];
120        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,) => // 2
121            ($A::INIT, $A::INIT)];
122        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A) => // 3
123            ($A::INIT, $A::INIT, $A::INIT)];
124        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A) => // 4
125            ($A::INIT, $A::INIT, $A::INIT, $A::INIT)];
126        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A) => // 5
127            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT)];
128        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A) => // 6
129            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT)];
130        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A,$A) => // 7
131            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT,
132             $A::INIT)];
133        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A,$A,$A) => // 8
134            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT,
135             $A::INIT, $A::INIT)];
136        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A,$A,$A,$A) => // 9
137            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT,
138             $A::INIT, $A::INIT, $A::INIT)];
139        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A,$A,$A,$A,$A) => // 10
140            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT,
141             $A::INIT, $A::INIT, $A::INIT, $A::INIT)];
142        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A,$A,$A,$A,$A,$A) => // 11
143            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT,
144             $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT)];
145        $crate::_impl_init![@tuple <$A:$BOUND> ($A,$A,$A,$A,$A,$A,$A,$A,$A,$A,$A,$A) => // 12
146            ($A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT,
147             $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT, $A::INIT)];
148    };
149    (@tuple <$A:ident:$BOUND:ident> $type:ty => $value:expr) => {
150        impl<$A: $crate::ConstInit> $crate::ConstInit for $type {
151            const INIT: Self = $value;
152        }
153    };
154}
