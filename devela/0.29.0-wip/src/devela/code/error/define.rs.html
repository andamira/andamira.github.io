14#[doc = crate::_tags!(construction error)]
15/// Defines individual and composite error types.
16#[doc = crate::_doc_meta!{location("code/error")}]
17///
18/// It can also implement `From` and `TryFrom` traits between them.
19///
20/** # Examples
21```
22# use devela::define_error;
23// Define simple individual error types
24define_error! {
25    individual:
26    /// A basic error with a code
27    pub struct ErrorCode(pub u32);
28    DOC_ERROR_CODE = "An error with a numeric code",
29    self + f => write!(f, "Error code: {}", self.0)
30}
31define_error! {
32    individual:
33    /// An error with message details
34    pub struct ErrorDetails {
35        pub code: u32,
36        pub message: &'static str,
37    }
38    DOC_ERROR_DETAILS = "An error with code and message details",
39    self + f => write!(f, "Error {}: {}", self.code, self.message)
40}
41
42// Define a composite error
43define_error! {
44    composite: fmt(f)
45    /// Container for multiple error types
46    pub enum CompositeError {
47        DOC_ERROR_CODE: +const CodeVariant(code|0: u32) => ErrorCode(*code),
48        DOC_ERROR_DETAILS: // ← this variant won't have const conversion
49            DetailsVariant { code: u32, message: &'static str } =>
50                ErrorDetails { code: *code, message: *message }
51    }
52}
53
54// Create individual errors
55let code_error = ErrorCode(42);
56let details_error = ErrorDetails {
57    code: 100,
58    message: "Example error",
59};
60
61// Automatic conversion to composite error
62let composite_from_code: CompositeError = code_error.into();
63let composite_from_details: CompositeError = details_error.into();
64
65// Try to convert back (only works for matching variants)
66match ErrorCode::try_from(composite_from_code) {
67    Ok(original) => println!("Successfully converted back: {}", original),
68    Err(_) => println!("Conversion failed - wrong variant"),
69}
70
71// Const conversion available for +const variants
72const CONST_ERROR: CompositeError = CompositeError::from_error_code(ErrorCode(99));
73```

75// NOTES:
76// - alternative sections for tuple-struct and field-struct variants are indicated in the margin.
77// - we are employing the trick `$(;$($_a:lifetime)?)?` for the optional semicolon terminator,
78//   where the never expected lifetime allows to refer to the non-identifier `;` later on;
79//   the same with `$(+const$($_c:lifetime)?)?` for the optional const fn implementation.
80#[macro_export]
81#[cfg_attr(cargo_primary_package, doc(hidden))]
82macro_rules! define_error· {
83    (
84    // Defines a standalone error struct with optional tuple or field variants.
85    //
86    // # Arguments
87    // - `first_attributes` :   Optional attributes applied first
88    // - `struct_vis`       :   Struct visibility (e.g., `pub`, `pub(crate)`)
89    // - `struct_name`      :   Name of the error struct
90    // - `last_attributes`  :   Optional attributes applied last
91    //
92    // ## Struct Body (choose one):
93    // - Unit variant  :  No body (just struct name)
94    // - Tuple variant : `(vis1 Type1, vis2 Type2, ...)`
95    // - Field variant : `{ #[attr] vis1 field1: Type1, #[attr] vis2 field2: Type2, ... }`
96    //
97    // ## Documentation:
98    // - `+tag: "tag text",`              : Optional additional doc tag
99    // - `DOC_CONST_NAME = "doc string",` : Constant name and documentation string
100    //
101    // ## Display Implementation:
102    // - `self_var + fmt_var => expression`: Display implementation where:
103    //   - `self_var`   : Name for self reference in display method
104    //   - `fmt_var`    : Name for formatter in display method
105    //   - `expression` : Expression writing to formatter
106    individual:
107        $(#[$first_attributes:meta])*
108        $struct_vis:vis struct $struct_name:ident
109        $(( $($e_vis:vis $e_ty:ty),+ $(,)? ))? $(;$($_a:lifetime)?)?              // tuple-struct↓
110        $({ $($(#[$f_attr:meta])* $f_vis:vis $f_name:ident: $f_ty:ty),+ $(,)? })? // field-struct↑
111        $($(#[$last_attributes:meta])*,)?
112        $(+location: $($location:literal)+ ,)?
113        $(+tag: $($tag:expr)+ ,)?
114        $DOC_NAME:ident = $doc_str:literal,
115        $self:ident + $fmt:ident => $display_expr:expr
116        $(,)?
117    ) => {
118        // IMPROVE: make it possibe to share publicly (conditional compilation, macro_export arms)
119        $crate::CONST! { pub(crate) $DOC_NAME = $doc_str; }
120
121        $( $(#[doc = $tag])+ )?
122        #[doc = $crate::_tags!(error)] // IMPROVE: make optional
123        #[doc = $DOC_NAME!()]
124        $(#[doc = $crate::_doc_location![$($location)?]])? // canonical location
125        $(#[$first_attributes])*
126        $($(#[$last_attributes])*)?
127        #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
128        $struct_vis struct $struct_name
129            $(( $($e_vis $e_ty),+ ) )? $(; $($_a)?)?                              // tuple-struct↓
130        $({ $( $(#[$f_attr])* $f_vis $f_name: $f_ty),+ })?                        // field-struct↑
131
132        impl $crate::Error for $struct_name {}
133        impl $crate::Display for $struct_name {
134            fn fmt(&$self, $fmt: &mut $crate::Formatter<'_>) -> $crate::FmtResult<()> {
135                $display_expr
136            }
137        }
138    };
139    (
140    // Defines a composite error enum that aggregates multiple individual error types.
141    //
142    // # Arguments
143    // - `fmt($fmt:ident)`      : Display formatter variable name for the Display implementation
144    // - `enum_attr`            : Optional attributes applied to the enum declaration
145    // - `vis`                  : Enum visibility (e.g., `pub`, `pub(crate)`)
146    // - `composite_error_name` : Name of the composite error enum
147    //
148    // ## Variant Definitions (one or more):
149    // For each variant:
150    // - `+tag: "tag text",` : Optional additional doc tag for the variant
151    // - `variant_attr`      : Optional attributes for the variant
152    // - `DOC_VARIANT`       : Constant name for variant documentation
153    // - `+const`            : Optional flag to generate const conversion methods
154    // - `variant_name`      : Name of the variant
155    //
156    // ## Variant Data (choose one per variant):
157    // - Unit variant  : No data (just the variant name)
158    // - Tuple variant : `(field_name|index: field_type, ...)`
159    // - Field variant : `{ #[attr] field_name: field_type, ... }`
160    //
161    // ## Conversion Mapping:
162    // - `=> individual_error_name`: Individual error type this variant maps to
163    // - Display expression mapping (matches variant data structure):
164    //   - Unit  : No mapping needed
165    //   - Tuple : `(expression_for_field1, expression_for_field2, ...)`
166    //   - Field : `{ field_name: expression_for_field, ... }`
167    //
168    // # Generates:
169    // - Composite enum with specified variants (unit, tuple or field)
170    // - Error and Display implementations
171    // - From/TryFrom implementations for each individual error type
172    // - Optional const conversion methods for variants marked with +const
173    composite: fmt($fmt:ident)
174        $(+location: $($location:literal)+ ,)?
175        $(+tag: $tag:expr ,)?
176        $(#[$enum_attr:meta])*
177        $vis:vis enum $composite_error_name:ident { $(
178            $(+tag: $tag_variant:expr ,)?
179            $(#[$variant_attr:meta])*
180            $DOC_VARIANT:ident:
181            $(+const$($_c:lifetime)?)?
182            $variant_name:ident
183                $(( $($e_name:ident| $e_numb:literal: $e_ty:ty),+ ))?             // tuple-struct↓
184                $({ $($(#[$f_attr:meta])* $f_name:ident: $f_ty:ty),+ })?          // field-struct↑
185                => $individual_error_name:ident
186                    $(( $($e_display_expr:expr),+ ))?                             // tuple-struct↓
187                    $({ $($f_display_name:ident: $f_display_exp:expr),+ })?       // field-struct↑
188        ),+ $(,)? }
189    ) => {
190        $(#[doc = $tag])?
191        #[doc = $crate::_tags!(error_composite)] // IMPROVE: make optional
192        $(#[$enum_attr])*
193        $(#[doc = $crate::_doc_location![$($location)?]])? // canonical location
194        #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
195        $vis enum $composite_error_name { $(
196            $(#[doc = $tag_variant])?
197            #[doc = $crate::_tags!(error)] // IMPROVE: make optional
198            #[doc = $DOC_VARIANT!()]
199            $(#[$variant_attr])*
200            $variant_name
201                $(( $($e_ty),+ ))?                                                // tuple-struct↓
202                $({ $($(#[$f_attr])* $f_name: $f_ty),+ })?                        // field-struct↑
203        ),+ }
204
205        // implements Error, & Display:
206        impl $crate::Error for $composite_error_name {}
207        impl $crate::Display for $composite_error_name  {
208            fn fmt(&self, $fmt: &mut $crate::Formatter<'_>) -> $crate::FmtResult<()> {
209                match self { $(
210                    $(#[$variant_attr])*
211                    #[allow(unused_doc_comments, reason = "repeated here for feature-gating")]
212                    $composite_error_name::$variant_name
213                    $(( $($e_name),+ ))?                                          // tuple-struct↓
214                    $({ $($f_name),+ })?                                          // field-struct↑
215                    =>
216                        $crate::Display::fmt(&$individual_error_name
217                            $(( $($e_display_expr),+ ))?                          // tuple-struct↓
218                            $({ $($f_display_name: $f_display_exp),+})?           // field-struct↑
219                        , $fmt),
220                )+ }
221            }
222        }
223        // implements From multiple individual errors for a composite error,
224        // and implements TryFrom in reverse:
225        $(
226            $(#[$variant_attr])*
227            $crate::define_error! { $(+const$($_c)?)?
228                from(_f): $individual_error_name, for: $composite_error_name => $variant_name
229                $(( $($e_name, $crate::field_of![_f, $e_numb] ),+ ))?             // tuple-struct↓
230                $({ $($f_name, $crate::field_of![_f, $f_name] ),+ })?             // field-struct↑
231            }
232        )+
233    };
234    (
235    // Implements `From` an individual error type to a composite error containing it, and
236    // implements `TryFrom` in reverse. Optionally implements a `const fn from_individual` method.
237    $(+const$($_c:lifetime)?)?
238    from($fn_arg:ident): $from_individual:ident, for: $for_composite:ident
239    => $variant_name:ident
240        $(( $($e_name:ident, $e_expr:expr),+ ))?                                  // tuple-struct↓
241        $({ $($f_name:ident,  $f_expr:expr),+ })?                                 // field-struct↑
242    $(,)? ) => {
243        $crate::paste! {
244            impl $for_composite {
245                $( #[doc = "*const*"] $($_c)?)?
246                #[doc = "method equivalent to `From<" $from_individual "> for " $for_composite "`."]
247                #[allow(dead_code, reason = "seldomly used")]
248                pub $(const$($_c)?)? fn
249                    [<from_ $from_individual:snake:lower>]($fn_arg: $from_individual)
250                    -> $for_composite {
251                        $for_composite::$variant_name
252                            $(( $($e_expr),+ ))?                                  // tuple-struct↓
253                            $({ $($f_name: $f_expr),+ })?                         // field-struct↑
254                    }
255            }
256        }
257
258        /* (example)
259        impl From<NotEnoughElements> for DataNotEnough {
260            fn from (_f: NotEnoughElements) -> DataNotEnough {
261                DataNotEnough::$Elements(_f.0)
262            }
263        } */
264        impl From<$from_individual> for $for_composite {
265            fn from($fn_arg: $from_individual) -> $for_composite {
266                $for_composite::$variant_name
267                    $(( $($e_expr),+ ))?                                          // tuple-struct↓
268                    $({ $($f_name: $f_expr),+ })?                                 // field-struct↑
269            }
270        }
271        /* (example)
272        impl TryFrom<DataNotEnough> for NotEnoughElements {
273            type Error = FailedErrorConversion;
274            fn from (_f: DataNotEnough) -> Result<NotEnoughElements, FailedErrorConversion> {
275                match _f {
276                    DataNotEnough::Elements(i) => NotEnoughElements(i),
277                    _ => Err(FailedErrorConversion)
278                }
279            }
280        } */
281        impl TryFrom<$for_composite> for $from_individual {
282            type Error = $crate::FailedErrorConversion;
283            fn try_from($fn_arg: $for_composite) -> Result<$from_individual, Self::Error> {
284                match $fn_arg {
285                    $for_composite::$variant_name
286                        $(( $($e_name),+ ))?                                      // tuple-struct↓
287                        $({ $($f_name),+ })?                                      // field-struct↑
288                        => Ok($from_individual
289                            $(( $($e_name),+ ))?                                  // tuple-struct↓
290                            $({ $($f_name),+ })?                                  // field-struct↑
291                        ),
292                        #[allow(unreachable_patterns)]
293                        _ => Err($crate::FailedErrorConversion)
294                }
295            }
296        }
297    };
298    (
299    // Implements `From` a composite error type and a composite error superset of it,
300    // as well as implementing `TryFrom` in the inverse direction.
301    // E.g. from(f): DataNotEnough for: DataError
302    composite: from($fn_arg:ident): $from_subset:ident, for: $for_superset:ident { $(
303        $from_variant:ident
304            $(( $($from_elem:ident),+ ))?                                         // tuple-struct↓
305            $({ $($from_field:ident),+ })?                                        // field-struct↑
306                => $for_variant:ident
307                    $(( $($for_elem:ident),+ ))?                                  // tuple-struct↓
308                    $({ $($for_field:ident),+ })?                                 // field-struct↑
309    ),+ $(,)? }
310    ) => {
311        impl From<$from_subset> for $for_superset {
312            fn from($fn_arg: $from_subset) -> $for_superset { match $fn_arg { $(
313                $from_subset::$from_variant
314                    $(( $($from_elem),+ ))?                                       // tuple-struct↓
315                    $({ $($from_field),+ })?                                      // field-struct↑
316                    => $for_superset::$for_variant
317                        $(( $($for_elem),+ ))?                                    // tuple-struct↓
318                        $({ $($for_field),+ })?                                   // field-struct↑
319            ),+ } }
320        }
321        impl TryFrom<$for_superset> for $from_subset {
322            type Error = $crate::FailedErrorConversion;
323            fn try_from($fn_arg: $for_superset)
324                -> Result<$from_subset, $crate::FailedErrorConversion> { match $fn_arg { $(
325                    $for_superset::$for_variant
326                        $(( $($for_elem),+ ))?                                    // tuple-struct↓
327                        $({ $($for_field),+ })?                                   // field-struct↑
328                            => Ok($from_subset::$from_variant
329                                $(( $($from_elem),+ ))?                           // tuple-struct↓
330                                $({ $($from_field),+ })?                          // field-struct↑
331                    ),)+
332                    _ => Err($crate::FailedErrorConversion)
333                }
334            }
335        }
336    };
337}

341#[cfg(test)]
342mod tests {
343    use super::define_error;
344
345    #[test]
346    fn error() {
347        /* define individual errors */
348
349        define_error! { individual: pub struct UnitStruct;
350            +tag: "tag",
351            DOC_UNIT_STRUCT = "docstring", self+f => write!(f, "display"),
352        }
353        define_error! { individual: pub struct SingleElement(pub(crate) Option<u8>);
354            DOC_SINGLE_ELEMENT = "docstring", self+f => write!(f, "display"),
355        }
356        define_error! { individual: pub struct MultipleElements(pub i32, u32,);
357            DOC_MULTI_ELEMENT = "docstring", self+f => write!(f, "display")
358        }
359        define_error! { individual: pub struct StructFields {
360                #[doc = "field1"] pub f1: bool,
361                #[doc = "field2"] f2: Option<char>
362            }
363            DOC_STRUCT_FIELDS = "docstring", self+f => write!(f, "display")
364        }
365
366        /* define composite errors
367         * (includes conversions between variants and indidual errors)
368         * (also enables some const conversion method implementations) */
369
370        define_error! { composite: fmt(f)
371            /// A composite error superset.
372            pub enum CompositeSuperset {
373                +tag: "tag",
374                DOC_UNIT_STRUCT: +const SuperUnit => UnitStruct,
375                DOC_SINGLE_ELEMENT: SuperSingle(i|0: Option<u8>) => SingleElement(*i),
376                DOC_MULTI_ELEMENT: SuperMultiple(i|0: i32, j|1: u32) => MultipleElements(*i, *j),
377                DOC_STRUCT_FIELDS: +const SuperStruct { f1: bool, f2: Option<char> }
378                    => StructFields { f1: *f1, f2: *f2 },
379            }
380        }
381        define_error! { composite: fmt(f)
382            /// A composite error subset.
383            // NOTE: removed the unit struct variant (the most trivial case)
384            pub enum CompositeSubset {
385                DOC_SINGLE_ELEMENT: SubSingle(i|0: Option<u8>) => SingleElement(*i),
386                DOC_MULTI_ELEMENT: SubMultiple(i|0: i32, j|1: u32) => MultipleElements(*i, *j),
387                DOC_STRUCT_FIELDS: +const SubStruct { f1: bool, f2: Option<char> }
388                    => StructFields { f1: *f1, f2: *f2 },
389            }
390        }
391
392        /* implement conversions between composite errors */
393
394        define_error! { composite: from(f): CompositeSubset, for: CompositeSuperset {
395            SubSingle(i) => SuperSingle(i),
396            SubMultiple(i, j) => SuperMultiple(i, j),
397            SubStruct { f1, f2 } => SuperStruct { f1, f2 },
398        }}
399
400        /* check the conversions between individual and composite errors */
401
402        assert_eq![CompositeSuperset::SuperUnit, UnitStruct.into()];
403        assert![UnitStruct::try_from(CompositeSuperset::SuperUnit).is_ok()];
404        assert![UnitStruct::try_from(CompositeSuperset::SuperSingle(None)).is_err()];
405
406        assert_eq![CompositeSuperset::SuperSingle(Some(1)), SingleElement(Some(1)).into()];
407        assert![SingleElement::try_from(CompositeSuperset::SuperSingle(Some(2))).is_ok()];
408
409        assert_eq![CompositeSuperset::SuperMultiple(3, 5), MultipleElements(3, 5).into()];
410        assert![MultipleElements::try_from(CompositeSuperset::SuperMultiple(7, 13)).is_ok()];
411
412        assert_eq![
413            CompositeSuperset::SuperStruct { f1: true, f2: Some('a') },
414            StructFields { f1: true, f2: Some('a') }.into()
415        ];
416        assert![
417            StructFields::try_from(CompositeSuperset::SuperStruct { f1: false, f2: None }).is_ok()
418        ];
419
420        /* check the conversions between composite errors */
421
422        assert_eq![
423            CompositeSuperset::SuperSingle(Some(1)),
424            CompositeSubset::SubSingle(Some(1)).into()
425        ];
426        assert_eq![
427            CompositeSubset::try_from(CompositeSuperset::SuperSingle(Some(2))),
428            Ok(CompositeSubset::SubSingle(Some(2)))
429        ];
430
431        assert_eq![
432            CompositeSuperset::SuperMultiple(4, 6),
433            CompositeSubset::SubMultiple(4, 6).into()
434        ];
435        assert_eq![
436            CompositeSubset::try_from(CompositeSuperset::SuperMultiple(5, 7)),
437            Ok(CompositeSubset::SubMultiple(5, 7))
438        ];
439
440        assert_eq![
441            CompositeSuperset::SuperStruct { f1: true, f2: Some('z') },
442            CompositeSubset::SubStruct { f1: true, f2: Some('z') }.into()
443        ];
444        assert_eq![
445            CompositeSubset::try_from(CompositeSuperset::SuperStruct { f1: true, f2: None }),
446            Ok(CompositeSubset::SubStruct { f1: true, f2: None })
447        ];
448
449        assert![CompositeSubset::try_from(CompositeSuperset::SuperUnit).is_err()];
450    }
