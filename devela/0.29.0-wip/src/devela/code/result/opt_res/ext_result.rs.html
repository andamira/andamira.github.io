11/// Extension trait providing additional methods for [`Result`].
12#[doc = crate::_doc_meta!{location("code/result")}]
13///
14/// This trait is sealed and cannot be implemented for any other type.
15///
16/// See also [`OptionExt`][crate::OptionExt], [`OptResExt`][crate::OptResExt].
17///
18/// Based on work from:
19/// - <https://github.com/rust-lang/rust/issues/62358> (contains→has).
20#[cfg_attr(nightly_doc, doc(notable_trait))]
21#[expect(private_bounds, reason = "Sealed")]
22pub trait ResultExt<T, E>: Sealed {
23    /// Returns `true` if the result is an [`Ok`] value having the given value.
24    ///
25    /// # Examples
26    /// ```
27    /// # extern crate devela as devela;
28    /// use devela::ResultExt;
29    /// assert_eq!(Ok::<_, ()>(1).has(&1), true);
30    /// assert_eq!(Ok::<_, ()>(1).has(&2), false);
31    /// assert_eq!(Err::<u8, _>("err").has(&1), false);
32    /// ```
33    #[must_use]
34    fn has<U: PartialEq<T>>(&self, x: &U) -> bool;
35
36    /// Returns `true` if the result is an [`Err`] value having the given value.
37    ///
38    /// # Examples
39    /// ```
40    /// # extern crate devela as devela;
41    /// use devela::ResultExt;
42    /// assert_eq!(Ok::<_, &str>(1).has_err(&"Some error message"), false);
43    /// assert_eq!(Err::<u8, _>("err").has_err(&"err"), true);
44    /// assert_eq!(Err::<u8, _>("err2").has_err(&"err"), false);
45    /// ```
46    #[must_use]
47    fn has_err<F: PartialEq<E>>(&self, f: &F) -> bool;

74impl<T, E> ResultExt<T, E> for Result<T, E> {
75    fn has<U: PartialEq<T>>(&self, x: &U) -> bool {
76        self.as_ref().is_ok_and(|y| x == y)
77    }
78
79    fn has_err<F: PartialEq<E>>(&self, f: &F) -> bool {
80        self.as_ref().err().is_some_and(|e| f == e)
81    }
