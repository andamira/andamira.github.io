16#[doc = crate::_tags!(code value)]
17/// Hooks operations into a value without breaking its flow.
18#[doc = crate::_doc_meta!{location("code/result")}]
19///
20/// `Hook` provides a way to intercept a value,
21/// allowing mutation (`hook`) or observation (`tap`) while preserving identity.
22///
23/// Use this trait when working with APIs that mutate in place or return `()`,
24/// but where fluent sequencing is still desired.
25///
26/// This is the ergonomic, runtime counterpart of the [`hook!`] macro.
27///
28/// # Examples
29/// ```
30/// # extern crate devela as devela;
31/// use devela::Hook;
32///
33/// # #[cfg(feature = "alloc")] {
34/// let v = vec![3, 2, 1, 5]
35///     .hook(|v| v.sort())
36///     .tap(|v| assert_eq![v, &[1, 2, 3, 5]])
37///     .hook(|v| v.push(7));
38/// assert_eq![v, vec![1, 2, 3, 5, 7]];
39/// # }
40/// ```
41/// # Design
42/// `Hook` preserves value identity.
43/// For transformations that replace the value, see [`Morph`].
44#[rustfmt::skip]
45#[doc = crate::_doc_vendor!("apply")]
46pub trait Hook: Sized {
47    /// Hooks a mutation step into the value and returns it.
48    ///
49    /// The provided function receives a mutable reference to the value,
50    /// allowing in-place modification while preserving flow.
51    ///
52    /// This is the method form of the [`hook!`] macro.
53    #[must_use]
54    fn hook<F>(mut self, f: F) -> Self where F: FnOnce(&mut Self) { f(&mut self); self }
55
56    /// Taps into the value for observation and returns it unchanged.
57    ///
58    /// The provided function receives a shared reference to the value.
59    /// This is useful for inspection, logging, or assertions.
60    #[must_use]
61    fn tap<F>(self, f: F) -> Self where F: FnOnce(&Self) { f(&self); self }
62}
63impl<T: Sized> Hook for T {}

65#[doc = crate::_tags!(code value)]
66/// Hooks one or more mutation steps into a value and returns it.
67#[doc = crate::_doc_meta!{location("code/result")}]
68///
69/// `hook!` binds a value to a local mutable name, applies a sequence of
70/// side-effecting operations to it, and then returns the value.
71///
72/// This macro preserves the value's identity: steps may mutate it, but do not replace it.
73///
74/// This is the trait-free, const-capable mechanical form of [`Hook`].
75///
76/// Steps can be written as:
77/// - a comma- or semicolon-separated list of expressions, or
78/// - a sequence of adjacent `{ ... }` blocks.
79///
80/// Block syntax is useful when a step requires multiple statements
81/// or local control flow.
82///
83/// All steps are applied in order to the same value.
84///
85/// # Examples
86/// ```
87/// # use devela::{const_assert, is, hook};
88/// /* expression syntax */
89/// const ARRAY: [u8; 4] = [3, 1, 4, 1];
90/// const DATA: [u8; 4] = hook!(ARRAY; |v| v[0] = 9; v[3] = 7);
91/// const_assert![eq_buf &DATA, &[9, 1, 4, 7]];
92///
93/// /* block syntax */
94/// const INPUT: [u8; 4] = [3, 0, 9, 1];
95/// const OUTPUT: [u8; 4] = hook!(INPUT, |v| { v[0] = 1; is!(v[2] >7 , v[2] -= 7) } { v[3] += 5 });
96/// const_assert![eq_buf &OUTPUT, &[1, 0, 2, 6]];
97/// ```
98/// For transformations that replace the value, see [`morph!`].
99#[macro_export]
100#[cfg_attr(cargo_primary_package, doc(hidden))]
101macro_rules! hook {
102    // (comma separated)
103    // expr-sequence form
104    ($t:expr, |$v:ident| $($step:expr),+ $(,)?) => {{
105        let mut $v = $t; $( $step; )+ $v
106    }};
107    // block-sequence form
108    ($t:expr, |$v:ident| $( { $($body:tt)* } $(,)? )+) => {{
109        let mut $v = $t; $( { $($body)* }; )+ $v
110    }};
111
112    // (semicolon separated)
113    // expr-sequence form
114    ($t:expr; |$v:ident| $($step:expr);+ $(;)?) => {{
115        let mut $v = $t; $( $step; )+ $v
116    }};
117    // block-sequence form
118    ($t:expr; |$v:ident| $( { $($body:tt)* } $(;)? )+) => {{
119        let mut $v = $t; $( { $($body)* }; )+ $v
120    }};
121}
122#[doc(inline)]
123pub use hook;

125#[doc = crate::_tags!(code value)]
126/// Morphs a value by threading it through a function.
127#[doc = crate::_doc_meta!{location("code/result")}]
128///
129/// `Morph` provides a fluent way to pass a value (by value, shared reference,
130/// or exclusive reference) into a transformation and return the result.
131///
132/// Unlike [`Hook`], `Morph` does not preserve value identity:
133/// each operation yields a new result.
134///
135/// This is the ergonomic, runtime counterpart of the [`morph!`] macro.
136///
137/// # Examples
138/// ```
139/// # extern crate devela as devela;
140/// use devela::Morph;
141///
142/// let x = 3.morph(|v| v * 2).morph(|v| v + 1);
143/// assert_eq![x, 7];
144/// ```
145///
146/// ```compile_fail
147/// # extern crate devela as devela;
148/// use devela::Morph;
149///
150/// // We can mutate it, but we do not receive the value back.
151/// let v: Vec<i32> = vec![3, 2, 1, 5].morph_mut(|it| it.sort());
152/// ```
153#[rustfmt::skip]
154#[doc = crate::_doc_vendor!("apply")]
155pub trait Morph<R> {
156    /// Morphs the value into a new one and returns it.
157    ///
158    /// The provided function consumes the value and produces a replacement.
159    #[must_use]
160    fn morph<F>(self, f: F) -> R where F: FnOnce(Self) -> R, Self: Sized { f(self) }
161
162    /// Morphs the value by shared reference and returns the result.
163    ///
164    /// The provided function receives `&Self` and may compute any result from it.
165    #[must_use]
166    fn morph_ref<F>(&self, f: F) -> R where F: FnOnce(&Self) -> R { f(self) }
167
168    /// Morphs the value by exclusive reference and returns the result.
169    ///
170    /// The provided function receives `&mut Self` and may mutate it before producing the result.
171    #[must_use]
172    fn morph_mut<F>(&mut self, f: F) -> R where F: FnOnce(&mut Self) -> R { f(self) }
173}
174impl<T: ?Sized, R> Morph<R> for T {}

176#[doc = crate::_tags!(code value)]
177/// Morphs a value through one or more transformation steps and returns the result.
178#[doc = crate::_doc_meta!{location("code/result")}]
179///
180/// `morph!` expresses a left-to-right transformation pipeline
181/// without explicit intermediate bindings in user code.
182///
183/// Conceptually, `morph!(x, a, b, c)` corresponds to `c(b(a(x)))`.
184///
185/// A single binding introduces the pipeline variable,
186/// which is reused implicitly by subsequent steps.
187/// Each step may change the value's type.
188///
189/// This macro is const-capable and does not create closures.
190/// It is the trait-free mechanical form of [`Morph`].
191///
192/// Steps can be written as:
193/// - a comma-separated list of expressions, or
194/// - a semicolon-separated list of expressions.
195///
196/// # Examples
197/// ```
198/// # use devela::{const_assert, is, morph};
199/// const S: &str = morph!(3u8, |v| v as usize, v * 2, v + 1, if v == 7 { "7" } else { "not7" });
200/// const_assert![eq_str S, "7"];
201/// ```
202/// For pipelines that preserve value identity and mutate in place, see [`hook!`].
203#[macro_export]
204#[cfg_attr(cargo_primary_package, doc(hidden))]
205macro_rules! morph {
206    // (comma separated)
207    // base case: single-step
208    ($t:expr, |$v:ident| $e:expr $(,)?) => {{ let $v = $t; $e }};
209    // recursive case: multiple steps
210    ($t:expr, |$v:ident| $e:expr, $($rest:expr),+ $(,)?) => {{
211        morph!( { let $v = $t; $e }, |$v| $($rest),+ )
212    }};
213
214    // (semicolon separated)
215    // base case: single-step
216    ($t:expr; |$v:ident| $e:expr $(;)?) => {{ let $v = $t; $e }};
217    // recursive case: multiple steps
218    ($t:expr; |$v:ident| $e:expr; $($rest:expr);+ $(;)?) => {{
219        morph!( { let $v = $t; $e }; |$v| $($rest);+ )
220    }};
221}
222#[doc(inline)]
223pub use morph;
