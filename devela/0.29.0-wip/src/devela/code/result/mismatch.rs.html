13#[doc = crate::_tags!(result)]
14/// Represents a mismatch between an expected `need` and an encountered `have`.
15#[doc = crate::_doc_meta!{location("code/result")}]
16///
17/// A diagnostic description of a mismatch between an expected requirement and an observed value.
18/// This type does not encode failure semantics on its own.
19pub struct Mismatch<N, H> {
20    /// Information about something that was needed, expected or anticipated.
21    pub need: N,
22
23    /// Information about something that was obtained, observed, or encountered.
24    pub have: H,
25
26    /// Contextual static information about the mismatch.
27    pub info: &'static str,
28}
29
30impl<N, H> Mismatch<N, H> {
31    /// Creates a new `Mismatch` with the specified `need` and `have`.
32    #[must_use]
33    pub const fn new(need: N, have: H) -> Self {
34        Self { need, have, info: "" }
35    }
36
37    /// Creates a new `Mismatch` with the specified `need`, `have`, and `info`.
38    #[must_use]
39    pub const fn with_info(need: N, have: H, info: &'static str) -> Self {
40        Self { need, have, info }
41    }
42}

44impl<N, H> Mismatch<Interval<N>, H> {
45    /// Creates a mismatch where `need` is an [`Interval::point`],
46    /// and `have` does not match.
47    #[must_use] #[rustfmt::skip]
48    pub fn in_point_interval(need_point: N, have: H, info: &'static str) -> Self where N: Clone {
49        Self { need: Interval::point(need_point), have, info }
50    }
51
52    /// Creates a mismatch where `need` is an [`Interval::empty`],
53    /// but `have` was provided.
54    #[must_use] #[rustfmt::skip]
55    pub fn in_empty_interval(have: H, info: &'static str) -> Self where N: Default {
56        Self { need: Interval::empty(), have, info }
57    }
58    /// Creates a mismatch where `need` is an [`Interval::empty_const`],
59    /// but `have` was provided.
60    #[must_use] #[rustfmt::skip]
61    pub const fn in_empty_const_interval(have: H, info: &'static str) -> Self
62    where N: ConstInit {
63        Self { need: Interval::empty_const(), have, info }
64    }
65    /// Creates a mismatch where `need` is an [`Interval::empty_with`],
66    /// but `have` was provided.
67    #[must_use] #[rustfmt::skip]
68    pub fn in_empty_interval_with(value: N, have: H, info: &'static str) -> Self where N: Clone {
69        Self { need: Interval::empty_with(value), have, info }
70    }
71
72    /// Creates a mismatch where `need` is an [`Interval::closed`],
73    /// and `have` is outside it.
74    #[must_use]
75    pub const fn in_closed_interval(lower: N, upper: N, have: H, info: &'static str) -> Self {
76        Self { need: Interval::closed(lower, upper), have, info }
77    }
78
79    /// Creates a mismatch where `need` is an [`Interval::closed_open`],
80    /// and `have` is outside it.
81    #[must_use]
82    pub const fn in_closed_open_interval(lower: N, upper: N, have: H, info: &'static str) -> Self {
83        Self {
84            need: Interval::closed_open(lower, upper),
85            have,
86            info,
87        }
88    }
89
90    /// Creates a mismatch where `need` is an [`Interval::open`],
91    /// and `have` is outside it.
92    #[must_use]
93    pub const fn in_open_interval(lower: N, upper: N, have: H, info: &'static str) -> Self {
94        Self { need: Interval::open(lower, upper), have, info }
95    }
96}

138                self.need, self.have, self.info

140        }
141    }

143    impl<N: PartialEq, H: PartialEq> PartialEq for Mismatch<N, H> {
144        fn eq(&self, other: &Self) -> bool {
145            self.need == other.need && self.have == other.have && self.info == other.info
146        }
147    }
148    impl<N: Eq, H: Eq> Eq for Mismatch<N, H> {}
149
150    impl<N: PartialOrd, H: PartialOrd> PartialOrd for Mismatch<N, H> {
151        /// Compare need first. If they are equal, then compare have.
152        fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
153            match self.need.partial_cmp(&other.need) {
154                Some(Ordering::Equal) => match self.have.partial_cmp(&other.have) {
155                    Some(Ordering::Equal) => self.info.partial_cmp(other.info),
156                    other => other,
157                },
158                other => other,
159            }
160        }
161    }
162    impl<N: Ord, H: Ord> Ord for Mismatch<N, H> {
163        /// Compare need first. If they are equal, then compare have.
164        fn cmp(&self, other: &Self) -> Ordering {
165            match self.need.cmp(&other.need) {
166                Ordering::Equal => match self.have.cmp(&other.have) {
167                    Ordering::Equal => self.info.cmp(other.info),
168                    order => order,
169                },
170                order => order,
171            }
172        }
173    }
174
175    impl<N: Hash, H: Hash> Hash for Mismatch<N, H> {
176        fn hash<HR: Hasher>(&self, state: &mut HR) {
177            self.need.hash(state);
178            self.have.hash(state);
179        }
180    }
