14#[doc = crate::_tags!(concurrency)]
15/// A spinlock providing mutual exclusion without blocking.
16#[doc = crate::_doc_meta!{location("work/sync")}]
17///
18/// Uses an atomic flag for synchronization, with a configurable backoff strategy.
19///
20/// # Examples
21/// ```
22/// # use devela::SpinLock;
23/// let lock = SpinLock::<i32, 5, 10, 100>::new(42);
24///
25/// let mut guard = lock.lock(); // Acquire the lock
26/// *guard += 1; // Modify the locked value
27///
28/// if let Some(mut guard) = lock.try_lock() {
29///     *guard *= 2; // Modify only if lock was acquired
30/// }
31/// // Lock is automatically released when `guard` goes out of scope
32/// ```
33#[derive(Default)]
34pub struct SpinLock<T, const SPIN: usize = 5, const YIELD: usize = 10, const SLEEP: u64 = 100> {
35    /// The protected value, wrapped in an `UnsafeCell` for interior mutability.
36    value: UnsafeCell<T>,
37    /// Atomic flag indicating whether the lock is held (`true`).
38    lock: AtomicBool,
39}
40
41// SAFETY: `SpinLock` can be safely sent across threads,
42// because it does not impose any additional restrictions beyond `T: Send`.
43unsafe impl<T> Send for SpinLock<T> where T: Send {}
44// SAFETY: `SpinLock` ensures exclusive access via atomic locking,
45// making it safe for shared references across threads as long as `T: Send`.
46unsafe impl<T> Sync for SpinLock<T> where T: Send {}

48impl<T, const SPIN: usize, const YIELD: usize, const SLEEP: u64> Debug
49    for SpinLock<T, SPIN, YIELD, SLEEP>
50{
51    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
52        let locked = self.lock.load(AtomicOrdering::Acquire);
53        f.debug_struct("SpinLock")
54            .field("type", &any_type_name::<T>())
55            .field("locked", &locked)
56            .finish()
57    }
58}

61impl<T, const SPIN: usize, const YIELD: usize, const SLEEP: u64> SpinLock<T, SPIN, YIELD, SLEEP> {
62    /// Creates a new spinlock with the given value.
63    pub const fn new(value: T) -> Self {
64        SpinLock { value: UnsafeCell::new(value), lock: AtomicBool::new(false) }
65    }
66
67    /// Acquires the lock, spinning until it is available.
68    ///
69    /// This method employs an **adaptive backoff strategy** to minimize CPU contention:
70    /// - Spins ([`spin_loop`]`()`) for `SPIN` iterations to avoid unnecessary thread switching.
71    /// - Yields ([`ThreadExt::yield_now`]`()`) for the next `YIELD - SPIN` iterations,
72    ///   allowing other threads to progress.
73    /// - Sleeps ([`ThreadExt::sleep_ns`]`(SLEEP)`) if `SLEEP > 0`,
74    ///   reducing CPU load under high contention.
75    ///
76    /// ## Environment
77    /// - In `std` environments, the full backoff strategy is used.
78    /// - In `no_std` environments, only `spin_loop()` is available.
79    ///
80    /// ## Compile-Time Optimization
81    /// Unnecessary branches are removed at compile time if their respective constant is set to 0.
82    pub fn lock(&self) -> SpinLockGuard<'_, T, SPIN, YIELD, SLEEP> {
83        #[cfg(feature = "std")]
84        let mut spin = 0usize;
85        while self.lock.compare_exchange_weak(false, true,
86            AtomicOrdering::Acquire, AtomicOrdering::Acquire).is_err() {
87            #[cfg(feature = "std")]
88            {
89                if spin < SPIN { spin_loop(); }
90                else if spin < YIELD { Thread::yield_now(); }
91                else if SLEEP > 0 { Thread::sleep_ns(SLEEP); }
92                spin += 1;
93            }
94            #[cfg(not(feature = "std"))]
95            { spin_loop(); }
96        }
97        // let mut sleep_spin = SleepSpin::<SPIN, YIELD, SLEEP>::new();
98        // while self
99        //     .lock
100        //     .compare_exchange_weak(false, true, Ordering::Acquire, Ordering::Acquire)
101        //     .is_err()
102        // {
103        //     sleep_spin.spin();
104        // }
105        SpinLockGuard(self)
106    }
107
108    /// Attempts to acquire the lock without blocking.
109    ///
110    /// Returns `Some(SpinLockGuard<T>)` if successful, otherwise `None`.
111    pub fn try_lock(&self) -> Option<SpinLockGuard<'_, T, SPIN, YIELD, SLEEP>> {
112        self.lock.compare_exchange(false, true, AtomicOrdering::Acquire, AtomicOrdering::Acquire)
113            .is_ok().then(|| SpinLockGuard(self))
114    }
115
116    /// Checks if the lock is currently held.
117    pub fn is_locked(&self) -> bool { self.lock.load(AtomicOrdering::Acquire) }
118
119    /// Consumes the lock and returns the inner value.
120    pub fn into_inner(self) -> T { self.value.into_inner() }
121
122    /// Tries to consume the lock and return the inner value.
123    ///
124    /// If the lock is currently **unlocked**, this method returns `Some(T)`.
125    /// Otherwise, it returns `None` without blocking or modifying the lock.
126    pub fn try_into_inner(&self) -> Option<T> {
127        // SAFETY: Safe because we check the lock
128        (!self.is_locked()).then(|| unsafe { self.value.get().read() })
129    }
130
131    // /// Provides an unsafe reference to the value without locking.
132    // ///
133    // /// # Safety
134    // /// - The caller must ensure no other thread is modifying the value simultaneously.
135    // /// - This breaks Rust's aliasing rules if misused.
136    // pub unsafe fn unsafe_as_ref(&self) -> &T { unsafe { &*self.value.get() } }
137
138    /// Forces the lock to be released without dropping the guard.
139    ///
140    /// # Safety
141    /// - This is a **manual unlocking mechanism**. If used incorrectly,
142    ///   it may allow multiple threads to access `T` simultaneously.
143    #[cfg(debug_assertions)]
144    #[cfg_attr(nightly_doc, doc(cfg(debug_assertions)))]
145    pub unsafe fn debug_force_unlock(&self) { self.lock.store(false, AtomicOrdering::SeqCst); }
146}

148/// A guard that grants exclusive access to a [`SpinLock`] value.
149///
150/// ## Guarantees
151/// - If a `SpinLockGuard` exists, the lock **is held**.
152/// - The protected value can be safely accessed through `Deref` and `DerefMut`.
153/// - Dropping `SpinLockGuard` **releases the lock**, allowing other threads to acquire it.
154pub struct SpinLockGuard<'a, T, const SPIN: usize, const YIELD: usize, const SLEEP: u64>(
155    &'a SpinLock<T, SPIN, YIELD, SLEEP>,
156);

178impl<T, const SPIN: usize, const YIELD: usize, const SLEEP: u64> Drop
179    for SpinLockGuard<'_, T, SPIN, YIELD, SLEEP>
180{
181    /// Releases the lock when the guard is dropped.
182    fn drop(&mut self) {
183        self.0.lock.store(false, AtomicOrdering::Release);
184    }
185}
186impl<T, const SPIN: usize, const YIELD: usize, const SLEEP: u64> Deref
187    for SpinLockGuard<'_, T, SPIN, YIELD, SLEEP>
188{
189    type Target = T;
190    /// Allows access to the locked value.
191    fn deref(&self) -> &Self::Target {
192        // SAFETY: The lock is held, ensuring exclusive access to `T`.
193        unsafe { &*self.0.value.get() }
194    }
195}
