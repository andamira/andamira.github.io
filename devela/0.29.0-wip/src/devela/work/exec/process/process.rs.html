19#[doc = crate::_tags!(platform concurrency namespace)]
20/// Extension trait providing additional methods for [`Process`]es.
21#[doc = crate::_doc_meta!{location("work/process")}]
22///
23/// It offers the standalone functions in `std::process` as associated methods.
24#[rustfmt::skip]
25#[cfg_attr(nightly_doc, doc(notable_trait))]
26#[expect(private_bounds, reason = "Sealed")]
27pub trait ProcessExt: Sealed {
28    /// Constructs a new `Command` for launching the `program`.
29    ///
30    /// See `Command::`[new][Command::new].
31    fn command<S: AsRef<OsStr>>(program: S) -> Command {
32        Command::new(program)
33    }
34
35    /// Constructs a new `Command` from shell-like command words.
36    ///
37    /// This does not invoke a shell. It only uses shell word syntax to split
38    /// `line` into a program and arguments.
39    ///
40    /// # Errors
41    /// Returns an error if `line` does not contain a program word, contains
42    /// invalid shell word syntax, or contains a decoded word that is not UTF-8.
43    #[cfg(feature = "shell")]
44    fn command_shell(line: &str) -> Result<Command, ShellWordError> {
45        let mut lex = ShellLex::new(line);
46        let mut buf = crate::vec_![0u8; line.len()];
47        let Some(len) = lex.next_word_to(&mut buf)? else {
48            return Err(ShellWordError::EmptyCommand);
49        };
50        let prog = Str::from_utf8(&buf[..len]).map_err(|_| ShellWordError::InvalidUtf8)?;
51        let mut cmd = Command::new(prog);
52        while let Some(len) = lex.next_word_to(&mut buf)? {
53            let arg = Str::from_utf8(&buf[..len]).map_err(|_| ShellWordError::InvalidUtf8)?;
54            cmd.arg(arg);
55        }
56        Ok(cmd)
57    }
58
59    /// Terminates the current process in an abnormal fashion.
60    ///
61    /// See `std::process::`[abort].
62    #[rustfmt::skip]
63    fn abort() -> ! { abort() }
64
65    /// Terminates the current process with the specified exit code.
66    ///
67    /// See `std::process::`[exit].
68    #[rustfmt::skip]
69    fn exit(code: i32) -> ! { exit(code) }
70
71    /// Returns the OS-assigned process identifier associated with this process.
72    ///
73    /// See `std::process::`[id].
74    #[must_use] #[rustfmt::skip]
75    fn self_pid() -> u32 { id() }
76}
77impl ProcessExt for Process {}
