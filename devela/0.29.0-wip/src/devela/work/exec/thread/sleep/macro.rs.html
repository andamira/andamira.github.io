3crate::CONST! {
4    _DOCS_SLEEP4 = r#"
5Sleeps for `$s` seconds, plus optional milliseconds, microseconds and nanoseconds.
6
7Forms:
8- `sleep4![s]`
9- `sleep4![s, ms]`
10- `sleep4![s, ms, us]`
11- `sleep4![s, ms, us, ns]`
12
13The single-argument form also accepts floating-point literals:
14`sleep4![0.5]`.
15
16# Examples
17```
18# use devela::sleep4;
19sleep4![1, 0, 500]; // sleeps for 1 second + 500 microseconds
20sleep4![1.5]; // sleeps for 1.5 seconds
21```
22
23# Features
24- With the `std` feature enabled it leverages [`Thread::sleep`].
25- With the `linux` feature enabled it leverages [`Linux::sleep`].
26
27[`Thread::sleep`]: crate::Thread#method.sleep
28[`Linux::sleep`]: crate::Linux::sleep
29"#;
30}

34/// A more compact thread sleep.
35#[doc = crate::_doc_meta!{location("work/thread")}]
36#[doc = _DOCS_SLEEP4!()]
37#[macro_export]
38#[cfg(feature = "std")]
39#[cfg_attr(nightly_doc, doc(cfg(any(feature = "linux", feature = "std"))))]
40#[cfg_attr(cargo_primary_package, doc(hidden))]
41macro_rules! sleep4 {
42    ($s:literal) => {
43        ::std::thread::sleep($crate::Duration::from_secs_f64($s as f64));
44    };
45    ($s:expr) => {
46        ::std::thread::sleep($crate::Duration::from_secs($s));
47    };
48    ($s:expr, $ms:expr) => {
49        ::std::thread::sleep($crate::Duration::from_millis($s * 1000 + $ms));
50    };
51    ($s:expr, $ms:expr, $us:expr) => {
52        ::std::thread::sleep($crate::Duration::from_micros($s * 1_000_000 + $ms * 1_000 + $us));
53    };
54    ($s:expr, $ms:expr, $us:expr, $ns:expr) => {
55        ::std::thread::sleep($crate::Duration::from_nanos(
56            $s * 1_000_000_000 + $ms * 1_000_000 + $us * 1_000 + $ns,
57        ));
58    };
59}
