70impl WaveletCompressionVec for WaveletHaar {
71    fn compress(&self, coeffs: &[f64], tolerance: f64) -> Vec<f64> {
72        coeffs.iter().map(|&c| if c.abs() < tolerance { 0.0 } else { c }).collect()
73    }
74}
75
76impl WaveletTransformVec for WaveletHaar {
77    fn forward(&self, input: &[f64]) -> Vec<f64> {
78        let mut output = input.to_vec();
79        let mut length = output.len();
80
81        // Iteratively apply averaging and differencing until we reach the coarsest level.
82        while length > 1 {
83            let mut temp = vec![0.0; length];
84            for i in 0..length / 2 {
85                // Calculate average and detail coefficients for each pair.
86                let average = (output[2 * i] + output[2 * i + 1]) / 2.0;
87                let difference = (output[2 * i] - output[2 * i + 1]) / 2.0;
88                temp[i] = average;
89                temp[length / 2 + i] = difference; // Store details in the second half
90            }
91            // Copy temporary results back to output for the next level.
92            output[..length].clone_from_slice(&temp);
93            length /= 2; // Reduce length to work on the next coarser level
94        }
95        output
96    }
97
98    fn inverse(&self, coeffs: &[f64]) -> Vec<f64> {
99        let mut output = coeffs.to_vec();
100        let mut length = 2;
101
102        // Iteratively reconstruct from coarsest to finest level
103        while length <= output.len() {
104            let mut temp = vec![0.0; length];
105            for i in 0..length / 2 {
106                // Reconstruct original values from average and detail coefficients.
107                let average = output[i];
108                let difference = output[length / 2 + i];
109                temp[2 * i] = average + difference; // First reconstructed value
110                temp[2 * i + 1] = average - difference; // Second reconstructed value
111            }
112            // Copy reconstructed values back to output for the next finer level.
113            output[..length].clone_from_slice(&temp);
114            length *= 2; // Expand length to work on the next finer level
115        }
116        output
117    }
118}
