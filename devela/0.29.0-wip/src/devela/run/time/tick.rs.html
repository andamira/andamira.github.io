3//! Defines [`RuntimeTick`].

6use crate::Ordering;
7
8#[doc = crate::_tags!(runtime time)]
9/// A deterministic logical execution-time counter.
10#[doc = crate::_doc_meta!{
11    location("run/time"),
12    test_size_of(RuntimeTick = 8|64),
13}]
14///
15/// `RuntimeTick` represents execution time as an explicitly advanced,
16/// monotonically increasing tick count.
17///
18/// It is **constructed**, not observed: ticks are advanced explicitly by
19/// program logic rather than sampled from external clocks.
20///
21/// ## Semantics
22/// - The origin (`0`) is synthetic and application-defined.
23/// - Ticks have no inherent unit; their meaning is determined by context
24///   (frames, cycles, steps, events, etc.).
25/// - Advancement is explicit and deterministic.
26///
27/// For sampled or system-backed time, use a [`TimeSource`][crate::TimeSource] instead.
28#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
29pub struct RuntimeTick {
30    ticks: u64,
31}
32#[rustfmt::skip]
33impl RuntimeTick {
34    /// Creates a new logical time value with the given tick count.
35    pub const fn new(ticks: u64) -> Self { Self { ticks } }
36
37    /// Returns the underlying tick count.
38    pub const fn ticks(self) -> u64 { self.ticks }
39
40    /// Advances by one tick in place.
41    pub const fn tick(&mut self) { self.ticks += 1; }
42
43    /// Returns the next tick value.
44    pub const fn next(self) -> Self { Self { ticks: self.ticks + 1 } }
45
46    /// Advances this time by `delta` ticks in place.
47    pub const fn advance_mut(&mut self, delta: u64) { self.ticks += delta; }
48
49    /// Returns a new `RuntimeTick` advanced by `delta` ticks.
50    pub const fn advanced(self, delta: u64) -> Self { Self { ticks: self.ticks + delta } }
51
52    /// Returns the difference in ticks between two times.
53    ///
54    /// The result is saturating and never negative.
55    pub const fn delta(self, earlier: Self) -> u64 { self.ticks.saturating_sub(earlier.ticks) }
56
57    /// Returns whether this time is strictly after `other`.
58    pub const fn is_after(self, other: Self) -> bool { self.ticks > other.ticks }
59
60    /// Returns whether this time is strictly before `other`.
61    pub const fn is_before(self, other: Self) -> bool { self.ticks < other.ticks }
62
63    /// Returns whether this time is equal to `other`.
64    pub const fn eq(self, other: Self) -> bool { self.ticks == other.ticks }
65
66    /// Compares two tick times.
67    pub const fn cmp(self, other: Self) -> Ordering {
68        if self.ticks < other.ticks { Ordering::Less }
69        else if self.ticks > other.ticks { Ordering::Greater }
70        else { Ordering::Equal }
71    }
72}
