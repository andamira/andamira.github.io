3//! Defines [`Runtime`].

6use crate::{RunCycle, RunPhase, RuntimeTick};
7
8#[doc = crate::_tags!(runtime)]
9/// A handle to the live execution context of a running system.
10#[doc = crate::_doc_meta!{
11    location("run/time"),
12    #[cfg(target_pointer_width = "32")]
13    test_size_of(__: Runtime<()> = 12|96),
14    #[cfg(target_pointer_width = "64")]
15    test_size_of(__: Runtime<()> = 16|128),
16}]
17///
18/// `Runtime` stores the long-lived state that defines an active run:
19/// - its committed regime,
20/// - its logical tick,
21/// - and its lifecycle phase.
22///
23/// It is a context object, not a scheduler or loop controller.
24#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
25pub struct Runtime<R> {
26    regime: R,
27    tick: RuntimeTick,
28    cycle: RunCycle,
29}
30
31impl<R> Runtime<R> {
32    /// Creates a new runtime from the given regime.
33    pub const fn new(regime: R) -> Self {
34        Self {
35            regime,
36            tick: RuntimeTick::new(0),
37            cycle: RunCycle::new(),
38        }
39    }
40
41    /// Returns a shared reference to the committed regime.
42    pub const fn regime(&self) -> &R {
43        &self.regime
44    }
45
46    /// Returns a mutable reference to the committed regime.
47    pub const fn regime_mut(&mut self) -> &mut R {
48        &mut self.regime
49    }
50
51    /// Returns the current logical tick.
52    pub const fn tick(&self) -> RuntimeTick {
53        self.tick
54    }
55
56    /// Returns the current run cycle.
57    pub const fn cycle(&self) -> RunCycle {
58        self.cycle
59    }
60
61    /// Returns the current run phase.
62    pub const fn phase(&self) -> RunPhase {
63        self.cycle.phase()
64    }
65
66    /// Returns `true` if logical progression is currently allowed.
67    pub const fn can_advance(&self) -> bool {
68        self.cycle.can_advance()
69    }
70
71    /// Advances the logical tick by one.
72    ///
73    /// This does not check the current phase.
74    pub const fn tick_once(&mut self) {
75        self.tick.tick();
76    }
77
78    /// Advances the logical tick by `delta`.
79    ///
80    /// This does not check the current phase.
81    pub const fn advance_mut(&mut self, delta: u64) {
82        self.tick.advance_mut(delta);
83    }
84
85    /// Attempts to transition to `next`.
86    ///
87    /// Returns `true` if the transition is valid.
88    pub const fn transition(&mut self, next: RunPhase) -> bool {
89        self.cycle.transition(next)
90    }
91}
