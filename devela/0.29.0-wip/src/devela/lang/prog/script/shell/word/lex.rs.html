6use crate::{ShellWordError, TextScanner, is, unwrap};
7
8#[doc = crate::_tags!(lang parser)]
9/// Splits shell-like input into command words.
10#[doc = crate::_doc_meta!{location("lang/prog/script/shell")}]
11#[derive(Clone, Debug, PartialEq, Eq, Hash)]
12pub struct ShellLex<'a> {
13    scanner: TextScanner<'a>,
14    line_no: usize,
15}
16impl<'a> ShellLex<'a> {
17    /// Creates a shell word lexer over bytes.
18    #[must_use]
19    pub const fn from_bytes(bytes: &'a [u8]) -> Self {
20        Self {
21            scanner: TextScanner::from_bytes(bytes),
22            line_no: 1,
23        }
24    }
25    /// Creates a shell word lexer over UTF-8 text.
26    #[must_use]
27    pub const fn new(s: &'a str) -> Self {
28        Self::from_bytes(s.as_bytes())
29    }
30    /// Returns the current one-based line number.
31    #[must_use]
32    pub const fn line_no(&self) -> usize {
33        self.line_no
34    }
35}
36impl<'a> ShellLex<'a> {
37    /// Decodes the next shell word into `out`.
38    ///
39    /// Returns `Ok(None)` at end of input.
40    ///
41    /// # Errors
42    /// Returns an error for unterminated quotes, trailing escapes,
43    /// or insufficient output space.
44    pub const fn next_word_to(&mut self, out: &mut [u8]) -> Result<Option<usize>, ShellWordError> {
45        self.skip_gap();
46        is! { self.scanner.peek_byte().is_none(), return Ok(None) }
47        let mut len = 0;
48        loop {
49            let Some(byte) = self.scanner.peek_byte() else {
50                return Ok(Some(len));
51            };
52            match byte {
53                b' ' | b'\t' | b'\n' => return Ok(Some(len)),
54                b'#' if len == 0 => {
55                    self.skip_comment();
56                    self.skip_gap();
57                    is! { self.scanner.peek_byte().is_none(), return Ok(None) }
58                }
59                b'\'' => {
60                    self.scanner.skip_byte();
61                    len = unwrap![ok? self.take_single_to(out, len)];
62                }
63                b'"' => {
64                    self.scanner.skip_byte();
65                    len = unwrap![ok? self.take_double_to(out, len)];
66                }
67                b'\\' => {
68                    self.scanner.skip_byte();
69                    let Some(next) = self.scanner.next_byte() else {
70                        return Err(ShellWordError::TrailingEscape);
71                    };
72                    self.bump_line(next);
73                    is! { next != b'\n', len = unwrap![ok? Self::push_byte(out, len, next)] }
74                }
75                _ => {
76                    self.scanner.skip_byte();
77                    self.bump_line(byte);
78                    len = unwrap![ok? Self::push_byte(out, len, byte)];
79                }
80            }
81        }
82    }
83}
84impl<'a> ShellLex<'a> {
85    const fn skip_gap(&mut self) {
86        loop {
87            match self.scanner.peek_byte() {
88                Some(b' ' | b'\t') => {
89                    self.scanner.skip_byte();
90                }
91                Some(b'\n') => {
92                    self.scanner.skip_byte();
93                    self.line_no += 1;
94                }
95                _ => break,
96            }
97        }
98    }
99    const fn skip_comment(&mut self) {
100        loop {
101            match self.scanner.next_byte() {
102                Some(b'\n') => {
103                    self.line_no += 1;
104                    break;
105                }
106                Some(_) => {}
107                None => break,
108            }
109        }
110    }
111    const fn bump_line(&mut self, byte: u8) {
112        is! { byte == b'\n', self.line_no += 1; }
113    }
114    const fn take_single_to(
115        &mut self,
116        out: &mut [u8],
117        mut len: usize,
118    ) -> Result<usize, ShellWordError> {
119        loop {
120            let Some(byte) = self.scanner.next_byte() else {
121                return Err(ShellWordError::UnterminatedSingleQuote);
122            };
123            if byte == b'\'' {
124                return Ok(len);
125            } else {
126                self.bump_line(byte);
127                len = unwrap![ok? Self::push_byte(out, len, byte)];
128            }
129        }
130    }
131    const fn take_double_to(
132        &mut self,
133        out: &mut [u8],
134        mut len: usize,
135    ) -> Result<usize, ShellWordError> {
136        loop {
137            let Some(byte) = self.scanner.next_byte() else {
138                return Err(ShellWordError::UnterminatedDoubleQuote);
139            };
140            match byte {
141                b'"' => return Ok(len),
142                b'\\' => {
143                    let Some(next) = self.scanner.next_byte() else {
144                        return Err(ShellWordError::UnterminatedDoubleQuote);
145                    };
146                    match next {
147                        b'$' | b'`' | b'"' | b'\\' => {
148                            len = unwrap![ok? Self::push_byte(out, len, next)];
149                        }
150                        b'\n' => {
151                            self.line_no += 1;
152                        }
153                        _ => {
154                            len = unwrap![ok? Self::push_byte(out, len, b'\\')];
155                            len = unwrap![ok? Self::push_byte(out, len, next)];
156                            self.bump_line(next);
157                        }
158                    }
159                }
160                _ => {
161                    self.bump_line(byte);
162                    len = unwrap![ok? Self::push_byte(out, len, byte)];
163                }
164            }
165        }
166    }
167    const fn push_byte(out: &mut [u8], len: usize, byte: u8) -> Result<usize, ShellWordError> {
168        if len == out.len() {
169            Err(ShellWordError::OutputTooSmall { needed: len + 1 })
170        } else {
171            out[len] = byte;
172            Ok(len + 1)
173        }
174    }
175}
