13///
14/// Represents the size of rendered text, measured by [`Web::measure_text`].
15///
16/// - <https://developer.mozilla.org/en-US/docs/Web/API/TextMetrics>
17#[repr(C)]
18#[derive(Copy, Clone, Debug, Default, PartialEq)]
19pub struct JsTextRenderMetrics {
20    /// The width of the rendered text.
21    pub width: f32,
22    /// The distance from the baseline to the highest point.
23    pub ascent: f32,
24    /// The distance from the baseline to the lowest point.
25    pub descent: f32,
26}

34///
35/// Represents the size of rendered text, measured by [`Web::measure_text_full`].
36///
37/// Includes all available text measurement properties.
38/// - <https://developer.mozilla.org/en-US/docs/Web/API/TextMetrics>
39#[repr(C)]
40#[derive(Copy, Clone, Debug, Default, PartialEq)]
41pub struct JsTextRenderMetricsFull {
42    /// The width of the rendered text.
43    pub width: f32,
44    /// Distance from the alignment point to the leftmost glyph edge.
45    pub left: f32,
46    /// Distance from the alignment point to the rightmost glyph edge.
47    pub right: f32,
48    /// Distance from the baseline to the highest glyph edge.
49    pub ascent: f32,
50    /// Distance from the baseline to the lowest glyph edge.
51    pub descent: f32,
52    /// The topmost possible bounding box for text.
53    pub font_ascent: f32,
54    /// The lowest possible bounding box for text.
55    pub font_descent: f32,
56    /// Height from the baseline to the top of the `em` square.
57    pub em_ascent: f32,
58    /// Height from the baseline to the bottom of the `em` square.
59    pub em_descent: f32,
60    /// Hanging baseline position.
61    pub hanging_baseline: f32,
62    /// Alphabetic baseline position.
63    pub alphabetic_baseline: f32,
64    /// Ideographic baseline position.
65    pub ideographic_baseline: f32,
66}
