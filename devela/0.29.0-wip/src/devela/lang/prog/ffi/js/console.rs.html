121/// Emits leveled diagnostics to the JavaScript console.
122///
123/// This implementation maps [`DiagOut`] directly onto the browser's native
124/// console levels, preserving the semantic distinction between trace, debug,
125/// info, warning, and error output.
126///
127/// See also [`TextOut`] for plain text emission.
128#[cfg(not(feature = "safe_lang"))]
129#[macro_apply(_js_unsafe_ffi_doc)]
130impl DiagOut for JsConsole {
131    type Error = Infallible;
132
133    fn diag(&mut self, level: DiagLevel, text: &str) -> Result<(), Self::Error> {
134        match level {
135            DiagLevel::Trace => Self::debug(text), // FUTURE `trace` when message-bearing
136            DiagLevel::Debug => Self::debug(text),
137            DiagLevel::Info => Self::info(text),
138            DiagLevel::Warn => Self::warn(text),
139            DiagLevel::Error => Self::error(text),
140        }
141        Ok(())
142    }
143}

145/// Emits UTF-8 text to the JavaScript console.
146///
147/// This is a convenience text-sink implementation over [`JsConsole`],
148/// forwarding plain text through the console's generic log channel.
149///
150/// For leveled diagnostic output, prefer [`DiagOut`].
151#[cfg(not(feature = "safe_lang"))]
152#[macro_apply(_js_unsafe_ffi_doc)]
153impl TextOut for JsConsole {
154    type Error = Infallible;
155
156    fn write_text(&mut self, text: &str) -> Result<(), Self::Error> {
157        Self::log(text);
158        Ok(())
159    }
