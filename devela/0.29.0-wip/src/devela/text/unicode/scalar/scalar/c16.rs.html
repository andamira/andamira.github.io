12    // SAFETY: this is not marked as unsafe because it's only used privately
13    // for a few selected operations in this module involving ASCII.
14    const fn from_char_unchecked(c: char) -> char16 {
15        char16::new_unchecked(c as u32 as u16)
16    }

18    // SAFETY: this is not marked as unsafe because it's only used privately
19    // for a few selected operations in this module and also by CharIter.
20    pub(crate) const fn new_unchecked(value: u16) -> char16 {
21        cfg_select! { all(feature = "unsafe_niche", not(feature = "safe_text")) => {
22            unsafe { char16(NonSurrogateU16::new_unchecked(value)) }
23        } _ => {
24            Self(crate::unwrap![some NonSurrogateU16::new(value)])
25        }}
26    }

57    /// Tries to convert a `charu` to `char16`.
58    ///
59    /// # Errors
60    /// Returns [`MismatchedCapacity`] if the character can't fit in 16 bits.
61    pub const fn try_from_charu(c: charu) -> Result<char16, MismatchedCapacity> {
62        let scalar = c.to_scalar();
63        if Char(scalar).len_bytes() == 1 {
64            Ok(char16::new_unchecked(scalar as u16))
65        } else {
66            Err(MismatchedCapacity::too_large(scalar as usize, u16::MAX as usize))
67        }
68    }
69    /// Tries to convert a `char` to `char16`.
70    ///
71    /// # Errors
72    /// Returns [`MismatchedCapacity`] if the character can't fit in 16 bits.
73    pub const fn try_from_char(c: char) -> Result<char16, MismatchedCapacity> {
74        if Char(c as u32).len_bytes() <= 2 {
75            Ok(char16::new_unchecked(c as u32 as u16))
76        } else {
77            Err(MismatchedCapacity::too_large(c as u32 as usize, u16::MAX as usize))
78        }
79    }

83    /// Tries to convert this `char16` to `CharAscii`.
84    ///
85    /// # Errors
86    /// Returns [`MismatchedCapacity`] if `self` can't fit in 7 bits.
87    ///
88    /// # Features
89    /// Makes use of the `unsafe_str` feature if enabled.
90    pub const fn try_to_char_ascii(self) -> Result<CharAscii, MismatchedCapacity> {
91        if Char(self.to_scalar()).is_ascii() {
92            cfg_select! { all(feature = "unsafe_str", not(feature = "safe_text")) => {
93                // SAFETY: we've already checked it's in range.
94                Ok(unsafe { CharAscii::from_u8_unchecked(self.0.get() as u8) })
95            } _ => {
96                if let Some(c) = CharAscii::from_u8(self.0.get() as u8) { Ok(c) }
97                else { unreachable![] }
98            }}
99        } else {
100            Err(MismatchedCapacity::too_large(self.to_scalar() as usize, 127))
101        }
102    }

122    /// Converts this `char16` to `char`.
123    ///
124    /// # Features
125    /// Uses the `unsafe_str` feature to avoid duplicated validation.
126    #[must_use]
127    #[rustfmt::skip]
128    pub const fn to_char(self) -> char {
129        cfg_select! { all(feature = "unsafe_str", not(feature = "safe_text")) => {
130            // SAFETY: we guarantee we contain a valid scalar value
131            unsafe { char::from_u32_unchecked(self.to_scalar()) }
132        } _ => {
133            if let Some(c) = char::from_u32(self.to_scalar()) { c } else { unreachable![] }
134        }}
135    }

142    /// Converts this `char16` to an UTF-8 encoded sequence of bytes.
143    ///
144    /// Note that this function always returns a 3-byte array, but the actual
145    /// UTF-8 sequence may be shorter. The unused bytes are set to 0.
146    //
147    // https://en.wikipedia.org/wiki/UTF-8#Encoding
148    #[must_use]
149    #[allow(clippy::unusual_byte_groupings)]
150    pub const fn to_utf8_bytes(self) -> [u8; 3] {
151        let c = self.0.get();
152        match c {
153            // From 0x0000 to 0x007F:
154            // the UTF-8 encoding is the same as the scalar value.
155            0x0000..=0x007F => [c as u8, 0, 0],
156
157            // from 0x0080 to 0x07FF:
158            // the UTF-8 encoding is 110xxxxx 10xxxxxx,
159            // where xxxxx and xxxxxx are the bits of the scalar value.
160            0x0080..=0x07FF => {
161                let y = 0b10_000000 | (Char::<u32>::CONT_MASK as u8 & (c as u8));
162                let x = 0b110_00000 | ((c >> 6) as u8);
163                [x, y, 0]
164            }
165
166            // From from 0x0800 to 0xFFFF:
167            // the UTF-8 encoding is 1110xxxx 10xxxxxx 10xxxxxx.
168            0x0800..=0xFFFF => {
169                let z = 0b10_000000 | (Char::<u32>::CONT_MASK as u8 & (c as u8));
170                let y = 0b10_000000 | ((c >> 6) & Char::<u32>::CONT_MASK as u16) as u8;
171                let x = 0b1110_0000 | ((c >> 12) as u8);
172                [x, y, z]
173            }
174        }
175    }

177    /// Makes a copy of the value in its ASCII upper case equivalent.
178    ///
179    /// ASCII letters 'a' to 'z' are mapped to 'A' to 'Z', but non-ASCII letters
180    /// are unchanged.
