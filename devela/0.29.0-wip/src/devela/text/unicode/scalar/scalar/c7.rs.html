19    pub(crate) const fn new_unchecked(value: u8) -> char7 {
20        cfg_select! { all(feature = "unsafe_niche", not(feature = "safe_text")) => {
21            unsafe { char7(NonMaxU8::new_unchecked(value)) }
22        } _ => {
23            if let Some(c) = NonMaxU8::new(value) { char7(c) } else { unreachable![] }
24        }}
25    }

40    /// Tries to convert a `char8` to `char7`.
41    ///
42    /// # Errors
43    /// Returns [`MismatchedCapacity`] if the character can't fit in 7 bits.
44    pub const fn try_from_char8(c: char8) -> Result<char7, MismatchedCapacity> {
45        let scalar = c.to_scalar();
46        if Char(scalar).is_ascii() { Ok(char7::new_unchecked(scalar as u8)) }
47        else { Err(MismatchedCapacity::too_large(scalar as usize, 127)) }
48    }

163    /// Tries to convert a `[u8; N]` into `[char7; N]`, ensuring ASCII.
164    ///
165    /// # Errors
166    /// Returns [`MismatchedCapacity`] if any byte is not valid ASCII (greater than `0x7F`).
167    pub const fn try_from_u8_array<const N: usize>(bytes: [u8; N])
168        -> Result<[char7; N], MismatchedCapacity> {
169        let mut out = [char7::INIT; N];
170        whilst! { i in 0..N; {
171            if bytes[i] <= 0x7F { out[i] = char7::new_unchecked(bytes[i]); }
172            else { return Err(MismatchedCapacity::too_large(bytes[i] as usize, 0x7F)); }
173        }}
174        Ok(out)
175    }
176
177    /// Converts a `[u8; N]` into `[char7; N]` without checks.
178    ///
179    /// # Safety
180    /// All bytes must be ASCII (`<= 0x7F`).
181    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_str")))]
182    #[cfg(all(not(feature = "safe_text"), feature = "unsafe_str"))]
183    pub const unsafe fn from_u8_array_unchecked<const N: usize>(bytes: [u8; N]) -> [char7; N] {
184        let mut out = [char7::INIT; N];
185        whilst! { i in 0..N; { out[i] = char7::new_unchecked(bytes[i]); }}
186        out
187    }

189    /* byte slices */
190
191    /// Reads ASCII bytes from `bytes` into `out`.
192    ///
193    /// Converts up to `min(bytes.len(), out.len())` elements and returns
194    /// the number of characters written.
195    ///
196    /// # Errors
197    /// Returns [`MismatchedCapacity`] if a non-ASCII byte (greater than `0x7F`) is encountered.
198    pub const fn read_u8_slice(bytes: &[u8], out: &mut [char7])
199        -> Result<usize, MismatchedCapacity> {
200        let len = cmp!(min bytes.len(), out.len());
201        whilst! { i in 0..len; {
202            let b = bytes[i];
203            if b <= 0x7F { out[i] = char7::new_unchecked(b); }
204            else { return Err(MismatchedCapacity::too_large(b as usize, 0x7F)); }
205        }}
206        Ok(len)
207    }
208
209    /// Reads ASCII bytes from `bytes` into `out`, stopping at the first non-ASCII byte.
210    ///
211    /// Converts up to `min(bytes.len(), out.len())` elements and returns
212    /// the number of characters written.
213    ///
214    /// Invalid (non-ASCII) bytes terminate the conversion but are not reported as an error.
215    pub const fn read_u8_slice_trunc(bytes: &[u8], out: &mut [char7]) -> usize {
216        let len = cmp!(min bytes.len(), out.len());
217        whilst! { i in 0..len; {
218            let b = bytes[i];
219            if b > 0x7F { break; }
220            out[i] = char7::new_unchecked(b);
221        }}
222        i
223    }
224
225    /// Writes ASCII bytes into `out`.
226    ///
227    /// Writes up to `min(chars.len(), out.len())` bytes and returns the number written.
228    pub const fn write_u8_slice(chars: &[char7], out: &mut [u8]) -> usize {
229        let len = cmp!(min chars.len(), out.len());
230        whilst! { i in 0..len; { out[i] = chars[i].to_byte(); }}
231        len
232    }
233
234    /* string slices */
235
236    /// Writes the UTF-8 bytes into `out` and returns a `&str` view into it.
237    ///
238    /// The returned string slice borrows `out`.
239    ///
240    /// # Features
241    /// Makes use of the `unsafe_str` feature if enabled.
242    pub const fn write_str<'s, const N: usize>(chars: &[char7; N], out: &'s mut [u8; N])
243        -> &'s str {
244        whilst! { i in 0..N; { out[i] = chars[i].to_byte(); }}
245        cfg_select! { all(feature = "unsafe_str", not(feature = "safe_text")) => {
246            // SAFETY: all bytes are ASCII by construction
247            unsafe { Str::from_utf8_unchecked(out) }
248        } _ => {
249            crate::unwrap![ok Str::from_utf8(out)]
250        }}
251    }

253    /* ASCII normalization / comparison */
254
255    /// Converts all characters in the array to their ASCII lowercase form.
256    pub const fn to_ascii_lowercase_array<const N: usize>(mut a: [char7; N]) -> [char7; N] {
257        whilst! { i in 0..N; { a[i] = a[i].to_ascii_lowercase(); }}
258        a
259    }
260
261    /// Compares two arrays for equality, ignoring ASCII case differences.
262    pub const fn eq_ignore_ascii_case<const N: usize>(a: &[char7; N], b: &[char7; N]) -> bool {
263        whilst! { i in 0..N; {
264            // if !a[i].to_ascii_lowercase().eq(b[i].to_ascii_lowercase()) { return false; }
265            if !a[i].to_byte().eq_ignore_ascii_case(&b[i].to_byte()) { return false; }
266        }}
267        true
268    }
