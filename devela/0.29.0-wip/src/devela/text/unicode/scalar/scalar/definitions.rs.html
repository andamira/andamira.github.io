17// This is a surrogate UTF-16 code point that can't ever be a Unicode scalar.
18pub(crate) type NonSurrogateU16 = NonValueU16<0xDFFF>;

20#[doc = crate::_tags!(construction text)]
21/// Concisely creates any kind of Unicode scalar.
22#[doc = crate::_doc_meta!{location("text/unicode/scalar")}]
23#[macro_export]
24#[cfg_attr(cargo_primary_package, doc(hidden))]
25macro_rules! ch {
26    ( // charu
27    u $char:literal) => { $crate::charu::from_char($char) };
28    (u str $str:literal) => { $crate::charu::from_str_unchecked($str) };
29    (u bytes $str:literal) => { $crate::unwrap![some $crate::charu::from_utf8($str)] };
30    ( // charu_niche
31    un $char:literal) => { $crate::charu_niche::from_char($char) };
32    (un str $str:literal) => { $crate::charu_niche::from_str_unchecked($str) };
33    (un bytes $str:literal) => { $crate::unwrap![some $crate::charu_niche::from_utf8($str)] };
34    (
35    c7 $char:literal) => { $crate::unwrap![some $crate::char7::try_from_char($char)] };
36    (c8 $char:literal) => { $crate::unwrap![some $crate::char8::try_from_char($char)] };
37    (c16 $char:literal) => { $crate::unwrap![some $crate::char16::try_from_char($char)] };
38}
39pub use ch;

44/// A 7-bit [Unicode scalar][scalar], limited to [basic latin][0w] subset
45/// (ASCII).
46#[doc = crate::_doc_meta!{location("text/unicode/scalar")}]
47///
48/// `Option<char7>` is the same size as `char7` or `char8` (1 byte).
49///
50/// See also: [`char8`], [`char16`], [`char`][crate::char].
51///
52/// [scalar]: https://www.unicode.org/glossary/#unicode_scalar_value
53/// [0w]: https://en.wikipedia.org/wiki/Basic_Latin_(Unicode_block)
54#[must_use]
55#[repr(transparent)]
56#[derive(Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
57pub struct char7(pub(super) NonMaxU8);
58
59#[doc = crate::_tags!(text)]
60/// An 8-bit [Unicode scalar][scalar], limited to [basic latin][0w]
61/// and [latin-1][1w] subsets.
62#[doc = crate::_doc_meta!{location("text/unicode/scalar")}]
63///
64/// `Option<char8>` is the same size as `char16` or `Option<char16>` (2 bytes),
65/// because each possible value is a valid Unicode scalar.
66///
67/// See also: [`char7`], [`char16`], [`char`][crate::char].
68///
69/// [scalar]: https://www.unicode.org/glossary/#unicode_scalar_value
70/// [0w]: https://en.wikipedia.org/wiki/Basic_Latin_(Unicode_block)
71/// [1w]: https://en.wikipedia.org/wiki/Latin-1_Supplement
72#[must_use]
73#[repr(transparent)]
74#[derive(Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
75pub struct char8(pub(crate) u8);
76
77#[doc = crate::_tags!(text)]
78/// A 16-bit [Unicode scalar][scalar], limited to the
79/// [Basic Multilingual Plane][0w] subset.
80#[doc = crate::_doc_meta!{location("text/unicode/scalar")}]
81///
82/// It can represent every scalar from the [Basic Multilingual Plane][0w] (BMP),
83/// the first and most important plane in the Unicode standard (also known as
84/// plane 0), containing nearly all commonly used writing systems and symbols.
85///
86/// `Option<char16>` is the same size as `char16` (2 bytes).
87///
88/// See also: [`char7`], [`char8`], [`char`][crate::char].
89///
90/// [scalar]: https://www.unicode.org/glossary/#unicode_scalar_value
91/// [0w]: https://en.wikipedia.org/wiki/Plane_(Unicode)#Basic_Multilingual_Plane
92#[must_use]
93#[repr(transparent)]
94#[derive(Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
95pub struct char16(pub(super) NonSurrogateU16);
96
97#[doc = crate::_tags!(text)]
98/// A 32-bit [Unicode scalar][scalar], with UTF-8 representation.
99#[doc = crate::_doc_meta!{
100    location("text/unicode/scalar"),
101    test_size_of(charu = 4),
102}]
103///
104/// It stores the UTF-8 bytes in big-endian order, similarly as a [`str`].
105///
106/// `Option<charu>` occupies extra space (8 bytes),
107/// prioritizing speed over memory. For the opposite trade-off see [`charu_niche`].
108///
109/// [scalar]: https://www.unicode.org/glossary/#unicode_scalar_value
110#[must_use]
111#[repr(transparent)]
112#[derive(Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
113pub struct charu(pub(super) NonNiche<u32>);
114
115#[doc = crate::_tags!(text)]
116/// A 32-bit [Unicode scalar][scalar], with UTF-8 representation,
117/// and niche-memory optimization.
118#[doc = crate::_doc_meta!{
119    location("text/unicode/scalar"),
120    test_size_of(charu_niche = 4; niche Option),
121}]
122///
123/// It stores the UTF-8 bytes in big-endian order, similarly as a [`str`].
124///
125/// `Option<charu_niche>` is the same size as `charu_niche` (4 bytes),
126/// prioritizing memory over speed. For the opposite trade-off see [`charu`].
127///
128/// [scalar]: https://www.unicode.org/glossary/#unicode_scalar_value
129#[must_use]
130#[repr(transparent)]
131#[derive(Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
132pub struct charu_niche(pub(super) NonMaxU32);
