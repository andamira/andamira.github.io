3use crate::Char;
4
5/// # Methods over `u16`.
6#[rustfmt::skip]
7impl Char<u16> {
8    /// Returns `true` if the given Unicode scalar code is a [surrogate code point][0].
9    ///
10    /// [0]: https://www.unicode.org/glossary/#surrogate_code_point
11    #[must_use]
12    pub const fn is_surrogate(self) -> bool { matches!(self.0, 0xD800..=0xDFFF) }
13
14    /// Returns `true` if the given Unicode scalar code is a [leading surrogate][0].
15    ///
16    /// [0]: https://www.unicode.org/glossary/#high_surrogate_code_point
17    #[must_use]
18    pub const fn is_surrogate_high(self) -> bool { matches!(self.0, 0xD800..=0xDBFF) }
19
20    /// Returns `true` if the given Unicode scalar code is a [trailing surrogate][0].
21    ///
22    /// [0]: https://www.unicode.org/glossary/#low_surrogate_code_point
23    #[must_use]
24    pub const fn is_surrogate_low(self) -> bool { matches!(self.0, 0xDC00..=0xDFFF) }
25
26    /// Decodes the given surrogate pair.
27    ///
28    /// # Features
29    /// Uses the `unsafe_str` feature to skip duplicated validation checks.
30    #[must_use]
31    pub const fn decode_surrogate_pair(high: u16, low: u16) -> Option<char> {
32        if Char(high).is_surrogate_high() && Char(low).is_surrogate_low() {
33            let code = 0x10000 + (((high as u32 - 0xD800) << 10) | (low as u32 - 0xDC00));
34            cfg_select! { all(feature = "unsafe_str", not(feature = "safe_text")) => {
35                Some(unsafe { char::from_u32_unchecked(code) }) // SAFETY: we just checked validity
36            } _ => { char::from_u32(code) }}
37        } else {
38            None
39        }
40    }
41}
