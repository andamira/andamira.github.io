16    /// Maximum Unicode code point.
17    pub const MAX_CODE_POINT: u32 = 0x10_FFFF;
18
19    /// Number of Unicode scalar values.
20    pub const SCALAR_COUNT: u32 = Self::MAX_CODE_POINT + 1 - Self::SURROGATE_COUNT;
21    /// Number of Unicode surrogate code points.
22    pub const SURROGATE_COUNT: u32 = Self::SURROGATE_END - Self::SURROGATE_START + 1;
23
24    /// First Unicode surrogate code point.
25    pub const SURROGATE_START: u32 = 0xD800;
26    /// Last Unicode surrogate code point.
27    pub const SURROGATE_END: u32 = 0xDFFF;
28
29    /// Last high-surrogate code point, also called a leading surrogate.
30    pub const SURROGATE_HIGH_END: u32 = 0xDBFF;
31    /// First low-surrogate code point, also called a trailing surrogate.
32    pub const SURROGATE_LOW_START: u32 = 0xDC00;

34    /// Maps a dense Unicode scalar rank to its scalar value.
35    ///
36    /// Scalar indices are contiguous in `0..SCALAR_COUNT`, with the surrogate
37    /// range omitted from the resulting values.
38    ///
39    /// # Panics
40    /// Panics if `rank >= SCALAR_COUNT`.
41    #[must_use]
42    #[inline(always)]
43    pub const fn scalar_from_rank(rank: u32) -> u32 {
44        assert!(rank < Self::SCALAR_COUNT);
45        if rank < Self::SURROGATE_START { rank } else { rank + Self::SURROGATE_COUNT }
46    }
47
48    /// Returns the dense scalar rank of this value,
49    /// or `None` if it is not a Unicode scalar value.
50    pub const fn scalar_rank(self) -> Option<u32> {
51        if !self.is_valid_scalar() {
52            None
53        } else if self.0 < Self::SURROGATE_START {
54            Some(self.0)
55        } else {
56            Some(self.0 - Self::SURROGATE_COUNT)
57        }
58    }

60    /// Checks whether the value is a Unicode code point.
61    ///
62    /// A valid Unicode code point is any integer in the range: `0..=MAX_CODE_POINT`.
63    ///
64    /// This includes surrogate code points which are
65    /// valid code points but cannot be represented as Unicode scalars.
66    ///
67    /// # Examples
68    /// ```
69    /// # use devela::Char;
70    /// assert!(Char('A' as u32).is_valid_code()); // regular character
71    /// assert!(Char(0x00).is_valid_code());       // NULL is valid
72    /// assert!(Char(0x10FFFF).is_valid_code());   // maximum Unicode code point
73    /// // surrogates are valid code points:
74    /// assert!(Char(0xD800).is_valid_code());     // high surrogate
75    /// assert!(Char(0xDFFF).is_valid_code());     // low surrogate
76    /// // invalid:
77    /// assert!(!Char(0x110000).is_valid_code());  // above max Unicode
78    /// ```
79    #[must_use] #[inline(always)]
80    pub const fn is_valid_code(self) -> bool {
81        self.0 <= Self::MAX_CODE_POINT
82    }
83
84    /// Checks whether the value is a Unicode scalar value representable as [`char`].
85    ///
86    /// This excludes surrogate code points, which are invalid in UTF-8
87    /// and cannot be represented as Unicode scalars.
88    ///
89    /// # Examples
90    /// ```
91    /// # use devela::Char;
92    /// assert!(Char('A' as u32).is_valid_scalar()); // regular character
93    /// assert!(Char(0x00).is_valid_scalar());       // NULL is valid
94    /// assert!(Char(0x10FFFF).is_valid_scalar());   // maximum Unicode scalar
95    /// // invalid:
96    /// assert!(!Char(0xD800).is_valid_scalar());    // high surrogate
97    /// assert!(!Char(0xDFFF).is_valid_scalar());    // low surrogate
98    /// assert!(!Char(0x110000).is_valid_scalar());  // above max Unicode
99    /// ```
100    #[must_use]
101    #[inline(always)]
102    pub const fn is_valid_scalar(self) -> bool {
103        self.is_valid_code() && !self.is_surrogate()
104    }

106    /// Returns `true` if the value is a Unicode [surrogate][0] code point.

110    pub const fn is_surrogate(self) -> bool {

119    }

122    /// [0]: https://www.unicode.org/glossary/#low_surrogate_code_point
123    #[must_use] #[inline(always)]

128    /// Returns the bytes required to store the given Unicode code point in a non-UTF encoding.
129    ///
130    /// This function does **not** determine the UTF-8 byte length.
131    /// It assumes a simple encoding where values up to `0xFF` use 1 byte,
132    /// `0x100..=0xFFFF` use 2 bytes, and anything larger uses 3 bytes.
133    #[must_use]
134    pub const fn len_bytes(self) -> usize {
135        match self.0 {
136            0x0000..=0x00FF => 1,
137            0x0100..=0xFFFF => 2,
138            _ => 3,
139        }
140    }
141    /// Returns the number of bytes required to encode the given Unicode scalar as UTF-8.
142    ///
143    /// Returns `None` if it's not a valid Unicode scalar.
144    #[must_use]
145    pub const fn len_utf8(self) -> Option<usize> {
146        if self.is_valid_scalar() { Some(self.len_utf8_unchecked()) } else { None }
147    }
148    /// Returns the UTF-8 byte length of the current Unicode scalar **without validation**.
149    ///
150    /// Assumes the code is a valid Unicode scalar.
151    /// Use [`len_utf8`][Self::len_utf8] for a checked version.
152    #[must_use]
153    pub const fn len_utf8_unchecked(self) -> usize {
154        match self.0 {
155            0x00_0000..=0x00_007F => 1,
156            0x00_0080..=0x00_07FF => 2,
157            0x00_0800..=0x00_FFFF => 3,
158            _ => 4,
159        }
160    }

162    /// Returns the monospace display width.
163    ///
164    /// - 0: Non-printing characters (controls, combining marks)
165    /// - 1: Regular characters (Latin, Greek, Cyrillic, etc.)
166    /// - 2: Wide characters (CJK, emoji, fullwidth forms)
167    #[must_use]
168    pub const fn width(self) -> usize {
169        if self.is_control() || self.is_combining() {
170            0
171        } else if self.is_fullwidth() {
172            2
173        } else {
174            1
175        }
176    }
177    /// Returns the monospace display width using faster calculation.
178    ///
179    /// Uses optimized checks that cover common cases but may incorrectly
180    /// report some obscure Unicode characters as 1 width instead of 2.
181    #[must_use]
182    pub const fn width_common(self) -> usize {
183        if self.is_control_common() || self.is_combining_common() {
184            0
185        } else if self.is_fullwidth_common() {
186            2
187        } else {
188            1
189        }
190    }

194    pub const fn is_ascii(self) -> bool { self.0 <= 0x7F }

196    /// Returns `true` if the given Unicode scalar code is a [noncharacter][0].
197    ///
198    /// Note that this also checks against reserved, potential non-characters.
199    ///
200    /// [0]: https://www.unicode.org/glossary/#noncharacter
201    #[must_use]
202    pub const fn is_noncharacter(self) -> bool {
203        // sub-block of 32 non-characters:
204        (self.0 >= 0xFDD0 && self.0 <= 0xFDEF)
205            // 2× non-characters at the end of each plane:
206            || (self.0 >= 0xFFFE && (self.0 & 0xFF) == 0xFE)
207            || (self.0 >= 0xFFFE && (self.0 & 0xFF) == 0xFF)
208            // unallocated range (16 potential non-characters):
209            || (self.0 >= 0x2FE0 && self.0 <= 0x2FEF)
210    }

360    /// Converts the Unicode scalar value to a UTF-8 encoded byte sequence array.
361    ///
362    /// Returns `None` if the value is not a valid Unicode scalar.
363    /// The result is always a `[u8; 4]` array, with unused bytes set to `0`.
364    ///
365    /// See also [`char::encode_utf8`].
366    #[must_use] #[inline(always)]
367    pub const fn to_utf8_bytes(self) -> Option<[u8; 4]> {
368        if self.is_valid_scalar() { Some(self.to_utf8_bytes_unchecked()) } else { None }
369    }
370
371    /// Converts the Unicode scalar value to a UTF-8 encoded byte sequence **without validation**.
372    ///
373    /// Assumes the value is a valid Unicode scalar.
374    /// Always returns a `[u8; 4]` array, with unused bytes set to `0`.
375    ///
376    /// See also [`Char::to_utf8_bytes`] for a checked version.
377    #[must_use]
378    #[allow(clippy::unusual_byte_groupings)]
379    pub const fn to_utf8_bytes_unchecked(self) -> [u8; 4] {
380        let mut buf = [0; 4];
381        let _ = self.write_utf8_to_unchecked(&mut buf);
382        buf
383    }
384    /// Writes this Unicode scalar as UTF-8 into `buf` **without validation**.
385    ///
386    /// Returns the number of bytes written.
387    ///
388    /// A buffer of 4 bytes is always large enough for any Unicode scalar.
389    ///
390    /// # Panics
391    /// Panics if `buf.len() < self.0.len_utf8()`.
392    #[must_use]
393    #[allow(clippy::unusual_byte_groupings)]
394    pub const fn write_utf8_to_unchecked(self, buf: &mut [u8]) -> usize {
395        __unicode_scalar_write_utf8_at![buf, 0, self.0]
396    }

398    /// Returns a Unicode scalar selected from the next output of `rng`.
399    #[must_use] #[inline(always)]
400    pub const fn random_next(rng: &mut Pcg32) -> u32 {
401        let rank = rng.next_bounded(Self::SCALAR_COUNT);
402        Self::scalar_from_rank(rank)
403    }
404
405    /// Returns a Unicode scalar deterministically selected from `seed`.
406    #[must_use] #[inline(always)]
407    pub const fn random_from_seed(seed: u64) -> u32 {
408        Self::random_next(&mut Pcg32::new(seed, 0))
409    }

413/// Writes a Unicode scalar value as UTF-8 bytes at the given offset.
414///
415/// Returns the number of bytes written. Does not validate scalar range.
416#[macro_export]
417macro_rules! __unicode_scalar_write_utf8_at {
418    ($buf:ident, $offset:expr, $value:expr) => {{
419        let off = $offset;
420        let val = $value as u32;
421        match val {
422            0x0000..=0x007F => {
423                $buf[off + 0] = val as u8;
424                1
425            }
426            0x0080..=0x07FF => {
427                $buf[off + 0] = 0b110_00000 | (val >> 6) as u8;
428                $buf[off + 1] = 0b10_000000 | (($crate::Char::<u32>::CONT_MASK & val) as u8);
429                2
430            }
431            0x0800..=0xFFFF => {
432                $buf[off + 0] = 0b1110_0000 | (val >> 12) as u8;
433                $buf[off + 1] = 0b10_000000 | ((val >> 6) & $crate::Char::<u32>::CONT_MASK) as u8;
434                $buf[off + 2] = 0b10_000000 | (($crate::Char::<u32>::CONT_MASK & val) as u8);
435                3
436            }
437            _ => {
438                $buf[off + 0] = 0b11110_000 | (val >> 18) as u8;
439                $buf[off + 1] = 0b10_000000 | ((val >> 12) & $crate::Char::<u32>::CONT_MASK) as u8;
440                $buf[off + 2] = 0b10_000000 | ((val >> 6) & $crate::Char::<u32>::CONT_MASK) as u8;
441                $buf[off + 3] = 0b10_000000 | (($crate::Char::<u32>::CONT_MASK & val) as u8);
442                4
443            }
444        }
445    }};
