34    /// Returns the expected UTF-8 byte length based on the given first byte, or `None` if invalid.
35    ///
36    /// LUT based (256-byte array).
37    #[must_use]
38    pub const fn len_utf8(self) -> Option<usize> {
39        let width = self.len_utf8_unchecked();
40        is![width == 0, None, Some(width)]
41    }
42    /// Returns the expected UTF-8 byte length based on the given first byte, or `0` if invalid.
43    ///
44    /// LUT based (256-byte array).
45    #[must_use]
46    pub const fn len_utf8_unchecked(self) -> usize {
47        Self::UTF8_CHAR_LEN[self.0 as usize] as usize
48    }
49    /// Returns the expected UTF-8 byte length based on the given first byte, or `None` if invalid.
50    ///
51    /// Match based, for when memory accesses are more expensive than branches.
52    #[must_use]
53    pub const fn len_utf8_match(self) -> Option<usize> {
54        match self.0 { // same logic as Self::UTF8_CHAR_LEN
55            0x00..=0x7F => Some(1),
56            0xC2..=0xDF => Some(2), // skips invalid C0, C1
57            0xE0..=0xEF => Some(3),
58            0xF0..=0xF4 => Some(4), // skips invalid 0xF5..0x=F7
59            _ => None, // invalid leading byte
60        }
61    }
62    /// Returns the expected UTF-8 byte length based on the given first byte.
63    ///
64    /// Match based, for when memory accesses are more expensive than branches.
65    ///
66    /// This function does **not** validate UTF-8 but determines how many bytes
67    /// a valid sequence **should** occupy based on the leading byte.
68    ///
69    /// ### Caveat
70    /// - If used on malformed UTF-8, it may suggest a length longer than the actual valid sequence.
71    /// - Always use in conjunction with proper UTF-8 validation if handling untrusted input.
72    #[must_use]
73    pub const fn len_utf8_match_naive(self) -> usize {
74        match self.0 {
75            0x00..=0x7F => 1, // 1-byte ASCII
76            0xC0..=0xDF => 2, // 2-byte sequence
77            0xE0..=0xEF => 3, // 3-byte sequence
78            0xF0..=0xF7 => 4, // 4-byte sequence
79            _ => 0,           // invalid leading byte
80        }
81    }

83    /// Returns `true` if this byte is a valid starting point for a UTF-8 sequence.
84    ///
85    /// This checks if the byte is not a UTF-8 continuation byte (i.e., it's either
86    /// an ASCII character or a valid leading byte of a multi-byte sequence).
87    #[must_use] #[inline(always)]
88    pub const fn is_utf8_boundary(self) -> bool {
89        // Equivalent to: b < 128 || b >= 192 (== not a continuation byte (0b10xxxxxx))
90        (self.0 as i8) >= -0x40
91    }
92    /// Returns `true` if this byte is a UTF-8 continuation byte.
93    ///
94    /// Continuation bytes have the bit pattern `10xxxxxx`.
95    #[must_use] #[inline(always)]
96    pub const fn is_utf8_continuation(self) -> bool { !self.is_utf8_boundary() }

102    pub const fn as_char(self) -> char { self.0 as char }
