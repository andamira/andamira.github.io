9///
10/// Used by the grapheme boundary detection algorithm to determine where
11/// grapheme cluster boundaries occur in text.
12///
13/// Based on the **Grapheme_Cluster_Break** property from [UAX#29 Section 3.1][0].
14///
15/// Note: `ExtendedPictographic` is derived from emoji code point tables
16/// but treated as mutually exclusive with other break properties.
17///
18/// [0]: https://www.unicode.org/reports/tr29/#Grapheme_Cluster_Break_Property_Values
19///
20#[doc = crate::_doc_vendor!("grapheme_machine")]
21#[repr(u8)]
22#[allow(missing_docs)]
23#[derive(Debug, Clone, Copy, Eq)]
24pub enum GraphemePropCb {
25    /// No specific grapheme cluster break property applies
26    None = 0x00,
27    /// Carriage Return (U+000D)
28    CR = 0x01,
29    /// Control code point (general category Cc, Cf, Zl, or Zp)
30    Control = 0x02,
31    /// Grapheme extender (general category Me or Mn)
32    Extend = 0x03,
33    /// Extended pictographic code point (emoji)
34    ExtendedPictographic = 0x04,
35    /// Hangul syllable L (leading consonant)
36    L = 0x05,
37    /// Line Feed (U+000A)
38    LF = 0x06,
39    /// Hangul syllable LV (vowel)
40    LV = 0x07,
41    /// Hangul syllable LVT (trailing consonant)
42    LVT = 0x08,
43    /// Prepend code point
44    Prepend = 0x09,
45    /// Regional indicator symbol (flag emojis)
46    RegionalIndicator = 0x0a,
47    /// Spacing mark
48    SpacingMark = 0x0b,
49    /// Hangul syllable T (trailing consonant)
50    T = 0x0c,
51    /// Hangul syllable V (vowel)
52    V = 0x0d,
53    /// Zero Width Joiner (U+200D)
54    Zwj = 0x0e,
55}
56
57impl_trait! { PartialEq for GraphemePropCb |self, other| Self::eq(*self, *other) }
58impl_trait! { Hash for GraphemePropCb |self, state| { Mem::discriminant(self).hash(state); } }

68                | (Self::Extend, Self::Extend)
69                | (Self::ExtendedPictographic, Self::ExtendedPictographic)
70                | (Self::L, Self::L)
71                | (Self::LF, Self::LF)
72                | (Self::LV, Self::LV)
73                | (Self::LVT, Self::LVT)
74                | (Self::Prepend, Self::Prepend)
75                | (Self::RegionalIndicator, Self::RegionalIndicator)
76                | (Self::SpacingMark, Self::SpacingMark)
77                | (Self::T, Self::T)
78                | (Self::V, Self::V)
79                | (Self::Zwj, Self::Zwj)
80        )
81    }

85/// Break property for Indic scripts that prevents splitting within orthographic syllables.
86#[doc = crate::_doc_meta!{location("text/unicode/grapheme")}]
87///
88/// Used by grapheme boundary rule [GB9c](https://www.unicode.org/reports/tr29/#GB9c)
89/// to avoid inappropriate breaks in conjunct sequences.
90///
91/// Based on the **Indic_Conjunct_Break** property from Unicode.
92///
93#[doc = crate::_doc_vendor!("grapheme_machine")]
94#[repr(u8)]
95#[allow(missing_docs)]
96#[derive(Debug, Clone, Copy, Eq)]
97pub enum GraphemePropInCb {
98    /// Code point is not part of an Indic conjunct sequence.
99    None = 0x00,
100    /// An Indic consonant code point.
101    Consonant = 0x10,
102    /// An extending code point in an Indic sequence.
103    Extend = 0x20,
104    /// A linker code point in an Indic sequence.
105    Linker = 0x30,
106}
107
108impl_trait! { PartialEq for GraphemePropInCb |self, other| Self::eq(*self, *other) }
109impl_trait! { Hash for GraphemePropInCb |self, state| { Mem::discriminant(self).hash(state); } }
110
111impl GraphemePropInCb {
112    /// Const-compatible `Eq`.
113    pub const fn eq(self, other: Self) -> bool {
114        matches!(
115            (self, other),
116            (Self::None, Self::None)
117                | (Self::Consonant, Self::Consonant)
118                | (Self::Extend, Self::Extend)
119                | (Self::Linker, Self::Linker)
120        )
121    }
122}

127///
128/// Packed representation of both [`GraphemePropCb`] and [`GraphemePropInCb`]
129/// properties used by Unicode grapheme cluster [boundary rules].
130///
131#[doc = crate::_doc_vendor!("grapheme_machine")]
132///
133/// [boundary rules]: https://www.unicode.org/reports/tr29/#Grapheme_Cluster_Boundary_Rules
134#[repr(transparent)]
135#[derive(Debug, Clone, Copy, Eq)]
136pub struct GraphemeProps {
137    ///  Stores `GraphemePropCb` in the low 4 bits and `GraphemePropInCb` in bits 4-5,
138    ///  allowing efficient property checks.
139    raw: u8,
140}
141
142impl_trait! { PartialEq for GraphemeProps |self, other| Self::eq(*self, *other) }
143impl_trait! { Hash for GraphemeProps |self, state| { self.raw.hash(state); } }
144
145impl GraphemeProps {
146    /// Creates a new property tuple from individual break properties.
147    pub const fn new(gcb: GraphemePropCb, incb: GraphemePropInCb) -> Self {
148        Self { raw: gcb as u8 | incb as u8 }
149    }
150
151    /// Looks up grapheme properties for a scalar.
152    ///
153    /// Uses an embedded trie to find the Grapheme_Cluster_Break and
154    /// Indic_Conjunct_Break properties for the scalar.
155    pub const fn for_charu(c: charu) -> Self {
156        Self { raw: graphemes_lookup(c) }
157    }
158
159    /// Looks up grapheme properties for a scalar.
160    ///
161    /// Convenience method that converts to `charu` internally.
162    /// Prefer `for_charu` if you already have a `charu`.
163    pub const fn for_char(c: char) -> Self {
164        Self { raw: graphemes_lookup(charu::from_char(c)) }
165    }
166
167    /// Returns the Grapheme_Cluster_Break property.
168    ///
169    /// # Features
170    /// Uses the `unsafe_layout` feature to transmute instead of match.
171    pub const fn gcb_property(self) -> GraphemePropCb {
172        cfg_select! { all(feature = "unsafe_layout", not(feature = "safe_text")) => {
173            let raw = self.raw & 0xf;
174            // SAFETY: The low nibble of raw repr matches the GraphemePropCb repr.
175            unsafe { core::mem::transmute(raw) }
176        } _ => {
177            match self.raw & 0xf {
178                0x00 => GraphemePropCb::None,
179                0x01 => GraphemePropCb::CR,
180                0x02 => GraphemePropCb::Control,
181                0x03 => GraphemePropCb::Extend,
182                0x04 => GraphemePropCb::ExtendedPictographic,
183                0x05 => GraphemePropCb::L,
184                0x06 => GraphemePropCb::LF,
185                0x07 => GraphemePropCb::LV,
186                0x08 => GraphemePropCb::LVT,
187                0x09 => GraphemePropCb::Prepend,
188                0x0a => GraphemePropCb::RegionalIndicator,
189                0x0b => GraphemePropCb::SpacingMark,
190                0x0c => GraphemePropCb::T,
191                0x0d => GraphemePropCb::V,
192                0x0e => GraphemePropCb::Zwj,
193                _ => unreachable!(), // mask ensures only 0x00-0x0E
194            }
195        }}
196    }
197
198    /// Returns the Indic_Conjunct_Break property.
199    ///
200    /// # Features
201    /// Uses the `unsafe_layout` feature to transmute instead of match.
202    pub const fn incb_property(self) -> GraphemePropInCb {
203        cfg_select! { all(feature = "unsafe_layout", not(feature = "safe_text")) => {
204            let raw = self.raw & 0x30;
205            // SAFETY: The selected bits of raw repr matches ththe GraphemePropInCb repr.
206            unsafe { core::mem::transmute(raw) }
207        } _ => {
208            match self.raw & 0x30 {
209                0x00 => GraphemePropInCb::None,
210                0x10 => GraphemePropInCb::Consonant,
211                0x20 => GraphemePropInCb::Extend,
212                0x30 => GraphemePropInCb::Linker,
213                _ => unreachable!(), // mask ensures only these values
214            }
215        }}
216    }
217
218    /// Returns `true` if this code point is a control code point.
219    ///
220    /// Matches CR, LF, or Control categories for rules [GB4]/[GB5].
221    ///
222    /// [GB4]: https://www.unicode.org/reports/tr29/#GB4
223    /// [GB5]: https://www.unicode.org/reports/tr29/#GB5
224    pub const fn is_any_control(self) -> bool {
225        matches!(
226            self.gcb_property(),
227            GraphemePropCb::LF | GraphemePropCb::CR | GraphemePropCb::Control,
228        )
229    }
230
231    /// Const-compatible `Eq`.
232    pub const fn eq(self, other: Self) -> bool {
233        self.raw == other.raw
234    }
235}
