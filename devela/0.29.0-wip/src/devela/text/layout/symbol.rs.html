10#[doc = crate::_tags!(text layout)]
11/// Spatial cohesion rules for a text symbol during layout.
12#[doc = crate::_doc_meta!{location("text/layout")}]
13///
14/// This enum defines what a layout engine is allowed to do with a symbol
15/// when negotiating limited inline space. It expresses layout permissions only;
16/// it does not encode meaning, style, or language semantics.
17#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
18pub enum TextCohesion {
19    /// Must be consumed as a whole or not at all.
20    ///
21    /// The symbol may not be split or removed.
22    /// If it does not fit, layout must stop or fail the fit decision.
23    Atomic,
24
25    /// May be split across layout steps.
26    ///
27    /// The symbol provides a soft break opportunity
28    /// and may be partially consumed if space runs out.
29    Breakable,
30
31    /// May be omitted to satisfy space constraints.
32    ///
33    /// The symbol may be entirely removed from the layout result
34    /// if needed to achieve a fit.
35    Elidable,
36}
37_impl_init![Self::Atomic => TextCohesion];

39#[doc = crate::_tags!(text layout quant)]
40/// Mapping between a contiguous text range and its consumed inline space.
41#[doc = crate::_doc_meta!{location("text/layout")}]
42///
43/// A `TextLayoutSpan` records that a contiguous range of symbols contributed
44/// a given amount of inline space during a layout step.
45///
46/// It is the primary structural output of text layout
47/// and enables stable mapping between text indices and spatial occupation.
48#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
49pub struct TextLayoutSpan {
50    /// Range between the first and last symbols included in this span.
51    pub range: TextRange,
52    /// Total inline space consumed by the symbols in this span.
53    pub units: TextUnit,
54}
55_impl_init![Self::from_prim(0, 0, 0) => TextLayoutSpan];
56
57impl TextLayoutSpan {
58    /// Creates a span from symbol indices and consumed units.
59    pub const fn new(start: TextIndex, end: TextIndex, units: TextUnit) -> Self {
60        Self { range: TextRange { start, end }, units }
61    }
62    /// Creates a span from symbol indices and consumed units.
63    pub const fn with_range(range: TextRange, units: TextUnit) -> Self {
64        Self { range, units }
65    }
66
67    /// Creates a span from symbol indices and consumed units.
68    pub const fn from_prim(start: TextUnit, end: TextUnit, units: TextUnit) -> Self {
69        Self::new(TextIndex(start), TextIndex(end), units)
70    }
71
72    /// Returns the start index.
73    pub const fn start(self) -> TextIndex {
74        self.range.start
75    }
76
77    /// Returns the end index.
78    pub const fn end(self) -> TextIndex {
79        self.range.end
80    }
81
82    /// Returns the interval from `start` to `end`.
83    pub const fn interval(self) -> Interval<TextIndex> {
84        Interval::closed_open(self.start(), self.end())
85    }
86}

88#[doc = crate::_tags!(text layout io)]
89/// Layout participation record for a single text symbol.
90#[doc = crate::_doc_meta!{location("text/layout")}]
91///
92/// A `TextSymbol` describes how a unit of text participates in layout:
93/// how much inline space it consumes and what spatial constraints apply under limited space.
94///
95/// It carries no identity or semantic meaning beyond layout behavior.
96#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
97pub struct TextSymbol {
98    /// Inline space consumed by this symbol when laid out as a whole.
99    pub units: TextUnit,
100    /// Spatial cohesion rules applied to this symbol under limited space.
101    pub cohesion: TextCohesion,
102}
103_impl_init![Self::new(TextUnit::INIT, TextCohesion::INIT) => TextSymbol];
104
105impl TextSymbol {
106    /// Creates a span from symbol indices and consumed units.
107    pub const fn new(units: TextUnit, cohesion: TextCohesion) -> Self {
108        Self { units, cohesion }
109    }
