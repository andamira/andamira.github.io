10#[doc = crate::_tags!(text layout result)]
11/// Result of testing whether text fits within an inline extent.
12#[doc = crate::_doc_meta!{location("text/layout")}]
13#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Hash)]
14pub enum TextFit {
15    /// All required symbols fit within the extent.
16    Full,
17
18    /// Some symbols fit, but space was exhausted.
19    Partial,
20
21    /// No symbols fit within the extent.
22    ///
23    /// This is the default.
24    #[default]
25    None,
26}
27_impl_init![Self::None => TextFit];
28
29#[doc = crate::_tags!(text layout result)]
30/// Result of a single text layout step.
31#[doc = crate::_doc_meta!{location("text/layout")}]
32///
33/// A layout step consumes a prefix of the symbol stream within a given inline extent,
34/// and reports:
35/// - how much space was used
36/// - which spans were produced
37/// - whether layout should continue
38#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
39pub struct TextLayoutStep {
40    /// Number of valid spans written to the output buffer.
41    pub span_count: usize,
42
43    /// Total inline space consumed by this step.
44    pub consumed: TextUnit,
45
46    /// Continuation cursor, if not all symbols were consumed.
47    ///
48    /// `None` indicates that layout reached the end of the symbol stream.
49    pub carry: Option<TextCursor>,
50
51    /// Result of testing whether the text fit the available extent.
52    pub fit: TextFit,
53}
54impl TextLayoutStep {
55    /// Creates a new text layout step outcome.
56    pub const fn new(
57        span_count: usize,
58        consumed: TextUnit,
59        carry: Option<TextCursor>,
60        fit: TextFit,
61    ) -> Self {
62        TextLayoutStep { span_count, consumed, carry, fit }
63    }
64}
65_impl_init![Self::new(0, TextUnit::INIT, None, TextFit::INIT) => TextLayoutStep];
