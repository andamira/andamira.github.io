8/// Construction and views.
9impl<'a> TextScanner<'a> {
10    /* constructors */
11
12    /// Creates a scanner over a string slice.
13    pub const fn new(string: &'a str) -> Self {
14        Self::from_bytes(string.as_bytes())
15    }
16    /// Creates a scanner over a byte slice.
17    pub const fn from_bytes(bytes: &'a [u8]) -> Self {
18        Self { bytes, cursor: TextCursor::INIT }
19    }
20
21    /* source views */
22
23    /// Returns the byte slice covered by `range`.
24    #[must_use]
25    pub const fn slice(&self, range: TextRange) -> &'a [u8] {
26        Slice::range(self.bytes, range.start.as_usize(), range.end.as_usize())
27    }
28
29    /// Returns the string slice covered by `range`, if valid UTF-8.
30    #[must_use]
31    pub const fn slice_str(&self, range: TextRange) -> Option<&'a str> {
32        unwrap![ok_some Str::from_utf8(self.slice(range))]
33    }
34    /// Returns the string slice covered by `range`, without checking if it's valid UTF-8.
35    ///
36    /// # Safety
37    /// The caller must ensure the covered bytes are valid UTF-8.
38    #[must_use]
39    #[cfg(all(not(feature = "safe_text"), feature = "unsafe_str"))]
40    pub const unsafe fn slice_str_unchecked(&self, range: TextRange) -> &'a str {
41        unsafe { Str::from_utf8_unchecked(self.slice(range)) }
42    }
43
44    /// Returns the remaining unread bytes.
45    #[must_use]
46    pub const fn rest(&self) -> &'a [u8] {
47        let pos = self.cursor.index.as_usize();
48        Slice::range_from(self.bytes, pos)
49    }
50}
51
52/// Cursor state and range construction.
53impl<'a> TextScanner<'a> {
54    // private cursor helpers
55    pub(super) const fn _cursor_bump(&mut self, n: u32) {
56        self.cursor = TextCursor::new_prim(self.cursor.index.0 + n);
57    }
58    pub(super) const fn _cursor_set(&mut self, pos: u32) {
59        self.cursor = TextCursor::new_prim(pos);
60    }
61
62    /// Returns the current cursor position.
63    pub const fn pos(&self) -> TextIndex {
64        self.cursor.index
65    }
66    /// Returns the current cursor for later range construction.
67    pub const fn mark(&self) -> TextCursor {
68        self.cursor
69    }
70
71    /// Returns the number of unread bytes remaining.
72    #[must_use]
73    pub const fn remaining_len(&self) -> TextUnit {
74        let pos = self.cursor.index.as_usize();
75        self.bytes.len().saturating_sub(pos) as TextUnit
76    }
77
78    /// Advances the cursor by up to `n` bytes.
79    ///
80    /// Returns the number of bytes actually consumed.
81    ///
82    /// If fewer than `n` bytes remain, advances to end of input.
83    #[must_use]
84    pub const fn advance(&mut self, n: TextUnit) -> TextUnit {
85        let consumed = crate::Cmp(n).min(self.remaining_len());
86        self._cursor_bump(consumed);
87        consumed
88    }
89
90    /// Returns `true` if the scanner reached the end of input.
91    #[must_use]
92    pub const fn is_eof(&self) -> bool {
93        self.cursor.index.as_usize() >= self.bytes.len()
94    }
95    /// Returns the half-open range from `mark` to the current cursor.
96    ///
97    /// # Panics
98    /// Panics if `mark` is ahead of the current cursor.
99    pub const fn range_from(&self, mark: TextCursor) -> TextRange {
100        TextRange::new(mark.index, self.cursor.index)
101    }
102    /// Consumes and returns the remaining unread range.
103    ///
104    /// After this call, the scanner is at end of input.
105    pub const fn take_rest(&mut self) -> TextRange {
106        let start = self.mark();
107        let _ = self.advance(self.remaining_len());
108        self.range_from(start)
109    }
110}

112/// Predicate-driven scanning adapters.
113impl<'a> TextScanner<'a> {
114    /// Consumes the next byte if it satisfies `f`.
115    ///
116    /// Returns `true` if a byte was consumed.
117    ///
118    /// This is the predicate-based counterpart to
119    /// [`eat_byte`][Self::eat_byte].
120    #[must_use]
121    pub fn eat_if<F: FnOnce(u8) -> bool>(&mut self, f: F) -> bool {
122        match self.peek_byte() {
123            Some(found) if f(found) => {
124                self._cursor_bump(1);
125                true
126            }
127            _ => false,
128        }
129    }
130
131    /// Advances while `byte` matches the predicate.
132    ///
133    /// Returns the number of consumed bytes.
134    pub fn skip_while<F: FnMut(u8) -> bool>(&mut self, mut f: F) -> TextUnit {
135        let start = self.cursor.index.0;
136        whilst! { let Some(byte) = self.peek_byte(); {
137            is! { !f(byte), break }
138            self._cursor_bump(1);
139        }}
140        self.cursor.index.0 - start
141    }
142
143    /// Consumes and returns the half-open range of bytes matching the predicate.
144    pub fn take_while<F: FnMut(u8) -> bool>(&mut self, mut f: F) -> TextRange {
145        let start = self.mark();
146        whilst! { let Some(byte) = self.peek_byte(); {
147            is! { !f(byte), break }
148            self._cursor_bump(1);
149        }}
150        self.range_from(start)
151    }
152}
