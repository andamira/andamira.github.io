3use crate::{TextCursor, TextParseError, TextRange, TextScanner, TextUnit};

6/// Byte inspection and exact consumption.
7impl<'a> TextScanner<'a> {
8    /// Returns the next byte without advancing.
9    #[must_use]
10    pub const fn peek_byte(&self) -> Option<u8> {
11        let pos = self.cursor.index.as_usize();
12        is! { pos < self.bytes.len(), Some(self.bytes[pos]), None }
13    }
14    /// Returns the byte at `offset` from the current cursor, without advancing.
15    ///
16    /// `offset == 0` returns the same value as [`peek_byte`][Self::peek_byte].
17    #[must_use]
18    pub const fn peek_byte_at(&self, offset: TextUnit) -> Option<u8> {
19        let start = self.cursor.index.as_usize();
20        let off = offset as usize;
21        match start.checked_add(off) {
22            Some(pos) if pos < self.bytes.len() => Some(self.bytes[pos]),
23            _ => None,
24        }
25    }
26
27    /// Returns `true` if the unread input starts with `bytes`.
28    ///
29    /// This method does not advance the scanner.
30    #[must_use]
31    pub const fn starts_with(&self, bytes: &[u8]) -> bool {
32        let start = self.cursor.index.as_usize();
33        let end = start + bytes.len();
34        is! { end > self.bytes.len(), return false }
35        whilst! { i in 0..bytes.len(); {
36            is! { self.bytes[start + i] != bytes[i], return false }
37        }}
38        true
39    }
40
41    /// Returns the next byte and advances the scanner.
42    #[must_use]
43    pub const fn next_byte(&mut self) -> Option<u8> {
44        let pos = self.cursor.index.as_usize();
45        if pos < self.bytes.len() {
46            let byte = self.bytes[pos];
47            self._cursor_bump(1);
48            Some(byte)
49        } else {
50            None
51        }
52    }
53
54    /// Advances the scanner by one byte.
55    ///
56    /// Returns `true` if a byte was consumed.
57    pub const fn skip_byte(&mut self) -> bool {
58        self.advance(1) == 1
59    }
60
61    #[must_use]
62    /// Consumes `byte` if it is next.
63    pub const fn eat_byte(&mut self, byte: u8) -> bool {
64        match self.peek_byte() {
65            Some(found) if found == byte => {
66                self._cursor_bump(1);
67                true
68            }
69            _ => false,
70        }
71    }
72    /// Consumes `bytes` if they are next.
73    #[must_use]
74    pub const fn eat_bytes(&mut self, bytes: &[u8]) -> bool {
75        let start = self.cursor.index.as_usize();
76        let end = start + bytes.len();
77        is! { end > self.bytes.len(), return false }
78        whilst! { i in 0..bytes.len(); {
79            is! { self.bytes[start + i] != bytes[i], return false }
80        }}
81        self._cursor_bump(bytes.len() as u32);
82        true
83    }
84
85    /// Consumes `byte` if it is next, or returns an error.
86    pub const fn expect_byte(&mut self, byte: u8) -> Result<(), TextParseError> {
87        match self.peek_byte() {
88            Some(found) if found == byte => {
89                self._cursor_bump(1);
90                Ok(())
91            }
92            found => Err(TextParseError::unexpected_byte(self.cursor, byte, found)),
93        }
94    }
95    /// Consumes `bytes` if they are next, or returns an error.
96    pub const fn expect_bytes(&mut self, bytes: &[u8]) -> Result<(), TextParseError> {
97        let start = self.cursor.index.as_usize();
98        let end = start + bytes.len();
99        is! { end > self.bytes.len(), return Err(TextParseError::unexpected_eof(self.cursor)) }
100        whilst! { i in 0..bytes.len(); {
101            let found = self.bytes[start + i];
102            let expected = bytes[i];
103            if found != expected {
104                let at = TextCursor::new_prim(self.cursor.index.0 + i as u32);
105                return Err(TextParseError::unexpected_byte(at, expected, Some(found)));
106            }
107        }}
108        self._cursor_bump(bytes.len() as u32);
109        Ok(())
110    }
111}

113/// Byte-delimited range scanning
114impl<'a> TextScanner<'a> {

116    /// Consumes and returns the range up to, but excluding, the next `byte`.
117    ///
118    /// If `byte` is not found, consumes to end of input.
119    pub const fn take_until_byte(&mut self, byte: u8) -> TextRange {
120        let start = self.mark();
121        whilst! { let Some(found) = self.peek_byte(); {
122            is! { found == byte, break }
123            self._cursor_bump(1);
124        }}
125        self.range_from(start)
126    }
127    /// Consumes and returns the range up to, but excluding, the next occurrence of `bytes`.
128    ///
129    /// If `bytes` is not found, consumes to end of input.
130    ///
131    /// If `bytes` is empty, returns an empty range and does not advance.
132    pub const fn take_until_bytes(&mut self, bytes: &[u8]) -> TextRange {
133        let start = self.mark();
134        is! { bytes.is_empty(), return self.range_from(start) }
135        let hay = self.bytes;
136        let needle_len = bytes.len();
137        let pos0 = self.cursor.index.as_usize();
138        if needle_len > hay.len().saturating_sub(pos0) {
139            self._cursor_set(hay.len() as u32);
140            return self.range_from(start);
141        }
142        whilst! { pos in pos0, ..=(hay.len() - needle_len); {
143            let mut matched = true;
144            whilst!{ i in 0..needle_len; {
145                if hay[pos + i] != bytes[i] {
146                    matched = false;
147                    break;
148                }
149            }}
150            if matched {
151                self._cursor_set(pos as u32);
152                return self.range_from(start);
153            }
154        }}
155        self._cursor_set(hay.len() as u32);
156        self.range_from(start)
157    }
158
159    /// Consumes and returns the range up to, but excluding, the next
160    /// occurrence of either `byte1` or `byte2`.
161    ///
162    /// If neither byte is found, consumes to end of input.
163    pub const fn take_until_any2(&mut self, byte1: u8, byte2: u8) -> TextRange {
164        let start = self.mark();
165        whilst! { let Some(byte) = self.peek_byte(); {
166            is! { byte == byte1 || byte == byte2, break }
167            self._cursor_bump(1);
168        }}
169        self.range_from(start)
170    }
171    /// Consumes and returns the range up to, but excluding, the next
172    /// occurrence of `byte1`, `byte2`, or `byte3`.
173    ///
174    /// If none of the bytes is found, consumes to end of input.
175    pub const fn take_until_any3(&mut self, byte1: u8, byte2: u8, byte3: u8) -> TextRange {
176        let start = self.mark();
177        whilst! { let Some(byte) = self.peek_byte(); {
178            is! { byte == byte1 || byte == byte2 || byte == byte3, break }
179            self._cursor_bump(1);
180        }}
181        self.range_from(start)
182    }
