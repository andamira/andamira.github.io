8/// ASCII whitespace, identifiers.
9impl<'a> TextScanner<'a> {
10    /// Skips ASCII horizontal and vertical whitespace.
11    ///
12    /// Currently skips: space, tab, line feed, carriage return,
13    /// form feed, and vertical tab.
14    pub const fn skip_ascii_ws(&mut self) {
15        whilst! { let Some(byte) = self.peek_byte(); {
16            match byte {
17                b' ' | b'\t' | b'\n' | b'\r' | 0x0C | 0x0B => self._cursor_bump(1),
18                _ => break,
19            }
20        }}
21    }
22    /// Skips ASCII horizontal whitespace.
23    ///
24    /// Currently skips: space, tab.
25    pub const fn skip_ascii_hws(&mut self) {
26        whilst! { let Some(byte) = self.peek_byte(); {
27            match byte {
28                b' ' | b'\t' => self._cursor_bump(1),
29                _ => break,
30            }
31        }}
32    }
33
34    /// Consumes and returns an ASCII identifier range.
35    ///
36    /// Identifier syntax:
37    /// - first byte: `A..=Z`, `a..=z`, or `_`
38    /// - following bytes: ASCII alphanumeric or `_`
39    ///
40    /// Returns `None` if the next byte is not a valid identifier start.
41    #[must_use]
42    pub const fn take_ascii_ident(&mut self) -> Option<TextRange> {
43        self.take_ascii_run(AsciiSet::IDENT_HEAD, AsciiSet::IDENT_TAIL)
44    }
45
46    /// Consumes and returns an ASCII identifier-tail range.
47    ///
48    /// Identifier-tail syntax:
49    /// - bytes: ASCII alphanumeric or `_`
50    ///
51    /// Unlike [`take_ascii_ident`][Self::take_ascii_ident],
52    /// this accepts digits as the first consumed byte.
53    ///
54    /// Returns `None` if no identifier-tail byte was consumed.
55    #[must_use]
56    pub const fn take_ascii_ident_tail(&mut self) -> Option<TextRange> {
57        self.take_ascii_set(AsciiSet::IDENT_TAIL)
58    }
59
60    /// Returns `range` trimmed of leading and trailing ASCII whitespace.
61    ///
62    /// Currently trims: space, tab, line feed, carriage return,
63    /// form feed, and vertical tab.
64    pub const fn trim_ascii_ws(&self, range: TextRange) -> TextRange {
65        let mut start = range.start.0;
66        let mut end = range.end.0;
67
68        whilst! { start < end; {
69            match self.bytes[start as usize] {
70                b' ' | b'\t' | b'\n' | b'\r' | 0x0C | 0x0B => start += 1,
71                _ => break,
72            }
73        }}
74        whilst! { start < end; {
75            match self.bytes[(end - 1) as usize] {
76                b' ' | b'\t' | b'\n' | b'\r' | 0x0C | 0x0B => end -= 1,
77                _ => break,
78            }
79        }}
80        TextRange::from_prim(start, end)
81    }
82
83    /// Returns `range` trimmed of leading and trailing ASCII horizontal whitespace.
84    ///
85    /// Currently trims: space, tab.
86    pub const fn trim_ascii_hws(&self, range: TextRange) -> TextRange {
87        let mut start = range.start.0;
88        let mut end = range.end.0;
89
90        whilst! { start < end; {
91            match self.bytes[start as usize] {
92                b' ' | b'\t' => start += 1,
93                _ => break,
94            }
95        }}
96        whilst! { start < end; {
97            match self.bytes[(end - 1) as usize] {
98                b' ' | b'\t' => end -= 1,
99                _ => break,
100            }
101        }}
102        TextRange::from_prim(start, end)
103    }
104}

106/// `AsciiSet` scanning.
107impl<'a> TextScanner<'a> {
108    /// Consumes the next byte if it belongs to `set`.
109    ///
110    /// Returns `true` if a byte was consumed.
111    ///
112    /// Non-ASCII bytes never match.
113    #[must_use]
114    pub const fn eat_ascii_set(&mut self, set: AsciiSet) -> bool {
115        match self.peek_byte() {
116            Some(byte) if set.contains_byte(byte) => {
117                self._cursor_bump(1);
118                true
119            }
120            _ => false,
121        }
122    }
123
124    /// Skips bytes while they belong to `set`.
125    ///
126    /// Returns the number of skipped bytes.
127    pub const fn skip_ascii_set(&mut self, set: AsciiSet) -> TextUnit {
128        let start = self.cursor.index.0;
129        whilst! { let Some(byte) = self.peek_byte(); {
130            is! { set.contains_byte(byte), self._cursor_bump(1), break }
131        }}
132        self.cursor.index.0 - start
133    }
134
135    /// Skips bytes until the next byte belonging to `set`.
136    ///
137    /// Stops before the matching byte.
138    ///
139    /// If no byte from `set` is found, skips to the end of input.
140    ///
141    /// Non-ASCII bytes never match.
142    pub const fn skip_until_ascii_set(&mut self, set: AsciiSet) -> TextUnit {
143        let start = self.cursor.index.0;
144        whilst! { let Some(byte) = self.peek_byte(); {
145            is! { set.contains_byte(byte), break }
146            self._cursor_bump(1);
147        }}
148        self.cursor.index.0 - start
149    }
150
151    /// Consumes and returns a range of bytes belonging to `set`.
152    ///
153    /// Returns `None` if no byte was consumed.
154    #[must_use]
155    pub const fn take_ascii_set(&mut self, set: AsciiSet) -> Option<TextRange> {
156        let start = self.mark();
157        whilst! { let Some(byte) = self.peek_byte(); {
158            is! { set.contains_byte(byte), self._cursor_bump(1), break }
159        }}
160        is! { self.cursor.index.0 == start.index.0, None, Some(self.range_from(start)) }
161    }
162
163    /// Consumes and returns an ASCII run with distinct head and tail sets.
164    ///
165    /// The first byte must belong to `head`.
166    /// Following bytes may belong to `tail`.
167    ///
168    /// Returns `None` if the first byte does not belong to `head`.
169    #[must_use]
170    pub const fn take_ascii_run(&mut self, head: AsciiSet, tail: AsciiSet) -> Option<TextRange> {
171        let start = self.mark();
172        // let Some(byte) = self.peek_byte() else { return None; };
173        // let byte = is![let Some(byte) = self.peek_byte(), byte, return None];
174        let byte = unwrap![some? self.peek_byte()];
175        is! { !head.contains_byte(byte), return None }
176        self._cursor_bump(1);
177        whilst! { let Some(byte) = self.peek_byte(); {
178            is! { tail.contains_byte(byte), self._cursor_bump(1), break }
179        }}
180        Some(self.range_from(start))
181    }
182
183    /// Consumes and returns the range up to, but excluding, the next byte belonging to `set`.
184    ///
185    /// If no byte from `set` is found, consumes to the end of input.
186    ///
187    /// Non-ASCII bytes never match.
188    pub const fn take_until_ascii_set(&mut self, set: AsciiSet) -> TextRange {
189        let start = self.mark();
190        whilst! { let Some(byte) = self.peek_byte(); {
191            is! { set.contains_byte(byte), break }
192            self._cursor_bump(1);
193        }}
194        self.range_from(start)
195    }
