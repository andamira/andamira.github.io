19/// A null-terminated UTF-8 string with up to `u8::MAX` bytes capacity.
20#[doc = crate::_doc_meta!{location("text/str")}]
21///
22/// Suited for write-once or protocol-level text where the length is rarely queried.
23/// Uses one less byte of storage, with length determined by scanning.
24/// For the opposite trade-off see [`StringU8`].
25///
26/// ## Methods
27///
28/// - [Constructors](#constructors):
29///   - [new](#method.new).
30///   - [from_str](#method.from_str),
31///     *([_truncate](#method.from_str_truncate),
32///       [_unchecked](#method.from_str_unchecked))*.
33///   - [from_char](#method.from_char)
34///     *([7](#method.from_char7),
35///       [8](#method.from_char8),
36///       [16](#method.from_char16),
37///       [utf8](#method.from_charu))*.
38///   - [from_array](#method.from_array)
39///     *([_unchecked](#method.from_array_unchecked)<sup title='unsafe function'>⚠</sup>)*.
40///
41/// - [Deconstructors](#deconstructors):
42///   - [into_array](#method.into_array).
43///   - [as_array](#method.as_array).
44///   - [as_bytes](#method.as_bytes)
45///     *([mut](#method.as_bytes_mut)<sup title='unsafe function'>⚠</sup>)*.
46///   - [as_str](#method.as_str)
47///     *([mut](#method.as_mut_str)<sup title='unsafe function'>⚠</sup>)*.
48///   - [chars](#method.chars).
49///
50/// - [Queries](#queries):
51///   - [len](#method.len).
52///   - [is_empty](#method.is_empty).
53///   - [is_full](#method.is_full).
54///   - [capacity](#method.capacity).
55///   - [remaining_capacity](#method.remaining_capacity).
56///
57/// - Operations:
58///   - [clear](#method.clear).
59///   - [pop](#method.pop)*([try](#method.try_pop))*.
60///   - [push](#method.push)*([try](#method.try_push))*.
61///   - [push_str](#method.push)*([try](#method.try_push_str))*.
62///   - [try_push_str_complete](#method.try_push_str_complete).

65pub struct StringNonul<const CAP: usize> {

67}

71    /// Creates a new empty `StringNonul`.
72    ///
73    /// # Panics
74    /// Panics if `CAP` > [`u8::MAX`].
75    ///
76    /// # Examples
77    /// ```
78    /// # use devela::StringNonul;
79    /// let mut s = StringNonul::<10>::new();
80    /// assert_eq![size_of_val(&s), 10];
81    /// ```
82    pub const fn new() -> Self {
83        assert![CAP <= u8::MAX as usize, "Mismatched capacity, greater than u8::MAX"];
84        Self { arr: [0; CAP] }
85    }

87    /// Creates a new empty `StringNonul`.
88    ///
89    /// # Errors
90    /// Returns [`MismatchedCapacity`] if `CAP` > [`u8::MAX`].
91    pub const fn new_checked() -> Result<Self, MismatchedCapacity> {
92        if CAP <= u8::MAX as usize {
93            Ok(Self { arr: [0; CAP] })
94        } else {
95            Err(MismatchedCapacity::too_large(CAP, u8::MAX as usize))
96        }
97    }

99    /// Creates a new `StringNonul` from a complete `&str`.
100    ///
101    /// # Errors
102    /// Returns [`MismatchedCapacity`] if `CAP` > [`u8::MAX`] or if `CAP < string.len()`.
103    ///
104    /// This is implemented via `Self::`[`try_push_str_complete()`](#method.try_push_str_complete).
105    ///
106    /// # Examples
107    /// ```
108    /// # use devela::StringU8;
109    /// let s = StringU8::<13>::from_str("Hello Wørld!").unwrap();
110    /// assert_eq![s.as_str(), "Hello Wørld!"];
111    /// ```
112    pub const fn from_str(string: &str) -> Result<Self, MismatchedCapacity> {
113        let mut new_string = unwrap![ok? Self::new_checked()];
114        if new_string.try_push_str_complete(string).is_ok() { Ok(new_string) }
115        else { Err(MismatchedCapacity::too_small(CAP, string.len())) }
116    }

152    /// Returns the current length.
153    ///
154    /// Time complexity: O(n) where n is the string length.
155    ///
156    /// # Examples
157    /// ```
158    /// # use devela::{StringNonul, MismatchedCapacity};
159    /// # fn main() -> Result<(), MismatchedCapacity> {
160    /// let mut s = StringNonul::<4>::new_checked()?;
161    /// assert_eq![0, s.len()];
162    ///
163    /// assert_eq![1, s.push('a')];
164    /// assert_eq![1, s.len()];
165    ///
166    /// assert_eq![3, s.push('€')];
167    /// assert_eq![4, s.len()];
168    /// # Ok(()) }
169    /// ```
170    #[must_use]
171    pub const fn len(&self) -> usize {
172        let mut position = 0;
173        while position < CAP {
174            is![self.arr[position] == 0, break];
175            position += 1;
176        }
177        position
178    }

188    /// Checks the equality of two strings, with the same capacity and length.
189    ///
190    /// It only checks the first `self.len()` bytes.
191    /// # Examples
192    /// ```
193    /// # use devela::StringNonul;
194    /// let mut a = StringNonul::<16>::from_str_unchecked("hello world!");
195    /// let mut b = StringNonul::<16>::from_str_unchecked("hello world!!!");
196    /// assert![!a.eq(&b)];
197    /// b.pop();
198    /// b.pop();
199    /// assert![a.eq(&b)];
200    /// ```
201    #[must_use] #[inline(always)]
202    pub const fn eq(&self, other: &Self) -> bool {
203        let mut i = 0;
204        while i < CAP {
205            if self.arr[i] != other.arr[i] { return false; }
206            if self.arr[i] == 0 { return true; }
207            i += 1;
208        }
209        true
210    }

305    /// Removes the last character and returns it.
306    ///
307    /// # Panics
308    /// Panics if the string is empty.
309    ///
310    /// # Examples
311    /// ```
312    /// # use devela::StringNonul;
313    /// let mut s = StringNonul::<16>::new();
314    /// s.push_str("hello worlð!");
315    /// assert_eq![s.len(), 13];
316    ///
317    /// assert_eq![s.pop_unchecked(), '!'];
318    /// assert_eq![s.len(), 12];
319    ///
320    /// assert_eq![s.pop_unchecked(), 'ð'];
321    /// assert_eq![s.len(), 10];
322    /// ```
323    #[must_use] #[rustfmt::skip]
324    pub const fn pop_unchecked(&mut self) -> char {
325        let (len, string) = (self.len(), self.as_str());
326        let mut idx_last_char = len - 1;
327        while idx_last_char > 0 && !string.is_char_boundary(idx_last_char) { idx_last_char -= 1; }
328
329        let range = Str::range(string, idx_last_char, len);
330        let last_char = unwrap![some CharIter::<&str>::new(range).next_char()];
331
332        let mut i = idx_last_char; while i < len { self.arr[i] = 0; i += 1; } // clean char bytes
333        last_char
334    }

336    /// Appends to the end of the string the given `character`.
337    ///
338    /// Returns the number of bytes written.
339    ///
340    /// It will return 0 bytes if the given `character` doesn't fit in
341    /// the remaining capacity, or if it is the nul character.
342    ///
343    /// # Examples
344    /// ```
345    /// # use devela::StringNonul;
346    /// let mut s = StringNonul::<16>::new();
347    /// s.push('h');
348    /// assert_eq![s.len(), 1];
349    ///
350    /// s.push('€');
351    /// assert_eq![s.len(), 4];
352    ///
353    /// assert_eq![s.as_str(), "h€"];
354    /// ```
355    pub const fn push(&mut self, character: char) -> usize {
356        let char_len = character.len_utf8();
357        if character != NUL_CHAR && self.remaining_capacity() >= char_len {
358            let len = self.len();
359            let _ = character.encode_utf8(slice![mut &mut self.arr, len, ..len + char_len]);
360            char_len
361        } else {
362            0
363        }
364    }

366    /// Tries to append to the end of the string the given `character`.
367    ///
368    /// Returns the number of bytes written.
369    ///
370    /// Trying to push a nul character does nothing and returns 0 bytes.
371    ///
372    /// # Errors
373    /// Returns [`NotEnoughSpace`]
374    /// if the available capacity is not enough to hold the given character.
375    pub const fn try_push(&mut self, character: char) -> Result<usize, NotEnoughSpace> {
376        let char_len = character.len_utf8();
377        if character == NUL_CHAR {
378            Ok(0)
379        } else if self.remaining_capacity() >= char_len {
380            let len = self.len();
381            let _ = character.encode_utf8(slice![mut &mut self.arr, len, ..len + char_len]);
382            Ok(char_len)
383        } else {
384            Err(NotEnoughSpace(Some(char_len)))
385        }
386    }

388    /// Appends to the end the fitting characters from the given `string` slice.
389    ///
390    /// Nul characters will be stripped out.
391    ///
392    /// Returns the number of bytes written, which will be 0
393    /// if not even the first non-nul character can fit.
394    ///
395    /// # Features
396    /// Uses the `unsafe_str` feature to skip validation checks.
397    pub const fn push_str(&mut self, string: &str) -> usize {
398        let mut rem_cap = self.remaining_capacity();
399        let mut bytes_written = 0;
400        let mut chars = CharIter::<&str>::new(string);
401        while let Some(character) = chars.next_char() {
402            if character != NUL_CHAR {
403                let char_len = character.len_utf8();
404                if char_len <= rem_cap {
405                    self.push(character);
406                    rem_cap -= char_len;
407                    bytes_written += char_len;
408                } else {
409                    break;
410                }
411            }
412        }
413        bytes_written
414    }

416    /// Tries to append to the end the fitting characters from the given `string` slice.
417    ///
418    /// Nul characters will be stripped out.
419    ///
420    /// Returns the number of bytes written.
421    ///
422    /// # Errors
423    /// Returns [`NotEnoughSpace`] if the capacity is not enough
424    /// to hold not even the first non-nul character.
425    ///
426    /// # Examples
427    /// ```
428    /// # use devela::StringNonul;
429    /// let mut s = StringNonul::<8>::new();
430    /// s.push_str("hello");
431    ///
432    /// // successfully appends a string while removing NUL characters
433    /// let result = s.try_push_str("\0\0\0\0\0 world");
434    /// assert!(result.is_ok());
435    /// assert_eq!(s.as_str(), "hello wo"); // truncates if it can't fit all
436    /// assert!(s.try_push_str("!").is_err()); // fails if it can't fit 1 char
437    ///
438    /// // Insufficient capacity for the first non-NUL character
439    /// let mut small = StringNonul::<3>::new();
440    /// assert!(small.try_push_str("🚀").is_err()); // Needs 4 bytes for the rocket
441    /// ```
442    /// # Features
443    /// Uses the `unsafe_str` feature to skip validation checks.
444    pub const fn try_push_str(&mut self, string: &str) -> Result<usize, NotEnoughSpace> {
445        let mut first_char_len = 0;
446        let mut chars = CharIter::<&str>::new(string);
447        while let Some(c) = chars.next_scalar() { // find the first non-zero length character:
448            if c != NUL_CHAR as u32 { first_char_len = Char(c).len_utf8_unchecked(); break; }
449        }
450        if self.remaining_capacity() < first_char_len {
451            Err(NotEnoughSpace(Some(first_char_len)))
452        } else {
453            Ok(self.push_str(string))
454        }
455    }

457    /// Tries to append the complete `string` slice to the end.
458    ///
459    /// Returns the number of bytes written in success.
460    ///
461    /// Nul characters will not be taken into account.
462    ///
463    /// # Errors
464    /// Returns [`NotEnoughSpace`] if the slice wont completely fit.
465    ///
466    /// # Features
467    /// Uses the `unsafe_str` feature to skip validation checks.
468    pub const fn try_push_str_complete(&mut self, string: &str) -> Result<usize, NotEnoughSpace> {
469        lets![mut non_nul_len=0, bytes=string.as_bytes(), str_len=bytes.len()];
470        whilst!{ i in 0..str_len; { is![bytes[i] != 0, non_nul_len += 1] }} // count !0 bytes
471        if self.remaining_capacity() >= non_nul_len {
472            Ok(self.push_str(string))
473        } else {
474            Err(NotEnoughSpace(Some(str_len)))
475        }
476    }

480    /// Creates a new `StringNonul` from a `char`.
481    ///
482    /// If `c` is NUL an empty string will be returned.
483    ///
484    /// # Errors
485    /// Returns [`MismatchedCapacity`] if `CAP` > [`u8::MAX`],
486    /// or  if `!c.is_nul()` and `CAP` < `c.`[`len_utf8()`][Char::len_utf8].
487    ///
488    /// Will always succeed if `CAP` >= 4.
489    /// # Examples
490    /// ```
491    /// # use devela::{StringNonul, char};
492    /// assert_eq![StringNonul::<4>::from_char('🐛').unwrap().as_str(), "🐛"];
493    /// assert_eq![StringNonul::<4>::from_char('\0').unwrap().as_str(), ""];
494    /// assert![StringNonul::<3>::from_char('🐛').is_err()];
495    /// ```
496    pub const fn from_char(c: char) -> Result<Self, MismatchedCapacity> {
497        let mut new = unwrap![ok? Self::new_checked()];
498        if c != '\0' {
499            let bytes = Char(c).to_utf8_bytes();
500            let len = Char(bytes[0]).len_utf8_unchecked();
501            is![CAP < len, return Err(MismatchedCapacity::too_small(CAP, len))];
502            new.arr[0] = bytes[0];
503            if len > 1 { new.arr[1] = bytes[1]; }
504            if len > 2 { new.arr[2] = bytes[2]; }
505            if len > 3 { new.arr[3] = bytes[3]; }
506        }
507        Ok(new)
508    }

511    ///
512    /// If `c`.[`is_nul()`][char7#method.is_nul] an empty string will be returned.
513    ///
514    /// # Errors
515    /// Returns [`MismatchedCapacity`] if `CAP` > [`u8::MAX`],
516    /// or if `!c.is_nul()` and `CAP` < 1.
517    ///
518    /// Will always succeed if `CAP` >= 1.
519    /// # Examples
520    /// ```
521    /// # use devela::{StringNonul, char7};
522    /// let s = StringNonul::<1>::from_char7(char7::try_from_char('@').unwrap()).unwrap();
523    /// assert_eq![s.as_str(), "@"];
524    /// let s = StringNonul::<1>::from_char7(char7::try_from_char('\0').unwrap()).unwrap();
525    /// assert_eq![s.as_str(), ""];
526    ///
527    /// assert![StringNonul::<0>::from_char7(char7::try_from_char('@').unwrap()).is_err()];
528    /// ```
529    pub const fn from_char7(c: char7) -> Result<Self, MismatchedCapacity> {
530        is![CAP == 0, return Err(MismatchedCapacity::too_small(CAP, 1))];
531        let mut new = unwrap![ok? Self::new_checked()];
532        if !c.is_nul() {
533            new.arr[0] = c.to_utf8_bytes()[0];
534        }
535        Ok(new)
536    }

538    /// Creates a new `StringNonul` from a `char8`.
539    ///
540    /// If `c`.[`is_nul()`][char8#method.is_nul] an empty string will be returned.
541    ///
542    /// # Errors
543    /// Returns [`MismatchedCapacity`] if `CAP` > [`u8::MAX`],
544    /// or if `!c.is_nul()` and `CAP` < `c.`[`len_utf8()`][char8#method.len_utf8].
545    ///
546    /// Will always succeed if `CAP` >= 2.
547    /// # Examples
548    /// ```
549    /// # use devela::{StringNonul, char8};
550    /// let s = StringNonul::<2>::from_char8(char8::try_from_char('ß').unwrap()).unwrap();
551    /// assert_eq![s.as_str(), "ß"];
552    /// let s = StringNonul::<2>::from_char8(char8::try_from_char('\0').unwrap()).unwrap();
553    /// assert_eq![s.as_str(), ""];
554    ///
555    /// assert![StringNonul::<1>::from_char8(char8::try_from_char('ß').unwrap()).is_err()];
556    /// ```
557    pub const fn from_char8(c: char8) -> Result<Self, MismatchedCapacity> {
558        let mut new = unwrap![ok? Self::new_checked()];
559        if !c.is_nul() {
560            let bytes = c.to_utf8_bytes();
561            let len = Char(bytes[0]).len_utf8_unchecked();
562            is![CAP < len, return Err(MismatchedCapacity::too_small(CAP, len))];
563            new.arr[0] = bytes[0];
564            if len > 1 { new.arr[1] = bytes[1]; }
565        }
566        Ok(new)
567    }

581    /// let s = StringNonul::<3>::from_char16(char16::try_from_char('€').unwrap()).unwrap();
582    /// assert_eq![s.as_str(), "€"];
583    /// let s = StringNonul::<3>::from_char16(char16::try_from_char('\0').unwrap()).unwrap();
584    /// assert_eq![s.as_str(), ""];
585    ///
586    /// assert![StringNonul::<2>::from_char16(char16::try_from_char('€').unwrap()).is_err()];
587    /// ```
588    pub const fn from_char16(c: char16) -> Result<Self, MismatchedCapacity> {
589        let mut new = unwrap![ok? Self::new_checked()];
590        if !c.is_nul() {
591            let bytes = c.to_utf8_bytes();
592            let len = Char(bytes[0]).len_utf8_unchecked();
593            is![CAP < len, return Err(MismatchedCapacity::too_small(CAP, len))];
594            new.arr[0] = bytes[0];
595            if len > 1 { new.arr[1] = bytes[1]; }
596            if len > 2 { new.arr[2] = bytes[2]; }
597        }
598        Ok(new)
599    }

601    /// Creates a new `StringNonul` from a `charu`.
602    ///
603    /// If `c`.[`is_nul()`][charu#method.is_nul] an empty string will be returned.
604    ///
605    /// # Errors
606    /// Returns [`MismatchedCapacity`] if `CAP` > [`u8::MAX`],
607    /// or if `CAP` < `c.`[`len_utf8()`][charu#method.len_utf8].
608    ///
609    /// Will always succeed if `CAP` >= 4.
610    /// # Examples
611    /// ```
612    /// # use devela::{StringNonul, charu};
613    /// let s = StringNonul::<4>::from_charu(charu::from_char('🐛')).unwrap();
614    /// assert_eq![s.as_str(), "🐛"];
615    /// let s = StringNonul::<4>::from_charu(charu::from_char('\0')).unwrap();
616    /// assert_eq![s.as_str(), ""];
617    ///
618    /// assert![StringNonul::<3>::from_charu(charu::from_char('🐛')).is_err()];
619    /// ```
620    pub const fn from_charu(c: charu) -> Result<Self, MismatchedCapacity> {
621        let mut new = unwrap![ok? Self::new_checked()];
622        if !c.is_nul() {
623            let len = c.len_utf8();
624            if len <= CAP {
625                let bytes = c.to_utf8_bytes();
626                slice![mut &mut new.arr, 0,..len].copy_from_slice(slice![&bytes, 0,..len]);
627            } else {
628                return Err(MismatchedCapacity::too_small(CAP, len));
629            }
630        }
631        Ok(new)
632    }

835    impl<const CAP: usize> TryFrom<&str> for StringNonul<CAP> {
836        type Error = MismatchedCapacity;
837
838        /// Tries to create a new `StringNonul` from the given string slice.
839        ///
840        /// # Errors
841        /// Returns [`MismatchedCapacity`] if `CAP > `[`u8::MAX`] or if `CAP < str.len()`.
842        fn try_from(string: &str) -> Result<Self, MismatchedCapacity> {
843            let non_nul_len = string.as_bytes().iter().filter(|x| **x != 0).count();
844            if CAP < non_nul_len {
845                Err(MismatchedCapacity::too_small(CAP, non_nul_len))
846            } else {
847                let mut new_string = Self::new_checked()?;
848                let copied_bytes = new_string.push_str(string);
849                debug_assert_eq![non_nul_len, copied_bytes];
850                Ok(new_string)
851            }
852        }
853    }

855    impl<const CAP: usize> TryFrom<&[u8]> for StringNonul<CAP> {
856        type Error = InvalidText;
857
858        /// Tries to create a new `StringNonul` from the given slice of `bytes`.
859        ///
860        /// The string will stop before the first nul character or the end of the slice.
861        ///
862        /// # Errors
863        /// Returns [`InvalidText::MismatchedCapacity`] if `CAP > `[u8::MAX`]
864        /// or if `CAP < bytes.len()`,
865        /// or [`InvalidText::Utf8`] if the `bytes` are not valid UTF-8.
866        fn try_from(bytes: &[u8]) -> Result<Self, InvalidText> {
867            let len = bytes.len();
868            if len >= CAP { return Err(MismatchedCapacity::too_small(CAP, len).into()); }
869            // IMPROVE: use Str
870            match ::core::str::from_utf8(bytes) {
871                Ok(_) => {
872                    let mut arr = [0; CAP];
873                    let mut idx = 0;
874                    for &byte in bytes.iter() {
875                        if byte != 0 {
876                            arr[idx] = byte;
877                            idx += 1;
878                        }
879                    }
880                    Ok(Self { arr })
881                }
882                Err(e) => Err(InvalidText::from_utf8_error(e)),
883            }
884        }
885    }
