8use crate::Str;
9#[cfg(feature = "alloc")]
10use crate::{Arc, Box, Rc};
11crate::_use! {basic::from_utf8}
12
13/// Marker trait to prevent downstream implementations of the [`StrExt`] trait.
14trait Sealed {}
15impl Sealed for str {}
16
17#[doc = crate::_tags!(string)]
18/// Extension trait providing additional methods for [`&str`].
19#[cfg_attr(nightly_doc, doc(notable_trait))]
20#[expect(private_bounds, reason = "Sealed")]
21pub trait StrExt: Sealed {
22    /// Converts the string slice into a `Box<str>`.
23    ///
24    /// Allows single ownership with exact allocation,
25    /// for when you don't need to clone or share.
26    #[cfg(feature = "alloc")]
27    fn to_box(&self) -> Box<str>;
28
29    /// Converts the string slice into an `Rc<str>`.
30    ///
31    /// Allows shared ownership with reference counting,
32    /// reducing memory duplication in single-threaded scenarios.
33    #[cfg(feature = "alloc")]
34    fn to_rc(&self) -> Rc<str>;
35
36    /// Converts the string slice into an `Arc<str>`.
37    ///
38    /// When you need shared ownership of a string slice across multiple threads.
39    #[cfg(feature = "alloc")]
40    fn to_arc(&self) -> Arc<str>;
41
42    /// Repeats a string a given number of times into the provided `buffer`.
43    /// and returns a reference to the new `&str`.
44    /// # Examples
45    /// ```
46    /// use devela::StrExt;
47    ///
48    /// let mut buf = [0_u8; 12];
49    /// let repeated = "ay".repeat_into(3, &mut buf);
50    /// assert_eq![repeated, "ayayay"];
51    /// ```
52    /// # Features
53    /// Makes use of the `unsafe_str` feature if enabled.
54    ///
55    /// For the *const* version see [`Str::repeat_into`].
56    #[must_use]
57    fn repeat_into<'input, const CAP: usize>(
58        &self,
59        n: usize,
60        buffer: &'input mut [u8; CAP],
61    ) -> &'input str;
62
63    /// Repeats `string` up to `n` times into `buffer`,
64    /// stopping early if it does not fit,
65    /// and returns the written prefix as `&str`.
66    ///
67    /// Like [`repeat_into`][Self::repeat_into], but accepts a dynamically sized buffer.
68    fn repeat_into_slice<'input, const CAP: usize>(
69        &self,
70        n: usize,
71        buffer: &'input mut [u8],
72    ) -> &'input str;
73
74    /// Returns a [`&str`] backed by a `buffer`, where you always know each
75    /// character's position.
76    ///
77    /// A [*counter string*][0] is a graduated string of arbitrary `length`,
78    /// with a `separator` positioned after the immediately preceding number.
79    /// # Examples
80    /// ```
81    /// use devela::StrExt;
82    ///
83    /// let mut buf = [0; 15];
84    /// assert_eq!("2*4*6*8*11*14*", str::new_counter(&mut buf, 14, '*'));
85    /// assert_eq!("_3_5_7_9_12_15_", str::new_counter(&mut buf, 15, '_'));
86    /// ```
87    /// # Panics
88    /// Panics if `buffer.len() < length`, or if `!separator.is_ascii()`.
89    ///
90    /// # Features
91    /// Makes use of the `unsafe_str` feature if enabled.
92    ///
93    /// For the *const* version see [`Str::new_counter`].
94    ///
95    /// [0]: https://www.satisfice.com/blog/archives/22
96    #[must_use]
97    fn new_counter(buffer: &mut [u8], length: usize, separator: char) -> &str;
98}
99
100impl StrExt for str {
101    #[cfg(feature = "alloc")]
102    fn to_box(&self) -> Box<str> {
103        Box::from(self)
104    }
105    #[cfg(feature = "alloc")]
106    fn to_rc(&self) -> Rc<str> {
107        Rc::from(self)
108    }
109    #[cfg(feature = "alloc")]
110    fn to_arc(&self) -> Arc<str> {
111        Arc::from(self)
112    }
113
114    fn repeat_into<'input, const CAP: usize>(
115        &self,
116        n: usize,
117        buffer: &'input mut [u8; CAP],
118    ) -> &'input str {
119        Str::repeat_into(self, n, buffer)
120    }
121    fn repeat_into_slice<'input, const CAP: usize>(
122        &self,
123        n: usize,
124        buffer: &'input mut [u8],
125    ) -> &'input str {
126        Str::repeat_into_slice(self, n, buffer)
127    }
128
129    fn new_counter(buffer: &mut [u8], length: usize, separator: char) -> &str {
130        Str::new_counter(buffer, length, separator)
131    }
132}
