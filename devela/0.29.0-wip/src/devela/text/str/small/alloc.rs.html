6use crate::{MismatchedCapacity, NotEnoughElements, String, StringU8};

8// (15 in 64-bit, 7 in 32-bit, 3 in 16 bit)
9const STRING_SMALL_ALLOC_DEFAULT_CAP: usize = 2 * size_of::<usize>() - 1;
10
11#[doc = crate::_tags!(string)]
12/// A UTF-8 string with inline storage and heap spillover.
13#[doc = crate::_doc_meta!{location("text/str")}]
14///
15/// Stores strings of up to `CAP` bytes inline,
16/// and spills to the heap when more capacity is needed.
17///
18/// The default capacity, `23`, matches the usual inline payload target for
19/// a `String`-sized compact string on 64-bit platforms.
20#[must_use]
21#[derive(Clone)]
22pub struct StringSmallAlloc<const CAP: usize = STRING_SMALL_ALLOC_DEFAULT_CAP> {
23    repr: StringSmallAllocRepr<CAP>,
24}
25#[derive(Clone)]
26enum StringSmallAllocRepr<const CAP: usize> {
27    Inline(StringU8<CAP>),
28    Heap(String),
29}
30
31impl<const CAP: usize> StringSmallAlloc<CAP> {

40    /// occupies the same stack size as `String`.

43    /// Inline capacity that makes `StringU8<CAP>` the same size as `String`.

48    /// the same size as `String`.

50    pub const STRING_SIZED_CAPACITY: usize = 2 * size_of::<usize>() - 1;

54    /// Creates a new empty string.
55    ///
56    /// # Panics
57    /// Panics if `CAP > StringU8::MAX_CAPACITY`.
58    #[inline(always)]
59    pub const fn new() -> Self {
60        Self {
61            repr: StringSmallAllocRepr::Inline(StringU8::<CAP>::new()),
62        }
63    }
64    /// Creates a new empty string.
65    ///
66    /// # Errors
67    /// Returns [`MismatchedCapacity`] if `CAP > StringU8::MAX_CAPACITY`.
68    pub fn new_checked() -> Result<Self, MismatchedCapacity> {
69        Ok(Self {
70            repr: StringSmallAllocRepr::Inline(StringU8::<CAP>::new_checked()?),
71        })
72    }

74    /// Creates a new string from `string`.
75    ///
76    /// Uses inline storage when `string.len() <= CAP`, otherwise spills to the heap.
77    ///
78    /// # Errors
79    /// Returns [`MismatchedCapacity`] if `CAP > StringU8::MAX_CAPACITY`.
80    #[allow(clippy::should_implement_trait, reason = "it does as well")]
81    pub fn from_str(string: &str) -> Result<Self, MismatchedCapacity> {
82        let _ = StringU8::<CAP>::new_checked()?;
83        if string.len() <= CAP {
84            Ok(Self {
85                repr: StringSmallAllocRepr::Inline(StringU8::<CAP>::from_str(string)?),
86            })
87        } else {
88            Ok(Self {
89                repr: StringSmallAllocRepr::Heap(String::from(string)),
90            })
91        }
92    }
93
94    /// Creates a new string from `string`.
95    ///
96    /// Reuses `string` directly when it does not fit inline.
97    ///
98    /// # Errors
99    /// Returns [`MismatchedCapacity`] if `CAP > StringU8::MAX_CAPACITY`.
100    pub fn from_string(string: String) -> Result<Self, MismatchedCapacity> {
101        let _ = StringU8::<CAP>::new_checked()?;
102        if string.len() <= CAP {
103            Ok(Self {
104                repr: StringSmallAllocRepr::Inline(StringU8::<CAP>::from_str(string.as_str())?),
105            })
106        } else {
107            Ok(Self { repr: StringSmallAllocRepr::Heap(string) })
108        }
109    }

200    /// Clears the string.
201    ///
202    /// If the string has spilled to the heap, the heap allocation is retained.

204    pub fn clear(&mut self) {
205        match &mut self.repr {
206            StringSmallAllocRepr::Inline(s) => s.clear(),
207            StringSmallAllocRepr::Heap(s) => s.clear(),
208        }
209    }

211    /// Truncates the string to `new_len` bytes.
212    ///
213    /// Does nothing if `new_len >= self.len()`.
214    ///
215    /// # Panics
216    /// Panics if `new_len` is not a UTF-8 boundary.
217    pub fn truncate(&mut self, new_len: usize) {
218        if new_len >= self.len() {
219            return;
220        }
221        assert![self.as_str().is_char_boundary(new_len), "not a UTF-8 boundary"];
222
223        match &mut self.repr {
224            StringSmallAllocRepr::Inline(s) => {
225                while s.len() > new_len {
226                    let _ = s.pop();
227                }
228            }
229            StringSmallAllocRepr::Heap(s) => s.truncate(new_len),
230        }
231    }
232
233    /// Reserves capacity for at least `additional` more bytes.
234    ///
235    /// Spills to the heap if the current inline storage is not enough.
236    pub fn reserve(&mut self, additional: usize) {
237        if additional <= self.remaining_capacity() {
238            return;
239        }
240
241        match &mut self.repr {
242            StringSmallAllocRepr::Inline(s) => {
243                let mut heap = String::with_capacity(s.len() + additional);
244                heap.push_str(s.as_str());
245                self.repr = StringSmallAllocRepr::Heap(heap);
246            }
247            StringSmallAllocRepr::Heap(s) => s.reserve(additional),
248        }
249    }
250
251    /// Moves the string back inline if it fits.
252    ///
253    /// Returns `true` if the string is inline after the call.
254    pub fn shrink_to_inline(&mut self) -> bool {
255        match &self.repr {
256            StringSmallAllocRepr::Inline(_) => true,
257            StringSmallAllocRepr::Heap(s) if s.len() <= CAP => {
258                let inline = StringU8::<CAP>::from_str(s.as_str())
259                    .expect("validated StringSmallAlloc inline capacity");
260                self.repr = StringSmallAllocRepr::Inline(inline);
261                true
262            }
263            StringSmallAllocRepr::Heap(_) => false,
264        }
265    }
266
267    /// Shrinks heap allocation if spilled, or moves back inline if possible.
268    pub fn shrink_to_fit(&mut self) {
269        if self.shrink_to_inline() {
270            return;
271        }
272        if let StringSmallAllocRepr::Heap(s) = &mut self.repr {
273            s.shrink_to_fit();
274        }
275    }
276
277    /// Removes the last character and returns it.
278    #[must_use]
279    pub fn pop(&mut self) -> Option<char> {
280        match &mut self.repr {
281            StringSmallAllocRepr::Inline(s) => s.pop(),
282            StringSmallAllocRepr::Heap(s) => s.pop(),
283        }
284    }
285
286    /// Tries to remove the last character and return it.
287    ///
288    /// # Errors
289    /// Returns [`NotEnoughElements`] if the string is empty.
290    pub fn try_pop(&mut self) -> Result<char, NotEnoughElements> {
291        match self.pop() {
292            Some(c) => Ok(c),
293            None => Err(NotEnoughElements(Some(1))),
294        }
295    }
296
297    /// Appends `character`, spilling to the heap if needed.
298    ///
299    /// Returns the number of bytes written.
300    pub fn push(&mut self, character: char) -> usize {
301        let len = character.len_utf8();
302
303        match &mut self.repr {
304            StringSmallAllocRepr::Inline(s) => {
305                if s.try_push(character).is_ok() {
306                    len
307                } else {
308                    let mut heap = String::with_capacity(s.len() + len);
309                    heap.push_str(s.as_str());
310                    heap.push(character);
311                    self.repr = StringSmallAllocRepr::Heap(heap);
312                    len
313                }
314            }
315            StringSmallAllocRepr::Heap(s) => {
316                s.push(character);
317                len
318            }
319        }
320    }
321
322    /// Appends `string`, spilling to the heap if needed.
323    ///
324    /// Returns the number of bytes written.
325    pub fn push_str(&mut self, string: &str) -> usize {
326        if string.is_empty() {
327            return 0;
328        }
329
330        match &mut self.repr {
331            StringSmallAllocRepr::Inline(s) => {
332                if let Ok(n) = s.try_push_str_complete(string) {
333                    n
334                } else {
335                    let mut heap = String::with_capacity(s.len() + string.len());
336                    heap.push_str(s.as_str());
337                    heap.push_str(string);
338                    self.repr = StringSmallAllocRepr::Heap(heap);
339                    string.len()
340                }
341            }
342            StringSmallAllocRepr::Heap(s) => {
343                s.push_str(string);
344                string.len()
345            }
346        }
347    }
