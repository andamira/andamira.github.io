8#[doc = crate::_tags!(fmt num)]
9/// Numeric digit grouping configuration.
10#[doc = crate::_doc_meta!{location("text/fmt")}]
11///
12/// This type controls structural digit grouping for formatted numbers,
13/// such as thousands separators, without performing layout or localization.
14///
15/// Grouping is applied independently to the integral (left) and fractional
16/// (right) digit sequences, counting from the radix point outward.
17///
18/// A grouping length of `0` disables grouping on that side,
19/// as well as having a `None` separator for that side.
20///
21/// This configuration is numeric-decoration only:
22/// it does not handle spacing, alignment, text direction, or locale rules.
23///
24/// Grouping assumes 1-byte separators and ASCII digits.
25#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
26pub struct FmtNumGroup {
27    /// Number of digits per group on the integral (left) side.
28    pub left_len: u8,
29    /// Number of digits per group on the fractional (right) side.
30    pub right_len: u8,
31    /// Separator glyph inserted between digit groups on the integral (left) side.
32    pub left_sep: Option<char>,
33    /// Separator glyph inserted between digit groups on the fractional (right) side.
34    pub right_sep: Option<char>,
35}
36#[rustfmt::skip]
37impl Default for FmtNumGroup { fn default() -> Self { Self::INIT } }
38impl ConstInit for FmtNumGroup {
39    const INIT: Self = Self {
40        left_len: 3,
41        right_len: 0,
42        left_sep: None,
43        right_sep: None,
44    };
45}

47#[rustfmt::skip]
48impl FmtNumGroup {
49    const _SIZE: () = const { assert![size_of::<Self>() == 12]; };
50
51    /* constructors */
52
53    /// Creates a default grouping configuration of 3 for the left part, and no separator.
54    #[inline(always)]
55    pub const fn new() -> Self { Self::INIT }
56
57    /// Sets the `_` underscore for the left part, with groups of 3.
58    pub const fn rust() -> Self { Self::INIT.with_sep('_') }
59    ///
60    /// Sets the `_` underscore for both parts, with groups of 3.
61    pub const fn both_rust() -> Self { Self::INIT.with_sep('_').with_len(3) }
62
63    /// Sets the `_` underscore for the left part, with groups of 4.
64    pub const fn rust_hex() -> Self { Self::INIT.with_left(4, '_') }
65
66    /// Sets the `_` underscore for both parts, with groups of 4.
67    pub const fn both_rust_hex() -> Self { Self::INIT.with_sep('_').with_len(4) }
68
69    /* individual lengths */
70
71    /// Returns a copy with the left length set to the given value.
72    pub const fn with_left_len(mut self, len: u8) -> Self { self.left_len = len; self }
73
74    /// Returns a copy with the right length set to the given value.
75    pub const fn with_right_len(mut self, len: u8) -> Self { self.right_len = len; self }
76
77    /* individual separators */
78
79    /// Returns a copy with the left separator set to the given character.
80    pub const fn with_left_sep(mut self, sep: char) -> Self { self.left_sep = Some(sep); self }
81    /// Return a copy with the left separator cleared.
82    pub const fn without_left_sep(mut self) -> Self { self.left_sep = None; self }
83
84    /// Returns a copy with the right separator set to the given character.
85    pub const fn with_right_sep(mut self, sep: char) -> Self { self.right_sep = Some(sep); self }
86    /// Return a copy with the right separator cleared.
87    pub const fn without_right_sep(mut self) -> Self { self.right_sep = None; self }
88
89    /* grouped */
90
91    /// Returns a copy with both lenghts set to different values.
92    pub const fn with_len(mut self, len: u8) -> Self {
93        self.left_len = len; self.right_len = len; self }
94
95    /// Returns a copy with both separators set to the same character.
96    pub const fn with_sep(mut self, sep: char) -> Self {
97        self.left_sep = Some(sep); self.right_sep = Some(sep); self }
98    /// Returns a copy with both separators cleared.
99    pub const fn without_sep(mut self) -> Self {
100        self.left_sep = None; self.right_sep = None; self }

102    /* sides */
103
104    /// Returns whether left-side grouping is enabled.
105    pub const fn has_left(&self) -> bool { self.left_len != 0 && self.left_sep.is_some() }
106
107    /// Returns whether right-side grouping is enabled.
108    pub const fn has_right(&self) -> bool { self.right_len != 0 && self.right_sep.is_some() }
109
110    /// Returns whether any grouping is enabled.
111    pub const fn is_enabled(&self) -> bool { self.has_left() || self.has_right() }
112
113    /// Returns a copy with the left length and separator set to the given values.
114    pub const fn with_left(mut self, len: u8, sep: char) -> Self {
115        self.left_len = len; self.left_sep = Some(sep); self }
116
117    /// Returns a copy with the right length and separator set to the given values.
118    pub const fn with_right(mut self, len: u8, sep: char) -> Self {
119        self.right_len = len; self.right_sep = Some(sep); self }

122impl FmtNumGroup {
123    /// Returns the grouping length for the given boundary side.
124    ///
125    /// For example, on the lower (left) side this corresponds to the
126    /// number of digits per group in the integral part.
127    pub const fn side_len(&self, side: Boundary1d) -> u8 {
128        match side {
129            Boundary1d::Lower => self.left_len,
130            Boundary1d::Upper => self.right_len,
131        }
132    }
133
134    /// Returns the grouping separator for the given boundary side, if any.
135    pub const fn side_sep(&self, side: Boundary1d) -> Option<char> {
136        match side {
137            Boundary1d::Lower => self.left_sep,
138            Boundary1d::Upper => self.right_sep,
139        }
140    }
141
142    /// Returns whether digit grouping is enabled on the given boundary side.
143    ///
144    /// Grouping is enabled when both a non-zero grouping length and
145    /// a separator are configured for that side.
146    pub const fn side_enabled(&self, side: Boundary1d) -> bool {
147        self.side_len(side) != 0 && self.side_sep(side).is_some()
148    }
149
150    /// Computes the minimal number of digits required so that, after grouping,
151    /// the rendered width on the given boundary side is at least `min_width`.
152    ///
153    /// This accounts only for digit grouping on the specified side and does not
154    /// include any sign or radix separator.
155    pub const fn digits_for_grouped_width(
156        &self,
157        side: Boundary1d,
158        digit_count: u16,
159        min_width: u16,
160    ) -> u16 {
161        is![!self.side_enabled(side), return Cmp(digit_count).max(min_width)];
162        let len = self.side_len(side) as u16;
163        let mut digits = Cmp((min_width * len) / (len + 1)).max(digit_count);
164        let seps = (digits.saturating_sub(1)) / len;
165        is![digits + seps < min_width, digits += 1];
166        digits
167    }
