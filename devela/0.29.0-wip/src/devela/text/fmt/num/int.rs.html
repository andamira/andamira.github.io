18            /// Writes the integer as ASCII decimal digits into `buf` starting at `pos`.
19            ///
20            /// Returns the number of bytes written, or `0` if the buffer is too small.
21            ///
22            /// # Behavioral guarantees
23            /// - Negative values are preceded by the `'-'` sign.
24            /// - The operation is atomic: on failure, nothing is written.
25            #[rustfmt::skip]
26            pub const fn write(self, buf: &mut [u8], mut pos: usize) -> usize {
27                use $crate::{Digits, write_at};
28                if self.0 < 0 {
29                    let digits = Digits(self.0.wrapping_neg().cast_unsigned()); // abs
30                    let needed = digits.count_digits10() as usize + 1;
31                    if needed > buf.len().saturating_sub(pos) { return 0; }
32                    write_at![buf, +=pos, b'-'];
33                    digits.write_digits10(buf, pos) + 1
34                } else {
35                    let digits = Digits(self.0.cast_unsigned());
36                    let needed = digits.count_digits10() as usize;
37                    if needed > buf.len().saturating_sub(pos) { return 0; }
38                    digits.write_digits10(buf, pos)
39                }
40            }

42            /// Writes the integer as ASCII decimal digits into `buf` starting at `pos`,
43            /// using the given formatting configuration.
44            ///
45            /// Returns the number of bytes written, or `0` if the buffer is too small.
46            ///
47            /// # Behavioral guarantees
48            /// - The emitted sign and any leading zero-padding are controlled by `conf`.
49            /// - The operation is atomic: on failure, nothing is written.
50            #[rustfmt::skip]
51            pub const fn write_fmt(self, buf: &mut [u8], mut pos: usize, conf: $crate::FmtNumConf)
52                -> usize {
53                use $crate::{Cmp, Digits, is, whilst, write_at};
54                let neg = self.0 < 0;
55                let emit_sign = conf.sign.emit_sign(neg);
56                let abs = is![neg, self.0.wrapping_neg().cast_unsigned(), self.0.cast_unsigned()];
57                // digit count
58                let digit_count = Digits(abs).count_digits10() as u16;
59                let left_digits = if conf.pad_sign && emit_sign {
60                    Cmp(digit_count + 1).max(conf.int) - 1
61                } else {
62                    Cmp(digit_count).max(conf.int)
63                };
64                // compute required space
65                let needed = (emit_sign as usize) + left_digits as usize;
66                if needed > buf.len().saturating_sub(pos) { return 0; }
67                // emit sign, zero padding and digits
68                if emit_sign { write_at![buf, +=pos, is![neg, b'-', b'+']]; }
69                whilst! { _i in 0..(left_digits - digit_count); { write_at![buf, +=pos, b'0']; }}
70                let _ = Digits(abs).write_digits10(buf, pos);
71                needed
72            }

74            /// Writes the formatted integer with optional digit grouping.
75            ///
76            /// Grouping is applied to the *rendered* integral digit sequence,
77            /// after sign emission and zero-padding have been accounted for.
78            ///
79            /// # Behavioral guarantees
80            /// - Grouping assumes 1-byte separators and ASCII digits.
81            /// - `conf.int` specifies the minimum width of the *entire* left block
82            ///   (digits + zero padding + grouping separators), excluding the sign.
83            /// - On failure, nothing is written (atomic operation).
84            pub const fn write_group(self, buf: &mut [u8], pos: usize, conf: $crate::FmtNumConf,
85                group: $crate::FmtNumGroup) -> usize {
86                use $crate::{Boundary1d, Digits, is, unwrap};
87                // Fast path: grouping disabled.
88                if !group.has_left() { return self.write_fmt(buf, pos, conf); }
89
90                // Determine whether a sign is emitted.
91                let neg = self.0 < 0;
92                let emit_sign = conf.sign.emit_sign(neg);
93
94                // Absolute value and digit count of the numeric magnitude.
95                let abs = is![neg, self.0.wrapping_neg().cast_unsigned(), self.0.cast_unsigned()];
96                let digit_count = Digits(abs).count_digits10() as u16;
97
98                // Target minimum width of the left block (excluding sign unless padded).
99                let mut target_left = conf.int;
100                if conf.pad_sign && emit_sign && target_left > 0 { target_left -= 1; }
101
102                // Compute minimal digit count so that: digits + grouping separators >= target_left.
103                let len = group.left_len as u16;
104                let left_digits = group.digits_for_grouped_width(Boundary1d::Left,
105                    digit_count, target_left);
106
107                // Translate back to write_fmt semantics.
108                let conf2_int = is![conf.pad_sign && emit_sign, left_digits + 1, left_digits];
109                let conf2 = conf.with_int(conf2_int);
110
111                // Measure final layout.
112                let raw_shape = self.measure_fmt(conf2);
113                let g_shape = self.measure_group(conf2, group);
114                let (raw_len, final_len) = (raw_shape.total(), g_shape.total());
115                if final_len > buf.len().saturating_sub(pos) { return 0; }
116
117                // Write ungrouped number at the start.
118                let written = self.write_fmt(buf, pos, conf2);
119                is![written == 0, return 0];
120
121                // Backward expansion: insert grouping separators in place.
122                let (mut src, mut dst) = (pos + raw_len, pos + final_len);
123                let (mut digits_left, mut group_count) = (raw_shape.left, 0u16);
124                while src > pos {
125                    src -= 1;
126                    dst -= 1;
127                    let b = buf[src];
128                    buf[dst] = b;
129                    // Only count digits belonging to the numeric magnitude.
130                    if b >= b'0' && b <= b'9' && digits_left > 0 {
131                        digits_left -= 1;
132                        group_count += 1;
133                        if digits_left > 0 && group_count == len {
134                            dst -= 1;
135                            buf[dst] = unwrap![some group.left_sep] as u8;
136                            group_count = 0;
137                        }
138                    }
139                }
140                final_len
141            }

166            }

309            ///
310            /// This first measures the formatted number using `measure_fmt`, then
311            /// accounts for any additional separator glyphs introduced by grouping.
312            pub const fn measure_group(self, conf: $crate::FmtNumConf, group: $crate::FmtNumGroup)
313                -> $crate::FmtNumShape {
314                let mut shape = self.measure_fmt(conf);
315                if shape.left > 0 && group.left_len != 0 && group.left_sep.is_some() {
316                    let groups = (shape.left.saturating_sub(1)) / group.left_len as u16;
317                    shape.left += groups;
318                }
319                shape
320            }
