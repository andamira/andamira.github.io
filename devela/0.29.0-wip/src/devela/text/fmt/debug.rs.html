8#[doc = crate::_tags!(fmt debug)]
9/// Extension for contextual debugging.
10#[doc = crate::_doc_meta!{location("text/fmt")}]
11///
12/// Types implementing this trait support formatting with a caller-supplied context.
13/// The context is defined per-type via the `Ctx` associated type, allowing each type
14/// to expose whatever formatting modes it needs without affecting other types.
15///
16/// # Examples
17/// ```
18/// # use devela::{DebugExt, FmtResult, Formatter};
19/// enum Mode { Hex, Dec }
20/// struct Value(u32);
21/// impl DebugExt for Value {
22///     type Ctx = Mode;
23///     fn fmt_with(&self, f: &mut Formatter, mode: &Self::Ctx) -> FmtResult<()> {
24///         match mode {
25///             Mode::Hex => write!(f, "0x{:X}", self.0),
26///             Mode::Dec => write!(f, "{}", self.0),
27///         }
28///     }
29/// }
30/// let v = Value(255);
31/// // caller chooses representation:
32/// // v.fmt_with(f, &Mode::Hex) → "0xFF"
33/// // v.fmt_with(f, &Mode::Dec) → "255"
34/// ```
35#[rustfmt::skip]
36pub trait DebugExt {
37    /// The context used to guide formatting.
38    type Ctx;
39    /// Formats `self` under the given context.
40    fn fmt_with(&self, f: &mut Formatter, ctx: &Self::Ctx) -> FmtResult<()>;
41    /// Wraps `self` for contextual debug formatting.
42    fn debug_with<'a>(&'a self, ctx: &'a Self::Ctx) -> DebugWith<'a, Self> where Self: Sized {
43        DebugWith::new(self, ctx)
44    }
45}
46
47/// A [`Debug`] adapter for formatting a value with a [`DebugExt`] context.
48pub struct DebugWith<'a, T: DebugExt + ?Sized> {
49    value: &'a T,
50    ctx: &'a T::Ctx,
51}
52impl<'a, T: DebugExt + ?Sized> DebugWith<'a, T> {
53    /// Creates a contextual debug adapter.
54    pub const fn new(value: &'a T, ctx: &'a T::Ctx) -> Self {
55        Self { value, ctx }
56    }
57}
58impl<T: DebugExt + ?Sized> Debug for DebugWith<'_, T> {
59    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
60        self.value.fmt_with(f, self.ctx)
61    }
