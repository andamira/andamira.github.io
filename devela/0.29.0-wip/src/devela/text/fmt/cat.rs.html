9#[doc = crate::_tags!(fmt)]
10/// Writes concatenated formatted data into a byte buffer.
11#[doc = crate::_doc_meta!{location("text/fmt")}]
12///
13/// Expands into sequential write calls using a mutable offset, combining:
14/// - String literals: `"text"`
15/// - Strings expressions: `@expr` — written via [`Slice::copy_str_into`]
16/// - Number expressions: `%expr` — formatted using [`FmtNum`] (only integers for now)
17///
18/// # Examples
19/// ```
20/// # use devela::{StringU8, fmtcat};
21/// let one = "Alice";
22/// let score = 42u8;
23/// let two = StringU8::<8>::from_str_unchecked("Bob");
24///
25/// let mut buf = [0u8; 64];
26/// let mut pos = 0;
27///
28/// fmtcat!(buf, pos, "Player ", @one, " scored ", %score, "! Now it's ", @two ,"'s turn.\n");
29/// assert_eq!(&buf[..pos], b"Player Alice scored 42! Now it's Bob's turn.\n");
30/// ```
31///
32/// # Notes
33/// - When passed a literal offset (`fmtcat!(buf, 0, ...)`) it doesn't update the variable.
34/// - All writes append sequentially; if the buffer is too small,
35///   the behavior depends on the helper functions.
36#[macro_export]
37#[cfg_attr(cargo_primary_package, doc(hidden))]
38macro_rules! fmtcat {
39    // Entry point: mutable offset variable (updated after writing)
40    ($buf:ident, $offset:ident, $($elem:tt)*) => {{
41        let mut pos = $offset;
42        $crate::fmtcat!(@proc $buf, pos, $($elem)*);
43        $offset = pos;
44    }};
45
46    // Entry point: literal offset (does not update)
47    ($buf:ident, $offset:literal, $($elem:tt)*) => {{
48        let mut pos = $offset;
49        $crate::fmtcat!(@proc $buf, pos, $($elem)*);
50    }};
51
52    // recursion terminator
53    (@proc $buf:ident, $pos:ident $(,)?) => {};
54
55    // literal string
56    (@proc $buf:ident, $pos:ident, $lit:literal , $($rest:tt)*) => {{
57        $pos += $crate::Slice::copy_str_into(&mut $buf, $pos, $lit);
58        $crate::fmtcat!(@proc $buf, $pos, $($rest)*);
59    }};
60    (@proc $buf:ident, $pos:ident, $lit:literal) => {{
61        $pos += $crate::Slice::copy_str_into(&mut $buf, $pos, $lit);
62    }};
63
64    // %number
65    (@proc $buf:ident, $pos:ident, %$num:expr , $($rest:tt)*) => {{
66        $pos += $crate::FmtNum($num).write(&mut $buf, $pos);
67        $crate::fmtcat!(@proc $buf, $pos, $($rest)*);
68    }};
69    (@proc $buf:ident, $pos:ident, %$num:expr) => {{
70        $pos += $crate::FmtNum($num).write(&mut $buf, $pos);
71    }};
72
73    // @string expression
74    (@proc $buf:ident, $pos:ident, @$s:expr, $($rest:tt)*) => {{
75        $pos += $crate::Slice::copy_utf8_into(&mut $buf, $pos, $s.as_bytes());
76        $crate::fmtcat!(@proc $buf, $pos, $($rest)*);
77    }};
78    (@proc $buf:ident, $pos:ident, @$s:expr) => {{
79        $pos += $crate::Slice::copy_utf8_into(&mut $buf, $pos, $s.as_bytes());
80    }};
81}
82pub use fmtcat;

84#[cfg(test)]
85mod tests {
86    #[test]
87    fn fmtargs() {
88        let name = "Alice";
89        let score = 42u8;
90        let mut buf = [0u8; 64];
91        let mut pos = 0;
92
93        fmtcat!(buf, pos,
94            "Hello ", @name, ", score=", %score, "!\n"
95        );
96
97        assert_eq!(&buf[..pos], b"Hello Alice, score=42!\n");
98
99        // can continue writing
100        fmtcat!(buf, pos, "Bye ", @name, "!");
101        assert_eq!(&buf[..pos], b"Hello Alice, score=42!\nBye Alice!");
102    }
103}
