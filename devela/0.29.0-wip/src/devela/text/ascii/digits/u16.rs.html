14    /// # Examples
15    /// ```
16    /// # use devela::text::Digits;
17    /// assert_eq![1, Digits(0_u16).count_digits10()];
18    /// assert_eq![4, Digits(9876_u16).count_digits10()];
19    /// ```
20    #[must_use]
21    pub const fn count_digits10(self) -> u8 {
22        is![self.0 == 0, 1, self.0.ilog10() as u8 + 1]
23    }

25    #[doc = _DOC_COUNT_DIGITS_16!()]
26    pub const fn count_digits16(self) -> u8 {
27        match self.0 {
28            0 => 1,
29            0x1..=0xF => 1,
30            0x10..=0xFF => 2,
31            0x100..=0xFFF => 3,
32            _ => 4,
33        }
34    }

38    /// Returns the ASCII decimal digit at the specified index.
39    ///
40    /// Returns `b'0'` if the index is beyond the number's decimal digits.
41    #[must_use]
42    #[inline(always)]
43    pub const fn digit_at_index10(self, index: u8) -> u8 {
44        is![index >= self.count_digits10(), return b'0'];
45        let power = AsciiLut::POWERS10[index as usize] as u16;
46        (self.0 / power % 10) as u8 + b'0'
47    }

52    #[must_use]

59    /// Returns the ASCII hexadecimal digit at the specified index (0 = least significant digit).
60    ///
61    /// For indices beyond the number's hexadecimal digits, returns the digit `b'0'`.
62    #[must_use]
63    pub const fn digit_at_index16(self, index: u8) -> u8 {
64        let shift = index as u32 * 4;
65        let digit = (self.0.unbounded_shr(shift) & 0xF) as usize;
66        AsciiLut::DIGITS_BASE36[digit]
67    }

121    #[must_use]
122    pub(crate) const fn digit_at_power10(self, divisor: u16) -> u8 {
123        (self.0 / divisor % 10) as u8 + b'0'
124    }
125    #[doc = _DOC_DIGIT_AT_POWER_16!()]
126    #[must_use]
127    pub(crate) const fn digit_at_power16(self, divisor: u16) -> u8 {
128        let digit = match divisor {
129            0x1 => self.0 & 0xF,
130            0x10 => (self.0 >> 4) & 0xF,
131            0x100 => (self.0 >> 8) & 0xF,
132            0x1000 => (self.0 >> 12) & 0xF,
133            _ => (self.0 / divisor) % 16,
134        };
135        AsciiLut::DIGITS_BASE36[digit as usize]
136    }

138    /// Converts a `u16` into a byte array of `5` ASCII decimal digits with leading zeros.
139    ///
140    /// You can trim the leading zeros with `Slice::`[`trim_leading()`][crate::Slice::trim_leading].
141    #[must_use]
142    pub const fn digits10(self) -> [u8; Self::MAX_DIGITS_10 as usize] {
143        [
144            //                    54321
145            //                    65535    ← u16::MAX
146            self.digit_at_power10(10000), // 5 digits
147            self.digit_at_power10(1000),
148            self.digit_at_power10(100),
149            self.digit_at_power10(10),
150            self.digit_at_power10(1),
151        ]
152    }

167    /// Writes 1..=5 decimal digits without leading zeros starting at `offset`,
168    /// returning the number of bytes written.
169    #[inline(always)]
170    const fn write_digits10_inner(self, buf: &mut [u8], offset: usize) -> usize {
171        let n = self.0;
172        if n < 10 {
173            buf[offset] = n as u8 + b'0';
174            1
175        } else if n < 100 {
176            buf[offset] = (n / 10) as u8 + b'0';
177            buf[offset + 1] = (n % 10) as u8 + b'0';
178            2
179        } else if n < 1000 {
180            buf[offset] = (n / 100) as u8 + b'0';
181            buf[offset + 1] = ((n / 10) % 10) as u8 + b'0';
182            buf[offset + 2] = (n % 10) as u8 + b'0';
183            3
184        } else if n < 10000 {
185            buf[offset] = (n / 1000) as u8 + b'0';
186            buf[offset + 1] = ((n / 100) % 10) as u8 + b'0';
187            buf[offset + 2] = ((n / 10) % 10) as u8 + b'0';
188            buf[offset + 3] = (n % 10) as u8 + b'0';
189            4
190        } else {
191            buf[offset] = (n / 10000) as u8 + b'0';
192            buf[offset + 1] = ((n / 1000) % 10) as u8 + b'0';
193            buf[offset + 2] = ((n / 100) % 10) as u8 + b'0';
194            buf[offset + 3] = ((n / 10) % 10) as u8 + b'0';
195            buf[offset + 4] = (n % 10) as u8 + b'0';
196            5
197        }
198    }

200    #[doc = _DOC_WRITE_DIGITS_10!(5)]
201    pub const fn write_digits10(self, buf: &mut [u8], offset: usize) -> usize {
202        let len = self.count_digits10() as usize;
203        is![len > buf.len().saturating_sub(offset), return 0];
204        self.write_digits10_inner(buf, offset)
205    }
206    #[doc = _DOC_WRITE_DIGITS_10_NONZERO!(5)]
207    pub const fn write_digits10_nonzero(self, buf: &mut [u8], offset: usize) -> usize {
208        is![self.0 == 0, return 0];
209        self.write_digits10(buf, offset)
210    }
