3//! Shows how to use the [`enumint!`] declarative macro.
4//!
5//! # Examples
6//! This will create the [`EnumintI8Example`] type.
7//! ```
8//! # use devela::enumint;
9//! enumint![
10//!     /// An example compact `i8` enum from -126 to 126, generated with [`enumint!`].
11//!     pub EnumintI8Example, i8, -126, 126
12//! ];
13//! fn main() {
14//!     assert_eq!(EnumintI8Example::VALUES, 253);
15//!     assert_eq!(EnumintI8Example::NICHES, 3);
16//!     // We can nest 3 Options before the memory representation increases:
17//!     assert_eq!(size_of::<EnumintI8Example>(), 1); // 0 niches used
18//!     assert_eq!(size_of::<Option<EnumintI8Example>>(), 1); // 1 niche used
19//!     assert_eq!(size_of::<Option<Option<EnumintI8Example>>>(), 1); // 2 niches used
20//!     assert_eq!(size_of::<Option<Option<Option<EnumintI8Example>>>>(), 1); // all 3 niches used
21//!     assert_eq!(size_of::<Option<Option<Option<Option<EnumintI8Example>>>>>(), 2);
22//! }
23//! ```
24//
25// Note that having a huge number of variants needs a lot of resources. E.g.:
26// enumint![pub EnumintI8Example, i8, -126, 126]; // +0.2s to compile
27// enumint![pub EnumintU16Example, u16, 0, 16384]; // +30s in safe mode, +10.0s unsafe
28// enumint![pub EnumintI16Example, i16, -16384, 16384]; // +30s +25GB RAM in unsafe mode
29
30use devela::enumint;
31
32enumint![
33    /// An example compact `i8` enum from -126 to 126, generated with [`enumint!`].
34    pub EnumintI8Example, i8, -126, 126
35];
36
37fn main() {
38    assert_eq!(EnumintI8Example::VALUES, 253);
39    assert_eq!(EnumintI8Example::NICHES, 3);
40    // We can nest 3 Options before the memory representation increases:
41    assert_eq!(size_of::<EnumintI8Example>(), 1); // 0 niches used
42    assert_eq!(size_of::<Option<EnumintI8Example>>(), 1); // 1 niche used
43    assert_eq!(size_of::<Option<Option<EnumintI8Example>>>(), 1); // 2 niches used
44    assert_eq!(size_of::<Option<Option<Option<EnumintI8Example>>>>(), 1); // all 3 niches used
45    assert_eq!(size_of::<Option<Option<Option<Option<EnumintI8Example>>>>>(), 2);
46}
