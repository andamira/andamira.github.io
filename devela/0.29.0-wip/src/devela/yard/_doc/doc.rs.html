3//! Defines [`_doc`].
4//
5
6#[doc = crate::_tags!(internal)]
7/// Generates a formatted meta-documentation string.
8#[doc = crate::_doc_meta!{location("yard")}]
9#[cfg_attr(cargo_primary_package, doc(hidden))]
10#[cfg_attr(not(feature = "__docs_internal"), doc(hidden))]
11#[cfg_attr(nightly_doc, doc(cfg(feature = "__docs_internal")))]
12#[macro_export]
13// #[allow(clippy::crate_in_macro_def, reason = "to invoke _std_core from crate of invocation")]
14macro_rules! _doc· {
15    // (newline) => { "<br/><br/>" };
16    (newline) => { "<br/><br style='display:block;content:\"\";margin-top:10px;' />" }; // DEPRECATE
17    (lf) => { "\n\n" };
18    (br) => { "<br/><br style='display:block;content:\"\";margin-top:10px;' />" };
19    (br+lf) => { "<br/><br style='display:block;content:\"\";margin-top:10px;' />\n\n" };
20    (br+hr) => { "<br/><hr/>" };
21    (hr) => { "<hr/>" };
22
23    // link to zall_::* associated module
24    (flat: $mod:literal) => {
25        concat!["<a title='See the flat view of the `", $mod, "` module' href='",
26            $crate::doclink![custom_current_crate concat!["zall_/_", $mod], @mod], "'>◉</a>"]//◉◦•
27    };
28    // link to the root * associated module
29    (root: $mod:literal) => {
30        concat!["<a title='See the hierarchical view of the `", $mod, "` module' href='",
31            $crate::doclink![custom_current_crate $mod, @mod], "'>▽</a>"]//▽▾⧩◬◌⊙⊜◎◎
32    };
33
34    /* list of submodules */
35
36    // IMPROVE:MAYBE pass optional feature-gate arg for each module
37    ( // no submodules:
38        modules: $path:path; $self:ident) => {
39        concat!(
40            $crate::_doc!(@meta_start_lf),
41            "[", stringify!($self), "][mod@", stringify!($path), "::", stringify!($self), "]",
42            $crate::_doc!(@meta_end),
43        )
44    };
45    ( // without submodules, but showing the module line.
46        modules: $path:path; $self:ident: _ $(,)?) => {
47        concat!(
48            $crate::_doc!(@meta_start_br),
49            "[", stringify!($self), "][mod@", stringify!($path), "::", stringify!($self), "]",
50            // $crate::_doc!(@modules: $path; $self: $($mod),+), "}",
51            $crate::_doc!(@meta_end),
52        )
53    };
54    ( // with submodules:
55        modules: $path:path; $self:ident: $($mod:ident),+ $(,)?) => {
56        concat!(
57            $crate::_doc!(@meta_start_br),
58            "[", stringify!($self), "][mod@", stringify!($path), "::", stringify!($self), "]::{",
59            $crate::_doc!(@modules: $path; $self: $($mod),+), "}",
60            $crate::_doc!(@meta_end),
61        )
62    };
63    // Handles the list of modules ensuring commas are only added between elements.
64    (@modules: $path:path; $self:ident: $first:ident $(, $rest:ident)*) => {
65        concat!(
66            "[", stringify!($first), "](mod@",
67                stringify!($path), "::", stringify!($self), "::",
68                stringify!($first), ")",
69            $(
70                ", [", stringify!($rest), "](mod@", stringify!($path), "::",
71                stringify!($self), "::", stringify!($rest), ")"
72            ),*
73        )
74    };
75
76    /* list of std modules */
77
78    // NOTE: it needs `_std_core!` to be defined in the crate where this is invoked.
79    (extends: $($mod:ident),+ $(,)?) => {
80        concat!(
81            $crate::_doc!(@meta_start_lf), "Extends: ",
82            // crate::_std_core!(), "::{", $crate::_doc!(@extends: $($mod),+), "}",
83            "std::{", $crate::_doc!(@extends: $($mod),+), "}",
84            $crate::_doc!(@meta_end_hr),
85        )
86    };
87    // Handles the list of modules ensuring commas are only added between elements.
88    (@extends: $first:ident $(, $rest:ident)*) => {
89        concat!(
90            // "[", stringify!($first), "](mod@", crate::_std_core!(), "::", stringify!($first), ")",
91            // $(
92            // ", [", stringify!($rest), "](mod@", crate::_std_core!(), "::", stringify!($rest), ")"
93            // ),*
94            "[", stringify!($first), "](https://doc.rust-lang.org/std/", stringify!($first), ")",
95            $(
96            ", [", stringify!($rest), "](https://doc.rust-lang.org/std/", stringify!($rest), ")"
97            ),*
98        )
99    };
100
101    /* start and end wrappers for modules: & extends: */
102    (@meta_start_br) => {
103        "<br/><i style='margin-left:0em;'></i><span style='font-size:90%;word-spacing:0px'>"
104    };
105    (@meta_start_lf) => {
106        "\n\n<i style='margin-left:0em;margin-top:-2em;'></i><span style='font-size:90%;word-spacing:0px'>"
107    };
108    (@meta_start_nobr) => {
109        "<i style='margin-left:0em;margin-top:-2em;'></i><span style='font-size:90%;word-spacing:0px'>"
110    };
111    // end
112    (@meta_end) => { "</span>" };
113    (@meta_end_hr) => { "</span><hr/>" };
114}
115#[doc(inline)]
116pub use _doc· as _doc;
