18/// A signed duration of time, stored as an `(i64, i32)` pair of secs and nanos.
19///
20/// Supports negative values, allowing representation of both past and future offsets.
21#[doc = crate::doc_!(vendor: "jiff")]
22#[derive(Clone, Copy, Debug, Default, Hash, PartialEq, Eq, PartialOrd, Ord)]
23pub struct TimeDelta {
24    secs: i64,
25    nanos: i32,
26}
27
28const NANOS_PER_SEC: i32 = 1_000_000_000;
29const NANOS_PER_MILLI: i32 = 1_000_000;
30const NANOS_PER_MICRO: i32 = 1_000;
31const MILLIS_PER_SEC: i64 = 1_000;
32const MICROS_PER_SEC: i64 = 1_000_000;
33const SECS_PER_MINUTE: i64 = 60;
34const MINS_PER_HOUR: i64 = 60;

38#[rustfmt::skip]
39mod impl_duration {
40    use crate::{Duration, TimeDelta};
41
42    /// # Additional APIs involving [`Duration`].
43    impl TimeDelta {
44        /// Converts a `Duration` to a `TimeDelta`.
45        pub const fn from_duration(duration: Duration) -> Self {
46            Self {
47                secs: duration.as_secs() as i64,
48                nanos: duration.subsec_nanos() as i32,
49            }
50        }
51
52        /// Converts a `TimeDelta` to a `Duration` if it is non-negative.
53        /// Returns `None` if the `TimeDelta` is negative.
54        pub const fn to_duration(&self) -> Option<Duration> {
55            if self.is_negative() { None }
56            else { Some(Duration::new(self.secs as u64, self.nanos as u32)) }
57        }
58
59        /// Returns the absolute value of this time delta as a [`Duration`].
60        ///
61        /// This method cannot panic because the absolute value of `TimeDelta::MIN`
62        /// is always representable in a `Duration`.
63        pub const fn abs_duration(self) -> Duration {
64            Duration::new(self.secs.unsigned_abs(), self.nanos.unsigned_abs())
65        }
66
67        /// Adds a `Duration` to this `TimeDelta`.
68        /// Returns `None` if the result overflows.
69        pub const fn checked_add_duration(&self, duration: Duration) -> Option<Self> {
70            let duration_delta = Self::from_duration(duration);
71            let Some(secs) = self.secs.checked_add(duration_delta.secs) else { return None; };
72            let nanos = self.nanos + duration_delta.nanos;
73            Some(Self::new(secs, nanos))
74        }
75
76        /// Subtracts a `Duration` from this `TimeDelta`.
77        /// Returns `None` if the result underflows.
78        pub const fn checked_sub_duration(&self, duration: Duration) -> Option<Self> {
79            let duration_delta = Self::from_duration(duration);
80            let Some(secs) = self.secs.checked_sub(duration_delta.secs) else { return None; };
81            let nanos = self.nanos - duration_delta.nanos;
82            Some(Self::new(secs, nanos))
83        }
84    }
85
86    impl TryFrom<Duration> for TimeDelta {
87        type Error = &'static str;
88
89        fn try_from(d: Duration) -> Result<TimeDelta, Self::Error> {
90            let secs = i64::try_from(d.as_secs())
91                .map_err(|_| "seconds in unsigned duration {d:?} overflowed i64")?;
92            // Guaranteed to succeed since 0<=nanos<=999,999,999.
93            let nanos = i32::try_from(d.subsec_nanos()).unwrap();
94            Ok(TimeDelta::new_unchecked(secs, nanos))
95        }
96    }
97    impl TryFrom<TimeDelta> for Duration {
98        type Error = &'static str;
99
100        fn try_from(sd: TimeDelta) -> Result<Duration, Self::Error> {
101            // This isn't needed, but improves error messages.
102            if sd.is_negative() {
103                return Err("cannot convert negative duration `{sd:?}` to \
104                     unsigned `std::time::Duration`");
105            }
106            let secs = u64::try_from(sd.as_secs())
107                .map_err(|_| "seconds in signed duration {sd:?} overflowed u64")?;
108            // Guaranteed to succeed because the above only succeeds
109            // when `sd` is non-negative. And when `sd` is non-negative,
110            // we are guaranteed that 0<=nanos<=999,999,999.
111            let nanos = u32::try_from(sd.subsec_nanos()).unwrap();
112            Ok(Duration::new(secs, nanos))
113        }
114    }
115}

119#[cfg_attr(nightly_doc, doc(cfg(feature = "js")))]
120mod impl_js {
121    use crate::{JsInstant, TimeDelta};
122    impl TimeDelta {
123        /// Converts a `JsInstant` into a `TimeDelta` relative to the time origin.
124        pub fn from_js(js: JsInstant) -> Self { Self::from_millis_f64(js.as_millis_f64()) }
125        /// Converts a `JsInstant` into a `TimeDelta` relative to the time origin.
126        pub const fn const_from_js(js: JsInstant) -> Self {
127            Self::const_from_millis_f64(js.as_millis_f64())
128        }
129        /// Converts a `TimeDelta` into a `JsInstant`, interpreting it as an absolute timestamp.
130        pub const fn to_js(self) -> JsInstant { JsInstant::from_millis_f64(self.as_millis_f64()) }
131    }

134#[rustfmt::skip]
135#[cfg(feature = "dep_jiff")]
136#[cfg_attr(nightly_doc, doc(cfg(feature = "dep_jiff")))]
137mod impl_jiff {
138    use crate::{TimeDelta, _dep::jiff::SignedDuration};
139    impl TimeDelta {
140        /// Converts [`SignedDuration`] into [`TimeDelta`].
141        pub const fn from_jiff(from: SignedDuration) -> TimeDelta {
142            TimeDelta::new(from.as_secs(), from.subsec_nanos())
143        }
144        /// Converts [`TimeDelta`] into [`SignedDuration`].
145        pub const fn to_jiff(self) -> SignedDuration { SignedDuration::new(self.secs, self.nanos) }
146    }
147    impl From<SignedDuration> for TimeDelta {
148        fn from(from: SignedDuration) -> TimeDelta { Self::from_jiff(from) }
149    }
150    impl From<TimeDelta> for SignedDuration {
151        fn from(from: TimeDelta) -> SignedDuration { from.to_jiff() }
152    }
