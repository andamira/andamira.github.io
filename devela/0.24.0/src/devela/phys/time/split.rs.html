38}

90impl TimeSplitNorm {
91    /// Converts a `Duration` into a full [`TimeSplit`].
92    ///
93    /// It assumes non-leap years and 30-day months for simplicity in calendar calculations.
94    pub const fn from_duration(duration: Duration) -> Self {
95        let secs = duration.as_secs();
96        let total_days = secs / 86400;
97        let y = total_days / 365;
98        let days_remaining = total_days % 365;
99        let mo = (days_remaining / 30) as u8;
100        let d = (days_remaining % 30) as u8;
101        let h = ((secs % 86400) / 3600) as u8;
102        let m = ((secs % 3600) / 60) as u8;
103        let s = (secs % 60) as u8;
104        let nanos = duration.subsec_nanos();
105        let ms = (nanos / 1_000_000) as u16;
106        let us = ((nanos % 1_000_000) / 1_000) as u16;
107        let ns = (nanos % 1_000) as u16;
108        TimeSplit { y, mo, d, h, m, s, ms, us, ns }
109    }
110}
