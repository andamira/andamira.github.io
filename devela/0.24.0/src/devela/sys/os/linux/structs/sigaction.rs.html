14/// Examine and change a signal action.
15///
16/// Represents the [`sigaction`] structure from libc.
17///
18/// [`sigaction`]: https://man7.org/linux/man-pages/man2/sigaction.2.html
19#[repr(C)]
20#[must_use]
21pub struct LinuxSigaction {
22    /// A union that stores either a simple handler or a `SA_SIGINFO` handler.
23    sa_handler: LinuxSigactionHandler,
24
25    /// A set of flags which modify the behavior of the signal.
26    sa_flags: usize,
27
28    /// A function that restores the program's state after the signal handler runs.
29    ///
30    /// This is required for proper signal handling on some architectures.
31    sa_restorer: Option<extern "C" fn()>,
32
33    /// A mask of signals that should be blocked.
34    sa_mask: LinuxSigset,
35}
36#[rustfmt::skip]
37impl Debug for LinuxSigaction {
38    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
39        write!(f, "LinuxSigaction {{ ")?;
40        #[cfg(feature = "unsafe_syscall")]
41        if self.sa_flags & LINUX_SIGACTION::SA_SIGINFO == 0 {
42            write!(f, "sa_handler: {:p}, ", unsafe { self.sa_handler.sa_handler })?;
43        } else {
44            write!(f, "sa_sigaction: {:p}, ", unsafe { self.sa_handler.sa_sigaction })?;
45        }
46        write!(f, "sa_flags: {:#x}, ", self.sa_flags)?;
47        match self.sa_restorer {
48            Some(ptr) => write!(f, "sa_restorer: {ptr:p}, ")?,
49            None => write!(f, "sa_restorer: None, ")?,
50        }
51        write!(f, "sa_mask: {:?}", self.sa_mask)?;
52        write!(f, " }}")
53    }
54}

56impl LinuxSigaction {
57    /// Returns a new `LinuxSigaction` for a simple handler.
58    pub const fn new(
59        handler: extern "C" fn(i32),
60        flags: usize,
61        mask: LinuxSigset,
62        restorer: Option<extern "C" fn()>,
63    ) -> Self {
64        Self {
65            sa_handler: LinuxSigactionHandler { sa_handler: handler },
66            sa_flags: flags,
67            sa_restorer: restorer,
68            sa_mask: mask,
69        }
70    }
71    /// Returns a new `LinuxSigaction` for a [`SA_SIGINFO`][LINUX_SIGACTION::SA_SIGINFO] handler.
72    pub const fn new_siginfo(
73        handler: extern "C" fn(i32, LinuxSiginfo, *mut c_void),
74        flags: usize,
75        mask: LinuxSigset,
76        restorer: Option<extern "C" fn()>,
77    ) -> Self {
78        Self {
79            sa_handler: LinuxSigactionHandler { sa_sigaction: handler },
80            sa_flags: flags,
81            sa_restorer: restorer,
82            sa_mask: mask,
83        }
84    }
85    /// Returns the simple handler, if set.
86    #[cfg(feature = "unsafe_syscall")]
87    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_syscall")))]
88    pub const fn sa_handler(&self) -> Option<extern "C" fn(i32)> {
89        if self.sa_flags & LINUX_SIGACTION::SA_SIGINFO == 0 {
90            Some(unsafe { self.sa_handler.sa_handler })
91        } else {
92            None
93        }
94    }
95    /// Returns the `SA_SIGINFO` handler, if set.
96    #[must_use]
97    #[cfg(feature = "unsafe_syscall")]
98    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_syscall")))]
99    pub const fn sa_sigaction(&self) -> Option<extern "C" fn(i32, LinuxSiginfo, *mut c_void)> {
100        if self.sa_flags & LINUX_SIGACTION::SA_SIGINFO != 0 {
101            Some(unsafe { self.sa_handler.sa_sigaction })
102        } else {
103            None
104        }
105    }
106}

117/// A set of linux signals.
118///
119/// Represents the [`sigset_t`] structure from libc.
120///
121/// [`sigset_t`]: https://man7.org/linux/man-pages/man7/system_data_types.7.html
122#[repr(C)]
123#[must_use]
124#[derive(Debug, Default, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
125pub struct LinuxSigset {
126    /// An array of signals.
127    ///
128    /// Its lenght is calculated from the number of signals divided by the bits of a usize.
129    pub sig: [usize; Self::LEN],
130}
131#[rustfmt::skip]
132impl LinuxSigset {
133    /// The number of bits in a `usize`.
134    const BITS_PER_USIZE: usize = usize::BITS as usize;
135    /// The hardcoded number of system signals defined in `LINUX_SIGNAL`.
136    const NSIG: usize = 31;
137    /// The size of the array is the number of signals divided by the bits of an usize.
138    const LEN: usize = { Self::NSIG.div_ceil(Self::BITS_PER_USIZE) };
139
140    /// Returns an empty set of signals.
141    pub const fn empty() -> Self { Self { sig: [0; Self::LEN] } }
142    /// Returns the size in bytes of `LinuxSigset`.
143    #[must_use]
144    pub const fn size() -> usize { size_of::<Self>() }
145}
146impl LinuxSigset {
147    /// Sets the bit corresponding to a `signal` in the `sig` array.
148    ///
149    /// # Arguments
150    /// * `signum` - The number of the signal. This should be between 1 and `NSIG`.
151    ///
152    /// # Panics
153    /// Panics if `signum` < 1 or > 36.
154    pub fn set_signal(&mut self, signal: i32) {
155        let signal = signal as usize;
156        assert![(1..=Self::NSIG).contains(&signal)];
157
158        // Subtract 1 from the signal number because signal numbers start from 1
159        // but array indices start from 0
160        let signal_index = (signal - 1) / Self::BITS_PER_USIZE;
161        let bit_position = (signal - 1) % Self::BITS_PER_USIZE;
162        self.sig[signal_index] |= 1 << bit_position;
163    }
164}

166/// Additional information about a signal.
167///
168/// Represents the [`siginfo_t`] structure from libc.
169///
170/// [`siginfo_t`]: https://man7.org/linux/man-pages/man2/sigaction.2.html
171#[repr(C)]
172#[must_use]
173pub struct LinuxSiginfo {
174    /// Signal number.
175    si_signo: i32,
176    /// Error number (if applicable).
177    si_errno: i32,
178    /// Indicates the reason or source of the signal.
179    si_code: i32,
180    /// Sender's PID.
181    si_pid: i32,
182    /// Sender's UID.
183    si_uid: u32,
184    /// Exit value or signal.
185    si_status: i32,
186    /// Faulting address (e.g., for SIGSEGV).
187    si_addr: *mut c_void,
188    /// Band event (for SIGPOLL).
189    si_band: i64,
190    /// Signal value.
191    si_value: LinuxSigval,
192}
193#[rustfmt::skip]
194impl LinuxSiginfo {
195    /// Returns the signal number.
196    pub fn signo(&self) -> i32 { self.si_signo }
197    /// Returns the error number (if applicable).
198    pub fn errno(&self) -> i32 { self.si_errno }
199    /// Returns the signal code.
200    pub fn code(&self) -> i32 { self.si_code } // IMPROVE: return si_codes
201    /// Returns the sender's PID.
202    pub fn pid(&self) -> i32 { self.si_pid }
203    /// Returns the sender's UID.
204    pub fn uid(&self) -> u32 { self.si_uid }
205    /// Returns the exit value or signal.
206    pub fn status(&self) -> i32 { self.si_status }
207    /// Returns the faulting address (e.g., for SIGSEGV).
208    pub fn addr(&self) -> *mut c_void { self.si_addr }
209    /// Returns the band event (for SIGPOLL).
210    pub fn band(&self) -> i64 { self.si_band }
211    /// Returns the signal value as an integer.
212    #[cfg(feature = "unsafe_syscall")]
213    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_syscall")))]
214    pub fn value_int(&self) -> i32 { unsafe { self.si_value.int } }
215    /// Returns the signal value as a pointer.
216    #[cfg(feature = "unsafe_syscall")]
217    #[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_syscall")))]
218    pub fn value_ptr(&self) -> *mut c_void { unsafe { self.si_value.ptr } }
219}
220#[rustfmt::skip]
221impl Debug for LinuxSiginfo {
222    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult<()> {
223        write!(f, "LinuxSiginfo {{ ")?;
224        write!(f, "si_signo: {}, ", self.si_signo)?;
225        write!(f, "si_errno: {}, ", self.si_errno)?;
226        write!(f, "si_code: {}, ", self.si_code)?;
227        write!(f, "si_pid: {}, ", self.si_pid)?;
228        write!(f, "si_uid: {}, ", self.si_uid)?;
229        write!(f, "si_status: {}, ", self.si_status)?;
230        write!(f, "si_addr: {:?}, ", self.si_addr)?;
231        write!(f, "si_band: {}, ", self.si_band)?;
232        // IMPROVE: use si_codes
233        #[cfg(feature = "unsafe_syscall")]
234        write!(f, "si_value: LinuxSigval {{ sival_int: {}, sival_ptr: {:?} }}",
235            unsafe { self.si_value.int }, unsafe { self.si_value.ptr })?;
236        write!(f, " }}")
237    }
