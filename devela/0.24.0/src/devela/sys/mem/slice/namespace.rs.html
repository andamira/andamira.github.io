31/// # `core::slice` namespaced methods
32impl<T> Slice<T> {
33    /// Copies all elements from `src` into `dst`.
34    ///
35    /// # Features
36    /// - Uses `Ptr::copy_nonoverlapping` when unsafe operations are allowed.
37    /// - Falls back to safe element-wise copy otherwise.
38    ///
39    /// # Panics
40    /// Panics if `src` and `dst` slices have different lengths.
41    // WAIT:1.87 [const_copy_from_slice](https://github.com/rust-lang/rust/issues/131415)
42    #[rustfmt::skip]
43    pub const fn copy_from_slice(dst: &mut [T], src: &[T]) where T: Copy {
44        if dst.len() != src.len() { panic!("`src` and `dst` slices have different lengths"); }
45
46        #[cfg(all(not(feature = "safe_mem"), unsafe··))]
47        // SAFETY: Lengths checked equal, T is Copy, and pointers are valid
48        unsafe { Ptr::copy_nonoverlapping(src.as_ptr(), dst.as_mut_ptr(), src.len()); }
49
50        #[cfg(any(feature = "safe_mem", not(unsafe··)))]
51        { let mut i = 0; while i < src.len() { dst[i] = src[i]; i += 1; } }
52    }
53
54    /// Converts a reference to `T` into a slice of length 1 (without copying).
55    ///
56    /// See `core::slice::`[`from_ref`].
57    #[must_use]
58    pub const fn from_ref(s: &T) -> &[T] {
59        from_ref(s)
60    }
61
62    /// Converts a reference to `T` into a slice of length 1 (without copying).
63    ///
64    /// See `core::slice::`[`from_mut`].
65    #[must_use]
66    pub const fn from_mut(s: &mut T) -> &mut [T] {
67        from_mut(s)
68    }
69
70    /// Forms a shared slice from a pointer and a length.
71    ///
72    /// # Safety
73    /// See `core::slice::`[`from_raw_parts`]
74    ///
75    /// See also `Ptr::`[`slice_from_raw_parts`][crate::Ptr::slice_from_raw_parts].
76    #[cfg_attr(nightly_doc, doc(cfg(unsafe··)))]
77    #[cfg(all(not(feature = "safe_mem"), unsafe··))]
78    pub const unsafe fn from_raw_parts<'a>(data: *const T, len: usize) -> &'a [T] {
79        // SAFETY: Caller must uphold the safety contract.
80        unsafe { from_raw_parts(data, len) }
81    }
82
83    /// Forms an exclusive slice from a pointer and a length.
84    ///
85    /// # Safety
86    /// See `core::slice::`[`from_raw_parts_mut`].
87    ///
88    /// See also `Ptr::`[`slice_from_raw_parts_mut`][crate::Ptr::slice_from_raw_parts_mut].
89    #[cfg_attr(nightly_doc, doc(cfg(unsafe··)))]
90    #[cfg(all(not(feature = "safe_mem"), unsafe··))]
91    pub const unsafe fn from_raw_parts_mut<'a>(data: *mut T, len: usize) -> &'a mut [T] {
92        // SAFETY: Caller must uphold the safety contract.
93        unsafe { from_raw_parts_mut(data, len) }
94    }

97/// # Methods for shared subslicing using index-based splitting.
98impl<T> Slice<T> {
99    /* panicking */
100
101    /// Returns a subslice up to the given `end` index.
102    ///
103    /// Equivalent to `&slice[..end]`.
104    ///
105    /// # Panics
106    /// Panics if `end` > `slice.len()`.
107    #[must_use]
108    pub const fn range_to(slice: &[T], end: usize) -> &[T] {
109        slice.split_at(end).0
110    }
111    /// Returns a subslice starting from the given `start` index.
112    ///
113    /// Equivalent to `&slice[start..]`.
114    ///
115    /// # Panics
116    /// Panics if `start` > `slice.len()`.
117    #[must_use]
118    pub const fn range_from(slice: &[T], start: usize) -> &[T] {
119        slice.split_at(start).1
120    }
121    /// Returns a subslice from `start` (inclusive) to `end` (exclusive).
122    ///
123    /// Equivalent to `&slice[start..end]`.
124    ///
125    /// # Panics
126    /// Panics if `start` > len or `end` > `slice.len()`.
127    #[must_use]
128    pub const fn range(slice: &[T], start: usize, end: usize) -> &[T] {
129        slice.split_at(start).1.split_at(end - start).0
130    }
131
132    /// Returns the first `n` elements of the slice.
133    ///
134    /// Equivalent to `&slice[..n]`.
135    ///
136    /// # Panics
137    /// Panics if `n` > `slice.len()`.
138    #[must_use]
139    pub const fn take_first(slice: &[T], n: usize) -> &[T] {
140        slice.split_at(n).0
141    }
142    /// Returns the last `n` elements of the slice.
143    ///
144    /// Equivalent to `&slice[slice.len() - n..]`.
145    ///
146    /// # Panics
147    /// Panics if `n` > `slice.len()`.
148    #[must_use]
149    pub const fn take_last(slice: &[T], n: usize) -> &[T] {
150        slice.split_at(slice.len() - n).1
151    }
152    /// Returns the slice omitting the last `n` elements.
153    ///
154    /// Equivalent to `&slice[..slice.len() - n]`.
155    ///
156    /// # Panics
157    /// Panics if `n` > `slice.len()`.
158    #[must_use]
159    pub const fn take_omit_last(slice: &[T], n: usize) -> &[T] {
160        slice.split_at(slice.len() - n).0
161    }
162
163    /* Checked */
164
165    /// Returns a subslice up to the given `end` index.
166    ///
167    /// Equivalent to `&slice[..end]`.
168    ///
169    /// Returns `None` if `end` > `slice.len()`.
170    #[must_use]
171    pub const fn range_to_checked(slice: &[T], end: usize) -> Option<&[T]> {
172        match slice.split_at_checked(end) {
173            Some((subslice, _)) => Some(subslice),
174            None => None,
175        }
176    }
177    /// Returns a subslice starting from the given `start` index.
178    ///
179    /// Equivalent to `&slice[start..]`.
180    ///
181    /// Returns `None` if `start` > `slice.len()`.
182    #[must_use]
183    pub const fn range_from_checked(slice: &[T], start: usize) -> Option<&[T]> {
184        match slice.split_at_checked(start) {
185            Some((_, subslice)) => Some(subslice),
186            None => None,
187        }
188    }
189    /// Returns a subslice from `start` (inclusive) to `end` (exclusive).
190    ///
191    /// Equivalent to `&slice[start..end]`.
192    ///
193    /// Returns `None` if `start` > `slice.len()` or `end` > `slice.len()`.
194    ///
195    /// # Features
196    /// This method makes use of of the `unsafe_slice` feature is enabled.
197    #[must_use]
198    pub const fn range_checked(slice: &[T], start: usize, end: usize) -> Option<&[T]> {
199        if start <= end && end <= slice.len() {
200            #[cfg(any(feature = "safe_mem", not(feature = "unsafe_slice")))]
201            return Some(slice.split_at(start).1.split_at(end - start).0);
202            #[cfg(all(not(feature = "safe_mem"), feature = "unsafe_slice"))]
203            // SAFETY: `start` and `end` are checked to be within bounds and valid
204            Some(unsafe { slice.split_at_unchecked(start).1.split_at_unchecked(end - start).0 })
205        } else {
206            None
207        }
208    }
209
210    /// Returns the first `n` elements of the slice.
211    ///
212    /// Equivalent to `&slice[..n]`.
213    ///
214    /// Returns `None` if `n` > `slice.len()`.
215    #[must_use]
216    pub const fn take_first_checked(slice: &[T], n: usize) -> Option<&[T]> {
217        match slice.split_at_checked(n) {
218            Some((subslice, _)) => Some(subslice),
219            None => None,
220        }
221    }
222    /// Returns the last `n` elements of the slice.
223    ///
224    /// Equivalent to `&slice[slice.len() - n..]`.
225    ///
226    /// Returns `None` if `n` > `slice.len()`.
227    ///
228    /// # Features
229    /// This method makes use of of the `unsafe_slice` feature is enabled.
230    #[must_use]
231    pub const fn take_last_checked(slice: &[T], n: usize) -> Option<&[T]> {
232        match slice.len().checked_sub(n) {
233            Some(index) => {
234                #[cfg(any(feature = "safe_mem", not(feature = "unsafe_slice")))]
235                return Some(slice.split_at(index).1);
236                #[cfg(all(not(feature = "safe_mem"), feature = "unsafe_slice"))]
237                // SAFETY: `n` is checked to be within bounds and valid
238                return Some(unsafe { slice.split_at_unchecked(index).1 });
239            }
240            None => None,
241        }
242    }
243    /// Returns the slice omitting the last `n` elements.
244    ///
245    /// Equivalent to `&slice[..slice.len() - n]`.
246    ///
247    /// Returns `None` if `n` > `slice.len()`.
248    ///
249    /// # Features
250    /// This method makes use of of the `unsafe_slice` feature is enabled.
251    #[must_use]
252    pub const fn take_omit_last_checked(slice: &[T], n: usize) -> Option<&[T]> {
253        match slice.len().checked_sub(n) {
254            Some(index) => {
255                #[cfg(any(feature = "safe_mem", not(feature = "unsafe_slice")))]
256                return Some(slice.split_at(index).0);
257                #[cfg(all(not(feature = "safe_mem"), feature = "unsafe_slice"))]
258                // SAFETY: `n` is checked to be within bounds and valid
259                return Some(unsafe { slice.split_at_unchecked(index).0 });
260            }
261            None => None,
262        }
263    }
264}

437/// # Methods for splitting.
438impl<T> Slice<T> {
439    /* left split */
440
441    /// Returns the leftmost sub-`slice` with the given maximum `len`.
442    ///
443    /// If `len > self.len()` it returns the full slice.
444    ///
445    /// # Example
446    /// ```
447    /// # use devela::Slice;
448    /// let v = [1, 2, 3, 4, 5, 6];
449    /// assert_eq!(Slice::lsplit(&v, 3), &[1, 2, 3]);
450    /// assert_eq!(Slice::lsplit(&v, 0), &[] as &[i32]);
451    /// assert_eq!(Slice::lsplit(&v, 10), &[1, 2, 3, 4, 5, 6]);
452    /// ```
453    pub const fn lsplit(slice: &[T], len: usize) -> &[T] {
454        let end_idx = Compare(len).clamp(0, slice.len());
455        let (left, _) = slice.split_at(end_idx);
456        left
457    }
458
459    /// Returns the leftmost exclusive sub-`slice` with the given maximum `len`.
460    ///
461    /// If `left_len > slice.len()` it returns the full slice.
462    ///
463    /// # Example
464    /// ```
465    /// # use devela::Slice;
466    /// let mut v = [1, 2, 3, 4, 5, 6];
467    /// assert_eq!(Slice::lsplit_mut(&mut v, 3), &mut [1, 2, 3]);
468    /// assert_eq!(Slice::lsplit_mut(&mut v, 0), &mut [] as &mut [i32]);
469    /// assert_eq!(Slice::lsplit_mut(&mut v, 10), &mut [1, 2, 3, 4, 5, 6]);
470    /// ```
471    /// See also [`Slice::lsplit_mut`].
472    pub const fn lsplit_mut(slice: &mut [T], len: usize) -> &mut [T] {
473        let end_idx = Compare(len).clamp(0, slice.len());
474        let (left, _) = slice.split_at_mut(end_idx);
475        left
476    }
477
478    /* right split */
479
480    /// Returns the rightmost sub-`slice` with the given maximum `len`.
481    ///
482    /// If `left_len > slice.len()` it returns the full slice.
483    ///
484    /// # Example
485    /// ```
486    /// # use devela::Slice;
487    /// let v = [1, 2, 3, 4, 5, 6];
488    /// assert_eq!(Slice::rsplit(&v, 3), &[4, 5, 6]);
489    /// assert_eq!(Slice::rsplit(&v, 0), &[] as &[i32]);
490    /// assert_eq!(Slice::rsplit(&v, 10), &[1, 2, 3, 4, 5, 6]);
491    /// ```
492    #[must_use]
493    pub const fn rsplit(slice: &[T], len: usize) -> &[T] {
494        let start_idx = slice.len().saturating_sub(len);
495        let (_, right) = slice.split_at(start_idx);
496        right
497    }
498
499    /// Returns the rightmost mutable sub-`slice` with the given maximum `len`.
500    ///
501    /// If `left_len > slice.len()` it returns the full slice.
502    ///
503    /// # Example
504    /// ```
505    /// # use devela::Slice;
506    /// let mut v = [1, 2, 3, 4, 5, 6];
507    /// assert_eq!(Slice::rsplit_mut(&mut v, 3), &mut [4, 5, 6]);
508    /// assert_eq!(Slice::rsplit_mut(&mut v, 0), &mut [] as &mut [i32]);
509    /// assert_eq!(Slice::rsplit_mut(&mut v, 10), &mut [1, 2, 3, 4, 5, 6]);
510    /// ```
511    /// See also [`Slice::lsplit_mut`].
512    #[must_use]
513    pub const fn rsplit_mut(slice: &mut [T], len: usize) -> &mut [T] {
514        let start_idx = slice.len().saturating_sub(len);
515        let (_, right) = slice.split_at_mut(start_idx);
516        right
517    }
518
519    /* middle split left biased */
520
521    /// Returns the middle sub-`slice` with the given maximum `len` and a left bias.
522    ///
523    /// In case of a non-perfect middle split, it will have one character more
524    /// on the left.
525    ///
526    /// If `len > slice.len()` returns the full `slice`.
527    ///
528    /// # Example
529    /// ```
530    /// # use devela::Slice;
531    /// let v = [1, 2, 3, 4, 5, 6];
532    /// assert_eq!(Slice::msplit_left(&v, 0), &[] as &[i32]);
533    /// assert_eq!(Slice::msplit_left(&v, 1), &[3]);
534    /// assert_eq!(Slice::msplit_left(&v, 2), &[3, 4]);
535    /// assert_eq!(Slice::msplit_left(&v, 3), &[2, 3, 4]);
536    /// assert_eq!(Slice::msplit_left(&v, 4), &[2, 3, 4, 5]);
537    /// assert_eq!(Slice::msplit_left(&v, 5), &[1, 2, 3, 4, 5]);
538    /// assert_eq!(Slice::msplit_left(&v, 10), &[1, 2, 3, 4, 5, 6]);
539    /// ```
540    /// See also [`Slice::msplit_right`].
541    #[must_use]
542    pub const fn msplit_left(slice: &[T], len: usize) -> &[T] {
543        let mid_idx = slice.len() / 2;
544        let half_len = len / 2;
545        let start_idx = mid_idx.saturating_sub(half_len + (len % 2));
546        let end_idx = Compare(mid_idx + half_len).min(slice.len());
547        let (_, right) = slice.split_at(start_idx);
548        let (middle, _) = right.split_at(end_idx - start_idx);
549        middle
550    }
551
552    /// Returns the middle exclusive sub-`slice` with the given maximum `len` and a
553    /// left bias.
554    ///
555    /// In case of a non-perfect middle split, it will have one character more
556    /// on the left.
557    ///
558    /// If `len > slice.len()` returns the full `slice`.
559    ///
560    /// # Example
561    /// ```
562    /// # use devela::Slice;
563    /// let mut v = [1, 2, 3, 4, 5, 6];
564    /// assert_eq!(Slice::msplit_left_mut(&mut v, 0), &mut [] as &mut [i32]);
565    /// assert_eq!(Slice::msplit_left_mut(&mut v, 1), &mut [3]);
566    /// assert_eq!(Slice::msplit_left_mut(&mut v, 2), &mut [3, 4]);
567    /// assert_eq!(Slice::msplit_left_mut(&mut v, 3), &mut [2, 3, 4]);
568    /// assert_eq!(Slice::msplit_left_mut(&mut v, 4), &mut [2, 3, 4, 5]);
569    /// assert_eq!(Slice::msplit_left_mut(&mut v, 5), &mut [1, 2, 3, 4, 5]);
570    /// assert_eq!(Slice::msplit_left_mut(&mut v, 10), &mut [1, 2, 3, 4, 5, 6]);
571    /// ```
572    /// See also [`Slice::msplit_right_mut`].
573    #[must_use]
574    pub const fn msplit_left_mut(slice: &mut [T], len: usize) -> &mut [T] {
575        let mid_idx = slice.len() / 2;
576        let half_len = len / 2;
577        let start_idx = mid_idx.saturating_sub(half_len + (len % 2));
578        let end_idx = Compare(mid_idx + half_len).min(slice.len());
579        let (_, right) = slice.split_at_mut(start_idx);
580        let (middle, _) = right.split_at_mut(end_idx - start_idx);
581        middle
582    }
583
584    /* middle split right biased */
585
586    /// Returns the middle sub-`slice` with the given maximum `len` and a right bias.
587    ///
588    /// In case of a non-perfect middle split, it will have one character more
589    /// on the right.
590    ///
591    /// If `len > slice.len()` returns the full `slice`.
592    ///
593    /// # Example
594    /// ```
595    /// # use devela::Slice;
596    /// let v = [1, 2, 3, 4, 5, 6];
597    /// assert_eq!(Slice::msplit_right(&v, 0), &[] as &[i32]);
598    /// assert_eq!(Slice::msplit_right(&v, 1), &[4]);
599    /// assert_eq!(Slice::msplit_right(&v, 2), &[3, 4]);
600    /// assert_eq!(Slice::msplit_right(&v, 3), &[3, 4, 5]);
601    /// assert_eq!(Slice::msplit_right(&v, 4), &[2, 3, 4, 5]);
602    /// assert_eq!(Slice::msplit_right(&v, 5), &[2, 3, 4, 5, 6]);
603    /// assert_eq!(Slice::msplit_right(&v, 10), &[1, 2, 3, 4, 5, 6]);
604    /// ```
605    /// See also [`Slice::msplit_left`].
606    #[must_use]
607    pub const fn msplit_right(slice: &[T], len: usize) -> &[T] {
608        let mid_idx = slice.len() / 2;
609        let half_len = len / 2;
610        let start_idx = mid_idx.saturating_sub(half_len);
611        let end_idx = Compare(mid_idx + half_len + (len % 2)).min(slice.len());
612        let (_, right) = slice.split_at(start_idx);
613        let (middle, _) = right.split_at(end_idx - start_idx);
614        middle
615    }
616
617    /// Returns the middle exclusive sub-`slice` with the given maximum `len` and a
618    /// right bias.
619    ///
620    /// In case of a non-perfect middle split, it will have one character more
621    /// on the right.
622    ///
623    /// If `len > slice.len()` returns the full `slice`.
624    /// # Example
625    /// ```
626    /// # use devela::Slice;
627    /// let mut v = [1, 2, 3, 4, 5, 6];
628    /// assert_eq!(Slice::msplit_right_mut(&mut v, 0), &mut [] as &mut[i32]);
629    /// assert_eq!(Slice::msplit_right_mut(&mut v, 1), &mut [4]);
630    /// assert_eq!(Slice::msplit_right_mut(&mut v, 2), &mut [3, 4]);
631    /// assert_eq!(Slice::msplit_right_mut(&mut v, 3), &mut [3, 4, 5]);
632    /// assert_eq!(Slice::msplit_right_mut(&mut v, 4), &mut [2, 3, 4, 5]);
633    /// assert_eq!(Slice::msplit_right_mut(&mut v, 5), &mut [2, 3, 4, 5, 6]);
634    /// assert_eq!(Slice::msplit_right_mut(&mut v, 10), &mut [1, 2, 3, 4, 5, 6]);
635    /// ```
636    /// See also [`Slice::msplit_left_mut`].
637    #[must_use]
638    pub const fn msplit_right_mut(slice: &mut [T], len: usize) -> &mut [T] {
639        let mid_idx = slice.len() / 2;
640        let half_len = len / 2;
641        let start_idx = mid_idx.saturating_sub(half_len);
642        let end_idx = Compare(mid_idx + half_len + (len % 2)).min(slice.len());
643        let (_, right) = slice.split_at_mut(start_idx);
644        let (middle, _) = right.split_at_mut(end_idx - start_idx);
645        middle
646    }

649/// # Methods for byte slices.
650#[rustfmt::skip]
651impl Slice<u8> {
652    /// Copies the `src` byte array into `dst` in compile-time.
653    pub const fn copy_array<const N: usize>(dst: &mut [u8; N], src: &[u8; N]) {
654        #[cfg(any(feature = "safe_mem", not(unsafe··)))]
655        { let mut i = 0; while i < N { dst[i] = src[i]; i += 1; } }
656
657        #[cfg(all(not(feature = "safe_mem"), unsafe··))]
658        unsafe { Ptr::copy_nonoverlapping(src.as_ptr(), dst.as_mut_ptr(), N); }
659    }
660
661    /// Copies all elements from `src` into a fixed-size array starting at `offset`.
662    ///
663    /// # Features
664    /// - Uses `Ptr::copy_nonoverlapping` when unsafe operations are allowed.
665    /// - Falls back to safe element-wise copy otherwise.
666    ///
667    /// # Panics
668    /// Panics if `src.len() + offset > LEN`.
669    pub const fn copy_array_at<const LEN: usize>(dst: &mut [u8; LEN], src: &[u8], offset: usize) {
670        assert!(src.len() + offset <= LEN, "source slice does not fit in destination array");
671
672        #[cfg(any(feature = "safe_mem", not(unsafe··)))]
673        { let mut i = 0; while i < src.len() { dst[offset + i] = src[i]; i += 1; } }
674
675        #[cfg(all(not(feature = "safe_mem"), unsafe··))]
676        // SAFETY: Length checked via assert, u8 is Copy, offset + src.len() is bounds-checked
677        unsafe { Ptr::copy_nonoverlapping(src.as_ptr(), dst.as_mut_ptr().add(offset), src.len()); }
678    }
679
680    /// A convenience wrapper over [`Slice::copy_array_at`].
681    // #[inline(always)] // MAYBE
682    // // Auto-select best version
683    // macro_rules! copy_array { // IDEA
684    //     ($dst:expr, $src:expr, $at:expr) => {
685    //         if $dst.len() <= 32 { copied_array_at($dst, $src, $at) }
686    //         else { let mut tmp = $dst; copy_array_at(&mut tmp, $src, $at); tmp }
687    //     }
688    // }
689    pub const fn copied_array_at<const LEN: usize>(base: [u8; LEN], src: &[u8], offset: usize)
690        -> [u8; LEN] {
691        let mut new = base;
692        Slice::copy_array_at(&mut new, src, offset);
693        new
694    }
695
696    /// Copies all elements from `src` into a new fixed-size array.
697    ///
698    /// # Features
699    /// - Uses `Ptr::copy_nonoverlapping` when unsafe operations are allowed
700    /// - Falls back to safe element-wise copy otherwise
701    ///
702    /// # Panics
703    /// Panics if `src.len() != LEN`.
704    pub const fn to_array<const LEN: usize>(src: &[u8]) -> [u8; LEN] {
705        assert!(src.len() == LEN, "source slice length must match destination array length");
706        let mut buf = [0; LEN];
707
708        #[cfg(any(feature = "safe_mem", not(unsafe··)))]
709        { let mut i = 0; while i < src.len() { buf[i] = src[i]; i += 1; } }
710
711        #[cfg(all(not(feature = "safe_mem"), unsafe··))]
712        // SAFETY: Lengths are equal (checked by assert), u8 is Copy, entire range is bounds-checked
713        unsafe { Ptr::copy_nonoverlapping(src.as_ptr(), buf.as_mut_ptr(), src.len()); }
714
715        buf
716    }
717
718    /// Returns a subslice without the given leading `byte`s.
719    #[must_use]
720    pub const fn trim_leading_bytes(slice: &[u8], byte: u8) -> &[u8] {
721        let mut start = 0;
722        while start < slice.len() && slice[start] == byte { start += 1; }
723        slice.split_at(start).1 // == &slice[start..]
724    }
725
726    /// Replaces the `old` leading byte with a `new` byte.
727    pub const fn replace_leading_bytes(slice: &mut [u8], old: u8, new: u8) {
728        let mut start = 0;
729        while start < slice.len() && slice[start] == old { slice[start] = new; start += 1; }
730    }
731}

733/// Helper for implementing slice operations for primitives.
734macro_rules! impl_prim {
735    () => {
736        impl_prim![
737            u8, u16, u32, u64, u128, usize,
738            i8, i16, i32, i64, i128, isize,
739            f32, f64,
740            bool, char
741        ];
742        #[cfg(nightly_float)]
743        impl_prim![f16, f128];
744    };
745    ($($t:ty),+) => { $( impl_prim![@$t]; )+ };
746    (@$t:ty) => {
747        impl Slice<$t> {
748            /// Checks the equality of two slices of primitives in compile-time.
749            #[must_use]
750            pub const fn eq(a: &[$t], b: &[$t]) -> bool {
751                is! { a.len() != b.len(); return false }
752                let mut i = 0;
753                while i < a.len() {
754                    is! { a[i] != b[i]; return false }
755                    i += 1;
756                }
757                true
758            }
759        }
760    };
761}
