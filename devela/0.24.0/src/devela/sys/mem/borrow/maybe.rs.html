12/// A container that may hold either a borrowed or owned value.
13///
14/// Unlike [`Cow`], this:
15/// - Uses the [`Ownership`] trait for flexible backing types.
16/// - Supports any `?Sized` type.
17/// - Has cleaner feature gating.
18///
19/// See also: [`Backing`][crate::Backing].
20///
21/// # Features
22/// If the `alloc` feature is disabled it can only contain borrowed values.
23#[non_exhaustive]
24#[derive(Clone, Debug)]
25pub enum MaybeOwned<'a, T: ?Sized + Ownership> {
26    /// A borrowed value.
27    Borrowed(&'a T),
28    /// An owned, heap-allocated value.
29    #[cfg(feature = "alloc")]
30    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
31    Owned(T::Backing),
32}
33
34impl<'a, T: ?Sized + Ownership> MaybeOwned<'a, T> {
35    /// Returns `true` if the value is borrowed.
36    pub const fn is_borrowed(&self) -> bool {
37        matches!(self, MaybeOwned::Borrowed(_))
38    }
39
40    /// Returns `true` if the value is owned.
41    pub const fn is_owned(&self) -> bool {
42        #[cfg(not(feature = "alloc"))]
43        return false;
44        #[cfg(feature = "alloc")]
45        matches!(self, MaybeOwned::Owned(_))
46    }
47
48    /// Returns a reference to the contained value, whether borrowed or owned.
49    pub fn borrowed(&self) -> &T {
50        match self {
51            MaybeOwned::Borrowed(r) => r,
52            #[cfg(feature = "alloc")]
53            MaybeOwned::Owned(b) => b.borrow(),
54        }
55    }
56
57    /// Converts to the owned backing type if not already owned.
58    #[cfg(feature = "alloc")]
59    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
60    pub fn owned(self) -> T::Backing
61    where
62        T: ToOwned<Owned = T::Backing>,
63    {
64        match self {
65            MaybeOwned::Borrowed(r) => r.to_owned(),
66            MaybeOwned::Owned(b) => b,
67        }
68    }
69
70    /// Converts the container into an owned value.
71    ///
72    /// If the value is borrowed, it will be cloned using `ToOwned`.
73    #[cfg(feature = "alloc")]
74    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
75    pub fn into_owned(self) -> T::Owned
76    where
77        T: ToOwned<Owned = <T as Ownership>::Backing>,
78    {
79        match self {
80            MaybeOwned::Borrowed(t) => t.to_owned(),
81            MaybeOwned::Owned(t) => t,
82        }
83    }
84}

88// AsRef, Deref
89impl<'a, T: ?Sized + Ownership> AsRef<T> for MaybeOwned<'a, T> {
90    fn as_ref(&self) -> &T {
91        match self {
92            MaybeOwned::Borrowed(t) => t,
93            #[cfg(feature = "alloc")]
94            MaybeOwned::Owned(t) => t.borrow(),
95        }
96    }
97}
98impl<'a, T: ?Sized + Ownership> Deref for MaybeOwned<'a, T> {
99    type Target = T;
100    fn deref(&self) -> &T {
101        self.as_ref()
102    }
103}
104
105// Eq, Ord, Hash
106impl<'a, T: ?Sized + Ownership + PartialEq> PartialEq for MaybeOwned<'a, T> {
107    fn eq(&self, other: &MaybeOwned<'a, T>) -> bool {
108        self.as_ref() == other.as_ref()
109    }
110}
111impl<'a, T: ?Sized + Ownership + Eq> Eq for MaybeOwned<'a, T> {}
112impl<'a, T: ?Sized + Ownership + PartialOrd> PartialOrd for MaybeOwned<'a, T> {
113    fn partial_cmp(&self, other: &MaybeOwned<'a, T>) -> Option<Ordering> {
114        self.as_ref().partial_cmp(other)
115    }
116}
117impl<'a, T: ?Sized + Ownership + Ord> Ord for MaybeOwned<'a, T> {
118    fn cmp(&self, other: &MaybeOwned<'a, T>) -> Ordering {
119        self.as_ref().cmp(other)
120    }
121}
122impl<'a, T: ?Sized + Ownership + Hash> Hash for MaybeOwned<'a, T> {
123    fn hash<H: Hasher>(&self, hasher: &mut H) {
124        self.as_ref().hash(hasher);
125    }
126}

128// From
129impl<'a, T: ?Sized + Ownership> From<&'a T> for MaybeOwned<'a, T> {
130    fn from(value: &'a T) -> Self {
131        MaybeOwned::Borrowed(value)
132    }
133}

135impl<'a, T: ?Sized + Ownership<Backing = Box<T>>> From<Box<T>> for MaybeOwned<'a, T> {
136    fn from(value: Box<T>) -> Self {
137        MaybeOwned::Owned(value)
