26/// # Safe methods.
27impl Mem {
28    /// Aligns `value` downward to the nearest multiple of `align`.
29    ///
30    /// This is equivalent to `value & !(align - 1)` but uses `wrapping_neg()`
31    /// which may generate better code on some architectures.
32    ///
33    /// # Requirements
34    /// - `align` must be a power of two (guaranteed when using [`MemLayout`][crate::MemLayout])
35    ///
36    /// # Examples
37    /// ```
38    /// # use devela::Mem;
39    /// assert_eq!(Mem::align_down(13, 8), 8);
40    /// assert_eq!(Mem::align_down(16, 8), 16);
41    /// ```
42    #[doc = crate::doc_!(vendor: "mini-alloc")]
43    #[must_use]
44    #[inline(always)]
45    pub const fn align_down(value: usize, align: usize) -> usize {
46        value & align.wrapping_neg()
47    }
48
49    /// Aligns `value` upward to the nearest multiple of `align`.
50    #[must_use]
51    #[inline(always)]
52    pub const fn align_up(value: usize, align: usize) -> usize {
53        (value + align - 1) & !(align - 1)
54    }
55
56    /// Checks if `value` is aligned to `align`.
57    #[must_use]
58    #[inline(always)]
59    pub const fn is_aligned(value: usize, align: usize) -> bool {
60        value & (align - 1) == 0
61    }
62
63    /// Runtime version of
64    /// [`MemAligned::is_compatible`][super::MemAligned::is_compatible] for raw pointers.
65    #[inline(always)]
66    pub fn is_aligned_to<T>(ptr: *const T, requirement: usize) -> bool {
67        let align = Mem::align_of::<T>();
68        requirement >= align && Mem::is_aligned(ptr as usize, align)
69    }
70
71    /// Returns the minimum alignment of the type in bytes.
72    ///
73    /// See `core::mem::`[`align_of`].
74    #[must_use]
75    pub const fn align_of<T>() -> usize {
76        align_of::<T>()
77    }
78
79    /// Returns the alignment of the pointed-to value in bytes.
80    ///
81    /// See `core::mem::`[`align_of_val`].
82    #[must_use]
83    pub const fn align_of_val<T: ?Sized>(val: &T) -> usize {
84        align_of_val(val)
85    }
86
87    /// Bitwise-copies a value.
88    ///
89    /// It is useful when you want to pass a function pointer to a combinator,
90    /// rather than defining a new closure.
91    ///
92    /// # Example
93    /// ```
94    /// # use devela::Mem;
95    /// let result_from_ffi_fn: Result<(), &i32> = Err(&1);
96    /// let result_copied: Result<(), i32> = result_from_ffi_fn.map_err(Mem::copy);
97    /// ```
98    // WAIT: [core::mem::copy](https://github.com/rust-lang/rust/issues/98262)
99    #[must_use]
100    pub const fn copy<T: Copy>(x: &T) -> T {
101        *x
102    }
103
104    /// Returns a value uniquely identifying the enum variant in v.
105    #[must_use]
106    pub const fn discriminant<T>(v: &T) -> Discriminant<T> {
107        discriminant(v)
108    }
109
110    /// Disposes of a value.
111    ///
112    /// See `core::mem::`[`drop`].
113    pub fn drop<T>(_x: T) {
114        drop(_x);
115    }
116
117    /// Takes ownership and “forgets” about `t` *without running its destructor*.
118    ///
119    /// See `core::mem::`[`forget`].
120    pub fn forget<T>(t: T) {
121        forget(t);
122    }
123
124    /// Returns true if dropping values of type T matters.
125    ///
126    /// See `core::mem::`[`needs_drop`].
127    #[must_use]
128    pub const fn needs_drop<T: ?Sized>() -> bool {
129        needs_drop::<T>()
130    }
131
132    /// Moves `src` into `dest`, returning the previous `dest` value.
133    ///
134    /// See `core::mem::`[`replace`].
135    #[must_use]
136    pub const fn replace<T>(dest: &mut T, src: T) -> T {
137        replace::<T>(dest, src)
138    }
139
140    /// Returns the size of a type in bytes.
141    ///
142    /// See `core::mem::`[`size_of`].
143    #[must_use]
144    pub const fn size_of<T>() -> usize {
145        size_of::<T>()
146    }
147
148    /// Returns the size of the pointed-to value in bytes.
149    /// See `core::mem::`[`size_of_val`].
150    #[must_use]
151    pub const fn size_of_val<T: ?Sized>(val: &T) -> usize {
152        size_of_val(val)
153    }
154
155    /// Swaps the values at two locations, without deinitializing either one.
156    ///
157    /// See `core::mem::`[`swap`].
158    pub const fn swap<T>(x: &mut T, y: &mut T) {
159        swap::<T>(x, y);
160    }
161
162    /// Replaces `dest` with `T::default()`, returning the previous `dest` value.
163    ///
164    /// See `core::mem::`[`take`].
165    #[must_use]
166    pub fn take<T: Default>(dest: &mut T) -> T {
167        take::<T>(dest)
168    }
169}

171/// # Extra methods
172impl Mem {
173    /// Returns the rounded up size in bytes from a size in bits.
174    ///
175    /// This is equivalent to `(bit_size + 7) / 8` but handles potential overflow.
176    #[must_use]
177    pub const fn bytes_from_bits(bit_size: usize) -> usize {
178        if let Some(t) = bit_size.checked_add(8 - 1) {
179            t / 8
180        } else {
181            Self::bytes_from_bits_cold()
182        }
183    }
184    #[cold] #[rustfmt::skip]
185    const fn bytes_from_bits_cold() -> usize { usize::MAX / 8 }
186}

226/// # Unsafe methods gated by `unsafe_slice`
227#[cfg(all(not(feature = "safe_data"), feature = "unsafe_slice"))]
228#[cfg_attr(nightly_doc, doc(cfg(feature = "unsafe_slice")))]
229impl Mem {
230    /// View any `T: Sync + Unpin + ?Sized` as `&[u8]`.
231    ///
232    /// This is a safer interface to `core::slice::`[`from_raw_parts`].
233    /// # Example
234    /// ```
235    /// # use devela::Mem;
236    /// #[repr(C)]
237    /// struct Data(u32);
238    ///
239    /// let data = Data(1234);
240    /// let bytes = Mem::as_bytes(&data);
241    ///
242    /// if cfg!(target_endian = "little") {
243    ///     assert!(bytes == &[210, 4, 0, 0]);
244    /// } else {
245    ///     assert!(bytes == &[0, 0, 4, 210]);
246    /// }
247    /// ```
248    #[doc = crate::doc_!(vendor: "rawbytes")]
249    #[must_use]
250    pub fn as_bytes<'t, T: Sync + Unpin + ?Sized + 't>(v: &T) -> &'t [u8] {
251        // SAFETY: `v` is valid; u8 has alignment 1, size_of_val(v) gives the exact byte length.
252        unsafe { from_raw_parts(v as *const _ as *const u8, size_of_val(v)) }
253    }
254
255    /// View any `T: Sync + Unpin + ?Sized` as `&mut [u8]`.
256    ///
257    /// This is a safer interface to `core::slice::`[`from_raw_parts_mut`].
258    /// # Examples
259    /// ```
260    /// # use devela::Mem;
261    /// #[repr(C)]
262    /// struct Data(u32);
263    ///
264    /// let mut data = Data(1234);
265    /// let bytes = Mem::as_bytes_mut(&mut data);
266    ///
267    /// if cfg!(target_endian = "little") {
268    ///     bytes[1] = 0;
269    ///     assert!(bytes == &[210, 0, 0, 0] && data.0 == 210);
270    /// } else {
271    ///     bytes[1] = 0;
272    ///     assert!(bytes == &[0, 0, 0, 210] && data.0 == 210);
273    /// }
274    /// ```
275    #[doc = crate::doc_!(vendor: "rawbytes")]
276    #[must_use]
277    pub fn as_bytes_mut<'t, T: Sync + Unpin + ?Sized + 't>(v: &mut T) -> &'t mut [u8] {
278        // SAFETY: `v` is a valid, exclusive reference;
279        // u8’s alignment is 1, and size_of_val(v) bounds the mutable slice.
280        unsafe { from_raw_parts_mut(v as *mut _ as *mut u8, size_of_val(v)) }
281    }
282
283    /// View any `T: Sync + Unpin + Sized` as `&[u8]` *compile-time* friendly.
284    ///
285    /// This is a safer interface to `core::slice::`[`from_raw_parts`], for `Sized` types.
286    /// # Examples
287    /// ```
288    /// # use devela::Mem;
289    /// const DATA: u32 = 1234;
290    /// const BYTES: &[u8] = Mem::as_bytes_sized(&DATA);
291    ///
292    /// if cfg!(target_endian = "little") {
293    ///     assert_eq!(BYTES, &[210, 4, 0, 0]);
294    /// } else {
295    ///     assert_eq!(BYTES, &[0, 0, 4, 210]);
296    /// }
297    /// ```
298    #[must_use]
299    pub const fn as_bytes_sized<T: Sync + Unpin>(v: &T) -> &[u8] {
300        // SAFETY: `v` is valid; casting to *const u8 is safe (u8 has alignment 1)
301        // and size_of::<T>() exactly covers the object.
302        unsafe { from_raw_parts(v as *const T as *const u8, size_of::<T>()) }
303    }
