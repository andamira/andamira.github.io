25define_error! { individual:
26    pub struct InvalidUtf8 {
27        /// The index in the given string up to which valid UTF-8 was verified.
28        pub valid_up_to: usize,
29        /// The length of the error in bytes, if known.
30        pub error_len: Option<usize>
31    }
32    DOC_INVALID_UTF8 = "Invalid Utf-8 found while interpreting a byte sequence.\n\n
33This is basically a replication of `core::str::`[`Utf8Error`]`.",
34    self+f => if let Some(len) = self.error_len {
35        write!(f, "Invalid UTF-8, valid up to: {}, len: {len}", self.valid_up_to)
36    } else { write!(f, "Invalid UTF-8, valid up to: {}", self.valid_up_to) }

51define_error! { composite: fmt(f)
52    #[doc = crate::TAG_TEXT!()]
53    /// An error composite of [`InvalidChar`] + [`InvalidUtf8`] + [`MismatchedCapacity`].
54    ///
55    /// Used in methods of:
56    /// [`StringNonul`][crate::StringNonul], and `StringU*`.
57    pub enum InvalidText {
58        DOC_INVALID_CHAR: Char(c|0: char) => InvalidChar(*c),
59        DOC_INVALID_UTF8: Utf8 {
60            #[doc = ""] valid_up_to: usize,
61            #[doc = ""] error_len: Option<usize>
62        } => InvalidUtf8 { valid_up_to: *valid_up_to, error_len: *error_len },
63        DOC_MISMATCHED_CAPACITY:
64            Capacity(c|0: Mismatch<Interval<usize>, usize>) => MismatchedCapacity(*c),
65    }
66}
67#[rustfmt::skip]
68impl From<Utf8Error> for InvalidText { fn from(f: Utf8Error) -> Self { Self::from_utf8_error(f) } }
69impl InvalidText {
70    /// Converts `core::str`[`Utf8Error`] to [`InvalidText::Utf8`] in compile-time.
71    #[must_use]
72    pub const fn from_utf8_error(from: Utf8Error) -> InvalidText {
73        let (valid_up_to, error_len) = (from.valid_up_to(), from.error_len());
74        InvalidText::Utf8 { valid_up_to, error_len }
75    }
76}

111        }
112    }
