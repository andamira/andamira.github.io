13impl Rgba8 {
14    /// New `Rgba8`.
15    pub const fn new(r: u8, g: u8, b: u8, a: u8) -> Rgba8 { Self { c: [r, g, b, a] } }
16    /// The red component.
17    pub const fn red(self) -> u8 { self.c[0] }
18    pub const fn r(self) -> u8 { self.c[0] }
19    /// The green component.
20    pub const fn green(self) -> u8 { self.c[1] }
21    pub const fn g(self) -> u8 { self.c[1] }
22    /// The blue component.
23    pub const fn blue(self) -> u8 { self.c[2] }
24    pub const fn b(self) -> u8 { self.c[2] }
25    /// The alpha component.
26    pub const fn alpha(self) -> u8 { self.c[3] }
27    pub const fn a(self) -> u8 { self.c[3] }
28}

30/// # General conversions
31impl Rgba8 {
32    /// Create from an array.
33    pub const fn from_array(c: [u8; 4]) -> Rgba8 {
34        Rgba8 { c }
35    }
36    /// Convert to an array.
37    pub const fn as_array(self) -> [u8; 4] {
38        self.c
39    }
40
41    /// Create from a tuple.
42    pub const fn from_tuple(c: (u8, u8, u8, u8)) -> Rgba8 {
43        Rgba8::new(c.0, c.1, c.2, c.3)
44    }
45    /// Convert to a tuple.
46    pub const fn to_tuple(self) -> (u8, u8, u8, u8) {
47        (self.r(), self.g(), self.b(), self.a())
48    }
49
50    /* Rgb8 */
51
52    /// Create from [`Rgb<u8>`].
53    ///
54    /// Adds the given `alpha` channel.
55    pub const fn from_rgb8(c: Rgb8, alpha: u8) -> Rgba8 {
56        Rgba8::new(c.r(), c.g(), c.b(), alpha)
57    }
58    /// Convert to [`Rgb<u8>`].
59    ///
60    /// Loses the alpha channel.
61    pub const fn to_rgb8(self) -> Rgb8 {
62        Rgb8::new(self.r(), self.g(), self.b())
63    }
64
65    /* packed u32 */
66
67    /// Create from a packed `u32` in `0xRRGGBB` format, with the given `alpha`.
68    ///
69    /// Any bits above `0x00FF_FFFF` are ignored.
70    pub const fn from_rgb8_packed(packed: u32, alpha: u8) -> Rgba8 {
71        Rgba8::new(
72            ((packed >> 16) & 0xFF) as u8,
73            ((packed >> 8) & 0xFF) as u8,
74            (packed & 0xFF) as u8,
75            alpha,
76        )
77    }
78    /// Convert to a packed `u32` in `0xRRGGBB` format, discarding alpha.
79    pub const fn to_rgb8_packed(self) -> u32 {
80        ((self.r() as u32) << 16) | ((self.g() as u32) << 8) | (self.b() as u32)
81    }
82
83    /// Create from a packed `u32` in `0xRRGGBBAA` format (big-endian).
84    pub const fn from_rgba8_packed(packed: u32) -> Rgba8 {
85        Rgba8::new(
86            ((packed >> 24) & 0xFF) as u8,
87            ((packed >> 16) & 0xFF) as u8,
88            ((packed >> 8) & 0xFF) as u8,
89            (packed & 0xFF) as u8,
90        )
91    }
92    /// Convert to a packed `u32` in `0xRRGGBBAA` format (big-endian).
93    pub const fn to_rgba8_packed(self) -> u32 {
94        ((self.r() as u32) << 24)
95            | ((self.g() as u32) << 16)
96            | ((self.b() as u32) << 8)
97            | (self.a() as u32)
98    }
99
100    /* u16 */
101
102    /// Create from `Rgba<u16>` by scaling each component proportionally.
103    pub fn from_rgba16(from: Rgba16) -> Rgba8 {
104        Rgba8::new(
105            ((from.c[0] + 128) / 257) as u8, // Rounding via +128
106            ((from.c[1] + 128) / 257) as u8,
107            ((from.c[2] + 128) / 257) as u8,
108            ((from.c[3] + 128) / 257) as u8,
109        )
110    }
111    /// Convert to Rgba16 by scaling each component proportionally.
112    pub fn to_rgba16(self) -> Rgba16 {
113        Rgba16::new(
114            (self.c[0] as u16) * 257, // 255 * 257 = 65535
115            (self.c[1] as u16) * 257,
116            (self.c[2] as u16) * 257,
117            (self.c[3] as u16) * 257,
118        )
119    }
120}

210#[cfg(test)]
211mod tests {
212    use super::*;
213
214    const C8: Rgb8 = Rgb8::new(10, 20, 30);
215    const CA8: Rgba8 = Rgba8::new(10, 20, 30, 40);
216    const CA16: Rgba16 = Rgba16::new(2570, 5140, 7710, 10280);
217    const H8: u32 = 0x_0A_14_1E;
218    const HA8: u32 = 0x_0A_14_1E_28;
219
220    #[test]
221    fn general_conversions() {
222        // array/tuple
223        assert_eq![Rgba8::from_array([10, 20, 30, 40]), CA8];
224        assert_eq![CA8.as_array(), [10, 20, 30, 40]];
225        assert_eq![Rgba8::from_tuple((10, 20, 30, 40)), CA8];
226        assert_eq![CA8.to_tuple(), (10, 20, 30, 40)];
227        // rgb
228        assert_eq![Rgba8::from_rgb8(C8, 40), CA8];
229        assert_eq![CA8.to_rgb8(), C8];
230        // packed rgba
231        assert_eq![Rgba8::from_rgba8_packed(HA8), CA8];
232        assert_eq![CA8.to_rgba8_packed(), HA8];
233        // packed rgb
234        assert_eq![Rgba8::from_rgb8_packed(H8, 40), CA8];
235        assert_eq![CA8.to_rgb8_packed(), H8];
236        // u16
237        assert_eq![Rgba8::from_rgba16(CA16), CA8];
238        assert_eq![CA8.to_rgba16(), CA16];
239    }
240
241    #[test]
242    #[cfg(feature = "_float_f32")]
243    fn f32_conversions() {
244        let f = RgbF32::new(0.039215688, 0.078431375, 0.11764706);
245        let fa = RgbaF32::new(0.039215688, 0.078431375, 0.11764706, 0.15686275);
246        assert_eq![Rgba8::from_rgb_f32(f, 0.156), CA8];
247        assert_eq![CA8.to_rgb_f32(), f];
248        assert_eq![Rgba8::from_rgba_f32(fa), CA8];
249        assert_eq![CA8.to_rgba_f32(), fa];
250    }
251
252    #[test]
253    #[cfg(feature = "_float_f64")]
254    fn f64_conversions() {
255        let f = RgbF64::new(0.0392156862745098, 0.0784313725490196, 0.11764705882352941);
256        let fa = RgbaF64::new(
257            0.0392156862745098,
258            0.0784313725490196,
259            0.11764705882352941,
260            0.1568627450980392,
261        );
262        assert_eq![Rgba8::from_rgb_f64(f, 0.156), CA8];
263        assert_eq![CA8.to_rgb_f64(), f];
264        assert_eq![Rgba8::from_rgba_f64(fa), CA8];
265        assert_eq![CA8.to_rgba_f64(), fa];
