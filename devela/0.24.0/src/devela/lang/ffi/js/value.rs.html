7#[cfg(feature = "alloc")]
8use devela::String;
9use devela::{js_bool, js_number, js_uint32};
10
11/// A JavaScript value for FFI communication.
12///
13/// # Variants
14/// - `Null`: JS `null`
15/// - `Undefined`: JS `undefined`  
16/// - `Boolean(bool)`: JS boolean (`true`/`false`)
17/// - `Object(u32)`: Handle to JS object (registry index)
18/// - `Str(&'static str)`: Static string (no allocation)
19/// - `String(String)`: Heap-allocated string (requires `alloc` feature)
20/// - `Number(f64)`: JS number (always 64-bit float)
21#[non_exhaustive]
22#[derive(Clone, Debug, PartialEq)]
23#[allow(missing_docs)]
24pub enum JsValue {
25    /* misc. */
26    Null,
27    Undefined,
28    Boolean(js_bool),
29    Object(js_uint32),
30
31    /* strings */
32    Str(&'static str),
33    #[cfg(feature = "alloc")]
34    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
35    String(String),
36
37    /* numbers */
38    Number(js_number),
39}
40
41impl JsValue {
42    /// Returns the number tag associated with the current variant.
43    #[inline(always)]
44    pub const fn type_tag(&self) -> u8 {
45        match self {
46            Self::Null => 0,
47            Self::Undefined => 1,
48            Self::Boolean(_) => 2,
49            Self::Object(_) => 3,
50            //
51            Self::Str(_) => 4,
52            #[cfg(feature = "alloc")]
53            Self::String(_) => 5,
54            //
55            Self::Number(_) => 6,
56        }
57    }
58
59    /// Encodes the value into a byte buffer for WASM FFI.
60    ///
61    /// Returns the total bytes written.
62    ///
63    /// # Layout
64    /// ```text
65    /// [1B: type tag][payload...]
66    /// ```
67    /// # Panics
68    /// If buffer is too small (debug builds only).
69    ///
70    pub fn encode_into(&self, buf: &mut [u8]) -> usize {
71        let mut offset = 0;
72        buf[offset] = self.type_tag(); // 1 Byte
73        offset += 1;
74
75        match self {
76            Self::Boolean(b) => {
77                buf[offset] = *b as u8;
78                offset += 1;
79            }
80            Self::Object(handle) => {
81                buf[offset..offset + 4].copy_from_slice(&handle.to_le_bytes());
82                offset += 4;
83            }
84            Self::Str(s) => {
85                offset += self.encode_str(s, &mut buf[offset..]);
86            }
87            #[cfg(feature = "alloc")]
88            Self::String(s) => {
89                offset += self.encode_str(s.as_str(), &mut buf[offset..]);
90            }
91            Self::Number(n) => {
92                buf[offset..offset + 8].copy_from_slice(&n.to_le_bytes());
93                offset += 8;
94            }
95            Self::Null | Self::Undefined => (),
96        }
97        offset
98    }
99
100    #[inline(always)]
101    fn encode_str(&self, s: &str, buf: &mut [u8]) -> usize {
102        let (bytes, len) = (s.as_bytes(), s.len());
103        buf[..4].copy_from_slice(&(len as u32).to_le_bytes()); // encode length prefix
104        buf[4..4 + len].copy_from_slice(bytes); // encode string byte content
105        4 + len // return total bytes written
106    }
107}
