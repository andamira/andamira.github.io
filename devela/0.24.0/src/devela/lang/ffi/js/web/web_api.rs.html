134    /// Callback dispatcher for WebAssembly events without args.
135    ///
136    /// - This function is used by JavaScript to invoke a Rust function pointer.
137    /// - It allows Rust event listeners to execute when triggered by JS.
138    /// - The `callback_ptr` is a function pointer cast from JS,
139    ///   which is then transmuted into a callable Rust function.
140    ///
141    /// # Safety
142    /// `callback_ptr` must be a valid function pointer.
143    ///
144    /// # Example
145    /// ```ignore
146    /// #[unsafe(no_mangle)]
147    /// pub extern "C" fn my_callback() { Web::console_log("Button clicked!"); }
148    /// Web::event_add_listener("#my_button", WebEventKind::Click, my_callback);
149    /// ```
150    #[unsafe(no_mangle)]
151    pub unsafe extern "C" fn wasm_callback(callback_ptr: usize) {
152        let callback = callback_ptr as *const ();
153        let callback: extern "C" fn() = unsafe { transmute(callback) };
154        callback();
155    }
156    /// WebAssembly mouse event callback dispatcher.
157    ///
158    /// - Called from JavaScript when a mouse event is fired.
159    /// - Passes the `WebEventMouse` struct to the Rust callback.
160    ///
161    /// # Safety
162    /// - `callback_ptr` must be a valid function pointer.
163    #[unsafe(no_mangle)]
164    pub unsafe extern "C" fn wasm_callback_mouse(callback_ptr: usize, button: js_int32,
165        buttons: js_int32, x: js_number, y: js_number, etype: js_int32, time_stamp: js_number) {
166        let callback = callback_ptr as *const ();
167        let callback: extern "C" fn(WebEventMouse) = unsafe { transmute(callback) };
168        let etype = WebEventKind::from_repr(etype as u8);
169        let time_stamp = JsInstant::from_millis_f64(time_stamp);
170        callback(WebEventMouse::new(x, y, button as u8, buttons as u8, etype, time_stamp));
171    }
172    /// WebAssembly mouse event callback dispatcher.
173    ///
174    /// - Called from JavaScript when a pointer event is fired.
175    /// - Passes the `WebEventPointer` struct to the Rust callback.
176    ///
177    /// # Safety
178    /// - `callback_ptr` must be a valid function pointer.
179    // WAIT: In firefox + linux + X11 needs the env variable `MOZ_USE_XINPUT2=1`
180    // https://bugzilla.mozilla.org/show_bug.cgi?id=1606832 (closed)
181    // https://bugzilla.mozilla.org/show_bug.cgi?id=1207700
182    // https://bugzilla.mozilla.org/show_bug.cgi?id=1822714
183    #[unsafe(no_mangle)]
184    pub unsafe extern "C" fn wasm_callback_pointer(callback_ptr: usize, id: js_int32,
185        x: js_number, y: js_number, pressure: js_number, tilt_x: js_int32, tilt_y: js_int32,
186        twist: js_int32, etype: js_int32, time_stamp: js_number) {
187        let callback = callback_ptr as *const ();
188        let callback: extern "C" fn(WebEventPointer) = unsafe { transmute(callback) };
189        let etype = WebEventKind::from_repr(etype as u8);
190        let time_stamp = JsInstant::from_millis_f64(time_stamp);
191        callback(WebEventPointer::new(x, y, pressure, id, tilt_x as i8, tilt_y as i8, twist as u16,
192            etype, time_stamp));
193    }

426/// Web API workers
427#[rustfmt::skip]
428#[cfg(all(feature = "unsafe_ffi", not(windows)))]
429#[cfg_attr(nightly_doc, doc(cfg(all(feature = "unsafe_ffi", target_arch = "wasm32"))))]
430impl Web {
431    /// Spawns a Web Worker and returns its ID.
432    pub fn worker_spawn(script: &str) -> Result<WebWorker, WebWorkerError> {
433        let id = unsafe { worker_spawn(script.as_ptr(), script.len()) };
434        if id == 0 { Err(WebWorkerError::InvalidScript) } else { Ok(WebWorker { id }) }
435    }
436    /// Checks if this worker is still active by querying JavaScript.
437    pub fn worker_is_active(worker: WebWorker) -> js_bool { worker_is_active(worker.id()) }
438    /// Stops a specific Web Worker by ID.
439    pub fn worker_stop(worker: WebWorker) { worker_stop(worker.id()); }
440    /// Stops all Web Workers.
441    pub fn worker_stop_all() { worker_stop_all(); }
442    /// Returns the number of active workers.
443    pub fn worker_list_len() -> usize { worker_list_len() as usize }
444    /// Returns the list of active worker IDs.
445    #[cfg(feature = "alloc")]
446    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
447    pub fn worker_list() -> Vec<WebWorker> {
448        let len = worker_list_len() as usize;
449        let mut workers = vec![WebWorker::default(); len];
450        let count = Web::worker_list_buf(&mut workers);
451        workers.truncate(count);
452        workers
453    }
454    /// Writes active worker handles into a buffer and returns the number written.
455    pub fn worker_list_buf(buffer: &mut [WebWorker]) -> usize {
456        let len = worker_list_len() as usize;
457        let count = len.min(buffer.len());
458        unsafe { worker_list(buffer.as_mut_ptr() as *mut js_uint32, count as js_uint32); }
459        count
460    }
461    /// Sends a message to a specific Web Worker.
462    pub fn worker_send_message(worker: WebWorker, msg: &str) {
463        unsafe { worker_send_message(worker.id(), msg.as_ptr(), msg.len()); }
464    }
465    /// Requests execution of JavaScript inside a worker.
466    pub fn worker_eval(worker: WebWorker, js_code: &str) -> WebWorkerJob {
467        let id = unsafe { worker_eval(worker.id(), js_code.as_ptr(), js_code.len()) };
468        WebWorkerJob { worker, id }
469    }
470    /// Polls for the result of a JavaScript execution in a worker, and writes it into `buffer`.
471    ///
472    /// If ready, returns the number of bytes written to the buffer.
473    pub fn worker_poll(job: WebWorkerJob, buffer: &mut [u8])
474        -> TaskPoll<Result<usize, WebWorkerError>> {
475        if !job.worker().is_active() { return TaskPoll::Ready(Err(WebWorkerError::WorkerNotFound)); }
476        let written = unsafe { worker_poll(job.id(), buffer.as_mut_ptr(), buffer.len()) };
477        match written {
478            0 => TaskPoll::Pending,
479            js_uint32::MAX => TaskPoll::Ready(Err(WebWorkerError::JobNotFound)),
480            _ => TaskPoll::Ready(Ok(written as usize)),
481        }
482    }
483    /// Polls for the result of a JavaScript execution in a worker.
484    #[cfg(feature = "alloc")]
485    #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
486    pub fn worker_poll_alloc(job: WebWorkerJob) -> TaskPoll<Result<String, WebWorkerError>> {
487        if !job.worker().is_active() { return TaskPoll::Ready(Err(WebWorkerError::WorkerNotFound)); }
488        let mut first_check = true;
489        let result = Js::read_string_capped(128, false, |ptr, cap| {
490            let res = unsafe { worker_poll(job.id(), ptr, cap as usize) as js_int32 };
491            if first_check {
492                first_check = false; // Intercept status codes before bothering with decoding
493                if res == 0 { return 0; } else if res == -1 { return -1; } // pending or not found
494            }
495            res
496        });
497        match result.as_str() {
498            "" => TaskPoll::Pending, // Covers 0 and -1 (mapped above)
499            _ => TaskPoll::Ready(Ok(result)),
500        }
501    }
502    /// Cancels an ongoing JavaScript evaluation.
503    pub fn worker_eval_cancel(job: WebWorkerJob) { worker_cancel_eval(job.id()); }
