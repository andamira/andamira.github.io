6/// Asserts the equality of a series of expressions.
7///
8/// Similar to [`assert_eq`] but supports more than 2 terms to test for equality.
9///
10/// # Panics
11// TODO
12#[macro_export]
13#[cfg_attr(cargo_primary_package, doc(hidden))]
14macro_rules! assert_eq_all {
15    ($first:expr, $($rest:expr),+ $(,)?) => {{
16        let first_val = &$first;
17        $(
18            let rest_val = &$rest;
19            assert!(
20                first_val == rest_val,
21                "Assertion failed: ({}) is not equal to ({})\n  left: {}\n right: {}",
22                stringify!($first),
23                stringify!($rest),
24                first_val,
25                rest_val,
26            );
27        )+
28    }};
29}
30#[doc(inline)]
31pub use assert_eq_all;

33/// Asserts the approximate equality of a series of expressions within `tolerance`.
34///
35/// This macro should work with any numeric type that supports comparison and
36/// subtraction, including signed and unsigned integers and floating-point numbers.
37///
38/// The given `$tolerance` must be a non-negative number.
39///
40/// # Panics
41// TODO
42///
43/// # Examples
44/// The following examples compile:
45/// ```
46/// # use devela::assert_approx_eq_all;
47/// assert_approx_eq_all![tolerance: 0.01, 1.0, 1.001, 0.999]; // up to 0.01 away from 1.0
48/// assert_approx_eq_all![tolerance: 1_u32, 4, 3, 5]; // up to 1 away from 4
49///
50/// assert_approx_eq_all![tolerance: 0.01, -1.0, -1.001, -0.999];
51/// assert_approx_eq_all![tolerance: 1_i32, -4, -3, -5];
52/// assert_approx_eq_all![tolerance: 0_i32, 3, 3];
53/// ```
54/// The following examples panic:
55/// ```should_panic
56/// # use devela::assert_approx_eq_all;
57/// assert_approx_eq_all![tolerance: 0.01, 1.0, 1.001, 0.989]; // |0.989 - 1.0| > |0.01|
58/// ```
59/// ```should_panic
60/// # use devela::assert_approx_eq_all;
61/// assert_approx_eq_all![tolerance: 1_u32, 4, 3, 5, 6]; // |6 - 4| > |1|
62/// ```
63/// ```should_panic
64/// # use devela::assert_approx_eq_all;
65/// assert_approx_eq_all![tolerance: 1_u32, 3, 4, 5]; // |5 - 3| > |1|
66/// ```
67/// ```should_panic
68/// # use devela::assert_approx_eq_all;
69/// assert_approx_eq_all![tolerance: -0.01, 1.0, 1.001, 0.999]; // tolerance: -0.01 < 0
70/// ```
71/// ```should_panic
72/// # use devela::assert_approx_eq_all;
73/// assert_approx_eq_all![tolerance: -1_i32, 4, 3, 5]; // tolerance: -1 < 0
74/// ```
75#[macro_export]
76#[cfg_attr(cargo_primary_package, doc(hidden))]
77macro_rules! assert_approx_eq_all {
78    (tolerance: $tolerance:expr, $first:expr, $($rest:expr),+ $(,)?) => {{
79        let first_val = $first;
80        $(
81            let rest_val = $rest;
82            // Calculate the absolute difference without relying on `abs`:
83            let difference =
84                if first_val > rest_val { first_val - rest_val } else { rest_val - first_val };
85            assert!(
86                difference <= $tolerance,
87                "Assertion failed: ({}) is not approximately equal to ({})
88      left: {}\n     right: {}\n tolerance: {}",
89                stringify!($first), stringify!($rest), first_val, rest_val, $tolerance,
90            );
91        )+
92    }};
93}
94#[doc(inline)]
95pub use assert_approx_eq_all;
