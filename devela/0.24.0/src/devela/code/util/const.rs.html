3//! Defines the CONST! macro.
4//
5
6/// A helper for constructing macro constants.
7///
8/// It accepts either a series of expressions or a series of functions.
9///
10/// # Examples
11/// ```
12/// # use devela::CONST;
13/// CONST!{ /* Supports empty declarations */ }
14///
15/// // Supports any expresion
16/// CONST!{EXPR = 2 * 15 / 3}
17/// assert_eq![EXPR![], 10u8];
18/// assert_eq![EXPR![], 10i8];
19///
20/// CONST!{
21///     /// Supports docs, attributes, and visibility.
22///     #[macro_export] // attribute needed if public
23///     pub ARRAY = [1, 2, 3]
24/// }
25/// assert_eq![ARRAY![], [1u16, 2, 3]];
26/// assert_eq![ARRAY![], [1i32, 2, 3]];
27///
28/// // Supports multiple definitions of constant expressions, ended with ;
29/// CONST! {
30///     DOC_1 = "Document expression." ;
31///     DOC_2 = "Document expression." ;
32/// }
33/// assert_eq![DOC_1![], "Document expression."];
34///
35/// // A good use-case is for repeated documentation
36/// /// Function 1, version a.
37/// #[doc = DOC_1!()]
38/// pub fn version_1a() {}
39/// /// Function 1, version b.
40/// #[doc = DOC_1!()]
41/// pub fn version_1b() {}
42///
43/// // Supports multiple definitions of constants functions, ended with ;
44/// CONST! {
45///     /// Supports *const* functions.
46///     FN_1 =
47///     /// Returns `n Ã— 5`.
48///     #[inline] #[must_use]
49///     pub const fn fn_1(n: i32) -> i64 { (n * 5) as i64 };
50///
51///     /// You can repeat functions.
52///     pub(crate) FN_2 = pub const fn fn_2(c: char) { };
53///
54///     /// Supports optional *unsafe*.
55///     FN_3 = pub const unsafe fn fn_3() {};
56///
57///     // NOTE: It's not possible to mix expressions and functions in the same invocation.
58///     // EXPR_ERR = "Compile fails if this line is uncommented";
59/// }
60/// pub struct Fns;
61/// impl Fns {
62///     FN_1!{}
63///     FN_2!{}
64///     FN_3!{}
65/// }
66///
67/// assert_eq![Fns::fn_1(0i32), 0i64];
68/// assert_eq![Fns::fn_1(5), 25];
69/// let _: () = Fns::fn_2('a');
70/// unsafe { Fns::fn_3(); }
71///
72/// // Supports giving a shared visibility for all defined constants
73/// CONST! { pub(crate),
74///     E1 = 1 + 1;
75///     E2 = 2 + 2;
76///     // pub E3 = 3 + 3; // shared visibility can't be overriden
77/// }
78/// CONST! { pub(crate),
79///     F1 = pub const fn f1(a: i32) -> i32 { a + 1 };
80///     F2 = pub const fn f2(a: i32) -> i32 { a + 2 };
81///     // pub F3 = pub const fn f3(a: i32) -> i32 { a + 3 };
82/// }
83/// ```
84// Related links
85// - https://doc.rust-lang.org/reference/items/external-blocks.html#functions
86#[doc(hidden)]
87#[macro_export]
88macro_rules! _CONST {
89    (
90    // Either multiple `const fn`
91    $(
92        $(#[$CONST_ATTRS:meta])*
93        $item_vis:vis $CONST_NAME:ident =
94            $(#[$fn_attrs:meta])*
95            $fn_vis:vis const
96            $(async$($_a:block)?)? $(safe$($_s:block)?)? $(unsafe$($_u:block)?)?
97            fn $fn:ident($($param:ident: $param_ty:ty),* $(,)?)
98        $(-> $fn_return:ty)?
99        $fn_body:block
100    );* $(;)?) => {
101        $(
102            $(#[$CONST_ATTRS])*
103            #[allow(unused_macros)]
104            macro_rules! $CONST_NAME {
105                () => {
106                    $(#[$fn_attrs])*
107                    $fn_vis const $(async$($_a)?)? $(safe$($_s)?)? $(unsafe$($_u)?)?
108                    fn $fn($($param: $param_ty),*) $(-> $fn_return)? $fn_body
109                }
110            }
111            $item_vis use $CONST_NAME;
112        )*
113    };
114    (
115    $shared_vis:vis, // (shared visibility alternative)
116    $(
117        $(#[$CONST_ATTRS:meta])*
118        $CONST_NAME:ident =
119            $(#[$fn_attrs:meta])*
120            $fn_vis:vis const
121            $(async$($_a:block)?)? $(safe$($_s:block)?)? $(unsafe$($_u:block)?)?
122            fn $fn:ident($($param:ident: $param_ty:ty),* $(,)?)
123        $(-> $fn_return:ty)?
124        $fn_body:block
125    );* $(;)?) => {
126        $(
127            $(#[$CONST_ATTRS])*
128            #[allow(unused_macros)]
129            macro_rules! $CONST_NAME {
130                () => {
131                    $(#[$fn_attrs])*
132                    $shared_vis const $(async$($_a)?)? $(safe$($_s)?)? $(unsafe$($_u)?)?
133                    fn $fn($($param: $param_ty),*) $(-> $fn_return)? $fn_body
134                }
135            }
136            $shared_vis use $CONST_NAME;
137        )*
138    };
139    (
140
141    // Either multiple expressions
142    $(
143        $(#[$CONST_ATTRS:meta])*
144        $item_vis:vis $CONST_NAME:ident = $expr:expr
145     );* $(;)?) => {
146        $(
147            $(#[$CONST_ATTRS])*
148            #[allow(unused_macro)]
149            macro_rules! $CONST_NAME { () => { $expr } }
150            $item_vis use $CONST_NAME;
151        )*
152    };
153    (
154    $shared_vis:vis, // (shared visibility alternative)
155    $(
156        $(#[$CONST_ATTRS:meta])*
157        $CONST_NAME:ident = $expr:expr
158     );* $(;)?) => {
159        $(
160            $(#[$CONST_ATTRS])*
161            #[allow(unused_macro)]
162            macro_rules! $CONST_NAME { () => { $expr } }
163            $shared_vis use $CONST_NAME;
164        )*
165    };
166}
167#[doc(inline)]
168pub use _CONST as CONST;
