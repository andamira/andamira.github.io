16#[doc = crate::TAG_QUANT!()]
17/// Creates an [`Interval`] using extended range notation.
18///
19/// # Syntax
20///
21/// ## Closed Lower Bound `[l, …`
22/// ```text
23/// [l, u]    interval![l, ..= u]    // closed-closed
24/// [l, u)    interval![l, .. u]     // closed-open
25/// [l, ∞)    interval![l, ..]       // closed-unbounded
26/// ```
27///
28/// ## Open Lower Bound `(l, …` *extension syntax*
29/// ```text
30/// (l, u)    interval![l <.. u]     // open-open
31/// (l, u]    interval![l <..= u]    // open-closed
32/// (l, ∞)    interval![l <..]       // open-unbounded
33/// ```
34///
35/// ## Unbounded Lower Bound `(-∞, …`
36/// ```text
37/// (-∞, u]   interval![..= u]       // unbounded-closed
38/// (-∞, u)   interval![.. u]        // unbounded-open
39/// (-∞, ∞)   interval![..]          // fully unbounded (inferred type)
40/// (-∞, ∞)   interval![.., T]       // fully unbounded (explicit type) (e.g., `.., i32`)
41/// ```
42///
43/// # Notes
44/// - Commas required for expressions: `interval![a(), ..= b()]`
45/// - Optional for literals/blocks: `interval![1..=3]` ≡ `interval![1, ..=3]`
46/// - `<..` indicates open-left bounds (non-standard Rust extension)
47///
48/// # Example
49/// ```
50/// # use devela::interval;
51/// # let (x, y) = (10, 20);
52/// # fn calc() -> i32 { 10 }
53/// # fn other() -> i32 { 20 }
54/// // Literals
55/// interval![1 ..=3];    // [1, 3]
56/// interval![1<.. 3];    // (1, 3)
57///
58/// // Expressions
59/// interval![(x+1), ..= (y*2)];
60/// interval![{ calc() }, <.. { other() }];
61///
62/// // Unbounded
63/// interval![.., i32]; // (-∞, ∞) as i32
64/// interval![..=10];   // (-∞, 10]
65/// interval![1..];     // [1, ∞)
66/// ```
67#[doc(hidden)]
68#[macro_export]
69#[rustfmt::skip]
70macro_rules! _interval {
71    (
72    /* expressions */
73
74     $l:expr,  ..= $u:expr) => { $crate::Interval::closed($l, $u) }; // lower closed
75    ($l:expr,  ..  $u:expr) => { $crate::Interval::closed_open($l, $u) };
76    ($l:expr,  ..         ) => { $crate::Interval::closed_unbounded($l) };
77    ($l:expr, <..  $u:expr) => { $crate::Interval::open($l, $u) }; // lower open
78    ($l:expr, <..= $u:expr) => { $crate::Interval::open_closed($l, $u) };
79    ($l:expr, <..         ) => { $crate::Interval::open_unbounded($l) };
80    (          ..= $u:expr) => { $crate::Interval::unbounded_closed($u) }; // lower unbounded
81    (          ..  $u:expr) => { $crate::Interval::unbounded_open($u) };
82    (
83    /* blocks */
84
85     $l:block  ..= $u:block) => { $crate::Interval::closed($l, $u) }; // lower closed
86    ($l:block  ..  $u:block) => { $crate::Interval::closed_open($l, $u) };
87    ($l:block  ..          ) => { $crate::Interval::closed_unbounded($l) };
88    ($l:block <..  $u:block) => { $crate::Interval::open($l, $u) }; // lower open
89    ($l:block <..= $u:block) => { $crate::Interval::open_closed($l, $u) };
90    ($l:block <..          ) => { $crate::Interval::open_unbounded($l) };
91    (          ..= $u:block) => { $crate::Interval::unbounded_closed($u) }; // lower unbounded
92    (          ..  $u:block) => { $crate::Interval::unbounded_open($u) };
93    (
94    /* literals */
95
96     $l:literal  ..= $u:literal) => { $crate::Interval::closed($l, $u) }; // lower closed
97    ($l:literal  ..  $u:literal) => { $crate::Interval::closed_open($l, $u) };
98    ($l:literal  ..            ) => { $crate::Interval::closed_unbounded($l) };
99    ($l:literal <..  $u:literal) => { $crate::Interval::open($l, $u) }; // lower open
100    ($l:literal <..= $u:literal) => { $crate::Interval::open_closed($l, $u) };
101    ($l:literal <..            ) => { $crate::Interval::open_unbounded($l) };
102    (            ..= $u:literal) => { $crate::Interval::unbounded_closed($u) }; // lower unbounded
103    (            ..  $u:literal) => { $crate::Interval::unbounded_open($u) };
104    (
105    /* fully unbounded variants (explicit and inferred type) */
106
107     .., $T:ty) => { $crate::Interval::<$T>::unbounded() }; // has to come first
108    (..       ) => { $crate::Interval::unbounded() };
109
110    (
111    /* syntax error msg */
112
113    $($t:tt)*) => {
114        compile_error! { "Invalid interval syntax. Expected forms like: \
115                      l..u, l..=u, l<..u, l<..=u, ..x, ..=x, ..,Type" }
116    };

121#[doc = crate::TAG_QUANT!()]
122/// A range of values with `lower` and `upper` [`Bound`]s.
123///
124/// The `Interval` type allows modeling ranges of values with optional inclusion
125/// or exclusion at each bound. This is useful for mathematical operations,
126/// range checks, and interval arithmetic.
127///
128/// See also the [`interval!`] macro.
129#[doc(alias = "Range")]
130#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
131pub struct Interval<T> {
132    /// The lower bound (also known as the *start* bound, or the *left* bound).
133    pub lower: Bound<T>,
134    /// The upper bound (also known as the *end*, bound or the *right* bound).
135    pub upper: Bound<T>,
136}
137
138/// # Methodical constructors
139impl<T> Interval<T> {
140    // lower-closed
141
142    /// Creates a closed interval $[l, u]$ `lower..=upper` [`RangeInclusive`].
143    #[must_use]
144    pub const fn closed(lower: T, upper: T) -> Self {
145        Self::new(Bound::Included(lower), Bound::Included(upper))
146    }
147    /// Creates a half-open interval $[l, u)$ `lower..upper` [`Range`].
148    #[must_use]
149    pub const fn closed_open(lower: T, upper: T) -> Self {
150        Self::new(Bound::Included(lower), Bound::Excluded(upper))
151    }
152    /// Creates an interval $[l, ∞)$ `lower..` [`RangeFrom`].
153    #[must_use]
154    pub const fn closed_unbounded(lower: T) -> Self {
155        Self::new(Bound::Included(lower), Bound::Unbounded)
156    }
157
158    // lower-open
159
160    /// Creates an open interval $(l, u)$ *`lower..<upper`*.
161    #[must_use]
162    pub const fn open(lower: T, upper: T) -> Self {
163        Self::new(Bound::Excluded(lower), Bound::Excluded(upper))
164    }
165
166    /// Creates a half-open interval $(l, u]$ *`lower..<=upper`*.
167    #[must_use]
168    pub const fn open_closed(lower: T, upper: T) -> Self {
169        Self::new(Bound::Excluded(lower), Bound::Included(upper))
170    }
171
172    /// Creates an interval $(l, ∞)$ *`lower..<`*.
173    #[must_use]
174    pub const fn open_unbounded(lower: T) -> Self {
175        Self::new(Bound::Excluded(lower), Bound::Unbounded)
176    }
177
178    // lower-unbounded
179
180    /// Creates an unbounded interval $(-∞, ∞)$ `..` [`RangeFull`].
181    #[must_use]
182    pub const fn unbounded() -> Self {
183        Self::new(Bound::Unbounded, Bound::Unbounded)
184    }
185    /// Creates an interval $(-∞, u]$ `..upper` [`RangeTo`].
186    #[must_use]
187    pub const fn unbounded_closed(upper: T) -> Self {
188        Self::new(Bound::Unbounded, Bound::Included(upper))
189    }
190    /// Creates an interval $(-∞, u)$ `..=upper` [`RangeToInclusive`].
191    #[must_use]
192    pub const fn unbounded_open(upper: T) -> Self {
193        Self::new(Bound::Unbounded, Bound::Excluded(upper))
194    }
195}

310impl<T: PartialOrd> Interval<T> {
311    /// Checks if the interval is empty (contains no values).
312    ///
313    /// An interval is empty if:
314    /// - The bounds exclude each other, such as `(x, x)`, `[x, x)`, or `(x, x]`.
315    /// - The `lower` bound is strictly greater than the `upper` bound.
316    ///
317    /// Unbounded intervals are never empty.
318    #[must_use]
319    pub fn is_empty(&self) -> bool {
320        match (&self.lower, &self.upper) {
321            (Bound::Unbounded, _) | (_, Bound::Unbounded) => false,
322            (Bound::Included(l), Bound::Included(u)) => l > u,
323            (Bound::Included(l), Bound::Excluded(u)) => l >= u,
324            (Bound::Excluded(l), Bound::Included(u)) => l >= u,
325            (Bound::Excluded(l), Bound::Excluded(u)) => l >= u,
326        }
327    }
328
329    /// Validates that the interval bounds are ordered correctly.
330    ///
331    /// Returns `true` if the lower bound is less than or equal to the upper bound.
332    /// Unbounded intervals are always considered well ordered.
333    #[must_use]
334    pub fn is_well_ordered(&self) -> bool {
335        match (&self.lower, &self.upper) {
336            (Bound::Unbounded, _) | (_, Bound::Unbounded) => true,
337            (Bound::Included(l), Bound::Included(u)) => l <= u,
338            (Bound::Included(l), Bound::Excluded(u)) => l < u,
339            (Bound::Excluded(l), Bound::Included(u)) => l < u,
340            (Bound::Excluded(l), Bound::Excluded(u)) => l < u,
341        }
342    }
343
344    /// Checks if the interval contains the given value.
345    ///
346    /// ```
347    /// # use devela::Interval;
348    /// let interval = Interval::closed(1, 5);
349    /// assert!(interval.contains(&3));
350    /// assert!(!interval.contains(&6));
351    /// ```
352    #[must_use]
353    pub fn contains(&self, value: &T) -> bool {
354        let lower_check = match &self.lower {
355            Bound::Included(lower) => *lower <= *value,
356            Bound::Excluded(lower) => *lower < *value,
357            Bound::Unbounded => true,
358        };
359        let upper_check = match &self.upper {
360            Bound::Included(upper) => *value <= *upper,
361            Bound::Excluded(upper) => *value < *upper,
362            Bound::Unbounded => true,
363        };
364        lower_check && upper_check
365    }
366
367    /// Returns the size of the interval, if finite.
368    #[must_use]
369    pub fn size(&self) -> Option<T>
370    where
371        T: Clone + core::ops::Sub<Output = T>,
372    {
373        match (&self.lower, &self.upper) {
374            (Bound::Included(l), Bound::Included(u)) => {
375                is![l <= u; Some(u.clone() - l.clone()); None]
376            }
377            (Bound::Included(l), Bound::Excluded(u)) => {
378                is![l < u; Some(u.clone() - l.clone()); None]
379            }
380            (Bound::Excluded(l), Bound::Included(u)) => {
381                is![l < u; Some(u.clone() - l.clone()); None]
382            }
383            (Bound::Excluded(l), Bound::Excluded(u)) => {
384                is![l < u; Some(u.clone() - l.clone()); None]
385            }
386            _ => None, // Unbounded intervals don't have a finite size
387        }
388    }
389}

532        impl<T: fmt::Display> fmt::Display for Interval<T> {
533            /// Formats the interval as a human-readable string.
534            ///
535            /// Examples:
536            /// - `(-∞, 5]` for an unbounded lower bound and inclusive upper bound.

539                let lower = match &self.lower {
540                    Bound::Included(value) => format!("[{value}"),
541                    Bound::Excluded(value) => format!("({value}"),
542                    Bound::Unbounded => String::from("(-∞"),
543                };
544                let upper = match &self.upper {
545                    Bound::Included(value) => format!("{lower}, {value}]"),
546                    Bound::Excluded(value) => format!("{lower}, {value})"),
547                    Bound::Unbounded => format!("{lower}, ∞)"),
548                };
549                write!(f, "{upper}")
550            }
551        }

554    /// Comparison Logic:
555    /// - We compare the lower bounds first.
556    /// - If the lower bounds are equal, we compare the upper bounds.
557    /// - We define Unbounded as less than any bounded value.
558    /// - We define that Included(l) < Excluded(l) at same point l.
559    impl<T: PartialOrd> PartialOrd for Interval<T> {
560        fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
561            match compare_bounds(&self.lower, &other.lower) {
562                Some(Ordering::Equal) => compare_bounds(&self.upper, &other.upper),
563                ord => ord,
564            }
565        }
566    }
567    /// Comparison Logic:
568    /// - We compare the lower bounds first.
569    /// - If the lower bounds are equal, we compare the upper bounds.
570    /// - We define Unbounded as less than any bounded value.
571    /// - We define that Included(l) < Excluded(l) at same point l.
572    impl<T: Ord> Ord for Interval<T> {
573        fn cmp(&self, other: &Self) -> Ordering {
574            match compare_bounds_ord(&self.lower, &other.lower) {
575                Ordering::Equal => compare_bounds_ord(&self.upper, &other.upper),
576                ord => ord,
577            }
578        }
579    }

581    /* helpers */

583    fn compare_bounds<T: PartialOrd>(l: &Bound<T>, u: &Bound<T>) -> Option<Ordering> {
584        use Bound::{Excluded, Included, Unbounded};
585        match (l, u) {
586            (Unbounded, Unbounded) => Some(Ordering::Equal),
587            (Unbounded, _) => Some(Ordering::Less),
588            (_, Unbounded) => Some(Ordering::Greater),
589            (Included(a_val), Included(b_val)) => a_val.partial_cmp(b_val),
590            (Excluded(a_val), Excluded(b_val)) => a_val.partial_cmp(b_val),
591            (Included(a_val), Excluded(b_val)) => {
592                match a_val.partial_cmp(b_val) {
593                    Some(Ordering::Equal) => Some(Ordering::Less),
594                    ord => ord,
595                }
596            }
597            (Excluded(a_val), Included(b_val)) => {
598                match a_val.partial_cmp(b_val) {
599                    Some(Ordering::Equal) => Some(Ordering::Greater),
600                    ord => ord,
601                }
602            }
603        }
604    }
605    fn compare_bounds_ord<T: Ord>(l: &Bound<T>, u: &Bound<T>) -> Ordering {
606        use Bound::{Excluded, Included, Unbounded};
607        match (l, u) {
608            (Unbounded, Unbounded) => Ordering::Equal,
609            (Unbounded, _) => Ordering::Less,
610            (_, Unbounded) => Ordering::Greater,
611            (Included(a_val), Included(b_val)) => a_val.cmp(b_val),
612            (Excluded(a_val), Excluded(b_val)) => a_val.cmp(b_val),
613            (Included(a_val), Excluded(b_val)) => {
614                match a_val.cmp(b_val) {
615                    Ordering::Equal => Ordering::Less,
616                    ord => ord,
617                }
618            }
619            (Excluded(a_val), Included(b_val)) => {
620                match a_val.cmp(b_val) {
621                    Ordering::Equal => Ordering::Greater,
622                    ord => ord,
623                }
624            }
625        }
626    }
