13/// Fundamental numeric constants for both integer and floating-point types.
14pub trait NumConst: PartialEq<Self::Num> {
15    /// The underlying numeric type implementing this trait.
16    type Num;
17
18    /* numeric type introspection */
19
20    /// Whether the number can represent big quantities.
21    const NUM_IS_BIG: bool;
22
23    /// Whether the number uses an integer representation.
24    const NUM_IS_INT: bool;
25
26    /// Whether the number uses a floating-point representation.
27    const NUM_IS_FLOAT: bool;
28
29    /// Whether the number uses a fixed-point representation.
30    const NUM_IS_FIXED: bool;
31
32    /// Whether the number includes the sign.
33    const NUM_IS_SIGNED: bool;
34
35    /// Whether the number has a memory niche optimization.
36    const NUM_IS_NICHE: bool;
37
38    /* numeric constant values */
39
40    /// The additive identity (`0`), if applicable.
41    const NUM_ZERO: Option<Self::Num>;
42
43    /// The multiplicative identity (`1`).
44    const NUM_ONE: Option<Self::Num>;
45
46    /// The only even prime and the fundamental doubling factor (`2`).
47    const NUM_TWO: Option<Self::Num>;
48
49    /// The smallest odd prime and the first nontrivial divisor (`3`).
50    const NUM_THREE: Option<Self::Num>;
51
52    /// The additive inverse of `ONE` (`-1`), if applicable.
53    const NUM_NEG_ONE: Option<Self::Num>;
54
55    /// The smallest representable value.
56    const NUM_MIN: Option<Self::Num>;
57
58    /// The greatest representable value.
59    const NUM_MAX: Option<Self::Num>;
60
61    /// The smallest representable positive value.
62    const NUM_MIN_POSITIVE: Option<Self::Num>;
63
64    /// The greatest representable negative value, if applicable.
65    const NUM_MAX_NEGATIVE: Option<Self::Num>;
66
67    /// The smallest normalized value (e.g. 0.0 for float, `MIN` for integers).
68    const NUM_MIN_NORM: Option<Self::Num>;
69
70    /// The greatest normalized value (e.g. 1.0 for float, `MAX` for integers).
71    const NUM_MAX_NORM: Option<Self::Num>;
72
73    /// The maximum representable power of two within the type's range.
74    ///
75    /// This is constructed using exact bit manipulation rather than arithmetic
76    /// operations to ensure consistent results across all platforms.
77    const NUM_MAX_POWER_OF_TWO: Option<Self::Num>;
78
79    /* auto-implemented convenience methods */
80
81    /// Whether `self` is equal to [`NUM_ZERO`][`Self::NUM_ZERO`].
82    fn is_num_zero(&self) -> bool {
83        Self::NUM_ZERO.is_some_and(|n| self == &n)
84    }
85
86    /// Whether `self` is equal to [`NUM_ONE`][`Self::NUM_ONE`].
87    fn is_num_one(&self) -> bool {
88        Self::NUM_ONE.is_some_and(|n| self == &n)
89    }
90
91    /// Whether `self` is equal to [`NUM_TWO`][`Self::NUM_TWO`].
92    fn is_num_two(&self) -> bool {
93        Self::NUM_TWO.is_some_and(|n| self == &n)
94    }
95
96    /// Whether `self` is equal to [`NUM_THREE`][`Self::NUM_THREE`].
97    fn is_num_three(&self) -> bool {
98        Self::NUM_THREE.is_some_and(|n| self == &n)
99    }
100
101    /// Whether `self` is equal to [`NUM_NEG_ONE`][`Self::NUM_NEG_ONE`].
102    fn is_num_neg_one(&self) -> bool {
103        Self::NUM_NEG_ONE.is_some_and(|n| self == &n)
104    }
105
106    /// Whether `self` is equal to [`NUM_MIN`][`Self::NUM_MIN`].
107    fn is_num_min(&self) -> bool {
108        Self::NUM_MIN.is_some_and(|n| self == &n)
109    }
110
111    /// Whether `self` is equal to [`NUM_MAX`][`Self::NUM_MAX`].
112    fn is_num_max(&self) -> bool {
113        Self::NUM_MAX.is_some_and(|n| self == &n)
114    }
115
116    /// Whether `self` is equal to `Some(`[`NUM_MIN_POSITIVE`][`Self::NUM_MIN_POSITIVE`]`)`.
117    fn is_num_min_positive(&self) -> bool {
118        Self::NUM_MIN_POSITIVE.is_some_and(|n| self == &n)
119    }
120
121    /// Whether `self` is equal to [`NUM_MAX_NEGATIVE`][`Self::NUM_MAX_NEGATIVE`].
122    fn is_num_max_negative(&self) -> bool {
123        Self::NUM_MAX_NEGATIVE.is_some_and(|n| self == &n)
124    }
125
126    /// Whether `self` is equal to [`NUM_MIN_NORM`][`Self::NUM_MIN_NORM`].
127    fn is_num_min_norm(&self) -> bool {
128        Self::NUM_MIN_NORM.is_some_and(|n| self == &n)
129    }
130
131    /// Whether `self` is equal to [`NUM_MAX_NORM`][`Self::NUM_MAX_NORM`].
132    fn is_num_max_norm(&self) -> bool {
133        Self::NUM_MAX_NORM.is_some_and(|n| self == &n)
134    }
135
136    /// Whether `self` is equal to [`NUM_MAX_POWER_OF_TWO`][`Self::NUM_MAX_POWER_OF_TWO`].
137    fn is_num_max_power_of_two(&self) -> bool {
138        Self::NUM_MAX_POWER_OF_TWO.is_some_and(|n| self == &n)
139    }
140}

142///
143macro_rules! impl_ext_num_const {
144    () => {
145        impl_ext_num_const![float: f32|u32, f64|u64];
146        impl_ext_num_const![int: i8|u8, i16|u16, i32|u32, i64|u64, i128|u128, isize|usize];
147        impl_ext_num_const![uint: u8|u8, u16|u16, u32|u32, u64|u64, u128|u128, usize|usize];
148        impl_ext_num_const![non0int: NonZeroI8|u8, NonZeroI16|u16, NonZeroI32|u32,
149            NonZeroI64|u64, NonZeroI128|u128, NonZeroIsize|usize];
150        impl_ext_num_const![non0uint: NonZeroU8|u8, NonZeroU16|u16, NonZeroU32|u32,
151            NonZeroU64|u64, NonZeroU128|u128, NonZeroUsize|usize];
152    };
153    ($T:ty | $U:ty: $ZERO:expr, $ONE:expr, $TWO:expr, $THREE:expr,
154     $NEG_ONE:expr, $MIN_POS:expr, $MAX_NEG:expr, $MAX_POW2:expr, $MIN_NORM:expr, $MAX_NORM:expr,
155     $IS_BIG:literal, $IS_INT:literal, $IS_FLOAT:literal, $IS_FIXED:literal,
156     $IS_SIGNED:literal, $IS_NICHE:literal) => {
157        impl_ext_num_const![@$T|$U: $ZERO, $ONE, $TWO, $THREE,
158        $NEG_ONE, $MIN_POS, $MAX_NEG, $MAX_POW2, $MIN_NORM, $MAX_NORM,
159        $IS_BIG, $IS_INT, $IS_FLOAT, $IS_FIXED, $IS_SIGNED, $IS_NICHE];
160    };
161    (@$T:ty | $U:ty: $ZERO:expr, $ONE:expr, $TWO:expr, $THREE:expr,
162     $NEG_ONE:expr, $MIN_POS:expr, $MAX_NEG:expr, $MAX_POW2:expr, $MIN_NORM:expr, $MAX_NORM:expr,
163     $IS_BIG:literal, $IS_INT:literal, $IS_FLOAT:literal, $IS_FIXED:literal,
164     $IS_SIGNED:literal, $IS_NICHE:literal) => {
165        impl NumConst for $T {
166            type Num = $T;
167            // introspection
168            const NUM_IS_BIG: bool = $IS_BIG;
169            const NUM_IS_INT: bool = $IS_INT;
170            const NUM_IS_FLOAT: bool = $IS_FLOAT;
171            const NUM_IS_FIXED: bool = $IS_FIXED;
172            const NUM_IS_SIGNED: bool = $IS_SIGNED;
173            const NUM_IS_NICHE: bool = $IS_NICHE;
174            // constant values
175            const NUM_ZERO: Option<$T> = $ZERO;
176            const NUM_ONE: Option<$T> = $ONE;
177            const NUM_TWO: Option<$T> = $TWO;
178            const NUM_THREE: Option<$T> = $THREE;
179            const NUM_NEG_ONE: Option<$T> = $NEG_ONE;
180            const NUM_MIN: Option<$T> = Some(<$T>::MIN); // FIXME: receive the some
181            const NUM_MAX: Option<$T> = Some(<$T>::MAX); // FIXME
182            const NUM_MIN_POSITIVE: Option<$T> = $MIN_POS;
183            const NUM_MAX_NEGATIVE: Option<$T> = $MAX_NEG;
184            const NUM_MIN_NORM: Option<$T> = $MIN_NORM;
185            const NUM_MAX_NORM: Option<$T> = $MAX_NORM;
186            const NUM_MAX_POWER_OF_TWO: Option<$T> = $MAX_POW2;
187        }
188    };
189
190    /* specific impls */
191
192    (float: $( $T:ty | $U:ty ),+) => { $(
193        impl_ext_num_const![$T|$U:
194            Some(0.0), Some(1.0), Some(2.0), Some(3.0),
195            Some(-1.0),               // NEG_ONE
196            Some(<$T>::MIN_POSITIVE), // MIN_POS
197            Some(-0.0),               // MAX_NEG // ↓ MAX_POW2
198            Some(<$T>::from_bits(((<$T>::EXPONENT_BIAS as $U << 1) << (<$T>::SIGNIFICAND_BITS)))),
199            Some(0.0), Some(1.0), // MIN_NORM, MAX_NORM
200            false, // IS_BIG
201            false, // IS_INT
202            true,  // IS_FLOAT
203            false, // IS_FIXED
204            true,  // IS_SIGNED
205            false  // IS_NICHE
206        ];
207    )+};
208    (int: $( $T:ty | $U:ty ),+) => { $(
209        impl_ext_num_const![$T|$U:
210            Some(0), Some(1), Some(2), Some(3),
211            Some(-1),   // NEG_ONE
212            Some(1),    // MIN_POS
213            Some(-1),   // MAX_NEG
214            Some(<$T>::MAX - (<$T>::MAX >> 1)), // MAX_POW2
215            Some(<$T>::MIN), // MIN_NORM
216            Some(<$T>::MAX), // MAX_NORM
217            false, // IS_BIG
218            true,  // IS_INT
219            false, // IS_FLOAT
220            false, // IS_FIXED
221            true,  // IS_SIGNED
222            false  // IS_NICHE
223        ];
224    )+};
225    (uint: $( $T:ty | $U:ty ),+) => { $(
226        impl_ext_num_const![$T|$U:
227            Some(0), Some(1), Some(2), Some(3),
228            None,    // NEG_ONE
229            Some(1), // MIN_POS
230            None,    // MAX_NEG
231            Some(<$T>::MAX ^ (<$T>::MAX >> 1)), // MAX_POW2
232            Some(<$T>::MIN), // MIN_NORM
233            Some(<$T>::MAX), // MAX_NORM
234            false, // IS_BIG
235            true,  // IS_INT
236            false, // IS_FLOAT
237            false, // IS_FIXED
238            false, // IS_SIGNED
239            false  // IS_NICHE
240          ];
241    )+};
242    (non0int: $( $T:ty | $U:ty ),+) => { $(
243        impl_ext_num_const![$T|$U:
244            None, <$T>::new(1), <$T>::new(2), <$T>::new(3),
245            <$T>::new(-1), // NEG_ONE
246            <$T>::new(1),  // MIN_POS
247            <$T>::new(-1), // MAX_NEG
248            <$T>::new(<$T>::MAX.get() - (<$T>::MAX.get() >> 1)), // MAX_POW2
249            Some(<$T>::MIN), // MIN_NORM
250            Some(<$T>::MAX), // MAX_NORM
251            false, // IS_BIG
252            true,  // IS_INT
253            false, // IS_FLOAT
254            false, // IS_FIXED
255            true,  // IS_SIGNED
256            true   // IS_NICHE
257        ];
258    )+};
259    (non0uint: $( $T:ty | $U:ty ),+) => { $(
260        impl_ext_num_const![$T|$U:
261            None, <$T>::new(1), <$T>::new(2), <$T>::new(3),
262            None,          // NEG_ONE
263            <$T>::new(1),  // MIN_POS
264            None,          // MAX_NEG
265            <$T>::new(<$T>::MAX.get() ^ (<$T>::MAX.get() >> 1)), // MAX_POW2
266            Some(<$T>::MIN), // MIN_NORM
267            Some(<$T>::MAX), // MAX_NORM
268            false, // IS_BIG
269            true,  // IS_INT
270            false, // IS_FLOAT
271            false, // IS_FIXED
272            false, // IS_SIGNED
273            true   // IS_NICHE
274        ];
275    )+};
276}

283    #[test]
284    fn float() {
285        assert_eq!(f32::NUM_ZERO, Some(0.0));
286        assert_eq!(f32::NUM_ONE, Some(1.0));
287        assert_eq!(f32::NUM_TWO, Some(2.0));
288        assert_eq!(f32::NUM_THREE, Some(3.0));
289        assert_eq!(f32::NUM_NEG_ONE, Some(-1.0));
290        assert_eq!(f32::NUM_MIN_POSITIVE, Some(f32::MIN_POSITIVE));
291        assert_eq!(f32::NUM_MAX_NEGATIVE, Some(-0.0));
292        assert_eq!(f32::NUM_MIN_NORM, Some(0.0));
293        assert_eq!(f32::NUM_MAX_NORM, Some(1.0));
294
295        // NOTE: The powi floating-point operation is non-deterministic.
296        // 1.70141183e38_f32, 8.98846567431158e307_f64, …
297        //
298        // #[cfg(all(target_arch = "x86_64", not(miri)))]
299        // {
300        //     assert_eq!(f32::NUM_MAX_POWER_OF_TWO, Some(2.0_f32.powi(f32::EXPONENT_BIAS as i32)));
301        //     assert_eq!(f64::NUM_MAX_POWER_OF_TWO, Some(2.0_f64.powi(f64::EXPONENT_BIAS as i32)));
302        // }
303        let expected_bits = ((f32::EXPONENT_BIAS as u32) << 1) << f32::SIGNIFICAND_BITS;
304        let actual_bits = f32::NUM_MAX_POWER_OF_TWO.unwrap().to_bits();
305        assert_eq!(actual_bits, expected_bits);
306        //
307        let expected_bits = ((f64::EXPONENT_BIAS as u64) << 1) << f64::SIGNIFICAND_BITS;
308        let actual_bits = f64::NUM_MAX_POWER_OF_TWO.unwrap().to_bits();
309        assert_eq!(actual_bits, expected_bits);
310    }
