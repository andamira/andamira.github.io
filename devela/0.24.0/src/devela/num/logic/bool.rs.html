12/// Allows to convert compile-time constants into type-level booleans.
13///
14/// See also the [`const_bool`] macro, and the [`True`] and [`False`] types.
15#[rustfmt::skip]
16#[diagnostic::on_unimplemented(
17    message = "Only expressions that evaluate to a constant 0 or 1 are valid for `ConstBool`.",
18    label = "This expression does not evaluate to a constant 0 or 1 (as usize)."
19)]
20pub trait ConstBool {
21    /// The resulting type-level boolean (`True` or `False`).
22    type Value: Sized;
23    /// The constant value of the type-level boolean.
24    const VALUE: Self::Value;
25}
26sf! {
27    impl ConstBool for False { type Value = False; const VALUE: Self::Value = False; }
28    impl ConstBool for [(); 0] { type Value = False; const VALUE: Self::Value = False; }
29    impl ConstBool for True { type Value = True; const VALUE: Self::Value = True; }
30    impl ConstBool for [(); 1] { type Value = True; const VALUE: Self::Value = True; }
31}

33/// Converts a *const* `bool` expression to a type-level boolean.
34///
35/// Internally, it leverages the [`ConstBool`] trait and a trick related to array sizes:
36/// - Arrays of size `[(); 0]` are mapped to [`False`].
37/// - Arrays of size `[(); 1]` are mapped to [`True`].
38///
39/// # Examples
40/// ```
41/// # use devela::{const_bool, True};
42/// const _: True = const_bool![4 == 4];
43/// ```
44/// ```compile_fail
45/// # use devela::{const_bool, True};
46/// const _: True = const_bool![3 == 4];
47/// ```
48#[macro_export]
49#[cfg_attr(cargo_primary_package, doc(hidden))]
50macro_rules! const_bool {
51    ($bool:expr) => {{ <[(); { $bool as usize }] as $crate::ConstBool>::VALUE }};
52}
53#[doc(inline)]
54pub use const_bool;
