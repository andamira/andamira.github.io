40            /// Returns a new bitmask of 1s from the `[start..=end]` range.
41            ///
42            /// Sets the rest of the bits to 0.
43            ///
44            /// # Panics
45            /// Panics if `start >= BITS || end >= BITS || start > end`.
46            #[doc = include_str!("../benches/mask_range.md")]
47            pub const fn mask_range(start: u32, end: u32) -> Self {
48                debug_assert![start <= end];
49                // a mask with all bits set, from 0 to end:
50                let mask_end = is![end == <$t>::BITS -1; !0; (1 << (end + 1)) - 1];
51                // a mask with all bits set from 0 to start - 1:
52                let mask_start = is![start == 0; 0; (1 << start) - 1];
53                Self(mask_end - mask_start)
54            }
55            /// Returns a new bitmask of ones from the `[start..=end]` checked range.
56            ///
57            /// Sets the rest of the bits to 0.
58            ///
59            /// # Errors
60            /// Returns [`IndexOutOfBounds`] if `start >= BITS || end >= BITS` and
61            /// [`MismatchedIndices`] if `start > end`.
62            #[doc = include_str!("../benches/mask_checked_range.md")]
63            pub const fn mask_checked_range(start: u32, end: u32)
64                -> Result<Self, MismatchedBounds> {
65                if start >= <$t>::BITS {
66                    Err(IndexOutOfBounds(Some(start as usize)))
67                } else if end >= <$t>::BITS {
68                    Err(IndexOutOfBounds(Some(end as usize)))
69                } else if start > end {
70                    Err(MismatchedIndices)
71                } else {
72                    // create a mask with all bits set, from 0 to end:
73                    let mask_end = is![end == <$t>::BITS -1; !0; (1 << (end + 1)) - 1];
74                    // create a mask with all bits set from 0 to start - 1:
75                    let mask_start = is![start == 0; 0; (1 << start) - 1];
76                    Ok(Self(mask_end - mask_start))
77                }
78            }

197            /// Leaves the rest of the bits unchanged.
198            /// # Errors
199            /// Returns [`IndexOutOfBounds`] if `start >= BITS || end >= BITS`,
200            /// [`MismatchedIndices`] if `start > end`, and
201            /// [`DataOverflow`] if `value` does not fit within the specified bit range.

208                        let value_shifted = (value << start) & mask.0;
209                        Ok(Self((self.0 & !mask.0) | value_shifted))
210                    },
211                    Err(e) => Err(e),
212                }

267            /// Reverses the order of the bits in `self` from the `[start..=end]` range.
268            ///
269            /// Leaves the rest of the bits unchanged.
270            /// # Panics
271            /// Panics if `start >= BITS || end >= BITS || start > end`.
272            pub const fn reverse_range(self, start: u32, end: u32) -> Self {
273                debug_assert![start <= end];
274                // If the entire range of bits is selected, simply reverse all bits
275                let range_bits = end - start + 1;
276                is![range_bits == Self::BITS; return Self(self.0.reverse_bits())];
277                // Create the mask for the range and reverse its bits
278                let mask = (((1 as $t) << range_bits) - 1) << start;
279                let bits_to_rev = (self.0 & mask) >> start;
280                let rev = bits_to_rev.reverse_bits();
281                // Shift the reversed bits back to their original position
282                let rev_shifted = (rev >> (Self::BITS - range_bits)) << start;
283                // Combine with the original number, preserving bits outside the range
284                Self((self.0 & !mask) | rev_shifted)
285            }
