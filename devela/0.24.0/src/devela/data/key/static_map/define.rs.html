9/// Build a custom static hashmap.
10///
11/// # Arguments
12/// - `$NAME`:      the name of the new hashmap.
13/// - `$KEY`:       the integer primitive type for the keys.
14///
15/// optional:
16/// - `$EMPTY`:     the `$KEY` value for empty entries.
17/// - `$TOMB`:      the `$KEY` value for deleted entries.
18/// - `$HASH_ARG`:  the argument representing the byte slice.
19/// - `$HASH_EXPR`: the const hasher expression using `$HASH_ARG`.
20///
21/// # Notes
22/// - values `V` have to be `Copy` + `ConstDefault`.
23/// - keys `$KEY` can be any primitive integers, floats or `char`.
24/// - Two specific `$KEY` values are reserved to indicate empty deleted keys.
25///   They default to `MIN` and `MAX`, respectively, but can be customized.
26/// - The default hasher is [`HasherFx`][crate::HasherFx].
27///
28/// # Examples
29///
30/// See the example type: [`ExampleStaticMapU16`].
31///
32/// Basic usage
33/// ```
34/// # use devela::define_static_map;
35/// // Define a static hashmap with `u16` keys and default hasher
36/// define_static_map![const ExampleMap, KEY: u16];
37///
38/// let mut map = ExampleMap::<u16, u32, 8>::new();
39///
40/// // Insert key-value pairs
41/// map.insert(1, 100).unwrap();
42/// map.insert(2, 200).unwrap();
43///
44/// // Retrieve values
45/// assert_eq!(map.get(1), Some(100));
46/// assert_eq!(map.get(2), Some(200));
47/// assert_eq!(map.get(3), None); // Key not found
48///
49/// // Delete a key
50/// assert!(map.remove(1));
51/// assert_eq!(map.get(1), None);
52///
53/// // Check introspection methods
54/// assert_eq!(map.len(), 1);
55/// assert!(!map.is_empty());
56/// assert!(!map.is_full());
57///
58/// // Rebuild after deletions to optimize probing
59/// if map.should_rebuild() {
60///     map.rebuild();
61/// }
62/// ```
63///
64/// Custom hashers
65/// ```
66/// # use devela::{define_static_map, HasherFx};
67/// // Define a static hashmap using `HasherFx` with a custom seed
68/// define_static_map![const ExampleMapFxSeeded, KEY: u16,
69///     HASHER: |b| HasherFx::<usize>::hash_bytes_with_seed(123, b)
70/// ];
71/// let mut map = ExampleMapFxSeeded::<u16, u32, 8>::new();
72/// map.insert(1, 100).unwrap();
73/// assert_eq!(map.get(1), Some(100));
74///
75/// # #[cfg(feature = "hash")] {
76/// # use devela::HasherPengy;
77/// // Define a static hashmap using a stateful pengy hasher
78/// # #[cfg(feature = "hash")]
79/// define_static_map![const ExampleMapPengy, KEY: u16,
80///     HASHER: |b| {
81///         let mut p = HasherPengy::new();
82///         p.process(b);
83///         p.digest() as usize
84///     }
85/// ];
86/// let mut map = ExampleMapPengy::<u16, u32, 8>::new();
87/// map.insert(1, 100).unwrap();
88/// assert_eq!(map.get(1), Some(100));
89/// # }
90/// ```

185            /// Retrieves some shared reference to the value associated with the given key.
186            pub const fn get_ref(&self, key: $KEY) -> Option<&V> {
187                Self::debug_assert_valid_key(key);
188                let mut index = self.hash_index(key);
189                let mut i = 0;
190                while i < N {
191                    if self.keys[index] == key { return Some(&self.values[index]); }

202                let mut index = self.hash_index(key);
203                let mut i = 0;
204                while i < N {

213            /// Retrieves an entry for a given key.
214            pub const fn entry(&mut self, key: $KEY) -> $crate::StaticMapEntry<'_, V> {
215                Self::debug_assert_valid_key(key);
216                let mut index = self.hash_index(key);
217                let mut i = 0;
218                let mut tombstone_index = None;
219                while i < N {
220                    if self.keys[index] == self.empty() {
221                        return $crate::StaticMapEntry::Vacant(index);

224                        return $crate::StaticMapEntry::Occupied(&mut self.values[index]);

228                    }

230                    i += 1;
231                }

233                $crate::StaticMapEntry::Vacant($crate::unwrap![some_or tombstone_index, N])
234            }

302        #[allow(unused)]
303        impl<V: Copy, const N: usize> $NAME<$KEY, V, N> {
304            /// Inserts a key-value pair.
305            ///
306            /// # Returns
307            /// - `Ok(())` if the insertion succeeds.
308            /// - `Err(`[`NotEnoughSpace`][crate::NotEnoughSpace]`)` if no slots are available.
309            ///
310            /// # Behavior
311            /// - Computes the **hash index** of the key.
312            /// - If the slot is `EMPTY`, inserts immediately.
313            /// - If the slot contains `TOMB`, the first `TOMB` encountered is
314            ///   **used if no empty slots exist earlier in probing**.
315            /// - If the slot contains another key, **probes forward** until an open slot is found.
316            /// - If no open slots exist, returns an error.
317            #[allow(clippy::float_cmp, clippy::float_cmp_const)]
318            pub const fn insert(&mut self, key: $KEY, value: V)
319                -> Result<(), $crate::NotEnoughSpace> {
320                Self::debug_assert_valid_key(key);
321                let mut index = self.hash_index(key);
322                let mut i = 0;
323                let mut tombstone_index = None;
324                while i < N {
325                    if self.keys[index] == self.empty() || self.keys[index] == self.tomb() {
326                        let slot = if let Some(tomb) = tombstone_index { tomb } else { index };
327                        self.keys[slot] = key;
328                        self.values[slot] = value;
329                        return Ok(());
330                    }
331                    if self.keys[index] == key {
332                        self.values[index] = value; // Overwrite existing value
333                        return Ok(());
334                    }
335                    if self.keys[index] == self.tomb() && tombstone_index.is_none() {
336                        tombstone_index = Some(index);
337                    }
338                    index = (index + 1) % N;
339                    i += 1;
340                }
341                Err($crate::NotEnoughSpace(Some(1)))
342            }
343
344            /// Retrieves a value by key.
345            ///
346            /// # Returns
347            /// - `Some(value)` if the key exists.
348            /// - `None` if the key is missing.
349            ///
350            /// # Behavior
351            /// - Searches for the key using **linear probing**.
352            /// - If a `TOMB` (deleted slot) is encountered, it **continues probing**.
353            /// - If an `EMPTY` slot is reached, the key is **not in the table**.
354            #[allow(clippy::float_cmp, clippy::float_cmp_const)]
355            pub const fn get(&self, key: $KEY) -> Option<V> {
356                Self::debug_assert_valid_key(key);
357                let mut index = self.hash_index(key);
358                let mut i = 0;
359                while i < N {
360                    if self.keys[index] == key { return Some(self.values[index]); }
361                    if self.keys[index] == self.empty() { return None; } // end of probe chain
362                    index = (index + 1) % N;
363                    i += 1;
364                }
365                None
366            }
367        }

369        #[allow(unused)]
370        impl<V: Copy + $crate::ConstDefault, const N: usize> $NAME<$KEY, V, N> {
371            /// Removes a key-value pair.
372            ///
373            /// # Returns
374            /// - `true` if the key was found and removed.
375            /// - `false` if the key was not found in the map.
376            ///
377            /// # Behavior
378            /// - Marks the slot as deleted (`TOMB`).
379            /// - Future lookups will continue probing past deleted entries.
380            /// - **Does NOT free the slot for immediate reuse**.
381            /// - New insertions only reuse a `TOMB` slot if no earlier `EMPTY` slots exist.
382            #[allow(clippy::float_cmp, clippy::float_cmp_const)]
383            pub const fn remove(&mut self, key: $KEY) -> bool {
384                Self::debug_assert_valid_key(key);
385                let mut index = self.hash_index(key);
386                let mut i = 0;
387                while i < N {
388                    if self.keys[index] == key { self.keys[index] = self.tomb(); return true; }
389                    if self.keys[index] == self.empty() { return false; }
390                    index = (index + 1) % N;
391                    i += 1;
392                }
393                false
394            }
395
396            /// Removes a key-value pair and optionally rebuilds the table.
397            ///
398            /// # Behavior
399            /// - Calls `remove()`, returning `true` if the key was found.
400            /// - If `should_rebuild()` returns `true`, calls `rebuild()`.
401            pub const fn remove_rebuild(&mut self, key: $KEY) -> bool {
402                let removed = self.remove(key);
403                if removed && self.should_rebuild() { self.rebuild(); }
404                removed
405            }
406
407            /// Rebuilds the table by removing `TOMB` slots and optimizing key placement.
408            ///
409            /// Calls [`Self::rebuilt()`] and replaces `self` with the optimized table.
410            ///
411            /// # When to Call?
412            /// - When **many deletions have occurred**.
413            /// - If lookups start taking significantly longer.
414            pub const fn rebuild(&mut self) { *self = self.rebuilt(); }
415
416            /// Returns a rebuilt version of the table with `TOMB` slots removed.
417            ///
418            /// Creates a new table and reinserts all valid keys while preserving the probe order.
419            ///
420            /// # Complexity
421            /// - **O(N)** worst-case when all slots are occupied.
422            pub const fn rebuilt(&self) -> Self {
423                let mut new_table = Self::new();
424                let mut i = 0;
425                while i < N {
426                    if self.keys[i] != self.empty() && self.keys[i] != self.tomb() {
427                        let _ = new_table.insert(self.keys[i], self.values[i]);
428                    }
429                    i += 1;
430                }
431                new_table
432            }
433        }

610                let mut count = 0;
611                let mut i = 0;
612                while i < N {
613                    if self.keys[i] == self.tomb() { count += 1; }
614                    i += 1;
615                }
616                count
617            }

642            /// Inserts a key-value pair.
643            ///
644            /// # Returns
645            /// - `Ok(())` if the insertion succeeds.
646            /// - `Err(`[`NotEnoughSpace`][crate::NotEnoughSpace]`)` if no slots are available.
647            ///
648            /// # Behavior
649            /// - Computes the **hash index** of the key.
650            /// - If the slot is `EMPTY`, inserts immediately.
651            /// - If the slot contains `TOMB`, the first `TOMB` encountered is
652            ///   **used if no empty slots exist earlier in probing**.
653            /// - If the slot contains another key, **probes forward** until an open slot is found.
654            /// - If no open slots exist, returns an error.
655            #[allow(clippy::float_cmp, clippy::float_cmp_const)]
656            pub fn insert(&mut self, key: $KEY, value: V)
657                -> Result<(), $crate::NotEnoughSpace> {
658                Self::debug_assert_valid_key(key);
659                let mut index = self.hash_index(key);
660                let mut i = 0;
661                let mut tombstone_index = None;
662                while i < N {
663                    if self.keys[index] == self.empty() || self.keys[index] == self.tomb() {
664                        let slot = if let Some(tomb) = tombstone_index { tomb } else { index };
665                        self.keys[slot] = key;
666                        self.values[slot] = value;
667                        return Ok(());
668                    }
669                    if self.keys[index] == key {
670                        self.values[index] = value; // Overwrite existing value
671                        return Ok(());
672                    }
673                    if self.keys[index] == self.tomb() && tombstone_index.is_none() {
674                        tombstone_index = Some(index);
675                    }
676                    index = (index + 1) % N;
677                    i += 1;
678                }
679                Err($crate::NotEnoughSpace(Some(1)))
680            }
681        }

683        #[allow(unused)]
684        impl<V: Copy, const N: usize> $NAME<$KEY, V, N> {
685            /// Retrieves a value by key.
686            ///
687            /// # Returns
688            /// - `Some(value)` if the key exists.
689            /// - `None` if the key is missing.
690            ///
691            /// # Behavior
692            /// - Searches for the key using **linear probing**.
693            /// - If a `TOMB` (deleted slot) is encountered, it **continues probing**.
694            /// - If an `EMPTY` slot is reached, the key is **not in the table**.
695            #[allow(clippy::float_cmp, clippy::float_cmp_const)]
696            pub fn get(&self, key: $KEY) -> Option<V> {
697                Self::debug_assert_valid_key(key);
698                let mut index = self.hash_index(key);
699                let mut i = 0;
700                while i < N {
701                    if self.keys[index] == key { return Some(self.values[index]); }
702                    if self.keys[index] == self.empty() { return None; } // end of probe chain
703                    index = (index + 1) % N;
704                    i += 1;
705                }
706                None
707            }
708
709            /// Removes a key-value pair.
710            ///
711            /// # Returns
712            /// - `true` if the key was found and removed.
713            /// - `false` if the key was not found in the map.
714            ///
715            /// # Behavior
716            /// - Marks the slot as deleted (`TOMB`).
717            /// - Future lookups will continue probing past deleted entries.
718            /// - **Does NOT free the slot for immediate reuse**.
719            /// - New insertions only reuse a `TOMB` slot if no earlier `EMPTY` slots exist.
720            #[allow(clippy::float_cmp, clippy::float_cmp_const)]
721            pub fn remove(&mut self, key: $KEY) -> bool {
722                Self::debug_assert_valid_key(key);
723                let mut index = self.hash_index(key);
724                let mut i = 0;
725                while i < N {
726                    if self.keys[index] == key { self.keys[index] = self.tomb(); return true; }
727                    if self.keys[index] == self.empty() { return false; }
728                    index = (index + 1) % N;
729                    i += 1;
730                }
731                false
732            }
733        }

736        impl<V: Copy + Default, const N: usize> $NAME<$KEY, V, N> {
737            /// Removes a key-value pair and optionally rebuilds the table.
738            ///
739            /// # Behavior
740            /// - Calls `remove()`, returning `true` if the key was found.
741            /// - If `should_rebuild()` returns `true`, calls `rebuild()`.
742            pub fn remove_rebuild(&mut self, key: $KEY) -> bool {
743                let removed = self.remove(key);
744                if removed && self.should_rebuild() { self.rebuild(); }
745                removed
746            }

748            /// Rebuilds the table by removing `TOMB` slots and optimizing key placement.
749            ///
750            /// Calls [`Self::rebuilt()`] and replaces `self` with the optimized table.
751            ///
752            /// # When to Call?
753            /// - When **many deletions have occurred**.
754            /// - If lookups start taking significantly longer.
755            pub fn rebuild(&mut self) { *self = self.rebuilt(); }
756
757            /// Returns a rebuilt version of the table with `TOMB` slots removed.
758            ///
759            /// Creates a new table and reinserts all valid keys while preserving the probe order.
760            ///
761            /// # Complexity
762            /// - **O(N)** worst-case when all slots are occupied.
763            pub fn rebuilt(&self) -> Self {
764                let mut new_table = Self::new();
765                let mut i = 0;
766                while i < N {
767                    if self.keys[i] != self.empty() && self.keys[i] != self.tomb() {
768                        let _ = new_table.insert(self.keys[i], self.values[i]);
769                    }
770                    i += 1;
771                }
772                new_table
773            }
774        }
