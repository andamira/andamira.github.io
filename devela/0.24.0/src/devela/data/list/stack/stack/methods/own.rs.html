1// devela::data::list::stack::stack::methods_own
2//
3//! owning stack const operations, in compile-time
4//! depends on T: Copy, S: Bare
5//
6// Algorithms are implemented in the unchecked versions by:
7// - creating a copy of the inner array.
8// - performing specific operations in it.
9// - creating the new stack.
10// - adjusting its length if necessary.
11// - returning it.
12
13use crate::{
14    Bare, DataNotEnough, MismatchedCapacity, NotEnoughElements, NotEnoughSpace, Own, Stack, cswap,
15    sf,
16};

18// helper macro to impl methods for a Stack with custom index size.
19//
20// $IDX : the index type. E.g. u8, usize
21macro_rules! impl_stack {
22    () => {
23        impl_stack![own
24            u8:"_stack_u8", u16:"_stack_u16", u32:"_stack_u32", usize:"_stack_usize"];
25    };
26
27    // general own_ ops
28    //
29    // $IDX : the index type. E.g. u8, usize
30    // $cap:  the capability feature that enables the given implementation. E.g "_stack_u8".
31    (own $( $IDX:ty: $cap:literal ),+) => {
32        $(
33            #[cfg(feature = $cap )]
34            impl_stack![@own $IDX:$cap];
35        )+
36    };
37    (@own $IDX:ty : $cap:literal) => { $crate::paste! {
38        // T: Copy, S: Bare
39        /// # Chainable *const* operations depending on `T: Copy`
40        ///
41        /// Every method is *const* and returns [`Own`][crate::Own]`<Self, V>`.
42        impl<T: Copy, const CAP: usize> Stack<T, CAP, $IDX, Bare> {
43            /* constructors */
44
45            /// Returns an empty stack, allocated in the stack,
46            /// copying `element` to fill the remaining free data.
47            ///
48            /// # Errors
49            #[doc = "Returns [`MismatchedCapacity`] if `CAP > `[`" $IDX "::MAX`]"]
50            /// or if `CAP > isize::MAX / size_of::<T>()`.
51            ///
52            /// # Examples
53            /// ```
54            #[doc = "# use devela::{Stack" $IDX:camel "};"]
55            #[doc = "const S: Stack" $IDX:camel "<i32, 16> = Stack" $IDX:camel
56                "::own_new(0).s_const_unwrap().s;"]
57            /// ```
58            pub const fn own_new(element: T) -> Own<Result<Self, MismatchedCapacity>, ()> {
59                Own::empty(Self::new_copied(element))
60            }

64            /// Clears the stack in compile-time.
65            ///
66            /// `( 1 2 3 -- )`
67            /// # Examples

73            /// ```
74            pub const fn own_clear(self) -> Own<Self, ()> {
75                let mut sta = self;
76                sta.len = 0;
77                Own::empty(sta)
78            }

82            /// Pushes a new `element` to the top of the stack in compile-time.
83            ///
84            /// `( 1 -- 1 2 )`
85            /// # Errors
86            /// Returns `Own<S,`[`NotEnoughSpace`]`>` if the stack is full.
87            /// # Examples
88            /// ```
89            #[doc = "# use devela::{Own, Stack" $IDX:camel "};"]
90            #[doc = "const S: Stack" $IDX:camel "<i32, 2> = Stack" $IDX:camel
91                "::own_new(0).s_const_unwrap().s"]
92            ///     .own_push(1).v_assert_ok().s
93            ///     .own_push(2).v_assert_ok().s;
94            /// assert_eq![S.as_slice(), &[1, 2]];
95            /// assert![S.own_push(3).v.is_err_and(|e| matches![e, NotEnoughSpace])];
96            /// ```
97            pub const fn own_push(self, element: T) -> Own<Self, Result<(), NotEnoughSpace>> {
98                if self.len as usize == CAP {
99                    Own::new(self, Err(NotEnoughSpace(Some(1))))
100                } else {
101                    self.own_push_unchecked(element).v_const_wrap_ok()
102                }
103            }

105            /// Pushes a new `element` to the top of the stack in compile-time, unchecked version.
106            ///
107            /// `( 1 -- 1 2 )`
108            /// # Panics
109            /// Panics if the stack is full.
110            /// # Examples
111            /// ```
112            #[doc = "# use devela::Stack" $IDX:camel ";"]
113            #[doc = "const S: Stack" $IDX:camel "<i32, 2> = Stack" $IDX:camel
114                "::own_new(0).s_const_unwrap().s"]
115            ///     .own_push_unchecked(1).s.own_push_unchecked(2).s;
116            /// assert_eq![S.as_slice(), &[1, 2]];
117            /// ```
118            pub const fn own_push_unchecked(self, element: T) -> Own<Self, ()> {
119                let mut arr = self.data.into_array_copy();
120                arr[self.len as usize] = element;
121                let mut sta = Self::from_array_copy(arr);
122                sta.len = self.len + 1;
123                Own::empty(sta)
124            }

509            /// Rotates the top three stack elements, clockwise, unchecked version.
510            ///
511            /// `( 1 2 3 -- 2 3 1 ) `
512            /// # Panics
513            /// Panics if the stack doesn’t contain at least 6 elements.
514            /// if the stack doesn't contain at least 3 elements.
515            /// # Examples
516            /// ```
517            #[doc = "# use devela::{Own, Stack" $IDX:camel "};"]
518            #[doc = "const S: Stack" $IDX:camel "<i32, 4> = Stack" $IDX:camel
519                "::from_array_copy([0, 1, 2, 3])"]
520            ///     .own_rot_unchecked().s;
521            /// assert_eq![S.as_slice(), &[0, 2, 3, 1]];
522            /// ```
523            // WAIT: [const_swap](https://github.com/rust-lang/rust/pull/134757)
524            pub const fn own_rot_unchecked(self) -> Own<Self, ()> {
525                let len = self.len as usize;
526                let mut arr = self.data.into_array_copy();
527                let [a, b, c] = [arr[len - 3], arr[len - 2], arr[len - 1]];
528                arr[len - 3] = b;
529                arr[len - 2] = c;
530                arr[len - 1] = a;
531                let sta = Self::from_array_copy(arr);
532                Own::empty(sta)
