35        /// # Stack resize.
36        // T, S: Bare
37        #[cfg_attr(nightly_doc, doc(cfg(feature = $cap)))]
38        impl<T: Default, const CAP: usize> Stack<T, CAP, $IDX, Bare> {
39            /// Converts the current stack to a different capacity
40            /// while preserving all existing elements.
41            ///
42            /// This method creates a new stack with the specified new capacity and moves the
43            /// current elements into it. The operation ensures that the new stack can accommodate
44            /// the number of elements currently held in the stack. It is designed to work with
45            /// both increases and decreases in capacity, as long as the new capacity can fit the
46            /// current number of elements.
47            ///
48            /// # Errors
49            /// Returns [`IndexOutOfBounds`]`(Some(NEW_CAP))` if `NEW_CAP < self.len()`,
50            #[doc = "if `CAP > `[`" $IDX "::MAX`]"]
51            /// or if `CAP > isize::MAX / size_of::<T>()`.
52            ///
53            /// # Examples
54            /// ```
55            #[doc = "# use devela::Stack" $IDX:camel ";"]
56            #[doc = "let s = Stack" $IDX:camel "::<_, 8>::from([1, 2, 3, 4]);"]
57            #[doc = "let less_cap: Stack" $IDX:camel "::<_, 4> = s.resize_default().unwrap();"]
58            /// assert_eq![s.as_slice(), less_cap.as_slice()];
59            #[doc = "let more_cap: Stack" $IDX:camel "::<_, 12> = s.resize_default().unwrap();"]
60            /// assert_eq![s.as_slice(), more_cap.as_slice()];
61            /// assert![s.resize_default::<2>().is_err()]; // too small
62            /// ```
63            pub fn resize_default<const NEW_CAP: usize>(self)
64                -> Result<Stack<T, NEW_CAP, $IDX, Bare>, IndexOutOfBounds> {
65                if NEW_CAP < (self.len() as usize) ||
66                    NEW_CAP > $IDX::MAX as usize ||
67                    NEW_CAP > isize::MAX as usize / size_of::<T>() {
68                    Err(IndexOutOfBounds(Some(NEW_CAP)))
69                } else {
70                    let old_arr: [T; CAP] = self.data.into_array();
71                    let mut new_arr =
72                        array_init![default [T; NEW_CAP], "safe_data", "unsafe_array"];
73                    for (i, item) in old_arr.into_iter().enumerate().take(NEW_CAP) {
74                        new_arr[i] = item;
75                    }
76                    Ok(Stack {
77                        data: Array::new(new_arr),
78                        len: self.len,
79                    })
80                }
81            }
82            /// Converts the current stack to a different capacity, dropping elements if needed.
83            ///
84            /// This method creates a new stack with the specified new capacity and moves the
85            /// current elements into it. The operation will drop any elements that can't fit
86            /// in the new capacity, starting with the first ones (from the front of the stack).
87            ///
88            /// # Examples
89            /// ```
90            #[doc = "# use devela::Stack" $IDX:camel ";"]
91            #[doc = "let s = Stack" $IDX:camel "::<_, 8>::from([1, 2, 3, 4]);"]
92            #[doc = "let less_cap: Stack" $IDX:camel "::<_, 4> = s.resize_default_truncate();"]
93            /// assert_eq![less_cap.as_slice(), s.as_slice()];
94            #[doc = "let more_cap: Stack" $IDX:camel "::<_, 12> = s.resize_default_truncate();"]
95            /// assert_eq![more_cap.as_slice(), s.as_slice()];
96            #[doc = "let drop_cap: Stack" $IDX:camel "::<_, 2> = s.resize_default_truncate();"]
97            /// assert_eq![drop_cap.as_slice(), &[3, 4]];
98            /// ```
99            pub fn resize_default_truncate<const NEW_CAP: usize>(self)
100                -> Stack<T, NEW_CAP, $IDX, Bare> {
101                let start_idx = if self.len() as usize > NEW_CAP {
102                    self.len() as usize - NEW_CAP
103                } else {
104                    0
105                };
106                let new_len = core::cmp::min(self.len(), NEW_CAP as $IDX);
107                let old_arr: [T; CAP] = self.data.into_array();
108                let mut new_arr = array_init![default [T; NEW_CAP], "safe_data", "unsafe_array"];
109                for (i, item) in old_arr.into_iter().enumerate().skip(start_idx).take(NEW_CAP) {
110                    new_arr[i - start_idx] = item;
111                }
112                Stack {
113                    data: Array::new(new_arr),
114                    len: new_len,
115                }
116            }
117        }

120        #[cfg(feature = "alloc")]
121        #[cfg_attr(nightly_doc, doc(cfg(feature = "alloc")))]
122        #[cfg_attr(nightly_doc, doc(cfg(feature = $cap)))]
123        impl<T: Default, const CAP: usize> Stack<T, CAP, $IDX, Boxed> {
124            /// Converts the current stack to a different capacity while preserving all existing
125            /// elements.
126            ///
127            /// This method creates a new stack with the specified new capacity and moves the
128            /// current elements into it. The operation ensures that the new stack can accommodate
129            /// the number of elements currently held in the stack. It is designed to work with
130            /// both increases and decreases in capacity, as long as the new capacity can fit the
131            /// current number of elements.
132            ///
133            /// # Errors
134            /// Returns [`IndexOutOfBounds`]`(Some(NEW_CAP))` if `NEW_CAP < self.len()`,
135            #[doc = "if `CAP > `[`" $IDX "::MAX`]"]
136            /// or if `CAP > isize::MAX / size_of::<T>()`.
137            ///
138            /// # Examples
139            /// ```
140            #[doc = "# use devela::{Boxed, Stack" $IDX:camel "};"]
141            #[doc = "let s = Stack" $IDX:camel "::<_, 8, Boxed>::from([1, 2, 3, 4]);"]
142            #[doc = "let less_cap: Stack" $IDX:camel
143                "::<_, 4, Boxed> = s.clone().resize_default().unwrap();"]
144            /// assert_eq![s.as_slice(), less_cap.as_slice()];
145            #[doc = "let more_cap: Stack" $IDX:camel
146                "::<_, 12, Boxed> = s.clone().resize_default().unwrap();"]
147            /// assert_eq![s.as_slice(), more_cap.as_slice()];
148            /// assert![s.resize_default::<2>().is_err()]; // too small
149            /// ```
150            pub fn resize_default<const NEW_CAP: usize>(self)
151            -> Result<Stack<T, NEW_CAP, $IDX, Boxed>, IndexOutOfBounds> {
152                if NEW_CAP < (self.len() as usize) ||
153                    NEW_CAP > $IDX::MAX as usize ||
154                    NEW_CAP > isize::MAX as usize / size_of::<T>() {
155                    Err(IndexOutOfBounds(Some(NEW_CAP)))
156                } else {
157                    let old_arr = self.data.into_vec();
158                    let mut v = Vec::with_capacity(NEW_CAP);
159                    for item in old_arr.into_iter().take(self.len as usize) {
160                        v.push(item);
161                    }
162                    v.resize_with(NEW_CAP, Default::default);
163
164                    let new_arr: Box<[T; NEW_CAP]> =
165                        v.into_boxed_slice().try_into().unwrap_or_else(|_| {
166                        panic!("Failed to convert Box<[T]> to Box<[T; NEW_CAP={}]>", NEW_CAP)
167                    });
168                    Ok(Stack {
169                        data: Array::new_boxed(new_arr),
170                        len: self.len,
171                    })
172                }
173            }
174            /// Converts the current stack to a different capacity
175            /// while preserving all existing elements.
176            ///
177            /// This method creates a new stack with the specified new capacity and moves the
178            /// current elements into it. The operation will drop any elements that can't fit
179            /// in the new capacity, starting with the first ones (from the front of the stack).
180            ///
181            /// # Examples
182            /// ```
183            #[doc = "# use devela::{Boxed, Stack" $IDX:camel "};"]
184            #[doc = "let s = Stack" $IDX:camel "::<_, 8, Boxed>::from([1, 2, 3, 4]);"]
185            #[doc = "let less_cap: Stack" $IDX:camel
186                "::<_, 4, Boxed> = s.clone().resize_default_truncate();"]
187            /// assert_eq![less_cap.as_slice(), s.as_slice()];
188            #[doc = "let more_cap: Stack" $IDX:camel
189                    "::<_, 12, Boxed> = s.clone().resize_default_truncate();"]
190            /// assert_eq![more_cap.as_slice(), s.as_slice()];
191            #[doc = "let drop_cap: Stack" $IDX:camel
192                    "::<_, 2, Boxed> = s.resize_default_truncate();"]
193            /// assert_eq![drop_cap.as_slice(), &[3, 4]];
194            /// ```
195            pub fn resize_default_truncate<const NEW_CAP: usize>(self)
196            -> Stack<T, NEW_CAP, $IDX, Boxed> {
197                let mut old_vec = self.data.into_vec();
198
199                // When reducing capacity, truncate elements from the front.
200                if NEW_CAP < self.len as usize {
201                    let excess = self.len as usize - NEW_CAP;
202                    old_vec.drain(0..excess);
203                }
204                // Ensure the vector's length matches NEW_CAP before conversion.
205                old_vec.resize_with(NEW_CAP, Default::default);
206
207                let new_arr: Box<[T; NEW_CAP]> =
208                    old_vec.into_boxed_slice().try_into().unwrap_or_else(|_| {
209                    panic!("Failed to convert Box<[T]> to Box<[T; NEW_CAP={}]>", NEW_CAP)
210                });
211                Stack {
212                    data: Array::new_boxed(new_arr),
213                    len: core::cmp::min(self.len as usize, NEW_CAP) as $IDX,
214                }
215            }
216        }

222            /// while preserving all existing elements.
223            ///
224            /// This method creates a new stack with the specified new capacity and moves the
225            /// current elements into it. The operation ensures that the new stack can accommodate
226            /// the number of elements currently held in the stack. It is designed to work with
227            /// both increases and decreases in capacity, as long as the new capacity can fit the
228            /// current number of elements.
229            ///
230            /// # Errors
231            /// Returns [`IndexOutOfBounds`]`(Some(NEW_CAP))` if `NEW_CAP < self.len()`,
232            #[doc = "if `CAP > `[`" $IDX "::MAX`]"]
233            /// or if `CAP > isize::MAX / size_of::<T>()`.
234            ///
235            /// # Examples
236            /// ```
237            #[doc = "# use devela::Stack" $IDX:camel ";"]
238            #[doc = "const S: Stack" $IDX:camel "<i32, 8> = Stack" $IDX:camel "::own_new(0)"]
239            ///     .s_const_unwrap().s.own_push(1).s.own_push(2).s.own_push(3).s;
240            #[doc = "const T: Stack" $IDX:camel
241                "<i32, 4> = S.own_resize_default().s_const_unwrap().s;"]
242            /// assert_eq![S.as_slice(), T.as_slice()];
243            /// let _ = S.own_resize_default::<2>().s_assert_err(); // too small
244            /// ```
245            pub const fn own_resize_default<const NEW_CAP: usize>(self)
246                -> Own<Result<Stack<T, NEW_CAP, $IDX, Bare>, IndexOutOfBounds>, ()> {
247                if NEW_CAP < (self.len as usize) ||
248                    NEW_CAP > $IDX::MAX as usize ||
249                    NEW_CAP > isize::MAX as usize / size_of::<T>() {
250                    Own::empty(Err(IndexOutOfBounds(Some(NEW_CAP))))
251                } else {
252                    let old_arr: [T; CAP] = self.data.into_array_copy();
253                    let mut new_arr = array_init![const_default [T; NEW_CAP]];
254
255                    let mut i = 0;
256                    while i < self.len as usize {
257                        new_arr[i] = old_arr[i];
258                        i += 1;
259                    }
260
261                    Own::empty(Ok(Stack {
262                        data: Array::new_bare(new_arr),
263                        len: self.len,
264                    }))
265                }
266            }
267            /// Converts the current stack to a different capacity, dropping elements if needed.
268            ///
269            /// This method creates a new stack with the specified new capacity and moves the
270            /// current elements into it. The operation will drop any elements that can't fit
271            /// in the new capacity, starting with the first ones (from the front of the stack).
272            ///
273            /// # Examples
274            /// ```
275            #[doc = "# use devela::Stack" $IDX:camel ";"]
276            #[doc = "const S: Stack" $IDX:camel "<i32, 8> = Stack" $IDX:camel "::own_new(0)"]
277            ///     .s_const_unwrap().s.own_push(1).s.own_push(2).s.own_push(3).s;
278            #[doc = "const T: Stack" $IDX:camel "<i32, 4> = S.own_resize_default_truncate().s;"]
279            /// assert_eq![S.as_slice(), T.as_slice()];
280            #[doc = "const U: Stack" $IDX:camel "<i32, 2> = S.own_resize_default_truncate().s;"]
281            /// assert_eq![U.as_slice(), &[2, 3]];
282            /// ```
283            pub const fn own_resize_default_truncate<const NEW_CAP: usize>(self)
284                -> Own<Stack<T, NEW_CAP, $IDX, Bare>, ()> {
285                let start_idx = if self.len as usize > NEW_CAP {
286                    self.len as usize - NEW_CAP
287                } else {
288                    0
289                };
290                let new_len = crate::num::Compare(NEW_CAP).min(self.len as usize);
291                let old_arr: [T; CAP] = self.data.into_array_copy();
292                let mut new_arr = array_init![const_default [T; NEW_CAP]];
293
294                let mut i = 0;
295                while i < new_len {
296                    new_arr[i] = old_arr[i + start_idx];
297                    i += 1;
298                }
299
300                Own::empty(Stack {
301                    data: Array::new_bare(new_arr),
302                    len: new_len as $IDX,
303                })
304            }
305        }
